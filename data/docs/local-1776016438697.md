**Autores:** Tech Lead

### CONTEXTO:
O V10 do DIYAPP está em produção há 18 meses com uma stack heterogênea (React 16, Node.js 14, MongoDB 4). Necessitamos modernizar para:
1. Manter suporte de versões de longo prazo (LTS)

3. Preparar para escalabilidade horizontal
4. Facilitar contratação de desenvolvedores

### DECISÃO:
Adotar a seguinte stack para V11:
- **Frontend:** React 18 com TypeScript 5, Vite como bundler
- **Backend:** Node.js 20 LTS, Express.js com TypeScript
- **Banco de Dados:** PostgreSQL 16 (relacional) + Redis 7 (cache/sessões)
- **Infraestrutura:** Docker + Docker Compose para desenvolvimento
- **CI/CD:** GitHub Actions com testes automatizados

### OPÇÕES CONSIDERADAS:
- **Opção A:** Next.js 14 (Full-stack) - Prós: SSR nativo, otimizações automáticas. Contras: Vendor lock-in, menos flexibilidade.
- **Opção B:** React 18 + Express separados - Prós: Separação clara de responsabilidades, escalabilidade independente. Contras: Mais complexidade inicial.

**Positivas:**
- Suporte LTS até 2026 (Node.js 20)
- Melhor performance com React 18 (concorrência)
- TypeScript em todo o stack reduz bugs em runtime
- PostgreSQL oferece ACID e relações complexas

- Curva de aprendizado para desenvolvedores V10

- Compatibilidade com bibliotecas legadas do V10

## 3. Estratégia de Cache com Redis

```typescript
// server/src/services/cache.service.ts
import Redis from 'ioredis';
import { logger } from '../utils/logger';

export class CacheService {
  private client: Redis;
  private static instance: CacheService;
  private defaultTTL = 3600; // 1 hora

  private constructor() {
    this.client = new Redis({
      host: process.env.REDIS_HOST || 'localhost',
      port: parseInt(process.env.REDIS_PORT || '6379'),
      password: process.env.REDIS_PASSWORD,
      retryStrategy: (times) => {
        const delay = Math.min(times * 50, 2000);
        return delay;
      },
    });

    this.client.on('error', (err) => {
      logger.error('Redis error:', err);
    });

    this.client.on('connect', () => {
      logger.info('Redis connected successfully');
    });
  }

  static getInstance(): CacheService {
    if (!CacheService.instance) {
      CacheService.instance = new CacheService();
    }
    return CacheService.instance;
  }

  async get<T>(key: string): Promise<T | null> {
    try {
      const data = await this.client.get(key);
      return data ? JSON.parse(data) : null;
    } catch (error) {
      logger.error(`Cache get error for key ${key}:`, error);
      return null;
    }
  }

  async set(key: string, value: any, ttl?: number): Promise<void> {
    try {
      const serialized = JSON.stringify(value);
      await this.client.setex(key, ttl || this.defaultTTL, serialized);
    } catch (error) {
      logger.error(`Cache set error for key ${key}:`, error);
    }
  }

  async delete(key: string): Promise<void> {
    try {
      await this.client.del(key);
    } catch (error) {
      logger.error(`Cache delete error for key ${key}:`, error);
    }
  }

  async deletePattern(pattern: string): Promise<void> {
    try {
      const keys = await this.client.keys(pattern);
      if (keys.length > 0) {
        await this.client.del(...keys);
      }
    } catch (error) {
      logger.error(`Cache deletePattern error for pattern ${pattern}:`, error);
    }
  }

  async invalidateUserCache(userId: string): Promise<void> {
    // Invalida todos os caches relacionados ao usuário
    await this.deletePattern(`user:${userId}:*`);
    await this.deletePattern(`session:${userId}:*`);
  }

  async healthCheck(): Promise<boolean> {
    try {
      await this.client.ping();
      return true;
    } catch {
      return false;
    }
  }
}

// Decorator para cache automático
export function Cacheable(ttl?: number, keyBuilder?: (...args: any[]) => string) {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    const cacheService = CacheService.getInstance();

    descriptor.value = async function (...args: any[]) {
      const cacheKey = keyBuilder 
        ? keyBuilder.apply(this, args)
        : `${target.constructor.name}:${propertyKey}:${JSON.stringify(args)}`;

      // Tentar obter do cache
      const cached = await cacheService.get(cacheKey);
      if (cached !== null) {
        return cached;
      }

      // Executar método original
      const result = await originalMethod.apply(this, args);

      // Armazenar no cache
      if (result !== undefined && result !== null) {
        await cacheService.set(cacheKey, result, ttl);
      }

      return result;
    };

    return descriptor;
  };
}
```

// 3. CIRCUIT BREAKER IMPLEMENTATION
// src/core/resilience/CircuitBreaker.js
class CircuitBreaker {
  constructor(name, options = {}) {
    this.name = name;
    this.failureThreshold = options.failureThreshold || 5;
    this.resetTimeout = options.resetTimeout || 60000; // 60 seconds
    this.halfOpenMaxAttempts = options.halfOpenMaxAttempts || 3;
    
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
    this.failureCount = 0;
    this.lastFailureTime = null;
    this.halfOpenAttempts = 0;
    
    this.metrics = {
      totalCalls: 0,
      successfulCalls: 0,
      failedCalls: 0,
      circuitOpens: 0,
      circuitCloses: 0
    };
    
    this.logger = require('../config/logger').createLogger(`circuit-breaker:${name}`);
  }
  
  async execute(fn, fallbackFn = null) {
    this.metrics.totalCalls++;
    
    // Check if circuit is open
    if (this.state === 'OPEN') {
      const timeSinceFailure = Date.now() - this.lastFailureTime;
      if (timeSinceFailure > this.resetTimeout) {
        this.state = 'HALF_OPEN';
        this.halfOpenAttempts = 0;
        this.logger.info(`${this.name}: Circuit moved to HALF_OPEN`);
      } else {
        this.logger.warn(`${this.name}: Circuit OPEN, using fallback`);
        return this.executeFallback(fallbackFn);
      }
    }
    
    try {
      // Execute the function
      const result = await fn();
      
      // Success - reset failure count
      if (this.state === 'HALF_OPEN') {
        this.halfOpenAttempts++;
        if (this.halfOpenAttempts >= this.halfOpenMaxAttempts) {
          this.closeCircuit();
        }
      } else {
        this.failureCount = 0;
      }
      
      this.metrics.successfulCalls++;
      return result;
      
    } catch (error) {
      this.metrics.failedCalls++;
      this.failureCount++;
      this.lastFailureTime = Date.now();
      
      this.logger.error(`${this.name}: Call failed`, {
        error: error.message,
        failureCount: this.failureCount,
        state: this.state
      });
      
      // Check if we should open the circuit
      if (this.state === 'HALF_OPEN' || 
          (this.state === 'CLOSED' && this.failureCount >= this.failureThreshold)) {
        this.openCircuit();
      }
      
      // Try fallback
      return this.executeFallback(fallbackFn, error);
    }
  }
  
  openCircuit() {
    this.state = 'OPEN';
    this.metrics.circuitOpens++;
    this.logger.error(`${this.name}: Circuit OPENED after ${this.failureCount} failures`);
  }
  
  closeCircuit() {
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.halfOpenAttempts = 0;
    this.metrics.circuitCloses++;
    this.logger.info(`${this.name}: Circuit CLOSED`);
  }
  
  async executeFallback(fallbackFn, originalError = null) {
    if (fallbackFn) {
      try {
        this.logger.info(`${this.name}: Executing fallback`);
        return await fallbackFn(originalError);
      } catch (fallbackError) {
        this.logger.error(`${this.name}: Fallback also failed`, {
          originalError: originalError?.message,
          fallbackError: fallbackError.message
        });
        throw new Error(`Service ${this.name} unavailable and fallback failed`);
      }
    }
    
    throw new Error(`Service ${this.name} unavailable`);
  }
  
  getStatus() {
    return {
      name: this.name,
      state: this.state,
      failureCount: this.failureCount,
      lastFailureTime: this.lastFailureTime,
      metrics: { ...this.metrics },
      isHealthy: this.state === 'CLOSED' || this.state === 'HALF_OPEN'
    };
  }
}

// src/core/resilience/RetryManager.js
class RetryManager {
  constructor(options = {}) {
    this.maxRetries = options.maxRetries || 3;
    this.baseDelay = options.baseDelay || 100; // milliseconds
    this.maxDelay = options.maxDelay || 10000; // 10 seconds
    this.retryableStatusCodes = options.retryableStatusCodes || [408, 429, 500, 502, 503, 504];
    this.retryableErrors = options.retryableErrors || ['ETIMEDOUT', 'ECONNRESET', 'ECONNREFUSED'];
    
    this.logger = require('../config/logger').createLogger('retry-manager');
  }
  
  async executeWithRetry(fn, context = {}) {
    let lastError;
    
    for (let attempt = 0; attempt <= this.maxRetries; attempt++) {
      try {
        const result = await fn();
        
        if (attempt > 0) {
          this.logger.info('Retry successful', {
            ...context,
            attempt,
            totalAttempts: attempt + 1
          });
        }
        
        return result;
        
      } catch (error) {
        lastError = error;
        
        // Check if error is retryable
        if (!this.shouldRetry(error) || attempt === this.maxRetries) {
          break;
        }
        
        // Calculate delay with exponential backoff and jitter
        const delay = this.calculateDelay(attempt);
        
        this.logger.warn('Retrying after error', {
          ...context,
          error: error.message,
          attempt: attempt + 1,
          maxRetries: this.maxRetries,
          delay,
          willRetry: attempt < this.maxRetries
        });
        
        // Wait before retry
        await this.sleep(delay);
      }
    }
    
    throw lastError;
  }
  
  shouldRetry(error) {
    // Check HTTP status codes
    if (error.response && error.response.status) {
      return this.retryableStatusCodes.includes(error.response.status);
    }
    
    // Check error codes
    if (error.code && this.retryableErrors.includes(error.code)) {
      return true;
    }
    
    // Check error message for timeout indicators
    if (error.message && (
      error.message.includes('timeout') ||
      error.message.includes('Timeout') ||
      error.message.includes('ETIMEDOUT')
    )) {
      return true;
    }
    
    return false;
  }
  
  calculateDelay(attempt) {
    // Exponential backoff: baseDelay * 2^attempt
    const exponentialDelay = this.baseDelay * Math.pow(2, attempt);
    
    // Add jitter (±20%)
    const jitter = 1 + (Math.random() * 0.4 - 0.2); // 0.8 to 1.2
    
    const delay = Math.min(exponentialDelay * jitter, this.maxDelay);
    
    return Math.floor(delay);
  }
  
  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

```markdown
- **Backend:** Node.js + Express (API Gateway)
- **Microserviços:** 
  - Auth Service (OAuth2/JWT)
  - User Service (gerenciamento de usuários)
  - Data Service (dados sensíveis)
  - LLM Service (integração com modelos de IA)
- **Banco de Dados:** PostgreSQL (dados estruturados) + Redis (cache/sessões)
- **Infra:** Docker + Kubernetes + API Gateway

## 2. Análise STRIDE por Componente

### 2.1 API Gateway
| Ameaça | Categoria | Impacto | Probabilidade | Controles |
|--------|-----------|---------|---------------|-----------|

| Tampering de requests | Tampering | Alto | Média | HTTPS obrigatório, assinatura de requests |
| Repudiation de acesso | Repudiation | Médio | Baixa | Logs imutáveis com correlation IDs |

## 3. Riscos Aceitos Conscientemente

1. **Risco BA-001:** Tokens JWT com 15min de expiração - Aceitamos risco de janela de 15min em caso de token theft em troca de melhor UX.
2. **Risco LLM-001:** Possibilidade de prompt injection básico - Mitigado por sanitização, mas risco residual aceito para funcionalidade.
3. **Risco INF-001:** Single point of failure no API Gateway - Aceito por simplicidade inicial, com plano de failover para Q2.
```

### 2.1 Configuração OAuth2 com PKCE

**Arquivo:** `src/auth/oauth2-config.js`

```javascript
/**
 * Configuração OAuth2 com PKCE para V11
 * Implementa OAuth 2.1 com segurança reforçada
 */

const crypto = require('crypto');
const jwt = require('jsonwebtoken');

class OAuth2Security {
  constructor() {
    this.config = {
      // Tokens de curta duração
      accessTokenExpiry: '15m', // 15 minutos
      refreshTokenExpiry: '7d', // 7 dias
      
      // Algoritmos seguros
      jwtAlgorithm: 'RS256',
      
      // PKCE obrigatório
      pkceRequired: true,
      codeVerifierLength: 64,
      
      // Rate limiting
      maxAuthAttempts: 5,
      lockoutDuration: 900, // 15 minutos
      
      // Escopos padrão
      defaultScopes: ['openid', 'profile', 'email'],
      protectedScopes: ['admin', 'write', 'delete']
    };
    
    this.privateKey = process.env.JWT_PRIVATE_KEY;
    this.publicKey = process.env.JWT_PUBLIC_KEY;
  }

  /**
   * Gera code_verifier e code_challenge para PKCE
   */
  generatePKCE() {
    const codeVerifier = crypto.randomBytes(this.config.codeVerifierLength)
      .toString('base64')
      .replace(/\+/g, '-')
      .replace(/\//g, '_')
      .replace(/=/g, '');
    
    const codeChallenge = crypto
      .createHash('sha256')
      .update(codeVerifier)
      .digest('base64')
      .replace(/\+/g, '-')
      .replace(/\//g, '_')
      .replace(/=/g, '');
    
    return { codeVerifier, codeChallenge };
  }

  /**
   * Valida PKCE flow
   */
  validatePKCE(codeVerifier, codeChallenge) {
    const calculatedChallenge = crypto
      .createHash('sha256')
      .update(codeVerifier)
      .digest('base64')
      .replace(/\+/g, '-')
      .replace(/\//g, '_')
      .replace(/=/g, '');
    
    return crypto.timingSafeEqual(
      Buffer.from(calculatedChallenge),
      Buffer.from(codeChallenge)
    );
  }

  /**
   * Gera JWT seguro com claims mínimas
   */
  generateAccessToken(user, scopes = []) {
    const payload = {
      sub: user.id,
      iss: 'diyapp-v11',
      aud: 'diyapp-api',
      exp: Math.floor(Date.now() / 1000) + 900, // 15 minutos
      iat: Math.floor(Date.now() / 1000),
      scope: scopes.join(' '),
      jti: crypto.randomBytes(16).toString('hex'), // Unique token ID
      // Claims adicionais seguras
      user: {
        id: user.id,
        email: user.email,
        role: user.role
      }
    };

    return jwt.sign(payload, this.privateKey, {
      algorithm: this.config.jwtAlgorithm
    });
  }

  /**
   * Gera refresh token rotativo
   */
  generateRefreshToken(userId) {
    const token = crypto.randomBytes(32).toString('hex');
    const hash = crypto.createHash('sha256').update(token).digest('hex');
    
    return {
      token,
      hash,
      expiresAt: new Date(Date.now() + (7 * 24 * 60 * 60 * 1000)) // 7 dias
    };
  }

  /**
   * Valida JWT com verificação completa
   */
  validateToken(token) {
    try {
      const decoded = jwt.verify(token, this.publicKey, {
        algorithms: [this.config.jwtAlgorithm],
        issuer: 'diyapp-v11',
        audience: 'diyapp-api',
        clockTolerance: 30 // 30 segundos de tolerância
      });

      // Verificações adicionais
      if (!decoded.jti) {
        throw new Error('Token sem JTI');
      }

      if (!decoded.scope) {
        throw new Error('Token sem escopo');
      }

      return decoded;
    } catch (error) {
      // Log detalhado para análise forense
      console.error('Token validation failed:', {
        error: error.message,
        tokenPreview: token.substring(0, 20) + '...'
      });
      throw error;
    }
  }

  /**
   * Verifica escopos do token
   */
  verifyScopes(token, requiredScopes) {
    const decoded = this.validateToken(token);
    const tokenScopes = decoded.scope.split(' ');
    
    return requiredScopes.every(scope => tokenScopes.includes(scope));
  }

  /**
   * Invalidação de tokens (logout, breach)
   */
  async invalidateToken(jti, reason = 'logout') {
    // Adiciona à blacklist no Redis
    const redis = require('../cache/redis-client');
    await redis.setex(
      `token:blacklist:${jti}`,
      900, // 15 minutos (mesmo que token expiry)
      JSON.stringify({
        reason,
        invalidatedAt: new Date().toISOString()
      })
    );
    
    // Log de auditoria
    this.logAudit('token_invalidated', {
      jti,
      reason,
      timestamp: new Date().toISOString()
    });
  }

  /**
   * Rate limiting para autenticação
   */
  async checkRateLimit(identifier, action = 'login') {
    const redis = require('../cache/redis-client');
    const key = `rate:auth:${action}:${identifier}`;
    
    const attempts = await redis.incr(key);
    if (attempts === 1) {
      await redis.expire(key, 60); // Reset a cada minuto
    }
    
    if (attempts > this.config.maxAuthAttempts) {
      // Bloqueio temporário
      const lockKey = `lockout:auth:${identifier}`;
      await redis.setex(lockKey, this.config.lockoutDuration, 'locked');
      
      throw new Error('Too many attempts. Account locked for 15 minutes.');
    }
    
    return attempts;
  }

  /**
   * Logs de auditoria
   */
  logAudit(event, data) {
    const auditLog = {
      event,
      data,
      timestamp: new Date().toISOString(),
      source: 'oauth2-security'
    };
    
    // Envia para sistema centralizado de logs
    require('../monitoring/logger').security(auditLog);
  }
}

module.exports = new OAuth2Security();
```

        documents: ['read', 'create', 'update', 'delete'],

      },
      admin: {
        '*': ['*'] // Admin tem acesso total
      }
    };
  }

  /**
   * Middleware para verificar autorização
   */
  authorize(resource, action) {
    return async (req, res, next) => {
      try {
        // Extrai token do header
        const authHeader = req.headers.authorization;
        if (!authHeader || !authHeader.startsWith('Bearer ')) {
          return res.status(401).json({
            error: 'Unauthorized',
            message: 'Missing or invalid authorization header'
          });
        }

        const token = authHeader.substring(7);
        
        // Valida token
        const decoded = oauth2.validateToken(token);
        
        // Verifica blacklist
        const isBlacklisted = await this.checkTokenBlacklist(decoded.jti);
        if (isBlacklisted) {
          return res.status(401).json({
            error: 'Unauthorized',
            message: 'Token has been revoked'
          });
        }

        // Verifica permissões
        const hasPermission = await this.checkPermission(
          decoded.user.role,
          resource,
          action,
          decoded.user.id,
          req.params.id // resourceId para verificação de ownership
        );

        if (!hasPermission) {
          return res.status(403).json({
            error: 'Forbidden',
            message: 'Insufficient permissions'
          });
        }

        // Adiciona usuário ao request
        req.user = decoded.user;
        req.token = decoded;
        
        // Log de auditoria
        this.logAccess(req, resource, action, true);
        
        next();
      } catch (error) {
        // Log de tentativa falha
        this.logAccess(req, resource, action, false, error.message);

            message: 'Invalid token'

          error: 'AuthorizationError',
          message: 'Internal authorization error'

```python
    def setup_data_quality_views(self):
```

import numpy as np
import pandas as pd
from typing import Dict, List, Optional, Tuple
from datetime import datetime, timedelta
import joblib
import json
import asyncio
from dataclasses import dataclass
from enum import Enum
import warnings
warnings.filterwarnings('ignore')

from sklearn.ensemble import IsolationForest
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import DBSCAN
from tensorflow import keras
from tensorflow.keras import layers

from ..utils.logger import get_logger
from ..utils.metrics_collector import MetricsCollector

logger = get_logger(__name__)

class AnomalyType(Enum):
    LATENCY_SPIKE = "latency_spike"
    ERROR_PATTERN = "error_pattern"
    RESOURCE_SATURATION = "resource_saturation"
    BEHAVIORAL_DRIFT = "behavioral_drift"
    CASCADE_FAILURE = "cascade_failure"

@dataclass
class Anomaly:
    service_name: str
    anomaly_type: AnomalyType
    severity: float  # 0.0 to 1.0
    timestamp: datetime
    metrics: Dict[str, float]
    features: List[float]
    confidence: float
    root_cause_hint: Optional[str] = None
    
class AnomalyDetector:
    def __init__(self, config: Dict):
        self.config = config
        self.metrics_collector = MetricsCollector()
        self.models = {}
        self.scalers = {}
        self.history = {}
        self.initialize_models()
        
    def initialize_models(self):
        """Initialize ML models for different anomaly types"""
        # Isolation Forest for general anomaly detection
        self.models['isolation_forest'] = IsolationForest(
            n_estimators=100,
            contamination=0.1,
            random_state=42
        )
        
        # LSTM for sequential pattern detection
        self.models['lstm_autoencoder'] = self._build_lstm_autoencoder()
        
        # DBSCAN for clustering similar anomalies
        self.models['dbscan'] = DBSCAN(eps=0.5, min_samples=5)
        
        # Initialize scalers
        self.scalers['standard'] = StandardScaler()
        
    def _build_lstm_autoencoder(self) -> keras.Model:
        """Build LSTM autoencoder for sequential anomaly detection"""
        model = keras.Sequential([
            layers.LSTM(64, activation='relu', 
                       input_shape=(20, 10), 
                       return_sequences=True),
            layers.LSTM(32, activation='relu', return_sequences=True),
            layers.LSTM(64, activation='relu', return_sequences=True),
            layers.TimeDistributed(layers.Dense(10))
        ])
        
        model.compile(optimizer='adam', loss='mse')
        return model
    
    async def detect_anomalies(self, service_name: str) -> List[Anomaly]:
        """Main anomaly detection method"""
        anomalies = []
        
        # Collect recent metrics
        metrics_data = await self.metrics_collector.get_service_metrics(
            service_name, 
            lookback_minutes=15
        )
        
        if not metrics_data:
            return anomalies
        
        # Convert to features
        features = self._extract_features(metrics_data)
        
        # Detect different types of anomalies
        anomalies.extend(self._detect_latency_anomalies(service_name, features, metrics_data))
        anomalies.extend(self._detect_error_patterns(service_name, features, metrics_data))
        anomalies.extend(self._detect_resource_anomalies(service_name, features, metrics_data))
        anomalies.extend(self._detect_behavioral_drift(service_name, features, metrics_data))
        
        # Filter by confidence threshold
        anomalies = [a for a in anomalies 
                    if a.confidence >= self.config['anomaly_detection']['confidence_threshold']]
        
        # Cluster similar anomalies
        anomalies = self._cluster_anomalies(anomalies)
        
        # Update history
        self._update_history(service_name, anomalies)
        
        return anomalies
    
    def _extract_features(self, metrics_data: pd.DataFrame) -> np.ndarray:
        """Extract features from metrics data"""
        features = []
        
        # Latency features
        if 'latency' in metrics_data.columns:
            features.append(metrics_data['latency'].mean())
            features.append(metrics_data['latency'].std())
            features.append(metrics_data['latency'].max())
            features.append(metrics_data['latency'].p95())
        
        # Error rate features
        if 'error_rate' in metrics_data.columns:
            features.append(metrics_data['error_rate'].mean())
            features.append(metrics_data['error_rate'].std())
            features.append(metrics_data['error_rate'].max())
        
        # Resource features
        for resource in ['cpu', 'memory', 'disk']:
            col = f'{resource}_usage'
            if col in metrics_data.columns:
                features.append(metrics_data[col].mean())
                features.append(metrics_data[col].max())
        
        # Rate of change features
        for col in metrics_data.columns:
            if metrics_data[col].dtype in [np.float64, np.int64]:
                diff = metrics_data[col].diff().dropna()
                if len(diff) > 0:
                    features.append(diff.mean())
                    features.append(diff.std())
        
        return np.array(features).reshape(1, -1)
    
    def _detect_latency_anomalies(self, service_name: str, 
                                 features: np.ndarray, 
                                 metrics_data: pd.DataFrame) -> List[Anomaly]:
        """Detect latency spikes and patterns"""
        anomalies = []
        
        # Check against thresholds
        thresholds = self.config['services']['core_services'].get(service_name, {}).get('thresholds', {})
        
        if 'latency_p95' in thresholds:
            current_p95 = metrics_data['latency'].quantile(0.95)
            threshold = thresholds['latency_p95']
            
            if current_p95 > threshold:
                severity = min(1.0, current_p95 / threshold)
                
                anomaly = Anomaly(
                    service_name=service_name,
                    anomaly_type=AnomalyType.LATENCY_SPIKE,
                    severity=severity,
                    timestamp=datetime.now(),
                    metrics={'latency_p95': current_p95, 'threshold': threshold},
                    features=features.flatten().tolist(),
                    confidence=0.9,
                    root_cause_hint=f"Latency P95 ({current_p95}ms) exceeds threshold ({threshold}ms)"
                )
                anomalies.append(anomaly)
        
        # Detect sudden spikes using statistical methods
        latency_series = metrics_data['latency'].values
        if len(latency_series) > 10:
            mean = np.mean(latency_series)
            std = np.std(latency_series)
            
            # Check for values beyond 3 standard deviations
            spikes = np.where(np.abs(latency_series - mean) > 3 * std)[0]
            
            if len(spikes) > 0:
                spike_value = latency_series[spikes[-1]]
                severity = min(1.0, (spike_value - mean) / (3 * std))
                
                anomaly = Anomaly(
                    service_name=service_name,
                    anomaly_type=AnomalyType.LATENCY_SPIKE,
                    severity=severity,
                    timestamp=datetime.now(),
                    metrics={'spike_value': spike_value, 'mean': mean, 'std': std},
                    features=features.flatten().tolist(),
                    confidence=0.85,
                    root_cause_hint=f"Statistical latency spike detected: {spike_value}ms"
                )
                anomalies.append(anomaly)
        
        return anomalies
    
    def _detect_error_patterns(self, service_name: str, 
                              features: np.ndarray, 
                              metrics_data: pd.DataFrame) -> List[Anomaly]:
        """Detect error patterns and sequences"""
        anomalies = []
        
        if 'error_rate' not in metrics_data.columns:
            return anomalies
        
        error_series = metrics_data['error_rate'].values
        
        # Check error rate threshold
        thresholds = self.config['services']['core_services'].get(service_name, {}).get('thresholds', {})
        
        if 'error_rate' in thresholds:
            current_error = error_series[-1] if len(error_series) > 0 else 0
            threshold = thresholds['error_rate']
            
            if current_error > threshold:
                severity = min(1.0, current_error / threshold)
                
                anomaly = Anomaly(
                    service_name=service_name,
                    anomaly_type=AnomalyType.ERROR_PATTERN,
                    severity=severity,
                    timestamp=datetime.now(),
                    metrics={'error_rate': current_error, 'threshold': threshold},
                    features=features.flatten().tolist(),
                    confidence=0.9,
                    root_cause_hint=f"Error rate ({current_error:.3f}) exceeds threshold ({threshold})"
                )
                anomalies.append(anomaly)
        
        # Detect error bursts (consecutive high error rates)
        if len(error_series) >= 5:
            error_burst = np.convolve(error_series > 0.01, np.ones(5), mode='valid')
            burst_indices = np.where(error_burst >= 4)[0]
            
            if len(burst_indices) > 0:
                burst_severity = error_burst[burst_indices[-1]] / 5
                
                anomaly = Anomaly(
                    service_name=service_name,
                    anomaly_type=AnomalyType.ERROR_PATTERN,
                    severity=burst_severity,
                    timestamp=datetime.now(),
                    metrics={'burst_strength': burst_severity},
                    features=features.flatten().tolist(),
                    confidence=0.8,
                    root_cause_hint="Error burst pattern detected"
                )
                anomalies.append(anomaly)
        
        return anomalies
    
    def _detect_resource_anomalies(self, service_name: str, 
                                  features: np.ndarray, 
                                  metrics_data: pd.DataFrame) -> List[Anomaly]:
        """Detect resource saturation anomalies"""
        anomalies = []

            col = f'{resource}_usage'
            if col in metrics_data.columns and f'{resource}_usage' in thresholds:
                current_usage = metrics_data[col].iloc[-1]
                threshold = thresholds[f'{resource}_usage']

                        service_name=service_name,
                        anomaly_type=AnomalyType.RESOURCE_SATURATION,
                        severity=severity,

                        metrics={f'{resource}_usage': current_usage, 'threshold': threshold},

```javascript
// Context Manager
class ContextManager {
  constructor() {
    this.userContexts = new Map();
    this.maxContextLength = 4000;
  }

  addToContext(userId, role, content) {
    if (!this.userContexts.has(userId)) {
      this.userContexts.set(userId, []);
    }
    
    const context = this.userContexts.get(userId);
    context.push({ role, content, timestamp: Date.now() });
    
    // Manter contexto dentro do limite
    while (this.getContextLength(context) > this.maxContextLength) {
      context.shift();
    }
    
    return context;
  }

  getContextLength(context) {
    return context.reduce((acc, msg) => acc + msg.content.length, 0);
  }

  getContext(userId) {
    return this.userContexts.get(userId) || [];
  }

  clearContext(userId) {
    this.userContexts.delete(userId);
  }
}

    // Adicionar mensagem do usuário ao contexto

        model: "gpt-4o-mini",
        messages,
        temperature: 0.7,
        max_tokens: 500,
      });

      // Detectar intenções específicas
      const intent = this.detectIntent(userMessage);
      
      return {
        message: assistantMessage,
        intent,
        contextLength: context.length,
        suggestions: this.generateSuggestions(intent, contextType)

      throw new Error('Falha ao processar mensagem com IA');

      return 'project_ideas';

      return 'step_by_step';

      return 'materials';
    } else if (lowerMessage.includes('dificuldade') || lowerMessage.includes('nível')) {
      return 'difficulty';
    } else if (lowerMessage.includes('tempo') || lowerMessage.includes('duração')) {
      return 'time_estimation';
    } else {
      return 'general_help';
    }
  }

  generateSuggestions(intent, contextType) {
    const suggestions = {
      project_ideas: [
        "Posso gerar 3 ideias de projetos personalizadas para você",
        "Quer que eu crie um plano passo a passo para alguma ideia?",

        "Quer dicas baseadas em projetos similares?",
        "Posso explicar técnicas comuns de marcenaria/elétrica/etc"
      ]
    };

    return suggestions[intent] || suggestions.general_help;
  }
}
```

```python
"""
Auditor de Conformidade LGPD para V11
"""
import json
import re
import hashlib
from datetime import datetime, timedelta
from typing import Dict, List, Optional, Set, Tuple
from dataclasses import dataclass, asdict
from enum import Enum
import logging

logger = logging.getLogger(__name__)

class DataCategory(Enum):
    """Categorias de dados pessoais conforme LGPD"""
    IDENTIFICADOR = "identificador"
    CONTATO = "contato"
    FINANCEIRO = "financeiro"
    LOCALIZACAO = "localizacao"
    COMPORTAMENTAL = "comportamental"
    PERFORMANCE = "performance"
    TECNICO = "tecnico"
    SENSIVEL = "sensivel"

class LegalBasis(Enum):
    """Bases legais conforme LGPD Artigo 7"""
    CONSENTIMENTO = "consentimento"
    CONTRATO = "contrato"
    OBRIGACAO_LEGAL = "obrigacao_legal"
    INTERESSE_LEGITIMO = "interesse_legitimo"
    VIDA = "protecao_da_vida"
    SAUDE = "tutela_da_saude"
    CREDITO = "protecao_do_credito"
    PUBLICO = "interesse_publico"

@dataclass
class DataFlow:
    """Fluxo de dados identificado"""
    id: str
    name: str
    source: str
    destination: str
    data_categories: List[DataCategory]
    legal_basis: Optional[LegalBasis]
    retention_days: int
    encrypted: bool
    has_consent: bool
    pii_detected: bool
    risk_level: str  # baixo, medio, alto
    
    def to_dict(self):
        return asdict(self)

class LGPDComplianceAuditor:
    """Auditor principal para conformidade LGPD"""
    
    # Padrões de PII (Personal Identifiable Information)
    PII_PATTERNS = {
        'cpf': r'\b\d{3}\.?\d{3}\.?\d{3}-?\d{2}\b',
        'cnpj': r'\b\d{2}\.?\d{3}\.?\d{3}/\d{4}-\d{2}\b',
        'email': r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b',
        'phone_br': r'\b(?:\+55\s?)?(?:\(?\d{2}\)?\s?)?\d{4,5}-?\d{4}\b',
        'credit_card': r'\b\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}\b',
        'rg': r'\b\d{1,2}\.?\d{3}\.?\d{3}-?\d{1}\b',
    }
    
    def __init__(self, config_path: str = None):
        self.data_flows: Dict[str, DataFlow] = {}
        self.violations: List[Dict] = []
        self.consent_records: Dict[str, Dict] = {}
        
        if config_path:
            self.load_config(config_path)
    
    def scan_data_flow(self, flow_data: Dict) -> DataFlow:
        """Analisa um fluxo de dados para conformidade"""
        
        # Identificar PII no fluxo
        pii_detected = self._detect_pii(flow_data.get('sample_data', ''))
        
        # Determinar categorias de dados
        categories = self._categorize_data(flow_data)
        
        # Avaliar nível de risco
        risk_level = self._assess_risk(categories, pii_detected)
        
        flow = DataFlow(
            id=flow_data.get('id', hashlib.md5(json.dumps(flow_data).encode()).hexdigest()[:8]),
            name=flow_data.get('name', 'Unnamed Flow'),
            source=flow_data.get('source', 'unknown'),
            destination=flow_data.get('destination', 'unknown'),
            data_categories=categories,
            legal_basis=self._determine_legal_basis(flow_data),
            retention_days=flow_data.get('retention_days', 30),
            encrypted=flow_data.get('encrypted', False),
            has_consent=flow_data.get('has_consent', False),
            pii_detected=pii_detected,
            risk_level=risk_level
        )
        
        self.data_flows[flow.id] = flow
        
        # Verificar violações
        self._check_violations(flow)
        
        return flow
    
    def _detect_pii(self, data: str) -> bool:
        """Detecta PII em dados de amostra"""
        if not data:
            return False
        
        data_lower = data.lower()
        
        for pii_type, pattern in self.PII_PATTERNS.items():
            if re.search(pattern, data_lower, re.IGNORECASE):
                logger.warning(f"PII detectado: {pii_type}")
                return True
        
        # Verificar padrões comuns de PII
        pii_indicators = [
            'nome', 'sobrenome', 'endereço', 'endereco', 'cidade', 'estado',
            'data de nascimento', 'nascimento', 'mae', 'pai', 'salario',
            'renda', 'senha', 'password', 'token', 'chave', 'assinatura'
        ]
        
        for indicator in pii_indicators:
            if indicator in data_lower:
                return True
        
        return False
    
    def _categorize_data(self, flow_data: Dict) -> List[DataCategory]:
        """Categoriza os dados processados"""
        categories = set()
        
        # Mapeamento de descrições para categorias
        category_mapping = {
            'user_id': DataCategory.IDENTIFICADOR,
            'email': DataCategory.CONTATO,
            'phone': DataCategory.CONTATO,
            'location': DataCategory.LOCALIZACAO,
            'ip_address': DataCategory.TECNICO,
            'device_info': DataCategory.TECNICO,
            'performance': DataCategory.PERFORMANCE,
            'behavior': DataCategory.COMPORTAMENTAL,
            'payment': DataCategory.FINANCEIRO,
            'health': DataCategory.SENSIVEL,
            'biometric': DataCategory.SENSIVEL,
        }
        
        data_types = flow_data.get('data_types', [])
        for data_type in data_types:
            if data_type in category_mapping:
                categories.add(category_mapping[data_type])
        
        return list(categories)
    
    def _determine_legal_basis(self, flow_data: Dict) -> Optional[LegalBasis]:
        """Determina a base legal apropriada"""
        
        data_types = flow_data.get('data_types', [])
        purpose = flow_data.get('purpose', '').lower()
        
        # Lógica para determinar base legal
        if 'consent_required' in flow_data and flow_data['consent_required']:
            return LegalBasis.CONSENTIMENTO
        
        if any(t in ['payment', 'subscription'] for t in data_types):
            return LegalBasis.CONTRATO
        
        if 'performance' in data_types and 'analytics' in purpose:
            return LegalBasis.INTERESSE_LEGITIMO
        
        if 'health' in data_types or 'biometric' in data_types:
            return LegalBasis.SAUDE
        
        return None
    
    def _assess_risk(self, categories: List[DataCategory], pii_detected: bool) -> str:
        """Avalia o nível de risco do fluxo"""
        
        risk_score = 0
        
        # Pontuar por categoria de dados
        category_weights = {
            DataCategory.SENSIVEL: 3,
            DataCategory.FINANCEIRO: 2,
            DataCategory.IDENTIFICADOR: 2,
            DataCategory.LOCALIZACAO: 1,
            DataCategory.CONTATO: 1,
            DataCategory.COMPORTAMENTAL: 1,
            DataCategory.PERFORMANCE: 0,
            DataCategory.TECNICO: 0,
        }
        
        for category in categories:
            risk_score += category_weights.get(category, 0)
        
        # Adicionar peso por PII
        if pii_detected:
            risk_score += 2
        
        # Determinar nível de risco
        if risk_score >= 4:
            return 'alto'
        elif risk_score >= 2:
            return 'medio'
        else:
            return 'baixo'
    
    def _check_violations(self, flow: DataFlow):
        """Verifica violações de conformidade"""
        
        violations = []
        
        # Violação: Fluxo sem base legal
        if not flow.legal_basis and flow.pii_detected:
            violations.append({
                'type': 'MISSING_LEGAL_BASIS',
                'severity': 'alto',
                'description': f'Fluxo {flow.name} processa PII sem base legal definida',
                'flow_id': flow.id
            })
        
        # Violação: Dados sensíveis sem consentimento explícito
        if DataCategory.SENSIVEL in flow.data_categories and not flow.has_consent:
            violations.append({
                'type': 'SENSITIVE_DATA_WITHOUT_CONSENT',
                'severity': 'alto',
                'description': f'Fluxo {flow.name} processa dados sensíveis sem consentimento',
                'flow_id': flow.id
            })
        
        # Violação: Dados não criptografados
        if not flow.encrypted and flow.pii_detected:
            violations.append({
                'type': 'UNENCRYPTED_PII',
                'severity': 'medio',
                'description': f'Fluxo {flow.name} transmite PII sem criptografia',
                'flow_id': flow.id
            })
        
        # Violação: Retenção indefinida
        if flow.retention_days <= 0 or flow.retention_days > 365 * 5:  # Máximo 5 anos
            violations.append({
                'type': 'INVALID_RETENTION',
                'severity': 'medio',
                'description': f'Fluxo {flow.name} tem período de retenção inválido: {flow.retention_days} dias',
                'flow_id': flow.id
            })
        
        self.violations.extend(violations)
    
    def generate_compliance_report(self) -> Dict:
        """Gera relatório completo de conformidade"""
        
        total_flows = len(self.data_flows)
        high_risk_flows = sum(1 for f in self.data_flows.values() if f.risk_level == 'alto')
        medium_risk_flows = sum(1 for f in self.data_flows.values() if f.risk_level == 'medio')
        
        return {
            'timestamp': datetime.now().isoformat(),
            'summary': {
                'total_data_flows': total_flows,
                'high_risk_flows': high_risk_flows,
                'medium_risk_flows': medium_risk_flows,
                'low_risk_flows': total_flows - high_risk_flows - medium_risk_flows,
                'total_violations': len(self.violations),
                'critical_violations': sum(1 for v in self.violations if v['severity'] == 'alto'),
            },
            'data_flows': [flow.to_dict() for flow in self.data_flows.values()],
            'violations': self.violations,
            'recommendations': self._generate_recommendations()
        }
    
    def _generate_recommendations(self) -> List[Dict]:
        """Gera recomendações para correção"""
        
        recommendations = []

            violation_types[v_type].append(violation)
```
