const billingRepo = require('../repositories/billingRepository');
const { CircuitBreaker } = require('../utils/resilience');
const { maskSensitiveData } = require('../utils/logging');

// Circuit Breaker para chamadas externas (ex: gateway de pagamento)
const paymentGatewayBreaker = new CircuitBreaker({
  threshold: 3, // Número de falhas antes de abrir o circuito
  resetTimeout: 30000, // 30 segundos para tentar novamente
});

// Função para calcular faturamento premium
async function calculatePremiumBilling(input) {
  // 1. Valida regras de negócio
  if (input.tokensUsed > input.plan.limit) {
    throw new Error('Token limit exceeded');
  }

  // 2. Calcula valor com fallback para cache
  const basePrice = input.plan.price;
  const extraTokens = Math.max(0, input.tokensUsed - input.plan.baseTokens);
  const extraCost = extraTokens * input.plan.extraTokenPrice;

  // 3. Chama gateway de pagamento com resiliência
  let paymentStatus;
  try {
    paymentStatus = await paymentGatewayBreaker.execute(() =>
      callPaymentGateway(input.paymentMethod)
    );
  } catch (err) {
    // Fallback: usa cache ou retorna erro claro
    console.warn({
      level: 'warn',
      message: 'Payment gateway failed, using fallback',
      error: err.message,
    });
    paymentStatus = { status: 'fallback', transactionId: 'fallback-' + Date.now() };
  }

  // 4. Persiste no banco
  const billingRecord = {
    userId: input.userId,
    amount: basePrice + extraCost,
    tokensUsed: input.tokensUsed,
    paymentStatus,
    createdAt: new Date(),
  };

  await billingRepo.saveBillingRecord(billingRecord);

  return {
    success: true,
    amount: basePrice + extraCost,
    transactionId: paymentStatus.transactionId,
    tokensUsed: input.tokensUsed,
  };
}

// Função para obter histórico de faturamento
async function getBillingHistory(userId) {
  return await billingRepo.getBillingHistory(userId);
}

// Helper para chamar gateway de pagamento (com timeout e retry)
async function callPaymentGateway(paymentMethod) {
  // Simula chamada externa com timeout
  return new Promise((resolve, reject) => {
    const timeout = setTimeout(() => {
      reject(new Error('Payment gateway timeout'));
    }, 5000);

    // Simula resposta do gateway
    setTimeout(() => {
      clearTimeout(timeout);
      resolve({ status: 'success', transactionId: 'txn_' + Math.random().toString(36).substr(2, 9) });
    }, 1000);
  });
}

module.exports = {
  calculatePremiumBilling,
  getBillingHistory,
};

const db = require('./db');

// Salva registro de faturamento (com query parametrizada)
async function saveBillingRecord(record) {
  const query = `
    INSERT INTO billing_records
    (user_id, amount, tokens_used, payment_status, created_at)
    VALUES ($1, $2, $3, $4, $5)
    RETURNING *;
  `;
  const params = [
    record.userId,
    record.amount,
    record.tokensUsed,
    JSON.stringify(record.paymentStatus),
    record.createdAt,
  ];
  const result = await db.query(query, params);
  return result.rows[0];
}

// Obtém histórico de faturamento (com mascaramento de dados sensíveis)
async function getBillingHistory(userId) {
  const query = `
    SELECT id, amount, tokens_used, payment_status, created_at
    FROM billing_records
    WHERE user_id = $1
    ORDER BY created_at DESC;
  `;
  const result = await db.query(query, [userId]);
  return result.rows.map(row => ({
    ...row,
    paymentStatus: JSON.parse(row.paymentStatus),
  }));
}

module.exports = {
  saveBillingRecord,
  getBillingHistory,
};

#### **[FILE: src/utils/resilience.js]**
```javascript
class CircuitBreaker {
  constructor({ threshold, resetTimeout }) {
    this.threshold = threshold;
    this.resetTimeout = resetTimeout;
    this.failures = 0;
    this.state = 'CLOSED'; // CLOSED | OPEN | HALF_OPEN
    this.nextAttempt = 0;
  }

  async execute(fn) {
    if (this.state === 'OPEN') {
      if (this.nextAttempt <= Date.now()) {
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('Circuit breaker is open');
      }
    }

    try {
      const result = await fn();
      this.reset();
      return result;
    } catch (err) {
      this.failures++;
      if (this.failures >= this.threshold) {
        this.open();
      }
      throw err;
    }
  }

  reset() {
    this.state = 'CLOSED';
    this.failures = 0;
    this.nextAttempt = 0;
  }

  open() {
    this.state = 'OPEN';
    this.nextAttempt = Date.now() + this.resetTimeout;
  }
}

module.exports = { CircuitBreaker };
```

function initMetrics(req, res, next) {

    const durationMs = duration[0] * 1000 + duration[1] / 1e6;

      endpoint,
      duration_ms: durationMs,
      status: res.statusCode,
    });
  });
