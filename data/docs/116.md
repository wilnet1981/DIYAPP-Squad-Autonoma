"""
observability/config.py
ConfiguraГ§ГЈo centralizada de observabilidade
"""

import logging
import json
from dataclasses import dataclass
from typing import Dict, Any
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.resources import Resource
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from prometheus_client import Counter, Histogram, Gauge, start_http_server

@dataclass
class ObservabilityConfig:
    """ConfiguraГ§ГЈo centralizada de observabilidade"""
    
    # MГ©tricas Prometheus
    metrics_port: int = 9090
    namespace: str = "diyapp_v12"
    
    # Logs
    log_level: str = "INFO"
    log_format: str = "json"
    
    # Traces
    trace_endpoint: str = "jaeger:4317"
    service_name: str = "unknown-service"
    
    def __post_init__(self):
        """Inicializar componentes de observabilidade"""
        self.setup_logging()
        self.setup_tracing()
        self.setup_metrics()
    
    def setup_logging(self):
        """Configurar logging estruturado"""
        logging.basicConfig(
            level=getattr(logging, self.log_level),
            format='%(asctime)s - %(name)s - %(levelname)s - %(message)s' 
            if self.log_format == "text" 
            else json.dumps({
                "timestamp": "%(asctime)s",
                "service": self.service_name,
                "level": "%(levelname)s",
                "message": "%(message)s",
                "logger": "%(name)s",
                "trace_id": self.get_trace_id()
            })
        )
    
    def setup_tracing(self):
        """Configurar distributed tracing"""
        resource = Resource.create({
            "service.name": self.service_name,
            "service.version": "v12"
        })
        
        tracer_provider = TracerProvider(resource=resource)
        tracer_provider.add_span_processor(
            BatchSpanProcessor(OTLPSpanExporter(endpoint=self.trace_endpoint))
        )
        trace.set_tracer_provider(tracer_provider)
    
    def setup_metrics(self):
        """Inicializar servidor de mГ©tricas"""
        start_http_server(self.metrics_port)
    
    def get_trace_id(self) -> str:
        """Obter trace ID atual para logs"""
        span = trace.get_current_span()
        if span and span.get_span_context().trace_id:
            return format(span.get_span_context().trace_id, '032x')
        return ""


class Metrics:
    """MГ©tricas padrГЈo para todos os serviГ§os"""
    
    # Contadores
    requests_total = Counter(
        'http_requests_total',
        'Total HTTP requests',
        ['method', 'endpoint', 'status']
    )
    
    errors_total = Counter(
        'application_errors_total',
        'Total application errors',
        ['error_type', 'service']
    )
    
    events_published = Counter(
        'events_published_total',
        'Total events published',
        ['event_type', 'topic']
    )
    
    events_consumed = Counter(
        'events_consumed_total',
        'Total events consumed',
        ['event_type', 'consumer_group']
    )
    
    # Histogramas
    request_duration = Histogram(
        'http_request_duration_seconds',
        'HTTP request duration',
        ['method', 'endpoint'],
        buckets=[0.1, 0.5, 1.0, 2.0, 5.0]
    )
    
    event_processing_time = Histogram(
        'event_processing_seconds',
        'Event processing time',
        ['event_type'],
        buckets=[0.01, 0.05, 0.1, 0.5, 1.0]
    )
    
    # Gauges
    active_connections = Gauge(
        'database_active_connections',
        'Active database connections',
        ['database']
    )
    
    queue_size = Gauge(
        'message_queue_size',
        'Current message queue size',
        ['topic']
    )
    
    memory_usage = Gauge(
        'process_memory_bytes',
        'Process memory usage in bytes'
    )


class StructuredLogger:
    """Logger estruturado com contexto"""
    
    def __init__(self, service_name: str):
        self.logger = logging.getLogger(service_name)
        self.service_name = service_name
    
    def info(self, message: str, **kwargs):
        """Log info level com contexto"""
        extra = {
            "service": self.service_name,
            "context": kwargs,
            "trace_id": ObservabilityConfig().get_trace_id()
        }
        self.logger.info(message, extra={"extra": extra})
    
    def error(self, message: str, error: Exception = None, **kwargs):
        """Log error level com contexto"""
        extra = {
            "service": self.service_name,
            "context": kwargs,
            "error_type": error.__class__.__name__ if error else "unknown",
            "error_message": str(error) if error else "",
            "trace_id": ObservabilityConfig().get_trace_id()
        }
        self.logger.error(message, extra={"extra": extra})

    async def register_user(
        self,
        email: str,
        username: str,
        password: str
    ) -> Tuple[User, str]:
        """Register new user with validation"""
        # Validation
        if not email or "@" not in email:
            raise ValidationError("Invalid email format")
        
        if len(password) < 8:
            raise ValidationError("Password must be at least 8 characters")
        
        # Check existing user
        existing = await self.user_repo.get_by_email(email)
        if existing:
            raise ConflictError("Email already registered")
        
        # Hash password
        hashed = bcrypt.hashpw(password.encode(), bcrypt.gensalt()).decode()
        
        # Create user
        user = User(
            id=str(uuid.uuid4()),
            email=email,
            username=username,
            hashed_password=hashed,
            status=UserStatus.PENDING,
            created_at=datetime.utcnow(),
            updated_at=datetime.utcnow()
        )
        
        # Save user
        saved_user = await self.user_repo.save(user)
        
        # Generate verification token
        verification_token = await self._generate_verification_token(saved_user.id)
        
        # Send verification email (async - don't await for response)
        await self.email_service.send_verification_email(email, verification_token)
        
        # Generate access token
        access_token = await self._generate_access_token(saved_user.id)
        
        return saved_user, access_token
    
    async def login(self, email: str, password: str) -> Tuple[str, str]:
        """Authenticate user and return tokens"""
        user = await self.user_repo.get_by_email(email)
        if not user:
            raise AuthenticationError("Invalid credentials")
        
        if not user.is_active():
            raise AuthenticationError("Account not active")
        
        # Verify password
        if not bcrypt.checkpw(password.encode(), user.hashed_password.encode()):
            raise AuthenticationError("Invalid credentials")
        
        # Update last login
        user.last_login_at = datetime.utcnow()
        await self.user_repo.save(user)
        
        # Generate tokens
        access_token = await self._generate_access_token(user.id)
        refresh_token = await self._generate_refresh_token(user.id)
        
        return access_token, refresh_token
    
    async def verify_email(self, token: str) -> bool:
        """Verify user email with token"""
        auth_token = await self.token_repo.get_valid_token(token, "verification")
        if not auth_token:
            raise AuthenticationError("Invalid or expired token")
        
        user = await self.user_repo.get_by_id(auth_token.user_id)
        if not user:
            raise NotFoundError("User not found")
        
        user.verify_email()
        user.activate()
        
        await self.user_repo.save(user)
        await self.token_repo.revoke_user_tokens(user.id, "verification")
        
        return True

            "type": "access",

        }

import asyncpg
from contextlib import asynccontextmanager
from typing import AsyncIterator

class PostgresConnectionPool:
    """PostgreSQL connection pool with circuit breaker"""
    
    def __init__(self, dsn: str, min_size: int = 5, max_size: int = 20):
        self.dsn = dsn
        self.min_size = min_size
        self.max_size = max_size
        self._pool = None
        self._circuit_open = False
        self._failure_count = 0
    
    async def initialize(self):
        """Initialize connection pool"""
        try:
            self._pool = await asyncpg.create_pool(
                dsn=self.dsn,
                min_size=self.min_size,
                max_size=self.max_size,
                command_timeout=30,  # 30 second timeout
                max_queries=50000,
                max_inactive_connection_lifetime=300
            )
            self._circuit_open = False
            self._failure_count = 0
        except Exception as e:
            self._failure_count += 1
            if self._failure_count >= 3:
                self._circuit_open = True
            raise ConnectionError(f"Failed to initialize pool: {e}")
    
    @asynccontextmanager
    async def acquire(self) -> AsyncIterator[asyncpg.Connection]:
        """Acquire connection with circuit breaker"""
        if self._circuit_open:
            raise ConnectionError("Database circuit breaker open")
        
        try:
            async with self._pool.acquire() as conn:
                yield conn
                self._failure_count = 0
        except Exception as e:
            self._failure_count += 1
            if self._failure_count >= 3:
                self._circuit_open = True
            raise
    
    async def close(self):
        """Close connection pool"""
        if self._pool:
            await self._pool.close()

    service_name VARCHAR(100),
    host_name VARCHAR(100),
    timestamp TIMESTAMP WITH TIME ZONE,
    tags JSONB,
    ingested_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()

-- ГЌndices para performance
CREATE INDEX idx_raw_app_events_timestamp ON raw.app_events(event_timestamp);
CREATE INDEX idx_raw_app_events_user ON raw.app_events(user_id);
CREATE INDEX idx_raw_llm_calls_timestamp ON raw.llm_calls(request_timestamp);
CREATE INDEX idx_raw_llm_calls_feature ON raw.llm_calls(feature_name);
CREATE INDEX idx_raw_user_sessions_start ON raw.user_sessions(start_timestamp);

-- PolГ­tica de retenГ§ГЈo de dados (30 dias para raw, 90 dias para staging, 365 dias para marts)
CREATE OR REPLACE FUNCTION raw.cleanup_old_data()
RETURNS void AS $$
BEGIN
    DELETE FROM raw.app_events 
    WHERE ingested_at < NOW() - INTERVAL '30 days';
    
    DELETE FROM raw.llm_calls 
    WHERE ingested_at < NOW() - INTERVAL '30 days';
    
    DELETE FROM raw.user_sessions 
    WHERE ingested_at < NOW() - INTERVAL '30 days';
    
    DELETE FROM raw.system_metrics 
    WHERE ingested_at < NOW() - INTERVAL '30 days';
END;
$$ LANGUAGE plpgsql;

-- Agendar limpeza diГЎria
SELECT cron.schedule(
    'cleanup-raw-data',
    '0 2 * * *', -- 2 AM daily

WITH user_events AS (
    SELECT
        DATE(event_timestamp) AS event_date,
        user_id,
        COUNT(DISTINCT session_id) AS session_count,
        COUNT(*) AS event_count,
        COUNT(DISTINCT event_name) AS unique_features_used
    FROM {{ ref('stg_app_events') }}
    WHERE event_type IN ('click', 'view', 'interaction')
    GROUP BY 1, 2
),

sessions AS (
    SELECT
        DATE(start_timestamp) AS session_date,
        user_id,
        COUNT(*) AS total_sessions,
        AVG(duration_seconds) AS avg_session_duration,
        SUM(page_views) AS total_page_views
    FROM {{ ref('stg_user_sessions') }}
    GROUP BY 1, 2
)

SELECT
    COALESCE(u.event_date, s.session_date) AS date,
    COALESCE(u.user_id, s.user_id) AS user_id,
    
    -- Atividade
    CASE 
        WHEN u.user_id IS NOT NULL THEN TRUE 
        ELSE FALSE 
    END AS was_active,
    
    -- MГ©tricas de engajamento
    COALESCE(u.session_count, 0) AS session_count,
    COALESCE(u.event_count, 0) AS event_count,
    COALESCE(u.unique_features_used, 0) AS unique_features_used,
    
    -- MГ©tricas de sessГЈo
    COALESCE(s.total_sessions, 0) AS total_sessions,
    COALESCE(s.avg_session_duration, 0) AS avg_session_duration,
    COALESCE(s.total_page_views, 0) AS total_page_views,
    
    -- ClassificaГ§ГЈo
    CASE
        WHEN u.event_count >= 10 THEN 'high'
        WHEN u.event_count >= 5 THEN 'medium'
        WHEN u.event_count >= 1 THEN 'low'
        ELSE 'inactive'
    END AS engagement_level
    
FROM user_events u
FULL OUTER JOIN sessions s 
    ON u.event_date = s.session_date 
    AND u.user_id = s.user_id
