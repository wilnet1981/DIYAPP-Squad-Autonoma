# DIYAPP Evolution - V12 Core - Arquitetura de MicrosserviГ§os

## 1. Architecture Decision Record (ADR-001)

```
ADR-001: Arquitetura de MicrosserviГ§os V12 para DIYAPP
Data: 2024-01-15
Status: Aceita
Autores: Tech Lead, Especialista Infra, Especialista SeguranГ§a

CONTEXTO:
O DIYAPP evoluiu de uma aplicaГ§ГЈo monolГӯtica para uma plataforma distribuГӯda
com mГәltiplos mГіdulos de IA. A arquitetura atual apresenta:
1. Acoplamento excessivo entre mГіdulos
2. Dificuldade de deploy independente
3. Escalabilidade limitada por gargalos compartilhados
4. Tempo de recuperaГ§ГЈo (MTTR) alto em falhas

DECISГғO:
Adotar arquitetura de microsserviГ§os com os seguintes princГӯpios:
- Cada mГіdulo de funcionalidade como serviГ§o independente
- ComunicaГ§ГЈo via gRPC para serviГ§os internos, REST para APIs pГәblicas
- ContainerizaГ§ГЈo Docker com orquestraГ§ГЈo Kubernetes
- Service Mesh (Istio) para observabilidade e resiliГӘncia
- Banco de dados por serviГ§o (Database-per-service)

OPГҮГ•ES CONSIDERADAS:
- OpГ§ГЈo A: MonГіlito modularizado - PrГіs: Simples deploy, debugging fГЎcil. Contras: Escalabilidade limitada, acoplamento.
- OpГ§ГЈo B: MicrosserviГ§os completos - PrГіs: Escalabilidade independente, resiliГӘncia. Contras: Complexidade operacional.
- OpГ§ГЈo C: Arquitetura hГӯbrida (microsserviГ§os crГӯticos) - PrГіs: Balanceamento. Contras: Gerenciamento misto.

OpГ§ГЈo escolhida: B - Justificativa: A plataforma jГЎ possui mГіdulos claramente separГЎveis
(LLM Orchestrator, Agent Factory, Memory Bank, etc.) e a escalabilidade independente
Г© crГӯtica para o modelo de negГіcio.

CONSEQUГҠNCIAS:
Positivas:
- Deploy independente por squad
- Escalabilidade granular
- Stack tecnolГіgica adequada por serviГ§o
- Isolamento de falhas

Negativas:
- Complexidade de debugging distribuГӯdo
- Overhead de comunicaГ§ГЈo entre serviГ§os
- Gerenciamento de transaГ§Гөes distribuГӯdas

Riscos:
- LatГӘncia aumentada em chamadas entre serviГ§os
- Gerenciamento de consistГӘncia eventual
- MitigaГ§ГЈo: Circuit breakers, caching estratГ©gico, design assГӯncrono

REVISГғO: 2024-04-15 (3 meses para avaliaГ§ГЈo de performance)
```

```proto
// protos/services.proto
syntax = "proto3";

package diyapp.v12;

import "google/protobuf/timestamp.proto";

// Common Types
message HealthCheckRequest {
  string service = 1;
}

message HealthCheckResponse {
  enum ServingStatus {
    UNKNOWN = 0;
    SERVING = 1;
    NOT_SERVING = 2;
  }
  ServingStatus status = 1;
  string version = 2;
  int64 timestamp = 3;
}

// LLM Orchestrator Service
service LLMOrchestrator {
  rpc ProcessPrompt(PromptRequest) returns (PromptResponse);
  rpc StreamResponse(StreamRequest) returns (stream StreamChunk);
  rpc HealthCheck(HealthCheckRequest) returns (HealthCheckResponse);
}

message PromptRequest {
  string prompt = 1;
  string model_id = 2;
  map<string, string> parameters = 3;
  string session_id = 4;
  int32 max_tokens = 5;
  float temperature = 6;
}

message PromptResponse {
  string text = 1;
  string model_used = 2;
  int32 tokens_used = 3;
  float processing_time = 4;
  repeated string citations = 5;
  google.protobuf.Timestamp generated_at = 6;
}

message StreamRequest {
  string prompt = 1;
  string model_id = 2;
  string session_id = 3;
}

message StreamChunk {
  string chunk = 1;
  bool is_final = 2;
  string chunk_id = 3;
}

// Agent Factory Service
service AgentFactory {
  rpc CreateAgent(AgentCreationRequest) returns (AgentCreationResponse);
  rpc UpdateAgent(AgentUpdateRequest) returns (AgentUpdateResponse);
  rpc ExecuteAgent(AgentExecutionRequest) returns (AgentExecutionResponse);
  rpc HealthCheck(HealthCheckRequest) returns (HealthCheckResponse);
}

message AgentCreationRequest {
  string agent_type = 1;
  string name = 2;
  string description = 3;
  map<string, string> configuration = 4;
  repeated string capabilities = 5;
  string owner_squad = 6;
}

message AgentCreationResponse {
  string agent_id = 1;
  string status = 2;
  google.protobuf.Timestamp created_at = 3;
  string deployment_url = 4;
  repeated string warnings = 5;
}

message AgentUpdateRequest {
  string agent_id = 1;
  map<string, string> updates = 2;
  string update_reason = 3;
}

message AgentUpdateResponse {
  string agent_id = 1;
  string status = 2;
  google.protobuf.Timestamp updated_at = 3;
  string version = 4;
}

message AgentExecutionRequest {
  string agent_id = 1;
  string action = 2;
  bytes payload = 3;
  string session_id = 4;
}

message AgentExecutionResponse {
  string execution_id = 1;
  string status = 2;
  bytes result = 3;
  google.protobuf.Timestamp executed_at = 4;
  float execution_time = 5;
}

// Memory Bank Service
service MemoryBank {
  rpc StoreMemory(MemoryStoreRequest) returns (MemoryStoreResponse);
  rpc RetrieveMemory(MemoryRetrieveRequest) returns (MemoryRetrieveResponse);
  rpc SearchMemories(MemorySearchRequest) returns (MemorySearchResponse);
  rpc HealthCheck(HealthCheckRequest) returns (HealthCheckResponse);
}

message MemoryStoreRequest {
  string key = 1;
  bytes value = 2;
  string namespace = 3;
  int32 ttl_seconds = 4;
  map<string, string> metadata = 5;
  repeated float embedding = 6;
}

message MemoryStoreResponse {
  string memory_id = 1;
  bool success = 2;
  google.protobuf.Timestamp stored_at = 3;
  int64 size_bytes = 4;
}

message MemoryRetrieveRequest {
  string key = 1;
  string namespace = 2;
  bool include_metadata = 3;
}

message MemoryRetrieveResponse {
  bytes value = 1;
  map<string, string> metadata = 2;
  google.protobuf.Timestamp stored_at = 3;
  google.protobuf.Timestamp accessed_at = 4;
}

message MemorySearchRequest {
  repeated float query_embedding = 1;
  string namespace = 2;
  int32 limit = 3;
  float similarity_threshold = 4;
}

message MemorySearchResponse {
  repeated MemoryResult results = 1;
  int32 total_matches = 2;
  float search_time = 3;
}

message MemoryResult {
  string key = 1;
  float similarity = 2;
  bytes value = 3;
  map<string, string> metadata = 4;
}

// Model Registry Service
service ModelRegistry {
  rpc RegisterModel(ModelRegistrationRequest) returns (ModelRegistrationResponse);
  rpc GetModel(ModelRequest) returns (ModelResponse);
  rpc ListModels(ModelListRequest) returns (ModelListResponse);
  rpc HealthCheck(HealthCheckRequest) returns (HealthCheckResponse);
}

message ModelRegistrationRequest {
  string model_id = 1;
  string model_type = 2;
  string version = 3;
  string storage_path = 4;
  map<string, string> metadata = 5;
  repeated string capabilities = 6;
  ModelRequirements requirements = 7;
}

message ModelRequirements {
  int32 min_memory_mb = 1;
  bool gpu_required = 2;
  float min_gpu_memory_gb = 3;
  repeated string supported_frameworks = 4;
}

message ModelRegistrationResponse {
  string registration_id = 1;
  string status = 2;
  google.protobuf.Timestamp registered_at = 3;
  string validation_report = 4;
}

message ModelRequest {
  string model_id = 1;
  string version = 2;
}

message ModelResponse {
  string model_id = 1;
  string version = 2;
  string storage_path = 3;
  map<string, string> metadata = 4;
  ModelStatus status = 5;
  ModelMetrics metrics = 6;
}

message ModelStatus {
  bool is_available = 1;
  string deployment_status = 2;
  google.protobuf.Timestamp last_health_check = 3;
}

message ModelMetrics {
  float avg_latency_ms = 1;
  float success_rate = 2;
  int64 total_requests = 3;
  google.protobuf.Timestamp last_updated = 4;
}

message ModelListRequest {
  string model_type = 1;
  bool only_available = 2;
  int32 limit = 3;
  int32 offset = 4;
}

message ModelListResponse {
  repeated ModelResponse models = 1;
  int32 total_count = 2;
  bool has_more = 3;
}
```

```javascript
require('dotenv').config();
require('express-async-errors');
const express = require('express');
const helmet = require('helmet');
const cors = require('cors');
const compression = require('compression');
const swaggerUi = require('swagger-ui-express');

const config = require('./config');
const logger = require('./utils/logger');
const metrics = require('./utils/metrics');
const errorHandler = require('./middleware/errorHandler');
const rateLimit = require('./middleware/rateLimit');

// Import services
const authRoutes = require('./services/auth/routes');
const taskRoutes = require('./services/tasks/routes');
const logRoutes = require('./services/logs/routes');

const app = express();

// Security middleware
app.use(helmet({
  contentSecurityPolicy: {
    directives: {
      defaultSrc: ["'self'"],
      styleSrc: ["'self'", "'unsafe-inline'"],
      scriptSrc: ["'self'"],
    },
  },
}));

// CORS configuration
app.use(cors({
  origin: config.cors.origin,
  credentials: true,
}));

// Compression
app.use(compression());

// JSON parsing with size limit
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true, limit: '10mb' }));

// Request logging middleware
app.use((req, res, next) => {
  const correlationId = req.headers['x-correlation-id'] || require('uuid').v4();
  req.correlationId = correlationId;
  
  logger.info({
    correlationId,
    method: req.method,
    url: req.url,
    userAgent: req.get('user-agent'),
    ip: req.ip,
  }, 'Request received');
  
  res.setHeader('X-Correlation-ID', correlationId);
  next();
});

// Metrics middleware
app.use(metrics.httpMiddleware);

// Health check endpoint
app.get('/health', (req, res) => {
  res.status(200).json({
    status: 'healthy',
    timestamp: new Date().toISOString(),
    uptime: process.uptime(),
    services: {
      database: 'connected', // Would check actual connection
      redis: 'connected',
    },
  });
});

// Metrics endpoint
app.get('/metrics', async (req, res) => {
  res.set('Content-Type', metrics.register.contentType);
  res.end(await metrics.register.metrics());
});

// API Documentation
if (process.env.NODE_ENV !== 'production') {
  const swaggerDocument = require('./api-docs/swagger.json');
  app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(swaggerDocument));
}

// Apply rate limiting to all API routes
app.use('/api/v1', rateLimit.apiLimiter);

// API Routes
app.use('/api/v1/auth', authRoutes);
app.use('/api/v1/tasks', taskRoutes);
app.use('/api/v1/logs', logRoutes);

// 404 handler
app.use((req, res) => {
  logger.warn({
    correlationId: req.correlationId,
    method: req.method,
    url: req.url,
  }, 'Route not found');
  
  res.status(404).json({
    error: 'Not Found',
    message: `Cannot ${req.method} ${req.url}`,
    correlationId: req.correlationId,
  });
});

// Global error handler
app.use(errorHandler);

// Start server
const PORT = config.server.port || 3000;
const server = app.listen(PORT, () => {
  logger.info(`Server running on port ${PORT}`);
  logger.info(`Environment: ${process.env.NODE_ENV}`);
  logger.info(`API Documentation: http://localhost:${PORT}/api-docs`);
});

// Graceful shutdown
const gracefulShutdown = () => {
  logger.info('Received shutdown signal, closing server...');
  
  server.close(() => {
    logger.info('HTTP server closed');
    
    // Close database connections
    const { pool } = require('./config/database');
    pool.end(() => {
      logger.info('Database connections closed');
      process.exit(0);
    });
  });

  // Force shutdown after timeout
  setTimeout(() => {
    logger.error('Could not close connections in time, forcefully shutting down');
    process.exit(1);
  }, 10000);
};

process.on('SIGTERM', gracefulShutdown);
process.on('SIGINT', gracefulShutdown);

// Unhandled promise rejection handler
process.on('unhandledRejection', (reason, promise) => {
```

            WHEN SAFE_CAST(status_code AS INT64) = 200 THEN 'success'
            ELSE 'other'
        END AS status_category,
        
        -- Endpoints padronizados
        REGEXP_REPLACE(
            REGEXP_REPLACE(LOWER(endpoint), r'\\d+', '{id}'),
            r'/[^/]+$', '/{resource}'
        ) AS endpoint_pattern,
        
        -- MГ©todos HTTP vГЎlidos
        CASE 
            WHEN UPPER(http_method) IN ('GET', 'POST', 'PUT', 'DELETE', 'PATCH', 'OPTIONS', 'HEAD')
            THEN UPPER(http_method)
            ELSE 'UNKNOWN'
        END AS http_method,
        
        -- Dados de sistema
        CAST(server_id AS STRING) AS server_id,
        CAST(region AS STRING) AS region,
        CAST(app_version AS STRING) AS app_version,
        
        -- MГ©tricas de carga (com validaГ§ГЈo)
        CASE 
            WHEN SAFE_CAST(concurrent_users AS INT64) BETWEEN 0 AND 10000 
            THEN SAFE_CAST(concurrent_users AS INT64)
            ELSE NULL 
        END AS concurrent_users,
        
        CASE 
            WHEN SAFE_CAST(memory_usage_mb AS FLOAT64) BETWEEN 0 AND 32000 
            THEN SAFE_CAST(memory_usage_mb AS FLOAT64)
            ELSE NULL 
        END AS memory_usage_mb,
        
        CASE 
            WHEN SAFE_CAST(cpu_usage_percent AS FLOAT64) BETWEEN 0 AND 100 
            THEN SAFE_CAST(cpu_usage_percent AS FLOAT64)
            ELSE NULL 
        END AS cpu_usage_percent,
        
        -- Dados de negГіcio
        CAST(feature_name AS STRING) AS feature_name,
        CAST(action_type AS STRING) AS action_type,
        
        -- Metadados
        ingested_at,
        batch_id
        
    FROM {{ ref('raw_diyapp_telemetry') }}
    WHERE event_timestamp IS NOT NULL
      AND event_id IS NOT NULL
)

SELECT 
    *,
    -- Flags de qualidade
    CASE 
        WHEN response_time_ms IS NULL THEN 1 
        ELSE 0 
    END AS is_response_time_missing,
    
    CASE 
        WHEN status_code IS NULL THEN 1 
        ELSE 0 
    END AS is_status_code_missing,
    
    CASE 
        WHEN concurrent_users IS NULL THEN 1 
        ELSE 0 
    END AS is_concurrent_users_missing
    
FROM raw_data
"""

daily_performance_metrics_sql = """
{{ config(
    materialized='table',
    schema='ops_marts',
    alias='daily_performance_metrics',
    tags=['marts', 'ops', 'performance']
) }}

WITH hourly_metrics AS (
    SELECT
        event_date,
        event_hour,
        region,
        server_id,
        endpoint_pattern,
        http_method,
        status_category,
        
        -- Contagem de eventos
        COUNT(*) AS total_requests,
        COUNT(DISTINCT session_id) AS unique_sessions,
        COUNT(DISTINCT user_id) AS unique_users,
        
        -- MГ©tricas de performance
        AVG(response_time_ms) AS avg_response_time_ms,
        PERCENTILE_CONT(response_time_ms, 0.5) OVER (
            PARTITION BY event_date, event_hour, endpoint_pattern
        ) AS p50_response_time_ms,
        PERCENTILE_CONT(response_time_ms, 0.95) OVER (
            PARTITION BY event_date, event_hour, endpoint_pattern
        ) AS p95_response_time_ms,
        PERCENTILE_CONT(response_time_ms, 0.99) OVER (
            PARTITION BY event_date, event_hour, endpoint_pattern
        ) AS p99_response_time_ms,
        
        -- Taxa de erro
        SUM(CASE WHEN status_category IN ('client_error', 'server_error') THEN 1 ELSE 0 END) 
        / COUNT(*) AS error_rate,
        
        -- MГ©tricas de carga
        AVG(concurrent_users) AS avg_concurrent_users,
        MAX(concurrent_users) AS peak_concurrent_users,
        AVG(memory_usage_mb) AS avg_memory_usage_mb,
        AVG(cpu_usage_percent) AS avg_cpu_usage_percent
        
    FROM {{ ref('stg_diyapp_telemetry') }}
    WHERE event_date >= DATE_SUB(CURRENT_DATE(), INTERVAL 30 DAY)
    GROUP BY 1, 2, 3, 4, 5, 6, 7
),

daily_aggregates AS (
    SELECT
        event_date,
        region,
        
        -- KPIs diГЎrios
        SUM(total_requests) AS daily_total_requests,
        COUNT(DISTINCT unique_users) AS daily_unique_users,
        
        -- SLA Performance (response time < 500ms)
        SUM(CASE WHEN avg_response_time_ms < 500 THEN total_requests ELSE 0 END) 
        / SUM(total_requests) AS sla_compliance_rate,
        
        -- Disponibilidade (error rate < 1%)
        CASE 
            WHEN SUM(CASE WHEN status_category IN ('client_error', 'server_error') THEN total_requests ELSE 0 END) 
                 / SUM(total_requests) < 0.01 
            THEN 1 
            ELSE 0 
        END AS availability_met,
        
        -- Picos de carga
        MAX(peak_concurrent_users) AS daily_peak_concurrent_users,
        
        -- Resource utilization
        AVG(avg_memory_usage_mb) AS avg_daily_memory_usage,
        AVG(avg_cpu_usage_percent) AS avg_daily_cpu_usage
        
    FROM hourly_metrics
    GROUP BY 1, 2
),

endpoint_performance AS (
    SELECT
        event_date,
        endpoint_pattern,
        
        -- Performance por endpoint
        AVG(avg_response_time_ms) AS endpoint_avg_response_time,
        AVG(p95_response_time_ms) AS endpoint_p95_response_time,
        
        -- Volume por endpoint
        SUM(total_requests) AS endpoint_total_requests,
        
        -- Taxa de erro por endpoint
        SUM(CASE WHEN status_category IN ('client_error', 'server_error') THEN total_requests ELSE 0 END)
        / SUM(total_requests) AS endpoint_error_rate
        
    FROM hourly_metrics
    GROUP BY 1, 2
)

SELECT
    d.event_date,
    d.region,
    d.daily_total_requests,
    d.daily_unique_users,
    d.sla_compliance_rate,
    d.availability_met,
    d.daily_peak_concurrent_users,
    d.avg_daily_memory_usage,
    d.avg_daily_cpu_usage,
    
    -- Top 5 endpoints mais lentos
    ARRAY_AGG(
        STRUCT(
            e.endpoint_pattern,
            e.endpoint_avg_response_time,
            e.endpoint_total_requests
        )
        ORDER BY e.endpoint_avg_response_time DESC 
        LIMIT 5
    ) AS top_slow_endpoints,
    
    -- Top 5 endpoints com mais erros
    ARRAY_AGG(
        STRUCT(
            e.endpoint_pattern,
            e.endpoint_error_rate,
            e.endpoint_total_requests
        )
        ORDER BY e.endpoint_error_rate DESC 
        LIMIT 5
    ) AS top_error_endpoints,
    
    -- Metadados
    CURRENT_TIMESTAMP() AS metrics_calculated_at
    
FROM daily_aggregates d
LEFT JOIN endpoint_performance e 
    ON d.event_date = e.event_date
GROUP BY 1, 2, 3, 4, 5, 6, 7, 8, 9
"""

```javascript
const logger = require('./logger');

class TokenCounter {
  constructor() {
    // Approximate token counting (4 chars вүҲ 1 token for English)
    this.CHARS_PER_TOKEN = 4;
  }

  countTokens(text) {
    if (!text || typeof text !== 'string') return 0;
    
    // Remove extra whitespace and count characters
    const cleanedText = text.replace(/\s+/g, ' ').trim();
    const charCount = cleanedText.length;
    
    // Approximate token count
    const approximateTokens = Math.ceil(charCount / this.CHARS_PER_TOKEN);
    
    logger.debug(`Token count: ${approximateTokens} for ${charCount} chars`);
    return approximateTokens;
  }

  calculateCost(tokens, model) {
    const pricing = {
      'gpt-4': { input: 0.03, output: 0.06 },
      'gpt-4-32k': { input: 0.06, output: 0.12 },
      'gpt-3.5-turbo': { input: 0.0015, output: 0.002 },
      'gpt-3.5-turbo-16k': { input: 0.003, output: 0.004 },
      'claude-2': { input: 0.01102, output: 0.03268 },
      'claude-instant': { input: 0.00163, output: 0.00551 },
      'claude-3-opus': { input: 0.015, output: 0.075 },
      'claude-3-sonnet': { input: 0.003, output: 0.015 },
      'claude-3-haiku': { input: 0.00025, output: 0.00125 },
      'gemini-pro': { input: 0.0005, output: 0.0015 }
    };

    const modelPricing = pricing[model] || pricing['gpt-3.5-turbo'];
    const costPer1K = (modelPricing.input + modelPricing.output) / 2;
    
    return (tokens / 1000) * costPer1K;
  }

  validateTokenLimit(text, maxTokens = 4000) {
    const tokens = this.countTokens(text);
    return {
      isValid: tokens <= maxTokens,
      tokenCount: tokens,
      exceedsBy: tokens > maxTokens ? tokens - maxTokens : 0
    };
  }
}

module.exports = new TokenCounter();
```

    try {
      const authHeader = req.headers.authorization;
      
      if (!authHeader?.startsWith('Bearer ')) {
        return res.status(401).json({ 
          error: 'Authentication required',
          code: 'AUTH_HEADER_MISSING'
        });
      }

      const token = authHeader.substring(7);
      
      // Check blacklist first
      const isBlacklisted = await this.redis.get(`blacklist:${token}`);
      if (isBlacklisted) {
        return res.status(401).json({ 
          error: 'Token revoked',
          code: 'TOKEN_REVOKED'
        });
      }

      // Verify JWT
      const decoded = jwt.verify(token, process.env.JWT_SECRET, {
        algorithms: ['RS256'],
        issuer: 'diyapp-v12',
        audience: 'api.diyapp.com'
      });

      // Check token expiration
      if (decoded.exp * 1000 < Date.now()) {
        return res.status(401).json({ 
          error: 'Token expired',
          code: 'TOKEN_EXPIRED'
        });
      }

      // Rate limiting per user
      const rateLimitKey = `ratelimit:${decoded.sub}:${req.path}`;
      const requests = await this.redis.incr(rateLimitKey);
      
      if (requests === 1) {
        await this.redis.expire(rateLimitKey, 60); // 1 minute window
      }
      
      if (requests > this.getRateLimit(req.path)) {

        });

      req.user = {
        id: decoded.sub,
        roles: decoded.roles || [],
        permissions: decoded.permissions || [],
        attributes: decoded.attributes || {},
        sessionId: decoded.jti

      this.logAuthEvent(req.user.id, 'SUCCESS', req.ip, req.path);

class DataProtectionFilter(logging.Filter):
    """Filtro para anonimizar dados pessoais em logs"""
    
    PERSONAL_DATA_PATTERNS = {
        'cpf': r'\b\d{3}\.\d{3}\.\d{3}-\d{2}\b',
        'cnpj': r'\b\d{2}\.\d{3}\.\d{3}/\d{4}-\d{2}\b',
        'email': r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b',
        'phone': r'\b(?:\+55\s?)?(?:\(?\d{2}\)?\s?)?\d{4,5}-?\d{4}\b',
        'credit_card': r'\b\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}\b'
    }
    
    def filter(self, record: logging.LogRecord) -> bool:
        """Filtra e anonimiza dados pessoais no registro de log"""
        
        if hasattr(record, 'msg') and record.msg:
            record.msg = self._anonymize_text(str(record.msg))
        
        if hasattr(record, 'args') and record.args:
            if isinstance(record.args, dict):
                record.args = self._anonymize_dict(record.args)
            elif isinstance(record.args, tuple):
                record.args = tuple(self._anonymize_text(str(arg)) 
                                  if isinstance(arg, str) else arg 
                                  for arg in record.args)
        
        return True
    
    def _anonymize_text(self, text: str) -> str:
        """Anonimiza dados pessoais em texto"""
        
        for data_type, pattern in self.PERSONAL_DATA_PATTERNS.items():
            if data_type == 'email':
                # Preserva domГӯnio para debugging, anonimiza usuГЎrio
                text = re.sub(
                    pattern, 
                    lambda m: f"***@{m.group().split('@')[1]}", 
                    text
                )
            else:
                text = re.sub(pattern, f'[{data_type.upper()}_ANONYMIZED]', text)
        
        return text
    
    def _anonymize_dict(self, data: Dict) -> Dict:
        """Anonimiza dados pessoais em dicionГЎrio"""
        
        anonymized = {}
        for key, value in data.items():
            if isinstance(value, str):
                anonymized[key] = self._anonymize_text(value)
            elif isinstance(value, dict):
                anonymized[key] = self._anonymize_dict(value)
            elif isinstance(value, list):
                anonymized[key] = [
                    self._anonymize_text(item) if isinstance(item, str) else item
                    for item in value
                ]
            else:
                anonymized[key] = value
        
        return anonymized

class RetentionPolicyManager:
    """Gerenciador de polГӯticas de retenГ§ГЈo de logs"""
    
    def __init__(self, retention_days: int = 180):
        self.retention_days = retention_days
        self.cleanup_thread = None
        self.running = False
        
    def start_cleanup_scheduler(self):
        """Inicia agendador de limpeza automГЎtica"""
        
        self.running = True
        self.cleanup_thread = threading.Thread(
            target=self._cleanup_scheduler,
            daemon=True
        )
        self.cleanup_thread.start()
    
    def _cleanup_scheduler(self):
        """Agendador que executa limpeza diГЎria"""
        
        import time
        while self.running:
            # Executa Г  01:00 todos os dias
            now = datetime.now()
            next_run = now.replace(hour=1, minute=0, second=0, microsecond=0)
            if now > next_run:
                next_run += timedelta(days=1)
            
            sleep_seconds = (next_run - now).total_seconds()
            time.sleep(sleep_seconds)
            
            self.cleanup_old_logs()
    
    def cleanup_old_logs(self):
        """Remove logs mais antigos que o perГӯodo de retenГ§ГЈo"""
        
        cutoff_date = datetime.now() - timedelta(days=self.retention_days)
        cutoff_timestamp = cutoff_date.timestamp()
        
        # Limpar logs do sistema de arquivos
        log_dir = Path("logs")
        if log_dir.exists():
            for log_file in log_dir.glob("*.log"):
                if log_file.stat().st_mtime < cutoff_timestamp:
                    log_file.unlink()
        
        # Limpar logs do banco de dados
        self._cleanup_database_logs(cutoff_date)
    
    def _cleanup_database_logs(self, cutoff_date: datetime):
        """Remove logs antigos do banco de dados"""
        
        try:
            conn = sqlite3.connect('logs/application.db')
            cursor = conn.cursor()
            
            cursor.execute('''
                DELETE FROM audit_logs 
                WHERE timestamp < ?
            ''', (cutoff_date.isoformat(),))
            
            cursor.execute('''
                DELETE FROM data_access_logs 
                WHERE access_time < ?
            ''', (cutoff_date.isoformat(),))
            
            conn.commit()
            conn.close()
            
            logging.info(f"Cleanup completed: removed logs older than {cutoff_date}")
            
        except Exception as e:
            logging.error(f"Failed to cleanup database logs: {e}")
