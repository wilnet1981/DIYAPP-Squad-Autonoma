## 1. ADR-001: Arquitetura Baseada em Eventos

**docs/architecture/ADR-001-event-driven-architecture.md**

```markdown
# ADR-001: Arquitetura Baseada em Eventos para V12

## Status
Aceita

## Data
2024-01-15

## Contexto
A V11 do DIYAPP enfrenta problemas de acoplamento forte entre módulos, dificuldade de escalar componentes individualmente e complexidade na manutenção de transações distribuídas. Necessitamos de uma arquitetura que permita:
- Desenvolvimento independente de squads
- Escalabilidade granular
- Resiliência a falhas
- Evolução contínua sem breaking changes

## Decisão
Adotar arquitetura baseada em eventos com os seguintes componentes:
1. **Event Bus**: Apache Kafka como backbone de eventos
2. **Event Store**: MongoDB para persistência de eventos
3. **Serviços Stateless**: Cada microsserviço processa eventos de forma independente
4. **API Gateway**: Kong para roteamento e rate limiting
5. **Service Mesh**: Istio para comunicação entre serviços

## Opções Consideradas

### Opção A: Arquitetura Monolítica com Módulos
- Prós: Simples de desenvolver inicialmente, transações ACID fáceis
- Contras: Acoplamento forte, difícil escalar, deploy único
- Rejeitada: Não atende requisitos de escalabilidade e autonomia de squads

### Opção B: Microsserviços REST Síncronos
- Prós: Separação clara de responsabilidades, deploy independente
- Contras: Acoplamento temporal, cascata de falhas, complexidade de transações
- Rejeitada: Problemas de resiliência e latência em cadeias longas

### Opção C: Arquitetura Baseada em Eventos (ESCOLHIDA)
- Prós: Desacoplamento temporal, resiliência, escalabilidade independente
- Contras: Complexidade inicial, eventual consistency, debugging distribuído
- Justificativa: Atende todos requisitos de autonomia e escalabilidade

## Consequências
### Positivas
- Squads podem desenvolver e deployar independentemente
- Sistema tolerante a falhas parciais
- Escalabilidade granular por componente
- Evolução de schemas sem breaking changes

### Negativas
- Maior complexidade operacional
- Necessidade de monitoramento distribuído
- Eventual consistency em alguns casos
- Maior latência em alguns cenários

### Riscos
1. **Perda de eventos**: Mitigado com persistent log no Kafka e dead letter queues
2. **Duplicação de eventos**: Mitigado com idempotency keys e deduplication
3. **Ordem de eventos**: Mitigado com partition keys e sequential processing por aggregate

## Revisão
Reavaliar em 6 meses após implementação completa
```

## 2. Template de Microsserviço

**templates/microservice/src/index.js**

```javascript
const express = require('express');
const { Kafka } = require('kafkajs');
const { metrics, tracer } = require('./utils/opentelemetry');
const logger = require('./utils/logger');
const healthCheck = require('./api/health-check');
const config = require('./config');

class Microservice {
  constructor() {
    this.app = express();
    this.port = config.port;
    this.kafka = new Kafka({
      clientId: config.serviceName,
      brokers: config.kafkaBrokers,
    });
    this.producer = this.kafka.producer();
    this.consumer = this.kafka.consumer({ groupId: config.serviceName });
  }

  async initialize() {
    try {
      // Setup OpenTelemetry
      await this.setupTelemetry();
      
      // Connect to Kafka
      await this.producer.connect();
      await this.consumer.connect();
      
      // Setup middleware
      this.setupMiddleware();
      
      // Setup routes
      this.setupRoutes();
      
      // Subscribe to events
      await this.subscribeToEvents();
      
      logger.info(`${config.serviceName} initialized successfully`);
    } catch (error) {
      logger.error('Failed to initialize microservice:', error);
      process.exit(1);
    }
  }

  setupMiddleware() {
    this.app.use(express.json());
    this.app.use(require('./api/middleware/requestLogger'));
    this.app.use(require('./api/middleware/errorHandler'));
  }

  setupRoutes() {
    this.app.use('/health', healthCheck);
    this.app.use('/api/v1', require('./api/routes'));
  }

  async setupTelemetry() {
    // Export metrics every 10 seconds
    setInterval(() => {
      metrics.getMeterProvider().getMeter('default').collect();
    }, 10000);
  }

  async subscribeToEvents() {
    const topics = config.subscribedTopics;
    
    await this.consumer.subscribe({ topics, fromBeginning: false });
    
    await this.consumer.run({
      eachMessage: async ({ topic, partition, message }) => {
        const span = tracer.startSpan(`process_${topic}`);
        
        try {
          const event = JSON.parse(message.value.toString());
          await this.handleEvent(topic, event, message.headers);
          span.setStatus({ code: 1 }); // OK
        } catch (error) {
          logger.error(`Error processing event from ${topic}:`, error);
          span.setStatus({ code: 2, message: error.message }); // ERROR
          await this.sendToDLQ(topic, message, error);
        } finally {
          span.end();
        }
      },
    });
  }

  async handleEvent(topic, event, headers) {
    // To be implemented by specific service
    logger.info(`Processing event from ${topic}:`, event);
  }

  async sendToDLQ(topic, message, error) {
    const dlqEvent = {
      originalTopic: topic,
      originalMessage: message.value.toString(),
      error: error.message,
      timestamp: new Date().toISOString(),
    };
    
    await this.producer.send({
      topic: `${topic}.dlq`,
      messages: [{ value: JSON.stringify(dlqEvent) }],
    });
  }

  async publishEvent(topic, event, headers = {}) {
    const span = tracer.startSpan(`publish_${topic}`);
    
    try {
      const message = {
        value: JSON.stringify(event),
        headers: {
          ...headers,
          'correlation-id': span.spanContext().traceId,
          'idempotency-key': headers['idempotency-key'] || require('crypto').randomUUID(),
          'timestamp': new Date().toISOString(),
        },
      };
      
      await this.producer.send({ topic, messages: [message] });
      span.setStatus({ code: 1 });
      logger.info(`Event published to ${topic}:`, event);
    } catch (error) {
      span.setStatus({ code: 2, message: error.message });
      logger.error(`Failed to publish event to ${topic}:`, error);
      throw error;
    } finally {
      span.end();
    }
  }

  start() {
    this.server = this.app.listen(this.port, () => {
      logger.info(`${config.serviceName} listening on port ${this.port}`);
    });
  }

  async shutdown() {
    logger.info('Shutting down microservice...');
    
    if (this.server) {
      this.server.close();
    }
    
    await this.producer.disconnect();
    await this.consumer.disconnect();
    
    process.exit(0);
  }
}

// Graceful shutdown
process.on('SIGTERM', () => {
  const microservice = require('./index').default;
  microservice.shutdown();
});

process.on('SIGINT', () => {
  const microservice = require('./index').default;
  microservice.shutdown();
});

module.exports = Microservice;
```

const { NodeSDK } = require('@opentelemetry/sdk-node');
const { getNodeAutoInstrumentations } = require('@opentelemetry/auto-instrumentations-node');
const { Resource } = require('@opentelemetry/resources');
const { SemanticResourceAttributes } = require('@opentelemetry/semantic-conventions');
const { OTLPTraceExporter } = require('@opentelemetry/exporter-trace-otlp-grpc');
const { OTLPMetricExporter } = require('@opentelemetry/exporter-metrics-otlp-grpc');
const { PeriodicExportingMetricReader } = require('@opentelemetry/sdk-metrics');
const { diag, DiagConsoleLogger, DiagLogLevel } = require('@opentelemetry/api');
const { ExpressInstrumentation } = require('@opentelemetry/instrumentation-express');
const { KafkaJsInstrumentation } = require('opentelemetry-instrumentation-kafkajs');
const { HttpInstrumentation } = require('@opentelemetry/instrumentation-http');
const { MongoDBInstrumentation } = require('@opentelemetry/instrumentation-mongodb');
const { RedisInstrumentation } = require('@opentelemetry/instrumentation-redis');

// Set up logging
diag.setLogger(new DiagConsoleLogger(), DiagLogLevel.INFO);

const resource = new Resource({
  [SemanticResourceAttributes.SERVICE_NAME]: process.env.SERVICE_NAME || 'unknown-service',
  [SemanticResourceAttributes.SERVICE_VERSION]: process.env.VERSION || '1.0.0',
  [SemanticResourceAttributes.DEPLOYMENT_ENVIRONMENT]: process.env.NODE_ENV || 'development',
});

const traceExporter = new OTLPTraceExporter({
  url: process.env.OTLP_ENDPOINT || 'http://jaeger:4317',
});

const metricExporter = new OTLPMetricExporter({
  url: process.env.OTLP_METRICS_ENDPOINT || 'http://prometheus:4317',
});

const metricReader = new PeriodicExportingMetricReader({
  exporter: metricExporter,
  exportIntervalMillis: 10000,
});

const sdk = new NodeSDK({
  resource,
  traceExporter,
  metricReader,
  instrumentations: [
    getNodeAutoInstrumentations({
      '@opentelemetry/instrumentation-express': {
        enabled: true,
      },
      '@opentelemetry/instrumentation-http': {
        enabled: true,
      },
      '@opentelemetry/instrumentation-mongodb': {
        enabled: true,
      },
      '@opentelemetry/instrumentation-redis': {
        enabled: true,
      },
    }),
    new ExpressInstrumentation(),
    new KafkaJsInstrumentation(),
    new HttpInstrumentation(),
    new MongoDBInstrumentation(),
    new RedisInstrumentation(),
  ],
});

sdk.start();

// Export tracer and metrics for use in application code
const { trace, metrics } = require('@opentelemetry/api');

module.exports = {
  tracer: trace.getTracer('diyapp-tracer'),
  metrics: metrics,
  sdk,
};

  CMD node health-check.js

import Brakes from 'brakes';
import { logger } from '../observability/Logger';

export interface CircuitBreakerOptions {
  name: string;
  timeout: number;
  threshold: number;
  resetTimeout: number;
  fallback?: () => Promise<any>;
}

export class CircuitBreaker {
  private brake: Brakes;
  private name: string;
  private fallback: () => Promise<any>;

  constructor(options: CircuitBreakerOptions) {
    this.name = options.name;
    this.fallback = options.fallback || this.defaultFallback;
    
    this.brake = new Brakes({
      name: options.name,
      timeout: options.timeout,
      threshold: options.threshold,
      waitThreshold: options.threshold,
      circuitDuration: options.resetTimeout,
      statInterval: 2500,
    });

    this.setupEventListeners();
  }

  private setupEventListeners(): void {
    this.brake.on('circuitOpen', () => {
      logger.warn(`Circuit breaker ${this.name} opened`, { circuit: this.name });
    });

    this.brake.on('circuitClosed', () => {
      logger.info(`Circuit breaker ${this.name} closed`, { circuit: this.name });
    });

    this.brake.on('failure', (err: Error) => {
      logger.error(`Circuit breaker ${this.name} failure`, { 
        circuit: this.name, 
        error: err.message 
      });
    });

    this.brake.on('timeout', () => {
      logger.warn(`Circuit breaker ${this.name} timeout`, { circuit: this.name });
    });
  }

  private async defaultFallback(): Promise<any> {
    logger.warn(`Using fallback for ${this.name}`, { circuit: this.name });
    throw new Error(`Service ${this.name} unavailable - fallback not implemented`);
  }

  async execute<T>(fn: () => Promise<T>): Promise<T> {
    try {
      return await this.brake.exec(fn);
    } catch (error) {
      if (this.brake.isOpen()) {
        logger.warn(`Circuit ${this.name} is open, using fallback`, { circuit: this.name });
        return this.fallback();
      }
      throw error;
    }
  }

  getStats() {
    return this.brake.getStats();
  }
}

// Circuit Breaker Registry
export class CircuitBreakerRegistry {
  private static instance: CircuitBreakerRegistry;
  private breakers: Map<string, CircuitBreaker> = new Map();

  private constructor() {}

  static getInstance(): CircuitBreakerRegistry {
    if (!CircuitBreakerRegistry.instance) {
      CircuitBreakerRegistry.instance = new CircuitBreakerRegistry();
    }
    return CircuitBreakerRegistry.instance;
  }

  register(name: string, breaker: CircuitBreaker): void {
    this.breakers.set(name, breaker);
  }

  get(name: string): CircuitBreaker | undefined {
    return this.breakers.get(name);
  }

  getAll(): Map<string, CircuitBreaker> {
    return new Map(this.breakers);
  }

  getStats() {
    const stats: Record<string, any> = {};
    this.breakers.forEach((breaker, name) => {
      stats[name] = breaker.getStats();
    });
    return stats;
  }
}

// src/core/resilience/RetryStrategy.ts
export interface RetryOptions {
  maxAttempts: number;
  baseDelay: number;
  maxDelay: number;
  jitter: boolean;
  retryableErrors?: number[];
}

export class RetryStrategy {
  static async execute<T>(
    fn: () => Promise<T>,
    options: RetryOptions = {
      maxAttempts: 3,
      baseDelay: 100,
      maxDelay: 10000,
      jitter: true,
      retryableErrors: [408, 429, 500, 502, 503, 504]
    }
  ): Promise<T> {
    let lastError: Error;
    
    for (let attempt = 1; attempt <= options.maxAttempts; attempt++) {
      try {
        return await fn();
      } catch (error: any) {
        lastError = error;
        
        // Check if error is retryable
        const statusCode = error.response?.status || error.statusCode || error.code;
        const isRetryable = options.retryableErrors?.includes(statusCode) || 
                          error.message?.includes('timeout') ||
                          error.message?.includes('ECONNREFUSED');
        
        if (!isRetryable || attempt === options.maxAttempts) {
          throw error;
        }
        
        // Calculate delay with exponential backoff and jitter
        const delay = this.calculateDelay(attempt, options);
        await this.sleep(delay);
      }
    }
    
    throw lastError!;
  }

  private static calculateDelay(attempt: number, options: RetryOptions): number {
    const exponentialDelay = options.baseDelay * Math.pow(2, attempt - 1);
    const cappedDelay = Math.min(exponentialDelay, options.maxDelay);
    
    if (options.jitter) {
      const jitter = Math.random() * 0.3 * cappedDelay; // Up to 30% jitter
      return cappedDelay + jitter;
    }
    
    return cappedDelay;
  }

  private static sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

      threshold: config.CIRCUIT_BREAKER_THRESHOLD,

      },
    });

    CircuitBreakerRegistry.getInstance().register('database', this.circuitBreaker);
    
    this.setupEventListeners();
    this.testConnection();
  }

  static getInstance(): Database {
    if (!Database.instance) {
      Database.instance = new Database();
    }
    return Database.instance;
  }

  private setupEventListeners(): void {
    this.pool.on('connect', () => {
      logger.debug('New database connection established');
    });

    this.pool.on('error', (err) => {
      logger.error('Database pool error', { error: err.message });
    });
  }

  private async testConnection(): Promise<void> {
    try {
      const client = await this.pool.connect();
      await client.query('SELECT NOW()');
      client.release();
      logger.info('Database connection test successful');
    } catch (error: any) {
      logger.error('Database connection test failed', { error: error.message });
    }
  }

  async query<T = any>(text: string, params?: any[]): Promise<{ rows: T[]; rowCount: number }> {
    return this.circuitBreaker.execute(async () => {
      const start = Date.now();
      
      try {
        const result = await this.pool.query(text, params);
        const duration = Date.now() - start;
        
        logger.debug('Database query executed', {
          query: text,
          params: params ? JSON.stringify(params) : null,
          duration,
          rowCount: result.rowCount,
        });
        
        return result;
      } catch (error: any) {
        logger.error('Database query failed', {
          query: text,
          params: params ? JSON.stringify(params) : null,
          error: error.message,
        });
        throw error;
      }
    });
  }

  async transaction<T>(callback: (client: any) => Promise<T>): Promise<T> {
    const client = await this.pool.connect();
    
    try {
      await client.query('BEGIN');
      const result = await callback(client);
      await client.query('COMMIT');
      return result;
    } catch (error) {
      await client.query('ROLLBACK');
      throw error;
    } finally {
      client.release();
    }
  }

  async close(): Promise<void> {
    await this.pool.end();
    logger.info('Database pool closed');
  }
}

#!/usr/bin/env python3
"""
Configuração inicial do Data Warehouse no BigQuery
Cria datasets, tabelas e configurações de segurança
"""
import os
from google.cloud import bigquery
from google.cloud.exceptions import NotFound
import json
from datetime import datetime, timedelta
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class BigQuerySetup:
    def __init__(self, project_id, location='US'):
        self.client = bigquery.Client(project=project_id)
        self.project_id = project_id
        self.location = location
        
    def create_dataset(self, dataset_id, description, retention_days=90):
        """Cria dataset com política de retenção"""
        dataset_ref = self.client.dataset(dataset_id)
        
        try:
            self.client.get_dataset(dataset_ref)
            logger.info(f"Dataset {dataset_id} já existe")
            return
        except NotFound:
            pass
        
        dataset = bigquery.Dataset(dataset_ref)
        dataset.description = description
        dataset.location = self.location
        
        # Configuração de retenção (Compliance)
        dataset.default_table_expiration_ms = retention_days * 24 * 60 * 60 * 1000
        
        dataset = self.client.create_dataset(dataset)
        logger.info(f"Dataset criado: {dataset_id}")
        
        # Configura IAM para a squad
        self._setup_dataset_iam(dataset_id)
        
    def _setup_dataset_iam(self, dataset_id):
        """Configura permissões IAM por papel da squad"""
        dataset_ref = self.client.dataset(dataset_id)
        
        # Policy para Data Engineer (acesso total)
        policy = self.client.get_iam_policy(dataset_ref)
        
        # PM: Apenas leitura nas marts de produto
        policy.bindings.append({
            "role": "roles/bigquery.dataViewer",
            "members": [
                f"serviceAccount:pm-{self.project_id}.iam.gserviceaccount.com"
            ],
            "condition": {
                "title": "ProductMartAccess",
                "description": "Acesso apenas às marts de produto",
                "expression": f"resource.name.startsWith('projects/{self.project_id}/datasets/{dataset_id}/tables/mart_product')"
            }
        })
        
        # AI Ops: Leitura nas marts de AI
        policy.bindings.append({
            "role": "roles/bigquery.dataViewer",
            "members": [
                f"serviceAccount:aiops-{self.project_id}.iam.gserviceaccount.com"
            ],
            "condition": {
                "title": "AIMartAccess",
                "description": "Acesso apenas às marts de AI",
                "expression": f"resource.name.startsWith('projects/{self.project_id}/datasets/{dataset_id}/tables/mart_ai')"
            }
        })
        
        # Compliance: Auditoria apenas
        policy.bindings.append({
            "role": "roles/bigquery.metadataViewer",
            "members": [
                f"serviceAccount:compliance-{self.project_id}.iam.gserviceaccount.com"
            ]
        })
        
        self.client.set_iam_policy(dataset_ref, policy)
        logger.info(f"IAM configurado para dataset {dataset_id}")
    
    def create_raw_tables(self, dataset_id):
        """Cria tabelas da camada raw"""
        schema_dir = "schemas/raw"
        
        tables = [
            {
                "name": "raw_app_logs",

    def setup_retention_policies(self, dataset_id):
        """Configura políticas de retenção automática (Compliance)"""
        retention_rules = [
            {
                "table": "raw_app_logs",
                "retention_days": 365,
                "description": "Logs de aplicação - retenção 1 ano"
            },
            {

                "retention_days": 180,
                "description": "Chamadas LLM - retenção 6 meses"
            },
            {
                "table": "raw_user_sessions",
                "retention_days": 90,
                "description": "Sessões de usuário - retenção 90 dias"
            }
        ]
        
        for rule in retention_rules:
            table_id = f"{self.project_id}.{dataset_id}.{rule['table']}"
            table = self.client.get_table(table_id)
            
            # Atualiza expiração da tabela
            table.expires = datetime.utcnow() + timedelta(days=rule['retention_days'])
            self.client.update_table(table, ["expires"])
            
            logger.info(f"Política de retenção aplicada: {rule['table']} - {rule['retention_days']} dias")

        """Sync ArgoCD application to target revision"""

            try:

                # Create sync operation
                sync_op = argocd.V1alpha1SyncOperation(
                    revision=spec.target_revision,
                    prune=spec.sync_policy.get("automated", {}).get("prune", True)
                )
                
                # Execute sync
                sync_response = self.argo_client.sync(
                    name=spec.application_name,
                    body=sync_op
                )
                
                # Wait for sync completion
                await self._wait_for_sync_completion(
                    spec.application_name,
                    spec.health_check_timeout
                )
                
                sync_result["success"] = True
                sync_result["sync_response"] = sync_response.to_dict()
                break
                
            except Exception as e:
                logger.warning(f"Sync attempt {attempt + 1} failed: {e}")
                sync_result["retries"] = attempt + 1
                await asyncio.sleep(5 * (attempt + 1))  # Exponential backoff
        
        return sync_result

const jwt = require('jsonwebtoken');

class JWTSecurity {
    constructor() {
        this.accessTokenSecret = process.env.JWT_ACCESS_SECRET || crypto.randomBytes(64).toString('hex');
        this.refreshTokenSecret = process.env.JWT_REFRESH_SECRET || crypto.randomBytes(64).toString('hex');
        this.tokenBlacklist = new Set();
    }

    generateAccessToken(user) {
        return jwt.sign(
            {
                userId: user.id,
                email: user.email,
                role: user.role,
                permissions: user.permissions
            },
            this.accessTokenSecret,
            {
                expiresIn: '15m',
                issuer: 'diyapp-v12',
                audience: 'diyapp-api',
                jwtid: crypto.randomUUID()
            }
        );
    }

    generateRefreshToken(user) {
        const refreshToken = jwt.sign(
            {
                userId: user.id,
                type: 'refresh'
            },
            this.refreshTokenSecret,
            {
                expiresIn: '7d',
                issuer: 'diyapp-v12',
                audience: 'diyapp-api'
            }
        );

        // Store refresh token hash in database
        const tokenHash = crypto.createHash('sha256').update(refreshToken).digest('hex');
        
        return {
            token: refreshToken,
            hash: tokenHash,
            expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000)
        };
    }

    verifyAccessToken(token) {
        try {
            const decoded = jwt.verify(token, this.accessTokenSecret, {
                issuer: 'diyapp-v12',
                audience: 'diyapp-api'
            });
            
            // Check blacklist
            if (this.tokenBlacklist.has(token)) {
                throw new Error('Token revoked');
            }
            
            return { valid: true, payload: decoded };
        } catch (error) {
            return { valid: false, error: error.message };
        }
    }

const JWTSecurity = require('./jwt');

class AuthMiddleware {
    constructor() {
        this.jwt = new JWTSecurity();
    }

    authenticate(req, res, next) {
        const authHeader = req.headers.authorization;
        
        if (!authHeader || !authHeader.startsWith('Bearer ')) {
            return res.status(401).json({
                error: 'Authentication required',
                code: 'AUTH_REQUIRED'
            });
        }

        const token = authHeader.substring(7);
        const result = this.jwt.verifyAccessToken(token);

        if (!result.valid) {
            return res.status(401).json({
                error: 'Invalid or expired token',
                code: 'TOKEN_INVALID',
                details: result.error
            });
        }

        req.user = result.payload;
        req.token = token;
        next();
    }

    authorize(requiredPermissions = []) {
        return (req, res, next) => {
            if (!req.user) {
                return res.status(401).json({ error: 'Unauthorized' });
            }

            // Check role-based access
            if (requiredPermissions.length > 0) {
                const userPermissions = req.user.permissions || [];
                const hasPermission = requiredPermissions.every(perm => 
                    userPermissions.includes(perm)
                );

                if (!hasPermission && req.user.role !== 'admin') {
                    return res.status(403).json({
                        error: 'Insufficient permissions',
                        code: 'PERMISSION_DENIED',
                        required: requiredPermissions,
                        has: userPermissions
                    });
                }
            }

            // Check resource ownership for non-admin users
            if (req.params.userId && req.user.role !== 'admin') {
                if (req.params.userId !== req.user.userId) {
                    return res.status(403).json({
                        error: 'Access denied to this resource',
                        code: 'RESOURCE_OWNERSHIP'
                    });
                }
            }

            next();
        };
    }

    async refreshToken(req, res) {
        const { refreshToken } = req.body;

        if (!refreshToken) {
            return res.status(400).json({ error: 'Refresh token required' });
        }

        const result = this.jwt.verifyRefreshToken(refreshToken);
        
        if (!result.valid) {
            return res.status(401).json({
                error: 'Invalid refresh token',
                code: 'REFRESH_INVALID'
            });
        }

        // In production: Verify refresh token exists in database and isn't revoked
        const tokenHash = crypto.createHash('sha256').update(refreshToken).digest('hex');
        
        // Generate new tokens
        const user = {
            id: result.payload.userId,
            email: 'user@example.com', // Fetch from DB
            role: 'user',
            permissions: ['read', 'write']
        };

        const newAccessToken = this.jwt.generateAccessToken(user);
        const newRefreshToken = this.jwt.generateRefreshToken(user);

        // Rotate refresh token (invalidate old)
        this.jwt.rotateRefreshToken(tokenHash, newRefreshToken.hash);

        res.json({
            accessToken: newAccessToken,
            refreshToken: newRefreshToken.token,
            expiresIn: 900 // 15 minutes in seconds
        });

        if (req.token) {
            this.jwt.revokeToken(req.token);
        }
        
        res.json({ message: 'Logged out successfully' });

const Redis = require('ioredis');

class RateLimiter {
    constructor() {
        this.redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');
        this.limits = {
            auth: { windowMs: 15 * 60 * 1000, max: 5 }, // 5 attempts per 15min
            api: { windowMs: 60 * 1000, max: 100 }, // 100 requests per minute
            llm: { windowMs: 60 * 1000, max: 10 }, // 10 LLM calls per minute
            public: { windowMs: 60 * 1000, max: 30 } // 30 public requests per minute
        };
    }

    async checkLimit(key, type = 'api') {
        const limit = this.limits[type] || this.limits.api;
        const now = Date.now();
        const windowStart = now - limit.windowMs;

        // Clean old entries
        await this.redis.zremrangebyscore(key, 0, windowStart);

        // Count requests in current window
        const requestCount = await this.redis.zcard(key);

        if (requestCount >= limit.max) {
            const oldestRequest = await this.redis.zrange(key, 0, 0, 'WITHSCORES');
            const retryAfter = Math.ceil((parseInt(oldestRequest[1]) + limit.windowMs - now) / 1000);
            
            return {
                allowed: false,
                retryAfter,
                remaining: 0,
                limit: limit.max
            };
        }

        // Add current request
        await this.redis.zadd(key, now, `${now}-${Math.random()}`);
        await this.redis.expire(key, Math.ceil(limit.windowMs / 1000) + 1);

        return {
            allowed: true,
            remaining: limit.max - requestCount - 1,
            limit: limit.max,
            reset: new Date(now + limit.windowMs)
        };
    }

    async middleware(req, res, next) {
        const identifier = req.ip || req.headers['x-forwarded-for'] || 'unknown';
        const path = req.path;
        
        let limitType = 'api';
        
        if (path.includes('/auth/')) {
            limitType = 'auth';
        } else if (path.includes('/llm/') || path.includes('/ai/')) {
            limitType = 'llm';
        } else if (path === '/' || path.includes('/public/')) {
            limitType = 'public';
        }

        const key = `rate_limit:${limitType}:${identifier}`;
        const result = await this.checkLimit(key, limitType);

        // Set rate limit headers (RFC 6585)
        res.set({
            'X-RateLimit-Limit': result.limit,
            'X-RateLimit-Remaining': result.remaining || 0,
            'X-RateLimit-Reset': Math.floor((result.reset || new Date()).getTime() / 1000)
        });

        if (!result.allowed) {
            res.set('Retry-After', result.retryAfter);
            return res.status(429).json({
                error: 'Too many requests',
                code: 'RATE_LIMITED',
                retryAfter: `${result.retryAfter} seconds`
            });
        }

        next();
    }

    async checkBruteForce(key, maxAttempts = 5, windowMs = 15 * 60 * 1000) {
        const now = Date.now();
        const windowStart = now - windowMs;

        await this.redis.zremrangebyscore(key, 0, windowStart);
        
        const attempts = await this.redis.zcard(key);
        
        if (attempts >= maxAttempts) {
            const oldestAttempt = await this.redis.zrange(key, 0, 0, 'WITHSCORES');
            const blockUntil = parseInt(oldestAttempt[1]) + windowMs;
            const remainingTime = Math.ceil((blockUntil - now) / 1000);
            
            return {
                blocked: true,
                remainingTime,
                attempts
            };
        }

        await this.redis.zadd(key, now, `${now}-${Math.random()}`);
        await this.redis.expire(key, Math.ceil(windowMs / 1000) + 1);

        return {
            blocked: false,
            attempts: attempts + 1,
            remainingAttempts: maxAttempts - attempts - 1
        };
    }

class ConsentManager:

        self.db = db_session
    
    def create_consent(self, consent_data: ConsentCreate) -> Consent:
        """Cria um novo registro de consentimento"""
        try:
            # Verifica se o data subject existe
            data_subject = self.db.query(DataSubject).filter(
                DataSubject.id == consent_data.data_subject_id
            ).first()
            
            if not data_subject:
                raise ValueError(f"Data subject {consent_data.data_subject_id} not found")
            
            # Cria o consentimento
            consent = Consent(
                data_subject_id=consent_data.data_subject_id,
                purpose=consent_data.purpose,
                legal_basis=consent_data.legal_basis,
                expires_at=consent_data.expires_at,
                metadata=consent_data.metadata or {},
                version="1.0"  # Versão da política de consentimento
            )
            
            self.db.add(consent)
            self.db.commit()
            self.db.refresh(consent)
            
            logger.info(f"Consent created: {consent.id} for subject {consent_data.data_subject_id}")
            return consent
            
        except Exception as e:
            self.db.rollback()
            logger.error(f"Error creating consent: {str(e)}")
            raise
    
    def update_consent_status(self, consent_id: str, update_data: ConsentUpdate) -> Consent:
        """Atualiza o status de um consentimento"""
        try:
            consent = self.db.query(Consent).filter(Consent.id == consent_id).first()
            
            if not consent:
                raise ValueError(f"Consent {consent_id} not found")
            
            consent.status = update_data.status
            
            if update_data.status == ConsentStatus.WITHDRAWN:
                consent.withdrawn_at = update_data.withdrawn_at or datetime.utcnow()
            
            self.db.commit()
            self.db.refresh(consent)
            
            logger.info(f"Consent {consent_id} status updated to {update_data.status}")
            return consent
            
        except Exception as e:
            self.db.rollback()
            logger.error(f"Error updating consent status: {str(e)}")
            raise
    
    def get_valid_consent(self, data_subject_id: str, purpose: str) -> Optional[Consent]:
        """Obtém consentimento válido para um propósito específico"""
        try:
            now = datetime.utcnow()
            
            consent = self.db.query(Consent).filter(
                and_(
                    Consent.data_subject_id == data_subject_id,
                    Consent.purpose == purpose,
                    Consent.status == ConsentStatus.GRANTED,
                    or_(
                        Consent.expires_at == None,
                        Consent.expires_at > now
                    ),
                    Consent.withdrawn_at == None
                )
            ).order_by(Consent.granted_at.desc()).first()
            
            return consent

describe('DIYAPP V12 - Integração API', () => {
  const API_BASE = Cypress.env('API_BASE_URL') || 'http://localhost:3001/api';
  
  before(() => {
    // Obtém token de autenticação

      method: 'POST',

        password: 'Test@123'

      expect(response.status).to.eq(200);
      Cypress.env('API_TOKEN', response.body.token);
    });
  });

  describe('API Projects', () => {
    let projectId;
    
    it('API-001: POST /api/projects - Criar projeto', () => {
      const projectData = {
        name: `API Project ${Date.now()}`,
        description: 'Projeto criado via API',
        type: 'web',
        config: { framework: 'react', version: '18' }
      };
      
      cy.apiRequest('POST', '/projects', projectData).then((response) => {
        expect(response.status).to.eq(201);
        expect(response.body).to.have.property('id');
        expect(response.body.name).to.eq(projectData.name);
        
        projectId = response.body.id;
      });
    });
    
    it('API-002: GET /api/projects - Listar projetos', () => {
      cy.apiRequest('GET', '/projects').then((response) => {
        expect(response.status).to.eq(200);
        expect(response.body).to.be.an('array');
        expect(response.body.length).to.be.greaterThan(0);
      });
    });
    
    it('API-003: GET /api/projects/:id - Obter projeto específico', () => {
      cy.apiRequest('GET', `/projects/${projectId}`).then((response) => {
        expect(response.status).to.eq(200);
        expect(response.body.id).to.eq(projectId);
      });
    });
    
    it('API-004: PUT /api/projects/:id - Atualizar projeto', () => {
      const updateData = {
        name: `Updated ${Date.now()}`,
        description: 'Descrição atualizada'
      };
      
      cy.apiRequest('PUT', `/projects/${projectId}`, updateData).then((response) => {
        expect(response.status).to.eq(200);
        expect(response.body.name).to.eq(updateData.name);
      });
    });
    
    it('API-005: DELETE /api/projects/:id - Excluir projeto', () => {
      cy.apiRequest('DELETE', `/projects/${projectId}`).then((response) => {
        expect(response.status).to.eq(204);
      });
      
      // Verifica que foi excluído
      cy.apiRequest('GET', `/projects/${projectId}`).then((response) => {
        expect(response.status).to.eq(404);
      });
    });
  });
  
  describe('API Validation', () => {
    it('API-006: Validação de erros 400', () => {
      cy.apiRequest('POST', '/projects', {
        name: '' // Nome vazio deve falhar
      }).then((response) => {
        expect(response.status).to.eq(400);
        expect(response.body).to.have.property('errors');
      });
    });
    
    it('API-007: Autenticação inválida', () => {
      cy.request({
        method: 'GET',
        url: `${API_BASE}/projects`,
        headers: {
          'Authorization': 'Bearer invalid-token'
        },
        failOnStatusCode: false
      }).then((response) => {
        expect(response.status).to.eq(401);
      });
    });
    
    it('API-008: Rate limiting', () => {
      const requests = Array(60).fill().map(() => 
        cy.apiRequest('GET', '/projects')
      );
      
      // Última requisição deve ser limitada
      cy.wrap(Promise.all(requests)).then((responses) => {
        const lastResponse = responses[responses.length - 1];
        expect(lastResponse.status).to.be.oneOf([200, 429]);
      });
    });
  });
});
