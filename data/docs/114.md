// shared/libs/logger/index.ts - Logger padronizado
import winston from 'winston';
import { ElasticsearchTransport } from 'winston-elasticsearch';

class Logger {
  private logger: winston.Logger;
  
  constructor(serviceName: string) {
    const transports: winston.transport[] = [
      new winston.transports.Console({
        format: winston.format.combine(
          winston.format.colorize(),
          winston.format.timestamp(),
          winston.format.printf(({ timestamp, level, message, service, ...meta }) => {
            return `[${timestamp}] ${service} ${level}: ${message} ${
              Object.keys(meta).length ? JSON.stringify(meta) : ''
            }`;
          })
        )
      })
    ];
    
    // Adiciona transporte para Elasticsearch em produção
    if (process.env.NODE_ENV === 'production') {
      transports.push(
        new ElasticsearchTransport({
          level: 'info',
          clientOpts: { node: process.env.ELASTICSEARCH_URL },
          indexPrefix: 'diyapp-logs'
        })
      );
    }
    
    this.logger = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      defaultMeta: { service: serviceName },
      transports
    });
  }
  
  info(message: string, meta?: Record<string, any>) {
    this.logger.info(message, meta);
  }
  
  error(message: string, error?: Error, meta?: Record<string, any>) {
    this.logger.error(message, { error, ...meta });
  }
  
  warn(message: string, meta?: Record<string, any>) {
    this.logger.warn(message, meta);
  }
  
  debug(message: string, meta?: Record<string, any>) {
    this.logger.debug(message, meta);
  }
}

export default Logger;

# Threat Model - DIYAPP V12 Core Architecture
**Data:** $(date)
**Versão:** 1.0
**Responsável:** Especialista em Segurança

## 1. Visão Geral da Arquitetura
- **Frontend:** React/Next.js com API Routes
- **Backend:** Node.js microservices (Auth, Data, LLM, Storage)
- **Banco de Dados:** PostgreSQL (principal) + Redis (cache)
- **LLM Services:** OpenAI GPT-4 + embeddings locais
- **Storage:** S3-compatible (MinIO)
- **Message Queue:** RabbitMQ
- **Container Runtime:** Docker + Kubernetes

## 2. Análise STRIDE por Componente

### 2.1 Serviço de Autenticação (Auth Service)
**Ativos:** Credenciais de usuário, tokens JWT, sessões

| Categoria STRIDE | Ameaça Identificada | Impacto | Probabilidade | Controles de Mitigação |
|-----------------|---------------------|---------|---------------|------------------------|
| **Spoofing** | Credential stuffing em endpoint /login | Alto | Alta | Rate limiting (10 tentativas/5min), MFA obrigatório, WAF com regras de login |
| **Tampering** | Manipulação de token JWT no localStorage | Crítico | Média | Tokens assinados com RSA-2048, short-lived (15min), refresh tokens com revogação |
| **Repudiation** | Usuário nega login bem-sucedido | Baixo | Baixa | Logs de auditoria imutáveis com IP, user agent, timestamp |
| **Info Disclosure** | Vazamento de hash de senha no DB | Crítico | Baixa | bcrypt com salt 12 rounds, PII criptografado em colunas separadas |
| **Denial of Service** | DDoS em endpoint /oauth/callback | Médio | Média | Rate limiting por IP, Cloudflare/WAF, circuit breaker |
| **Elevation of Privilege** | Acesso a admin bypassando RBAC | Crítico | Baixa | Verificação de permissão por recurso, princípio do menor privilégio |

### 2.2 Serviço de Dados (Data Service)
**Ativos:** Dados PII, documentos de usuário, metadados

| Categoria STRIDE | Ameaça Identificada | Impacto | Probabilidade | Controles de Mitigação |
|-----------------|---------------------|---------|---------------|------------------------|
| **Spoofing** | SQL injection via parâmetros de query | Crítico | Média | Prepared statements, ORM com escaping, validação de schema |
| **Tampering** | Manipulação de dados em repouso via backup | Alto | Baixa | Criptografia AES-256-GCM em nível de coluna, HMAC para integridade |
| **Repudiation** | Negação de modificação de dados | Médio | Baixa | Audit trail com hash chain, logs assinados digitalmente |
| **Info Disclosure** | IDOR (Insecure Direct Object Reference) | Alto | Alta | Verificação de ownership em cada query, UUIDs não sequenciais |
| **Denial of Service** | Query complexa derruba DB | Médio | Média | Query timeout (30s), connection pooling, read replicas |
| **Elevation of Privilege** | Acesso cross-tenant via bug de multi-tenancy | Crítico | Média | Tenant isolation com row-level security, contextos separados |

### 2.3 Serviço LLM (AI Service)
**Ativos:** Prompts do sistema, embeddings, dados de treinamento

| Categoria STRIDE | Ameaça Identificada | Impacto | Probabilidade | Controles de Mitigação |
|-----------------|---------------------|---------|---------------|------------------------|
| **Spoofing** | Prompt injection para extrair system prompt | Alto | Alta | Separação rígida system/user prompt, sanitização de input |
| **Tampering** | Manipulação de embeddings no vector DB | Médio | Baixa | Checksums de embeddings, validação de integridade |
| **Repudiation** | Negação de geração de conteúdo prejudicial | Médio | Média | Logs de prompts/responses com hash, moderation API |
| **Info Disclosure** | Data exfiltration via contexto do LLM | Crítico | Média | Sanitização de contexto, não incluir PII no prompt |
| **Denial of Service** | Custo explosion via prompts infinitos | Alto | Média | Rate limiting por usuário, budget alerts, max tokens |
| **Elevation of Privilege** | LLM usado para gerar código malicioso | Crítico | Baixa | Sandbox para execução de código, análise estática de output |

### 2.4 Storage Service (S3/MinIO)
**Ativos:** Arquivos de usuário, backups, logs

| Categoria STRIDE | Ameaça Identificada | Impacto | Probabilidade | Controles de Mitigação |
|-----------------|---------------------|---------|---------------|------------------------|
| **Spoofing** | Acesso não autorizado via presigned URLs vazadas | Alto | Média | URLs com expiração curta (5min), validação de assinatura |
| **Tampering** | Modificação de arquivos estáticos (JS/CSS) | Crítico | Baixa | Subresource Integrity (SRI), checksums em deploy |
| **Repudiation** | Negação de upload de conteúdo malicioso | Baixo | Baixa | Logs de acesso ao bucket, versionamento habilitado |
| **Info Disclosure** | Bucket público por configuração errada | Crítico | Média | SCP (Service Control Policies), scanner de buckets públicos |
| **Denial of Service** | Upload massivo esgota storage | Médio | Baixa | Quotas por usuário, monitoramento de espaço |
| **Elevation of Privilege** | Acesso a arquivos de outros usuários | Crítico | Média | Prefixos por tenant, políticas de bucket restritivas |

```typescript
import amqp, { Channel, Connection } from 'amqplib';
import { Logger } from './logger';
import { RabbitMQMessage } from '../types';

export class RabbitMQClient {
  private connection: Connection | null = null;
  private channel: Channel | null = null;
  private logger: Logger;
  private isConnected = false;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 10;
  private reconnectDelay = 5000;

  constructor(private serviceName: string) {
    this.logger = new Logger(`RabbitMQ-${serviceName}`);
  }

  async connect(): Promise<void> {
    try {
      const url = process.env.RABBITMQ_URL || 'amqp://localhost:5672';
      this.connection = await amqp.connect(url);
      this.channel = await this.connection.createChannel();
      
      this.connection.on('close', () => {
        this.logger.warn('RabbitMQ connection closed, attempting to reconnect...');
        this.isConnected = false;
        this.handleReconnect();
      });

      this.connection.on('error', (error) => {
        this.logger.error('RabbitMQ connection error', error);
      });

      this.isConnected = true;
      this.reconnectAttempts = 0;
      this.logger.info('RabbitMQ connected successfully');
    } catch (error) {
      this.logger.error('Failed to connect to RabbitMQ', error as Error);
      this.handleReconnect();
      throw error;
    }
  }

  private async handleReconnect(): Promise<void> {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      this.logger.error('Max reconnection attempts reached');
      return;
    }

    this.reconnectAttempts++;
    this.logger.info(`Reconnection attempt ${this.reconnectAttempts}/${this.maxReconnectAttempts}`);

    setTimeout(async () => {
      try {
        await this.connect();
      } catch (error) {
        // Continue reconnection attempts
      }
    }, this.reconnectDelay * this.reconnectAttempts);
  }

  async publish(exchange: string, routingKey: string, message: RabbitMQMessage): Promise<boolean> {
    if (!this.isConnected || !this.channel) {
      throw new Error('RabbitMQ not connected');
    }

    try {
      await this.channel.assertExchange(exchange, 'topic', { durable: true });
      const published = this.channel.publish(
        exchange,
        routingKey,
        Buffer.from(JSON.stringify(message)),
        { persistent: true }
      );

      if (published) {
        this.logger.debug(`Message published to ${exchange}.${routingKey}`, {
          correlationId: message.correlationId,
          type: message.type,
        });
      }

      return published;
    } catch (error) {
      this.logger.error('Failed to publish message', error as Error, {
        exchange,
        routingKey,
        correlationId: message.correlationId,
      });
      throw error;
    }
  }

  async consume(queue: string, onMessage: (message: RabbitMQMessage) => Promise<void>): Promise<void> {
    if (!this.isConnected || !this.channel) {
      throw new Error('RabbitMQ not connected');
    }

    try {
      await this.channel.assertQueue(queue, { durable: true });
      await this.channel.consume(queue, async (msg) => {
        if (!msg) return;

        try {
          const message: RabbitMQMessage = JSON.parse(msg.content.toString());
          this.logger.debug(`Message received from queue ${queue}`, {
            correlationId: message.correlationId,
            type: message.type,
          });

          await onMessage(message);
          this.channel!.ack(msg);
        } catch (error) {
          this.logger.error('Error processing message', error as Error);
          this.channel!.nack(msg, false, false); // Don't requeue
        }
      });

      this.logger.info(`Started consuming queue: ${queue}`);
    } catch (error) {
      this.logger.error('Failed to consume queue', error as Error, { queue });
      throw error;
    }
  }

  async bindQueue(queue: string, exchange: string, routingKey: string): Promise<void> {
    if (!this.isConnected || !this.channel) {
      throw new Error('RabbitMQ not connected');
    }

    try {
      await this.channel.assertQueue(queue, { durable: true });
      await this.channel.assertExchange(exchange, 'topic', { durable: true });
      await this.channel.bindQueue(queue, exchange, routingKey);
      this.logger.debug(`Queue ${queue} bound to ${exchange}.${routingKey}`);
    } catch (error) {
      this.logger.error('Failed to bind queue', error as Error, {
        queue,
        exchange,
        routingKey,
      });
      throw error;
    }
  }

  async close(): Promise<void> {
    try {
      if (this.channel) {
        await this.channel.close();
      }
      if (this.connection) {
        await this.connection.close();
      }
      this.isConnected = false;
      this.logger.info('RabbitMQ connection closed');
    } catch (error) {
      this.logger.error('Error closing RabbitMQ connection', error as Error);
    }
  }

  get isReady(): boolean {
    return this.isConnected;
  }
}

export interface CircuitBreakerConfig {

export enum CircuitState {
  CLOSED = 'CLOSED',
  OPEN = 'OPEN',
  HALF_OPEN = 'HALF_OPEN',
}

export class CircuitBreaker {
  private state: CircuitState = CircuitState.CLOSED;
  private failureCount = 0;
  private lastFailureTime: number | null = null;
  private halfOpenAttempts = 0;
  private logger: Logger;

  constructor(
    private serviceName: string,
    private config: CircuitBreakerConfig = {
      failureThreshold: 5,
      resetTimeout: 30000, // 30 seconds
      halfOpenMaxAttempts: 3,
    }
  ) {
    this.logger = new Logger(`CircuitBreaker-${serviceName}`);
  }

  async execute<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === CircuitState.OPEN) {
      if (this.canAttemptReset()) {
        this.state = CircuitState.HALF_OPEN;
        this.halfOpenAttempts = 0;
        this.logger.info(`Circuit breaker transitioning to HALF_OPEN for ${this.serviceName}`);
      } else {
        throw new Error(`Circuit breaker is OPEN for ${this.serviceName}`);
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private canAttemptReset(): boolean {
    if (!this.lastFailureTime) return false;
    return Date.now() - this.lastFailureTime > this.config.resetTimeout;
  }

  private onSuccess(): void {
    this.failureCount = 0;
    this.halfOpenAttempts = 0;
    
    if (this.state === CircuitState.HALF_OPEN) {
      this.state = CircuitState.CLOSED;
      this.logger.info(`Circuit breaker reset to CLOSED for ${this.serviceName}`);
    }
  }

  private onFailure(): void {
    this.failureCount++;
    this.lastFailureTime = Date.now();

    if (this.state === CircuitState.HALF_OPEN) {
      this.halfOpenAttempts++;
      if (this.halfOpenAttempts >= this.config.halfOpenMaxAttempts) {
        this.state = CircuitState.OPEN;
        this.logger.warn(`Circuit breaker transitioned to OPEN for ${this.serviceName} (half-open attempts exceeded)`);
      }
    } else if (this.failureCount >= this.config.failureThreshold) {
      this.state = CircuitState.OPEN;
      this.logger.warn(`Circuit breaker transitioned to OPEN for ${this.serviceName}`);
    }
  }

  getState(): CircuitState {
    return this.state;
  }

  getMetrics() {
    return {
      state: this.state,
      failureCount: this.failureCount,
      lastFailureTime: this.lastFailureTime,
      halfOpenAttempts: this.halfOpenAttempts,
    };
  }

  reset(): void {
    this.state = CircuitState.CLOSED;
    this.failureCount = 0;
    this.lastFailureTime = null;
    this.halfOpenAttempts = 0;
    this.logger.info(`Circuit breaker manually reset for ${this.serviceName}`);
  }
}

```python
import os
import asyncio
from typing import Dict, List, Optional, Any, Tuple
from dataclasses import dataclass
from datetime import datetime, timedelta
import tiktoken
from tenacity import retry, stop_after_attempt, wait_exponential

from langchain_openai import AzureChatOpenAI, ChatOpenAI
from langchain_anthropic import ChatAnthropic
from langchain_google_genai import ChatGoogleGenerativeAI
from langchain.schema import HumanMessage, SystemMessage, AIMessage
from pydantic import BaseModel, Field

from .rate_limiter import RateLimiter
from .fallback_strategy import FallbackStrategy
from ..utils.logger import get_logger
from ..monitoring.cost_tracker import CostTracker
from ..monitoring.metrics_collector import MetricsCollector

logger = get_logger(__name__)

@dataclass
class LLMResponse:
    content: str
    model_used: str
    provider: str
    tokens_used: Dict[str, int]
    latency_ms: float
    cost_usd: float
    fallback_triggered: bool = False

class LLMManager:
    """Gerencia múltiplos provedores LLM com rate limiting e fallback"""
    
    def __init__(self, config_path: str = "config/llm_config.yaml"):
        self.config = self._load_config(config_path)
        self.rate_limiter = RateLimiter(
            requests_per_minute=self.config['rate_limiting']['requests_per_minute'],
            tokens_per_minute=self.config['rate_limiting']['tokens_per_minute']
        )
        self.fallback_strategy = FallbackStrategy(
            primary=self.config['fallback_strategy']['primary'],
            fallback_chain=self.config['fallback_strategy']['fallback_chain'],
            retry_attempts=self.config['fallback_strategy']['retry_attempts']
        )
        self.cost_tracker = CostTracker(self.config['cost_tracking']['token_prices'])
        self.metrics = MetricsCollector()
        
        # Inicializa clientes LLM
        self.clients = self._initialize_clients()
        
        # Cache para tokenizers
        self.tokenizers = {}
        
        logger.info("LLMManager inicializado com rate limiting e fallback")
    
    def _load_config(self, config_path: str) -> Dict:
        import yaml
        from pathlib import Path
        
        config_path = Path(config_path)
        if not config_path.exists():
            raise FileNotFoundError(f"Config file not found: {config_path}")
        
        with open(config_path, 'r') as f:
            config = yaml.safe_load(f)
        
        # Substitui variáveis de ambiente
        import re
        def replace_env_vars(value):
            if isinstance(value, str) and value.startswith('${') and value.endswith('}'):
                env_var = value[2:-1]
                return os.getenv(env_var, '')
            return value
        
        # Processa recursivamente
        def process_config(obj):
            if isinstance(obj, dict):
                return {k: process_config(v) for k, v in obj.items()}
            elif isinstance(obj, list):
                return [process_config(item) for item in obj]
            else:
                return replace_env_vars(obj)
        
        return process_config(config)
    
    def _initialize_clients(self) -> Dict[str, Any]:
        """Inicializa clientes para todos os provedores configurados"""
        clients = {}
        
        # Azure OpenAI
        azure_config = self.config['llm_providers']['azure_openai']
        if azure_config.get('api_key'):
            clients['azure_openai'] = AzureChatOpenAI(
                azure_endpoint=azure_config['api_base'],
                api_key=azure_config['api_key'],
                api_version=azure_config['api_version'],
                azure_deployment=azure_config['deployment_name'],
                max_tokens=azure_config['max_tokens'],
                temperature=azure_config['temperature'],
                timeout=azure_config['timeout']
            )
        
        # Anthropic
        anthropic_config = self.config['llm_providers']['anthropic']
        if anthropic_config.get('api_key'):
            clients['anthropic'] = ChatAnthropic(
                anthropic_api_key=anthropic_config['api_key'],
                model=anthropic_config['model'],
                max_tokens=anthropic_config['max_tokens'],
                temperature=anthropic_config['temperature']
            )
        
        # Google
        google_config = self.config['llm_providers']['google']
        if google_config.get('api_key'):
            clients['google'] = ChatGoogleGenerativeAI(
                google_api_key=google_config['api_key'],
                model=google_config['model'],
                max_tokens=google_config['max_tokens'],
                temperature=google_config['temperature']
            )
        
        return clients
    
    def _count_tokens(self, text: str, provider: str) -> int:
        """Conta tokens para um texto específico baseado no provedor"""
        if provider not in self.tokenizers:
            if provider == 'azure_openai':
                self.tokenizers[provider] = tiktoken.encoding_for_model("gpt-4")
            elif provider == 'anthropic':
                # Anthropic tem contagem diferente, usando aproximação
                self.tokenizers[provider] = tiktoken.get_encoding("cl100k_base")
            elif provider == 'google':
                # Gemini usa contagem similar a OpenAI
                self.tokenizers[provider] = tiktoken.get_encoding("cl100k_base")
            else:
                self.tokenizers[provider] = tiktoken.get_encoding("cl100k_base")
        
        return len(self.tokenizers[provider].encode(text))
    
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=4, max=10)
    )
    async def generate(
        self,
        messages: List[Dict[str, str]],
        provider: str = None,
        feature: str = "unknown",
        metadata: Dict = None
    ) -> LLMResponse:
        """
        Gera resposta do LLM com rate limiting e fallback automático
        
        Args:
            messages: Lista de mensagens no formato [{"role": "user", "content": "..."}]
            provider: Provedor específico (None para usar estratégia de fallback)
            feature: Nome da feature para tracking
            metadata: Metadados adicionais para logging
        
        Returns:
            LLMResponse com resposta e métricas
        """
        start_time = datetime.now()
        metadata = metadata or {}
        
        # Converte mensagens para formato LangChain
        lc_messages = []
        for msg in messages:
            if msg['role'] == 'system':
                lc_messages.append(SystemMessage(content=msg['content']))
            elif msg['role'] == 'user':
                lc_messages.append(HumanMessage(content=msg['content']))
            elif msg['role'] == 'assistant':
                lc_messages.append(AIMessage(content=msg['content']))
        
        # Calcula tokens de input
        input_text = " ".join([msg.content for msg in lc_messages])
        
        # Determina provedor a ser usado
        if provider is None:
            provider = self.fallback_strategy.get_next_provider()
        
        # Aplica rate limiting
        tokens_estimated = self._count_tokens(input_text, provider)
        await self.rate_limiter.wait_if_needed(provider, tokens_estimated)
        
        # Tenta gerar resposta com fallback automático
        response = None
        fallback_used = False
        last_error = None
        
        for attempt, current_provider in enumerate(
            self.fallback_strategy.get_fallback_chain(provider)
        ):
            try:
                if current_provider not in self.clients:
                    logger.warning(f"Provider {current_provider} não disponível")
                    continue
                
                logger.info(f"Tentando com provedor {current_provider} (tentativa {attempt + 1})")
                
                # Gera resposta
                llm_response = await self.clients[current_provider].agenerate([lc_messages])
                response_content = llm_response.generations[0][0].text
                
                # Calcula tokens de output
                output_tokens = self._count_tokens(response_content, current_provider)
                
                # Calcula custo
                cost = self.cost_tracker.calculate_cost(
                    provider=current_provider,
                    input_tokens=tokens_estimated,
                    output_tokens=output_tokens
                )
                
                # Registra métricas
                latency = (datetime.now() - start_time).total_seconds() * 1000
                self.metrics.record_call(
                    provider=current_provider,
                    feature=feature,
                    input_tokens=tokens_estimated,
                    output_tokens=output_tokens,
                    latency_ms=latency,
                    success=True,
                    cost_usd=cost
                )
                
                response = LLMResponse(
                    content=response_content,
                    model_used=self.config['llm_providers'][current_provider].get('model', current_provider),
                    provider=current_provider,
                    tokens_used={
                        'input': tokens_estimated,
                        'output': output_tokens,
                        'total': tokens_estimated + output_tokens
                    },
                    latency_ms=latency,
                    cost_usd=cost,
                    fallback_triggered=fallback_used
                )
                
                logger.info(
                    f"LLM call successful",
                    provider=current_provider,
                    feature=feature,
                    tokens=input_tokens,
                    latency_ms=latency,
                    cost_usd=cost
                )
                
                break
                
            except Exception as e:
                last_error = e
                fallback_used = True
                logger.error(
                    f"Falha com provedor {current_provider}",
                    error=str(e),
                    attempt=attempt + 1
                )
                
                # Registra falha nas métricas
                self.metrics.record_call(
                    provider=current_provider,
                    feature=feature,
                    input_tokens=tokens_estimated,
                    output_tokens=0,
                    latency_ms=(datetime.now() - start_time).total_seconds() * 1000,
                    success=False,
                    cost_usd=0
                )
                
                # Espera antes de tentar próximo provedor
                if attempt < len(self.fallback_strategy.get_fallback_chain(provider)) - 1:
                    await asyncio.sleep(self.config['fallback_strategy']['retry_delay'])
        
        if response is None:
            raise Exception(f"Todos os provedores falharam: {last_error}")
        
        return response
    
    def get_usage_stats(self, time_window_hours: int = 24) -> Dict:
        """Retorna estatísticas de uso"""
        return self.metrics.get_usage_stats(time_window_hours)
    
    def get_cost_by_feature(self, start_date: datetime, end_date: datetime) -> Dict:
        """Retorna custo agrupado por feature"""
        return self.metrics.get_cost_by_feature(start_date, end_date)
    
    def get_provider_health(self) -> Dict:
        """Retorna saúde dos provedores baseado em métricas recentes"""
        return self.metrics.get_provider_health()

import React, { createContext, useContext } from 'react';
import { tokens } from './tokens';

interface ThemeContextType {
  tokens: typeof tokens;
  mode: 'light' | 'dark';
  toggleMode: () => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [mode, setMode] = React.useState<'light' | 'dark'>('light');

  const toggleMode = () => {
    setMode(prev => prev === 'light' ? 'dark' : 'light');
  };

  const value = {
    tokens,
    mode,
    toggleMode
  };

  return (
    <ThemeContext.Provider value={value}>
      <div className={`theme-${mode}`} data-theme={mode}>
        {children}
      </div>
    </ThemeContext.Provider>
  );
};

export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
};

# ADR-001: Arquitetura Core do DIYAPP V12

## Status
Aceito

## Contexto
O DIYAPP evoluiu para uma plataforma multi-agente autônoma que requer:
- 100% de disponibilidade operacional
- Escalabilidade horizontal para múltiplas squads
- Monitoramento proativo com auto-recuperação
- Documentação como código integrada

## Decisão
Adotar arquitetura baseada em microserviços com os seguintes componentes:

### 1. Core Engine
- **Orquestrador de Agentes**: Gerencia ciclo de vida dos agentes IA
- **Message Bus**: Comunicação assíncrona entre componentes
- **State Manager**: Persistência de estado distribuído

### 2. API Gateway
- Rate limiting e autenticação centralizada
- Roteamento inteligente baseado em carga
- Cache de respostas frequentes

### 3. Monitoring Stack
- Coleta de métricas em tempo real
- Alertas automáticos com auto-remediação

# Runbook: Resposta a Incidentes - DIYAPP V12

## Owner
SRE - Squad Core
Data: 2024-01-15
Última atualização: 2024-01-15

## 1. Classificação de Incidentes

### P1 - Crítico
- Sistema completamente inacessível
- Perda de dados
- Violação de segurança

**Tempo de resposta: 5 minutos**
**Tempo de resolução: 1 hora**

### P2 - Alto Impacto
- Degradação significativa de performance
- Funcionalidade core indisponível
- Erros em múltiplos usuários

**Tempo de resposta: 15 minutos**
**Tempo de resolução: 4 horas**

### P3 - Médio Impacto
- Funcionalidade não-core indisponível
- Erros isolados
- Performance degradada

**Tempo de resposta: 1 hora**
**Tempo de resolução: 24 horas**

### P4 - Baixo Impacto
- Issues cosméticas
- Melhorias
- Documentação

**Tempo de resposta: 8 horas**
**Tempo de resolução: 7 dias**

## 2. Fluxo de Resposta

### Fase 1: Detecção e Triagem
1. **Alerta recebido** via PagerDuty/Telegram
2. **Confirmar incidente** no dashboard de monitoramento
3. **Classificar** conforme matriz acima
4. **Criar canal** #incident-{ID} no Slack
5. **Notificar** Squad Leader e stakeholders

### Fase 2: Contenção
1. **Isolar componente afetado**
   ```bash
   # Exemplo: Isolar agente problemático
   kubectl scale deployment/agent-{id} --replicas=0
   ```

2. **Ativar fallback** se disponível
   ```bash
   # Exemplo: Ativar modo degradado
   curl -X POST https://api.diyapp.dev/v12/degraded-mode
   ```

3. **Coletar logs** para análise
   ```bash
   kubectl logs deployment/agent-{id} --tail=1000 > incident-{id}-logs.txt
   ```

### Fase 3: Resolução
1. **Identificar root cause** usando:
   - Logs do aplicativo
   - Métricas do sistema
   - Traces distribuídos

2. **Aplicar correção**
   - Hotfix se necessário
   - Rollback se apropriado

3. **Validar recuperação**
   ```bash
   # Verificar saúde do sistema
   curl https://api.diyapp.dev/v12/health | jq '.status'
   ```

### Fase 4: Pós-Incidente
1. **Documentar** no postmortem template
2. **Agendar** reunião de revisão
3. **Criar issues** para ações preventivas
4. **Atualizar** este runbook se necessário

## 3. Comandos Úteis

### Verificar Status do Cluster
```bash
kubectl get pods -n diyapp-production
kubectl get deployments -n diyapp-production
kubectl describe pod/{pod-name} -n diyapp-production
```

### Acessar Logs
```bash
# Logs em tempo real
kubectl logs -f deployment/{deployment-name}

# Logs com filtro de erro
kubectl logs deployment/{deployment-name} | grep -i error

# Logs de múltiplos pods
kubectl logs -l app={app-label} --tail=100
```

### Métricas e Monitoramento
```bash
# Acessar Prometheus
open https://prometheus.diyapp.dev

# Acessar Grafana
open https://grafana.diyapp.dev

# Verificar métricas via CLI
curl https://prometheus.diyapp.dev/api/v1/query?query=up
```

### Rollback de Deployment
```bash
# Listar histórico
kubectl rollout history deployment/{deployment-name}

# Rollback para versão anterior
kubectl rollout undo deployment/{deployment-name}

# Rollback para versão específica
kubectl rollout undo deployment/{deployment-name} --to-revision=2
```
