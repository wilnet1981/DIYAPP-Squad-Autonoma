## 3. ADR 001: Arquitetura Hexagonal com Módulos Paralelos

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead + Arquitetos de Software

### CONTEXTO:
Precisamos de uma arquitetura que suporte desenvolvimento paralelo de módulos (Modo Hive) enquanto mantém coesão do sistema. Módulos como Task Automation, LLM Orchestrator e Security Auditor devem evoluir independentemente mas compartilhar recursos comuns.

### DECISÃO:
Adotar Arquitetura Hexagonal (Ports & Adapters) com Clean Architecture principles, organizada em módulos feature-based com dependências unidirecionais.

### OPÇÕES CONSIDERADAS:
- **Opção A:** Monólito modular tradicional - Prós: Simples, baixa latência. Contras: Acoplamento alto, difícil evolução paralela.
- **Opção B:** Microserviços completos - Prós: Isolamento total. Contras: Complexidade operacional, overhead de comunicação.
- **Opção C:** Arquitetura Hexagonal com módulos - Prós: Isolamento lógico, testes fáceis, evolução independente. Contras: Curva de aprendizado.

**Opção escolhida:** C - Justificativa: Balanceia autonomia dos módulos com coesão do sistema, permitindo desenvolvimento paralelo (Modo Hive) sem complexidade operacional excessiva.

### CONSEQUÊNCIAS:
**Positivas:**
- Módulos podem ser desenvolvidos em paralelo
- Testes isolados por módulo
- Substituição fácil de implementações
- Baixo acoplamento entre módulos

**Negativas:**
- Boilerplate inicial maior
- Necessidade de disciplina nos contratos
- Overhead de abstrações

**Riscos:**
- Módulos criarem dependências diretas (mitigado por CI checks)
- Complexidade excessiva (mitigado por code reviews)

## 6. Protocolo de Desenvolvimento Paralelo (Modo Hive)

### 6.1 Branch Strategy para Módulos
```
main
├── develop
│   ├── feature/task-automation-v2
│   ├── feature/llm-orchestrator-v3
│   └── feature/security-auditor-v1
└── release/v12.0.0
```

### 6.2 CI/CD Pipeline por Módulo
```yaml
# .github/workflows/module-ci.yml
name: Module CI
on:
  push:
    paths:
      - 'src/modules/task-automation/**'
      - 'src/modules/llm-orchestrator/**'
      - 'src/modules/security-auditor/**'

jobs:
  test-module:
    runs-on: ubuntu-latest
    steps:
      - name: Test Module Isolation
        run: |
          # Testa módulo independentemente
          npm run test:module -- ${{ github.event.path }}
          
      - name: Integration Test
        run: |
          # Testa integração com módulos dependentes
          npm run test:integration
```

### 6.3 Dependency Management
```json
{
  "package.json": {

      "build:module": "lerna run build",

    }
  }
}
```

```typescript
// src/store/AppStateContext.tsx
import React, { createContext, useContext, useReducer, ReactNode, useEffect } from 'react'
import { Project, AppState, Action } from '@/types/app'
import { fetchProjects } from '@/services/api'

interface AppStateContextType {
  state: AppState
  dispatch: React.Dispatch<Action>
}

const initialState: AppState = {
  projects: [],
  isLoading: false,
  error: null,
  user: null,
  theme: 'light'
}

const AppStateContext = createContext<AppStateContextType | undefined>(undefined)

function appReducer(state: AppState, action: Action): AppState {
  switch (action.type) {
    case 'SET_PROJECTS':
      return { ...state, projects: action.payload, isLoading: false, error: null }
    case 'ADD_PROJECT':
      return { ...state, projects: [...state.projects, action.payload] }
    case 'UPDATE_PROJECT':
      return {
        ...state,
        projects: state.projects.map(project =>
          project.id === action.payload.id ? action.payload : project
        )
      }
    case 'DELETE_PROJECT':
      return {
        ...state,
        projects: state.projects.filter(project => project.id !== action.payload)
      }
    case 'SET_LOADING':
      return { ...state, isLoading: action.payload }
    case 'SET_ERROR':
      return { ...state, error: action.payload, isLoading: false }
    case 'SET_USER':
      return { ...state, user: action.payload }
    case 'SET_THEME':
      return { ...state, theme: action.payload }
    default:
      return state
  }
}

export function AppStateProvider({ children }: { children: ReactNode }) {
  const [state, dispatch] = useReducer(appReducer, initialState)

  useEffect(() => {
    const loadProjects = async () => {
      dispatch({ type: 'SET_LOADING', payload: true })
      try {
        const projects = await fetchProjects()
        dispatch({ type: 'SET_PROJECTS', payload: projects })
      } catch (error) {
        dispatch({ 
          type: 'SET_ERROR', 
          payload: error instanceof Error ? error.message : 'Failed to load projects' 
        })
      }
    }

    loadProjects()
  }, [])

  return (
    <AppStateContext.Provider value={{ state, dispatch }}>
      {children}
    </AppStateContext.Provider>
  )
}

export function useAppState() {
  const context = useContext(AppStateContext)
  if (context === undefined) {
    throw new Error('useAppState must be used within an AppStateProvider')
  }
  return context
}
```

```typescript
// src/services/api.ts
import { Project } from '@/types/app'

const API_BASE_URL = import.meta.env.VITE_API_BASE_URL || 'http://localhost:3001/api'

async function handleResponse<T>(response: Response): Promise<T> {
  if (!response.ok) {
    const error = await response.text()
    throw new Error(error || `HTTP error! status: ${response.status}`)
  }
  return response.json()
}

export async function fetchProjects(): Promise<Project[]> {
  const response = await fetch(`${API_BASE_URL}/projects`, {
    headers: {
      'Content-Type': 'application/json',
    },
  })
  return handleResponse<Project[]>(response)
}

export async function fetchProject(id: string): Promise<Project> {
  const response = await fetch(`${API_BASE_URL}/projects/${id}`, {
    headers: {
      'Content-Type': 'application/json',
    },
  })
  return handleResponse<Project>(response)
}

export async function createProject(project: Omit<Project, 'id' | 'createdAt' | 'updatedAt'>): Promise<Project> {
  const response = await fetch(`${API_BASE_URL}/projects`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(project),
  })
  return handleResponse<Project>(response)
}

export async function updateProject(id: string, updates: Partial<Project>): Promise<Project> {
  const response = await fetch(`${API_BASE_URL}/projects/${id}`, {
    method: 'PATCH',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(updates),
  })
  return handleResponse<Project>(response)
}

export async function deleteProject(id: string): Promise<void> {
  const response = await fetch(`${API_BASE_URL}/projects/${id}`, {
    method: 'DELETE',
  })
  if (!response.ok) {
    throw new Error(`Failed to delete project: ${response.status}`)
  }
}
```

# DIYAPP Evolution - V12 Core - Arquitetura Técnica

## 1. DECISÃO ARQUITETURAL - Stack Tecnológica V12

**ADR-001: Stack de Tecnologia para V12 Core**
**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead V12

### CONTEXTO:
A V11 do DIYAPP atingiu limitações de performance devido a:
1. Código monolítico sem separação clara de responsabilidades
2. Ausência de estratégia de caching estruturada
3. Dificuldade em escalar componentes individualmente
4. Acoplamento alto entre frontend e backend
5. Débito técnico acumulado em dependências desatualizadas

### DECISÃO:
Adotar arquitetura de micro-frontends com backend modular, utilizando:
- **Frontend:** React 18 + TypeScript + Vite
- **Backend:** Node.js 20 + Express + TypeScript
- **Banco de Dados:** PostgreSQL 16 + Redis 7 para caching
- **ORM:** Prisma 5 para type-safe database access
- **Comunicação:** REST APIs versionadas + WebSockets para real-time
- **Containerização:** Docker + Docker Compose para ambiente local
- **CI/CD:** GitHub Actions com testes automatizados

### OPÇÕES CONSIDERADAS:
- **Opção A (Vue 3 + NestJS):** Vue 3 com Composition API + NestJS com TypeORM
  - Prós: Vue mais fácil para novos devs, NestJS com arquitetura robusta
  - Contras: Ecossistema menor que React, TypeORM menos type-safe que Prisma
- **Opção B (React + Fastify):** React 18 + Fastify + Drizzle ORM
  - Prós: Fastify mais performático que Express, Drizzle mais leve
  - Contras: Menor adoção em produção, documentação menos madura
- **Opção C (Status Quo - Angular + Express):** Manter Angular 15 + Express atual
  - Prós: Zero custo de migração, time já conhece
  - Contras: Performance inferior, difícil contratação de devs Angular

**Opção escolhida:** Stack React/Node.js/PostgreSQL/Redis
**Justificativa:** Maior ecossistema, melhor performance, type-safety completo com Prisma, e alinhamento com tendências de mercado para facilitar contratações.

### CONSEQUÊNCIAS:
**Positivas:**
- Performance 3-5x melhor com Vite + React 18 features
- Type-safety end-to-end com TypeScript + Prisma
- Escalabilidade horizontal facilitada
- Developer experience significativamente melhorada

**Negativas:**
- Curva de aprendizado para time acostumado com Angular
- Migração complexa exigindo Modo Hive
- Aumento inicial em complexidade de configuração

**Riscos:**
- Migração pode introduzir bugs se não for faseada
- Mitigação: Modo Hive com refatoração paralela e feature flags
- Prisma pode ter performance em queries complexas
- Mitigação: Query otimização + caching estratégico com Redis

```typescript
  constructor(
    public message: string,
    public statusCode: number = 500,
    public isOperational: boolean = true,
    public code?: string
  ) {
    super(message);
    Object.setPrototypeOf(this, AppError.prototype);

  }
}

export class ValidationError extends AppError {
  constructor(message: string, details?: any) {
    super(message, 400, true, 'VALIDATION_ERROR');
    this.details = details;
  }
  details?: any;
}

export class NotFoundError extends AppError {
  constructor(resource: string) {
    super(`${resource} not found`, 404, true, 'NOT_FOUND');
  }
}

export class UnauthorizedError extends AppError {
  constructor(message: string = 'Unauthorized') {
    super(message, 401, true, 'UNAUTHORIZED');
  }
}

// Global error handler middleware
export const errorHandler = (
  err: Error,
  req: Request,
  res: Response,
  next: NextFunction
) => {
  const logger = req.app.get('logger');
  
  if (err instanceof AppError) {
    logger.warn(`Operational error: ${err.message}`, {
      code: err.code,
      statusCode: err.statusCode
    });
    
    return res.status(err.statusCode).json({
      success: false,
      error: {
        message: err.message,
        code: err.code,
        ...(err instanceof ValidationError && { details: err.details })
      }
    });
  }
  
  // Unexpected errors
  logger.error('Unexpected error:', {
    message: err.message,
    stack: err.stack,
    url: req.url,
    method: req.method
  });
  
  return res.status(500).json({
    success: false,
    error: {
      message: 'Internal server error',
      code: 'INTERNAL_ERROR'
    }
  });
};
```

## 4. ESTRATÉGIA DE CACHING - Redis Implementation

```typescript
// packages/core/src/utils/cache.ts
import Redis from 'ioredis';
import { logger } from '../lib/logger';

export interface CacheConfig {
  ttl: number; // Time to live in seconds
  prefix: string;
  fallbackToDB?: boolean;
}

export class CacheService {
  private client: Redis;
  private isConnected: boolean = false;

  constructor() {
    this.client = new Redis({
      host: process.env.REDIS_HOST || 'localhost',
      port: parseInt(process.env.REDIS_PORT || '6379'),
      password: process.env.REDIS_PASSWORD,
      retryStrategy: (times) => {
        const delay = Math.min(times * 50, 2000);
        return delay;
      },
    });

    this.client.on('connect', () => {
      this.isConnected = true;
      logger.info('Redis connected successfully');
    });

    this.client.on('error', (err) => {
      this.isConnected = false;
      logger.error('Redis connection error:', err);
    });
  }

  // Pattern: Cache-Aside (Lazy Loading)
  async getOrSet<T>(
    key: string,
    fetchData: () => Promise<T>,
    config: CacheConfig
  ): Promise<T> {
    const cacheKey = `${config.prefix}:${key}`;
    
    try {
      if (this.isConnected) {
        const cached = await this.client.get(cacheKey);
        if (cached) {
          logger.debug(`Cache hit: ${cacheKey}`);
          return JSON.parse(cached);
        }
      }
    } catch (error) {
      logger.warn(`Cache read error for ${cacheKey}:`, error);
    }

    // Cache miss or Redis down
    logger.debug(`Cache miss: ${cacheKey}`);
    const data = await fetchData();
    
    try {
      if (this.isConnected) {
        await this.client.setex(
          cacheKey,
          config.ttl,
          JSON.stringify(data)
        );
        logger.debug(`Cache set: ${cacheKey} (TTL: ${config.ttl}s)`);
      }
    } catch (error) {
      logger.warn(`Cache write error for ${cacheKey}:`, error);
    }
    
    return data;
  }

  // Pattern: Write-Through for critical data
  async writeThrough<T>(
    key: string,
    data: T,
    writeToDB: () => Promise<void>,
    config: CacheConfig
  ): Promise<void> {
    const cacheKey = `${config.prefix}:${key}`;
    
    // 1. Write to database first
    await writeToDB();
    
    // 2. Update cache
    try {
      if (this.isConnected) {
        await this.client.setex(
          cacheKey,
          config.ttl,
          JSON.stringify(data)
        );
        logger.debug(`Write-through cache: ${cacheKey}`);
      }
    } catch (error) {
      logger.warn(`Write-through cache error:`, error);
    }
  }

  // Pattern: Cache Invalidation
  async invalidatePattern(pattern: string): Promise<void> {
    try {
      if (this.isConnected) {
        const keys = await this.client.keys(pattern);
        if (keys.length > 0) {
          await this.client.del(...keys);
          logger.info(`Invalidated ${keys.length} cache keys matching: ${pattern}`);
        }
      }
    } catch (error) {
      logger.error(`Cache invalidation error:`, error);
    }
  }

  // Health check
  async healthCheck(): Promise<boolean> {
    try {
      await this.client.ping();
      return true;
    } catch {
      return false;
    }
  }

  // Close connection
  async disconnect(): Promise<void> {
    await this.client.quit();
  }
}

// Singleton instance
export const cacheService = new CacheService();
```

```javascript
// src/database/pool.js
const { Pool } = require('pg');
const config = require('../config/env');
const logger = require('../utils/logger');

class DatabasePool {
  constructor() {
    this.pool = new Pool({
      connectionString: config.DATABASE_URL,
      max: config.DB_POOL_MAX,
      min: config.DB_POOL_MIN,
      idleTimeoutMillis: config.DB_IDLE_TIMEOUT,
      connectionTimeoutMillis: config.DB_CONNECTION_TIMEOUT,
    });

    this.setupEventListeners();
  }

  setupEventListeners() {
    this.pool.on('connect', () => {
      logger.debug('New database connection established');
    });

    this.pool.on('error', (err) => {
      logger.error('Unexpected database pool error', { error: err.message });
    });

    this.pool.on('remove', () => {
      logger.debug('Database connection removed from pool');
    });
  }

  async query(text, params) {
    const start = Date.now();
    try {
      const result = await this.pool.query(text, params);
      const duration = Date.now() - start;
      
      logger.debug('Database query executed', {
        query: text,
        duration,
        rowCount: result.rowCount
      });
      
      return result;
    } catch (error) {
      const duration = Date.now() - start;
      logger.error('Database query failed', {
        query: text,
        params: this.maskSensitiveData(params),
        duration,
        error: error.message
      });
      throw error;
    }
  }

  maskSensitiveData(params) {
    if (!params) return params;
    return params.map(param => {
      if (typeof param === 'string' && (
        param.includes('@') || // email
        param.match(/^\d{3}\.\d{3}\.\d{3}-\d{2}$/) || // CPF
        param.match(/^\d{16}$/) // credit card
      )) {
        return '[MASKED]';
      }
      return param;
    });
  }

  async connect() {
    return this.pool.connect();
  }

  async end() {
    await this.pool.end();
    logger.info('Database pool ended');
  }

  getPool() {
    return this.pool;
  }
}

module.exports = new DatabasePool();
```

```javascript
// src/database/migrations/run-migrations.js
const db = require('../pool');
const fs = require('fs');
const path = require('path');
const logger = require('../../utils/logger');

class MigrationRunner {
  constructor() {
    this.migrationsDir = path.join(__dirname, '.');
    this.migrationsTable = 'schema_migrations';
  }

  async init() {
    await db.query(`
      CREATE TABLE IF NOT EXISTS ${this.migrationsTable} (
        version VARCHAR(255) PRIMARY KEY,
        applied_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
      );
    `);
  }

  async getAppliedMigrations() {
    const result = await db.query(`SELECT version FROM ${this.migrationsTable} ORDER BY version`);
    return result.rows.map(row => row.version);
  }

  async getMigrationFiles() {
    const files = fs.readdirSync(this.migrationsDir)
      .filter(file => file.endsWith('.js') && file !== 'run-migrations.js' && file !== 'create-migration.js')
      .sort();
    
    return files.map(file => ({
      version: file.split('_')[0],
      name: file,
      path: path.join(this.migrationsDir, file)
    }));
  }

  async up() {
    await this.init();
    const applied = await this.getAppliedMigrations();
    const files = await this.getMigrationFiles();
    
    const pending = files.filter(file => !applied.includes(file.version));
    
    if (pending.length === 0) {
      logger.info('No pending migrations');
      return;
    }

    logger.info(`Found ${pending.length} pending migration(s)`);
    
    for (const migration of pending) {
      try {
        logger.info(`Applying migration: ${migration.name}`);
        
        // Start transaction
        await db.query('BEGIN');
        
        // Load and run migration
        const migrationModule = require(migration.path);
        await migrationModule.up(db);
        
        // Record migration
        await db.query(
          `INSERT INTO ${this.migrationsTable} (version) VALUES ($1)`,
          [migration.version]
        );
        
        await db.query('COMMIT');
        logger.info(`Migration ${migration.name} applied successfully`);
        
      } catch (error) {
        await db.query('ROLLBACK');
        logger.error(`Failed to apply migration ${migration.name}`, { error: error.message });
        throw error;
      }
    }
  }

  async down(version = null) {
    await this.init();
    const applied = await this.getAppliedMigrations();
    
    if (applied.length === 0) {
      logger.info('No migrations to rollback');
      return;
    }

    const files = await this.getMigrationFiles();
    const targetVersion = version || applied[applied.length - 1];
    
    const migrationToRollback = files.find(file => file.version === targetVersion);
    
    if (!migrationToRollback) {
      throw new Error(`Migration ${targetVersion} not found`);
    }

    try {
      logger.info(`Rolling back migration: ${migrationToRollback.name}`);
      
      await db.query('BEGIN');
      
      const migrationModule = require(migrationToRollback.path);
      await migrationModule.down(db);
      
      await db.query(
        `DELETE FROM ${this.migrationsTable} WHERE version = $1`,
        [targetVersion]
      );
      
      await db.query('COMMIT');

      logger.error(`Failed to rollback migration ${migrationToRollback.name}`, { error: error.message });
      throw error;
    }
  }

  const command = process.argv[2];
  const version = process.argv[3] || null;
  
  const runner = new MigrationRunner();
  
  try {
    switch (command) {
      case 'up':
        await runner.up();
        break;
      case 'down':
        await runner.down(version);
        break;
      default:
        console.log('Usage: node run-migrations.js [up|down] [version]');
        process.exit(1);
    }
  } catch (error) {
    logger.error('Migration failed', { error: error.message });
    process.exit(1);
  } finally {
    await db.end();
  }
}

  run();
```

// ARQUITETURA TÉCNICA V12 - DIYAPP EVOLUTION
// Arquivo: architecture-v12.md
// Data: 2024-01-15
// Status: Aceita
// Autores: Tech Lead + Squad Autônoma

# ADR-001: Arquitetura Técnica V12 Core

## CONTEXTO:
A DIYAPP evoluiu para uma plataforma complexa com múltiplos módulos autônomos.
A V11 apresentou problemas de performance devido a:
1. Acoplamento excessivo entre componentes
2. Estratégia de caching inconsistente
3. Estrutura de pastas não escalável
4. Dificuldade em deploy independente de módulos

Forças em jogo:
- Necessidade de 100% autonomia das squads
- Performance crítica em tempo real (chat, processamento LLM)
- Manutenibilidade com equipe distribuída
- Observabilidade em produção

## DECISÃO:
Adotar arquitetura de micro-frontends com backend monolítico modular (MonoRepo),
utilizando Domain-Driven Design para organização e estratégia de caching em múltiplas camadas.

## OPÇÕES CONSIDERADAS:

### Opção A: Microserviços completos
- **Prós**: Desacoplamento máximo, deploy independente, escalabilidade granular
- **Contras**: Complexidade operacional, latência entre serviços, consistência de dados complexa
- **Custo**: Alto (necessidade de orquestração, service mesh, monitoramento distribuído)

### Opção B: Monolito tradicional
- **Prós**: Simplicidade operacional, consistência forte, debugging facilitado
- **Contras**: Acoplamento, dificuldade de escalar componentes específicos, deploy all-or-nothing
- **Custo**: Baixo inicial, alto de manutenção

### Opção C: MonoRepo com módulos autônomos (ESCOLHIDA)
- **Prós**: 
  - Desacoplamento lógico mantendo operacionalidade simples
  - Compartilhamento de código via pacotes internos
  - Deploy seletivo possível via feature flags
  - Estrutura DDD clara para organização
- **Contras**: 
  - Build times maiores (mitigado com incremental builds)
  - Necessidade de disciplina na definição de contratos
- **Custo**: Médio (tooling MonoRepo necessário)

## CONSEQUÊNCIAS:

### Positivas:
1. **Autonomia das squads**: Cada domínio pode evoluir independentemente
2. **Performance otimizada**: Caching em múltiplas camadas (Redis, CDN, browser)
3. **Manutenibilidade**: Estrutura clara, testes isolados por módulo
4. **Onboarding**: Nova dev entende estrutura em minutos

### Negativas (trade-offs aceitos):
1. **Complexidade inicial**: Setup do MonoRepo requer configuração cuidadosa
2. **Dependência do tooling**: Lerna/Turborepo/Nx como dependência crítica
3. **Overhead de comunicação**: Contratos entre módulos precisam ser bem definidos

### Riscos e Mitigações:
- **Risco**: Degradação de performance no build
  - **Mitigação**: Implementar cache de builds, parallel execution
- **Risco**: Quebra de contratos entre módulos
  - **Mitigação**: Testes de integração automatizados, versionamento semântico
- **Risco**: Complexidade excessiva para pequenas features
  - **Mitigação**: Manter módulo "shared" para utilitários comuns

### Estratégia de Caching (Multi-layer)

```typescript
// Exemplo de implementação em packages/shared/src/cache/multiLayerCache.ts
export class MultiLayerCache {
  constructor(
    private memoryCache: Map<string, { data: any; expiry: number }>,
    private redisClient: Redis,
    private cdnClient: CDNClient
  ) {}

  async get<T>(key: string, options: CacheOptions = {}): Promise<T | null> {
    const {
      ttl = 300, // 5 minutos default
      layer = 'all', // 'memory' | 'redis' | 'cdn' | 'all'
      fallbackFn // Função para gerar dados se cache miss
    } = options;

    // 1. Tentar memory cache (nanoseconds)
    if (layer === 'all' || layer === 'memory') {
      const memoryHit = this.memoryCache.get(key);
      if (memoryHit && memoryHit.expiry > Date.now()) {
        metrics.cacheHit('memory');
        return memoryHit.data as T;
      }
    }

    // 2. Tentar Redis (microseconds)
    if (layer === 'all' || layer === 'redis') {
      try {
        const redisData = await this.redisClient.get(key);
        if (redisData) {
          const parsed = JSON.parse(redisData);
          
          // Popular memory cache
          this.memoryCache.set(key, {
            data: parsed,
            expiry: Date.now() + (ttl * 1000)
          });
          
          metrics.cacheHit('redis');
          return parsed as T;
        }
      } catch (error) {
        logger.warn('Redis cache failed, continuing', { key, error });
      }
    }

    // 3. Se tem fallback function, executar e cachear
    if (fallbackFn) {
      const freshData = await fallbackFn();
      
      // Cache em todas as camadas (async, não-blocking)
      this.set(key, freshData, ttl).catch(err => {
        logger.error('Failed to cache data', { key, err });
      });
      
      return freshData as T;
    }

    metrics.cacheMiss(key);
    return null;
  }

  async set(key: string, data: any, ttl: number): Promise<void> {
    const expiry = Date.now() + (ttl * 1000);
    
    // 1. Memory cache (síncrono)
    this.memoryCache.set(key, { data, expiry });
    
    // 2. Redis cache (assíncrono)
    await this.redisClient.setex(
      key,
      ttl,
      JSON.stringify(data)
    );
    
    // 3. CDN para assets estáticos (se aplicável)
    if (this.isStaticAsset(key)) {
      await this.cdnClient.purge(key);
    }
  }
}

// Tipos de cache por domínio:
export const CACHE_STRATEGIES = {
  USER_PROFILE: {
    ttl: 300, // 5 minutos
    layer: 'memory+redis',
    invalidateOn: ['USER_UPDATE', 'PROFILE_CHANGE']
  },
  CHAT_HISTORY: {
    ttl: 3600, // 1 hora
    layer: 'redis',

  },

  },
```

```typescript
    total?: number;

      hit: boolean;
      source: 'memory' | 'redis' | 'database';
    };
  };

  error: {
    code: string;
    message: string;
    details?: Record<string, any>;
  };
  data?: never;
}

// 3. Middleware de cache automático para GET
import { RequestHandler } from 'express';

export const cacheMiddleware = (ttl: number): RequestHandler => {
  return async (req, res, next) => {
    // Apenas cache GET requests
    if (req.method !== 'GET') {
      return next();
    }
    
    const cacheKey = `api:${req.originalUrl}:${JSON.stringify(req.query)}`;
    
    try {
      const cached = await redis.get(cacheKey);
      if (cached) {
        // Header para indicar cache hit
        res.set('X-Cache', 'HIT');
        return res.json(JSON.parse(cached));
      }
      
      // Interceptar response para cachear
      const originalJson = res.json;
      res.json = function(body) {
        // Cachear apenas respostas bem sucedidas
        if (res.statusCode >= 200 && res.statusCode < 300) {
          redis.setex(cacheKey, ttl, JSON.stringify(body))
            .catch(err => logger.warn('Cache set failed', { err }));
        }
        return originalJson.call(this, body);
      };
      
      res.set('X-Cache', 'MISS');
      next();
    } catch (error) {
      // Se cache falhar, continuar sem cache
      logger.warn('Cache middleware failed', { error });
      next();
    }
  };
};
```

```javascript
    client.release();
  }

// Query wrapper with error handling and logging
const query = async (text, params) => {
  const start = Date.now();
  const correlation_id = global.correlationId || 'system';
  
  try {
    const result = await pool.query(text, params);
    const duration = Date.now() - start;
    
    logger.debug('Database query executed', {
      correlation_id,
      query: text,
      params: maskSensitiveData(params),
      duration_ms: duration,
      row_count: result.rowCount
    });
    
    return result;
  } catch (error) {
    logger.error('Database query failed', {
      correlation_id,
      query: text,
      params: maskSensitiveData(params),
      error: error.message,
      stack: error.stack
    });
    throw error;
  }
};

// Mask sensitive data in logs
const maskSensitiveData = (params) => {
  if (!params) return params;
  
  const sensitiveFields = ['password', 'token', 'secret', 'api_key', 'credit_card'];
  return params.map(param => {
    if (typeof param === 'string' && sensitiveFields.some(field => 
      param.toLowerCase().includes(field))) {
      return '***MASKED***';
    }
    return param;
  });
};
```

const fs = require('fs').promises;
const path = require('path');
const { query } = require('../connection');
const logger = require('../../utils/logger');

class MigrationRunner {
  constructor() {
    this.migrationsDir = path.join(__dirname, 'scripts');
    this.migrationsTable = 'schema_migrations';
  }

  async ensureMigrationsTable() {
    await query(`
      CREATE TABLE IF NOT EXISTS ${this.migrationsTable} (
        version VARCHAR(255) PRIMARY KEY,
        name VARCHAR(255) NOT NULL,
        applied_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
      );
    `);
  }

  async getAppliedMigrations() {
    const result = await query(`SELECT version FROM ${this.migrationsTable} ORDER BY applied_at`);
    return result.rows.map(row => row.version);
  }

  async getMigrationFiles(direction) {
    const files = await fs.readdir(this.migrationsDir);
    return files
      .filter(file => file.endsWith(`.${direction}.sql`))
      .sort();
  }

  async runMigration(file, direction) {
    const version = file.split('.')[0];
    const filePath = path.join(this.migrationsDir, file);
    const sql = await fs.readFile(filePath, 'utf8');

    try {
      await query('BEGIN');
      await query(sql);
      
      if (direction === 'up') {
        await query(
          `INSERT INTO ${this.migrationsTable} (version, name) VALUES ($1, $2)`,
          [version, file]
        );
        logger.info(`Migration applied: ${file}`);
      } else {
        await query(
          `DELETE FROM ${this.migrationsTable} WHERE version = $1`,
          [version]
        );
        logger.info(`Migration rolled back: ${file}`);
      }
      
      await query('COMMIT');
    } catch (error) {
      await query('ROLLBACK');
      logger.error(`Migration failed: ${file}`, error);
      throw error;
    }
  }

    try {
      await this.ensureMigrations
