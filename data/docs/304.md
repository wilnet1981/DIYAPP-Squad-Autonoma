// Controller com validaГ§ГЈo e tratamento de erros
@Controller('/api/v1/users')
export class UserController {
  constructor(private readonly userService: UserService) {}

  @Get(':id')
  @UsePipes(new ValidationPipe())
  async getUser(
    @Param('id', ParseUUIDPipe) id: string,
    @Res() response: Response
  ): Promise<void> {
    try {
      const user = await this.userService.findById(id);
      
      if (!user) {
        throw new NotFoundException('User not found');
      }

      response.json({
        success: true,
        data: user,
        timestamp: new Date().toISOString()
      });
    } catch (error) {
      this.logger.error(`Failed to fetch user ${id}:`, error);
      
      if (error instanceof NotFoundException) {
        throw error;
      }
      
      throw new InternalServerErrorException('Failed to fetch user');
    }
  }

  // Rate limiting e caching
  @Get(':id/profile')
  @UseGuards(RateLimitGuard)
  @CacheTTL(300) // 5 minutes cache
  async getUserProfile(@Param('id') id: string) {
    return this.userService.getProfileWithCache(id);
  }
}

// Service com Domain-Driven Design
@Service()
export class UserService {
  constructor(
    @InjectRepository(User)
    private userRepository: Repository<User>,
    private eventEmitter: EventEmitter2
  ) {}

  async createUser(createUserDto: CreateUserDto): Promise<User> {
    // ValidaГ§ГЈo de negГіcio
    if (await this.userRepository.exists({ email: createUserDto.email })) {
      throw new BusinessRuleException('Email already registered');
    }

    const user = this.userRepository.create(createUserDto);
    const savedUser = await this.userRepository.save(user);

    // Evento de domГ­nio
    this.eventEmitter.emit(
      'user.created',
      new UserCreatedEvent(savedUser.id, savedUser.email)
    );

    return savedUser;
  }
}

**Status**: Aceita

## Contexto

- TransaГ§Гµes ACID
- Consultas complexas com JOINs
- Controle de concorrГЄncia
- Backup e recovery confiГЎveis

**Positivas**:
- TransaГ§Гµes ACID garantidas
- Suporte a JSONB para dados semi-estruturados
- ReplicaГ§ГЈo nativa para alta disponibilidade
- Comunidade ativa e documentaГ§ГЈo extensa

**Negativas**:

# ADR-002: Cache em Duas Camadas (Redis + MemГіria)

**Data**: 2024-01-01
**Status**: Aceita
**Autores**: Tech Lead, Especialista Performance

## Contexto
Necessidade de reduzir latГЄncia para dados frequentemente acessados:
- Perfil de usuГЎrio: 95% das requisiГ§Гµes sГЈo leitura
- ConfiguraГ§Гµes do sistema: alteradas raramente
- Dados de referГЄncia: atualizados diariamente

## DecisГЈo
Implementar cache em duas camadas:
1. **L1**: Cache em memГіria (Node.js) para dados ultra-quentes
2. **L2**: Redis compartilhado para dados quentes

## ImplementaГ§ГЈo
```typescript
class TwoLayerCache {
  private localCache = new Map<string, { data: any; expiry: number }>();
  
  async get(key: string): Promise<any> {
    // Check L1 cache
    const localItem = this.localCache.get(key);
    if (localItem && localItem.expiry > Date.now()) {
      return localItem.data;
    }
    
    // Check L2 cache (Redis)
    const redisData = await redisClient.get(key);
    if (redisData) {
      // Populate L1 cache
      this.localCache.set(key, {
        data: JSON.parse(redisData),
        expiry: Date.now() + 60000 // 1 minute
      });
      return JSON.parse(redisData);
    }
    
    return null;
  }
}
```

## ConsequГЄncias
**Performance Esperada**:
- L1 cache: < 1ms
- L2 cache: < 5ms
- Database: 20-50ms
```

name: CI Pipeline - Staging V12

on:
  push:
    branches: [main, staging/v12]
  pull_request:
    branches: [main]

  quality-gates:

    - uses: actions/checkout@v3

      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt

    - name: Lint with flake8
      run: |
        flake8 src/ --count --select=E9,F63,F7,F82 --show-source --statistics
        flake8 src/ --count --exit-zero --max-complexity=10 --max-line-length=127 --statistics
    
    - name: Type check with mypy
      run: |
        mypy src/ --ignore-missing-imports
    
    - name: Format check with black
      run: |
        black --check src/
    
    - name: Run unit tests
      run: |
        pytest src/tests/ -v --cov=src --cov-report=xml --cov-report=html
    
    - name: Upload coverage to Codecov
      uses: codecov/codecov-action@v3
      with:
        file: ./coverage.xml
        flags: unittests
    
    - name: Security scan
      uses: snyk/actions/python@master
      env:
        SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
      with:
        args: --severity-threshold=high
    
    - name: Build Docker image
      run: |
        docker build -t $IMAGE_NAME:${{ github.sha }} -f infrastructure/docker/Dockerfile .
    
    - name: Run integration tests
      run: |
        docker-compose -f infrastructure/docker/docker-compose.test.yml up --abort-on-container-exit
    
    - name: Trivy vulnerability scanner
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: '$IMAGE_NAME:${{ github.sha }}'
        format: 'sarif'
        output: 'trivy-results.sarif'
    
    - name: Upload Trivy scan results
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: 'trivy-results.sarif'
    
    - name: Check SLO compliance
      run: |
        python scripts/check-slo-compliance.py

  deploy-to-staging:
    needs: quality-gates
    if: github.ref == 'refs/heads/main' || github.ref == 'refs/heads/staging/v12'
    runs-on: ubuntu-latest
    environment: staging-v12
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Configure AWS credentials
      uses: aws-actions/configure-aws-credentials@v2
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: us-east-1
    
    - name: Login to Container Registry
      uses: docker/login-action@v2
      with:
        registry: ${{ env.REGISTRY }}
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}
    
    - name: Build and push Docker image
      run: |
        docker build -t $IMAGE_NAME:${{ github.sha }} -f infrastructure/docker/Dockerfile .
        docker tag $IMAGE_NAME:${{ github.sha }} $IMAGE_NAME:latest-staging
        docker push $IMAGE_NAME:${{ github.sha }}
        docker push $IMAGE_NAME:latest-staging
    
    - name: Deploy to Staging

name: CD Pipeline - Auto Healing

on:
  schedule:
    - cron: '*/5 * * * *'  # Run every 5 minutes
  workflow_dispatch:

jobs:
  health-monitoring:
    runs-on: ubuntu-latest
    environment: staging-v12
    
    steps:
    - name: Check staging health
      run: |
        STAGING_URL=${{ secrets.STAGING_URL }}
        RESPONSE=$(curl -s -o /dev/null -w "%{http_code}" -H "X-Health-Check: true" $STAGING_URL/health)
        
        if [ "$RESPONSE" != "200" ]; then
          echo "Staging health check failed: $RESPONSE"
          
          # Attempt auto-healing
          ./scripts/auto-heal.sh
          
          # If still failing after 3 attempts, trigger rollback
          ATTEMPTS=3
          for i in $(seq 1 $ATTEMPTS); do
            sleep 30
            RESPONSE=$(curl -s -o /dev/null -w "%{http_code}" $STAGING_URL/health)
            if [ "$RESPONSE" = "200" ]; then
              echo "Auto-healing successful on attempt $i"
              exit 0
            fi
          done
          
          # Trigger rollback
          echo "Triggering automatic rollback"
          ./scripts/rollback-staging.sh
          
          # Create incident
          curl -X POST ${{ secrets.PAGERDUTY_URL }} \
            -H "Content-Type: application/json" \
            -d '{
              "incident": {
                "type": "incident",
                "title": "Staging V12 Auto-Rollback Triggered",
                "service": {
                  "id": "staging-v12",
                  "type": "service_reference"
                },
                "priority": {
                  "id": "P2",
                  "type": "priority_reference"
                }
              }
            }'
        else
          echo "Staging health check passed: $RESPONSE"
        fi
    
    - name: Monitor SLO compliance

const { RetentionPolicy } = require('../models/RetentionPolicy');
const { DataProcessingRecord } = require('../models/DataProcessingRecord');
const logger = require('../../utils/logger');

class DataRetentionController {
  constructor() {
    this.policy = require('../policies/retentionPolicy.json');
    this.retentionJobs = new Map();
  }

  /**
   * Inicializa jobs de retenГ§ГЈo automГЎtica
   */
  async initializeRetentionScheduler() {
    try {
      // Job diГЎrio para verificaГ§ГЈo de expiraГ§ГЈo
      this.retentionJobs.set('dailyCheck', {
        interval: '0 2 * * *', // 2 AM daily
        handler: this.executeRetentionCycle.bind(this)
      });

      // Job semanal para limpeza de backups
      this.retentionJobs.set('backupCleanup', {
        interval: '0 3 * * 0', // 3 AM every Sunday
        handler: this.cleanupOldBackups.bind(this)
      });

      logger.info('Scheduler de retenГ§ГЈo inicializado');
      return true;
    } catch (error) {
      logger.error('Erro ao inicializar scheduler de retenГ§ГЈo:', error);
      throw error;
    }
  }

  /**
   * Executa ciclo de retenГ§ГЈo para todas as categorias
   */
  async executeRetentionCycle() {
    const results = {
      processed: 0,
      deleted: 0,
      anonymized: 0,
      errors: []
    };

    try {
      const categories = Object.keys(this.policy.dataCategories);
      
      for (const category of categories) {
        const categoryResult = await this.processCategoryRetention(category);
        
        results.processed += categoryResult.processed;
        results.deleted += categoryResult.deleted;
        results.anonymized += categoryResult.anonymized;
        
        if (categoryResult.error) {
          results.errors.push({
            category,
            error: categoryResult.error
          });
        }
      }

      // Registrar auditoria
      await this.logRetentionAudit(results);
      
      logger.info(`Ciclo de retenГ§ГЈo executado: ${JSON.stringify(results)}`);
      return results;
    } catch (error) {
      logger.error('Erro no ciclo de retenГ§ГЈo:', error);
      throw error;
    }
  }

  /**
   * Processa retenГ§ГЈo para uma categoria especГ­fica
   */
  async processCategoryRetention(category) {
    const policy = this.policy.dataCategories[category];
    if (!policy) {
      throw new Error(`PolГ­tica nГЈo encontrada para categoria: ${category}`);
    }

    const retentionDays = parseInt(policy.retentionPeriod);
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - retentionDays);

    const result = {
      category,
      processed: 0,
      deleted: 0,
      anonymized: 0,
      error: null
    };

    try {
      // Buscar registros expirados
      const expiredRecords = await DataProcessingRecord.findExpired(
        category,
        cutoffDate
      );

      result.processed = expiredRecords.length;

      for (const record of expiredRecords) {
        try {
          switch (policy.deletionMethod) {
            case 'automated':
              if (policy.postDeletionAction === 'anonymize') {
                await this.anonymizeRecord(record, policy);
                result.anonymized++;
              } else {
                await this.deleteRecord(record, policy);
                result.deleted++;
              }
              break;

            case 'manual_review':
              await this.flagForManualReview(record, policy);
              break;

            default:
              logger.warn(`MГ©todo de deleГ§ГЈo desconhecido: ${policy.deletionMethod}`);
          }
        } catch (recordError) {
          logger.error(`Erro ao processar registro ${record.id}:`, recordError);
        }
      }

      return result;
    } catch (error) {
      result.error = error.message;
      return result;
    }
  }

  /**
   * Anonimiza registro conforme polГ­tica
   */
  async anonymizeRecord(record, policy) {
    const anonymizedData = this.generateAnonymizedData(record.data);
    
    await DataProcessingRecord.update(record.id, {
      data: anonymizedData,
      status: 'anonymized',
      anonymizedAt: new Date(),
      retentionAction: 'automatic_anonymization'
    });

    // Log de auditoria
    await this.logAnonymizationAudit(record, policy);
  }

  /**
   * Exclui registro permanentemente
   */
  async deleteRecord(record, policy) {
    // Verificar se hГЎ dependГЄncias legais
    const legalHolds = await this.checkLegalHolds(record);
    if (legalHolds.length > 0) {
      await this.suspendDeletion(record, legalHolds);
      return;
    }

    // ExclusГЈo real (soft delete primeiro)
    await DataProcessingRecord.softDelete(record.id, {
      deletedAt: new Date(),
      deletionReason: 'retention_policy',
      policyVersion: this.policy.version
    });

    // ExclusГЈo fГ­sica programada
    await this.schedulePhysicalDeletion(record.id);

    // Log de auditoria
    await this.logDeletionAudit(record, policy);
  }

  generateAnonymizedData(originalData) {
    return {
      originalId: originalData.id,
      anonymizedAt: new Date().toISOString(),
      dataCategory: originalData.category,
      metadata: {
        originalRecordType: originalData.type,
        anonymizationMethod: 'pseudonymization_v2'
      }
    };
  }

  /**
   * Limpa backups antigos
   */
  async cleanupOldBackups() {
    const backupPolicy = this.policy.backupRetention;
    const now = new Date();
    
    const backupTypes = ['daily', 'weekly', 'monthly'];
    
    for (const type of backupTypes) {
      const retentionDays = parseInt(backupPolicy[type]);
      const cutoffDate = new Date(now);
      cutoffDate.setDate(cutoffDate.getDate() - retentionDays);
      
      // Implementar lГіgica de limpeza de backups
      // Esta Г© uma implementaГ§ГЈo genГ©rica - adaptar ao sistema de backup real
      await this.deleteOldBackupFiles(type, cutoffDate);
    }
  }

  /**
   * Registra auditoria de retenГ§ГЈo
   */
  async logRetentionAudit(results) {
    const AuditLog = require('../models/AuditLog');
    
    await AuditLog.create({
      eventType: 'retention_cycle',
      userId: 'system',
      timestamp: new Date(),
      details: {
        results,
        policyVersion: this.policy.version,
        executionTime: new Date().toISOString()
      },
      complianceFrameworks: ['LGPD', 'GDPR', 'SOC2']
    });
  }

  /**
   * Verifica retenГ§ГЈo para nova feature
   */
  async assessFeatureRetention(featureData) {
    const assessment = {
      featureId: featureData.id,
      assessmentDate: new Date(),
      dataCategories: [],
      retentionRequirements: [],
      complianceStatus: 'pending',
      recommendations: []
    };

    // Identificar categorias de dados
    const identifiedCategories = this.identifyDataCategories(featureData);
    assessment.dataCategories = identifiedCategories;

    // Avaliar requisitos por categoria
    for (const category of identifiedCategories) {
      const policy = this.policy.dataCategories[category];
      if (policy) {
        assessment.retentionRequirements.push({
          category,
          retentionPeriod: `${policy.retentionPeriod} ${policy.unit}`,
          legalBasis: policy.legalBasis,
          deletionMethod: policy.deletionMethod
        });

        // Verificar conformidade
        if (!this.isCompliant(featureData, category, policy)) {
          assessment.complianceStatus = 'non_compliant';
          assessment.recommendations.push(
            `Ajustar retenГ§ГЈo para categoria ${category}: ${policy.justification}`
          );
        }
      }
    }

    // Gerar DPIA se necessГЎrio
    if (this.requiresDPIA(featureData)) {
      assessment.dpiaRequired = true;
      assessment.dpiaRecommendation = 'Conduzir DPIA antes da implementaГ§ГЈo';
    }

    return assessment;
  }

  /**
   * Identifica categorias de dados em uma feature
   */
  identifyDataCategories(featureData) {
    const categories = [];
    
    // AnГЎlise heurГ­stica dos dados
    if (featureData.collectsPersonalInfo) {
      categories.push('userProfile');
    }
    
    if (featureData.processesPayments) {
      categories.push('paymentInformation');
    }
    
    if (featureData.usesLLM) {
      categories.push('llmInteractions');
      
      // ClassificaГ§ГЈo EU AI Act
      const aiActClassification = this.classifyAIUsage(featureData);
      featureData.aiActClassification = aiActClassification;
    }
    
    if (featureData.storesUserActivity) {
      categories.push('userActivity');
    }
    
    return categories;
  }

  /**
   * Classifica uso de IA conforme EU AI Act
   */
  classifyAIUsage(featureData) {
    const { aiUsage } = featureData;
    
    if (aiUsage.includes('biometric_identification')) {
      return 'high_risk';
    }
    
    if (aiUsage.includes('employment_decisions') || 
        aiUsage.includes('credit_scoring')) {
      return 'high_risk';
    }
    
    if (aiUsage.includes('chatbot') || 
        aiUsage.includes('content_generation')) {
      return 'limited_risk';
    }
    
    if (aiUsage.includes('spam_filter') || 
        aiUsage.includes('recommendation')) {
      return 'minimal_risk';
    }
    
    return 'limited_risk'; // Default conservador
  }

  /**
   * Verifica conformidade com polГ­tica
   */
  isCompliant(featureData, category, policy) {
    // Implementar lГіgica de verificaГ§ГЈo
    return featureData.retentionPeriod <= parseInt(policy.retentionPeriod);
  }

  /**
   * Verifica necessidade de DPIA
   */
  requiresDPIA(featureData) {
    const highRiskCategories = [
      'paymentInformation',
      'biometricData',
      'healthData',
      'childrenData'
    ];
    
    return featureData.dataCategories.some(cat => 
      highRiskCategories.includes(cat) || 
      featureData.aiActClassification === 'high_risk'
    );
  }
}

class CorrelationMiddleware:
    async def __call__(self, request: Request, call_next):
        correlation_id = request.headers.get('X-Correlation-ID', str(uuid.uuid4()))
        
        # Adiciona correlation_id ao logger
        logger = logging.LoggerAdapter(
            logging.getLogger(__name__),
            {'correlation_id': correlation_id}
        )
        
        request.state.correlation_id = correlation_id
        request.state.logger = logger
        
        ACTIVE_REQUESTS.inc()
        start_time = time.time()
        
        try:
            response = await call_next(request)
            duration = time.time() - start_time
            
            # Log estruturado
            logger.info(
                "Request completed",
                extra={
                    "method": request.method,
                    "endpoint": request.url.path,
                    "status_code": response.status_code,
                    "duration_ms": round(duration * 1000, 2),
                    "user_agent": request.headers.get("user-agent", ""),
                    "client_ip": request.client.host if request.client else None
                }
            )
            
            # MГ©tricas
            REQUEST_COUNT.labels(
                method=request.method,
                endpoint=request.url.path,
                status=response.status_code
            ).inc()
            
            REQUEST_LATENCY.labels(
                method=request.method,
                endpoint=request.url.path
            ).observe(duration)
            
            response.headers["X-Correlation-ID"] = correlation_id
            return response
            
        except Exception as e:
            duration = time.time() - start_time
            logger.error(
                f"Request failed: {str(e)}",
                extra={
                    "method": request.method,
                    "endpoint": request.url.path,
                    "duration_ms": round(duration * 1000, 2),
                    "error_type": type(e).__name__
                },
                exc_info=True
            )
            raise
        finally:
            ACTIVE_REQUESTS.dec()

class DatabaseConnection:
    """Pool de conexГµes com circuit breaker e retry"""
    
    _pool = None
    
    @classmethod
    @circuit(
        failure_threshold=config.CB_FAILURE_THRESHOLD,
        recovery_timeout=config.CB_RECOVERY_TIMEOUT,

    )
    @retry(

        wait=wait_exponential(multiplier=1, min=4, max=10),

    )
    async def get_pool(cls):
        if cls._pool is None:
            try:
                cls._pool = await asyncpg.create_pool(

                )
                DB_CONNECTIONS.set(1)
                logger.info("Database connection pool created")
            except Exception as e:
                logger.error(f"Failed to create database pool: {str(e)}")
                raise
        return cls._pool
    
    @classmethod
    async def close_pool(cls):
        if cls._pool:
            await cls._pool.close()
            cls._pool = None
            DB_CONNECTIONS.set(0)
            logger.info("Database connection pool closed")

class BaseRepository:
    """Repository base com queries parametrizadas e cache"""
    
    def __init__(self, redis_client=None):
        self.redis = redis_client
        self.table_name = None
        
    async def _execute_query(self, query: str, *args) -> List[Dict]:
        """Executa query com parГўmetros preparados"""
        pool = await DatabaseConnection.get_pool()
        try:
            async with pool.acquire() as conn:
                # TODAS as queries usam parГўmetros preparados
                rows = await conn.fetch(query, *args)
                return [dict(row) for row in rows]
        except asyncpg.exceptions.TooManyConnectionsError:
            logger.error("Database connection limit exceeded")
            raise HTTPException(
                status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
                detail="Database temporarily unavailable"
            )
        except Exception as e:
            logger.error(f"Database query failed: {str(e)}")
            raise
    
    async def _get_cached(self, key: str) -> Optional[Any]:
        """ObtГ©m dados do cache Redis"""
        if not self.redis:
            return None
            
        try:
            data = await self.redis.get(key)
            if data:
                CACHE_HITS.labels(cache_type="redis").inc()
                return json.loads(data)
            CACHE_MISSES.labels(cache_type="redis").inc()
            return None
        except Exception as e:
            logger.warning(f"Cache read failed: {str(e)}")
            return None
    
    async def _set_cached(self, key: str, data: Any, ttl: int = 300):
        """Armazena dados no cache Redis"""
        if not self.redis:
            return
            
        try:
            await self.redis.setex(key, ttl, json.dumps(data))
        except Exception as e:
            logger.warning(f"Cache write failed: {str(e)}")

-- DIYAPP V12 - Esquemas Otimizados para Performance
-- EstratГ©gias: Partitioning, IndexaГ§ГЈo Inteligente, Materialized Views

-- ==================== TABELAS PRINCIPAIS COM PARTITIONING ====================

-- 1. EVENTOS DO USUГЃRIO (Partitioning por data)
CREATE TABLE user_events (
    event_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL,
    event_type VARCHAR(50) NOT NULL,
    event_data JSONB NOT NULL,
    session_id UUID,
    device_info JSONB,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    processed_at TIMESTAMPTZ,
    -- Campos otimizados para queries frequentes
    event_date DATE GENERATED ALWAYS AS (created_at::DATE) STORED,
    hour_of_day SMALLINT GENERATED ALWAYS AS (EXTRACT(HOUR FROM created_at)) STORED
) PARTITION BY RANGE (event_date);

-- PartiГ§Гµes mensais para eventos (retГ©m 24 meses)
CREATE TABLE user_events_2024_01 PARTITION OF user_events
    FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

CREATE TABLE user_events_2024_02 PARTITION OF user_events
    FOR VALUES FROM ('2024-02-01') TO ('2024-03-01');

-- ГЌndices otimizados para queries de eventos
CREATE INDEX idx_user_events_user_date ON user_events(user_id, event_date);
CREATE INDEX idx_user_events_type_date ON user_events(event_type, event_date);
CREATE INDEX idx_user_events_created_at ON user_events(created_at DESC);
CREATE INDEX idx_user_events_json_gin ON user_events USING GIN(event_data);

-- 2. MГ‰TRICAS DE PERFORMANCE LLM (Partitioning por modelo)
CREATE TABLE llm_metrics (
    metric_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    model_name VARCHAR(100) NOT NULL,
    provider VARCHAR(50) NOT NULL,
    operation_type VARCHAR(50) NOT NULL,
    input_tokens INTEGER NOT NULL,
    output_tokens INTEGER NOT NULL,
    total_tokens INTEGER GENERATED ALWAYS AS (input_tokens + output_tokens) STORED,
    latency_ms DECIMAL(10,2) NOT NULL,
    cost_usd DECIMAL(10,6) NOT NULL,
    success BOOLEAN NOT NULL,
    error_message TEXT,
    request_id UUID NOT NULL,
    user_id UUID,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    -- Campos calculados para agregaГ§Гµes rГЎpidas
    cost_per_token DECIMAL(12,8) GENERATED ALWAYS AS (
        CASE WHEN total_tokens > 0 THEN cost_usd / total_tokens ELSE 0 END
    ) STORED
) PARTITION BY LIST (model_name);

-- PartiГ§Гµes por modelo principal
CREATE TABLE llm_metrics_gpt4 PARTITION OF llm_metrics
    FOR VALUES IN ('gpt-4', 'gpt-4-turbo', 'gpt-4-32k');

CREATE TABLE llm_metrics_claude PARTITION OF llm_metrics
    FOR VALUES IN ('claude-3-opus', 'claude-3-sonnet', 'claude-3-haiku');

CREATE TABLE llm_metrics_gemini PARTITION OF llm_metrics
    FOR VALUES IN ('gemini-pro', 'gemini-ultra');

-- ГЌndices para queries de custo e performance
CREATE INDEX idx_llm_metrics_model_date ON llm_metrics(model_name, created_at);
CREATE INDEX idx_llm_metrics_cost ON llm_metrics(cost_usd DESC);
CREATE INDEX idx_llm_metrics_latency ON llm_metrics(latency_ms);
CREATE INDEX idx_llm_metrics_user_model ON llm_metrics(user_id, model_name) WHERE user_id IS NOT NULL;

-- 3. DADOS DE PRODUTO (EstratГ©gia de IndexaГ§ГЈo Composta)
CREATE TABLE product_metrics (
    metric_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    feature_name VARCHAR(100) NOT NULL,
    user_id UUID NOT NULL,
    session_count INTEGER NOT NULL DEFAULT 1,
    duration_seconds INTEGER NOT NULL,
    completion_rate DECIMAL(5,4),
    error_count INTEGER DEFAULT 0,
    date DATE NOT NULL,

## 2. Detector de Anomalias Preditivo

### `src/monitoring/predictive/anomaly_detector.py`
```python
import numpy as np
import pandas as pd
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass
from datetime import datetime, timedelta
from sklearn.ensemble import IsolationForest
from sklearn.preprocessing import StandardScaler
import joblib
import json
import asyncio
from ..logs.structured_logger import StructuredLogger

@dataclass
class Anomaly:
    timestamp: datetime
    feature: str
    metric: str
    value: float
    expected_range: Tuple[float, float]
    severity: str
    description: str
    model_used: str
    confidence: float

class PredictiveAnomalyDetector:
    def __init__(self, config: Dict):
        self.config = config
        self.logger = StructuredLogger("anomaly_detector")
        self.models = {}
        self.scalers = {}
        self.historical_data = {}
        self.initialize_models()
    
    def initialize_models(self):
        """Inicializa modelos de detecГ§ГЈo de anomalias para diferentes mГ©tricas"""
        metrics = ['cost', 'latency', 'tokens', 'error_rate', 'fallback_rate']
        
        for metric in metrics:
            self.models[metric] = IsolationForest(
                n_estimators=100,
                contamination=0.1,
                random_state=42
            )
            self.scalers[metric] = StandardScaler()
    
    async def train_models(self, historical_data: pd.DataFrame):
        """Treina modelos com dados histГіricos"""
        try:
            for metric in self.models.keys():
                if metric in historical_data.columns:
                    # Prepara dados para treinamento
                    metric_data = historical_data[metric].values.reshape(-1, 1)
                    
                    # Normaliza dados
                    scaled_data = self.scalers[metric].fit_transform(metric_data)
                    
                    # Treina modelo
                    self.models[metric].fit(scaled_data)
                    
                    # Armazena dados histГіricos para referГЄncia
                    self.historical_data[metric] = {
                        'mean': float(metric_data.mean()),
                        'std': float(metric_data.std()),
                        'percentiles': {
                            'p5': float(np.percentile(metric_data, 5)),
                            'p25': float(np.percentile(metric_data, 25)),
                            'p50': float(np.percentile(metric_data, 50)),
                            'p75': float(np.percentile(metric_data, 75)),
                            'p95': float(np.percentile(metric_data, 95))
                        }
                    }
            
            self.logger.info(
                "models_trained",
                message="Modelos de anomalia treinados com sucesso",
                metrics_trained=list(self.models.keys())
            )
            
        except Exception as e:
            self.logger.error(
                "model_training_failed",
                error=str(e),
                stack_trace=self.logger.format_exception(e)
            )
            raise
    
    async def detect_anomalies(self, current_data: Dict) -> List[Anomaly]:
        """Detecta anomalias em dados atuais"""
        anomalies = []
        
        for metric, value in current_data.items():
            if metric in self.models and value is not None:
                try:
                    # Prepara dado para prediГ§ГЈo
                    value_array = np.array([[value]])
                    scaled_value = self.scalers[metric].transform(value_array)
                    
                    # Prediz se Г© anomalia
                    prediction = self.models[metric].predict(scaled_value)
                    decision_score = self.models[metric].decision_function(scaled_value)
                    
                    # Calcula limites esperados
                    hist_data = self.historical_data.get(metric, {})
                    if hist_data:
                        expected_min = hist_data['percentiles']['p5']
                        expected_max = hist_data['percentiles']['p95']
                        
                        # Verifica se Г© anomalia
                        is_anomaly = (prediction[0] == -1 or 
                                     value < expected_min or 
                                     value > expected_max)
                        
                        if is_anomaly:
                            # Determina severidade
                            if abs(value - hist_data['mean']) > 3 * hist_data['std']:
                                severity = "critical"
                            elif abs(value - hist_data['mean']) > 2 * hist_data['std']:
                                severity = "high"
                            else:
                                severity = "medium"
                            
                            anomaly = Anomaly(
                                timestamp=datetime.now(),
                                feature=current_data.get('feature', 'unknown'),
                                metric=metric,
                                value=value,
                                expected_range=(expected_min, expected_max),
                                severity=severity,
                                description=f"Anomaly detected in {metric}: {value} outside expected range [{expected_min:.2f}, {expected_max:.2f}]",
                                model_used="IsolationForest",
                                confidence=abs(float(decision_score[0]))
                            )
                            
                            anomalies.append(anomaly)
                            
                            self.logger.warning(
                                "anomaly_detected",
                                metric=metric,
                                value=value,
                                expected_min=expected_min,
                                expected_max=expected_max,
                                severity=severity,
                                confidence=anomaly.confidence,
                                feature=current_data.get('feature')
                            )
                    
                except Exception as e:
                    self.logger.error(
                        "anomaly_detection_failed",
                        metric=metric,
                        error=str(e)
                    )
        
        return anomalies
    
    async def detect_pattern_anomalies(self, time_series: pd.DataFrame, window: int = 24) -> List[Dict]:
        """Detecta anomalias de padrГЈo em sГ©ries temporais"""
        anomalies = []
        
        try:
            for column in time_series.columns:
                if column in ['cost', 'latency', 'tokens']:
                    # Calcula mГ©dia mГіvel
                    rolling_mean = time_series[column].rolling(window=window).mean()
                    rolling_std = time_series[column].rolling(window=window).std()
                    
                    # Identifica pontos fora de 3 desvios padrГЈo
                    z_scores = (time_series[column] - rolling_mean) / rolling_std
                    anomaly_indices = np.where(np.abs(z_scores) > 3)[0]
                    
                    for idx in anomaly_indices:
                        if not np.isnan(z_scores[idx]):
                            anomaly = {
                                'timestamp': time_series.index[idx],
                                'metric': column,
                                'value': float(time_series[column].iloc[idx]),
                                'z_score': float(z_scores[idx]),
                                'rolling_mean': float(rolling_mean.iloc[idx]),
                                'rolling_std': float(rolling_std.iloc[idx]),
                                'type': 'pattern_anomaly'
                            }
                            anomalies.append(anomaly)
            
            self.logger.info(
                "pattern_anomalies_detected",
                count=len(anomalies),
                metrics_analyzed=list(time_series.columns)
            )
            
        except Exception as e:
            self.logger.error(
                "pattern_anomaly_detection_failed",
                error=str(e)
            )
        
        return anomalies
    
    async def save_model_state(self, path: str):
        """Salva estado dos modelos"""
        try:
            model_state = {
                'models': {k: joblib.dump(v, f'{path}/{k}_model.joblib') for k, v in self.models.items()},
                'scalers': {k: joblib.dump(v, f'{path}/{k}_scaler.joblib') for k, v in self.scalers.items()},
                'historical_data': self.historical_data,
                'timestamp': datetime.now().isoformat()
            }
            
            with open(f'{path}/model_state.json', 'w') as f:
                json.dump(model_state, f, default=str, indent=2)
            
            self.logger.info("model_state_saved", path=path)
            
        except Exception as e:
            self.logger.error("model_state_save_failed", error=str(e))
    
    async def load_model_state(self, path: str):
        """Carrega estado dos modelos"""
        try:
            with open(f'{path}/model_state.json', 'r') as f:
                model_state = json.load(f)
            
            for metric in self.models.keys():
                model_path = f'{path}/{metric}_model.joblib'
                scaler_path = f'{path}/{metric}_scaler.joblib'
                
                if os.path.exists(model_path):
                    self.models[metric] = joblib.load(model_path)
                if os.path.exists(scaler_path):
                    self.scalers[metric] = joblib.load(scaler_path)
            
            self.historical_data = model_state.get('historical_data', {})
            
            self.logger.info("model_state_loaded", path=path)
            
        except Exception as e:
            self.logger.error("model_state_load_failed", error=str(e))
