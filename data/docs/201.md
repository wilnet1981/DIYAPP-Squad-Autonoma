            user.status,

```sql
-- Materialized view for real-time event aggregation
CREATE MATERIALIZED VIEW events_hourly_mv TO events_hourly AS
SELECT
    toDate(created_at) AS event_date,
    toStartOfHour(created_at) AS event_hour,
    event_type,
    feature_id,
    uniqState(user_id) AS user_count,
    count() AS event_count,
    groupUniqArrayState(toString(properties)) AS properties_agg
FROM events
GROUP BY event_date, event_hour, event_type, feature_id;

-- Materialized view for LLM cost monitoring
CREATE MATERIALIZED VIEW llm_calls_hourly_mv TO llm_calls_hourly AS
SELECT
    toDate(created_at) AS call_date,
    toStartOfHour(created_at) AS call_hour,
    provider,
    model,
    feature_id,
    count() AS total_calls,
    sum(prompt_tokens) AS total_prompt_tokens,
    sum(completion_tokens) AS total_completion_tokens,
    sum(cost_usd) AS total_cost_usd,
    avg(latency_ms) AS avg_latency_ms,
    avg(if(success, 1, 0)) AS success_rate,
    uniqState(user_id) AS unique_users
FROM llm_calls
GROUP BY call_date, call_hour, provider, model, feature_id;
```

```javascript
const OAuth2Server = require('oauth2-server');
const model = require('./oauth-model');
const { oauth } = require('./config');

class OAuthServer {
  constructor() {
    this.server = new OAuth2Server({
      model,
      grants: ['authorization_code', 'refresh_token'],
      accessTokenLifetime: oauth.tokens.accessTokenLifetime,
      refreshTokenLifetime: oauth.tokens.refreshTokenLifetime,
      allowBearerTokensInQueryString: false,
      requireClientAuthentication: {
        authorization_code: true,
        refresh_token: true
      }
    });
  }

  async authenticate(request, response, options = {}) {
    try {
      const token = await this.server.authenticate(request, response, options);
      
      // Log authentication attempt
      this.logAuthAttempt(token.userId, 'success', request);
      
      return token;
    } catch (error) {
      // Log failed authentication attempt
      this.logAuthAttempt(null, 'failed', request, error.message);
      throw error;
    }
  }

  async authorize(request, response, options = {}) {
    try {
      const code = await this.server.authorize(request, response, options);
      
      // PKCE validation
      if (options.codeChallenge && options.codeChallengeMethod) {
        await this.validatePKCE(
          code.authorizationCode,
          options.codeChallenge,
          options.codeChallengeMethod
        );
      }
      
      return code;
    } catch (error) {
      console.error('Authorization error:', error);
      throw error;
    }
  }

  async token(request, response, options = {}) {
    try {
      const token = await this.server.token(request, response, options);
      
      // Add custom claims to JWT
      if (token.accessToken) {
        token.accessToken = await this.addCustomClaims(token.accessToken, token.user);
      }
      
      return token;
    } catch (error) {
      console.error('Token error:', error);
      throw error;
    }
  }

  async validatePKCE(authorizationCode, codeChallenge, method) {
    // Implementation of PKCE validation
    const codeVerifier = await model.getCodeVerifier(authorizationCode);
    
    if (!codeVerifier) {
      throw new OAuth2Server.InvalidGrantError('PKCE code verifier not found');
    }

    let calculatedChallenge;
    if (method === 'S256') {
      calculatedChallenge = crypto
        .createHash('sha256')
        .update(codeVerifier)
        .digest('base64')
        .replace(/\+/g, '-')
        .replace(/\//g, '_')
        .replace(/=/g, '');
    } else if (method === 'plain') {
      calculatedChallenge = codeVerifier;
    } else {
      throw new OAuth2Server.InvalidGrantError('Unsupported PKCE method');
    }

    if (calculatedChallenge !== codeChallenge) {
      throw new OAuth2Server.InvalidGrantError('PKCE validation failed');
    }
  }

  async addCustomClaims(token, user) {
    // Add custom claims to JWT
    const payload = this.decodeToken(token);
    
    payload.sub = user.id;
    payload.email = user.email;
    payload.name = user.name;
    payload.roles = user.roles;
    payload.permissions = await this.getUserPermissions(user.id);
    payload.tenant = user.tenantId;
    
    // Add security context
    payload.auth_time = Math.floor(Date.now() / 1000);
    payload.amr = ['pwd', 'mfa']; // Authentication Methods Reference
    
    return this.signToken(payload);
  }

  decodeToken(token) {
    // Implementation depends on your JWT library
    return JSON.parse(Buffer.from(token.split('.')[1], 'base64').toString());
  }

  signToken(payload) {
    // Implementation depends on your JWT library
    const header = {
      alg: 'RS256',
      typ: 'JWT'
    };
    
    const encodedHeader = Buffer.from(JSON.stringify(header)).toString('base64url');
    const encodedPayload = Buffer.from(JSON.stringify(payload)).toString('base64url');
    
    // In production, use proper signing
    return `${encodedHeader}.${encodedPayload}.signature`;
  }

  logAuthAttempt(userId, status, request, error = null) {
    const logEntry = {
      timestamp: new Date().toISOString(),
      userId,
      status,
      ip: request.ip,
      userAgent: request.get('User-Agent'),
      endpoint: request.originalUrl,
      error
    };
    
    // Send to audit log

  }
```

```python
@app.post("/api/v1/llm/completion")
async def llm_completion(request: Request):
    """
    Main LLM completion endpoint with monitoring and fallback
    """
    start_time = time.time()
    request_id = str(uuid.uuid4())
    
    try:
        # Parse request
        body = await request.json()
        feature = body.get("feature", "unknown")
        model = body.get("model")
        prompt = body.get("prompt")
        temperature = body.get("temperature", 0.7)
        max_tokens = body.get("max_tokens", 1000)
        
        # Validate request
        if not prompt:
            raise HTTPException(status_code=400, detail="Prompt is required")
        
        # Log request
        logger.info("LLM request received", 
                   request_id=request_id,
                   feature=feature,
                   model=model,
                   prompt_length=len(prompt))
        
        # Calculate input tokens (simplified - in production use tiktoken)
        input_tokens = len(prompt) // 4
        
        # Try primary provider
        try:
            provider_response = await llm_provider_manager.complete(
                model=model,
                prompt=prompt,
                temperature=temperature,
                max_tokens=max_tokens,
                feature=feature
            )
            
            output_tokens = len(provider_response["content"]) // 4
            used_fallback = False
            fallback_reason = None
            
        except Exception as primary_error:
            # Primary provider failed, try fallback
            logger.warning("Primary provider failed, attempting fallback",
                         request_id=request_id,
                         error=str(primary_error))
            
            fallback_result = await fallback_manager.execute_fallback(
                original_model=model,
                prompt=prompt,
                temperature=temperature,
                max_tokens=max_tokens,
                feature=feature,
                error=str(primary_error)
            )
            
            provider_response = fallback_result["response"]
            output_tokens = len(provider_response["content"]) // 4
            used_fallback = True
            fallback_reason = fallback_result["reason"]
        
        # Calculate cost
        cost = cost_calculator.calculate_cost(
            provider=provider_response["provider"],
            model=provider_response["model"],
            input_tokens=input_tokens,
            output_tokens=output_tokens
        )
        
        # Calculate latency
        latency_ms = (time.time() - start_time) * 1000
        
        # Collect metrics
        await metrics_collector.record_completion(
            request_id=request_id,
            feature=feature,
            provider=provider_response["provider"],
            model=provider_response["model"],
            input_tokens=input_tokens,
            output_tokens=output_tokens,
            latency_ms=latency_ms,
            cost=cost,
            used_fallback=used_fallback,
            fallback_reason=fallback_reason,
            success=True
        )
        
        # Check for anomalies
        await anomaly_detector.check_request(
            request_id=request_id,
            feature=feature,
            input_tokens=input_tokens,
            cost=cost,
            latency_ms=latency_ms
        )
        
        # Prepare response
        response_data = {
            "request_id": request_id,
            "content": provider_response["content"],
            "model": provider_response["model"],
            "provider": provider_response["provider"],
            "used_fallback": used_fallback,
            "fallback_reason": fallback_reason,
            "tokens": {
                "input": input_tokens,
                "output": output_tokens,
                "total": input_tokens + output_tokens
            },
            "cost": cost,
            "latency_ms": latency_ms
        }
        
        return JSONResponse(content=response_data)
        
    except Exception as e:
        # Record failure
        latency_ms = (time.time() - start_time) * 1000
        await metrics_collector.record_completion(
            request_id=request_id,
            feature=feature if 'feature' in locals() else "unknown",
            provider="unknown",
            model=model if 'model' in locals() else "unknown",
            input_tokens=0,
            output_tokens=0,
            latency_ms=latency_ms,
            cost=0,
            used_fallback=False,
            fallback_reason=None,
            success=False,
            error=str(e)
        )
        
        logger.error("LLM completion failed",
                    request_id=request_id,
                    error=str(e))
        
        raise HTTPException(status_code=500, detail=f"LLM completion failed: {str(e)}")

if __name__ == "__main__":
    uvicorn.run(
        "main:app",
        host="0.0.0.0",
        port=8000,
        reload=settings.DEBUG,
        log_level="info"
    )
```
