export const contrastUtils = {
  /**
   * Verifica se o contraste entre duas cores atende WCAG AA
   * @param {string} foreground - Cor do texto
   * @param {string} background - Cor do fundo
   * @returns {boolean} - True se atende WCAG AA
   */
  checkContrastAA: (foreground, background) => {
    // Implementação simplificada - em produção usar biblioteca como chroma.js
    const lum1 = getLuminance(foreground);
    const lum2 = getLuminance(background);
    const ratio = (Math.max(lum1, lum2) + 0.05) / (Math.min(lum1, lum2) + 0.05);
    return ratio >= 4.5;
  },

  /**
   * Verifica se o contraste atende WCAG AAA
   */
  checkContrastAAA: (foreground, background) => {
    const lum1 = getLuminance(foreground);
    const lum2 = getLuminance(background);
    const ratio = (Math.max(lum1, lum2) + 0.05) / (Math.min(lum1, lum2) + 0.05);
    return ratio >= 7;
  }
};

function getLuminance(hex) {
  const rgb = hexToRgb(hex);
  const [r, g, b] = Object.values(rgb).map(val => {
    val /= 255;
    return val <= 0.03928 ? val / 12.92 : Math.pow((val + 0.055) / 1.055, 2.4);
  });
  return 0.2126 * r + 0.7152 * g + 0.0722 * b;
}

function hexToRgb(hex) {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result ? {
    r: parseInt(result[1], 16),
    g: parseInt(result[2], 16),
    b: parseInt(result[3], 16)
  } : { r: 0, g: 0, b: 0 };
}

- **bff-admin**: Dashboard interno com permissões granulares

### 2. Responsabilidades do BFF:
- **Orquestração**: Agrega dados de múltiplos microsserviços
- **Transformação**: Formata dados específicos para cada cliente
- **Caching**: Cache em memória para dados frequentemente acessados
- **Adaptação**: Traduz erros de backend para respostas amigáveis
- **Autenticação**: Gerencia sessões específicas por cliente

### 3. Stack dos BFFs:
- **Framework**: Node.js + Fastify + TypeScript
- **API Gateway**: Kong (roteamento, rate limiting, autenticação)
- **Cache**: Redis (dados de sessão e cache)
- **Monitoramento**: Datadog/New Relic

## OPÇÕES CONSIDERADAS

### Opção A: API Gateway Único
- Prós: Simplicidade, um ponto de entrada
- Contras: Acoplamento entre clientes, difícil otimização específica

### Opção B: BFF por Cliente
- Prós: Otimização específica, desacoplamento, evolução independente
- Contras: Duplicação de código, mais serviços para operar

### Opção C: GraphQL como camada única
- Prós: Flexibilidade do cliente, evita over-fetching
- Contras: Complexidade, performance imprevisível, caching difícil

### Opção escolhida: B
**Justificativa**: Permite otimizações específicas por cliente enquanto mantém benefícios de microsserviços. Duplicação controlada é trade-off aceitável.

  toggleTheme: () => void;

export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [theme, setTheme] = useState<Theme>(() => {
    if (typeof window !== 'undefined') {
      const saved = localStorage.getItem('diyapp-theme');
      const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
      const prefersContrast = window.matchMedia('(prefers-contrast: more)').matches;
      
      if (saved && ['light', 'dark', 'high-contrast'].includes(saved)) {
        return saved as Theme;
      }
      
      if (prefersContrast) return 'high-contrast';
      if (prefersDark) return 'dark';
    }
    return 'light';
  });

  useEffect(() => {
    const root = document.documentElement;
    root.setAttribute('data-theme', theme);
    localStorage.setItem('diyapp-theme', theme);
    
    // Apply theme-specific styles
    const style = document.createElement('style');
    style.id = 'theme-styles';
    
    const highContrastStyles = theme === 'high-contrast' ? `
      * {
        border-color: ${tokens.colors.neutral[900]} !important;
      }
      button, a {
        outline: 3px solid ${tokens.colors.neutral[900]} !important;
      }
    ` : '';
    
    style.textContent = `
      :root[data-theme="dark"] {
        --color-bg: ${tokens.colors.neutral[900]};
        --color-text: ${tokens.colors.neutral[50]};
      }
      :root[data-theme="light"] {

      }

        --color-text: black;
      }

      if (current === 'light') return 'dark';

      return 'light';
    });
  };

import React, { useRef, useCallback, useEffect, useState, useMemo } from 'react';
import { tokens } from '../../../design-system/tokens';

interface VirtualListProps<T> {
  items: T[];
  itemHeight: number;
  renderItem: (item: T, index: number) => React.ReactNode;
  overscan?: number;
  className?: string;
  ariaLabel?: string;
  role?: string;
  onScroll?: (scrollTop: number) => void;
  loading?: boolean;
  loadingComponent?: React.ReactNode;
  emptyComponent?: React.ReactNode;
}

export function VirtualList<T>({
  items,
  itemHeight,
  renderItem,
  overscan = 3,
  className = '',
  ariaLabel = 'Lista virtual',
  role = 'list',
  onScroll,
  loading = false,
  loadingComponent,
  emptyComponent,
}: VirtualListProps<T>) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [scrollTop, setScrollTop] = useState(0);
  const [containerHeight, setContainerHeight] = useState(0);

  // Calculate visible items
  const { startIndex, endIndex, visibleItems } = useMemo(() => {
    if (!containerHeight || items.length === 0) {
      return { startIndex: 0, endIndex: 0, visibleItems: [] };
    }

    const startIndex = Math.max(0, Math.floor(scrollTop / itemHeight) - overscan);
    const visibleCount = Math.ceil(containerHeight / itemHeight) + 2 * overscan;
    const endIndex = Math.min(items.length, startIndex + visibleCount);

    return {
      startIndex,
      endIndex,
      visibleItems: items.slice(startIndex, endIndex),
    };
  }, [scrollTop, containerHeight, itemHeight, items, overscan]);

  // Handle scroll
  const handleScroll = useCallback((e: React.UIEvent<HTMLDivElement>) => {
    const newScrollTop = e.currentTarget.scrollTop;
    setScrollTop(newScrollTop);
    onScroll?.(newScrollTop);
  }, [onScroll]);

  // Resize observer
  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;

    const resizeObserver = new ResizeObserver((entries) => {
      for (const entry of entries) {
        setContainerHeight(entry.contentRect.height);
      }
    });

    resizeObserver.observe(container);
    return () => resizeObserver.disconnect();
  }, []);

  // Total height for scroll container
  const totalHeight = items.length * itemHeight;

  // Loading state
  if (loading) {
    return (
      <div className={className} style={{ minHeight: '200px' }}>
        {loadingComponent || (
          <div 
            role="status" 
            aria-label="Carregando lista"
            style={{
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              height: '100%',
              color: tokens.colors.neutral[500],
            }}
          >
            <span>Carregando...</span>
          </div>
        )}
      </div>
    );
  }

  // Empty state
  if (items.length === 0) {
    return (
      <div className={className} style={{ minHeight: '200px' }}>
        {emptyComponent || (
          <div 
            role="status" 
            aria-label="Lista vazia"
            style={{
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              height: '100%',
              color: tokens.colors.neutral[500],
            }}
          >
            <span>Nenhum item encontrado</span>
          </div>
        )}
      </div>
    );
  }

  return (
    <div
      ref={containerRef}
      className={className}
      style={{
        overflowY: 'auto',
        position: 'relative',
        height: '100%',
        scrollBehavior: 'smooth',
      }}
      onScroll={handleScroll}
      role={role}
      aria-label={ariaLabel}
      aria-busy={loading}
      aria-setsize={items.length}
      aria-posinset={startIndex + 1}
    >
      <div
        style={{
          height: totalHeight,
          position: 'relative',
        }}
        aria-hidden="true"
      >
        <div
          style={{
            position: 'absolute',
            top: startIndex * itemHeight,
            left: 0,
            right: 0,
          }}
        >
          {visibleItems.map((item, index) => {
            const absoluteIndex = startIndex + index;
            return (
              <div
                key={absoluteIndex}
                role={role === 'list' ? 'listitem' : undefined}
                aria-setsize={items.length}
                aria-posinset={absoluteIndex + 1}
                style={{
                  height: itemHeight,
                  position: 'absolute',
                  top: 0,
                  left: 0,
                  right: 0,
                  transform: `translateY(${index * itemHeight}px)`,
                }}
              >
                {renderItem(item, absoluteIndex)}
              </div>
            );
          })}
        </div>
      </div>
    </div>
  );
}

            raise
        except Exception as e:
            logger.error(f"Erro inesperado no Redis: {e}")
            raise
        finally:
            if conn:
                self._release_connection(conn)
    
    def _acquire_connection(self) -> redis.Redis:
        """Adquire uma conexão do pool"""
        # Health check periódico
        self._check_pool_health()
        
        if not self._available:
            if len(self._pool) < self.max_connections:
                conn = self._create_connection()
                if conn:
                    self._pool.append(conn)
                    self._available.append(conn)
                else:
                    raise redis.ConnectionError("Não foi possível criar nova conexão")
            else:
                raise redis.ConnectionError("Pool de conexões esgotado")
        
        conn = self._available.pop()
        self._in_use.add(conn)
        return conn
    
    def _release_connection(self, conn: redis.Redis):
        """Libera uma conexão de volta para o pool"""
        if conn in self._in_use:
            self._in_use.remove(conn)
            # Verificar se a conexão ainda está saudável
            try:
                conn.ping()
                self._available.append(conn)
            except:
                # Conexão morta, remover do pool
                if conn in self._pool:
                    self._pool.remove(conn)
                conn.close()
    
    def _check_pool_health(self):
        """Verifica a saúde do pool de conexões"""
        current_time = time.time()
        if current_time - self._last_health_check > self._health_check_interval:
            try:
                # Testar uma conexão disponível
                if self._available:
                    test_conn = self._available[0]
                    test_conn.ping()
                    self._is_healthy = True
                else:
                    # Criar conexão de teste
                    test_conn = self._create_connection()
                    if test_conn:
                        self._is_healthy = True
                        test_conn.close()
                    else:
                        self._is_healthy = False
                
                self._last_health_check = current_time
                logger.debug(f"Health check Redis: {'HEALTHY' if self._is_healthy else 'UNHEALTHY'}")
                
            except Exception as e:
                self._is_healthy = False
                logger.error(f"Health check Redis falhou: {e}")
    
    def get_health_status(self) -> Dict[str, Any]:
        """Retorna status de saúde do pool"""
        return {
            "healthy": self._is_healthy,
            "total_connections": len(self._pool),
            "available_connections": len(self._available),
            "in_use_connections": len(self._in_use),
            "last_health_check": self._last_health_check,
            "metrics": {
                "hits": self.metrics.hits,
                "misses": self.metrics.misses,
                "hit_ratio": self.metrics.hits / (self.metrics.hits + self.metrics.misses) 
                    if (self.metrics.hits + self.metrics.misses) > 0 else 0
            }
        }
    
    def close_all(self):
        """Fecha todas as conexões"""
        for conn in self._pool:
            try:
                conn.close()
            except:
                pass
        self._pool.clear()
        self._available.clear()
        self._in_use.clear()

class CacheManager:
    """Gerenciador de cache com estratégias avançadas e circuit breaker"""
    
    def __init__(self, connection_pool: RedisConnectionPool, 
                 default_ttl: int = 300,
                 circuit_breaker_threshold: int = 5,
                 circuit_breaker_timeout: int = 30):
        self.pool = connection_pool
        self.default_ttl = default_ttl
        self.circuit_breaker_threshold = circuit_breaker_threshold
        self.circuit_breaker_timeout = circuit_breaker_timeout
        self.circuit_open = False
        self.circuit_open_since = 0
        self.failure_count = 0
        
        # Prefixos para organização
        self.key_prefix = "diyapp:v12:"
        
    def _get_full_key(self, key: str) -> str:
        """Adiciona prefixo à chave"""
        return f"{self.key_prefix}{key}"
    
    def _circuit_breaker_check(self):
        """Verifica se o circuit breaker está aberto"""
        if self.circuit_open:
            current_time = time.time()
            if current_time - self.circuit_open_since > self.circuit_breaker_timeout:
                # Tentar meio-abrir o circuito
                self.circuit_open = False
                self.failure_count = 0
                logger.info("Circuit breaker: tentando meio-abrir")
            else:
                raise redis.ConnectionError("Circuit breaker aberto - cache indisponível")
    
    def _record_failure(self):
        """Registra falha e gerencia circuit breaker"""
        self.failure_count += 1
        if self.failure_count >= self.circuit_breaker_threshold:
            self.circuit_open = True
            self.circuit_open_since = time.time()
            logger.warning(f"Circuit breaker ABERTO após {self.failure_count} falhas")
    
    def _record_success(self):
        """Registra sucesso e reseta contador de falhas"""
        self.failure_count = 0
        if self.circuit_open:
            self.circuit_open = False
            logger.info("Circuit breaker FECHADO - operação bem-sucedida")
    
    def get(self, key: str, default: Any = None) -> Any:
        """Obtém valor do cache com fallback"""
        start_time = time.time()
        
        try:
            self._circuit_breaker_check()
            
            with self.pool.get_connection() as conn:
                full_key = self._get_full_key(key)
                value = conn.get(full_key)
                
                if value is not None:
                    self.pool.metrics.hits += 1
                    try:
                        # Tentar deserializar JSON primeiro
                        return json.loads(value)
                    except:
                        # Fallback para pickle
                        return pickle.loads(value)
                else:
                    self.pool.metrics.misses += 1
                    return default
                    
        except redis.ConnectionError as e:
            self._record_failure()
            logger.warning(f"Cache miss (circuit breaker): {key} - {e}")
            return default
        except Exception as e:
            logger.error(f"Erro ao obter do cache: {key} - {e}")
            return default
        finally:
            duration = (time.time() - start_time) * 1000
            self.pool.metrics.avg_read_time_ms = (
                self.pool.metrics.avg_read_time_ms * 0.9 + duration * 0.1
            )
    
    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Define valor no cache"""
        start_time = time.time()
        
        try:
            self._circuit_breaker_check()
            
            with self.pool.get_connection() as conn:
                full_key = self._get_full_key(key)
                
                # Tentar serializar como JSON primeiro
                try:
                    serialized = json.dumps(value)
                    conn.setex(full_key, ttl or self.default_ttl, serialized)
                except (TypeError, OverflowError):
                    # Fallback para pickle para objetos complexos
                    serialized = pickle.dumps(value)
                    conn.setex(full_key, ttl or self.default_ttl, serialized)
                
                self.pool.metrics.write_operations += 1
                self._record_success()
                return True
                
        except redis.ConnectionError as e:
            self._record_failure()
            logger.warning(f"Cache write failed (circuit breaker): {key} - {e}")
            return False
        except Exception as e:
            logger.error(f"Erro ao definir cache: {key} - {e}")
            return False
        finally:
            duration = (time.time() - start_time) * 1000
            self.pool.metrics.avg_write_time_ms = (
                self.pool.metrics.avg_write_time_ms * 0.9 + duration * 0.1
            )
    
    def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        try:
            self._circuit_breaker_check()
            
            with self.pool.get_connection() as conn:
                full_key = self._get_full_key(key)
                result = conn.delete(full_key)
                return result > 0
                
        except redis.ConnectionError as e:
            self._record_failure()
            logger.warning(f"Cache delete failed: {key} - {e}")
            return False
        except Exception as e:
            logger.error(f"Erro ao deletar cache: {key} - {e}")
            return False
    
    def get_or_set(self, key: str, factory: callable, ttl: Optional[int] = None) -> Any:
        """Obtém do cache ou executa factory e cacheia resultado"""
        cached = self.get(key)
        if cached is not None:
            return cached
        
        value = factory()
        self.set(key, value, ttl)
        return value
    
    def clear_pattern(self, pattern: str) -> int:
        """Limpa todas as chaves que correspondem ao padrão"""
        try:
            self._circuit_breaker_check()
            
            with self.pool.get_connection() as conn:
                full_pattern = self._get_full_key(pattern)
                keys = conn.keys(full_pattern)
                
                if keys:
                    deleted = conn.delete(*keys)
                    self.pool.metrics.evictions += deleted
                    return deleted
                return 0
                
        except redis.ConnectionError as e:
            self._record_failure()
            logger.warning(f"Clear pattern failed: {pattern} - {e}")
            return 0
        except Exception as e:
            logger.error(f"Erro ao limpar padrão: {pattern} - {e}")
            return 0
    
    def get_stats(self) -> Dict[str, Any]:
        """Retorna estatísticas do cache"""
        stats = self.pool.get_health_status()
        stats.update({
            "circuit_breaker": {
                "open": self.circuit_open,
                "failure_count": self.failure_count,
                "open_since": self.circuit_open_since if self.circuit_open else None
            },
            "performance": {
                "avg_read_time_ms": round(self.pool.metrics.avg_read_time_ms, 2),
                "avg_write_time_ms": round(self.pool.metrics.avg_write_time_ms, 2)
            }
        })
        return stats

from functools import wraps

class APIOptimizer:
    """Otimizador de APIs com cache inteligente"""
    
    def __init__(self, cache_manager: CacheManager):
        self.cache = cache_manager
        self.endpoint_stats = {}
    
    def cached_endpoint(self, ttl: int = 60, 
                       key_prefix: str = "api",
                       vary_by_params: List[str] = None,
                       vary_by_headers: List[str] = None,
                       bypass_cache: Callable = None):
        """
        Decorador para cache de endpoints
        
        Args:
            ttl: Tempo de vida do cache em segundos
            key_prefix: Prefixo para chave de cache
            vary_by_params: Lista de parâmetros que diferenciam o cache
            vary_by_headers: Lista de headers que diferenciam o cache
            bypass_cache: Função que retorna True para bypass do cache
        """
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                # Verificar se deve bypassar cache
                if bypass_cache and bypass_cache(*args, **kwargs):
                    return func(*args, **kwargs)
                
                # Construir chave de cache
                cache_key = self._build_cache_key(
                    func, args, kwargs, key_prefix, 
                    vary_by_params, vary_by_headers
                )
                
                # Tentar obter do cache
                cached_response = self.cache.get(cache_key)
                if cached_response is not None:
                    # Adicionar header indicando cache hit
                    if isinstance(cached_response, dict):
                        cached_response['_cache'] = {
                            'hit': True,
                            'cached_at': time.time(),
                            'ttl': ttl
                        }
                    return cached_response
                
                # Executar função original
                start_time = time.time()
                response = func(*args, **kwargs)
                execution_time = time.time() - start_time
                
                # Cachear resposta
                if response is not None:
                    self.cache.set(cache_key, response, ttl)
                
                # Adicionar informações de cache
                if isinstance(response, dict):
                    response['_cache'] = {
                        'hit': False,
                        'execution_time_ms': round(execution_time * 1000, 2),
                        'cached_until': time.time() + ttl
                    }
                
                # Atualizar estatísticas
                self._update_endpoint_stats(func.__name__, execution_time, False)
                
                return response
            
            return wrapper
        return decorator
    
    def _build_cache_key(self, func, args, kwargs, key_prefix,
                        vary_by_params, vary_by_headers) -> str:
        """Constrói chave única para cache baseada nos parâmetros"""
        key_parts = [key_prefix, func.__module__, func.__name__]
        
        # Adicionar parâmetros relevantes
        if vary_by_params:
            for param in vary_by_params:
                if param in kwargs:
                    key_parts.append(f"{param}:{kwargs[param]}")
        
        # Adicionar headers relevantes (se disponível no request)
        if vary_by_headers and len(args) > 0:
            request = args[0]
            if hasattr(request, 'headers'):
                for header in vary_by_headers:
                    if header in request.headers:
                        key_parts.append(f"{header}:{request.headers[header]}")
        
        # Hash para chave compacta
        key_string = ":".join(str(part) for part in key_parts)
        return hashlib.md5(key_string.encode()).hexdigest()
    
    def _update_endpoint_stats(self, endpoint_name: str, 
                              execution_time: float, cache_hit: bool):
        """Atualiza estatísticas do endpoint"""
        if endpoint_name not in self.endpoint_stats:
            self.endpoint_stats[endpoint_name] = {
                'total_requests': 0,
                'cache_hits': 0,
                'cache_misses': 0,
                'avg_response_time': 0,
                'p95_response_time': [],
                'last_updated': time.time()
            }
        
        stats = self.endpoint_stats[endpoint_name]
        stats['total_requests'] += 1
        
        if cache_hit:
            stats['cache_hits'] += 1
        else:
            stats['cache_misses']
