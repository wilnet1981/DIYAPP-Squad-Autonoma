```typescript
// api-gateway/src/index.ts
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import rateLimit from 'express-rate-limit';
import { createProxyMiddleware } from 'http-proxy-middleware';
import { logger } from './utils/logger';
import { errorHandler } from './middleware/errorHandler';
import { requestLogger } from './middleware/requestLogger';
import { authMiddleware } from './middleware/auth';
import { metricsMiddleware } from './middleware/metrics';

const app = express();
const PORT = process.env.PORT || 8080;

// Security middleware
app.use(helmet());
app.use(cors({
  origin: process.env.CORS_ORIGIN || 'http://localhost:3000',
  credentials: true
}));

// Rate limiting
const limiter = rateLimit({
  windowMs: parseInt(process.env.RATE_LIMIT_WINDOW || '900000'),
  max: parseInt(process.env.RATE_LIMIT_MAX || '100'),
  message: 'Too many requests from this IP'
});
app.use('/api/', limiter);

// Metrics middleware
app.use(metricsMiddleware);

// Request logging
app.use(requestLogger);

// Health check endpoint
app.get('/health', (req, res) => {
  res.status(200).json({
    status: 'healthy',
    timestamp: new Date().toISOString(),
    uptime: process.uptime()
  });
});

// Service routing with authentication
app.use('/api/auth', createProxyMiddleware({
  target: 'http://auth-service:3001',
  changeOrigin: true,
  pathRewrite: { '^/api/auth': '' }
}));

app.use('/api/users', authMiddleware, createProxyMiddleware({
  target: 'http://user-service:3002',
  changeOrigin: true,
  pathRewrite: { '^/api/users': '' }
}));

app.use('/api/tasks', authMiddleware, createProxyMiddleware({
  target: 'http://task-service:3003',
  changeOrigin: true,
  pathRewrite: { '^/api/tasks': '' }
}));

app.use('/api/notifications', authMiddleware, createProxyMiddleware({
  target: 'http://notification-service:3004',
  changeOrigin: true,
  pathRewrite: { '^/api/notifications': '' }
}));

// Error handling
app.use(errorHandler);

app.listen(PORT, () => {
  logger.info(`API Gateway running on port ${PORT}`);
});
```

```typescript
// services/task/src/middleware/metrics.ts
import client from 'prom-client';
import { Request, Response, NextFunction } from 'express';

// Create Registry
const register = new client.Registry();

// Enable default metrics
client.collectDefaultMetrics({ register });

// Custom metrics
const httpRequestDurationMicroseconds = new client.Histogram({
  name: 'http_request_duration_seconds',
  help: 'Duration of HTTP requests in seconds',
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.1, 0.5, 1, 2, 5]
});

const httpRequestsTotal = new client.Counter({
  name: 'http_requests_total',
  help: 'Total number of HTTP requests',
  labelNames: ['method', 'route', 'status_code']
});

register.registerMetric(httpRequestDurationMicroseconds);
register.registerMetric(httpRequestsTotal);

export const metricsMiddleware = (req: Request, res: Response, next: NextFunction) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = Date.now() - start;
    const route = req.route?.path || req.path;
    
    httpRequestDurationMicroseconds
      .labels(req.method, route, res.statusCode.toString())
      .observe(duration / 1000);
    
    httpRequestsTotal
      .labels(req.method, route, res.statusCode.toString())
      .inc();
  });
  
  next();
};

export const metricsEndpoint = async (req: Request, res: Response) => {
  res.set('Content-Type', register.contentType);
  res.end(await register.metrics());
};

// P95 Alert Rule
export const checkP95Latency = async (): Promise<boolean> => {
  const metrics = await register.getMetricsAsJSON();
  const histogram = metrics.find(m => m.name === 'http_request_duration_seconds');
  
  if (histogram && histogram.values) {
    const p95 = histogram.values.find(v => v.quantile === 0.95);
    return p95 && p95.value < 0.2; // P95 < 200ms
  }
  
  return true;
};
```

```javascript
const express = require('express');
const helmet = require('helmet');
const cors = require('cors');
const rateLimit = require('express-rate-limit');
const { createTerminus } = require('@godaddy/terminus');
const { ApolloServer } = require('apollo-server-express');
const swaggerUi = require('swagger-ui-express');

const config = require('./config');
const logger = require('./shared/utils/logger');
const { errorHandler } = require('./shared/errors/errorHandler');
const { requestLogger, correlationId } = require('./api/rest/middlewares/logging');
const { authMiddleware } = require('./infrastructure/auth/middleware');
const restRoutes = require('./api/rest/routes');
const graphqlSchema = require('./api/graphql/schema');
const { connectDatabase, disconnectDatabase } = require('./data/database');
const { connectRedis, disconnectRedis } = require('./infrastructure/cache/redis');
const { connectQueue, disconnectQueue } = require('./infrastructure/queue/bull');
const openApiSpec = require('./docs/api/openapi');

class Server {
  constructor() {
    this.app = express();
    this.apolloServer = null;
    this.setupMiddleware();
    this.setupRoutes();
    this.setupErrorHandling();
  }

  setupMiddleware() {
    // Security middleware
    this.app.use(helmet({
      contentSecurityPolicy: {
        directives: {
          defaultSrc: ["'self'"],
          styleSrc: ["'self'", "'unsafe-inline'"],
          scriptSrc: ["'self'"],
          imgSrc: ["'self'", "data:", "https:"],
        },
      },
    }));

    // CORS configuration
    this.app.use(cors({
      origin: config.cors.origins,
      credentials: true,
      methods: ['GET', 'POST', 'PUT', 'DELETE', 'PATCH', 'OPTIONS'],
      allowedHeaders: ['Content-Type', 'Authorization', 'X-Requested-With'],
    }));

    // Rate limiting
    const limiter = rateLimit({
      windowMs: config.rateLimit.windowMs,
      max: config.rateLimit.max,
      message: {
        error: 'Too many requests',
        message: 'Please try again later.',
      },
      standardHeaders: true,
      legacyHeaders: false,
    });

    this.app.use('/api/', limiter);

    // Body parsing
    this.app.use(express.json({ limit: '10mb' }));
    this.app.use(express.urlencoded({ extended: true, limit: '10mb' }));

    // Logging and correlation ID
    this.app.use(correlationId);
    this.app.use(requestLogger);

    // Health check endpoint
    this.app.get('/health', (req, res) => {
      res.status(200).json({
        status: 'healthy',
        timestamp: new Date().toISOString(),
        uptime: process.uptime(),
        version: config.version,
      });
    });

    // API Documentation
    this.app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(openApiSpec));
  }

  setupRoutes() {
    // Public routes
    this.app.use('/api/v1/auth', require('./api/rest/routes/auth'));

    // Protected routes
    this.app.use('/api/v1', authMiddleware, restRoutes);
  }

  async setupGraphQL() {
    this.apolloServer = new ApolloServer({
      schema: graphqlSchema,
      context: ({ req }) => ({
        user: req.user,
        correlationId: req.correlationId,
      }),
      formatError: (error) => {
        logger.error('GraphQL Error', {
          correlationId: error.extensions?.correlationId,
          error: error.message,
          path: error.path,
          stack: error.stack,
        });

        // Don't expose internal errors to client
        const isInternalError = error.extensions?.code === 'INTERNAL_SERVER_ERROR';
        return {
          message: isInternalError ? 'Internal server error' : error.message,
          code: error.extensions?.code,
          path: error.path,
        };
      },
      plugins: [{
        requestDidStart: () => ({
          willSendResponse(requestContext) {
            logger.info('GraphQL Request Completed', {
              correlationId: requestContext.context.correlationId,
              duration: Date.now() - requestContext.request.queryStartTime,
              operationName: requestContext.request.operationName,
            });
          },
        }),
      }],
    });

    await this.apolloServer.start();
    this.apolloServer.applyMiddleware({
      app: this.app,
      path: '/api/graphql',
      cors: false,
    });
  }

  setupErrorHandling() {
    // 404 handler
    this.app.use((req, res, next) => {
      const error = new Error(`Route not found: ${req.method} ${req.url}`);
      error.status = 404;
      next(error);
    });

    // Global error handler
    this.app.use(errorHandler);
  }

  async start() {
    try {
      // Connect to dependencies
      await connectDatabase();
      await connectRedis();
      await connectQueue();

      // Setup GraphQL
      await this.setupGraphQL();

      // Setup graceful shutdown
      const onSignal = async () => {
        logger.info('Starting graceful shutdown...');
        await disconnectDatabase();
        await disconnectRedis();
        await disconnectQueue();
        if (this.apolloServer) {
          await this.apolloServer.stop();
        }
      };

      const onShutdown = async () => {
        logger.info('Cleanup finished, server is shutting down');
      };

      const healthChecks = {
        '/health/ready': async () => {
          // Check database connection
          const dbHealthy = await checkDatabaseHealth();
          const redisHealthy = await checkRedisHealth();
          
          if (!dbHealthy || !redisHealthy) {
            throw new Error('Service dependencies unhealthy');
          }
          
          return {
            status: 'healthy',
            database: dbHealthy ? 'connected' : 'disconnected',
            redis: redisHealthy ? 'connected' : 'disconnected',
          };
        },
      };

      createTerminus(this.app, {
        signal: 'SIGINT',
        healthChecks,
        onSignal,
        onShutdown,
        logger: (msg, err) => logger.error(msg, { error: err }),
      });

      // Start server
      const server = this.app.listen(config.port, () => {
        logger.info(`Server running on port ${config.port}`, {
          environment: config.env,
          version: config.version,
          graphqlPath: '/api/graphql',
          docsPath: '/api-docs',
        });
      });

      return server;
    } catch (error) {
      logger.error('Failed to start server', { error: error.message, stack: error.stack });
      process.exit(1);
    }
  }
}
```

```typescript
import React from 'react';
import clsx from 'clsx';
import './Button.css';

export type ButtonVariant = 'primary' | 'secondary' | 'outline' | 'ghost' | 'danger';
export type ButtonSize = 'sm' | 'md' | 'lg';

export interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: ButtonVariant;
  size?: ButtonSize;
  loading?: boolean;
  fullWidth?: boolean;
  leftIcon?: React.ReactNode;
  rightIcon?: React.ReactNode;
  children: React.ReactNode;
}

const Button = React.forwardRef<HTMLButtonElement, ButtonProps>(
  (
    {
      variant = 'primary',
      size = 'md',
      loading = false,
      fullWidth = false,
      leftIcon,
      rightIcon,
      children,
      className,
      disabled,
      ...props
    },
    ref
  ) => {
    const isDisabled = disabled || loading;

    return (
      <button
        ref={ref}
        className={clsx(
          'button',
          `button--${variant}`,
          `button--${size}`,
          {
            'button--loading': loading,
            'button--full-width': fullWidth,
            'button--disabled': isDisabled,
          },
          className
        )}
        disabled={isDisabled}
        aria-busy={loading}
        {...props}
      >
        {loading && (
          <span className="button__loader" aria-hidden="true">
            <div className="button__loader-dot" />
            <div className="button__loader-dot" />
            <div className="button__loader-dot" />
          </span>
        )}
        
        {!loading && leftIcon && (
          <span className="button__icon button__icon--left">{leftIcon}</span>
        )}
        
        <span className="button__content">{children}</span>
        
        {!loading && rightIcon && (
          <span className="button__icon button__icon--right">{rightIcon}</span>
        )}
      </button>
    );
  }
);

Button.displayName = 'Button';

export default Button;
```

```sql
-- Apply triggers to all tables with updated_at
DO $$ 
DECLARE
    t text;
BEGIN
    FOR t IN 
        SELECT table_name 
        FROM information_schema.columns 
        WHERE column_name = 'updated_at' 
        AND table_schema = 'public'
    LOOP
        EXECUTE format('
            DROP TRIGGER IF EXISTS update_%s_updated_at ON %s;
            CREATE TRIGGER update_%s_updated_at
            BEFORE UPDATE ON %s
            FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
        ', t, t, t, t);
    END LOOP;
END;
$$;

-- Function to create monthly partitions
CREATE OR REPLACE FUNCTION create_monthly_partition(
    parent_table TEXT,
    year INTEGER,
    month INTEGER
) RETURNS VOID AS $$
DECLARE
    partition_name TEXT;
    start_date DATE;
    end_date DATE;
BEGIN
    partition_name := format('%s_%s_%s', parent_table, year, LPAD(month::TEXT, 2, '0'));
    start_date := format('%s-%s-01', year, LPAD(month::TEXT, 2, '0'))::DATE;
    end_date := start_date + INTERVAL '1 month';
    
    EXECUTE format('
        CREATE TABLE IF NOT EXISTS %I PARTITION OF %I
        FOR VALUES FROM (%L) TO (%L);
    ', partition_name, parent_table, start_date, end_date);
END;
$$ LANGUAGE plpgsql;
```

# DIYAPP Evolution - V12 Core - Segurança

## 1. Threat Modeling (STRIDE) - Documento

```markdown
# THREAT MODELING - DIYAPP V12 CORE

## Data: 2024
## Sistema: DIYAPP Evolution V12
## Analista: Especialista em Segurança

## 1. ATIVOS CRÍTICOS
- Dados de usuários (PII, credenciais, histórico)
- Chaves de API e secrets
- Sistema de pagamento/integrações financeiras
- LLM prompts e configurações
- Infraestrutura (servidores, banco de dados)

## 2. ANÁLISE STRIDE POR COMPONENTE

### 2.1 Autenticação/Login
**Spoofing**: Ataque de força bruta em login
- Controle: Rate limiting (10 tentativas/hora), MFA obrigatório
- Risco: Médio

**Tampering**: Manipulação de tokens JWT
- Controle: Tokens assinados com algoritmo forte (RS256), expiração curta (15min)
- Risco: Baixo

**Repudiation**: Usuário nega login bem-sucedido
- Controle: Logs de auditoria imutáveis com IP, timestamp, user agent
- Risco: Baixo

### 2.2 API REST
**Information Disclosure**: Exposição de dados sensíveis em responses
- Controle: Mascaramento de PII, validação de autorização por recurso
- Risco: Alto

**Denial of Service**: Ataque DDoS nas APIs
- Controle: WAF com rate limiting, circuit breakers, auto-scaling
- Risco: Médio

**Elevation of Privilege**: Acesso não autorizado a endpoints admin
- Controle: RBAC granular, verificação de permissão em cada endpoint
- Risco: Alto

### 2.3 Banco de Dados
**Tampering**: SQL Injection
- Controle: Prepared statements, ORM com sanitização
- Risco: Crítico

**Information Disclosure**: Acesso direto ao DB
- Controle: Criptografia em repouso (AES-256), IAM restritivo
- Risco: Alto

### 2.4 LLM/IA
**Spoofing**: Prompt injection
- Controle: Separar system prompt de user input, sanitização de output
- Risco: Alto

**Information Disclosure**: Vazamento de dados via contexto do LLM
- Controle: Não incluir PII no contexto, sanitizar outputs
- Risco: Médio

## 3. RISCOS ACEITOS CONSCIENTEMENTE
1. Rate limiting pode bloquear usuários legítimos em caso de ataque DDoS distribuído
```

```javascript
const rateLimit = require('express-rate-limit');
const helmet = require('helmet');
const { body, validationResult } = require('express-validator');
const mongoSanitize = require('express-mongo-sanitize');
const xss = require('xss-clean');
const hpp = require('hpp');

class WAF {
  constructor() {
    this.rules = {
      sqlInjectionPatterns: [
        /(\%27)|(\')|(\-\-)|(\%23)|(#)/gi,
        /((\%3D)|(=))[^\n]*((\%27)|(\')|(\-\-)|(\%3B)|(;))/gi,
        /\w*((\%27)|(\'))((\%6F)|o|(\%4F))((\%72)|r|(\%52))/gi,
        /((\%27)|(\'))union/gi
      ],
      xssPatterns: [
        /<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi,
        /javascript:/gi,
        /on\w+\s*=/gi
      ],
      pathTraversalPatterns: [
        /\.\.\//gi,
        /\.\.\\/gi
      ]
    };
  }

  initialize(app) {
    // 1. Headers de segurança
    app.use(helmet({
      contentSecurityPolicy: {
        directives: {
          defaultSrc: ["'self'"],
          styleSrc: ["'self'", "'unsafe-inline'"],
          scriptSrc: ["'self'"],
          imgSrc: ["'self'", "data:", "https:"],
          connectSrc: ["'self'"],
          fontSrc: ["'self'"],
          objectSrc: ["'none'"],
          mediaSrc: ["'self'"],
          frameSrc: ["'none'"]
        }
      },
      hsts: {
        maxAge: 31536000,
        includeSubDomains: true,
        preload: true
      }
    }));

    // 2. Rate limiting por tipo de endpoint
    const apiLimiter = rateLimit({
      windowMs: 15 * 60 * 1000, // 15 minutos
      max: 100, // 100 requests por IP
      message: 'Too many requests from this IP, please try again later.',
      standardHeaders: true,
      legacyHeaders: false
    });

    const authLimiter = rateLimit({
      windowMs: 60 * 60 * 1000, // 1 hora
      max: 10, // 10 tentativas de login por hora
      message: 'Too many login attempts, please try again later.'
    });

    // 3. Sanitização
    app.use(mongoSanitize());
    app.use(xss());
    app.use(hpp());

    // 4. Middleware de validação customizado
    app.use(this.validateRequest.bind(this));

    // Aplicar rate limiting
    app.use('/api/', apiLimiter);
    app.use('/auth/', authLimiter);

    return app;
  }

  validateRequest(req, res, next) {
    // Verificar padrões maliciosos na URL
    const url = req.originalUrl;
    if (this.detectMaliciousPatterns(url)) {
      this.logAttackAttempt(req, 'URL_MALICIOUS_PATTERN');
      return res.status(400).json({ error: 'Invalid request' });
    }

    // Verificar padrões maliciosos nos headers
    for (const [header, value] of Object.entries(req.headers)) {
      if (this.detectMaliciousPatterns(value)) {
        this.logAttackAttempt(req, 'HEADER_MALICIOUS_PATTERN', header);
        return res.status(400).json({ error: 'Invalid request headers' });
      }
    }

    // Verificar tamanho do body (prevenção de DoS)
    const contentLength = parseInt(req.headers['content-length'] || '0');
    if (contentLength > 10 * 1024 * 1024) { // 10MB max
      return res.status(413).json({ error: 'Request too large' });
    }

    next();
  }

  detectMaliciousPatterns(input) {
    if (!input || typeof input !== 'string') return false;

    const patterns = [
      ...this.rules.sqlInjectionPatterns,
      ...this.rules.xssPatterns,
      ...this.rules.pathTraversalPatterns
    ];

    for (const pattern of patterns) {
      if (pattern.test(input)) {
        return true;
      }
    }

    return false;
  }

  logAttackAttempt(req, attackType, details = '') {
    const logEntry = {
      timestamp: new Date().toISOString(),
      ip: req.ip,
      method: req.method,
      url: req.originalUrl,
      attackType,
      details,
      userAgent: req.headers['user-agent'],
      referer: req.headers['referer']
    };

    // Log para sistema de SIEM
    console.warn(`[WAF] Attack detected: ${JSON.stringify(logEntry)}`);
    
    // Aqui você integraria com seu sistema de logging (ELK, Splunk, etc.)
    this.sendToSIEM(logEntry);
  }

  sendToSIEM(logEntry) {
    // Implementação de integração com SIEM
    // Exemplo: enviar para Elasticsearch
    fetch(process.env.SIEM_ENDPOINT, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(logEntry)
    }).catch(console.error);
  }
}

module.exports = new WAF();
```

```javascript
      abortEarly: false,
      stripUnknown: true
    });

        } else {
          safeQuery[key] = value;
        }
      }
      
      return safeQuery;
    }
```

```javascript
/**
 * Registro de Atividades de Processamento (ROPA) - LGPD Art. 37
 * Data Processing Activity Register - GDPR Article 30
 */
class DataProcessingActivity {
    constructor() {
        this.activities = new Map();
        this.nextId = 1;
    }

    /**
     * Registra nova atividade de processamento
     * @param {Object} activityData - Dados da atividade
     */
    registerActivity(activityData) {
        const activity = {
            id: this.nextId++,
            timestamp: new Date().toISOString(),
            name: activityData.name,
            description: activityData.description,
            purpose: activityData.purpose,
            legalBasis: this.validateLegalBasis(activityData.legalBasis),
            dataCategories: activityData.dataCategories || [],
            dataSubjects: activityData.dataSubjects || [],
            retentionPeriod: activityData.retentionPeriod,
            securityMeasures: activityData.securityMeasures || [],
            processors: activityData.processors || [],
            internationalTransfers: activityData.internationalTransfers || [],
            dpiaRequired: activityData.dpiaRequired || false,
            dpiaId: activityData.dpiaId,
            status: 'active',
            lastReview: new Date().toISOString(),
            nextReview: this.calculateNextReviewDate(activityData.reviewFrequency)
        };

        this.activities.set(activity.id, activity);
        this.logActivity(activity, 'REGISTERED');
        return activity;
    }

    /**
     * Valida base legal conforme LGPD Art. 7 / GDPR Art. 6
     */
    validateLegalBasis(basis) {
        const validBases = [
            'consent',           // Consentimento
            'contract',          // Contrato
            'legal_obligation',  // Obrigação legal
            'vital_interest',    // Interesse vital
            'public_interest',   // Interesse público
            'legitimate_interest' // Legítimo interesse
        ];

        if (!validBases.includes(basis)) {
            throw new Error(`Base legal inválida: ${basis}. Bases válidas: ${validBases.join(', ')}`);
        }

        return basis;
    }

    /**
     * Calcula data da próxima revisão
     */
    calculateNextReviewDate(frequency) {
        const date = new Date();
        switch(frequency) {
            case 'monthly':
                date.setMonth(date.getMonth() + 1);
                break;
            case 'quarterly':
                date.setMonth(date.getMonth() + 3);
                break;
            case 'biannual':
                date.setMonth(date.getMonth() + 6);
                break;
            case 'annual':
                date.setFullYear(date.getFullYear() + 1);
                break;
            default:
                date.setFullYear(date.getFullYear() + 1);
        }
        return date.toISOString();
    }

    /**
     * Log de auditoria
     */
    logActivity(activity, action) {
        console.log(`[COMPLIANCE-AUDIT] ${action} - Activity ID: ${activity.id}, Name: ${activity.name}, Legal Basis: ${activity.legalBasis}`);
    }

    /**
     * Exporta ROPA para auditoria
     */
    exportROPA(format = 'json') {
        const activities = Array.from(this.activities.values());
        
        if (format === 'csv') {
            return this.convertToCSV(activities);
        }
        
        return {
            exportDate: new Date().toISOString(),
            totalActivities: activities.length,
            activities: activities
        };
    }

        const headers = ['ID', 'Name', 'Purpose', 'Legal Basis', 'Data Categories', 'Retention Period', 'Status'];

            a.id,
            a.name,
            a.purpose,
            a.legalBasis,
            a.dataCategories.join(';'),
            a.retentionPeriod,
            a.status
        ]);
        
        return [headers, ...rows].map(row => row.join(',')).join('\n');
    }
}
```

```javascript
        } else if (retentionPolicy.includes('day')) {

            date.setDate(date.getDate() - days);
```

class FeatureStore:
    """Feature Store para versionamento e reuso de features"""
    
    def __init__(self, config: Dict):
        self.redis_client = redis.Redis(
            host=config['host'],
            port=config['port'],
            decode_responses=False
        )
        self.ttl = timedelta(hours=config.get('ttl_hours', 720))
        
    def store_features(self, 
                      model_id: str,
                      inference_id: str,
                      features: Dict[str, Any],
                      metadata: Optional[Dict] = None) -> str:
        """Armazena features para um modelo específico"""
        
        feature_key = f"features:{model_id}:{inference_id}"
        feature_data = {
            'features': features,
            'timestamp': datetime.utcnow().isoformat(),
            'metadata': metadata or {},
            'feature_hash': self._calculate_hash(features)
        }
        
        # Serializa e armazena
        serialized = pickle.dumps(feature_data)
        self.redis_client.setex(
            feature_key,
            int(self.ttl.total_seconds()),
            serialized
        )
        
        # Indexa por modelo e timestamp
        index_key = f"index:{model_id}"
        self.redis_client.zadd(
            index_key,
            {inference_id: datetime.utcnow().timestamp()}
        )
        
        logger.info(f"Features stored for {model_id}:{inference_id}")
        return inference_id
    
    def retrieve_features(self, model_id: str, inference_id: str) -> Optional[Dict]:
        """Recupera features armazenadas"""
        feature_key = f"features:{model_id}:{inference_id}"
        serialized = self.redis_client.get(feature_key)
        
        if serialized:
            return pickle.loads(serialized)
        return None
    
    def get_feature_statistics(self, model_id: str, feature_name: str, 
                              start_time: datetime, end_time: datetime) -> Dict:
        """Calcula estatísticas de uma feature específica"""
        # Implementação simplificada - em produção usar RedisTimeSeries
        stats = {
            'mean': 0.0,
            'std': 0.0,
            'min': float('inf'),
            'max': float('-inf'),
            'count': 0
        }
        
        values = []
        index_key = f"index:{model_id}"
        inference_ids = self.redis_client.zrangebyscore(
            index_key,
            start_time.timestamp(),
            end_time.timestamp()
        )
        
        for inf_id in inference_ids:
            features = self.retrieve_features(model_id, inf_id.decode())
            if features and feature_name in features['features']:
                value = features['features'][feature_name]
                if isinstance(value, (int, float)):
                    values.append(value)
        
        if values:
            import numpy as np
            stats.update({
                'mean': float(np.mean(values)),
                'std': float(np.std(values)),
                'min': float(np.min(values)),
                'max': float(np.max(values)),
                'count': len(values)
            })

import numpy as np
from scipy import stats
from typing import Dict, List, Tuple, Any
import pandas as pd
from datetime import datetime, timedelta
import logging

from evidently.report import Report

class DriftDetector:
    """Detector de drift de dados e conceito"""
    
    def __init__(self, config: Dict):
        self.psi_threshold = config.get('threshold_psi', 0.1)
        self.ks_threshold = config.get('threshold_ks', 0.05)
        self.window_size = config.get('window_size', 1000)
        self.reference_data = {}
        
    def calculate_psi(self, expected: np.ndarray, actual: np.ndarray, 
                     buckets: int = 10) -> float:
        """Calcula Population Stability Index"""
        # Cria buckets baseados nos dados esperados
        breakpoints = np.percentile(expected, np.linspace(0, 100, buckets + 1))
        
        expected_hist, _ = np.histogram(expected, bins=breakpoints)
        actual_hist, _ = np.histogram(actual, bins=breakpoints)
        
        # Adiciona pequeno valor para evitar divisão por zero
        expected_hist = expected_hist + 0.0001
        actual_hist = actual_hist + 0.0001
        
        # Normaliza
        expected_perc = expected_hist / len(expected)
        actual_perc = actual_hist / len(actual)
        
        # Calcula PSI
        psi = np.sum((actual_perc - expected_perc) * np.log(actual_perc / expected_perc))
        return psi
    
    def detect_data_drift(self, reference_data: Dict[str, np.ndarray],
                         current_data: Dict[str, np.ndarray]) -> Dict[str, Any]:
        """Detecta drift em múltiplas features"""
        
        drift_report = {
            'drift_detected': False,
            'features': {},
            'overall_psi': 0.0,
            'timestamp': datetime.utcnow().isoformat()
        }
        
        feature_drifts = []
        
        for feature_name in reference_data.keys():
            if feature_name not in current_data:
                continue
                
            ref_data = reference_data[feature_name]
            curr_data = current_data[feature_name]
            
            if len(ref_data) < 100 or len(curr_data) < 100:
                continue
            
            # KS Test
            ks_stat, ks_pvalue = stats.ks_2samp(ref_data, curr_data)
            
            # PSI
            psi = self.calculate_psi(ref_data, curr_data)
            
            # Chi-square para features categóricas
            chi2_stat, chi2_pvalue = None, None
            if len(np.unique(ref_data)) < 20:  # Considera categórica
                # Converte para contagens
                unique_vals = np.unique(np.concatenate([ref_data, curr_data]))
                ref_counts = np.array([np.sum(ref_data == val) for val in unique_vals])
                curr_counts = np.array([np.sum(curr_data == val) for val in unique_vals])
                
                chi2_stat, chi2_pvalue = stats.chisquare(curr_counts, ref_counts)
            
            feature_drift = {
                'feature': feature_name,
                'ks_statistic': float(ks_stat),
                'ks_pvalue': float(ks_pvalue),
                'psi': float(psi),
                'chi2_pvalue': float(chi2_pvalue) if chi2_pvalue else None,
                'drift_detected': psi > self.psi_threshold or ks_pvalue < self.ks_threshold
            }
            
            drift_report['features'][feature_name] = feature_drift
            feature_drifts.append(feature_drift)
            
            if feature_drift['drift_detected']:
                drift_report['drift_detected'] = True
        
        # PSI geral (média ponderada)
        if feature_drifts:
            drift_report['overall_psi'] = float(np.mean([f['psi'] for f in feature_drifts]))
        
        return drift_report
    
    def detect_concept_drift(self, predictions: np.ndarray, 
                           actuals: np.ndarray, 
                           window_size: int = 100) -> Dict[str, Any]:
        """Detecta drift de conceito usando accuracy móvel"""
        
        if len(predictions) < window_size * 2:
            return {'error': 'Insufficient data'}
        
        accuracies = []
        for i in range(0, len(predictions) - window_size, window_size // 2):
            window_preds = predictions[i:i+window_size]
            window_actuals = actuals[i:i+window_size]
            
            if len(window_preds) == window_size:
                accuracy = np.mean(window_preds == window_actuals)
                accuracies.append(accuracy)
        
        # Testa se há tendência decrescente
        if len(accuracies) >= 3:
            x = np.arange(len(accuracies))
            slope, intercept = np.polyfit(x, accuracies, 1)
            
            # Teste t para significância
            t_stat, p_value = stats.ttest_1samp(accuracies, np.mean(accuracies))
            
            drift_detected = slope < -0.01 and p_value < 0.05
            
            return {
                'drift_detected': drift_detected,
                'slope': float(slope),
                'p_value': float(p_value),
                'mean_accuracy': float(np.mean(accuracies)),
                'accuracy_trend': accuracies,
                'timestamp': datetime.utcnow().isoformat()
            }
        
        return {'drift_detected': False}
