## ADR-001: Arquitetura de MicroserviГ§os

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

### CONTEXTO:
DIYAPP V11 Г© uma aplicaГ§ГЈo monolГ­tica que estГЎ enfrentando:
1. Dificuldade de escalar componentes especГ­ficos
2. Acoplamento tГ©cnico entre funcionalidades
3. Deploys lentos e arriscados
4. Dificuldade em adotar diferentes stacks tecnolГіgicas por componente

### DECISГѓO:
Adotar arquitetura de microserviГ§os com os seguintes princГ­pios:
1. Cada serviГ§o Г© responsГЎvel por um domГ­nio de negГіcio especГ­fico
2. ComunicaГ§ГЈo via API REST sГ­ncrona e eventos assГ­ncronos
3. Cada serviГ§o possui seu prГіprio banco de dados
4. API Gateway como ponto Гєnico de entrada
5. Service Mesh para comunicaГ§ГЈo entre serviГ§os

### OPГ‡Г•ES CONSIDERADAS:
- **Monolito modularizado:** MantГ©m cГіdigo Гєnico com separaГ§ГЈo lГіgica
- **MicrosserviГ§os leves:** 4-6 serviГ§os com responsabilidades bem definidas
- **NanosserviГ§os:** 10+ serviГ§os altamente especializados

**OpГ§ГЈo escolhida:** MicrosserviГ§os leves (4 serviГ§os iniciais)

### CONSEQUГЉNCIAS:

- Escalabilidade independente por serviГ§o
- Deploys mais rГЎpidos e seguros

- Monitoramento distribuГ­do necessГЎrio

```typescript
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import rateLimit from 'express-rate-limit';
import { createProxyMiddleware } from 'http-proxy-middleware';
import { logger } from '../shared/lib/logger';
import { errorHandler } from '../shared/middleware/error-handler';
import { requestLogger } from '../shared/middleware/request-logger';
import { authMiddleware } from '../shared/middleware/auth';
import { metricsMiddleware } from '../shared/middleware/metrics';

const app = express();
const PORT = process.env.PORT || 3000;

// Security middleware
app.use(helmet());
app.use(cors({
  origin: process.env.ALLOWED_ORIGINS?.split(',') || ['http://localhost:8080'],
  credentials: true
}));

// Rate limiting
const limiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 minutes
  max: 100 // limit each IP to 100 requests per windowMs
});
app.use('/api/', limiter);

// Logging and metrics
app.use(requestLogger);
app.use(metricsMiddleware);

// Health check
app.get('/health', (req, res) => {
  res.json({ 
    status: 'healthy',
    timestamp: new Date().toISOString(),
    service: 'api-gateway',
    version: process.env.npm_package_version || '1.0.0'
  });
});

// Authentication middleware for protected routes
app.use('/api/v1/*', authMiddleware);

// Service routing
const services = {
  users: process.env.USER_SERVICE_URL || 'http://user-service:3001',
  tasks: process.env.TASK_SERVICE_URL || 'http://task-service:3002',
  llm: process.env.LLM_SERVICE_URL || 'http://llm-orchestrator:3003',
  notifications: process.env.NOTIFICATION_SERVICE_URL || 'http://notification-service:3004'
};

// Proxy configuration
Object.entries(services).forEach(([path, target]) => {
  app.use(`/api/v1/${path}`, createProxyMiddleware({
    target,
    changeOrigin: true,
    pathRewrite: {
      [`^/api/v1/${path}`]: '/api/v1'
    },
    onProxyReq: (proxyReq, req, res) => {
      logger.info(`Proxying ${req.method} ${req.path} to ${target}`);
    },
    onError: (err, req, res) => {
      logger.error(`Proxy error for ${req.path}:`, err);
      res.status(502).json({ 
        error: 'Service temporarily unavailable',
        path: req.path 
      });
    }
  }));
});

// 404 handler
app.use('*', (req, res) => {
  res.status(404).json({ 
    error: 'Route not found',
    path: req.path,
    method: req.method 
  });
});

// Global error handler
app.use(errorHandler);

app.listen(PORT, () => {
  logger.info(`API Gateway running on port ${PORT}`);
  logger.info(`Proxying to services:`, services);
});

export default app;
```

## LLM Orchestrator (Node.js/TypeScript)

### `src/services/llm-orchestrator/src/core/llm-manager.ts`
```typescript
import { OpenAI } from 'openai';
import { GoogleGenerativeAI } from '@google/generative-ai';
import { Anthropic } from '@anthropic-ai/sdk';
import { logger } from '../../shared/lib/logger';
import { CircuitBreaker } from '../../shared/lib/circuit-breaker';
import { CacheService } from './cache-service';

export interface LLMRequest {
  prompt: string;
  model: string;
  temperature?: number;
  maxTokens?: number;
  systemPrompt?: string;
}

export interface LLMResponse {
  content: string;
  model: string;
  tokensUsed: number;
  latency: number;
  cost: number;
}

export class LLMManager {
  private openai: OpenAI;
  private gemini: GoogleGenerativeAI;
  private anthropic: Anthropic;
  private circuitBreakers: Map<string, CircuitBreaker>;
  private cache: CacheService;

  constructor() {
    this.openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
    this.gemini = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);
    this.anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
    this.circuitBreakers = new Map();
    this.cache = new CacheService();
    
    // Initialize circuit breakers for each provider
    ['openai', 'gemini', 'anthropic'].forEach(provider => {
      this.circuitBreakers.set(provider, new CircuitBreaker({
        failureThreshold: 5,
        resetTimeout: 60000,
        halfOpenMaxAttempts: 3
      }));
    });
  }

  async generateCompletion(request: LLMRequest): Promise<LLMResponse> {
    const startTime = Date.now();
    
    // Check cache first
    const cacheKey = this.generateCacheKey(request);
    const cached = await this.cache.get(cacheKey);
    if (cached) {
      logger.debug('Cache hit for LLM request', { cacheKey });
      return {
        ...cached,
        latency: Date.now() - startTime,
        cost: 0 // Cached responses have no cost
      };
    }

    // Select provider based on model
    const provider = this.getProviderForModel(request.model);
    const circuitBreaker = this.circuitBreakers.get(provider);
    
    if (!circuitBreaker?.call()) {
      logger.warn(`Circuit breaker open for ${provider}, trying fallback`);
      return this.fallbackCompletion(request);
    }

    try {
      let response: LLMResponse;
      
      switch (provider) {
        case 'openai':
          response = await this.callOpenAI(request);
          break;
        case 'gemini':
          response = await this.callGemini(request);
          break;
        case 'anthropic':
          response = await this.callAnthropic(request);
          break;
        default:
          throw new Error(`Unsupported provider: ${provider}`);
      }
      
      circuitBreaker.success();
      
      // Cache successful response
      await this.cache.set(cacheKey, response, 3600); // Cache for 1 hour
      
      return {
        ...response,
        latency: Date.now() - startTime
      };
      
    } catch (error) {
      circuitBreaker.failure();
      logger.error(`LLM call failed for ${provider}:`, error);
      
      // Try fallback
      return this.fallbackCompletion(request);
    }
  }

  private async callOpenAI(request: LLMRequest): Promise<LLMResponse> {
    const completion = await this.openai.chat.completions.create({
      model: request.model,
      messages: [
        { role: 'system', content: request.systemPrompt || 'You are a helpful assistant' },
        { role: 'user', content: request.prompt }
      ],
      temperature: request.temperature || 0.7,
      max_tokens: request.maxTokens || 1000
    });

    return {
      content: completion.choices[0]?.message?.content || '',
      model: request.model,
      tokensUsed: completion.usage?.total_tokens || 0,
      latency: 0, // Will be calculated by caller
      cost: this.calculateCost('openai', request.model, completion.usage?.total_tokens || 0)
    };
  }

  private async callGemini(request: LLMRequest): Promise<LLMResponse> {
    const model = this.gemini.getGenerativeModel({ model: request.model });
    const result = await model.generateContent(request.prompt);
    const response = await result.response;
    
    return {
      content: response.text(),
      model: request.model,
      tokensUsed: response.usageMetadata?.totalTokenCount || 0,
      latency: 0,
      cost: this.calculateCost('gemini', request.model, response.usageMetadata?.totalTokenCount || 0)
    };
  }

  private calculateCost(provider: string, model: string, tokens: number): number {
    // Simplified cost calculation
    const rates: Record<string, number> = {
      'openai:gpt-4': 0.03 / 1000,
      'openai:gpt-3.5-turbo': 0.002 / 1000,
      'gemini:gemini-pro': 0.0005 / 1000,
      'anthropic:claude-2': 0.01102 / 1000
    };
    
    const key = `${provider}:${model}`;
    return (rates[key] || 0.001) * tokens;
  }

  private generateCacheKey(request: LLMRequest): string {
    return `llm:${request.model}:${Buffer.from(request.prompt).toString('base64')}`;
  }

  private getProviderForModel(model: string): string {
    if (model.includes('gpt')) return 'openai';
    if (model.includes('gemini')) return 'gemini';
    if (model.includes('claude')) return 'anthropic';
    return 'openai'; // Default
  }

  private async fallbackCompletion(request: LLMRequest): Promise<LLMResponse> {
    // Implement fallback logic
    logger.info('Using fallback completion');
    
    // Try different model from same provider
    const fallbackModels: Record<string, string> = {
      'gpt-4': 'gpt-3.5-turbo',
      'gemini-pro': 'gemini-pro-vision',
      'claude-2': 'claude-instant-1'
    };
    
    const fallbackModel = fallbackModels[request.model] || 'gpt-3.5-turbo';
    
    return this.generateCompletion({
      ...request,
      model: fallbackModel
    });
  }
}
```

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

class JWTBearer(HTTPBearer):
    def __init__(self, auto_error: bool = True):
        super().__init__(auto_error=auto_error)
    
    async def __call__(self, request: Request) -> Optional[HTTPAuthorizationCredentials]:
        credentials = await super().__call__(request)
        
        if not credentials:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Invalid authorization code"
            )
        
        if credentials.scheme != "Bearer":
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Invalid authentication scheme"
            )
        
        return credentials

def verify_password(plain_password: str, hashed_password: str) -> bool:
    """Verifica senha com bcrypt"""
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    """Gera hash de senha com bcrypt"""
    return pwd_context.hash(password)

def create_access_token(data: Dict[str, Any], expires_delta: Optional[timedelta] = None) -> str:
    """Cria token JWT de acesso"""
    to_encode = data.copy()
    
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES)
    
    to_encode.update({"exp": expire, "type": "access"})
    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    
    # Log seguro (sem dados sensГ­veis)
    user_id = data.get("sub")
    logger.info("Access token created", extra={
        "user_id": user_id,
        "token_type": "access",
        "expires_in": expires_delta.total_seconds() if expires_delta else settings.ACCESS_TOKEN_EXPIRE_MINUTES * 60
    })
    
    return encoded_jwt

def create_refresh_token(data: Dict[str, Any]) -> str:
    """Cria token JWT de refresh"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(days=settings.REFRESH_TOKEN_EXPIRE_DAYS)
    
    to_encode.update({"exp": expire, "type": "refresh"})
    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    
    # Armazena refresh token no Redis para invalidar se necessГЎrio
    redis_client = get_redis_client()
    user_id = data.get("sub")
    if user_id:
        key = f"refresh_token:{user_id}"
        redis_client.setex(key, settings.REFRESH_TOKEN_EXPIRE_DAYS * 24 * 3600, encoded_jwt)
    
    return encoded_jwt

def verify_token(token: str, token_type: str = "access") -> Dict[str, Any]:
    """Verifica e decodifica token JWT"""
    try:
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        
        # Verifica tipo de token
        if payload.get("type") != token_type:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid token type"
            )
        
        # Para refresh tokens, verifica se estГЎ na lista de vГЎlidos
        if token_type == "refresh":
            redis_client = get_redis_client()
            user_id = payload.get("sub")
            stored_token = redis_client.get(f"refresh_token:{user_id}")
            
            if not stored_token or stored_token.decode() != token:
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail="Refresh token invalid or expired"
                )
        
        return payload
        
    except JWTError as e:
        logger.warning(f"JWT verification failed: {str(e)}", extra={"token_type": token_type})
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Could not validate credentials"
        )

    """Invalida refresh token do usuГЎrio"""
    redis_client = get_redis_client()
    redis_client.delete(f"refresh_token:{user_id}")
    logger.info(f"Refresh token invalidated for user {user_id}")

class RedisClient:
    """Cliente Redis com circuit breaker e fallback"""
    
    def __init__(self):
        self._client = None
        self._connected = False
    
    @circuit_breaker(
        failure_threshold=settings.CIRCUIT_BREAKER_FAILURE_THRESHOLD,
        recovery_timeout=settings.CIRCUIT_BREAKER_RECOVERY_TIMEOUT
    )
    def connect(self) -> redis.Redis:
        """Conecta ao Redis com timeout e retry"""
        if self._connected and self._client:
            return self._client
        
        try:
            self._client = redis.Redis(
                host=settings.REDIS_HOST,
                port=settings.REDIS_PORT,
                password=settings.REDIS_PASSWORD,
                db=settings.REDIS_DB,
                socket_connect_timeout=5,
                socket_timeout=5,
                retry_on_timeout=True,
                max_connections=50,
                health_check_interval=30
            )
            
            # Testa conexГЈo
            self._client.ping()
            self._connected = True
            logger.info("Redis connected successfully")
            
            return self._client
            
        except (ConnectionError, TimeoutError) as e:
            logger.error(f"Redis connection failed: {str(e)}")
            self._connected = False
            raise
    
    def get(self, key: str) -> Optional[Any]:
        """ObtГ©m valor do cache com fallback silencioso"""
        try:
            client = self.connect()
            value = client.get(key)
            return json.loads(value) if value else None
        except RedisError as e:
            logger.warning(f"Redis get failed for key {key}: {str(e)}")
            return None
    
    def set(self, key: str, value: Any, expire: int = 3600) -> bool:
        """Define valor no cache com fallback silencioso"""
        try:
            client = self.connect()
            client.setex(key, expire, json.dumps(value))
            return True
        except RedisError as e:
            logger.warning(f"Redis set failed for key {key}: {str(e)}")
            return False
    
    def delete(self, key: str) -> bool:
        """Remove valor do cache com fallback silencioso"""
        try:
            client = self.connect()
            client.delete(key)
            return True
        except RedisError as e:
            logger.warning(f"Redis delete failed for key {key}: {str(e)}")
            return False
    
    def clear_pattern(self, pattern: str) -> int:
        """Remove todas as chaves que correspondem ao padrГЈo"""
        try:
            client = self.connect()
            keys = client.keys(pattern)
            if keys:
                return client.delete(*keys)
            return 0
        except RedisError as e:
            logger.warning(f"Redis clear pattern failed for {pattern}: {str(e)}")
            return 0

// Metrics middleware for Express
export const metricsMiddleware = (req: any, res: any, next: Function) => {
  const start = Date.now();
  
  // Increment active connections
  activeConnections.inc();
  
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    
    // Record request duration
    httpRequestDuration
      .labels(req.method, req.route?.path || req.path, res.statusCode.toString())
      .observe(duration);
    
    // Increment request counter
    httpRequestsTotal
      .labels(req.method, req.route?.path || req.path, res.statusCode.toString())
      .inc();
    
    // Decrement active connections
    activeConnections.dec();
  });
  
  next();
};

# src/core/circuit_breaker.py
import time
from enum import Enum
from typing import Any, Callable, Optional
import structlog
from prometheus_client import Counter, Histogram

logger = structlog.get_logger(__name__)

# MГ©tricas Prometheus
CIRCUIT_OPENED = Counter('circuit_breaker_opened_total', 'Total times circuit opened', ['service'])
CIRCUIT_CLOSED = Counter('circuit_breaker_closed_total', 'Total times circuit closed', ['service'])
CIRCUIT_HALF_OPEN = Counter('circuit_breaker_half_open_total', 'Total times circuit half opened', ['service'])
REQUEST_DURATION = Histogram('external_service_request_duration_seconds', 'Request duration to external services', ['service', 'status'])

class CircuitState(Enum):
    CLOSED = "CLOSED"
    OPEN = "OPEN"
    HALF_OPEN = "HALF_OPEN"

class CircuitBreaker:
    """Circuit Breaker pattern implementation with Prometheus metrics"""
    
    def __init__(
        self,
        service_name: str,
        failure_threshold: int = 5,
        recovery_timeout: int = 30,
        half_open_max_calls: int = 3
    ):
        self.service_name = service_name
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_max_calls = half_open_max_calls
        
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.last_failure_time = None
        self.half_open_success_count = 0
        
    def __call__(self, func: Callable) -> Callable:
        """Decorator para aplicar circuit breaker a uma funГ§ГЈo"""
        def wrapper(*args, **kwargs) -> Any:
            return self.execute(func, *args, **kwargs)
        return wrapper
    
    def execute(self, func: Callable, *args, **kwargs) -> Any:
        """Executa a funГ§ГЈo com circuit breaker"""
        if self.state == CircuitState.OPEN:
            # Verifica se jГЎ passou tempo suficiente para tentar recovery
            if self._should_try_recovery():
                self.state = CircuitState.HALF_OPEN
                self.half_open_success_count = 0
                CIRCUIT_HALF_OPEN.labels(service=self.service_name).inc()
                logger.info("Circuit half-opened", service=self.service_name)
            else:
                raise CircuitOpenException(self.service_name, self.recovery_timeout)
        
        try:
            start_time = time.time()
            result = func(*args, **kwargs)
            duration = time.time() - start_time
            
            # Registra mГ©trica de sucesso
            REQUEST_DURATION.labels(
                service=self.service_name,
                status="success"
            ).observe(duration)
            
            self._on_success()
            return result
            
        except Exception as e:
            duration = time.time() - start_time if 'start_time' in locals() else 0
            
            # Registra mГ©trica de falha
            REQUEST_DURATION.labels(
                service=self.service_name,
                status="failure"
            ).observe(duration)
            
            self._on_failure()
            raise
    
    def _on_success(self):
        """Lida com sucesso na execuГ§ГЈo"""
        if self.state == CircuitState.HALF_OPEN:
            self.half_open_success_count += 1
            if self.half_open_success_count >= self.half_open_max_calls:
                self.state = CircuitState.CLOSED
                self.failure_count = 0
                CIRCUIT_CLOSED.labels(service=self.service_name).inc()
                logger.info("Circuit closed", service=self.service_name)
        
        elif self.state == CircuitState.CLOSED:
            self.failure_count = 0
    
    def _on_failure(self):
        """Lida com falha na execuГ§ГЈo"""
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        if self.state == CircuitState.HALF_OPEN:
            self.state = CircuitState.OPEN
            CIRCUIT_OPENED.labels(service=self.service_name).inc()
            logger.warning("Circuit re-opened", service=self.service_name)
        
        elif (self.state == CircuitState.CLOSED and 
              self.failure_count >= self.failure_threshold):
            self.state = CircuitState.OPEN
            CIRCUIT_OPENED.labels(service=self.service_name).inc()
            logger.error("Circuit opened", 
                        service=self.service_name,
                        failure_count=self.failure_count)
    
    def _should_try_recovery(self) -> bool:
        """Verifica se pode tentar recovery"""
        if not self.last_failure_time:
            return False
        
        elapsed = time.time() - self.last_failure_time
        return elapsed >= self.recovery_timeout
    
    def get_status(self) -> dict:
        """Retorna status atual do circuit breaker"""
        return {
            "service": self.service_name,
            "state": self.state.value,
            "failure_count": self.failure_count,
            "last_failure_time": self.last_failure_time,
            "half_open_success_count": self.half_open_success_count
        }

class CircuitOpenException(Exception):
    """ExceГ§ГЈo lanГ§ada quando circuit estГЎ aberto"""
    def __init__(self, service_name: str, recovery_timeout: int):
        super().__init__(
            f"Circuit breaker for {service_name} is OPEN. "
            f"Try again in {recovery_timeout} seconds"
        )
        self.service_name = service_name
        self.recovery_timeout = recovery_timeout

    """PolГ­tica de retry com backoff exponencial e jitter"""
    
    @staticmethod
    def external_service(
        max_attempts: int = 3,
        initial_wait: float = 1.0,
        max_wait: float = 10.0,
        jitter: bool = True
    ):
        """
        Decorator para chamadas a serviГ§os externos
        Aplica apenas para exceГ§Гµes transitГіrias (timeout, 5xx)
        """

            @retry(
                stop=stop_after_attempt(max_attempts),
                wait=wait_exponential(
                    multiplier=initial_wait,
                    max=max_wait,

                before_sleep=before_sleep_log(logger, logging.WARNING),
                after=after_log(logger, logging.INFO)
            )

                return func(*args, **kwargs)
            return wrapper
        return decorator

```sql
WITH daily_events AS (
    SELECT
        event_date,
        user_id,
        COUNT(DISTINCT session_id) as daily_sessions,
        COUNT(*) as daily_events,
        SUM(duration_seconds) as total_time_seconds
    FROM {{ ref('stg_events') }}

),

daily_metrics AS (
    SELECT
        event_date,
        COUNT(DISTINCT user_id) as daily_active_users,
        COUNT(DISTINCT CASE WHEN daily_sessions >= 3 THEN user_id END) as power_users,
        COUNT(DISTINCT CASE WHEN daily_events >= 10 THEN user_id END) as engaged_users,
        AVG(daily_sessions) as avg_sessions_per_user,
        AVG(daily_events) as avg_events_per_user,
        AVG(total_time_seconds) as avg_time_per_user_seconds,
        SUM(total_time_seconds) as total_time_all_users_seconds
    FROM daily_events
    GROUP BY 1
),

retention AS (
    SELECT
        first_date,
        event_date,
        COUNT(DISTINCT user_id) as retained_users
    FROM (
        SELECT
            user_id,
            MIN(event_date) as first_date,
            event_date
        FROM {{ ref('stg_events') }}
        GROUP BY 1, 3
    )
    GROUP BY 1, 2
)

SELECT
    dm.event_date,
    dm.daily_active_users,
    dm.power_users,
    dm.engaged_users,
    dm.avg_sessions_per_user,
    dm.avg_events_per_user,
    dm.avg_time_per_user_seconds,
    dm.total_time_all_users_seconds,
    COALESCE(r.retained_users, 0) as day_1_retention,
    LAG(dm.daily_active_users, 7) OVER (ORDER BY dm.event_date) as dau_7d_ago,
    LAG(dm.daily_active_users, 30) OVER (ORDER BY dm.event_date) as dau_30d_ago,
    CASE 
        WHEN LAG(dm.daily_active_users, 7) OVER (ORDER BY dm.event_date) > 0 
        THEN ROUND(100.0 * dm.daily_active_users / LAG(dm.daily_active_users, 7) OVER (ORDER BY dm.event_date), 2)
        ELSE NULL
    END as wow_growth_percent
FROM daily_metrics dm
LEFT JOIN retention r ON dm.event_date = r.event_date AND r.first_date = dm.event_date - INTERVAL '1 day'
```

```sql
SELECT
    dlc.request_date,
    dlc.feature_name,
    dlc.model_name,
    dlc.provider,
    dlc.total_calls,
    dlc.successful_calls,
    dlc.failed_calls,
    ROUND(100.0 * dlc.failed_calls / NULLIF(dlc.total_calls, 0), 2) as error_rate_percent,
    dlc.total_prompt_tokens,
    dlc.total_completion_tokens,
    dlc.total_tokens,
    dlc.total_cost_usd,
    fct.cost_7d_running_sum,
    fct.cost_7d_running_avg,
    dlc.avg_latency_ms,
    dlc.p50_latency_ms,
    dlc.p95_latency_ms,
    dlc.avg_processing_time_seconds,
    ROUND(dlc.total_cost_usd / NULLIF(dlc.total_tokens, 0) * 1000, 4) as cost_per_1k_tokens_usd,
    ROUND(dlc.total_cost_usd / NULLIF(dlc.total_calls, 0), 4) as cost_per_call_usd
FROM daily_llm_calls dlc
LEFT JOIN feature_cost_trend fct ON dlc.request_date = fct.request_date AND dlc.feature_name = fct.feature_name
ORDER BY dlc.request_date DESC, dlc.total_cost_usd DESC
```
