## 2. Padrões de Código e Engenharia

### 2.1 Engineering Standards Document

```markdown
# ENGINEERING_STANDARDS_V12.md

## 1. Princípios de Design

### 1.1 SOLID + Clean Architecture
- Single Responsibility: Cada módulo/componente tem uma única razão para mudar
- Open/Closed: Aberto para extensão, fechado para modificação
- Liskov Substitution: Subtipos devem ser substituíveis por seus tipos base
- Interface Segregation: Múltiplas interfaces específicas > uma interface geral
- Dependency Inversion: Depender de abstrações, não de implementações

### 1.2 Domain-Driven Design (DDD Lite)
- Entidades de domínio claramente definidas
- Value Objects imutáveis
- Aggregates com boundaries consistentes
- Repositories para persistência
- Domain Events para comunicação assíncrona

class CacheManager:

        """Connect to Redis with resilience"""
        try:
            self._redis = redis.from_url(
                self.redis_url,
                encoding="utf-8",

                socket_connect_timeout=5,
                socket_timeout=5,
                retry_on_timeout=True,

            await self._redis.ping()
            logger.info("Redis connected successfully")
            
        except RedisError as e:
            logger.error(f"Redis connection failed: {str(e)}")
            self._circuit_open = True
            self._circuit_last_failure = time.time()
            raise
    
    async def disconnect(self):
        """Disconnect from Redis"""

            await self._redis.close()
            logger.info("Redis disconnected")
    
    def _check_circuit(self) -> bool:
        """Check if circuit breaker should be closed"""
        if self._circuit_open:
            if time.time() - self._circuit_last_failure > self._circuit_reset_timeout:
                self._circuit_open = False
                logger.info("Redis circuit breaker closed")
                return True
            return False
        return True

    async def get(self, key: str) -> Optional[Any]:
        """Get value from cache with circuit breaker"""
        if not self._check_circuit():
            logger.warning("Redis circuit open, returning None")
            return None
            
        try:
            start_time = time.time()
            value = await self._redis.get(key)
            duration = (time.time() - start_time) * 1000
            
            if duration > 10:  # Log slow cache operations
                logger.warning(f"Slow cache get: {duration:.2f}ms for key {key}")

                try:
                    return json.loads(value)
                except:
                    return pickle.loads(value.encode('latin1'))
            return None
            
        except RedisError as e:
            logger.error(f"Redis get failed: {str(e)}")
            self._circuit_open = True
            self._circuit_last_failure = time.time()
            return None

-- DIYAPP Analytics Database Schema V12
-- PostgreSQL with TimeScaleDB for time-series data

-- Enable extensions
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "timescaledb";

-- ==================== RAW LAYER (Bronze) ====================
-- Raw tables - never modified, append-only

-- User Events (from frontend/backend)
CREATE TABLE raw.user_events (
    event_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id VARCHAR(255),
    session_id VARCHAR(255),
    event_type VARCHAR(100) NOT NULL,
    event_name VARCHAR(200) NOT NULL,
    event_timestamp TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    page_url VARCHAR(500),
    referrer VARCHAR(500),
    user_agent TEXT,
    device_type VARCHAR(50),
    browser VARCHAR(100),
    os VARCHAR(100),
    country_code CHAR(2),
    city VARCHAR(100),
    event_properties JSONB,
    user_properties JSONB,
    app_version VARCHAR(20),
    platform VARCHAR(50),
    ingested_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    source_system VARCHAR(100) NOT NULL,
    batch_id VARCHAR(100)
);

-- Create hypertable for time-series partitioning
SELECT create_hypertable('raw.user_events', 'event_timestamp');

-- LLM Gateway Calls
CREATE TABLE raw.llm_calls (
    call_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    request_id VARCHAR(255) UNIQUE NOT NULL,
    user_id VARCHAR(255),
    session_id VARCHAR(255),
    feature_name VARCHAR(100) NOT NULL,
    model_provider VARCHAR(50) NOT NULL,
    model_name VARCHAR(100) NOT NULL,
    prompt_tokens INTEGER NOT NULL,
    completion_tokens INTEGER NOT NULL,

    cost_usd DECIMAL(10,6) NOT NULL,
    latency_ms INTEGER NOT NULL,
    status_code INTEGER NOT NULL,
    error_message TEXT,
    request_timestamp TIMESTAMPTZ NOT NULL,
    response_timestamp TIMESTAMPTZ NOT NULL,
    request_metadata JSONB,
    response_metadata JSONB,
    ingested_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    source_system VARCHAR(100) DEFAULT 'llm_gateway'
);

SELECT create_hypertable('raw.llm_calls', 'request_timestamp');

-- API Logs
CREATE TABLE raw.api_logs (
    log_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

    user_id VARCHAR(255),
    endpoint VARCHAR(500) NOT NULL,
    method VARCHAR(10) NOT NULL,
    status_code INTEGER NOT NULL,
    response_time_ms INTEGER NOT NULL,
    request_size_bytes INTEGER,
    response_size_bytes INTEGER,
    ip_address INET,
    user_agent TEXT,
    request_headers JSONB,
    request_body JSONB,
    response_body JSONB,
    error_stack TEXT,
    log_timestamp TIMESTAMPTZ NOT NULL,

);

SELECT create_hypertable('raw.api_logs', 'log_timestamp');

-- Deployments

    failure_reason TEXT,
    deployment_metadata JSONB,
    ingested_at TIMESTAMPTZ NOT NULL DEFAULT NOW()

-- Incidents
CREATE TABLE raw.incidents (

    service_affected VARCHAR(100) NOT NULL,

    status VARCHAR(20) NOT NULL CHECK (status IN ('open', 'investigating', 'resolved', 'closed')),
    root_cause TEXT,
    resolution TEXT,
    assigned_to VARCHAR(100),
    incident_metadata JSONB,
    ingested_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ==================== STAGING VIEWS (Silver) ====================
-- Cleaned and standardized views

CREATE VIEW staging.user_events AS
SELECT 
    event_id,
    user_id,
    session_id,
    LOWER(event_type) AS event_type,
    LOWER(event_name) AS event_name,
    event_timestamp,
    page_url,
    referrer,
    user_agent,
    CASE 
        WHEN user_agent ILIKE '%mobile%' THEN 'mobile'
        WHEN user_agent ILIKE '%tablet%' THEN 'tablet'
        ELSE 'desktop'
    END AS device_type,
    CASE 
        WHEN user_agent ILIKE '%chrome%' THEN 'chrome'
        WHEN user_agent ILIKE '%firefox%' THEN 'firefox'
        WHEN user_agent ILIKE '%safari%' THEN 'safari'
        WHEN user_agent ILIKE '%edge%' THEN 'edge'
        ELSE 'other'
    END AS browser,
    country_code,
    city,
    event_properties,
    user_properties,
    app_version,
    platform,
    ingested_at,
    source_system,
    DATE(event_timestamp) AS event_date
FROM raw.user_events
WHERE event_timestamp IS NOT NULL
  AND event_type IS NOT NULL
  AND event_name IS NOT NULL;

CREATE VIEW staging.llm_calls AS
SELECT 
    call_id,
    request_id,
    user_id,
    session_id,
    LOWER(feature_name) AS feature_name,
    LOWER(model_provider) AS model_provider,
    model_name,
    prompt_tokens,
    completion_tokens,
    total_tokens,
    cost_usd,
    latency_ms,
    status_code,
    error_message,
    request_timestamp,
    response_timestamp,
    request_metadata,
    response_metadata,
    ingested_at,
    source_system,
    DATE(request_timestamp) AS request_date,
    CASE 
        WHEN status_code BETWEEN 200 AND 299 THEN 'success'
        WHEN status_code BETWEEN 400 AND 499 THEN 'client_error'
        WHEN status_code BETWEEN 500 AND 599 THEN 'server_error'
        ELSE 'other_error'
    END AS status_category
FROM raw.llm_calls
WHERE request_timestamp IS NOT NULL
  AND model_provider IS NOT NULL
  AND model_name IS NOT NULL
  AND total_tokens >= 0
  AND cost_usd >= 0;

-- ==================== INDEXES ====================

-- Raw layer indexes
CREATE INDEX idx_raw_user_events_user_date ON raw.user_events(user_id, event_timestamp);
CREATE INDEX idx_raw_user_events_session ON raw.user_events(session_id);
CREATE INDEX idx_raw_user_events_type_date ON raw.user_events(event_type, event_timestamp);
CREATE INDEX idx_raw_llm_calls_user_feature ON raw.llm_calls(user_id, feature_name, request_timestamp);
CREATE INDEX idx_raw_llm_calls_model_date ON raw.llm_calls(model_name, request_timestamp);
CREATE INDEX idx_raw_api_logs_endpoint_date ON raw.api_logs(endpoint, log_timestamp);
CREATE INDEX idx_raw_api_logs_status_date ON raw.api_logs(status_code, log_timestamp);

-- JSONB indexes for frequently queried fields
CREATE INDEX idx_raw_user_events_properties ON raw.user_events USING GIN(event_properties);
CREATE INDEX idx_raw_llm_calls_metadata ON raw.llm_calls USING GIN(request_metadata);

-- ==================== RETENTION POLICIES ====================

-- Raw data retention: 90 days for events, 180 days for LLM calls (cost tracking)
SELECT add_retention_policy('raw.user_events', INTERVAL '90 days');
SELECT add_retention_policy('raw.llm_calls', INTERVAL '180 days');
SELECT add_retention_policy('raw.api_logs', INTERVAL '60 days');

-- ==================== DATA QUALITY VIEWS ====================

CREATE VIEW monitoring.data_quality_metrics AS
SELECT 
    'user_events' AS table_name,
    COUNT(*) AS total_rows,
    COUNT(DISTINCT user_id) AS unique_users,
    COUNT(DISTINCT session_id) AS unique_sessions,
    MIN(event_timestamp) AS earliest_event,
    MAX(event_timestamp) AS latest_event,
    SUM(CASE WHEN user_id IS NULL THEN 1 ELSE 0 END) AS null_user_ids,
    SUM(CASE WHEN event_type IS NULL THEN 1 ELSE 0 END) AS null_event_types,
    NOW() AS checked_at
FROM raw.user_events
WHERE event_timestamp >= NOW() - INTERVAL '1 day'
UNION ALL
SELECT 
    'llm_calls' AS table_name,
    COUNT(*) AS total_rows,
    COUNT(DISTINCT user_id) AS unique_users,
    COUNT(DISTINCT feature_name) AS unique_features,
    MIN(request_timestamp) AS earliest_request,
    MAX(request_timestamp) AS latest_request,
    SUM(CASE WHEN cost_usd < 0 THEN 1 ELSE 0 END) AS negative_costs,
    SUM(CASE WHEN total_tokens < 0 THEN 1 ELSE 0 END) AS negative_tokens,
    NOW() AS checked_at
FROM raw.llm_calls
WHERE request_timestamp >= NOW() - INTERVAL '1 day';

-- ==================== REAL-TIME METRICS VIEWS ====================

CREATE MATERIALIZED VIEW realtime.active_users_1h
WITH (timescaledb.continuous) AS
SELECT 
    time_bucket('1 minute', event_timestamp) AS bucket,
    COUNT(DISTINCT user_id) AS active_users
FROM raw.user_events
WHERE event_timestamp >= NOW() - INTERVAL '1 hour'
  AND user_id IS NOT NULL
GROUP BY bucket
WITH NO DATA;

-- Refresh policy for real-time views
SELECT add_continuous_aggregate_policy('realtime.active_users_1h',
    start_offset => INTERVAL '1 hour',
    end_offset => INTERVAL '1 minute',
    schedule_interval => INTERVAL '1 minute');

// src/security/waf/rules.js
class SecurityRules {
  constructor() {
    this.rules = {
      sqlInjection: {
        patterns: [
          /(\%27)|(\')|(\-\-)|(\%23)|(#)/i,
          /((\%3D)|(=))[^\n]*((\%27)|(\')|(\-\-)|(\%3B)|(;))/i,
          /\w*((\%27)|(\'))((\%6F)|o|(\%4F))((\%72)|r|(\%52))/i,
          /((\%27)|(\'))union/i,
          /exec(\s|\+)+(s|x)p\w+/i,
          /insert(\s|\+)+into/i,
          /select(\s|\+)+from/i,
          /delete(\s|\+)+from/i,
          /update(\s|\+)+set/i,
          /drop(\s|\+)+table/i
        ],
        severity: 'HIGH',
        action: 'BLOCK'
      },

      xss: {
        patterns: [
          /<script\b[^>]*>([\s\S]*?)<\/script>/i,
          /javascript:/i,
          /on\w+\s*=/i,
          /<\s*iframe/i,
          /<\s*object/i,
          /<\s*embed/i,
          /<\s*applet/i,
          /<\s*meta/i,
          /<\s*link/i,
          /expression\s*\(/i
        ],
        severity: 'HIGH',
        action: 'BLOCK'
      },

      pathTraversal: {
        patterns: [
          /\.\.\//g,
          /\.\.\\/g,
          /\/etc\/passwd/i,
          /\/etc\/shadow/i,
          /\/proc\/self/i,
          /\.\.%2f/i,
          /\.\.%5c/i
        ],
        severity: 'HIGH',
        action: 'BLOCK'
      },

      commandInjection: {
        patterns: [
          /;\s*\w+/,
          /\|\s*\w+/,
          /&\s*\w+/,
          /\$\s*\(/,
          /`\s*\w+/,
          /\|\|/,
          /&&/,
          /nc\s+-l/,
          /wget\s+/,
          /curl\s+/
        ],
        severity: 'CRITICAL',
        action: 'BLOCK'
      },

      nosqlInjection: {
        patterns: [
          /\$where/i,
          /\$ne/i,
          /\$gt/i,
          /\$lt/i,
          /\$in/i,
          /\$nin/i,
          /\$regex/i,
          /\$exists/i,
          /\$type/i,
          /\$mod/i
        ],
        severity: 'HIGH',
        action: 'BLOCK'
      }
    };

    this.violationCounts = new Map();
    this.blockedIPs = new Set();
  }

  checkRequest(req) {
    const violations = [];
    const ip = req.ip || req.connection.remoteAddress;
    
    // Skip if IP is already blocked
    if (this.blockedIPs.has(ip)) {
      return { blocked: true, reason: 'IP_PERMANENTLY_BLOCKED' };
    }

    // Check all inputs
    const inputs = this._extractInputs(req);
    
    for (const [ruleName, rule] of Object.entries(this.rules)) {
      for (const input of inputs) {
        for (const pattern of rule.patterns) {
          if (pattern.test(input.value)) {
            violations.push({
              rule: ruleName,
              severity: rule.severity,
              input: input.field,
              value: this._maskSensitiveData(input.value),
              pattern: pattern.toString(),
              timestamp: new Date().toISOString()
            });
            
            // Increment violation count for this IP
            const count = this.violationCounts.get(ip) || 0;
            this.violationCounts.set(ip, count + 1);
            
            // Block IP if threshold exceeded
            if (count + 1 >= 10) {
              this.blockedIPs.add(ip);
              return {
                blocked: true,
                reason: 'THRESHOLD_EXCEEDED',
                violations: violations
              };
            }
            
            break;
          }
        }
      }
    }

    return {
      blocked: violations.length > 0,
      violations: violations,
      action: violations.length > 0 ? 'BLOCK' : 'ALLOW'
    };
  }

  _extractInputs(req) {
    const inputs = [];
    
    // URL parameters
    if (req.query) {
      Object.entries(req.query).forEach(([key, value]) => {
        if (typeof value === 'string') {
          inputs.push({ field: `query.${key}`, value });
        }
      });
    }
    
    // Body parameters
    if (req.body) {
      Object.entries(req.body).forEach(([key, value]) => {
        if (typeof value === 'string') {
          inputs.push({ field: `body.${key}`, value });
        } else if (typeof value === 'object') {
          // Handle nested objects
          this._extractNestedInputs(value, `body.${key}`, inputs);
        }
      });
    }
    
    // Headers (skip common headers)
    const skipHeaders = ['user-agent', 'accept', 'accept-encoding', 'accept-language', 'cache-control'];
    Object.entries(req.headers || {}).forEach(([key, value]) => {
      if (!skipHeaders.includes(key.toLowerCase()) && typeof value === 'string') {
        inputs.push({ field: `header.${key}`, value });
      }
    });
    
    return inputs;
  }

  _extractNestedInputs(obj, prefix, inputs) {
    Object.entries(obj).forEach(([key, value]) => {
      const field = `${prefix}.${key}`;
      if (typeof value === 'string') {
        inputs.push({ field, value });
      } else if (typeof value === 'object' && value !== null) {
        this._extractNestedInputs(value, field, inputs);
      }
    });
  }

  _maskSensitiveData(value) {
    // Mask sensitive data in logs
    const sensitivePatterns = [
      /password=([^&]*)/gi,
      /token=([^&]*)/gi,
      /secret=([^&]*)/gi,
      /api[_-]?key=([^&]*)/gi,
      /authorization:\s*bearer\s+(\S+)/gi,
      /"password"\s*:\s*"([^"]*)"/gi,
      /"token"\s*:\s*"([^"]*)"/gi
    ];
    
    let masked = value;
    sensitivePatterns.forEach(pattern => {
      masked = masked.replace(pattern, (match, p1) => match.replace(p1, '***MASKED***'));
    });
    
    return masked;
  }

  getStats() {
    return {
      totalViolations: Array.from(this.violationCounts.values()).reduce((a, b) => a + b, 0),
      blockedIPs: this.blockedIPs.size,
      activeRules: Object.keys(this.rules).length
    };
  }

  clearBlockedIP(ip) {
    this.blockedIPs.delete(ip);
    this.violationCounts.delete(ip);
  }
}

module.exports = SecurityRules;

const SecurityRules = require('./rules');
const AuditLogger = require('../audit/logger');

class WAFMiddleware {
  constructor(config) {
    this.rules = new SecurityRules();
    this.auditLogger = new AuditLogger(config.audit);
    this.config = config;
  }

  middleware() {
    return async (req, res, next) => {
      if (!this.config.enabled) {
        return next();
      }

      const result = this.rules.checkRequest(req);
      
      if (result.blocked) {
        // Log the block
        await this.auditLogger.logSecurityEvent({
          event: 'WAF_BLOCK',
          ip: req.ip,
          method: req.method,
          url: req.url,
          userAgent: req.headers['user-agent'],
          violations: result.violations,
          timestamp: new Date().toISOString()
        });

        // Send block response
        res.status(403).json({
          error: 'Access Denied',
          message: 'Request blocked by security rules',
          requestId: req.id,
          timestamp: new Date().toISOString()
        });
        
        return;
      }

      // Add security headers
      this._addSecurityHeaders(res);
      
      // Log allowed requests in learning mode
      if (this.config.learningMode) {
        await this.auditLogger.logSecurityEvent({
          event: 'WAF_ALLOWED',
          ip: req.ip,
          method: req.method,
          url: req.url,
          timestamp: new Date().toISOString()
        });
      }

      next();
    };
  }

  _addSecurityHeaders(res) {
    res.setHeader('X-Content-Type-Options', 'nosniff');
    res.setHeader('X-Frame-Options', 'DENY');
    res.setHeader('X-XSS-Protection', '1; mode=block');
    res.setHeader('Referrer-Policy', 'strict-origin-when-cross-origin');
    res.setHeader('Permissions-Policy', 'geolocation=(), microphone=(), camera=()');
  }

  getStats() {
    return this.rules.getStats();
  }

  async exportRules() {
    return {
      rules: this.rules.rules,
      stats: this.getStats(),
      timestamp: new Date().toISOString()
    };
  }
}

module.exports = WAFMiddleware;

const express = require('express');
const http = require('http');
const socketIo = require('socket.io');
const path = require('path');
const dotenv = require('dotenv');

// Carregar configurações
dotenv.config();

// Importar componentes do sistema
const AlertEngine = require('./core/alert-engine');
const RemediationEngine = require('./core/remediation-engine');
const HealthMonitor = require('./core/health-monitor');
const AlertManager = require('./alerts/alert-manager');
const NotificationService = require('./alerts/notification-service');
const logger = require('./utils/logger');

// Configurar aplicação Express
const app = express();
const server = http.createServer(app);
const io = socketIo(server);

// Middleware
app.use(express.json());
app.use(express.static(path.join(__dirname, '../public')));

// Rotas da API
app.use('/api/alerts', require('./dashboard/api'));

// Servir dashboard
app.get('/', (req, res) => {
    res.sendFile(path.join(__dirname, '../index.html'));
});

// Inicializar componentes do sistema
async function initializeSystem() {
    try {
        logger.info('Inicializando Sistema de Alertas DIYAPP V12...');

        // Inicializar serviços
        const alertManager = new AlertManager();
        const notificationService = new NotificationService();
        const healthMonitor = new HealthMonitor();
        const remediationEngine = new RemediationEngine();
        const alertEngine = new AlertEngine({
            alertManager,
            notificationService,
            healthMonitor,
            remediationEngine
        });

        // Configurar WebSocket para atualizações em tempo real
        io.on('connection', (socket) => {
            logger.info('Novo cliente conectado ao dashboard');
            
            socket.on('disconnect', () => {
                logger.info('Cliente desconectado do dashboard');
            });

            // Enviar estado inicial
            socket.emit('systemStatus', {
                status: 'operational',
                timestamp: new Date().toISOString()
            });
        });

        // Iniciar monitoramento
        await healthMonitor.start();
        await alertEngine.start();
        await remediationEngine.start();

        // Expor instâncias para uso global (apenas para desenvolvimento)
        global.alertEngine = alertEngine;
        global.remediationEngine = remediationEngine;
        global.healthMonitor = healthMonitor;

        logger.info('Sistema de Alertas inicializado com sucesso!');

        // Iniciar servidor
        const PORT = process.env.PORT || 3000;
        server.listen(PORT, () => {
            logger.info(`Dashboard disponível em: http://localhost:${PORT}`);
            logger.info(`API disponível em: http://localhost:${PORT}/api`);
        });

    } catch (error) {
        logger.error('Erro ao inicializar sistema:', error);
        process.exit(1);
    }
}

// Manipuladores de shutdown
process.on('SIGTERM', gracefulShutdown);

async function gracefulShutdown() {
    logger.info('Iniciando shutdown gracioso...');
    
    try {
        //
