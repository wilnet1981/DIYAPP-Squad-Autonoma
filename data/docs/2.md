```python
"""
Integração com API da LearnWorlds
- Autenticação
- Busca de usuários
- Busca de conteúdos
"""

import requests

class LearnWorldsAPI:
    def __init__(self):
        self.base_url = "https://vanusagrando.learnworlds.com/api/v2"
        self.headers = {
            "Authorization": "Bearer SEU_TOKEN_AQUI",  # Substituir pelo token real
            "Content-Type": "application/json"
        }

    def verificar_matricula(self, email: str, produto: str) -> bool:
        """Verifica se usuário está matriculado no produto"""
        endpoint = f"{self.base_url}/users"
        params = {'email': email}
        response = requests.get(endpoint, headers=self.headers, params=params)
        
        if response.status_code == 200:
            user_data = response.json()
            # Verifica se o usuário está matriculado no produto específico
            return produto in user_data.get('cursos_inscritos', [])
        return False

    def buscar_conteudo(self, email: str, pergunta: str) -> dict:
        """Busca conteúdo relacionado a pergunta do cliente"""
        # Implementação fictícia: lógica de tradução de pergunta para chamada de API
        mapping = {
            "cover letter": "curso_de_cover_letter",
            "módulo X": "curso_de_modulo_x"
        }
        curso_id = mapping.get(pergunta.lower())
        
        if curso_id:
            endpoint = f"{self.base_url}/courses/{curso_id}/content"
            response = requests.get(endpoint, headers=self.headers)
            if response.status_code == 200:
                return response.json()
        return {}
```

```python
        try:
            response = requests.get(
                f"{self.base_url}/users",

            return False
```
