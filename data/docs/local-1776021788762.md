// src/metrics/dashboard.js
class V12MetricsDashboard {
  constructor() {
    this.metrics = {
      webVitals: {
        lcp: { target: 2.5, current: null },
        fid: { target: 100, current: null },
        cls: { target: 0.1, current: null }
      },
      business: {
        nps: { target: 50, current: null },
        retention: { target: 40, current: null },
        uptime: { target: 99.9, current: null }
      }
    };
  }

  async collectMetrics() {
    // Coleta Core Web Vitals
    if (typeof window !== 'undefined') {
      import('web-vitals').then(({ getLCP, getFID, getCLS }) => {
        getLCP(this.setMetric('lcp'));
        getFID(this.setMetric('fid'));
        getCLS(this.setMetric('cls'));
      });
    }

    // Coleta métricas de negócio
    await this.fetchBusinessMetrics();
  }

  setMetric(type) {
    return (metric) => {
      this.metrics.webVitals[type].current = metric.value;
      this.checkThreshold(type, metric.value);
    };
  }

  async fetchBusinessMetrics() {
    const response = await fetch('/api/metrics/business');
    const data = await response.json();
    
    this.metrics.business.nps.current = data.nps;
    this.metrics.business.retention.current = data.retention;
    this.metrics.business.uptime.current = data.uptime;
  }

  checkThreshold(metric, value) {
    const config = this.metrics.webVitals[metric] || this.metrics.business[metric];
    if (!config) return;

    const threshold = config.target;
    const isBelowTarget = value > threshold;
    
    if (isBelowTarget) {
      this.triggerAlert(metric, value, threshold);
    }
  }

  triggerAlert(metric, value, threshold) {
    console.warn(`ALERTA: ${metric} = ${value} (meta: ${threshold})`);
    
    // Envia para sistema de alertas
    fetch('/api/alerts', {
      method: 'POST',
      body: JSON.stringify({
        metric,
        value,
        threshold,
        timestamp: new Date().toISOString()
      })
    });
  }

  getHealthScore() {
    let score = 100;
    let totalMetrics = 0;

    // Calcula score baseado em todas as métricas
    for (const category in this.metrics) {
      for (const metric in this.metrics[category]) {
        totalMetrics++;
        const config = this.metrics[category][metric];
        if (config.current === null) continue;

        const percentage = (config.current / config.target) * 100;
        if (percentage > 100) {
          score -= (percentage - 100) / 10;
        }
      }
    }

    return Math.max(0, Math.round(score));
  }
}

async function calculateNPS() {
  return new Promise((resolve, reject) => {
    db.all(`
      SELECT value 
      FROM metrics 
      WHERE metric_type = 'nps_score' 
      AND timestamp > datetime('now', '-30 days')
    `, (err, rows) => {
      if (err) reject(err);
      
      const scores = rows.map(r => r.value);
      if (scores.length === 0) resolve(0);
      
      const promoters = scores.filter(s => s >= 9).length;
      const detractors = scores.filter(s => s <= 6).length;
      const nps = ((promoters - detractors) / scores.length) * 100;
      
      resolve(Math.round(nps));
    });
  });
}

async function calculateRetention() {
  return new Promise((resolve, reject) => {
    db.all(`
      SELECT COUNT(DISTINCT user_id) as returned_users
      FROM user_sessions
      WHERE session_date BETWEEN datetime('now', '-7 days') AND datetime('now')
      AND user_id IN (
        SELECT DISTINCT user_id 
        FROM user_sessions 
        WHERE session_date BETWEEN datetime('now', '-14 days') AND datetime('now', '-7 days')
      )
    `, (err, rows) => {
      if (err) reject(err);
      
      const returned = rows[0]?.returned_users || 0;
      
      db.get(`
        SELECT COUNT(DISTINCT user_id) as total_users
        FROM user_sessions
        WHERE session_date BETWEEN datetime('now', '-14 days') AND datetime('now', '-7 days')
      `, (err, totalRow) => {
        if (err) reject(err);
        
        const total = totalRow?.total_users || 1;
        const retention = (returned / total) * 100;
        
        resolve(Math.round(retention));
      });
    });
  });
}

# DIYAPP Evolution - V12 Core - Arquitetura Técnica

## 1. ADR-001: Arquitetura Híbrida Microsserviços/Monólito

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

**CONTEXTO:**
DIYAPP precisa evoluir para suportar 100k+ usuários simultâneos com 99.9% uptime. O sistema atual é um monólito que enfrenta problemas de escalabilidade e deploy lento. Precisamos balancear velocidade de desenvolvimento com capacidade de escalar componentes críticos independentemente.

**DECISÃO:**
Adotar arquitetura híbrida "Monólito Modular com Microsserviços Estratégicos":
- Core: Monólito modular em Node.js (API Gateway + Serviços Core)
- Serviços Estratégicos: Microsserviços independentes para:
  - Chat/WebSocket (real-time)
  - Processamento de IA/ML
  - Upload/Processamento de arquivos
  - Notificações em tempo real

**OPÇÕES CONSIDERADAS:**
- Opção A: Microsserviços puros - Prós: Escalabilidade máxima, deploy independente. Contras: Complexidade operacional alta, latência de rede, desenvolvimento mais lento.
- Opção B: Monólito puro - Prós: Desenvolvimento rápido, debugging fácil. Contras: Escalabilidade limitada, deploy de todo sistema para mudanças pequenas.
- Opção C: Híbrida (escolhida) - Prós: Balanceia velocidade e escalabilidade, isola componentes críticos. Contras: Gerenciamento de duas arquiteturas.

**CONSEQUÊNCIAS:**
Positivas: Escalabilidade seletiva, deploy independente de serviços críticos, manutenção simplificada do core.
Negativas: Complexidade aumentada, necessidade de orquestração de serviços.
Riscos: Latência entre serviços - mitigado com cache Redis e connection pooling.

**REVISÃO:** 2024-04-15

```typescript
// apps/api/src/index.ts
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import compression from 'compression';
import { config } from './config';
import { errorHandler, notFoundHandler } from './middleware/error';
import { requestLogger } from './middleware/logger';
import apiRoutes from './routes';

const app = express();

// Middleware
app.use(helmet());
app.use(cors(config.cors));
app.use(compression());
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true }));
app.use(requestLogger);

// Health check
app.get('/health', (req, res) => {
  res.json({ 
    status: 'healthy', 
    timestamp: new Date().toISOString(),
    version: config.version 
  });
});

// API Routes
app.use('/api/v1', apiRoutes);

// Error handling
app.use(notFoundHandler);
app.use(errorHandler);

// Start server
const server = app.listen(config.port, () => {
  console.log(`🚀 API Server running on port ${config.port}`);
  console.log(`📊 Environment: ${config.nodeEnv}`);
  console.log(`🔗 Health check: http://localhost:${config.port}/health`);
});

// Graceful shutdown
process.on('SIGTERM', () => {
  console.log('SIGTERM received. Starting graceful shutdown...');
  server.close(() => {
    console.log('Server closed. Process terminated.');
    process.exit(0);
  });
});

export default app;
```

```typescript
import React from 'react';
import clsx from 'clsx';
import './Button.css';

export type ButtonVariant = 'primary' | 'secondary' | 'outline' | 'ghost' | 'danger';
export type ButtonSize = 'sm' | 'md' | 'lg';

export interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: ButtonVariant;
  size?: ButtonSize;
  isLoading?: boolean;
  leftIcon?: React.ReactNode;
  rightIcon?: React.ReactNode;
  fullWidth?: boolean;
}

const Button = React.forwardRef<HTMLButtonElement, ButtonProps>(
  (
    {
      children,
      variant = 'primary',
      size = 'md',
      isLoading = false,
      leftIcon,
      rightIcon,
      fullWidth = false,
      disabled,
      className,
      ...props
    },
    ref
  ) => {
    const isDisabled = disabled || isLoading;

    return (
      <button
        ref={ref}
        className={clsx(
          'button',
          `button--${variant}`,
          `button--${size}`,
          {
            'button--loading': isLoading,
            'button--full-width': fullWidth,
            'button--disabled': isDisabled,
          },
          className
        )}
        disabled={isDisabled}
        aria-busy={isLoading}
        {...props}
      >
        {isLoading && (
          <span className="button__loader" aria-hidden="true">
            <span className="button__loader-dot" />
            <span className="button__loader-dot" />
            <span className="button__loader-dot" />
          </span>
        )}
        
        {!isLoading && leftIcon && (
          <span className="button__icon button__icon--left" aria-hidden="true">
            {leftIcon}
          </span>
        )}
        
        <span className="button__content">
          {children}
        </span>
        
        {!isLoading && rightIcon && (
          <span className="button__icon button__icon--right" aria-hidden="true">
            {rightIcon}
          </span>
        )}
        
        {isLoading && (
          <span className="sr-only">Loading...</span>
        )}
      </button>
    );
  }
);

Button.displayName = 'Button';

export default Button;
```

### src/config/database.js
```javascript
const { Pool } = require('pg');
const config = require('./environment');
const logger = require('../utils/logger');

class Database {
  constructor() {
    this.pool = new Pool({
      host: config.database.host,
      port: config.database.port,
      database: config.database.name,
      user: config.database.user,
      password: config.database.password,
      max: config.database.maxConnections,
      idleTimeoutMillis: config.database.idleTimeout,
      connectionTimeoutMillis: config.database.connectionTimeout,
    });

    this.pool.on('error', (err) => {
      logger.error('Unexpected database pool error', {
        error: err.message,
        stack: err.stack,
      });
    });

    this.pool.on('connect', () => {
      logger.debug('New database connection established');
    });
  }

  async query(text, params) {
    const start = Date.now();
    try {
      const result = await this.pool.query(text, params);
      const duration = Date.now() - start;
      
      logger.debug('Database query executed', {
        query: text,
        params: this.maskSensitiveData(params),
        duration,
        rowCount: result.rowCount,
      });
      
      return result;
    } catch (error) {
      const duration = Date.now() - start;
      logger.error('Database query failed', {
        query: text,
        params: this.maskSensitiveData(params),
        duration,
        error: error.message,
        code: error.code,
      });
      throw error;
    }
  }

  async transaction(callback) {
    const client = await this.pool.connect();
    
    try {
      await client.query('BEGIN');
      const result = await callback(client);
      await client.query('COMMIT');
      return result;
    } catch (error) {
      await client.query('ROLLBACK');
      throw error;
    } finally {
      client.release();
    }
  }

  maskSensitiveData(params) {
    if (!params) return params;
    
    return params.map(param => {
      if (typeof param === 'string') {
        // Mask emails
        if (param.includes('@')) {
          const [local, domain] = param.split('@');
          return `${local.charAt(0)}***@${domain}`;
        }
```

# DIYAPP Evolution - V12 Core - Segurança

## 1. Threat Modeling STRIDE - V12 Architecture

### Assets Críticos:
1. **Dados de Usuário**: PII, credenciais, histórico de conversas
2. **Sistema de Agentes**: Configurações, prompts, comportamentos
3. **Integrações Externas**: WhatsApp, APIs de IA, bancos de dados
4. **Infraestrutura**: Servidores, containers, redes

### Análise STRIDE:

#### Spoofing (Falsificação):
- **Ameaça**: Ataque de força bruta em login
- **Controle**: MFA obrigatório, rate limiting, tokens JWT com expiração curta
- **Risco**: Médio → Mitigado com controles

#### Tampering (Manipulação):
- **Ameaça**: Manipulação de prompts de sistema via API
- **Controle**: Validação de schema, assinatura digital de prompts
- **Risco**: Alto → Requer validação rigorosa

#### Repudiation (Repúdio):
- **Ameaça**: Negação de ações administrativas
- **Controle**: Logs imutáveis com auditoria completa
- **Risco**: Baixo → Mitigado com logging

#### Information Disclosure (Divulgação):
- **Ameaça**: Vazamento de dados via logs ou erro messages
- **Controle**: Mascaramento de PII, criptografia em repouso
- **Risco**: Alto → Crítico para LGPD

#### Denial of Service (Negação):
- **Ameaça**: Ataque DDoS em endpoints de IA
- **Controle**: Rate limiting por IP/user, circuit breakers
- **Risco**: Médio → Mitigado com infra escalável

#### Elevation of Privilege (Elevação):
- **Ameaça**: Exploração de vulnerabilidade para acesso admin
- **Controle**: Principle of least privilege, verificações de autorização
- **Risco**: Alto → Monitoramento contínuo

### 2.2 Middleware de Autenticação e Autorização

**security/src/middleware/auth.js:**
```javascript
const jwt = require('jsonwebtoken');
const crypto = require('crypto');
const SecurityConfig = require('../config/securityConfig');
const AuditLogger = require('../utils/auditLogger');

class AuthMiddleware {
    constructor() {
        this.secretKey = process.env.JWT_SECRET || this.generateSecret();
        this.tokenExpiry = '15m';
        this.refreshExpiry = '7d';
        this.auditLogger = new AuditLogger();
    }
    
    generateSecret() {
        return crypto.randomBytes(64).toString('hex');
    }
    
    async authenticate(req, res, next) {
        try {
            const token = this.extractToken(req);
            
            if (!token) {
                this.auditLogger.log('AUTH_FAIL', {
                    ip: req.ip,
                    path: req.path,
                    reason: 'No token provided'
                });
                return res.status(401).json({ 
                    error: 'Authentication required',
                    code: 'AUTH_REQUIRED'
                });
            }
            
            const decoded = await this.verifyToken(token);
            
            // Verificar se o token foi revogado
            if (await this.isTokenRevoked(decoded.jti)) {
                this.auditLogger.log('AUTH_FAIL', {
                    userId: decoded.userId,
                    ip: req.ip,
                    reason: 'Token revoked'
                });
                return res.status(401).json({ 
                    error: 'Token revoked',
                    code: 'TOKEN_REVOKED'
                });
            }
            
            req.user = decoded;
            req.tokenId = decoded.jti;
            
            // Log de acesso bem-sucedido
            this.auditLogger.log('AUTH_SUCCESS', {
                userId: decoded.userId,
                ip: req.ip,
                path: req.path
            });
            
            next();
        } catch (error) {
            this.auditLogger.log('AUTH_FAIL', {
                ip: req.ip,
                path: req.path,
                reason: error.message
            });
            
            return res.status(401).json({ 
                error: 'Invalid token',
                code: 'INVALID_TOKEN'
            });
        }
    }
    
    async authorize(requiredPermissions = []) {
        return async (req, res, next) => {
            try {
                const userPermissions = req.user?.permissions || [];
                
                // Verificar se o usuário tem todas as permissões necessárias
                const hasPermission = requiredPermissions.every(perm => 
                    userPermissions.includes(perm)
                );
                
                if (!hasPermission) {
                    this.auditLogger.log('AUTHZ_FAIL', {
                        userId: req.user?.userId,
                        ip: req.ip,
                        path: req.path,
                        required: requiredPermissions,
                        has: userPermissions
                    });
                    
                    return res.status(403).json({ 
                        error: 'Insufficient permissions',
                        code: 'INSUFFICIENT_PERMISSIONS'
                    });
                }
                
                next();
            } catch (error) {
                this.auditLogger.log('AUTHZ_ERROR', {
                    error: error.message,
                    path: req.path
                });
                
                return res.status(500).json({ 
                    error: 'Authorization error',
                    code: 'AUTHZ_ERROR'
                });
            }
        };
    }
    
    async generateTokens(userId, permissions = []) {
        const tokenId = crypto.randomBytes(16).toString('hex');
        const accessToken = jwt.sign(
            { 
                userId, 
                permissions,
                jti: tokenId,
                type: 'access'
            },
            this.secretKey,
            { expiresIn: this.tokenExpiry }
        );
        
        const refreshToken = jwt.sign(
            { 
                userId, 
                jti: tokenId,
                type: 'refresh'
            },
            this.secretKey,
            { expiresIn: this.refreshExpiry }
        );
        
        // Armazenar token no registro de tokens ativos
        await this.storeToken(tokenId, userId, 'active');
        
        return { accessToken, refreshToken, tokenId };
    }
    
    extractToken(req) {
        const authHeader = req.headers.authorization;
        if (authHeader && authHeader.startsWith('Bearer ')) {
            return authHeader.substring(7);
        }
        return req.cookies?.access_token || req.query?.token;
    }
    
    verifyToken(token) {
        return new Promise((resolve, reject) => {
            jwt.verify(token, this.secretKey, (err, decoded) => {
                if (err) reject(err);
                else resolve(decoded);
            });
        });
    }
    
    async storeToken(tokenId, userId, status) {
        // Implementar armazenamento em Redis ou banco de dados
        const tokenData = {
            tokenId,
            userId,
            status,
            createdAt: new Date().toISOString(),
            lastUsed: new Date().toISOString()
        };
        
        // Simulação - em produção usar Redis
        if (global.tokenStore) {
            global.tokenStore[tokenId] = tokenData;
        }
    }
    
    async isTokenRevoked(tokenId) {
        // Verificar no Redis/banco de dados
        if (global.tokenStore && global.tokenStore[tokenId]) {
            return global.tokenStore[tokenId].status === 'revoked';
        }
        return false;
    }
    
    async revokeToken(tokenId) {
        if (global.tokenStore && global.tokenStore[tokenId]) {
            global.tokenStore[tokenId].status = 'revoked';
            global.tokenStore[tokenId].revokedAt = new Date().toISOString();
            return true;
        }
        return false;
    }
    
    async requireMFA(req, res, next) {
        if (req.user?.mfaEnabled && !req.session?.mfaVerified) {
```

```javascript
class WAFMiddleware {
    constructor() {
        this.securityConfig = SecurityConfig;
        this.auditLogger = new AuditLogger();
        this.blockedIPs = new Map();
        this.suspiciousPatterns = this.compilePatterns();
    }
    
    compilePatterns() {
        return {
            sqlInjection: new RegExp(
                this.securityConfig.OWASP_RULES.SQL_INJECTION_PATTERNS
                    .map(p => p.source)
                    .join('|'),
                'gi'
            ),
            xss: new RegExp(
                this.securityConfig.OWASP_RULES.XSS_PATTERNS
                    .map(p => p.source)
                    .join('|'),
                'gi'
            ),
            pathTraversal: /(\.\.\/|\.\.\\|\.\.%2f|\.\.%5c)/gi,
            commandInjection: /(\||\&|\;|\`|\$\(|\n|\r)/gi,
            nosqlInjection: /(\$where|\$ne|\$gt|\$lt|\$regex)/gi
        };
    }
    
    async checkRequest(req, res, next) {
        const clientIP = req.ip || req.connection.remoteAddress;
        
        // Verificar se IP está bloqueado
        if (this.isIPBlocked(clientIP)) {
            this.auditLogger.log('WAF_BLOCK_IP', {
                ip: clientIP,
                reason: 'IP blocked',
                path: req.path
            });
            return res.status(403).json({
                error: 'Access denied',
                code: 'IP_BLOCKED'
            });
        }
        
        // Analisar query parameters
        if (req.query && this.detectAttack(req.query, 'query')) {
            this.blockIP(clientIP, 'Malicious query parameters');
            return this.blockResponse(res, clientIP, 'SQL Injection attempt');
        }
        
        // Analisar body
        if (req.body && this.detectAttack(req.body, 'body')) {
            this.blockIP(clientIP, 'Malicious request body');
            return this.blockResponse(res, clientIP, 'XSS attempt');
        }
        
        // Analisar headers
        if (this.detectAttack(req.headers, 'headers')) {
            this.blockIP(clientIP, 'Malicious headers');
            return this.blockResponse(res, clientIP, 'Header injection attempt');
        }
        
        // Analisar URL path
        if (this.detectAttack({ path: req.path }, 'path')) {
            this.blockIP(clientIP, 'Path traversal attempt');
            return this.blockResponse(res, clientIP, 'Path traversal');
        }
        
        next();
    }
    
    detectAttack(data, context) {
        const dataString = JSON.stringify(data).toLowerCase();
        
        for (const [attackType, pattern] of Object.entries(this.suspiciousPatterns)) {
            if (pattern.test(dataString)) {
                this.auditLogger.log('WAF_DETECT', {
                    attackType,
                    context,
                    data: dataString.substring(0, 500) // Log parcial por segurança
                });
                return true;
            }
        }
        
        return false;
    }
    
    isIPBlocked(ip) {
        const blockInfo = this.blockedIPs.get(ip);
        if (!blockInfo) return false;
        
        // Verificar se o bloqueio expirou
        if (Date.now() > blockInfo.expiresAt) {
            this.blockedIPs.delete(ip);
            return false;
        }
        
        return true;
    }
    
    blockIP(ip, reason, duration = 3600000) { // 1 hora padrão
        this.blockedIPs.set(ip, {
            reason,
            blockedAt: Date.now(),
            expiresAt: Date.now() + duration
        });
        
        this.auditLogger.log('WAF_BLOCK', {
            ip,
            reason,
            duration
        });
    }
    
    blockResponse(res, ip, reason) {
        this.auditLogger.log('WAF_BLOCK_RESPONSE', {
            ip,
            reason,
            timestamp: new Date().toISOString()
        });
        
        return res.status(403).json({
            error: 'Request blocked by security filter',
            code: 'REQUEST_BLOCKED',
            reference: `SEC-${Date.now()}`
        });
    }
    
    getSecurityHeaders() {
        return (req, res, next) => {
            Object.entries(this.securityConfig.SECURITY_HEADERS).forEach(([header, value]) => {
                res.setHeader(header, value);
            });
            next();
        };
    }
```

```sql
import psycopg2
from psycopg2.extensions import ISOLATION_LEVEL_AUTOCOMMIT
import logging
from datetime import datetime, timedelta
import json

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class DatabaseOptimizer:
    def __init__(self, db_params):
        self.db_params = db_params
        self.conn = None
        
    def connect(self):
        """Estabelece conexão com o banco"""
        try:
            self.conn = psycopg2.connect(**self.db_params)
            self.conn.set_isolation_level(ISOLATION_LEVEL_AUTOCOMMIT)
            logger.info("Conexão estabelecida com sucesso")
            return True
        except Exception as e:
            logger.error(f"Erro na conexão: {e}")
            return False
    
    def create_schema(self):
        """Cria schema otimizado com particionamento e indexação"""
        commands = [
            # Schema principal
            "CREATE SCHEMA IF NOT EXISTS diyapp;",
            "CREATE SCHEMA IF NOT EXISTS analytics;",
            
            # Tabela de usuários com particionamento por tenant
            """
            CREATE TABLE IF NOT EXISTS diyapp.users (
                user_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
                tenant_id VARCHAR(50) NOT NULL,
                email VARCHAR(255) NOT NULL,
                username VARCHAR(100) NOT NULL,
                created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                status VARCHAR(20) DEFAULT 'active',
                metadata JSONB,
                CONSTRAINT unique_user_email UNIQUE(tenant_id, email)
            ) PARTITION BY LIST (tenant_id);
            """,
            
            # Partições para tenants
            "CREATE TABLE IF NOT EXISTS diyapp.users_tenant_default PARTITION OF diyapp.users FOR VALUES IN ('default');",
            "CREATE TABLE IF NOT EXISTS diyapp.users_tenant_enterprise PARTITION OF diyapp.users FOR VALUES IN ('enterprise');",
            
            # Tabela de eventos com particionamento por data
            """
            CREATE TABLE IF NOT EXISTS diyapp.events (
                event_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
                user_id UUID REFERENCES diyapp.users(user_id),
                event_type VARCHAR(100) NOT NULL,
                event_name VARCHAR(200) NOT NULL,
                properties JSONB,
                session_id VARCHAR(100),
                device_info JSONB,
                created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                tenant_id VARCHAR(50) NOT NULL
            ) PARTITION BY RANGE (created_at);
            """,
            
            # Partições mensais para eventos (últimos 12 meses + futuro)
            """
            CREATE TABLE IF NOT EXISTS diyapp.events_2024_01 
            PARTITION OF diyapp.events 
            FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');
            """,
            
            """
            CREATE TABLE IF NOT EXISTS diyapp.events_2024_02 
            PARTITION OF diyapp.events 
            FOR VALUES FROM ('2024-02-01') TO ('2024-03-01');
            """,
            
            # Tabela de transações
            """
            CREATE TABLE IF NOT EXISTS diyapp.transactions (
                transaction_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
                user_id UUID REFERENCES diyapp.users(user_id),
                amount DECIMAL(10,2) NOT NULL,
                currency VARCHAR(3) DEFAULT 'USD',
                status VARCHAR(20) NOT NULL,
                payment_method VARCHAR(50),
                metadata JSONB,
                created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                tenant_id VARCHAR(50) NOT NULL
            ) PARTITION BY RANGE (created_at);
            """,
            
            # Tabela de logs do sistema
            """
            CREATE TABLE IF NOT EXISTS diyapp.system_logs (
                log_id BIGSERIAL PRIMARY KEY,
                log_level VARCHAR(20) NOT NULL,
                service_name VARCHAR(100) NOT NULL,
                message TEXT NOT NULL,
                context JSONB,
                created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                INDEX idx_logs_service_created (service_name, created_at)
            ) PARTITION BY RANGE (created_at);
            """,
            
            # Tabela de configurações
            """
            CREATE TABLE IF NOT EXISTS diyapp.configurations (
                config_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

                is_active BOOLEAN DEFAULT TRUE,
                created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
                expires_at TIMESTAMP WITH TIME ZONE

            cursor = self.conn.cursor()

            raise
    
    def create_indexes(self):
        """Cria índices otimizados para performance"""
        indexes = [
            # Índices para tabela users
            "CREATE INDEX IF NOT EXISTS idx_users_tenant_status ON diyapp.users(tenant_id, status);",
            "CREATE INDEX IF NOT EXISTS idx_users_created_at ON diyapp.users(created_at);",
            "CREATE INDEX IF NOT EXISTS idx_users_email ON diyapp.users(email);",
            
            # Índices para tabela events (por partição)
            "CREATE INDEX IF NOT EXISTS idx_events_user_created ON diyapp.events(user_id, created_at);",
            "CREATE INDEX IF NOT EXISTS idx_events_type_created ON diyapp.events(event_type, created_at);",
            "CREATE INDEX IF NOT EXISTS idx_events_tenant_created ON diyapp.events(tenant_id, created_at);",
            "CREATE INDEX IF NOT EXISTS idx_events_session ON diyapp.events(session_id);",
            "CREATE INDEX IF NOT EXISTS idx_events_properties_gin ON diyapp.events USING GIN(properties);",
            
            # Índices para tabela transactions
            "CREATE INDEX IF NOT EXISTS idx_transactions_user_status ON diyapp.transactions(user_id, status);",
            "CREATE INDEX IF NOT EXISTS idx_transactions_created_amount ON diyapp.transactions(created_at, amount);",
            "CREATE INDEX IF NOT EXISTS idx_transactions_tenant_created ON diyapp.transactions(tenant_id, created_at);",
            
            # Índices para tabela system_logs
            "CREATE INDEX IF NOT EXISTS idx_system_logs_level_created ON diyapp.system_logs(log_level, created_at);",
            "CREATE INDEX IF NOT EXISTS idx_system_logs_created_at ON diyapp.system_logs(created_at DESC);"
        ]
        
        try:
            cursor = self.conn.cursor()
            for index in indexes:
                cursor.execute(index)
            cursor.close()
            logger.info("Índices criados com sucesso")
        except Exception as e:
            logger.error(f"Erro ao criar índices: {e}")
            raise
    
    def create_partitions_for_next_months(self, months=6):
        """Cria partições futuras automaticamente"""
        cursor = self.conn.cursor()
        today = datetime.now()
        
        for i in range(months):
            month_date = today + timedelta(days=30*i)
            year = month_date.year
            month = month_date.month
            
            # Partições para eventos
            partition_name = f"events_{year}_{month:02d}"
            start_date = f"{year}-{month:02d}-01"
            
            if month == 12:
                end_date = f"{year+1}-01-01"
            else:
                end_date = f"{year}-{month+1:02d}-01"
            
            create_partition_sql = f"""
            CREATE TABLE IF NOT EXISTS diyapp.{partition_name}
            PARTITION OF diyapp.events
            FOR VALUES FROM ('{start_date}') TO ('{end_date}');
            """
            
            try:
                cursor.execute(create_partition_sql)
                logger.info(f"Partição criada: {partition_name}")
            except Exception as e:
                logger.warning(f"Partição {partition_name} já existe ou erro: {e}")
        
        cursor.close()
    
    def setup_retention_policies(self):

        policies = [

                WHERE created_at < NOW() - INTERVAL '90 days';
            END;
            $$ LANGUAGE plpgsql;
            """,
            
            # Retenção de eventos após 2 anos (mover para cold storage)
            """
            CREATE OR REPLACE FUNCTION diyapp.archive_old_events()
            RETURNS void AS $$
            BEGIN
                -- Implementar lógica de archive para S3/Glacier

                WHERE created_at < NOW() - INTERVAL '2 years';
            END;
            $$ LANGUAGE plpgsql;
            """
        ]
        
        try:
            cursor = self.conn.cursor()
            for policy in policies:
                cursor.execute(policy)
            cursor.close()
            logger.info("Políticas de retenção configuradas")
        except Exception as e:
            logger.error(f"Erro ao configurar políticas: {e}")
```

# DIYAPP Evolution - V12 Core - Segurança

## 1. Threat Modeling STRIDE - V12 Core

### Ativos Críticos Identificados:
1. **Dados de Usuário**: PII, credenciais, tokens de sessão
2. **Sistema de Autenticação**: OAuth 2.0, JWT, refresh tokens
3. **APIs de IA**: LLM endpoints, prompts do sistema, contextos de usuário
4. **Banco de Dados**: PostgreSQL com dados sensíveis
5. **Integrações Externas**: WhatsApp API, serviços de terceiros
6. **Arquivos de Configuração**: Secrets, chaves de API, variáveis de ambiente

### Análise STRIDE:

#### **Spoofing (Falsificação)**
- **Ameaça**: Ataque de força bruta em login, token JWT roubado, OAuth redirect hijacking
- **Controles**: 
  - MFA obrigatório para contas administrativas
  - Rate limiting em endpoints de autenticação (10 tentativas/hora)
  - Validação de origem em callbacks OAuth
  - Tokens JWT com expiração curta (15min access, 7d refresh)

#### **Tampering (Manipulação)**
- **Ameaça**: Manipulação de requests API, SQL injection, NoSQL injection
- **Controles**:
  - Validação de schema com Zod em todas as APIs
  - Prepared statements/ORM para SQL
  - Assinatura digital em cookies sensíveis
  - WAF com regras OWASP Core Rule Set

#### **Repudiation (Repúdio)**
- **Ameaça**: Negação de ações administrativas, falta de logs de auditoria
- **Controles**:
  - Logs imutáveis em CloudWatch/Splunk
  - Auditoria de todas as operações CRUD em dados sensíveis
  - Assinatura de logs com hash chain

#### **Information Disclosure (Divulgação)**
- **Ameaça**: Vazamento de dados via erro messages, logs, API responses
- **Controles**:
  - PII criptografado em repouso (AES-256-GCM)
  - Mascaramento de dados em logs e respostas de erro
  - Headers de segurança (HSTS, CSP, X-Frame-Options)
  - Sanitização de outputs LLM

#### **Denial of Service (Negação)**
- **Ameaça**: Ataques DDoS, resource exhaustion, rate limiting bypass
- **Controles**:
  - WAF com rate limiting por IP/usuário
  - Circuit breakers em chamadas externas
  - Quotas de API por plano de usuário
  - Auto-scaling com limites máximos

#### **Elevation of Privilege (Elevação)**
- **Ameaça**: IDOR, JWT tampering, bypass de autorização
- **Controles**:
  - RBAC com verificação por recurso
  - Validação de escopo em tokens OAuth
  - Princípio do menor privilégio em todas as permissões
  - Revisão periódica de permissões

**src/security/auth/middleware.ts**
```typescript
import jwt from 'jsonwebtoken';
import { Request, Response, NextFunction } from 'express';
import { SecretsManager } from '../encryption/secrets-manager';
import { auditLogger } from '../../monitoring/audit-logger';

export interface AuthUser {
  id: string;
  email: string;
  roles: string[];
  permissions: string[];
  mfaEnabled: boolean;
  sessionId: string;
}

declare global {
  namespace Express {
    interface Request {
      user?: AuthUser;
    }
  }
}

export class AuthMiddleware {
  private static secretsManager = new SecretsManager();
  private static tokenBlacklist = new Set<string>();

  // Middleware de autenticação JWT
  static async authenticate(req: Request, res: Response, next: NextFunction) {
    const authHeader = req.headers.authorization;
    
    if (!authHeader?.startsWith('Bearer ')) {
      return res.status(401).json({ error: 'No token provided' });
    }

    const token = authHeader.substring(7);
    
    // Verifica se token está na blacklist
    if (this.tokenBlacklist.has(token)) {
      auditLogger.logBlacklistedToken(req.ip);
      return res.status(401).json({ error: 'Token revoked' });
    }

    try {
      const secret = await this.secretsManager.getSecret('JWT_SECRET');
      const decoded = jwt.verify(token, secret) as AuthUser;
      
      // Verifica se o token está perto de expirar
      const now = Math.floor(Date.now() / 1000);
      const decodedAny = decoded as any;
      
      if (decodedAny.exp && decodedAny.exp - now < 300) { // 5 minutos
        // Token perto de expirar, adiciona header para refresh
        res.setHeader('X-Token-Near-Expiry', 'true');
      }
      
      req.user = decoded;
      next();
    } catch (error) {
      auditLogger.logInvalidToken(req.ip, error.message);
      
      if (error.name === 'TokenExpiredError') {
        return res.status(401).json({ error: 'Token expired' });
      }
      
      return res.status(401).json({ error: 'Invalid token' });
    }
  }

  // Middleware de autorização RBAC
  static authorize(requiredPermissions: string[] | string, requireAll: boolean = true) {
    return (req: Request, res: Response, next: NextFunction) => {
      if (!req.user) {
        return res.status(401).json({ error: 'Unauthorized' });
      }

      const permissions = Array.isArray(requiredPermissions) 
        ? requiredPermissions 
        : [requiredPermissions];

      const hasPermission = requireAll
        ? permissions.every(p => req.user!.permissions.includes(p))
        : permissions.some(p => req.user!.permissions.includes(p));

      if (!hasPermission) {
        auditLogger.logUnauthorizedAction(req.user.id, permissions);
        return res.status(403).json({ 
          error: 'Insufficient permissions',
          required: permissions
        });
      }

      next();
    };
  }

  // Verificação de MFA
  static requireMFA(req: Request, res: Response, next: NextFunction) {
    if (!req.user) {
      return res.status(401).json({ error: 'Unauthorized' });
    }

    if (req.user.mfaEnabled && !req.headers['x-mfa-verified']) {
      return res.status(403).json({ 
        error: 'MFA verification required',
        code: 'MFA_REQUIRED'
      });
    }

    next();
  }

  // Logout e invalidação de token
  static async logout(req: Request, res: Response) {
    const token = req.headers.authorization?.substring(7);
    
    if (token) {
      // Adiciona token à blacklist
      this.tokenBlacklist.add(token);
      
      // Agenda remoção da blacklist após expiração
      setTimeout(() => {
        this.tokenBlacklist.delete(token);
      }, 15 * 60 * 1000); // 15 minutos
    }
    
    auditLogger.logLogout(req.user?.id);
    res.json({ message: 'Logged out successfully' });
  }

  // Geração segura de tokens
  static async generateTokens(user: Omit<AuthUser, 'sessionId'>) {
    const accessSecret = await this.secretsManager.getSecret('JWT_ACCESS_SECRET');
    const refreshSecret = await this.secretsManager.getSecret('JWT_REFRESH_SECRET');
    const sessionId = crypto.randomUUID();

    const accessToken = jwt.sign(
      { ...user, sessionId },
      accessSecret,
      { expiresIn: '15m' }
    );

    const refreshToken = jwt.sign(
      { userId: user.id, sessionId },
      refreshSecret,
      { expiresIn: '7d' }
    );

    return { accessToken, refreshToken, sessionId };
  }
}
```
