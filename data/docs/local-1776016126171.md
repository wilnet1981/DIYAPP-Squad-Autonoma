## 2. ADR-001: Arquitetura de Microsserviços

**docs/architecture/ADR-001-service-boundaries.md**
```markdown
# ADR-001: Arquitetura de Microsserviços Baseada em Domínios

## Status
Aceito

## Data
2024-01-15

## Contexto
A V10 do DIYAPP enfrenta problemas de escalabilidade devido à arquitetura monolítica. 
Necessitamos de:
1. Deploy independente por funcionalidade
2. Escala horizontal seletiva
3. Isolamento de falhas
4. Time-to-market reduzido

## Decisão
Adotar arquitetura de microsserviços com os seguintes princípios:
- Cada serviço é responsável por um domínio de negócio específico
- Comunicação via API REST (síncrona) e eventos (assíncrona)
- Banco de dados por serviço (Database per Service)
- API Gateway como ponto único de entrada

## Opções Consideradas
1. **Monolito modularizado**: Mais simples, mas não resolve problemas de escalabilidade
2. **Microsserviços leves (6-8 serviços)**: Balance entre complexidade e benefícios
3. **Nanosserviços (15+ serviços)**: Muito complexo para nossa equipe atual

**Escolha: Opção 2** - 6 serviços iniciais com possibilidade de divisão futura

## Consequências
### Positivas
- Deploy independente
- Escala por demanda
- Stack tecnológica adequada por serviço
- Isolamento de falhas

### Negativas
- Complexidade operacional aumentada
- Necessidade de orquestração (Kubernetes)
- Monitoramento distribuído necessário
- Transações distribuídas complexas

### Riscos
1. Latência de rede entre serviços
   - Mitigação: Cache estratégico, P95 < 200ms
2. Consistência eventual
   - Mitigação: Sagas para transações complexas
3. Observabilidade
   - Mitigação: OpenTelemetry + Grafana stack

## Revisão
Reavaliar em 6 meses ou quando atingir 50k usuários ativos
```

```typescript
// shared/libs/logger/src/logger.ts
import winston from 'winston';
import { AsyncLocalStorage } from 'async_hooks';

const asyncLocalStorage = new AsyncLocalStorage<Map<string, string>>();

export class Logger {
  private logger: winston.Logger;

  constructor(serviceName: string) {
    this.logger = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.json()
      ),
      defaultMeta: { service: serviceName },
      transports: [
        new winston.transports.Console(),
        new winston.transports.File({ 
          filename: `logs/${serviceName}-error.log`, 
          level: 'error' 
        }),
        new winston.transports.File({ 
          filename: `logs/${serviceName}-combined.log` 
        }),
      ],
    });
  }

  private getContext(): Record<string, string> {
    const store = asyncLocalStorage.getStore();
    const context: Record<string, string> = {};
    
    if (store) {
      for (const [key, value] of store) {
        context[key] = value;
      }
    }
    
    return context;
  }

  runWithContext<T>(context: Record<string, string>, fn: () => T): T {
    const store = new Map(Object.entries(context));
    return asyncLocalStorage.run(store, fn);
  }

  info(message: string, meta?: Record<string, any>) {
    this.logger.info(message, { 
      ...meta, 
      context: this.getContext() 
    });
  }

  error(message: string, error?: Error, meta?: Record<string, any>) {
    this.logger.error(message, {
      error: {
        message: error?.message,
        stack: error?.stack,
        name: error?.name
      },
      ...meta,
      context: this.getContext()
    });
  }

  warn(message: string, meta?: Record<string, any>) {
    this.logger.warn(message, { 
      ...meta, 
      context: this.getContext() 
    });
  }

  debug(message: string, meta?: Record<string, any>) {
    this.logger.debug(message, { 
      ...meta, 
      context: this.getContext() 
    });
  }
}

// shared/libs/metrics/src/metrics.ts
import client from 'prom-client';

export class Metrics {
  private static registry = new client.Registry();
  
  static httpRequestDuration = new client.Histogram({
    name: 'http_request_duration_seconds',
    help: 'Duration of HTTP requests in seconds',
    labelNames: ['method', 'route', 'status_code', 'service'],
    buckets: [0.1, 0.5, 1, 2, 5]
  });

  static databaseQueryDuration = new client.Histogram({
    name: 'database_query_duration_seconds',
    help: 'Duration of database queries in seconds',
    labelNames: ['operation', 'table', 'service'],
    buckets: [0.01, 0.05, 0.1, 0.5, 1]
  });

  static activeConnections = new client.Gauge({
    name: 'active_connections',
    help: 'Number of active connections',
    labelNames: ['service']
  });

  static businessEvents = new client.Counter({
    name: 'business_events_total',
    help: 'Total business events',
    labelNames: ['event_type', 'service']
  });

  static errors = new client.Counter({
    name: 'errors_total',
    help: 'Total errors',
    labelNames: ['error_type', 'service']
  });

  static initialize(serviceName: string) {
    this.registry.setDefaultLabels({ service: serviceName });
    
    [
      this.httpRequestDuration,
      this.databaseQueryDuration,
      this.activeConnections,
      this.businessEvents,
      this.errors
    ].forEach(metric => this.registry.registerMetric(metric));

    return this.registry;
  }
}
```

```typescript
import express from 'express';
import helmet from 'helmet';
import cors from 'cors';
import { Logger } from '@diyapp/shared/logger';
import { Metrics } from '@diyapp/shared/metrics';
import { errorHandler } from '@diyapp/shared/errors';
import { config } from './config';
import { userRoutes } from './routes/user.routes';
import { healthRoutes } from './routes/health.routes';
import { metricsMiddleware } from './middleware/metrics.middleware';
import { requestIdMiddleware } from './middleware/request-id.middleware';
import { contextMiddleware } from './middleware/context.middleware';

const logger = new Logger('user-service');
const app = express();
const port = config.port;

// Security middleware
app.use(helmet());
app.use(cors(config.cors));

// Observability middleware
app.use(requestIdMiddleware);
app.use(contextMiddleware);
app.use(metricsMiddleware);

// Body parsing
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true }));

// Routes
app.use('/health', healthRoutes);
app.use('/api/v1/users', userRoutes);

// Error handling (must be last)
app.use(errorHandler);

// Graceful shutdown
process.on('SIGTERM', () => {
  logger.info('Received SIGTERM, starting graceful shutdown');
  // Close database connections, etc.
  process.exit(0);
});

process.on('SIGINT', () => {
  logger.info('Received SIGINT, starting graceful shutdown');
  process.exit(0);
});

// Start server
if (require.main === module) {
  app.listen(port, () => {
    logger.info(`User service started on port ${port}`);
    logger.info(`Environment: ${config.nodeEnv}`);
    logger.info(`Metrics available at http://localhost:${port}/metrics`);
  });
}

export { app };
```

```yaml
name: CI Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]

jobs:
  lint-and-test:
    name: Lint & Test
    runs-on: ubuntu-latest
    
    strategy:
      matrix:
        service: [api-gateway, auth-service, user-service, project-service, notification-service, ai-orchestrator]
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Setup Node.js
      uses: actions/setup-node@v3
      with:
        node-version: '18'
        cache: 'npm'
        cache-dependency-path: services/${{ matrix.service }}/package-lock.json
    
    - name: Install Dependencies
      working-directory: services/${{ matrix.service }}
      run: npm ci
    
    - name: Lint
      working-directory: services/${{ matrix.service }}
      run: npm run lint
    
    - name: Type Check
      working-directory: services/${{ matrix.service }}
      run: npm run type-check
    
    - name: Run Tests
      working-directory: services/${{ matrix.service }}
      run: npm test -- --coverage
      env:
        NODE_ENV: test
        DB_HOST: localhost
        DB_PORT: 5432
        DB_USERNAME: test
        DB_PASSWORD: test
        DB_NAME: test_${{ matrix.service }}
    
    - name: Upload Coverage
      uses: codecov/codecov-action@v3
      with:
        file: services/${{ matrix.service }}/coverage/lcov.info
        flags: ${{ matrix.service }}
    
    - name: Build
      working-directory: services/${{ matrix.service }}
      run: npm run build
  
  security-scan:
    name: Security Scan
    runs-on: ubuntu-latest
    needs: lint-and-test
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Run Snyk to check for vulnerabilities
      uses: snyk/actions/node@master
      env:
        SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
      with:
        args: --severity-threshold=high
    
    - name: Run OWASP Dependency Check
      uses: dependency-check/Dependency-Check_Action@main
      with:
        project: 'DIYAPP-V11'
        path: '.'
        format: 'HTML'
        out: 'reports'
    
    - name: Upload Security Report
      uses: actions/upload-artifact@v3
      with:
        name: security-report
        path: reports/
  
  performance-test:
    name: Performance Test
    runs-on: ubuntu-latest
    needs: lint-and-test
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Setup k6
      uses: grafana/setup-k6-action@v1
    
    - name: Run Performance Tests
      run: |
        k6 run --out json=test-results.json \
          --summary-export=summary.json \
          scripts/performance/load-test.js
    
    - name: Check P95 Latency
      run: |
        P95=$(jq '.metrics.http_req_duration."p(95)"' summary.json)
        if (( $(echo "$P95 > 200" | bc -l) )); then
          echo "P95 latency $P95 exceeds 200ms threshold"
          exit 1
        fi
    
    - name: Upload Performance Results
      uses: actions/upload-artifact@v3
      with:
        name: performance-results
        path: |
          test-results.json
          summary.json
```

-- Staging layer indexes (para transformações eficientes)
CREATE INDEX idx_stg_events_date_hour ON staging.events(date, hour);

CREATE INDEX idx_stg_users_last_seen ON staging.users(last_seen_at);

  - Rate limiting em endpoints de autenticação (10 tentativas/hora)
  - Tokens JWT com expiração curta (15min access, 7d refresh)
  - Validação de origem em webhooks (assinatura HMAC)

#### **Tampering (Manipulação)**
- **Ameaça**: Manipulação de dados em trânsito, SQL injection, prompt injection
- **Controles Implementados**:
  - TLS 1.3 obrigatório
  - Validação de schema em todas as entradas (Zod)
  - Prepared statements para SQL
  - Sanitização de prompts LLM

#### **Repudiation (Repúdio)**
- **Ameaça**: Negação de ações administrativas, operações críticas sem log
- **Controles Implementados**:
  - Logs de auditoria imutáveis (Winston + CloudWatch)
  - Assinatura digital em logs críticos
  - Session tracking com correlation IDs

#### **Information Disclosure (Divulgação)**
- **Ameaça**: Vazamento de PII, secrets no código, erros com stack trace
- **Controles Implementados**:
  - Criptografia AES-256 para PII em repouso
  - Secrets em AWS Secrets Manager
  - Mascaramento de dados em logs

#### **Denial of Service (Negação)**
- **Ameaça**: Ataque DDoS, resource exhaustion, rate limit bypass
- **Controles Implementados**:
  - WAF com regras OWASP

  - Auto-scaling com limites

- **Ameaça**: Acesso não autorizado a admin, IDOR, LLM jailbreak
- **Controles Implementados**:
  - RBAC com princípio do menor privilégio

```javascript
    this.secretsCache.set(secretName, {
      value: secretValue,
      timestamp: Date.now()
    });

    return secretValue;
  }

    // In production, use KMS or similar
    // This is a simplified version for demonstration
    const algorithm = 'aes-256-gcm';
    const key = crypto.scryptSync(process.env.ENCRYPTION_KEY || 'default-key', 'salt', 32);

    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    return {
      iv: iv.toString('hex'),
      encrypted: encrypted,
      authTag: authTag.toString('hex')
    };
  }

  decryptSecret(encryptedData) {
    if (typeof encryptedData === 'string') {
      // Simple XOR for development (not secure for production!)
      return encryptedData.split('').map(char => 
        String.fromCharCode(char.charCodeAt(0) ^ 0x42)
      ).join('');
    }
    
    // Proper decryption for production format
    const algorithm = 'aes-256-gcm';
    const key = crypto.scryptSync(process.env.ENCRYPTION_KEY, 'salt', 32);
    const iv = Buffer.from(encryptedData.iv, 'hex');
    
    const decipher = crypto.createDecipheriv(algorithm, key, iv);
    decipher.setAuthTag(Buffer.from(encryptedData.authTag, 'hex'));
    
    let decrypted = decipher.update(encryptedData.encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return JSON.parse(decrypted);
  }
```

from sqlalchemy.orm import Session
from sqlalchemy import func, desc, and_, or_
from datetime import datetime, timedelta
import pytz
from typing import List, Optional, Dict, Any
from . import models

class CRUD:
    def __init__(self, db: Session):
        self.db = db
    
    # LLM Calls operations
    def create_llm_call(self, call_data: Dict[str, Any]) -> models.LLMCall:
        db_call = models.LLMCall(**call_data)
        self.db.add(db_call)
        self.db.commit()
        self.db.refresh(db_call)
        return db_call
    
    def get_calls_by_time_range(
        self, 
        start_time: datetime, 
        end_time: datetime,
        provider_id: Optional[int] = None,
        feature_id: Optional[int] = None
    ) -> List[models.LLMCall]:
        query = self.db.query(models.LLMCall).filter(
            models.LLMCall.created_at.between(start_time, end_time)
        )
        
        if provider_id:
            query = query.filter(models.LLMCall.provider_id == provider_id)
        
        if feature_id:
            query = query.filter(models.LLMCall.feature_id == feature_id)
        
        return query.order_by(models.LLMCall.created_at.desc()).all()
    
    def get_cost_by_feature(
        self, 
        start_time: datetime, 
        end_time: datetime
    ) -> List[Dict[str, Any]]:
        result = self.db.query(
            models.Feature.name,
            func.sum(models.LLMCall.cost).label("total_cost"),
            func.count(models.LLMCall.id).label("call_count"),
            func.avg(models.LLMCall.total_tokens).label("avg_tokens")
        ).join(
            models.LLMCall, models.LLMCall.feature_id == models.Feature.id
        ).filter(
            models.LLMCall.created_at.between(start_time, end_time)
        ).group_by(
            models.Feature.name
        ).order_by(
            desc("total_cost")
        ).all()
        
        return [
            {
                "feature": row[0],
                "total_cost": float(row[1] or 0),
                "call_count": row[2],
                "avg_tokens": float(row[3] or 0)
            }
            for row in result
        ]
    
    def get_provider_metrics(
        self,
        start_time: datetime,
        end_time: datetime,
        window_minutes: int = 5
    ) -> List[models.ProviderMetrics]:
        return self.db.query(models.ProviderMetrics).filter(
            models.ProviderMetrics.timestamp.between(start_time, end_time),
            models.ProviderMetrics.window_minutes == window_minutes
        ).order_by(models.ProviderMetrics.timestamp).all()
    
    # Anomaly operations
    def create_anomaly(self, anomaly_data: Dict[str, Any]) -> models.Anomaly:
        db_anomaly = models.Anomaly(**anomaly_data)
        self.db.add(db_anomaly)
        self.db.commit()
        self.db.refresh(db_anomaly)
        return db_anomaly
    
    def get_active_anomalies(self) -> List[models.Anomaly]:
        return self.db.query(models.Anomaly).filter(
            models.Anomaly.status.in_(["detected", "investigating"])
        ).order_by(desc(models.Anomaly.detected_at)).all()
    
    def update_anomaly_status(
        self, 
        anomaly_id: int, 
        status: str,
        notes: Optional[str] = None,
        resolved_by: Optional[str] = None
    ) -> Optional[models.Anomaly]:
        anomaly = self.db.query(models.Anomaly).filter(
            models.Anomaly.id == anomaly_id
        ).first()
        
        if anomaly:
            anomaly.status = status
            if notes:
                anomaly.investigation_notes = notes
            if status == "resolved":
                anomaly.resolved_at = datetime.now(pytz.UTC)
                if resolved_by:
                    anomaly.resolved_by = resolved_by
            self.db.commit()

**Data da última atualização:** 2024-01-15

## Visão Geral

A API DIYAPP V11 permite que desenvolvedores integrem funcionalidades do DIYAPP Evolution em seus próprios aplicativos. Esta versão introduz melhorias de performance, novos endpoints de IA e melhor tratamento de erros.

## Autenticação

### Método: Bearer Token JWT

Todas as requisições (exceto login) requerem autenticação via token JWT.

```http
Authorization: Bearer <seu_token_jwt>
```

### Fluxo de Autenticação

1. **Login** → Obtenha tokens de acesso
2. **Usar access_token** → Para requisições API
3. **Refresh token** → Renovar quando expirar

### Exemplo de Login

```bash
curl -X POST https://api.diyapp.example.com/v11/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "usuario@example.com",
    "password": "senhaSegura123"
  }'
```

**Resposta:**
```json
{
  "access_token": "eyJhbGciOiJIUzI1NiIs...",

}
```

```javascript
const db = require('../../database/connection');

class DataProcessingActivity {
    static async create(activity) {
        const sql = `
            INSERT INTO data_processing_activities 
            (activity_name, description, data_categories, purpose, legal_basis,

            JSON.stringify(activity.data_categories || []),
            activity.purpose,
            activity.legal_basis,
            JSON.stringify(activity.data_subjects || []),
            activity.retention_period_days,
            activity.deletion_procedure,
            JSON.stringify(activity.processors || [])
        ];
        
        return db.run(sql, params);
    }

    static async findAll() {
        const sql = `SELECT * FROM data_processing_activities WHERE is_active = 1 ORDER BY created_at DESC`;
        return db.all(sql);
    }

    static async findById(id) {
        const sql = `SELECT * FROM data_processing_activities WHERE id = ? AND is_active = 1`;
        return db.get(sql, [id]);
    }

    static async update(id, updates) {
        const fields = [];
        const values = [];
        
        Object.keys(updates).forEach(key => {
            if (['data_categories', 'data_subjects', 'processors'].includes(key)) {
                fields.push(`${key} = ?`);
                values.push(JSON.stringify(updates[key]));
            } else {
                fields.push(`${key} = ?`);
                values.push(updates[key]);
            }
        });
        
        fields.push('updated_at = CURRENT_TIMESTAMP');
        values.push(id);
        
        const sql = `UPDATE data_processing_activities SET ${fields.join(', ')} WHERE id = ?`;
        return db.run(sql, values);
    }

    static async deactivate(id) {
        const sql = `UPDATE data_processing_activities SET is_active = 0, updated_at = CURRENT_TIMESTAMP WHERE id = ?`;
        return db.run(sql, [id]);
    }

    static async getROPAReport() {
        const sql = `
            SELECT 
                activity_name,
                purpose,
                legal_basis,
                json_extract(data_categories, '$') as data_categories,
                retention_period_days,
                json_extract(processors, '$') as processors,
                created_at,
                updated_at
            FROM data_processing_activities 
            WHERE is_active = 1
            ORDER BY activity_name
        `;
        return db.all(sql);
    }
```

```javascript
const { v4: uuidv4 } = require('uuid');

class DataSubjectRequest {
    static async create(request) {
        const requestId = uuidv4();
        const deadlineDate = new Date();

        const sql = `
            INSERT INTO data_subject_requests 
            (request_id, user_id, request_type, request_data, deadline_date)
            VALUES (?, ?, ?, ?, ?)
        `;
        
        const params = [
            requestId,
            request.user_id,
            request.request_type,
            JSON.stringify(request.request_data || {}),
            deadlineDate.toISOString().split('T')[0]
        ];
        
        await db.run(sql, params);
        return requestId;
    }

    static async findByUserId(userId) {
        const sql = `SELECT * FROM data_subject_requests WHERE user_id = ? ORDER BY submitted_at DESC`;
        return db.all(sql, [userId]);
    }

    static async findPendingRequests() {
        const sql = `
            SELECT * FROM data_subject_requests 
            WHERE status = 'pending' 
            AND deadline_date >= date('now')
            ORDER BY deadline_date ASC
        `;
        return db.all(sql);
    }

    static async updateStatus(requestId, status, responseData = null, notes = null) {
        const updates = {
            status,
            updated_at: new Date().toISOString()
        };
        
        if (responseData) {
            updates.response_data = JSON.stringify(responseData);
        }
        
        if (notes) {
            updates.notes = notes;
        }
        
        if (status === 'completed') {
            updates.completed_at = new Date().toISOString();
        }
        
        const fields = Object.keys(updates).map(key => `${key} = ?`);
        const values = Object.values(updates);
        values.push(requestId);
        
        const sql = `UPDATE data_subject_requests SET ${fields.join(', ')} WHERE request_id = ?`;
        return db.run(sql, values);
    }

    static async getSLACompliance() {
        const sql = `
            SELECT 
                COUNT(*) as total_requests,
                SUM(CASE WHEN status = 'completed' AND completed_at <= deadline_date THEN 1 ELSE 0 END) as on_time,
                SUM(CASE WHEN status = 'completed' AND completed_at > deadline_date THEN 1 ELSE 0 END) as late,
                SUM(CASE WHEN status != 'completed' AND deadline_date < date('now') THEN 1 ELSE 0 END) as overdue
            FROM data_subject_requests
        `;
        return db.get(sql);
    }
```

```javascript
                    granted: granted,
                    version: version,
                    preferences: preferences
                })
            });

            logger.info(`Consentimento registrado: ${userId} - ${consentType} - ${granted}`);

            logger.error(`Erro ao registrar consentimento: ${error.message}`);
            throw error;
        }
    }

    async getUserConsents(userId) {
        try {
            const consents = await Consent.findByUserId(userId);
            
            // Agrupa por tipo, mantendo a versão mais recente
            const grouped = {};
            consents.forEach(consent => {
                if (!grouped[consent.consent_type] || 
                    new Date(consent.granted_at) > new Date(grouped[consent.consent_type].granted_at)) {
                    grouped[consent.consent_type] = consent;
                }
            });
            
            return Object.values(grouped);
        } catch (error) {
            logger.error(`Erro ao buscar consentimentos: ${error.message}`);
            throw error;
        }
    }

    async validateConsent(userId, consentType) {
        try {
            const consents = await Consent.findByUserIdAndType(userId, consentType);
            if (consents.length === 0) return false;
            
            // Pega o consentimento mais recente
            const latestConsent = consents.reduce((latest, current) => {
                return new Date(current.granted_at) > new Date(latest.granted_at) ? current : latest;
            });
            
            return latestConsent.granted;
        } catch (error) {
            logger.error(`Erro ao validar consentimento: ${error.message}`);
            return false;
        }
    }

        const userConsents = await this.getUserConsents(userId);

                null
        };
    }

        try {
            await this.recordConsent(userId, consentType, false);

            }
            
            return true;
        } catch (error) {

            throw error;
        }
    }
```
