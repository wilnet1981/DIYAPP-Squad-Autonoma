class ADRStatus(Enum):
    PROPOSED = "Proposed"
    ACCEPTED = "Accepted"
    SUPERSEDED = "Superseded"
    DEPRECATED = "Deprecated"

@dataclass
class ArchitectureDecisionRecord:
    """ADR Template for V12 Microservices Architecture"""
    id: str
    title: str
    date: str
    status: ADRStatus
    authors: List[str]
    context: str
    decision: str
    options_considered: Dict[str, Dict[str, List[str]]]
    consequences: Dict[str, List[str]]
    review_date: Optional[str]
    superseded_by: Optional[str] = None
    
    def to_markdown(self) -> str:
        """Convert ADR to markdown format"""
        options_text = "\n".join([
            f"- **{name}**: {desc}\n  - Prós: {', '.join(pros)}\n  - Contras: {', '.join(cons)}"
            for name, (desc, pros, cons) in self.options_considered.items()
        ])
        
        return f"""# ADR-{self.id}: {self.title}

**Data**: {self.date}
**Status**: {self.status.value}
**Autores**: {', '.join(self.authors)}
{'**Substituído por**: ' + self.superseded_by if self.superseded_by else ''}

## CONTEXTO
{self.context}

## DECISÃO
{self.decision}

## OPÇÕES CONSIDERADAS
{options_text}

## CONSEQUÊNCIAS
### Positivas
{chr(10).join(f'- {c}' for c in self.consequences['positive'])}

### Negativas
{chr(10).join(f'- {c}' for c in self.consequences['negative'])}

### Riscos
{chr(10).join(f'- {c}' for c in self.consequences.get('risks', []))}

{'## REVISÃO: ' + self.review_date if self.review_date else ''}
"""

```javascript
// libs/circuit-breaker/index.js
const EventEmitter = require('events');

class CircuitBreaker extends EventEmitter {
  constructor(options = {}) {
    super();
    
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.successCount = 0;
    this.nextAttempt = Date.now();
    
    // Configurações
    this.failureThreshold = options.failureThreshold || 5;
    this.resetTimeout = options.resetTimeout || 60000; // 60 segundos
    this.halfOpenMaxAttempts = options.halfOpenMaxAttempts || 2;
    this.timeout = options.timeout || 10000; // 10 segundos
    
    // Métricas
    this.metrics = {
      totalRequests: 0,
      successfulRequests: 0,
      failedRequests: 0,
      circuitOpened: 0,
      circuitClosed: 0,
      circuitHalfOpened: 0
    };
  }

  async call(fn, ...args) {
    this.metrics.totalRequests++;
    
    // Verifica se está no período de timeout (OPEN state)
    if (this.state === 'OPEN') {
      if (Date.now() >= this.nextAttempt) {
        this.state = 'HALF_OPEN';
        this.metrics.circuitHalfOpened++;
        this.emit('halfOpen');
      } else {
        this.metrics.failedRequests++;
        throw new Error('Circuit breaker is OPEN');
      }
    }

    try {
      // Executa a função com timeout
      const result = await this.executeWithTimeout(fn, args);
      
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  async executeWithTimeout(fn, args) {
    return new Promise((resolve, reject) => {
      const timeoutId = setTimeout(() => {
        reject(new Error('Request timeout'));
      }, this.timeout);

      Promise.resolve(fn(...args))
        .then(resolve)
        .catch(reject)
        .finally(() => clearTimeout(timeoutId));
    });
  }

  onSuccess() {
    this.successCount++;
    this.failureCount = 0;

    if (this.state === 'HALF_OPEN') {
      if (this.successCount >= this.halfOpenMaxAttempts) {
        this.state = 'CLOSED';
        this.successCount = 0;
        this.metrics.circuitClosed++;
        this.emit('closed');
      }
    }
    
    this.metrics.successfulRequests++;
  }

  onFailure() {
    this.failureCount++;
    this.successCount = 0;

    if (this.state === 'HALF_OPEN') {
      this.state = 'OPEN';
      this.nextAttempt = Date.now() + this.resetTimeout;
      this.metrics.circuitOpened++;
      this.emit('opened');
    } else if (this.state === 'CLOSED' && this.failureCount >= this.failureThreshold) {
      this.state = 'OPEN';
      this.nextAttempt = Date.now() + this.resetTimeout;
      this.metrics.circuitOpened++;
      this.emit('opened');
    }
    
    this.metrics.failedRequests++;
  }

  getMetrics() {
    return {
      ...this.metrics,
      state: this.state,
      failureCount: this.failureCount,
      successCount: this.successCount,
      nextAttempt: this.nextAttempt
    };
  }

  reset() {
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.successCount = 0;
    this.nextAttempt = Date.now();
    this.emit('reset');
  }
}

module.exports = CircuitBreaker;
```

### Retry Strategy Library

```javascript
// libs/retry-strategy/index.js
class RetryStrategy {
  constructor(options = {}) {
    this.maxAttempts = options.maxAttempts || 3;
    this.baseDelay = options.baseDelay || 100; // 100ms
    this.maxDelay = options.maxDelay || 10000; // 10 segundos
    this.retryableErrors = options.retryableErrors || [
      'ECONNRESET',
      'ETIMEDOUT',
      'ECONNREFUSED',
      'ENOTFOUND',
      'EAI_AGAIN'
    ];
    this.retryableStatusCodes = options.retryableStatusCodes || [408, 429, 500, 502, 503, 504];
  }

  async execute(fn, context = {}) {
    let lastError;
    
    for (let attempt = 1; attempt <= this.maxAttempts; attempt++) {
      try {
        const result = await fn();
        return result;
      } catch (error) {
        lastError = error;
        
        // Verifica se o erro é retryable
        if (!this.shouldRetry(error, attempt)) {
          throw error;
        }
        
        // Calcula delay com exponential backoff + jitter
        const delay = this.calculateDelay(attempt);
        
        // Log do retry
        console.warn(`Retry attempt ${attempt}/${this.maxAttempts} after ${delay}ms`, {
          context,
          error: error.message,
          stack: error.stack
        });
        
        // Aguarda antes da próxima tentativa
        if (attempt < this.maxAttempts) {
          await this.sleep(delay);
        }
      }
    }
    
    throw lastError;
  }

  shouldRetry(error, attempt) {
    if (attempt >= this.maxAttempts) return false;
    
    // Verifica códigos de status HTTP
    if (error.statusCode && this.retryableStatusCodes.includes(error.statusCode)) {
      return true;
    }
    
    // Verifica códigos de erro do sistema
    if (error.code && this.retryableErrors.includes(error.code)) {
      return true;
    }
    
    // Verifica se é timeout
    if (error.message && error.message.includes('timeout')) {
      return true;
    }
    
    // Não retry em erros 4xx (exceto 408, 429)
    if (error.statusCode && error.statusCode >= 400 && error.statusCode < 500) {
      return error.statusCode === 408 || error.statusCode === 429;
    }
    
    return false;
  }

  calculateDelay(attempt) {
    // Exponential backoff: baseDelay * 2^(attempt-1)
    const exponentialDelay = this.baseDelay * Math.pow(2, attempt - 1);
    
    // Adiciona jitter (até 30% de variação)
    const jitter = exponentialDelay * 0.3 * Math.random();
    
    // Delay final com jitter
    const delayWithJitter = exponentialDelay + jitter;
    
    // Limita ao máximo permitido
    return Math.min(delayWithJitter, this.maxDelay);
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

module.exports = RetryStrategy;
```

```javascript
class Logger {

      ),
      defaultMeta: { service: this.serviceName },
      transports: [
        new winston.transports.Console({
          format: winston.format.combine(
            winston.format.colorize(),
            winston.format.simple()
          )
        }),
        new winston.transports.File({ 
          filename: 'logs/error.log', 
          level: 'error' 
        }),
        new winston.transports.File({ 
          filename: 'logs/combined.log' 
        })
      ]
    });
  }

  setContext(correlationId, userId = null) {
    this.correlationId = correlationId;
    this.userId = userId;
  }

  generateCorrelationId() {
    return uuidv4();
  }

  maskSensitiveData(data) {
    if (typeof data !== 'object' || data === null) return data;
    
    const masked = { ...data };
    const sensitiveFields = [
      'password',
      'token',
      'secret',
      'key',
      'credit_card',
      'cvv',
      'cpf',
      'email'
    ];
    
    for (const field of sensitiveFields) {
      if (masked[field]) {
        masked[field] = '***MASKED***';
      }
    }
    
    return masked;
  }

  info(message, meta = {}) {
    this.logger.info(message, this.enrichLog(meta));
  }

  error(message, meta = {}) {
    this.logger.error(message, this.enrichLog(meta));
  }

  warn(message, meta = {}) {
    this.logger.warn(message, this.enrichLog(meta));
  }

  debug(message, meta = {}) {
    this.logger.debug(message, this.enrichLog(meta));
  }

  logRequest(req, res, duration) {
    const logData = {
      correlation_id: this.correlationId || req.headers['x-correlation-id'] || this.generateCorrelationId(),
      user_id: this.userId || req.user?.id || null,
      method: req.method,
      endpoint: req.originalUrl,
      status_code: res.statusCode,
      duration_ms: duration,
      user_agent: req.get('user-agent'),
      ip: req.ip
    };

    // Mascara dados sensíveis nos headers
    const headers = { ...req.headers };
    if (headers.authorization) headers.authorization = '***MASKED***';
    if (headers.cookie) headers.cookie = '***MASKED***';
    
    logData.headers = headers;

    this.info('HTTP Request', logData);
  }

  enrichLog(meta) {
    const enriched = {
      ...this.maskSensitiveData(meta),
      correlation_id: this.correlationId,
      user_id: this.userId,
      timestamp: new Date().toISOString()
    };

    return enriched;
  }
}
```
