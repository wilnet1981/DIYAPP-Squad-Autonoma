const fs = require('fs').promises;
const path = require('path');

class KPIMonitor {
    constructor() {
        this.kpis = {
            uptime: { target: 99.99, current: 99.87, unit: '%' },
            responseTimeP95: { target: 200, current: 240, unit: 'ms' },
            errorRate: { target: 0.1, current: 0.12, unit: '%' },
            autoHealingRate: { target: 100, current: 87, unit: '%' }
        };
        
        this.autoHealFeatures = [
            {
                id: 1,
                name: "Health Check Automático",
                priority: "P1",
                impact: "Alto",
                effort: "Médio",
                status: "pending",
                description: "Monitoramento contínuo com recuperação automática"
            },
            {
                id: 2,
                name: "Circuit Breaker Inteligente",
                priority: "P1",
                impact: "Alto",
                effort: "Baixo",
                status: "in_progress",
                description: "Isola falhas e tenta reconexão automática"
            },
            {
                id: 3,
                name: "Rollback Automático de Deploy",
                priority: "P2",
                impact: "Médio",
                effort: "Alto",
                status: "pending",
                description: "Reverte versões com alta taxa de erro"
            },
            {
                id: 4,
                name: "Escalonamento Automático",
                priority: "P2",
                impact: "Médio",
                effort: "Alto",
                status: "planned",
                description: "Ajusta capacidade baseado em métricas"
            }
        ];
        
        this.logFile = path.join(__dirname, '../logs/stability.log');
    }

    async checkKPIs() {
        const results = {};
        
        for (const [name, data] of Object.entries(this.kpis)) {
            const status = data.current >= data.target ? 'healthy' : 
                          data.current >= data.target * 0.95 ? 'degraded' : 'critical';
            
            results[name] = {
                current: data.current,
                target: data.target,
                unit: data.unit,
                status: status,
                meetsTarget: data.current >= data.target
            };
            
            if (status !== 'healthy') {
                await this.logAlert(name, data.current, data.target, status);
            }
        }
        
        return results;
    }

    async logAlert(kpiName, current, target, status) {
        const timestamp = new Date().toISOString();
        const message = `[${timestamp}] ALERTA: ${kpiName} - ${current} (meta: ${target}) - Status: ${status.toUpperCase()}\n`;
        
        try {
            await fs.appendFile(this.logFile, message);
            console.log(`Alerta registrado: ${kpiName}`);
        } catch (error) {
            console.error('Erro ao registrar alerta:', error);
        }
    }

    prioritizeFeatures() {
        // Prioridade: P1 > P2, impacto Alto > Médio, esforço Baixo > Médio > Alto
        return this.autoHealFeatures.sort((a, b) => {
            if (a.priority !== b.priority) {
                return a.priority === 'P1' ? -1 : 1;
            }
            if (a.impact !== b.impact) {
                const impactOrder = { 'Alto': 0, 'Médio': 1, 'Baixo': 2 };
                return impactOrder[a.impact] - impactOrder[b.impact];
            }
            const effortOrder = { 'Baixo': 0, 'Médio': 1, 'Alto': 2 };
            return effortOrder[a.effort] - effortOrder[b.effort];
        });
    }

    generateSprintRecommendation() {
        const prioritized = this.prioritizeFeatures();
        const kpiStatus = this.checkKPIs();
        
        return {
            sprintFocus: "Estabilidade 100% Autônoma",
            timeframe: "Próximo Sprint (2 semanas)",
            recommendedFeatures: prioritized.slice(0, 2), // Top 2 para sprint
            kpiTargets: this.kpis,
            monitoring: {
                frequency: "30 segundos",
                alertChannels: ["dashboard", "logs", "webhook"],
                autoHealThreshold: 85 // % mínimo para ativar auto-cura
            }
        };
    }

    async simulateAutoHeal() {
        console.log('Simulando ciclo de auto-cura...');
        
        // Simula melhoria gradual nos KPIs
        this.kpis.uptime.current = Math.min(99.99, this.kpis.uptime.current + 0.02);
        this.kpis.responseTimeP95.current = Math.max(180, this.kpis.responseTimeP95.current - 5);
        this.kpis.autoHealingRate.current = Math.min(100, this.kpis.autoHealingRate.current + 1);
        
        const timestamp = new Date().toISOString();
        const logMessage = `[${timestamp}] AUTO-CURA: KPIs ajustados - Uptime: ${this.kpis.uptime.current}%, Response: ${this.kpis.responseTimeP95.current}ms\n`;
        
        await fs.appendFile(this.logFile, logMessage);
        
        return {
            success: true,
            message: "Auto-cura aplicada",
            newValues: this.kpis
        };
    }
}

// Servidor simples para dashboard
const http = require('http');

const monitor = new KPIMonitor();

const server = http.createServer(async (req, res) => {

});

**docker/healthcheck.sh**
```bash
#!/bin/bash

# Health check para o container DIYAPP
# Retorna 0 se saudável, 1 se não saudável

HEALTH_ENDPOINT="http://localhost:3000/health"
MAX_RETRIES=3
RETRY_DELAY=2

for i in $(seq 1 $MAX_RETRIES); do
    # Verificar se a aplicação está respondendo
    response=$(curl -s -o /dev/null -w "%{http_code}" $HEALTH_ENDPOINT)
    
    if [ "$response" = "200" ]; then
        # Verificar métricas de saúde internas
        metrics_response=$(curl -s "$HEALTH_ENDPOINT/metrics")
        
        # Extrair status geral do JSON (assumindo que retorna JSON)
        status=$(echo $metrics_response | grep -o '"status":"[^"]*"' | cut -d'"' -f4)
        
        if [ "$status" = "healthy" ] || [ -z "$status" ]; then
            echo "DIYAPP is healthy"
            exit 0
        fi
    fi
    
    if [ $i -lt $MAX_RETRIES ]; then
        sleep $RETRY_DELAY
    fi
done

echo "DIYAPP health check failed"
exit 1

DesignTokens.checkContrast = function(background, foreground) {
  // Implementação simplificada - em produção usar biblioteca como chroma.js
  const hexToRgb = (hex) => {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ? {
      r: parseInt(result[1], 16),
      g: parseInt(result[2], 16),
      b: parseInt(result[3], 16)
    } : null;
  };

  const rgb1 = hexToRgb(background);
  const rgb2 = hexToRgb(foreground);
  
  if (!rgb1 || !rgb2) return false;
  
  const luminance = (r, g, b) => {
    const a = [r, g, b].map(v => {
      v /= 255;
      return v <= 0.03928 ? v / 12.92 : Math.pow((v + 0.055) / 1.055, 2.4);
    });
    return a[0] * 0.2126 + a[1] * 0.7152 + a[2] * 0.0722;
  };

  const lum1 = luminance(rgb1.r, rgb1.g, rgb1.b);
  const lum2 = luminance(rgb2.r, rgb2.g, rgb2.b);
  const brightest = Math.max(lum1, lum2);
  const darkest = Math.min(lum1, lum2);
  const ratio = (brightest + 0.05) / (darkest + 0.05);
  
  return ratio >= 4.5; // WCAG AA mínimo para texto normal
};

import time
import threading
from enum import Enum
from typing import Any, Callable, Optional, Dict
import logging
from dataclasses import dataclass
from datetime import datetime, timedelta

logger = logging.getLogger(__name__)


class CircuitState(Enum):
    CLOSED = "CLOSED"      # Normal operation
    OPEN = "OPEN"          # Circuit is open, failing fast
    HALF_OPEN = "HALF_OPEN" # Testing if service recovered


@dataclass
class CircuitBreakerConfig:
    failure_threshold: int = 5           # Failures before opening circuit
    reset_timeout: int = 30              # Seconds before trying half-open
    half_open_max_requests: int = 3      # Max requests in half-open state
    success_threshold: int = 2           # Successes to close circuit again
    excluded_exceptions: tuple = ()      # Exceptions that don't count as failures


class CircuitBreaker:
    """Circuit breaker pattern implementation with thread safety"""
    
    def __init__(self, name: str, config: Optional[CircuitBreakerConfig] = None):
        self.name = name
        self.config = config or CircuitBreakerConfig()
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.success_count = 0
        self.last_failure_time = None
        self.last_state_change = datetime.now()
        self._lock = threading.RLock()
        
        # Metrics
        self.metrics = {
            "total_requests": 0,
            "successful_requests": 0,
            "failed_requests": 0,
            "circuit_opens": 0,
            "fast_failures": 0,
        }
    
    def execute(self, func: Callable, *args, **kwargs) -> Any:
        """Execute function with circuit breaker protection"""
        with self._lock:
            self.metrics["total_requests"] += 1
            
            # Check if circuit is open
            if self.state == CircuitState.OPEN:
                if self._should_try_half_open():
                    self.state = CircuitState.HALF_OPEN
                    self.success_count = 0
                    self.last_state_change = datetime.now()
                    logger.info(f"Circuit {self.name} transitioning to HALF_OPEN")
                else:
                    self.metrics["fast_failures"] += 1
                    raise CircuitOpenError(
                        f"Circuit {self.name} is OPEN. "
                        f"Last failure: {self.last_failure_time}"
                    )
            
            # Check half-open limits
            if (self.state == CircuitState.HALF_OPEN and 
                self.success_count >= self.config.half_open_max_requests):
                self.metrics["fast_failures"] += 1
                raise CircuitOpenError(
                    f"Circuit {self.name} is HALF_OPEN - request limit reached"
                )
        
        # Try to execute the function
        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result
        except Exception as e:
            self._on_failure(e)
            raise
    
    def _should_try_half_open(self) -> bool:
        """Check if enough time has passed to try half-open"""
        if not self.last_failure_time:
            return False
        
        time_since_failure = datetime.now() - self.last_failure_time
        return time_since_failure.total_seconds() >= self.config.reset_timeout
    
    def _on_success(self):
        """Handle successful execution"""
        with self._lock:
            self.success_count += 1
            self.metrics["successful_requests"] += 1
            
            if self.state == CircuitState.HALF_OPEN:
                if self.success_count >= self.config.success_threshold:
                    self._close_circuit()
            elif self.state == CircuitState.CLOSED:
                # Reset failure count on consecutive successes
                if self.failure_count > 0:
                    self.failure_count = max(0, self.failure_count - 1)
    
    def _on_failure(self, exception: Exception):
        """Handle failed execution"""
        with self._lock:
            self.failure_count += 1
            self.metrics["failed_requests"] += 1
            self.last_failure_time = datetime.now()
            
            # Check if exception should be excluded
            if isinstance(exception, self.config.excluded_exceptions):
                return
            
            # Check if we should open the circuit
            if (self.state == CircuitState.CLOSED and 
                self.failure_count >= self.config.failure_threshold):
                self._open_circuit()
            elif self.state == CircuitState.HALF_OPEN:
                self._open_circuit()
    
    def _open_circuit(self):
        """Open the circuit"""
        self.state = CircuitState.OPEN
        self.last_state_change = datetime.now()
        self.metrics["circuit_opens"] += 1
        logger.warning(f"Circuit {self.name} OPENED. Failures: {self.failure_count}")
    
    def _close_circuit(self):
        """Close the circuit"""
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.success_count = 0
        self.last_state_change = datetime.now()
        logger.info(f"Circuit {self.name} CLOSED")
    
    def get_state(self) -> Dict[str, Any]:
        """Get current circuit state and metrics"""
        with self._lock:
            return {
                "name": self.name,
                "state": self.state.value,
                "failure_count": self.failure_count,
                "success_count": self.success_count,
                "last_failure_time": self.last_failure_time.isoformat() if self.last_failure_time else None,
                "last_state_change": self.last_state_change.isoformat(),
                "metrics": self.metrics.copy(),
                "config": {
                    "failure_threshold": self.config.failure_threshold,
                    "reset_timeout": self.config.reset_timeout,
                    "half_open_max_requests": self.config.half_open_max_requests,
                    "success_threshold": self.config.success_threshold,
                }
            }

import random
from typing import Callable, Optional, Type, Tuple
from functools import wraps


class RetryPolicy:
    """Configurable retry policy with exponential backoff and jitter"""
    
    def __init__(
        self,
        max_retries: int = 3,
        base_delay: float = 0.1,  # seconds
        max_delay: float = 10.0,  # seconds
        jitter: bool = True,
        retry_on_exceptions: Tuple[Type[Exception], ...] = (Exception,),
        no_retry_on_exceptions: Tuple[Type[Exception], ...] = ()
    ):
        self.max_retries = max_retries
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.jitter = jitter
        self.retry_on_exceptions = retry_on_exceptions
        self.no_retry_on_exceptions = no_retry_on_exceptions
    
    def __call__(self, func: Callable) -> Callable:
        """Decorator to apply retry policy to a function"""
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            
            for attempt in range(self.max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    
                    # Check if we should retry this exception
                    if not self._should_retry(e):
                        raise
                    
                    # Check if this was the last attempt
                    if attempt == self.max_retries:
                        logger.warning(
                            f"Max retries ({self.max_retries}) exceeded for {func.__name__}"
                        )
                        raise
                    
                    # Calculate delay with exponential backoff
                    delay = self._calculate_delay(attempt)
                    
                    logger.info(
                        f"Retry {attempt + 1}/{self.max_retries} for {func.__name__} "
                        f"after {delay:.2f}s. Error: {str(e)}"
                    )
                    
                    time.sleep(delay)
            
            # This should never be reached
            raise last_exception
        
        return wrapper
    
    def _should_retry(self, exception: Exception) -> bool:
        """Determine if an exception should be retried"""
        # Check no-retry exceptions first
        if isinstance(exception, self.no_retry_on_exceptions):
            return False
        
        # Check retry-on exceptions
        if self.retry_on_exceptions:
            return isinstance(exception, self.retry_on_exceptions)
        
        return True
    
    def _calculate_delay(self, attempt: int) -> float:
        """Calculate delay with exponential backoff and optional jitter"""
        delay = min(
            self.max_delay,
            self.base_delay * (2 ** attempt)
        )
        
        if self.jitter:
            # Add ±25% jitter
            jitter_factor = 0.25
            delay *= 1 + (random.random() * 2 * jitter_factor - jitter_factor)
        
        return delay


# src/core/resilience/fallback.py
from typing import Any, Callable, Optional
from functools import wraps


class FallbackStrategy:
    """Fallback strategy for service degradation"""
    
    def __init__(
        self,
        fallback_func: Optional[Callable] = None,
        fallback_value: Any = None,
        cache_ttl: int = 300,  # seconds
        degrade_gracefully: bool = True
    ):
        self.fallback_func = fallback_func
        self.fallback_value = fallback_value
        self.cache_ttl = cache_ttl
        self.degrade_gracefully = degrade_gracefully
        self._cache = {}
        self._cache_timestamps = {}
    
    def __call__(self, func: Callable) -> Callable:
        """Decorator to apply fallback strategy"""
        @wraps(func)
        def wrapper(*args, **kwargs):
            cache_key = self._generate_cache_key(func, args, kwargs)
            
            # Try to get from cache first
            if cache_key in self._cache:
                if time.time() - self._cache_timestamps[cache_key] < self.cache_ttl:
                    logger.info(f"Using cached result for {func.__name__}")
                    return self._cache[cache_key]
            
            try:
                result = func(*args, **kwargs)
                # Cache successful result
                self._cache[cache_key] = result
                self._cache_timestamps[cache_key] = time.time()
                return result
            except Exception as e:
                logger.warning(f"Fallback triggered for {func.__name__}: {str(e)}")
                
                # Try fallback function
                if self.fallback_func:
                    try:
                        return self.fallback_func(*args, **kwargs)
                    except Exception as fallback_error:
                        logger.error(f"Fallback function also failed: {str(fallback_error)}")
                
                # Return fallback value
                if self.fallback_value is not None:
                    return self.fallback_value
                
                # Degrade gracefully if configured
                if self.degrade_gracefully:
                    # Return cached value even if stale
                    if cache_key in self._cache:
                        logger.warning(f"Using stale cache for {func.__name__}")
                        return self._cache[cache_key]
                    
                    # Return empty/default response
                    return self._get_degraded_response(func)
                
                # Re-raise if no fallback available
                raise
        
        return wrapper
    
    def _generate_cache_key(self, func: Callable, args: tuple, kwargs: dict) -> str:
        """Generate cache key from function and arguments"""
        # Simple key generation - can be improved for complex cases
        key_parts = [func.__name__]
        key_parts.extend(str(arg) for arg in args)
        key_parts.extend(f"{k}={v}" for k, v in sorted(kwargs.items()))
        return "|".join(key_parts)
    
    def _get_degraded_response(self, func: Callable) -> Any:
        """Generate degraded response based on function type"""
        # This can be customized per application
        func_name = func.__name__.lower()
        
        if "list" in func_name or "get_all" in func_name:
            return []
        elif "get" in func_name or "find" in func_name:
            return None
        elif "count" in func_name:
            return 0
        else:
            return {}

  skeletonClassName,
  ...props
}: LazyImageProps) {
  const [isLoading, setIsLoading] = useState(true);
  const [hasError, setHasError] = useState(false);
  const [isInView, setIsInView] = useState(false);
  const imgRef = useRef<HTMLImageElement>(null);
  const observerRef = useRef<IntersectionObserver>();

  useEffect(() => {
    if (!imgRef.current) return;

    observerRef.current = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsInView(true);
          observerRef.current?.disconnect();
        }
      },
      { rootMargin: '50px' }
    );

    observerRef.current.observe(imgRef.current);

    return () => {
      observerRef.current?.disconnect();
    };
  }, []);

  const handleLoad = () => {
    setIsLoading(false);
  };

  const handleError = () => {
    setIsLoading(false);
    setHasError(true);
  };

  const imageSrc = hasError && fallbackSrc ? fallbackSrc : src;

  return (
    <div className="relative overflow-hidden">
      {isLoading && (
        <Skeleton
          className={cn('absolute inset-0 z-10', skeletonClassName)}
          aria-hidden="true"
        />
      )}
      {isInView && (
        <img
          ref={imgRef}
          src={imageSrc}
          alt={alt}
          className={cn(
            'transition-opacity duration-300',
            isLoading ? 'opacity-0' : 'opacity-100',
            className
          )}
          onLoad={handleLoad}
          onError={handleError}
          loading="lazy"
          decoding="async"
          {...props}
        />
      )}
      {!isInView && (
        <div
          ref={imgRef}
          className={cn('bg-muted', className)}
          aria-hidden="true"
          data-lazy-placeholder
        />
      )}
    </div>
  );
}

const { ElasticsearchTransformer } = require('winston-elasticsearch');
const { SeverityNumber } = require('@opentelemetry/api-logs');
const { logs } = require('@opentelemetry/api-logs');

class StructuredLogger {
  constructor() {
    this.logger = null;
    this.otelLogger = null;
    this.initialize();
  }

  initialize() {
    const { combine, timestamp, json, errors, metadata } = winston.format;

    // Custom format for structured logging
    const structuredFormat = winston.format((info) => {
      const span = require('@opentelemetry/api').trace.getActiveSpan();
      
      if (span) {
        const spanContext = span.spanContext();
        info.traceId = spanContext.traceId;
        info.spanId = spanContext.spanId;
        info.traceFlags = spanContext.traceFlags;
      }

      // Add service context
      info.service = 'diyapp-v12-core';
      info.environment = process.env.NODE_ENV || 'development';
      info.timestamp = new Date().toISOString();

      // Ensure error objects are properly serialized
      if (info.error instanceof Error) {
        info.error = {
          message: info.error.message,
          stack: info.error.stack,
          name: info.error.name,
          code: info.error.code
        };
      }

      return info;
    });

    const transports = [
      new winston.transports.Console({
        format: combine(
          errors({ stack: true }),
          structuredFormat(),
          timestamp(),
          json()
        )
      }),
      new winston.transports.File({
        filename: 'logs/error.log',
        level: 'error',
        format: combine(
          errors({ stack: true }),
          structuredFormat(),
          timestamp(),
          json()
        ),
        maxsize: 5242880, // 5MB
        maxFiles: 5,
      }),
      new winston.transports.File({
        filename: 'logs/combined.log',
        format: combine(
          errors({ stack: true }),
          structuredFormat(),
          timestamp(),
          json()
        ),
        maxsize: 5242880, // 5MB
        maxFiles: 10,
      })
    ];

    // Add Elasticsearch transport if configured
    if (process.env.ELASTICSEARCH_HOST) {
      const { ElasticsearchTransport } = require('winston-elasticsearch');
      
      const esTransport = new ElasticsearchTransport({
        level: 'info',
        clientOpts: {
          node: process.env.ELASTICSEARCH_HOST,
          auth: {
            username: process.env.ELASTICSEARCH_USERNAME,
            password: process.env.ELASTICSEARCH_PASSWORD
          }
        },
        transformer: (logData) => {
          const transformed = ElasticsearchTransformer(logData);
          transformed['@timestamp'] = new Date().toISOString();
          return transformed;
        }
      });
      
      transports.push(esTransport);
    }

    this.logger = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      format: combine(
        errors({ stack: true }),
        metadata()
      ),
      transports,
      exceptionHandlers: [
        new winston.transports.File({ filename: 'logs/exceptions.log' })
      ],
      rejectionHandlers: [
        new winston.transports.File({ filename: 'logs/rejections.log' })
      ]
    });

    // Initialize OpenTelemetry logger if available
    try {
      this.otelLogger = logs.getLogger('diyapp-v12-logger');
    } catch (error) {
      // OTEL not available, continue without it
    }
  }

  log(level, message, meta = {}) {
    const logData = {
      message,
      ...meta,
      timestamp: new Date().toISOString()
    };

    // Log to winston
    this.logger.log(level, message, meta);

    // Also log to OpenTelemetry if available
    if (this.otelLogger) {
      const severityNumber = this.mapLevelToSeverity(level);
      
      this.otelLogger.emit({
        severityNumber,
        severityText: level.toUpperCase(),
        body: message,
        attributes: {
          ...meta,
          'log.level': level,
          'service.name': 'diyapp-v12-core'
        }
      });
    }
  }

  mapLevelToSeverity(level) {
    const levelMap = {
      error: SeverityNumber.ERROR,
      warn: SeverityNumber.WARN,
      info: SeverityNumber.INFO,
      debug: SeverityNumber.DEBUG,
      trace: SeverityNumber.TRACE
    };
    
    return levelMap[level] || SeverityNumber.INFO;
  }

  // Convenience methods
  error(message, meta = {}) {
    this.log('error', message, meta);
  }

  warn(message, meta = {}) {
    this.log('warn', message, meta);
  }

  info(message, meta = {}) {
    this.log('info', message, meta);
  }

  debug(message, meta = {}) {
    this.log('debug', message, meta);
  }

  // Business-specific logging
  pluginLoaded(pluginName, version, loadTime) {
    this.info('Plugin loaded', {
      plugin: pluginName,
      version,
      loadTime,
      event: 'plugin_load'
    });
  }

    this.warn('Circuit breaker state changed', {
      service: serviceName,
      state,
      errorCount,
      event: 'circuit_breaker_state_change'
    });

    this.log(level, 'Auto-heal action triggered', {
      action,
      reason,
      success,
      event: 'auto_heal_action'

// src/middleware/resilience.js
import CircuitBreaker from 'opossum';
import axios from 'axios';
import { EventEmitter } from 'events';
import { logger } from '../utils/logger.js';

class ResilienceManager extends EventEmitter {
    constructor() {
        super();
        this.circuits = new Map();
        this.metrics = {
            circuitStateChanges: [],
            retryAttempts: [],
            fallbackTriggers: []
        };
    }

    createCircuit(serviceName, options = {}) {
        const defaultOptions = {
            timeout: options.timeout || 10000,
            errorThresholdPercentage: 50,
            resetTimeout: 30000,
            rollingCountTimeout: 10000,
            rollingCountBuckets: 10,
            name: serviceName,
            group: serviceName
        };

        const circuit = new CircuitBreaker(
            async (url, config) => {
                const response = await axios({
                    url,
                    timeout: defaultOptions.timeout,
                    ...config
                });
                return response.data;
            },
            defaultOptions
        );

        // Event listeners for circuit state changes
        circuit.on('open', () => {
            logger.warn(`Circuit ${serviceName} OPEN`, { serviceName });
            this.metrics.circuitStateChanges.push({
                timestamp: Date.now(),
                service: serviceName,
                state: 'open'
            });
            this.emit('circuitOpen', { serviceName });
        });

        circuit.on('close', () => {
            logger.info(`Circuit ${serviceName} CLOSED`, { serviceName });
            this.metrics.circuitStateChanges.push({
                timestamp: Date.now(),
                service: serviceName,
                state: 'closed'
            });
            this.emit('circuitClosed', { serviceName });
        });

        circuit.on('halfOpen', () => {
            logger.info(`Circuit ${serviceName} HALF_OPEN`, { serviceName });
            this.metrics.circuitStateChanges.push({
                timestamp: Date.now(),
                service: serviceName,
                state: 'halfOpen'
            });
        });

        circuit.fallback(async (url, config, err) => {
            logger.error(`Circuit ${serviceName} fallback triggered`, {
                serviceName,
                error: err.message,
                url
            });
            
            this.metrics.fallbackTriggers.push({
                timestamp: Date.now(),
                service: serviceName,
                url,
                error: err.message
            });

            // Return cached data or default response
            return this.getFallbackResponse(serviceName, url);
        });

        this.circuits.set(serviceName, circuit);
        return circuit;
    }

    getFallbackResponse(serviceName, url) {
        // Implement service-specific fallback responses
        const fallbacks = {
            'payment-service': {
                status: 'degraded',
                message: 'Payment service temporarily unavailable',
                timestamp: Date.now()
            },
            'email-service': {
                status: 'queued',
                message: 'Email queued for later delivery',
                timestamp: Date.now()
            },
            'llm-service': {
                status: 'fallback',
                message: 'AI service unavailable, using cached response',
                cached: true,
                timestamp: Date.now()
            }
        };

        return fallbacks[serviceName] || {
            status: 'service_unavailable',
            message: 'Service temporarily unavailable',
            timestamp: Date.now()
        };
    }

    async callWithRetry(serviceName, url, config = {}, maxRetries = 3) {
        const circuit = this.circuits.get(serviceName) || this.createCircuit(serviceName);
        
        let lastError;
        for (let attempt = 1; attempt <= maxRetries; attempt++) {
            try {
                const result = await circuit.fire(url, config);
                
                if (attempt > 1) {
                    logger.info(`Retry successful on attempt ${attempt}`, {
                        serviceName,
                        url,
                        attempt
                    });
                }
                
                return result;
            } catch (error) {
                lastError = error;
                
                this.metrics.retryAttempts.push({
                    timestamp: Date.now(),
                    service: serviceName,
                    attempt,
                    error: error.message,
                    url
                });

                // Only retry on timeout or 5xx errors
                const isRetryable = error.code === 'ECONNABORTED' || 
                                  (error.response && error.response.status >= 500);
                
                if (!isRetryable || attempt === maxRetries) {
                    break;
                }

                // Exponential backoff with jitter
                const delay = Math.min(1000 * Math.pow(2, attempt - 1) + Math.random() * 100, 10000);
                await new Promise(resolve => setTimeout(resolve, delay));
                
                logger.warn(`Retry attempt ${attempt} for ${serviceName}`, {
                    serviceName,
                    attempt,
                    delay,
                    error: error.message
                });
            }
        }

        throw lastError;
    }

    getCircuitState(serviceName) {
        const circuit = this.circuits.get(serviceName);
        if (!circuit) return null;
        
        return {
            state: circuit.status,
            stats: circuit.stats,
            openedAt: circuit.opened,
            lastError: circuit.lastError
        };
    }

    getAllMetrics() {
        return {
            circuits: Array.from(this.circuits.entries()).map(([name, circuit]) => ({
                name,
                state: circuit.status,
                stats: circuit.stats
            })),
            stateChanges: this.metrics.circuitStateChanges.slice(-100),
            retryAttempts: this.metrics.retryAttempts.slice(-100),
            fallbackTriggers: this.metrics.fallbackTriggers.slice(-100)
        };
    }
}

            const health = {

                uptime: process.uptime(),
                memory: process.memoryUsage(),

            // Check external dependencies

            health.dependencies = dependencyChecks;

            // Check circuit states
            const circuits = resilienceManager.getAllMetrics().circuits;
            health.circuits = circuits;

            // Determine overall status
            const unhealthyDeps = Object.values(dependencyChecks).filter(dep => dep.status !== 'healthy');
            const openCircuits = circuits.filter(circuit => circuit.state === 'open');
            
            if (unhealthyDeps.length > 0 || openCircuits.length > 0) {
                health.status = 'degraded';
                health.issues = {
                    dependencies: unhealthyDeps,
                    circuits: openCircuits
                };
            }

            const responseTime = Date.now() - startTime;
            logger.info('Health check completed', {
                status: health.status,
                responseTime,
                dependencies: Object.keys(health.dependencies).length
            });

            res.status(health.status === 'healthy' ? 200 : 503).json(health);
        } catch (error) {
            logger.error('Health check failed', { error: error.message });
            res.status(503).json({
                status: 'unhealthy',
                timestamp: new Date().toISOString(),
                error: 'Health check failed',
                details: process.env.NODE_ENV === 'development' ? error.message : undefined
            });
        }
    }

        try {

                status: 'healthy',
                latency: 0,
                timestamp: Date.now()

        } catch (error) {
            checks.database = {
                status: 'unhealthy',
                error: error.message,
                timestamp: Date.now()
            };
        }

            await redisClient.ping();

                status: 'healthy',
                latency: Date.now() - start,

                    service: attempt.service,
                    outcome: attempt.error ? 'failed' : 'success'
                });
            });

            res.end(await register.metrics());
        } catch (error) {
            logger.error('Failed to get metrics', { error: error.message });
            res.status(500).json({ error: 'Failed to collect metrics' });
        }

            httpRequestDurationMicroseconds

                .observe(duration);
            
            httpRequestsTotal

                .inc();
        });

# DIYAPP Evolution - V12 Core - Arquitetura de Microsserviços

## ADR-001: Arquitetura de Microsserviços V12

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead + Especialista Infra + Especialista Backend

### CONTEXTO:
A V11 do DIYAPP evoluiu para uma arquitetura monolítica que está enfrentando problemas de escalabilidade, deploy independente e resiliência. O sistema atual tem:
- Acoplamento forte entre módulos
- Tempos de deploy longos (30+ minutos)
- Dificuldade em escalar componentes específicos
- Single point of failure no core
- Dificuldade em adotar novas tecnologias por módulo

### DECISÃO:
Adotar arquitetura de microsserviços com os seguintes princípios:
1. **12 microsserviços independentes** com bounded contexts claros
2. **API Gateway** como único ponto de entrada
3. **Message Broker** (RabbitMQ) para comunicação assíncrona
4. **Service Discovery** via Consul
5. **Circuit Breaker pattern** em todas as chamadas síncronas
6. **Banco de dados por serviço** (Database per Service)
7. **Containerização** via Docker com orquestração Kubernetes

### OPÇÕES CONSIDERADAS:

**Opção A: Arquitetura de Microsserviços Completa** (ESCOLHIDA)
- Prós: Escalabilidade independente, deploy rápido, resiliência, polyglot persistence
- Contras: Complexidade operacional, necessidade de observabilidade robusta, latência de rede
- Justificativa: Alinha com meta de 100% autônoma e permite evolução independente por squad

**Opção B: Arquitetura de Módulos (Modular Monolith)**
- Prós: Mais simples operacionalmente, debugging mais fácil, transações ACID
- Contras: Escalabilidade limitada, deploy ainda lento, acoplamento técnico
- Justificativa: Rejeitada por não atender requisito de autonomia total por squad

**Opção C: Arquitetura Híbrida (Strangler Pattern)**
- Prós: Migração gradual, menor risco, permite coexistência
- Contras: Complexidade dupla durante transição, overhead de manutenção
- Justificativa: Rejeitada por diluir foco e atrasar benefícios completos

### CONSEQUÊNCIAS:

**Positivas:**
- Cada squad pode desenvolver, testar e deployar independentemente
- Escalabilidade granular (escalar apenas serviços sob carga)
- Resiliência (falha em um serviço não derruba sistema inteiro)
- Adoção de tecnologias específicas por serviço
- Deploy em segundos por serviço

**Negativas:**
- Complexidade de transações distribuídas
- Necessidade de observabilidade robusta (tracing distribuído)
- Overhead de comunicação de rede
- Gerenciamento de dados eventualmente consistentes

**Riscos:**
1. **Latência de rede**: Mitigar com cache Redis e comunicação assíncrona
2. **Consistência de dados**: Implementar SAGA pattern para transações
3. **Debugging complexo**: Implementar OpenTelemetry com Jaeger
4. **Deploy coordination**: Usar feature flags e canary releases

# 3. SISTEMA DE MONITORAMENTO DE SLOs
# scripts/monitor-canary.sh
#!/bin/bash
# Monitora SLOs durante deploy canary e decide rollback

set -e

VERSION=$1
NAMESPACE="diyapp-production"
SLO_AVAILABILITY=99.9
SLO_LATENCY=300
SLO_ERROR_RATE=0.1
CANARY_DURATION=120  # 2 minutos

echo "Starting canary monitoring for version: $VERSION"

# Função para coletar métricas do Prometheus
get_metric() {
    local metric_name=$1
    local query=$2
    local result=$(curl -s "http://prometheus.$NAMESPACE:9090/api/v1/query" \
        --data-urlencode "query=$query" | jq -r '.data.result[0].value[1]')
    echo $result
}

# Monitora por CANARY_DURATION segundos
for i in $(seq 1 $CANARY_DURATION); do
    echo "Check $i/$CANARY_DURATION"
    
    # 1. VERIFICA DISPONIBILIDADE
    availability=$(get_metric "availability" \
        "sum(rate(http_requests_total{app=\"diyapp-backend\", version=\"$VERSION\", status!~\"5..\"}[1m])) / sum(rate(http_requests_total{app=\"diyapp-backend\", version=\"$VERSION\"}[1m])) * 100")
    
    # 2. VERIFICA LATÊNCIA P95
    latency=$(get_metric "latency_p95" \
        "histogram_quantile(0.95, sum(rate(http_request_duration_seconds_bucket{app=\"diyapp-backend\", version=\"$VERSION\"}[1m])) by (le)) * 1000")
    
    # 3. VERIFICA ERROR RATE
    error_rate=$(get_metric "error_rate" \
        "sum(rate(http_requests_total{app=\"diyapp-backend\", version=\"$VERSION\", status=~\"5..\"}[1m])) / sum(rate(http_requests_total{app=\"diyapp-backend\", version=\"$VERSION\"}[1m])) * 100")
    
    echo "Metrics - Availability: ${availability:-0}%, Latency: ${latency:-0}ms, Error Rate: ${error_rate:-0}%"
    
    # VERIFICA VIOLAÇÃO DE SLOs
    if (( $(echo "${availability:-0} < $SLO_AVAILABILITY" | bc -l) )); then
        echo "SLO Violation: Availability ${availability}% < ${SLO_AVAILABILITY}%"
        exit 1
    fi

    def client(self) -> redis.Redis:
        if self._client is None or self._circuit_open:
            self._connect()
        return self._client
    
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=4, max=10)
    )
    def _connect(self):
        """Conecta ao Redis com retry exponencial"""
        try:
            self._client = redis.Redis(
                host=self.settings.REDIS_HOST,
                port=self.settings.REDIS_PORT,
                password=self.settings.REDIS_PASSWORD,
                db=self.settings.REDIS_DB,
                socket_timeout=self.settings.REDIS_TIMEOUT,
                socket_connect_timeout=self.settings.REDIS_TIMEOUT,
                decode_responses=False,  # Mantém bytes para pickle
            )
            # Testa conexão
            self._client.ping()
            self._circuit_open = False
            logger.info("redis_connected", host=self.settings.REDIS_HOST)
        except Exception as e:
            self._circuit_open = True
            logger.error("redis_connection_failed", error=str(e))
            raise
    
    @circuit(
        failure_threshold=settings.CIRCUIT_BREAKER_FAILURE_THRESHOLD,
        recovery_timeout=settings.CIRCUIT_BREAKER_RECOVERY_TIMEOUT
    )
    def get(self, key: str) -> Optional[Any]:
        """Obtém valor do cache com circuit breaker"""
        try:
            value = self.client.get(key)
            if value:
                return pickle.loads(value)
            return None
        except redis.RedisError as e:
            logger.error("redis_get_failed", key=key, error=str(e))
            return None  # Fallback: retorna None, não propaga erro
    
    @circuit(
        failure_threshold=settings.CIRCUIT_BREAKER_FAILURE_THRESHOLD,
        recovery_timeout=settings.CIRCUIT_BREAKER_RECOVERY_TIMEOUT
    )
    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Define valor no cache com TTL"""
        try:
            serialized = pickle.dumps(value)
            if ttl:
                return bool(self.client.setex(key, ttl, serialized))
            return bool(self.client.set(key, serialized))
        except redis.RedisError as e:
            logger.error("redis_set_failed", key=key, error=str(e))
            return False  # Fallback: cache falhou, mas não quebra o fluxo
    
    @circuit(
        failure_threshold=settings.CIRCUIT_BREAKER_FAILURE_THRESHOLD,
        recovery_timeout=settings.CIRCUIT_BREAKER_RECOVERY_TIMEOUT
    )
    def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        try:
            return bool(self.client.delete(key))
        except redis.RedisError as e:
            logger.error("redis_delete_failed", key=key, error=str(e))
            return False
    
    @circuit(
        failure_threshold=settings.CIRCUIT_BREAKER_FAILURE_THRESHOLD,
        recovery_timeout=settings.CIRCUIT_BREAKER_RECOVERY_TIMEOUT
    )
    def increment(self, key: str, amount: int = 1) -> Optional[int]:
        """Incrementa valor (para rate limiting)"""
        try:
            return self.client.incrby(key, amount)
        except redis.RedisError as e:
            logger.error("redis_increment_failed", key=key, error=str(e))
            return None
    
    def get_with_fallback(self, key: str, fallback_func, ttl: int = 300):
        """Pattern: Cache-Aside com fallback"""
        # Tenta cache primeiro
        cached = self.get(key)
        if cached is not None:
            return cached
        
        # Se não tem cache, executa função
        result = fallback_func()
        
        # Armazena no cache (async, não bloqueia resposta)
        if result is not None:
            self.set(key, result, ttl)
        
        return result

@contextmanager
def track_request(method: str, endpoint: str):
    """Context manager para trackear requests"""
    ACTIVE_REQUESTS.inc()
    start_time = time.time()
    
    try:
        yield
        status = "200"
    except Exception as e:
        status = "500" if getattr(e, 'status_code', 500) >= 500 else "400"
        raise
    finally:
        duration = time.time() - start_time
        REQUEST_LATENCY.labels(method=method, endpoint=endpoint).observe(duration)
        REQUEST_COUNT.labels(method=method, endpoint=endpoint, status=status).inc()
        ACTIVE_REQUESTS.dec()
        
        # Log estruturado
        logger.info(
            "request_completed",
            method=method,
            endpoint=endpoint,
            status=status,
            duration_ms=round(duration * 1000, 2),
            correlation_id="auto-generated"  # Será injetado pelo middleware
        )

export const useMediaQuery = (breakpoint: keyof typeof tokens.breakpoints) => {
  const query = `(min-width: ${tokens.breakpoints[breakpoint]})`;
  const [matches, setMatches] = React.useState(false);

  React.useEffect(() => {
    const media = window.matchMedia(query);
    setMatches(media.matches);
    
    const listener = (e: MediaQueryListEvent) => setMatches(e.matches);
    media.addEventListener('change', listener);
    
    return () => media.removeEventListener('change', listener);
  }, [query]);

  return matches;
};

export const responsiveProp = <T,>(
  value: T | T[],
  mapper: (val: T) => string
): string => {
  if (Array.isArray(value)) {
    return value.map((val, i) => {
      if (i === 0) return mapper(val);
      return `@media (min-width: ${Object.values(tokens.breakpoints)[i - 1]}) { ${mapper(val)} }`;
    }).join(' ');
  }
  return mapper(value);

export const useAsyncState = <T,>() => {
  const [state, setState] = React.useState<{
    data: T | null;
    loading: boolean;
    error: Error | null;
  }>({
    data: null,
    loading: false,
    error: null,
  });

  const execute = React.useCallback(async (promise: Promise<T>) => {
    setState(prev => ({ ...prev, loading: true, error: null }));
    
    try {
      const data = await promise;
      setState({ data, loading: false, error: null });
      return data;
    } catch (error) {
      setState({
        data: null,
        loading: false,
        error: error instanceof Error ? error : new Error('Unknown error'),
      });
      throw error;
    }
  }, []);

  return { ...state, execute };
};

# DIYAPP Evolution - V12 Core - Arquitetura de Microsserviços

## ADR-001: Arquitetura de Microsserviços V12

**Data:** 2024-03-15
**Status:** Aceita
**Autores:** Tech Lead, Especialista Infra, Especialista Backend

### CONTEXTO:
A V11 do DIYAPP evoluiu para uma arquitetura monolítica complexa que dificulta:
1. Deploy independente de componentes
2. Escalabilidade granular
3. Isolamento de falhas
4. Evolução tecnológica por componente

Forças em jogo:
- Necessidade de 100% autonomia operacional
- Time-to-market reduzido para novas features
- Manutenção de estabilidade durante transição
- Recursos limitados da squad (6 membros)

### DECISÃO:
Adotar arquitetura de microsserviços baseada em:
1. **6 serviços independentes** com responsabilidades bem definidas
2. **API Gateway** como ponto único de entrada
3. **Message Broker** para comunicação assíncrona
4. **Service Discovery** automático
5. **Observabilidade unificada** (logs, métricas, traces)

### OPÇÕES CONSIDERADAS:

**Opção A: Arquitetura Hexagonal (Ports & Adapters)**
- Prós: Desacoplamento total, testabilidade máxima, fácil substituição de tecnologias
- Contras: Complexidade inicial alta, curva de aprendizado, overhead de código

**Opção B: Clean Architecture**
- Prós: Separação clara de responsabilidades, independência de frameworks
- Contras: Muito boilerplate, difícil para times pequenos

**Opção C: Arquitetura em Camadas com Event Sourcing**
- Prós: Rastreabilidade completa, fácil debugging
- Contras: Complexidade operacional, necessidade de reprocessamento

**Opção escolhida: Arquitetura Hexagonal adaptada** - Justificativa: Balanceia desacoplamento com produtividade, alinha com expertise da squad, permite evolução gradual.

### CONSEQUÊNCIAS:
**Positivas:**
- Deploy independente por serviço
- Escalabilidade granular
- Stack tecnológica específica por domínio
- Isolamento de falhas
- Facilita contratação de especialistas por tecnologia

**Negativas:**
- Complexidade operacional aumentada
- Latência de rede entre serviços
- Gerenciamento distribuído de transações
- Monitoramento mais complexo

**Riscos:**
1. **Network partitioning:** Mitigar com circuit breakers e retry policies
2. **Consistência eventual:** Implementar SAGA pattern para transações distribuídas
3. **Observabilidade fragmentada:** Centralizar com OpenTelemetry + Grafana stack

**REVISÃO:** 2024-06-15 (3 meses para avaliar adoção)

// apps/user-service/src/application/createUser.usecase.ts
import { Injectable, Inject } from '@nestjs/common';
import { IUserRepository } from '../domain/repositories/user.repository.interface';
import { User } from '../domain/entities/user.entity';
import { CreateUserDTO } from '../interfaces/dto/create-user.dto';
import { EventPublisher } from '../infrastructure/events/event.publisher';
import { MetricsService } from '../infrastructure/observability/metrics.service';

@Injectable()
export class CreateUserUseCase {
  constructor(
    @Inject('IUserRepository') private readonly userRepository: IUserRepository,
    private readonly eventPublisher: EventPublisher,
    private readonly metrics: MetricsService,
  ) {}

  async execute(dto: CreateUserDTO): Promise<User> {
    const timer = this.metrics.startTimer('user_creation_duration');
    
    try {
      // Validação de domínio
      const user = User.create(dto);
      
      // Persistência
      const savedUser = await this.userRepository.save(user);
      
      // Publicação de evento de domínio
      await this.eventPublisher.publish('user.created', {
        userId: savedUser.id,
        email: savedUser.email,
        timestamp: new Date().toISOString(),
      });
      
      this.metrics.incrementCounter('users_created_total');
      return savedUser;
      
    } catch (error) {
      this.metrics.incrementCounter('user_creation_errors_total');
      throw error;
    } finally {
      timer();
    }
  }
}

import logging
from typing import Optional
from dataclasses import dataclass
from abc import ABC, abstractmethod
import aiosmtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
import opentelemetry.trace

logger = logging.getLogger(__name__)
tracer = opentelemetry.trace.get_tracer(__name__)

@dataclass
class EmailConfig:
    host: str
    port: int
    username: str
    password: str
    use_tls: bool = True
    timeout: int = 30

class IEmailAdapter(ABC):
    @abstractmethod
    async def send(self, to: str, subject: str, body: str) -> bool:
        pass

class SMTPEmailAdapter(IEmailAdapter):
    def __init__(self, config: EmailConfig):
        self.config = config
        self.client = aiosmtplib.SMTP(
            hostname=config.host,
            port=config.port,
            use_tls=config.use_tls,
            timeout=config.timeout,
        )
    
    async def send(self, to: str, subject: str, body: str) -> bool:
        with tracer.start_as_current_span("send_email") as span:
            span.set_attribute("email.to", to)
            span.set_attribute("email.subject", subject)
            
            message = MIMEMultipart("alternative")
            message["From"] = self.config.username
            message["To"] = to
            message["Subject"] = subject
            
            html_part = MIMEText(body, "html")
            message.attach(html_part)
            
            try:
                await self.client.connect()
                await self.client.login(
                    self.config.username, 
                    self.config.password
                )
                await self.client.send_message(message)
                
                logger.info(f"Email sent to {to}", extra={
                    "correlation_id": span.get_span_context().trace_id
                })
                return True
                
            except Exception as e:
                logger.error(f"Failed to send email to {to}: {str(e)}", 
                           extra={"error": str(e)})
                span.record_exception(e)
                return False
                
            finally:
                await self.client.quit()

groups:
- name: slo-rules
  interval: 30s
  rules:
  - record: sli:availability:ratio
    expr: |
      avg_over_time(
        up{job="diyapp-api-service", namespace="diyapp-production"}[5m]
      ) or vector(0)
  
  - record: sli:latency:p95
    expr: |
      histogram_quantile(0.95,
        sum by(le) (
          rate(http_request_duration_seconds_bucket{job="diyapp-api-service", status!~"5.."}[5m])
        )
      )
  
  - record: sli:error_rate:ratio
    expr: |
      sum(rate(http_requests_total{job="diyapp-api-service", status=~"5.."}[5m]))
      /
      sum(rate(http_requests_total{job="diyapp-api-service"}[5m]))
  
  - record: sli:llm_latency:p95
    expr: |
      histogram_quantile(0.95,
        sum by(le, provider) (
          rate(llm_request_duration_seconds_bucket{job="diyapp-llm-service"}[5m])
        )
      )
  
  - record: error_budget:availability:consumed
    expr: |
      (1 - sli:availability:ratio) / (1 - 0.999)
  
  - record: error_budget:latency:consumed
    expr: |
      max(0, sli:latency:p95 - 0.3) / 0.3
  
  - record: error_budget:error_rate:consumed
    expr: |
      max(0, sli:error_rate:ratio - 0.001) / 0.001
  
  - record: error_budget:total:remaining
    expr: |
      1 - (
        clamp_max(error_budget:availability:consumed, 1) * 0.4 +
        clamp_max(error_budget:latency:consumed, 1) * 0.3 +
        clamp_max(error_budget:error_rate:consumed, 1) * 0.3
      )
```

class RedisCache:
    """Cache distribuído com Redis com circuit breaker e fallback"""
    
    def __init__(self):
        self._client = None
        self._connected = False
        
    def connect(self) -> bool:
        """Conecta ao Redis com tratamento de erro"""
        try:
            self._client = redis.Redis(
                host=settings.REDIS_HOST,
                port=settings.REDIS_PORT,
                password=settings.REDIS_PASSWORD,
                db=settings.REDIS_DB,
                socket_connect_timeout=2,
                socket_timeout=2,
                retry_on_timeout=True,
                max_connections=50
            )
            # Test connection
            self._client.ping()
            self._connected = True
            logger.info("redis_connected", host=settings.REDIS_HOST)
            return True
        except RedisError as e:
            logger.error("redis_connection_failed", error=str(e))
            self._connected = False
            return False
    
    @circuit(failure_threshold=5, recovery_timeout=30)
    def get(self, key: str) -> Optional[Any]:
        """Obtém valor do cache com circuit breaker"""
        if not self._connected:
            return None
            
        try:
            value = self._client.get(key)
            if value:
                return json.loads(value)
            return None
        except RedisError as e:
            logger.warning("redis_get_failed", key=key, error=str(e))
            return None
    
    @circuit(failure_threshold=5, recovery_timeout=30)
    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Define valor no cache com circuit breaker"""
        if not self._connected:
            return False
            
        try:
            ttl = ttl or settings.CACHE_TTL
            serialized = json.dumps(value)
            result = self._client.setex(key, ttl, serialized)
            return bool(result)
        except RedisError as e:
            logger.warning("redis_set_failed", key=key, error=str(e))
            return False
    
    @circuit(failure_threshold=5, recovery_timeout=30)
    def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        if not self._connected:
            return False
            
        try:
            result = self._client.delete(key)
            return bool(result)
        except RedisError as e:
            logger.warning("redis_delete_failed", key=key, error=str(e))
            return False
    
    @circuit(failure_threshold=5, recovery_timeout=30)
    def increment(self, key: str, amount: int = 1) -> Optional[int]:
        """Incrementa valor atômico"""
        if not self._connected:
            return None
            
        try:
            return self._client.incrby(key, amount)
        except RedisError as e:
            logger.warning("redis_increment_failed", key=key, error=str(e))
            return None

class AsyncQueue:
    """Sistema de filas assíncronas com Celery + Redis"""
    
    def __init__(self):
        self._app = Celery(
            'diyapp_tasks',
            broker=settings.CELERY_BROKER_URL,
            backend=settings.CELERY_RESULT_BACKEND,
            include=['src.tasks.email_tasks', 'src.tasks.report_tasks']
        )
        
        # Configuração de performance
        self._app.conf.update(
            task_serializer='json',
            accept_content=['json'],
            result_serializer='json',
            timezone='UTC',
            enable_utc=True,
            task_track_started=True,
            task_time_limit=30 * 60,  # 30 minutos máximo
            task_soft_time_limit=25 * 60,  # 25 minutos soft limit
            worker_prefetch_multiplier=1,  # Prefetch baixo para tarefas longas
            worker_max_tasks_per_child=100,  # Reinicia worker após 100 tarefas
            broker_connection_retry_on_startup=True,
            broker_connection_max_retries=10,
            result_expires=3600,  # 1 hora
        )
    
    def send_task(self, task_name: str, args: tuple = None, kwargs: dict = None, 
                  priority: int = 5) -> str:
        """Envia tarefa para fila com prioridade"""
        try:
            task = self._app.send_task(
                task_name,
                args=args,
                kwargs=kwargs,
                queue='high_priority' if priority <= 3 else 'default'
            )
            logger.info("task_sent", task_id=task.id, task_name=task_name)
            return task.id
        except CeleryError as e:
            logger.error("task_send_failed", task_name=task_name, error=str(e))
            raise
    
    def get_task_result(self, task_id: str, timeout: int = 5) -> Any:
        """Obtém resultado de tarefa com timeout"""
        try:
            result = self._app.AsyncResult(task_id)
            return result.get(timeout=timeout)
        except CeleryError as e:
            logger.error("task_result_failed", task_id=task_id, error=str(e))
            return None

    'http_requests_active',
    'Active HTTP requests'

    'cache_hits_total',
    'Total cache hits',
    ['cache_type']
)

CACHE_MISSES = Counter(
    'cache_misses_total',
    'Total cache misses',
    ['cache_type']
)

QUEUE_TASKS = Counter(
    'queue_tasks_total',
    'Total tasks queued',
    ['task_type', 'priority']
)

@contextmanager
def track_request(method: str, endpoint: str):
    """Track request metrics with context manager"""
    ACTIVE_REQUESTS.inc()
    start_time = time.time()
    
    try:
        yield
        status = "success"
    except Exception:
        status = "error"
        raise
    finally:
        duration = time.time() - start_time
        ACTIVE_REQUESTS.dec()
        REQUEST_COUNT.labels(method=method, endpoint=endpoint, status=status).inc()
        REQUEST_LATENCY.labels(method=method, endpoint=endpoint).observe(duration)
        
        # Log performance
        if duration > 0.2:  # >200ms
            logger.warning("slow_request", 
                          method=method, 
                          endpoint=endpoint, 
                          duration_ms=duration*1000)

import React, { forwardRef } from 'react';
import { tokens } from '../../design-system/tokens';
import { Spinner } from '../Spinner/Spinner';
import './Button.css';

export type ButtonVariant = 'primary' | 'secondary' | 'outline' | 'ghost' | 'danger';
export type ButtonSize = 'xs' | 'sm' | 'md' | 'lg';

export interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: ButtonVariant;
  size?: ButtonSize;
  isLoading?: boolean;
  loadingText?: string;
  leftIcon?: React.ReactNode;
  rightIcon?: React.ReactNode;
  fullWidth?: boolean;
}

export const Button = forwardRef<HTMLButtonElement, ButtonProps>(
  (
    {
      children,
      variant = 'primary',
      size = 'md',
      isLoading = false,
      loadingText,
      leftIcon,
      rightIcon,
      fullWidth = false,
      disabled,
      className = '',
      ...props
    },
    ref
  ) => {
    const isDisabled = disabled || isLoading;

    const getVariantStyles = () => {
      const base = 'button-base';
      
      switch (variant) {
        case 'primary':
          return `${base} button-primary`;
        case 'secondary':
          return `${base} button-secondary`;
        case 'outline':
          return `${base} button-outline`;
        case 'ghost':
          return `${base} button-ghost`;
        case 'danger':
          return `${base} button-danger`;
        default:
          return `${base} button-primary`;
      }
    };

    const getSizeStyles = () => {
      switch (size) {
        case 'xs':
          return 'button-xs';
        case 'sm':
          return 'button-sm';
        case 'lg':
          return 'button-lg';
        default:
          return 'button-md';
      }
    };

    const buttonClasses = [
      getVariantStyles(),
      getSizeStyles(),
      fullWidth ? 'button-full-width' : '',
      isDisabled ? 'button-disabled' : '',
      className,
    ]
      .filter(Boolean)
      .join(' ');

    return (
      <button
        ref={ref}
        className={buttonClasses}
        disabled={isDisabled}
        aria-busy={isLoading}
        {...props}
      >
        {isLoading && (
          <span className="button-loading-spinner">
            <Spinner size={size === 'xs' || size === 'sm' ? 'sm' : 'md'} />
          </span>
        )}
        
        {!isLoading && leftIcon && (
          <span className="button-icon-left" aria-hidden="true">
            {leftIcon}
          </span>
        )}
        
        <span className="button-content">
          {isLoading && loadingText ? loadingText : children}
        </span>
        
        {!isLoading && rightIcon && (
          <span className="button-icon-right" aria-hidden="true">
            {rightIcon}
          </span>
        )}
      </button>
    );
  }
);

Button.displayName = 'Button';

import React, { forwardRef } from 'react';
import './Card.css';

export interface CardProps extends React.HTMLAttributes<HTMLDivElement> {
  variant?: 'elevated' | 'outline' | 'filled';
  padding?: 'none' | 'sm' | 'md' | 'lg';
  hoverable?: boolean;
}

export const Card = forwardRef<HTMLDivElement, CardProps>(
  (
    {
      children,
      variant = 'elevated',
      padding = 'md',
      hoverable = false,
      className = '',
      ...props
    },
    ref
  ) => {
    const cardClasses = [
      'card',
      `card-${variant}`,
      `card-padding-${padding}`,
      hoverable ? 'card-hoverable' : '',
      className,
    ]
      .filter(Boolean)
      .join(' ');

    return (
      <div ref={ref} className={cardClasses} {...props}>
        {children}
      </div>
    );
  }
);

Card.displayName = 'Card';

export interface CardHeaderProps extends React.HTMLAttributes<HTMLDivElement> {
  title?: string;
  subtitle?: string;
  action?: React.ReactNode;
}

export const CardHeader = forwardRef<HTMLDivElement, CardHeaderProps>(
  ({ title, subtitle, action, children, className = '', ...props }, ref) => {
    return (
      <div ref={ref} className={`card-header ${className}`} {...props}>
        {(title || subtitle) && (
          <div className="card-header-content">
            {title && <h3 className="card-title">{title}</h3>}
            {subtitle && <p className="card-subtitle">{subtitle}</p>}
          </div>
        )}
        {action && <div className="card-header-action">{action}</div>}
        {children}
      </div>
    );
  }
);

CardHeader.displayName = 'CardHeader';
