const ADR_001 = `
# ADR-001: Arquitetura de Microsserviços/Plugins V12

## CONTEXTO:
O DIYAPP V11 possui arquitetura monolítica que limita:
1. Escalabilidade independente de componentes
2. Deploy contínuo sem downtime total
3. Isolamento de falhas entre funcionalidades
4. Adoção de diferentes stacks tecnológicas por domínio

Forças em jogo:
- Necessidade de 100% estabilidade autônoma
- Time distribuído trabalhando em paralelo (Modo Hive)
- Manutenibilidade de longo prazo
- Performance em cenários de alta carga

## DECISÃO:
Adotar arquitetura de microsserviços orientados a eventos com sistema de plugins dinâmicos.

Componentes principais:
1. **Core Service**: API Gateway + Service Registry + Plugin Manager
2. **Microsserviços Independentes**: Cada domínio de negócio como serviço isolado
3. **Message Bus**: Event-driven communication entre serviços
4. **Plugin System**: Carregamento dinâmico de funcionalidades
5. **Shared Library**: Utilitários comuns versionados independentemente

## OPÇÕES CONSIDERADAS:

### Opção A: Monólito Modularizado (Status Quo)
- Prós: Simplicidade inicial, deploy único, debugging centralizado
- Contras: Acoplamento alto, escalabilidade limitada, risco de falha total

### Opção B: Microsserviços Tradicionais REST
- Prós: Desacoplamento, deploy independente, stack específica por serviço
- Contras: Complexidade de comunicação síncrona, latência acumulada

### Opção C: Microsserviços Orientados a Eventos + Plugins (ESCOLHIDA)
- Prós: Desacoplamento máximo, resiliência a falhas, escalabilidade granular
- Contras: Complexidade inicial maior, necessidade de infraestrutura de eventos
- Justificativa: Alinha com necessidade de estabilidade 100% autônoma e Modo Hive

## CONSEQUÊNCIAS:

### Positivas:
- Cada squad pode desenvolver/deployar independentemente
- Falhas isoladas não derrubam sistema inteiro
- Plugins podem ser ativados/desativados sem downtime
- Escalabilidade granular por serviço

### Negativas:
- Maior complexidade operacional
- Necessidade de observabilidade distribuída
- Overhead de comunicação entre serviços

### Riscos:
1. **Event Storm**: Mitigar com schema validation e rate limiting
2. **Data Consistency**: Mitigar com SAGA pattern e compensação
3. **Debugging Distribuído**: Mitigar com correlation IDs e logging centralizado

## REVISÃO:
Reavaliar em 6 meses (2024-07-15) após métricas de produção
`;

class PluginSystem {
  private plugins: Map<string, Plugin> = new Map();
  private eventBus: EventBus;
  
  constructor(eventBus: EventBus) {
    this.eventBus = eventBus;
  }
  
  async loadPlugin(pluginPath: string, config: PluginConfig): Promise<Plugin> {
    try {
      // Validação do plugin
      await this.validatePlugin(pluginPath);
      
      // Carregamento dinâmico
      const pluginModule = await import(pluginPath);
      const plugin = new pluginModule.default(config);
      
      // Validação de interface
      this.validatePluginInterface(plugin);
      
      // Registro
      this.plugins.set(plugin.id, plugin);
      
      // Inicialização
      await plugin.initialize();
      
      // Subscribe to events
      this.setupEventHandlers(plugin);
      
      console.log(`Plugin ${plugin.id} loaded successfully`);
      return plugin;
    } catch (error) {
      console.error(`Failed to load plugin ${pluginPath}:`, error);
      throw new PluginLoadError(`Failed to load plugin: ${error.message}`);
    }
  }
  
  async unloadPlugin(pluginId: string): Promise<void> {
    const plugin = this.plugins.get(pluginId);
    if (!plugin) {
      throw new Error(`Plugin ${pluginId} not found`);
    }
    
    try {
      await plugin.cleanup();
      this.plugins.delete(pluginId);
      console.log(`Plugin ${pluginId} unloaded successfully`);
    } catch (error) {
      console.error(`Failed to unload plugin ${pluginId}:`, error);
      throw error;
    }
  }

  }
  
  private validatePluginInterface(plugin: Plugin): void {
    const requiredMethods = ['initialize', 'cleanup', 'handleEvent'];
    for (const method of requiredMethods) {
      if (typeof plugin[method] !== 'function') {
        throw new Error(`Plugin missing required method: ${method}`);
      }
    }
  }
  
  private setupEventHandlers(plugin: Plugin): void {
    if (plugin.eventHandlers) {
      for (const [eventType, handler] of Object.entries(plugin.eventHandlers)) {
        this.eventBus.subscribe(eventType, async (event) => {
          try {
            await handler.call(plugin, event);
          } catch (error) {
            console.error(`Plugin ${plugin.id} failed to handle event ${eventType}:`, error);
          }
        });
      }
    }
  }
  
  getPlugin(pluginId: string): Plugin | undefined {
    return this.plugins.get(pluginId);
  }
  
  listPlugins(): PluginInfo[] {
    return Array.from(this.plugins.values()).map(p => ({
      id: p.id,
      name: p.name,
      version: p.version,
      enabled: p.enabled,
      description: p.description
    }));
  }
}

  enabled: boolean;
  description: string;

  handleEvent?(event: any): Promise<void>;
  eventHandlers?: Record<string, Function>;

// src/components/VirtualList.tsx
import React, { useState, useRef, useEffect, useCallback, memo } from 'react';
import { useDesignSystem } from '../design-system';

interface VirtualListProps<T> {
  items: T[];
  itemHeight: number;
  renderItem: (item: T, index: number) => React.ReactNode;
  overscan?: number;
  className?: string;
  'aria-label'?: string;
}

const VirtualListComponent = <T,>({
  items,
  itemHeight,
  renderItem,
  overscan = 3,
  className = '',
  'aria-label': ariaLabel = 'List of items'
}: VirtualListProps<T>) => {
  const tokens = useDesignSystem();
  const containerRef = useRef<HTMLDivElement>(null);
  const [scrollTop, setScrollTop] = useState(0);
  const [containerHeight, setContainerHeight] = useState(0);

  // Calculate visible items
  const totalHeight = items.length * itemHeight;
  const startIndex = Math.max(0, Math.floor(scrollTop / itemHeight) - overscan);
  const endIndex = Math.min(
    items.length - 1,
    Math.floor((scrollTop + containerHeight) / itemHeight) + overscan
  );

  const visibleItems = items.slice(startIndex, endIndex + 1);

  // Handle scroll
  const handleScroll = useCallback(() => {
    if (containerRef.current) {
      setScrollTop(containerRef.current.scrollTop);
    }
  }, []);

  // Measure container height
  useEffect(() => {
    const updateHeight = () => {
      if (containerRef.current) {
        setContainerHeight(containerRef.current.clientHeight);
      }
    };

    updateHeight();
    window.addEventListener('resize', updateHeight);
    
    return () => window.removeEventListener('resize', updateHeight);
  }, []);

  return (
    <div
      ref={containerRef}
      className={`virtual-list ${className}`}
      onScroll={handleScroll}
      style={{
        height: '100%',
        overflowY: 'auto',
        position: 'relative',
        backgroundColor: tokens.colors.background.primary
      }}
      role="list"
      aria-label={ariaLabel}
      aria-setsize={items.length}
      aria-posinset={startIndex + 1}
    >
      {/* Spacer for total height */}
      <div style={{ height: `${totalHeight}px` }} />

      {/* Visible items positioned absolutely */}
      <div
        style={{
          position: 'absolute',
          top: 0,
          left: 0,
          right: 0,
          transform: `translateY(${startIndex * itemHeight}px)`
        }}
      >
        {visibleItems.map((item, index) => (
          <div
            key={startIndex + index}
            style={{
              height: `${itemHeight}px`,
              padding: tokens.spacing.md,
              borderBottom: `1px solid ${tokens.colors.border.primary}`
            }}
            role="listitem"
            aria-setsize={items.length}
            aria-posinset={startIndex + index + 1}
          >
            {renderItem(item, startIndex + index)}
          </div>
        ))}
      </div>
    </div>
  );
};

// Memoize to prevent unnecessary re-renders
export const VirtualList = memo(VirtualListComponent) as typeof VirtualListComponent;

# Circuit breaker implementation
class CircuitBreaker:
    def __init__(self, failure_threshold=5, recovery_timeout=60):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.failures = 0
        self.last_failure_time = None
        self.state = "closed"  # closed, open, half-open
        
    def record_failure(self):
        self.failures += 1
        self.last_failure_time = time.time()
        if self.failures >= self.failure_threshold:
            self.state = "open"
            logger.warning(f"Circuit breaker opened after {self.failures} failures")
            
    def record_success(self):
        self.failures = 0
        self.state = "closed"
        
    def can_execute(self):
        if self.state == "closed":
            return True
        elif self.state == "open":
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = "half-open"
                return True
            return False
        return True  # half-open

# Rate limiting middleware
class RateLimiter:
    def __init__(self, redis_client: redis.Redis, limit: int = 100, window: int = 60):
        self.redis = redis_client
        self.limit = limit
        self.window = window
        
    async def is_rate_limited(self, key: str) -> bool:
        """Check if request should be rate limited"""
        try:
            current = await self.redis.get(key)
            if current is None:
                await self.redis.setex(key, self.window, 1)
                return False
            
            current_count = int(current)
            if current_count >= self.limit:
                return True
                
            await self.redis.incr(key)
            return False
        except Exception as e:
            # Fail open - don't block requests if Redis fails
            logger.error(f"Rate limiting error: {str(e)}")
            return False

# Cache service
class CacheService:
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.default_ttl = Config.CACHE_TTL
        
    async def get(self, key: str) -> Optional[str]:
        """Get value from cache with circuit breaker"""
        if not redis_circuit_breaker.can_execute():
            return None
            
        try:
            value = await self.redis.get(key)
            if value:
                logger.info(f"Cache hit for key: {key}")
            return value
        except Exception as e:
            redis_circuit_breaker.record_failure()
            logger.error(f"Cache get error: {str(e)}")
            return None
            
    async def set(self, key: str, value: str, ttl: Optional[int] = None) -> bool:
        """Set value in cache with circuit breaker"""
        if not redis_circuit_breaker.can_execute():
            return False
            
        try:
            ttl = ttl or self.default_ttl
            await self.redis.setex(key, ttl, value)
            redis_circuit_breaker.record_success()
            return True
        except Exception as e:
            redis_circuit_breaker.record_failure()
            logger.error(f"Cache set error: {str(e)}")
            return False
            
    async def delete(self, key: str) -> bool:
        """Delete key from cache"""
        try:
            await self.redis.delete(key)
            return True
        except Exception as e:
            logger.error(f"Cache delete error: {str(e)}")
            return False
            
    async def get_or_set(self, key: str, fetch_func, ttl: Optional[int] = None) -> Any:
        """Get from cache or fetch and set"""
        cached = await self.get(key)
        if cached is not None:
            return json.loads(cached)
            
        data = await fetch_func()
        if data is not None:
            await self.set(key, json.dumps(data), ttl)
        return data

                ROUND((idx_scan::NUMERIC / (idx_scan + seq_scan + 1) * 100), 2)
            ELSE 0 
        END as index_usage_ratio
    FROM pg_stat_user_indexes
    JOIN pg_index USING (indexrelid)
    WHERE NOT indisunique
    ORDER BY index_usage_ratio ASC, pg_relation_size(indexrelid) DESC;
END;
$$ LANGUAGE plpgsql;

    mean_exec_time,
    rows,
    shared_blks_hit,
    shared_blks_read

LIMIT 50;

    schemaname,

    pg_size_pretty(pg_total_relation_size(schemaname || '.' || tablename)) as total_size,
    pg_size_pretty(pg_relation_size(schemaname || '.' || tablename)) as table_size,
    pg_size_pretty(pg_total_relation_size(schemaname || '.' || tablename) - 
                   pg_relation_size(schemaname || '.' || tablename)) as index_size,
    n_live_tup as live_rows,
    n_dead_tup as dead_rows,
    ROUND((n_dead_tup::NUMERIC / GREATEST(n_live_tup + n_dead_tup, 1)) * 100, 2) as dead_row_percentage
FROM pg_stat_user_tables

```python
# auth/oauth2_service.py
"""
Serviço OAuth2.1 com PKCE e MFA
Implementação conforme RFC 6749, RFC 7636, OAuth 2.1 Draft
"""

import os
import secrets
import hashlib
import base64
import json
import time
from datetime import datetime, timedelta
from typing import Dict, Optional, Tuple, List
import jwt
from cryptography.hazmat.primitives import serialization
from cryptography.hazmat.primitives.asymmetric import rsa
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.primitives import hashes
import bcrypt
import pyotp
import qrcode
from io import BytesIO
import base64

class OAuth21Service:
    def __init__(self):
        self.private_key = self._load_or_generate_private_key()
        self.public_key = self.private_key.public_key()
        self.auth_codes = {}  # code_verifier -> auth_data
        self.access_tokens = {}  # token -> token_data
        self.refresh_tokens = {}  # token -> refresh_data
        self.mfa_secrets = {}  # user_id -> mfa_secret
        
    def _load_or_generate_private_key(self):
        """Carrega ou gera chave privada RSA para assinatura JWT"""
        key_path = "keys/oauth_private.pem"
        
        if os.path.exists(key_path):
            with open(key_path, "rb") as f:
                private_key = serialization.load_pem_private_key(
                    f.read(),
                    password=None
                )
        else:
            # Gerar nova chave RSA 4096
            private_key = rsa.generate_private_key(
                public_exponent=65537,
                key_size=4096
            )
            
            os.makedirs("keys", exist_ok=True)
            with open(key_path, "wb") as f:
                f.write(private_key.private_bytes(
                    encoding=serialization.Encoding.PEM,
                    format=serialization.PrivateFormat.PKCS8,
                    encryption_algorithm=serialization.NoEncryption()
                ))
            
            # Salvar chave pública
            pub_path = "keys/oauth_public.pem"
            with open(pub_path, "wb") as f:
                f.write(private_key.public_key().public_bytes(
                    encoding=serialization.Encoding.PEM,
                    format=serialization.PublicFormat.SubjectPublicKeyInfo
                ))
        
        return private_key
    
    def generate_pkce_challenge(self, code_verifier: str) -> Tuple[str, str]:
        """Gera code_challenge a partir do code_verifier (PKCE S256)"""
        # SHA-256 do code_verifier
        digest = hashlib.sha256(code_verifier.encode()).digest()
        # Base64 URL-safe
        code_challenge = base64.urlsafe_b64encode(digest).decode().replace('=', '')
        
        return code_challenge, code_verifier
    
    def validate_pkce(self, code_verifier: str, code_challenge: str) -> bool:
        """Valida PKCE code_verifier contra code_challenge"""
        expected_challenge, _ = self.generate_pkce_challenge(code_verifier)
        return secrets.compare_digest(expected_challenge, code_challenge)
    
    def create_auth_code(self, client_id: str, user_id: str, scope: List[str], 
                        code_challenge: str, redirect_uri: str) -> str:
        """Cria authorization code com PKCE"""
        code = secrets.token_urlsafe(32)
        code_verifier = secrets.token_urlsafe(32)
        
        # Validar que o code_verifier gera o code_challenge
        if not self.validate_pkce(code_verifier, code_challenge):
            raise ValueError("Invalid PKCE code_challenge")
        
        self.auth_codes[code] = {
            "client_id": client_id,
            "user_id": user_id,
            "scope": scope,
            "code_verifier": code_verifier,
            "redirect_uri": redirect_uri,
            "created_at": datetime.now(),
            "expires_at": datetime.now() + timedelta(minutes=10)
        }
        
        return code
    
    def exchange_code_for_token(self, code: str, code_verifier: str, 
                               client_id: str, redirect_uri: str) -> Dict:
        """Exchange authorization code for access token (PKCE validation)"""
        auth_data = self.auth_codes.get(code)
        
        if not auth_data:
            raise ValueError("Invalid authorization code")
        
        if datetime.now() > auth_data["expires_at"]:
            del self.auth_codes[code]
            raise ValueError("Authorization code expired")
        
        if auth_data["client_id"] != client_id:
            raise ValueError("Client ID mismatch")
        
        if auth_data["redirect_uri"] != redirect_uri:
            raise ValueError("Redirect URI mismatch")
        
        # Validar PKCE
        if not self.validate_pkce(code_verifier, auth_data["code_verifier"]):
            raise ValueError("PKCE validation failed")
        
        # Gerar tokens
        access_token = self._generate_access_token(
            auth_data["user_id"],
            auth_data["scope"],
            client_id
        )
        
        refresh_token = self._generate_refresh_token(
            auth_data["user_id"],
            auth_data["scope"],
            client_id
        )
        
        # Limpar código usado
        del self.auth_codes[code]
        
        return {
            "access_token": access_token,
            "token_type": "Bearer",
            "expires_in": 900,  # 15 minutos
            "refresh_token": refresh_token,
            "scope": " ".join(auth_data["scope"])
        }
    
    def _generate_access_token(self, user_id: str, scope: List[str], 
                              client_id: str) -> str:
        """Gera JWT access token"""
        now = datetime.now()
        expires = now + timedelta(minutes=15)  # Tokens curtos
        
        payload = {
            "iss": "diyapp-v12",
            "sub": user_id,
            "aud": client_id,
            "exp": int(expires.timestamp()),
            "iat": int(now.timestamp()),
            "jti": secrets.token_urlsafe(16),
            "scope": " ".join(scope),
            "token_type": "access",
            "mfa_verified": True  # Assumindo MFA já verificado
        }
        
        token = jwt.encode(
            payload,
            self.private_key,
            algorithm="RS256"
        )
        
        self.access_tokens[token] = {
            "user_id": user_id,
            "expires_at": expires,
            "scope": scope,
            "client_id": client_id
        }
        
        return token
    
    def _generate_refresh_token(self, user_id: str, scope: List[str], 
                               client_id: str) -> str:
        """Gera refresh token (opaco, armazenado com hash)"""
        token = secrets.token_urlsafe(64)
        token_hash = hashlib.sha256(token.encode()).hexdigest()
        
        expires = datetime.now() + timedelta(days=7)  # Refresh tokens de 7 dias
        
        self.refresh_tokens[token_hash] = {
            "user_id": user_id,
            "expires_at": expires,
            "scope": scope,
            "client_id": client_id,
            "last_used": datetime.now()
        }
        
        return token
    
    def refresh_access_token(self, refresh_token: str, client_id: str) -> Dict:
        """Usa refresh token para obter novo access token"""
        token_hash = hashlib.sha256(refresh_token.encode()).hexdigest()
        refresh_data = self.refresh_tokens.get(token_hash)
        
        if not refresh_data:
            raise ValueError("Invalid refresh token")
        
        if datetime.now() > refresh_data["expires_at"]:

            raise ValueError("Refresh token expired")
        
        if refresh_data["client_id"] != client_id:
            raise ValueError("Client ID mismatch")
        
        # Rotação de refresh token (RTTR)
        del self.refresh_tokens[token_hash]
        
        # Gerar novos tokens
        access_token = self._generate_access_token(
            refresh_data["user_id"],
            refresh_data["scope"],

            refresh_data["scope"],

        return {
            "access_token": access_token,
            "token_type": "Bearer",
            "expires_in": 900,
            "refresh_token": new_refresh_token,
            "scope": " ".join(refresh_data["scope"])
