## 1. Princípios Fundamentais

### 1.1 SOLID + Clean Architecture
- **S**ingle Responsibility: Cada classe/função tem um único motivo para mudar
- **O**pen/Closed: Aberto para extensão, fechado para modificação
- **L**iskov Substitution: Subtipos devem ser substituíveis por seus tipos base
- **I**nterface Segregation: Múltiplas interfaces específicas são melhores que uma geral
- **D**ependency Inversion: Depender de abstrações, não de implementações

```
Presentation → Application → Domain → Infrastructure
```

1. Codebase: Um repositório por serviço
2. Dependencies: Declaradas explicitamente
3. Config: Armazenada em ambiente
4. Backing Services: Tratados como recursos anexados
5. Build, release, run: Estágios estritamente separados
6. Processes: Stateless, compartilham nada
7. Port binding: Auto-contidos
8. Concurrency: Scale-out via processos
9. Disposability: Inicialização rápida, shutdown gracioso
10. Dev/prod parity: Ambientes similares

12. Admin processes: Executados como processos one-off

export class CacheService {
  private readonly DEFAULT_TTL = 300; // 5 minutos
  
  async getOrSet<T>(
    key: string,
    fetchFn: () => Promise<T>,
    ttl: number = this.DEFAULT_TTL
  ): Promise<T> {
    const cached = await this.redis.get(key);
    
    if (cached) {
      return JSON.parse(cached) as T;
    }
    
    const data = await fetchFn();
    await this.redis.setex(key, ttl, JSON.stringify(data));
    
    return data;
  }
  
  async invalidatePattern(pattern: string): Promise<void> {
    const keys = await this.redis.keys(pattern);
    if (keys.length > 0) {
      await this.redis.del(...keys);
    }
  }
}

      },
      process.env.JWT_SECRET!,
      { expiresIn: '24h' }
    );
  }

    return jwt.verify(token, process.env.JWT_SECRET!) as JwtPayload;
  }
}

export const requireRole = (...roles: UserRole[]) => {
  return (req: Request, res: Response, next: NextFunction) => {
    if (!req.user) {
      throw new UnauthorizedError('Authentication required');
    }
    
    if (!roles.includes(req.user.role)) {
      throw new ForbiddenError('Insufficient permissions');
    }
    
    next();
  };

  encrypt(text: string): string {
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv(
      this.algorithm,
      Buffer.from(process.env.ENCRYPTION_KEY!, 'hex'),
      iv
    );
    
    let encrypted = cipher.update(text, 'utf8', 'hex');
    encrypted += cipher.final('hex');

    return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;

import time
from typing import Any, Callable, Optional
from functools import wraps
import logging
from dataclasses import dataclass, field
from enum import Enum

logger = logging.getLogger(__name__)

class CircuitState(Enum):
    CLOSED = "CLOSED"      # Operação normal
    OPEN = "OPEN"          # Circuito aberto - falhas
    HALF_OPEN = "HALF_OPEN" # Testando recuperação

@dataclass
class CircuitBreakerConfig:
    failure_threshold: int = 5          # Falhas para abrir circuito
    recovery_timeout: int = 30          # Segundos antes de tentar half-open
    expected_exceptions: tuple = (Exception,)  # Exceções que contam como falha
    name: str = "default"

@dataclass
class CircuitBreakerMetrics:
    failures: int = 0
    successes: int = 0
    state_changes: list = field(default_factory=list)
    last_failure_time: Optional[float] = None

class CircuitBreaker:
    """Implementação de Circuit Breaker com estado thread-safe"""
    
    def __init__(self, config: CircuitBreakerConfig):
        self.config = config
        self.state = CircuitState.CLOSED
        self.metrics = CircuitBreakerMetrics()
        self._lock = False
        
    def call(self, func: Callable, *args, **kwargs) -> Any:
        """Executa função com proteção de circuit breaker"""
        
        # Se circuito aberto, verifica timeout
        if self.state == CircuitState.OPEN:
            if time.time() - self.metrics.last_failure_time > self.config.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
                logger.info(f"Circuit {self.config.name} moved to HALF_OPEN")
            else:
                raise CircuitOpenException(f"Circuit {self.config.name} is OPEN")
        
        try:
            result = func(*args, **kwargs)
            
            # Sucesso em half-open fecha circuito
            if self.state == CircuitState.HALF_OPEN:
                self.state = CircuitState.CLOSED
                self.metrics.failures = 0
                logger.info(f"Circuit {self.config.name} reset to CLOSED")
            
            self.metrics.successes += 1
            return result
            
        except self.config.expected_exceptions as e:
            self.metrics.failures += 1
            self.metrics.last_failure_time = time.time()
            
            logger.warning(f"Circuit {self.config.name} failure {self.metrics.failures}/{self.config.failure_threshold}: {str(e)}")
            
            # Abre circuito se threshold atingido
            if self.metrics.failures >= self.config.failure_threshold:
                self.state = CircuitState.OPEN
                logger.error(f"Circuit {self.config.name} OPENED due to {self.metrics.failures} failures")
            
            raise
    
    def get_metrics(self) -> dict:
        """Retorna métricas para observabilidade"""
        return {
            "name": self.config.name,
            "state": self.state.value,
            "failures": self.metrics.failures,
            "successes": self.metrics.successes,
            "last_failure_time": self.metrics.last_failure_time,
            "is_open": self.state == CircuitState.OPEN
        }

class CircuitOpenException(Exception):
    """Exceção lançada quando circuito está aberto"""
    pass

def circuit_breaker(config: CircuitBreakerConfig):
    """Decorator para circuit breaker"""
    breaker = CircuitBreaker(config)
    
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            return breaker.call(func, *args, **kwargs)
        
        # Expõe métricas via função
        wrapper.get_metrics = breaker.get_metrics
        wrapper.breaker = breaker
        
        return wrapper
    return decorator

class RateLimitStrategy(Enum):
    FIXED_WINDOW = "fixed_window"

class RateLimiter:
    """Rate Limiter distribuído usando Redis"""
    
    def __init__(self, redis_url: str, strategy: RateLimitStrategy = RateLimitStrategy.SLIDING_WINDOW):
        self.redis = redis.from_url(redis_url, decode_responses=True)
        self.strategy = strategy
        
        # Configura storage baseado em Redis
        self.storage = storage.RedisStorage(redis_url)
        self.limiter = strategies.MovingWindowRateLimiter(self.storage)
    
    def is_rate_limited(self, key: str, limit: RateLimitItem) -> Tuple[bool, dict]:
        """
        Verifica se requisição excede rate limit
        
        Args:
            key: Identificador único (ex: user_id:ip:endpoint)
            limit: RateLimitItem (ex: "100 per minute")
            
        Returns:
            Tuple[bool, dict]: (is_limited, metadata)
        """
        try:
            # Usa limits library para lógica complexa
            if self.limiter.hit(limit, key):
                return False, {"remaining": limit.amount - 1}
            else:
                # Calcula tempo para reset
                reset_time = self._get_reset_time(key, limit)
                return True, {
                    "reset_time": reset_time,
                    "limit": limit.amount,
                    "window": limit.get_expiry()
                }
        except Exception as e:
            logger.error(f"Rate limiter error: {str(e)}")
            # Em caso de falha no Redis, permite requisição (fail open)
            return False, {"error": "limiter_unavailable"}
    
    def _get_reset_time(self, key: str, limit: RateLimitItem) -> int:
        """Calcula tempo para reset do rate limit"""
        # Implementação simplificada - na prática usaría lógica do limits
        window = limit.get_expiry()
        current_window = int(time.time() / window) * window
        return current_window + window
    
    def get_limits_for_key(self, key: str) -> dict:
        """Retorna status atual dos rate limits para uma chave"""
        # Implementação para múltiplos limites por chave
        return {
            "key": key,
            "timestamp": time.time(),
            "strategy": self.strategy.value
        }

WITH daily_users AS (
    SELECT
        event_date,
        user_id,
        COUNT(DISTINCT session_id) AS sessions_count,
        COUNT(*) AS events_count,
        MIN(event_timestamp) AS first_event_time,
        MAX(event_timestamp) AS last_event_time,
        ARRAY_AGG(DISTINCT event_type) AS event_types
    FROM {{ ref('stg_user_events') }}
    GROUP BY 1, 2
),

daily_metrics AS (
    SELECT
        event_date,
        COUNT(DISTINCT user_id) AS daily_active_users,
        COUNT(DISTINCT CASE WHEN sessions_count >= 3 THEN user_id END) AS engaged_users,
        COUNT(DISTINCT session_id) AS total_sessions,
        AVG(sessions_count) AS avg_sessions_per_user,
        AVG(events_count) AS avg_events_per_user,
        SUM(events_count) AS total_events
    FROM daily_users
    GROUP BY 1
),

rolling_metrics AS (
    SELECT
        event_date,
        daily_active_users,
        engaged_users,
        total_sessions,
        avg_sessions_per_user,
        avg_events_per_user,
        total_events,
        AVG(daily_active_users) OVER (ORDER BY event_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS dau_7d_avg,
        AVG(engaged_users) OVER (ORDER BY event_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS engaged_7d_avg
    FROM daily_metrics
)

SELECT
    event_date,
    daily_active_users,
    engaged_users,
    total_sessions,
    avg_sessions_per_user,
    avg_events_per_user,
    total_events,
    dau_7d_avg,
    engaged_7d_avg,
    CASE
        WHEN LAG(daily_active_users) OVER (ORDER BY event_date) > 0
        THEN ROUND((daily_active_users - LAG(daily_active_users) OVER (ORDER BY event_date)) * 100.0 / LAG(daily_active_users) OVER (ORDER BY event_date), 2)
        ELSE NULL
    END AS dau_growth_pct,
    CURRENT_TIMESTAMP AS _calculated_at
FROM rolling_metrics

  // STRIDE: MFA obrigatório para contas administrativas
  async setupMFA(userId, userEmail) {
    const secret = speakeasy.generateSecret({
      length: 20,
      name: `${config.mfa.totp.issuer}:${userEmail}`,
      issuer: config.mfa.totp.issuer
    });

    // Gera QR Code para aplicativo autenticador
    const qrCodeUrl = await qrcode.toDataURL(secret.otpauth_url);

    // Gera códigos de backup
    const backupCodes = this.generateBackupCodes();

    // Armazena secret de forma segura (hash)
    const secretHash = crypto
      .createHash('sha256')
      .update(secret.base32)
      .digest('hex');

    await this.auditLogger.log({
      event: 'mfa_setup_initiated',
      userId,
      level: 'medium',
      details: { method: 'totp' }
    });

    return {
      secret: secret.base32, // Apenas para exibição inicial
      secretHash,
      qrCodeUrl,
      backupCodes,
      backupCodesHash: backupCodes.map(code => 
        crypto.createHash('sha256').update(code).digest('hex')
      )
    };

    const codes = [];
    for (let i = 0; i < config.mfa.backupCodes.count; i++) {
      codes.push(crypto.randomBytes(config.mfa.backupCodes.length).toString('hex'));
    }
    return codes;

          event: 'mfa_verified',

class ZAPSecurityScanner {
  constructor(apiKey, proxyUrl = 'http://localhost:8080') {
    this.api = new zaproxy(proxyUrl, apiKey);
    this.scanId = null;
    this.results = {};
  }

  async setupContext(baseUrl) {
    console.log('Configurando contexto de segurança...');
    
    // Criar contexto
    const contextName = 'DIYAPP-V12';
    const context = await this.api.context.newContext(contextName);
    
    // Incluir URLs no contexto
    await this.api.context.includeInContext(contextName, `${baseUrl}.*`);
    
    // Configurar autenticação (se necessário)
    await this.configureAuthentication(baseUrl);
    
    return contextName;
  }

  async configureAuthentication(baseUrl) {
    // Configurar login form-based
    const loginUrl = `${baseUrl}/api/v1/auth/login`;
    const loginData = 'email=%USERNAME%&password=%PASSWORD%';
    
    await this.api.authentication.setAuthenticationMethod(
      '1', // contextId
      'formBasedAuthentication',
      JSON.stringify({
        loginUrl: loginUrl,
        loginRequestData: loginData,
      })
    );
    
    // Configurar usuários de teste
    await this.api.users.newUser('1', 'security-tester');
    await this.api.users.setAuthenticationCredentials(
      '1',
      '1', // userId
      JSON.stringify({
        username: 'test@security.com',
        password: 'TestSecurity@123',
      })
    );
    
    await this.api.users.setUserEnabled('1', '1', true);
  }

  async runSpider(baseUrl) {
    console.log('Executando spider...');
    
    const spiderScan = await this.api.spider.scan(baseUrl, null, null, null, null);
    this.spiderId = spiderScan.scan;
    
    // Aguardar conclusão
    while (true) {
      const status = await this.api.spider.status(this.spiderId);
      console.log(`Spider progress: ${status.status}%`);
      
      if (parseInt(status.status) >= 100) break;
      await this.sleep(5000);
    }
    
    console.log('Spider concluído');
  }

  async runActiveScan(baseUrl) {
    console.log('Iniciando varredura ativa...');
    
    const scan = await this.api.ascan.scan(baseUrl, null, null, null, null, null);
    this.scanId = scan.scan;
    
    // Monitorar progresso
    while (true) {
      const status = await this.api.ascan.status(this.scanId);
      console.log(`Scan progress: ${status.status}%`);
      
      if (parseInt(status.status) >= 100) break;
      await this.sleep(10000);
    }
    
    console.log('Varredura ativa concluída');
  }

  async generateReport(outputDir) {
    console.log('Gerando relatórios...');
    
    // Relatório HTML
    const htmlReport = await this.api.core.htmlreport();
    fs.writeFileSync(
      path.join(outputDir, 'zap-security-report.html'),
      htmlReport
    );
    
    // Relatório JSON
    const alerts = await this.api.core.alerts();
    fs.writeFileSync(
      path.join(outputDir, 'zap-alerts.json'),
      JSON.stringify(alerts, null, 2)
    );
    
    // Relatório de vulnerabilidades por risco
    const summary = this.summarizeAlerts(alerts);
    fs.writeFileSync(
      path.join(outputDir, 'vulnerability-summary.json'),
      JSON.stringify(summary, null, 2)
    );
    
    console.log(`Relatórios gerados em: ${outputDir}`);
  }

  summarizeAlerts(alerts) {
    const summary = {
      high: [],
      medium: [],
      low: [],
      informational: [],
      totals: {
        high: 0,
        medium: 0,
        low: 0,
        informational: 0,
      },
    };
    
    alerts.forEach(alert => {
      const risk = alert.risk.toLowerCase();
      summary[risk].push({
        name: alert.name,
        url: alert.url,
        description: alert.description,
        solution: alert.solution,
      });
      summary.totals[risk]++;
    });
    
    return summary;
  }

  async sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  async fullScan(baseUrl, outputDir = './reports') {
    try {
      // Criar diretório de saída
      if (!fs.existsSync(outputDir)) {
        fs.mkdirSync(outputDir, { recursive: true });
      }
      
      // Verificar se ZAP está rodando
      await this.api.core.version();
      console.log('ZAP conectado com sucesso');
      
      // Configurar contexto
      await this.setupContext(baseUrl);
      
      // Executar spider
      await this.runSpider(baseUrl);
      
      // Executar varredura ativa
      await this.runActiveScan(baseUrl);
      
      // Gerar relatórios
      await this.generateReport(outputDir);
      
      console.log('Scan de segurança completo!');
      
    } catch (error) {
      console.error('Erro no scan de segurança:', error);
      throw error;
    }
  }
}

// Script de execução
