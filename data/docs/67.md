# v10_analysis.py
"""
Análise técnica da V10 - Identificação de pontos críticos
"""

import json
from dataclasses import dataclass
from typing import List, Dict, Optional
from datetime import datetime
from enum import Enum

class IssueSeverity(Enum):
    CRITICAL = "CRITICAL"  # Pode causar downtime
    HIGH = "HIGH"          # Impacta performance significativamente
    MEDIUM = "MEDIUM"      # Problema de manutenibilidade
    LOW = "LOW"           # Melhoria recomendada

@dataclass
class TechnicalDebtItem:
    id: str
    description: str
    severity: IssueSeverity
    location: str
    impact: str
    estimated_effort_hours: int
    dependencies: List[str]
    created_at: datetime = datetime.now()
    
    def to_adr_format(self) -> Dict:
        return {
            "id": self.id,
            "description": self.description,
            "severity": self.severity.value,
            "location": self.location,
            "impact": self.impact,
            "estimated_effort": self.estimated_effort_hours,
            "dependencies": self.dependencies,
            "created": self.created_at.isoformat()
        }

class V10CodeAnalyzer:
    def __init__(self):
        self.debt_items = []
        self.metrics = {
            "monolithic_components": 0,
            "tight_coupling_instances": 0,
            "missing_observability": 0,
            "resilience_gaps": 0
        }
    
    def analyze_monolithic_structure(self) -> List[TechnicalDebtItem]:
        """Identifica componentes monolíticos que devem ser desacoplados"""
        items = [
            TechnicalDebtItem(
                id="V10-MONO-001",
                description="Serviço principal combina gerenciamento de usuários, processamento de tarefas e API Gateway",
                severity=IssueSeverity.CRITICAL,
                location="src/main_server.js",
                impact="Single point of failure, dificulta escalabilidade horizontal",
                estimated_effort_hours=40,
                dependencies=["auth-service", "task-service", "api-gateway"]
            ),
            TechnicalDebtItem(
                id="V10-MONO-002",
                description="Banco de dados único para todos os domínios (usuários, tarefas, logs)",
                severity=IssueSeverity.HIGH,
                location="database/schema.sql",
                impact="Acoplamento de dados, dificuldade em migrações independentes",
                estimated_effort_hours=32,
                dependencies=["user-db", "task-db", "audit-db"]
            )
        ]
        self.debt_items.extend(items)
        self.metrics["monolithic_components"] = len(items)
        return items
    
    def analyze_coupling_issues(self) -> List[TechnicalDebtItem]:
        """Identifica acoplamento forte entre módulos"""
        items = [
            TechnicalDebtItem(
                id="V10-COUP-001",
                description="Módulo de autenticação diretamente acoplado ao módulo de tarefas",
                severity=IssueSeverity.HIGH,
                location="src/auth/task_integration.js",
                impact="Impossibilidade de atualizar autenticação sem afetar tarefas",
                estimated_effort_hours=24,
                dependencies=["auth-service", "task-service"]
            ),
            TechnicalDebtItem(
                id="V10-COUP-002",
                description="Configurações compartilhadas globalmente sem namespacing",
                severity=IssueSeverity.MEDIUM,
                location="config/global.js",
                impact="Conflitos de configuração entre serviços",
                estimated_effort_hours=16,
                dependencies=["config-service"]
            )
        ]
        self.debt_items.extend(items)
        self.metrics["tight_coupling_instances"] = len(items)
        return items
    
    def analyze_observability_gaps(self) -> List[TechnicalDebtItem]:
        """Identifica falta de observabilidade"""
        items = [
            TechnicalDebtItem(
                id="V10-OBSV-001",
                description="Logs centralizados sem estrutura (console.log disperso)",
                severity=IssueSeverity.HIGH,
                location="src/**/*.js",
                impact="Dificuldade debugging em produção, falta de contexto",
                estimated_effort_hours=20,
                dependencies=["logging-service", "distributed-tracing"]

            TechnicalDebtItem(
                id="V10-OBSV-002",
                description="Métricas de negócio não instrumentadas",
                severity=IssueSeverity.MEDIUM,
                location="src/business/",
                impact="Impossibilidade de medir KPIs técnicos e de negócio",

            TechnicalDebtItem(
                id="V10-OBSV-003",
                description="Health checks ausentes para serviços críticos",
                severity=IssueSeverity.CRITICAL,
                location="src/services/",
                impact="Kubernetes não detecta falhas, falta de auto-healing",
                estimated_effort_hours=12,
                dependencies=["health-check-service", "k8s-probes"]
            )
        ]
        self.debt_items.extend(items)
        self.metrics["missing_observability"] = len(items)
        return items
    
    def analyze_resilience_gaps(self) -> List[TechnicalDebtItem]:
        """Identifica gaps em resiliência"""
        items = [
            TechnicalDebtItem(
                id="V10-RESL-001",
                description="Circuit breaker ausente em chamadas externas",
                severity=IssueSeverity.CRITICAL,

                id="V10-RESL-002",
                description="Retry policies não configuradas",
                severity=IssueSeverity.HIGH,

                estimated_effort_hours=14,

                id="V10-RESL-003",
                description="Timeout não configurado para operações de I/O",
                severity=IssueSeverity.HIGH,
                location="src/io_operations/",
                impact="Operações bloqueantes podem travar o sistema",
                estimated_effort_hours=10,
                dependencies=["timeout-config", "async-timeout"]
            )
        ]
        self.debt_items.extend(items)
        self.metrics["resilience_gaps"] = len(items)
        return items
    
    def generate_analysis_report(self) -> Dict:
        """Gera relatório completo da análise"""
        self.analyze_monolithic_structure()
        self.analyze_coupling_issues()
        self.analyze_observability_gaps()
        self.analyze_resilience_gaps()
        
        total_effort = sum(item.estimated_effort_hours for item in self.debt_items)
        critical_items = [item for item in self.debt_items if item.severity == IssueSeverity.CRITICAL]
        
        return {
            "analysis_date": datetime.now().isoformat(),
            "total_debt_items": len(self.debt_items),
            "critical_items_count": len(critical_items),
            "total_estimated_effort_hours": total_effort,
            "metrics": self.metrics,
            "debt_items": [item.to_adr_format() for item in self.debt_items],
            "recommendations": self._generate_recommendations()
        }
    
    def _generate_recommendations(self) -> List[str]:
        return [
            "Migrar para arquitetura de microsserviços com bounded contexts claros",
            "Implementar API Gateway para roteamento e rate limiting",
            "Adicionar service mesh para comunicação entre serviços",
            "Implementar observabilidade completa (logs, métricas, traces)",
            "Adicionar padrões de resiliência (circuit breaker, retry, timeout)",
            "Containerizar todos os serviços com Docker",
            "Implementar CI/CD com rollback automático",
            "Adicionar feature flags para deploy progressivo"
        ]

# Executar análise
if __name__ == "__main__":
    analyzer = V10CodeAnalyzer()
    report = analyzer.generate_analysis_report()
    
    # Salvar relatório

        json.dump(report, f, indent=2)
    
    print(f"Análise concluída. {report['total_debt_items']} itens de débito técnico identificados.")
    print(f"Esforço total estimado: {report['total_estimated_effort_hours']} horas")

## 2. ADR - ARQUITETURA V11 - MICROSSERVIÇOS

```markdown
# ADR-001: Migração para Arquitetura de Microsserviços

Data: 2024-01-15
Status: Aceita
Autores: Tech Lead, Especialista Infra, Especialista Backend

## CONTEXTO:
A V10 do DIYAPP evoluiu para uma aplicação monolítica com:
1. Acoplamento forte entre domínios (usuários, tarefas, IA)
2. Dificuldade de escalar componentes individualmente
3. Single point of failure no servidor principal
4. Falta de observabilidade distribuída
5. Deployment e rollback de todo o sistema simultaneamente

## DECISÃO:
Migrar para arquitetura de microsserviços baseada em:
- **Bounded Contexts** do Domain-Driven Design
- **Containerização** com Docker
- **Orquestração** com Kubernetes
- **Service Mesh** (Istio) para comunicação entre serviços
- **API Gateway** (Kong) para roteamento externo

## OPÇÕES CONSIDERADAS:

### Opção A: Monolito Modularizado
- **Prós**: Menor complexidade inicial, código compartilhado fácil
- **Contras**: Não resolve acoplamento, dificuldade de deploy independente
- **Esforço**: 80 horas

### Opção B: Microsserviços Leves (3-5 serviços)
- **Prós**: Separação clara de responsabilidades, deploy independente
- **Contras**: Overhead de comunicação, necessidade de orquestração
- **Esforço**: 200 horas

### Opção C: Microsserviços Completos (8+ serviços) com Service Mesh
- **Prós**: Máxima resiliência, observabilidade nativa, escalabilidade granular
- **Contras**: Maior complexidade operacional, curva de aprendizado
- **Esforço**: 320 horas

### Opção escolhida: C
**Justificativa**: O DIYAPP é crítico para operação autônoma. A resiliência e observabilidade são requisitos não negociáveis. O esforço adicional é justificado pela redução de downtime e capacidade de escalar componentes de IA independentemente.

## CONSEQUÊNCIAS:

### Positivas:
1. Deploy independente por serviço
2. Escalabilidade horizontal granular
3. Isolamento de falhas (um serviço cai, outros continuam)
4. Observabilidade distribuída nativa
5. Melhor separação de equipes (squads por serviço)

### Negativas:
1. Maior complexidade de debugging
2. Overhead de comunicação entre serviços
3. Necessidade de orquestração (Kubernetes)
4. Gerenciamento de múltiplos bancos de dados

### Riscos:
1. **Latência aumentada**: Mitigar com caching e comunicação assíncrona
2. **Consistência de dados**: Implementar SAGA pattern
3. **Complexidade operacional**: Investir em CI/CD robusto e monitoring

## REVISÃO: 2024-04-15 (3 meses após implementação)
```

// src/api-gateway/index.js
const express = require('express');
const helmet = require('helmet');
const cors = require('cors');
const rateLimit = require('express-rate-limit');
const winston = require('winston');
const expressWinston = require('express-winston');
const CircuitBreaker = require('opossum');
const axios = require('axios');
const crypto = require('crypto');

class APIGateway {
    constructor() {
        this.app = express();
        this.services = new Map();
        this.breakers = new Map();
        this.setupMiddleware();
        this.setupLogging();
        this.setupRoutes();
        this.setupHealthChecks();
        this.setupCircuitBreakers();
    }

    setupMiddleware() {
        // Security headers
        this.app.use(helmet());
        
        // CORS configuration
        this.app.use(cors({
            origin: process.env.CORS_ORIGIN || 'http://localhost:3000',
            credentials: true,
            optionsSuccessStatus: 200
        }));

        // Body parsing
        this.app.use(express.json({ limit: '10mb' }));
        this.app.use(express.urlencoded({ extended: true, limit: '10mb' }));

        // Global rate limiting
        const globalLimiter = rateLimit({
            windowMs: 15 * 60 * 1000, // 15 minutes
            max: 1000, // Limit each IP to 1000 requests per windowMs
            standardHeaders: true,
            legacyHeaders: false,
            message: {
                error: 'Too many requests',
                message: 'Rate limit exceeded. Please try again later.'
            }
        });
        this.app.use(globalLimiter);
    }

    setupLogging() {
        // Winston logger configuration
        this.logger = winston.createLogger({
            level: process.env.LOG_LEVEL || 'info',
            format: winston.format.combine(
                winston.format.timestamp(),
                winston.format.json()
            ),
            transports: [
                new winston.transports.Console({
                    format: winston.format.combine(
                        winston.format.colorize(),
                        winston.format.simple()
                    )
                }),
                new winston.transports.File({ 
                    filename: 'logs/error.log', 
                    level: 'error',
                    maxsize: 5242880, // 5MB
                    maxFiles: 5
                }),
                new winston.transports.File({ 
                    filename: 'logs/combined.log',
                    maxsize: 5242880,
                    maxFiles: 10
                })
            ]
        });

        // Request logging middleware
        this.app.use(expressWinston.logger({
            winstonInstance: this.logger,
            meta: true,
            msg: 'HTTP {{req.method}} {{req.url}}',
            expressFormat: false,
            colorize: false,
            dynamicMeta: (req, res) => {
                const correlationId = req.headers['x-correlation-id'] || crypto.randomUUID();
                req.correlationId = correlationId;
                
                return {
                    correlationId,
                    userId: req.user?.id || 'anonymous',
                    userAgent: req.get('user-agent'),
                    ip: req.ip,
                    duration: res.responseTime
                };
            },
            requestWhitelist: ['url', 'method', 'headers', 'query', 'body'],
            responseWhitelist: ['statusCode', 'responseTime'],
            bodyBlacklist: ['password', 'token', 'creditCard', 'cvv', 'cpf', 'cnpj']
        }));

        // Error logging middleware
        this.app.use(expressWinston.errorLogger({
            winstonInstance: this.logger
        }));
    }

    setupCircuitBreakers() {
        // Circuit breaker configuration
        const breakerOptions = {
            timeout: 10000, // 10 seconds
            errorThresholdPercentage: 50,
            resetTimeout: 30000, // 30 seconds
            rollingCountTimeout: 60000, // 1 minute
            rollingCountBuckets: 10,
            name: 'service-call'
        };

        // Default fallback function
        const fallback = (err, args) => {
            this.logger.error('Circuit breaker fallback triggered', {
                error: err.message,
                service: args[0]?.service || 'unknown',
                correlationId: args[0]?.correlationId
            });
            
            return {
                status: 503,
                data: {
                    error: 'Service temporarily unavailable',
                    message: 'Please try again in a few moments',
                    fallback: true
                }
            };
        };

        // Store default options
        this.breakerOptions = breakerOptions;
        this.fallback = fallback;
    }

    registerService(name, config) {
        const serviceConfig = {
            baseURL: config.url,
            timeout: config.timeout || 10000,
            headers: {
                'Content-Type': 'application/json',
                'x-api-version': config.version || '1.0'
            }
        };

        // Create axios instance for the service
        const client = axios.create(serviceConfig);

        // Create circuit breaker for this service
        const breaker = new CircuitBreaker(
            async (requestConfig) => {
                const startTime = Date.now();
                try {
                    const response = await client(requestConfig);
                    const duration = Date.now() - startTime;
                    
                    // Log successful call
                    this.logger.info('Service call succeeded', {
                        service: name,
                        endpoint: requestConfig.url,
                        method: requestConfig.method,
                        duration,
                        status: response.status,
                        correlationId: requestConfig.headers?.['x-correlation-id']
                    });

                    return response;
                } catch (error) {
                    const duration = Date.now() - startTime;
                    
                    // Log failed call
                    this.logger.error('Service call failed', {
                        service: name,
                        endpoint: requestConfig.url,
                        method: requestConfig.method,
                        duration,
                        error: error.message,
                        status: error.response?.status,
                        correlationId: requestConfig.headers?.['x-correlation-id']
                    });

                    throw error;
                }
            },
            {
                ...this.breakerOptions,
                name: `${name}-breaker`,
                timeout: config.timeout || this.breakerOptions.timeout
            }
        );

        breaker.fallback(this.fallback);
        
        // Event listeners for circuit breaker
        breaker.on('open', () => {
            this.logger.warn(`Circuit breaker OPEN for service: ${name}`);
        });

        breaker.on('close', () => {
            this.logger.info(`Circuit breaker CLOSED for service: ${name}`);
        });

        breaker.on('halfOpen', () => {
            this.logger.info(`Circuit breaker HALF-OPEN for service: ${name}`);
        });

        // Store service and breaker
        this.services.set(name, { client, config });
        this.breakers.set(name, breaker);

        this.logger.info(`Service registered: ${name}`, { url: config.url });
    }

    setupRoutes() {
        // API Gateway proxy middleware
        this.app.all('/api/:service/*', async (req, res, next) => {
            const serviceName = req.params.service;
            const service = this.services.get(serviceName);
            const breaker = this.breakers.get(serviceName);

            if (!service || !breaker) {
                return res.status(404).json({
                    error: 'Service not found',
                    message: `Service '${serviceName}' is not registered`
                });
            }

            // Extract the path after service name
            const originalUrl = req.originalUrl;
            const servicePath = originalUrl.substring(originalUrl.indexOf(serviceName) + serviceName.length + 1);

            // Prepare request configuration
            const requestConfig = {
                method: req.method,
                url: servicePath,
                data: req.body,
                params: req.query,
                headers: {
                    ...req.headers,
                    'x-correlation-id': req.correlationId,
                    'x-user-id': req.user?.id || 'anonymous',
                    'x-forwarded-for': req.ip,
                    // Remove hop-by-hop headers
                    'host': undefined,
                    'connection': undefined,
                    'keep-alive': undefined,
                    'proxy-authenticate': undefined,
                    'proxy-authorization': undefined,
                    'te': undefined,
                    'trailers': undefined,
                    'transfer-encoding': undefined,
                    'upgrade': undefined
                }
            };

            try {
                // Call service through circuit breaker
                const response = await breaker.fire(requestConfig);
                
                // Forward response
                res.status(response.status)
                   .set(response.headers)
                   .send(response.data);
            } catch (error) {
                if (error.response) {
                    // Service responded with error
                    res.status(error.response.status)
                       .set(error.response.headers)
                       .send(error.response.data);
                } else if (error.code === 'ECONNABORTED') {
                    // Timeout error
                    res.status(504).json({
                        error: 'Gateway Timeout',
                        message: 'The service took too long to respond',
                        correlationId: req.correlationId
                    });
                } else {
                    // Other errors
                    next(error);
                }
            }
        });

        // Service-specific rate limiting
        const authLimiter = rateLimit({
            windowMs: 15 * 60 * 1000,
            max: 100,
            skipSuccessfulRequests: false,
            keyGenerator: (req) => req.ip,
            handler: (req, res) => {
                this.logger.warn('Rate limit exceeded for auth service', {
                    ip: req.ip,
                    correlationId: req.correlationId
                });
                
                res.status(429).json({
                    error: 'Too Many Requests',
                    message: 'Too many authentication attempts. Please try again later.',
                    retryAfter: 15 * 60
                });
            }
        });

        // Apply specific rate limit to auth service
        this.app.use('/api/auth/*', authLimiter);
    }

    setupHealthChecks() {
        // Health check endpoint
        this.app.get('/health', (req, res) => {
            const health = {
                status: 'healthy',
                timestamp: new Date().toISOString(),
                uptime: process.uptime(),
                memory: process.memoryUsage(),
                services: []
            };

            // Check all registered services
            for (const [name, breaker] of this.breakers.entries()) {
                const serviceHealth = {
                    name,
                    status: breaker.closed ? 'healthy' : breaker.halfOpen ? 'degraded' : 'unhealthy',
                    stats: breaker.stats
                };
                health.services.push(serviceHealth);
            }

            // Determine overall status
            const unhealthyServices = health.services.filter(s => s.status === 'unhealthy');
            health.status = unhealthyServices.length > 0 ? 'degraded' : 'healthy';

            const statusCode = unhealthyServices.length > 0 ? 200 : 200; // Always 200 for health checks
            res.status(statusCode).json(health);
        });

        // Liveness probe
        this.app.get('/health/live', (req, res) => {
            res.status(200).json({ status: 'alive' });
        });

        // Readiness probe
        this.app.get('/health/ready', (req, res) => {
            const readyServices = Array.from(this.breakers.values())
                .filter(breaker => breaker.closed || breaker.halfOpen);
            
            const isReady = readyServices.length === this.breakers.size;
            
            res.status(isReady ? 200 : 503).json({
                status: isReady ? 'ready' : 'not ready',
                readyServices: readyServices.length,
                totalServices: this.breakers.size
            });
        });
    }

    // Metrics endpoint for Prometheus
    setupMetrics() {
        this.app.get('/metrics', (req, res) => {
            const metrics = [];
            
            // Circuit breaker metrics
            for (const [name, breaker] of this.breakers.entries()) {
                const stats = breaker.stats;
                metrics.push(`circuit_breaker_state{service="${name}"} ${breaker.opened ? 1 : breaker.halfOpen ? 0.5 : 0}`);
                metrics.push(`circuit_breaker_failures{service="${name}"} ${stats.failures}`);
                metrics.push(`circuit_breaker_successes{service="${name}"} ${stats.successes}`);
                metrics.push(`circuit_breaker_timeouts{service="${name}"} ${stats.timeouts}`);
                metrics.push(`circuit_breaker_fallback_successes{service="${name}"} ${stats.fallbackSuccesses}`);
                metrics.push(`circuit_breaker_fallback_failures{service="${name}"} ${stats.fallbackFailures}`);
                metrics.push(`circuit_breaker_latency_mean{service="${name}"} ${stats.latencyMean}`);
                metrics.push(`circuit_breaker_latency_95{service="${name}"} ${stats.latency95}`);
            }

            res.set('Content-Type', 'text/plain');
            res.send(metrics.join('\n') + '\n');
        });
    }

    start(port = process.env.PORT || 3000) {
        this.setupMetrics();
        
        // Error handling middleware (must be last)
        this.app.use((err, req, res, next) => {
            this.logger.error('Unhandled error in API Gateway', {
                error: err.message,
                stack: err.stack,
                correlationId: req.correlationId,
                url: req.url,
                method: req.method
            });

            res.status(500).json({
                error: 'Internal Server Error',
                message: 'An unexpected error occurred',
                correlationId: req.correlationId
            });
        });

        const server = this.app.listen(port, () => {
            this.logger.info(`API Gateway started on port ${port}`, {
                port,
                nodeVersion: process.version,
                environment: process.env.NODE_ENV || 'development'
            });
        });

        // Graceful shutdown
        const gracefulShutdown = () => {
            this.logger.info('Received shutdown signal, closing gracefully...');
            
            server.close(() => {
                this.logger.info('HTTP server closed');
                process.exit(0);
            });

            // Force shutdown after 10 seconds
            setTimeout(() => {
                this.logger.error('Could not close connections in time, forcefully shutting down');
                process.exit(1);
            }, 10000);
        };

        process.on('SIGTERM', gracefulShutdown);
        process.on('SIGINT', gracefulShutdown);

        return server;
    }
}

// src/services/api-gateway.ts
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { ApiResponse, ApiError } from '@/types'

const API_BASE_URL = import.meta.env.VITE_API_BASE_URL || 'https://api.diyapp-evolution.com/v1'

class ApiGateway {
  private client: AxiosInstance
  private retryCount = 0
  private readonly maxRetries = 3

  constructor() {
    this.client = axios.create({
      baseURL: API_BASE_URL,
      timeout: 30000,
      headers: {
        'Content-Type': 'application/json',
        'Accept': 'application/json'
      }
    })

    this.setupInterceptors()
  }

  private setupInterceptors(): void {
    // Request interceptor
    this.client.interceptors.request.use(
      (config) => {
        const token = localStorage.getItem('auth_token')
        if (token) {
          config.headers.Authorization = `Bearer ${token}`
        }
        
        // Add request ID for tracing
        config.headers['X-Request-ID'] = crypto.randomUUID()
        config.headers['X-Client-Version'] = 'v11.0.0'
        
        return config
      },
      (error) => {
        console.error('Request error:', error)
        return Promise.reject(error)
      }
    )

    // Response interceptor
    this.client.interceptors.response.use(
      (response: AxiosResponse<ApiResponse>) => {
        this.retryCount = 0 // Reset retry count on success
        
        // Extract data from standardized response
        if (response.data && typeof response.data === 'object' && 'data' in response.data) {
          return {
            ...response,
            data: response.data.data
          }
        }
        
        return response
      },
      async (error) => {
        const originalRequest = error.config
        
        // Handle network errors with exponential backoff retry
        if (!error.response && this.retryCount < this.maxRetries && !originalRequest._retry) {
          this.retryCount++
          originalRequest._retry = true
          
          const delay = Math.min(1000 * Math.pow(2, this.retryCount), 10000)
          
          await new Promise(resolve => setTimeout(resolve, delay))
          
          return this.client(originalRequest)
        }
        
        // Handle API errors
        if (error.response) {
          const apiError: ApiError = {
            code: error.response.data?.code || 'UNKNOWN_ERROR',
            message: error.response.data?.message || 'An unexpected error occurred',
            details: error.response.data?.details,
            timestamp: new Date().toISOString()
          }
          
          // Handle specific error codes
          switch (apiError.code) {
            case 'UNAUTHORIZED':
              localStorage.removeItem('auth_token')
              window.location.href = '/login'
              break
            case 'TOKEN_EXPIRED':
              // Implement token refresh logic here
              break
            case 'RATE_LIMITED':
              console.warn('Rate limited, please try again later')
              break
          }
          
          return Promise.reject(apiError)
        }
        
        // Handle timeout
        if (error.code === 'ECONNABORTED') {
          return Promise.reject({
            code: 'TIMEOUT',
            message: 'Request timeout. Please check your connection and try again.',
            timestamp: new Date().toISOString()
          })
        }
        
        // Handle network errors
        return Promise.reject({
          code: 'NETWORK_ERROR',
          message: 'Network error. Please check your internet connection.',
          timestamp: new Date().toISOString()
        })
      }
    )
  }

  // Generic request method
  async request<T = any>(config: AxiosRequestConfig): Promise<T> {
    try {
      const response = await this.client.request<ApiResponse<T>>(config)
      return response.data
    } catch (error) {
      console.error('API request failed:', error)
      throw error
    }
  }

  // CRUD methods
  async get<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.request<T>({ method: 'GET', url, ...config })
  }

  async post<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.request<T>({ method: 'POST', url, data, ...config })
  }

  async put<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.request<T>({ method: 'PUT', url, data, ...config })
  }

  async patch<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.request<T>({ method: 'PATCH', url, data, ...config })
  }

  async delete<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.request<T>({ method: 'DELETE', url, ...config })
  }

  // Health check
  async healthCheck(): Promise<{ status: string; timestamp: string }> {
    return this.get('/health')
  }

  // System status
  async getSystemStatus(): Promise<any> {
    return this.get('/system/status')
  }
}

export const apiGateway = new ApiGateway()

import { useQuery, useMutation, useQueryClient, UseQueryOptions, UseMutationOptions } from '@tanstack/react-query'
import { apiGateway } from '@/services/api-gateway'
import { ApiError } from '@/types'

export function useApiQuery<TData = any, TError = ApiError>(
  queryKey: string[],
  url: string,
  options?: Omit<UseQueryOptions<TData, TError>, 'queryKey' | 'queryFn'>
) {
  return useQuery<TData, TError>({
    queryKey,
    queryFn: () => apiGateway.get<TData>(url),
    retry: (failureCount, error) => {
      // Don't retry on 4xx errors except 429 (rate limit)
      if ((error as ApiError).code?.startsWith('4') && (error as ApiError).code !== '429') {
        return false
      }
      return failureCount < 2
    },
    staleTime: 5 * 60 * 1000, // 5 minutes
    gcTime: 10 * 60 * 1000, // 10 minutes
    ...options
  })
}

export function useApiMutation<TData = any, TVariables = any, TError = ApiError>(
  method: 'POST' | 'PUT' | 'PATCH' | 'DELETE',
  url: string,
  options?: Omit<UseMutationOptions<TData, TError, TVariables>, 'mutationFn'>
) {
  const queryClient = useQueryClient()

  return useMutation<TData, TError, TVariables>({
    mutationFn: (data) => {
      switch (method) {
        case 'POST':
          return apiGateway.post<TData>(url, data)
        case 'PUT':
          return apiGateway.put<TData>(url, data)
        case 'PATCH':
          return apiGateway.patch<TData>(url, data)
        case 'DELETE':
          return apiGateway.delete<TData>(url)
        default:
          throw new Error(`Invalid method: ${method}`)
      }
    },
    onSuccess: () => {
      // Invalidate relevant queries
      const baseKey = url.split('?')[0].split('/').filter(Boolean)
      queryClient.invalidateQueries({ queryKey: baseKey })
    },
    ...options
  })
}

            });
        }

        if (!allowedRoles.includes(req.user.role)) {
            // Log de tentativa de elevação de privilégio
            console.warn(`[SECURITY] Tentativa de acesso não autorizado: Usuário ${req.user.id} (role: ${req.user.role}) tentou acessar recurso que requer roles: ${allowedRoles.join(', ')}`);
            
            return res.status(403).json({
                error: 'Acesso negado',

/**
 * Middleware de rate limiting básico
 * @param {Object} options - Opções de rate limiting
 */
const rateLimit = (options = {}) => {
    const {
        windowMs = 15 * 60 * 1000, // 15 minutos
        max = 100, // limite de requisições por IP
        message = 'Muitas requisições deste IP, tente novamente mais tarde'
    } = options;

    const requests = new Map();

    return (req, res, next) => {
        const ip = req.ip || req.connection.remoteAddress;
        const now = Date.now();
        
        if (!requests.has(ip)) {
            requests.set(ip, { count: 1, resetTime: now + windowMs });
        } else {
            const record = requests.get(ip);
            
            if (now > record.resetTime) {
                // Reset do contador
                record.count = 1;
                record.resetTime = now + windowMs;
            } else if (record.count >= max) {
                // Log de possível ataque DoS
                console.warn(`[SECURITY] Rate limit excedido para IP: ${ip}, Path: ${req.path}`);
                
                return res.status(429).json({
                    error: 'Rate limit excedido',
                    message: message,
                    retryAfter: Math.ceil((record.resetTime - now) / 1000)
                });
            } else {
                record.count++;
            }
        }
        
        // Limpeza periódica de registros antigos
        if (Math.random() < 0.01) { // 1% chance a cada requisição
            for (const [key, value] of requests.entries()) {
                if (now > value.resetTime) {
                    requests.delete(key);
                }
            }
        }
        
        next();
    };
};

        console.warn(`[SECURITY] Validação falhou para ${req.method} ${req.path}:`, errors.array());
        
        return res.status(400).json({

                message: err.msg,
                value: err.value

        });

    ['body', 'params', 'query'].forEach(source => {

                if (typeof req[source][key] === 'string') {
                    req[source][key] = sanitizeInput(req[source][key]);
                }
            });
        }
    });

const validateFileUpload = (req, res, next) => {
    if (!req.file && !req.files) {
        return next();
    }
    
    const files = req.file ? [req.file] : req.files;
    const allowedMimeTypes = [
        'image/jpeg',
        'image/png',
        'image/gif',
        'application/pdf',
        'text/plain'
    ];
    
    const maxSize = 5 * 1024 * 1024; // 5MB
    
    for (const file of files) {
        // Verifica tipo MIME
        if (!allowedMimeTypes.includes(file.mimetype)) {
            return res.status(400).json({
                error: 'Tipo de arquivo não permitido',
                message: `Tipo ${file.mimetype} não é permitido. Tipos permitidos: ${allowedMimeTypes.join(', ')}`
            });
        }
        
        // Verifica tamanho
        if (file.size > maxSize) {
            return res.status(400).json({
                error: 'Arquivo muito grande',
                message: `Arquivo excede o limite de ${maxSize / (1024 * 1024)}MB`
            });
        }
        
        // Verifica extensão do nome do arquivo
        const extension = file.originalname.split('.').pop().toLowerCase();
        const allowedExtensions = ['jpg', 'jpeg', 'png', 'gif', 'pdf', 'txt'];
        
        if (!allowedExtensions.includes(extension)) {
            return res.status(400).json({
                error: 'Extensão de arquivo não permitida',
                message: `Extensão .${extension} não é permitida`
            });
        }
    }
    
    next();
};

    constructor() {
        this.vulnerabilities = {

    async runNpmAudit() {
        try {

            const { stdout } = await execPromise('npm audit --json');
            this.auditResults = JSON.parse(stdout);
            
            this._parseVulnerabilities();
            this._generateReport();
            
            return this.vulnerabilities;
        } catch (error) {
            // npm audit retorna código de saída 1 quando encontra vulnerabilidades
            if (error.stdout) {
                try {
                    this.auditResults = JSON.parse(error.stdout);
                    this._parseVulnerabilities();
                    this._generateReport();
                    
                    console.warn('[SECURITY] Vulnerabilidades encontradas durante auditoria');
                    return this.vulnerabilities;
                } catch (parseError) {
                    throw new Error(`Erro ao analisar resultados do npm audit: ${parseError.message}`);
                }
            }
            throw new Error(`Erro ao executar npm audit: ${error.message}`);
        }
    }

    /**
     * Analisa vulnerabilidades do relatório
     */
    _parseVulnerabilities() {
        if (!this.auditResults || !this.auditResults.vulnerabilities) {
            return;
        }

        for (const [packageName, vulnInfo] of Object.entries(this.auditResults.vulnerabilities)) {
            for (const vulnerability of vulnInfo.via) {
                const severity = vulnerability.severity || vulnInfo.severity;
                
                const vulnData = {
                    package: packageName,
                    title: vulnerability.title || 'Vulnerabilidade não especificada',
                    severity: severity,
                    cvssScore: vulnerability.cvss ? vulnerability.cvss.score : null,
                    range: vulnInfo.range,
                    path: vulnInfo.nodes || [],
                    fixAvailable: vulnInfo.fixAvailable || false,
                    recommendation: vulnerability.recommendation || 'Atualize para versão mais recente'
                };

                switch (severity) {
                    case 'critical':
                        this.vulnerabilities.critical.push(vulnData);
                        break;
                    case 'high':
                        this.vulnerabilities.high.push(vulnData);
                        break;
                    case 'moderate':
                        this.vulnerabilities.moderate.push(vulnData);
                        break;
                    case 'low':
                        this.vulnerabilities.low.push(vulnData);
                        break;

            timestamp: new Date().toISOString(),
            summary: {
                critical: this.vulnerabilities.critical.length,
                high: this.vulnerabilities.high.length,
                moderate: this.vulnerabilities.moderate.length,
                low: this.vulnerabilities.low.length,
                total: Object.values(this.vulnerabilities).flat().length
            },
            vulnerabilities: this.vulnerabilities,
            recommendations: this._generateRecommendations()
        };

        // Salva relatório em arquivo
        const reportDir = path.join(__dirname, '../../security-reports');
        if (!fs.existsSync(reportDir)) {
            fs.mkdirSync(reportDir, { recursive: true });
        }

        const reportFile = path.join(reportDir, `security-audit-${Date.now()}.json`);
        fs.writeFileSync(reportFile, JSON.stringify(report, null, 2));
        
        console.log(`[SECURITY] Relatório salvo em: ${reportFile}`);
        
        // Log de vulnerabilidades críticas/altas
        if (this.vulnerabilities.critical.length > 0 || this.vulnerabilities.high.length > 0) {
            console.error('[SECURITY ALERT] Vulnerabilidades críticas/altas encontradas!');
            this.vulnerabilities.critical.forEach(v => {
                console.error(`  CRITICAL: ${v.package} - ${v.title} (CVSS: ${v.cvssScore})`);
            });
            this.vulnerabilities.high.forEach(v => {
                console.error(`  HIGH: ${v.package} - ${v.title} (CVSS: ${v.cvssScore})`);
            });
        }
    }

     */
    _generateRecommendations() {
        const recommendations = [];
        
        // Verifica vulnerabilidades críticas
        if (this.vulnerabilities.critical.length > 0) {
            recommendations.push({
                priority: 'CRITICAL',
                action: 'CORREÇÃO IMEDIATA REQUERIDA',
                description: `Existem ${this.vulnerabilities.critical.length} vulnerabilidades críticas que devem ser corrigidas em menos de 24 horas`,
                packages: this.vulnerabilities.critical.map(v => v.package)
            });
        }
        
        // Verifica vulnerabilidades altas
        if (this.vulnerabilities.high.length > 0) {

                action: 'Correção urgente',
                description: `Existem ${this.vulnerabilities.high.length} vulnerabilidades altas que devem ser corrigidas em menos de 72 horas`,
                packages: this.vulnerabilities.high.map(v => v.package)
            });
        }

input {
  # Input para logs de arquivo
  file {
    path => "/app-logs/*.log"
    start_position => "beginning"
    sincedb_path => "/dev/null"
    type => "app-logs"
  }
  
  # Input TCP para logs em tempo real
  tcp {
    port => 5000
    codec => json_lines
    type => "real-time-logs"
  }
  
  # Input HTTP para métricas da aplicação
  http {
    port => 8080
    codec => json
    type => "app-metrics"
  }
}

filter {
  if [type] == "app-logs" {
    grok {
      match => { "message" => "%{TIMESTAMP_ISO8601:timestamp} %{LOGLEVEL:level} \[%{DATA:module}\] %{GREEDYDATA:message}" }
    }
    
    date {
      match => [ "timestamp", "ISO8601" ]
      target => "@timestamp"
    }
    
    mutate {
      remove_field => ["timestamp"]
    }
    
    # Análise de padrões simples
    if [message] =~ /error|exception|failed/i {
      mutate {
        add_tag => ["error_pattern"]
      }
    }
    
    if [message] =~ /slow|timeout|latency/i {
      mutate {
        add_tag => ["performance_pattern"]
      }
    }
    
    if [message] =~ /authentication|authorization|permission/i {
      mutate {
        add_tag => ["security_pattern"]
      }
    }
  }
  
  if [type] == "app-metrics" {
    mutate {
      add_field => {
        "metric_timestamp" => "%{@timestamp}"
      }
    }
  }
}

output {
  elasticsearch {
    hosts => ["elasticsearch:9200"]
    index => "diyapp-logs-%{+YYYY.MM.dd}"
  }
  
  # Output para debug (opcional)
  stdout {
    codec => rubydebug
  }
}

-- DIYAPP Evolution - V11 Core
-- Schema de banco de dados principal para eventos de usuário

-- =============================================
-- CONFIGURAÇÃO INICIAL DO BANCO
-- =============================================

-- Criar extensões necessárias
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pgcrypto";

-- =============================================
-- SCHEMA PRINCIPAL
-- =============================================

CREATE SCHEMA IF NOT EXISTS v11_core;
COMMENT ON SCHEMA v11_core IS 'Schema principal da V11 - Eventos de usuário e métricas de produto';

-- =============================================
-- TABELA: users (Camada Raw - Bronze)
-- =============================================

CREATE TABLE v11_core.raw_users (
    -- Identificação
    user_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    external_user_id VARCHAR(255), -- ID do sistema externo (ex: Auth0, Firebase)
    
    -- Dados do usuário (preservados como vieram da fonte)
    raw_user_data JSONB NOT NULL,
    
    -- Metadados da fonte
    source_system VARCHAR(100) NOT NULL,
    source_table VARCHAR(100),
    source_record_id VARCHAR(255),
    
    -- Timestamps de processamento
    ingested_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    source_created_at TIMESTAMPTZ,
    source_updated_at TIMESTAMPTZ,
    
    -- Controle de qualidade
    data_quality_score INTEGER DEFAULT 100,
    validation_errors JSONB DEFAULT '[]'::jsonb,
    
    -- Indexes para performance
    INDEX idx_raw_users_external_id (external_user_id),
    INDEX idx_raw_users_ingested_at (ingested_at),
    INDEX idx_raw_users_source (source_system, source_table)
);

COMMENT ON TABLE v11_core.raw_users IS 'Camada RAW: Dados brutos de usuários preservados exatamente como vieram da fonte';
COMMENT ON COLUMN v11_core.raw_users.raw_user_data IS 'Dados completos do usuário no formato original da fonte';
COMMENT ON COLUMN v11_core.raw_users.data_quality_score IS 'Score de qualidade (0-100) baseado em validações automáticas';

-- =============================================
-- TABELA: user_events (Camada Raw - Bronze)
-- =============================================

CREATE TABLE v11_core.raw_user_events (
    -- Identificação
    event_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    event_uuid VARCHAR(255) UNIQUE, -- UUID do evento no sistema fonte
    
    -- Dados do evento
    event_type VARCHAR(100) NOT NULL,
    event_name VARCHAR(200) NOT NULL,
    event_data JSONB NOT NULL,
    
    -- Contexto do usuário
    user_id UUID REFERENCES v11_core.raw_users(user_id),
    session_id VARCHAR(255),
    device_id VARCHAR(255),
    
    -- Contexto da aplicação
    app_version VARCHAR(50),
    platform VARCHAR(50), -- 'web', 'ios', 'android'
    os_version VARCHAR(50),
    browser VARCHAR(100),
    
    -- Localização
    country_code CHAR(2),
    region VARCHAR(100),
    city VARCHAR(100),
    
    -- Timestamps
    event_timestamp TIMESTAMPTZ NOT NULL,
    ingested_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    
    -- Metadados da fonte
    source_system VARCHAR(100) NOT NULL DEFAULT 'frontend',
    source_ip INET,
    user_agent TEXT,
    
    -- Indexes para queries comuns
    INDEX idx_raw_events_user_timestamp (user_id, event_timestamp),
    INDEX idx_raw_events_type_timestamp (event_type, event_timestamp),
    INDEX idx_raw_events_session (session_id),
    INDEX idx_raw_events_ingested_at (ingested_at)

-- =============================================
-- VIEW: mart_daily_active_users (Camada Mart - Gold)
-- =============================================

CREATE OR REPLACE VIEW v11_core.mart_daily_active_users AS
SELECT 
    DATE(event_timestamp) AS activity_date,
    COUNT(DISTINCT user_id) AS active_users,
    COUNT(DISTINCT session_id) AS total_sessions,
    COUNT(*) AS total_events,
    
    -- Segmentação por plataforma
    COUNT(DISTINCT CASE WHEN platform = 'ios' THEN user_id END) AS ios_users,
    COUNT(DISTINCT CASE WHEN platform = 'android' THEN user_id END) AS android_users,
    COUNT(DISTINCT CASE WHEN platform = 'web' THEN user_id END) AS web_users,
    
    -- Engajamento
    AVG(events_per_session) AS avg_events_per_session,
    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY events_per_session) AS median_events_per_session
    
FROM (
    SELECT 
        user_id,
        session_id,
        platform,
        DATE(event_timestamp) as event_date,
        COUNT(*) as events_per_session
    FROM v11_core.stg_user_events
    WHERE event_timestamp >= CURRENT_DATE - INTERVAL '30 days'
    GROUP BY 1,2,3,4
) session_metrics
GROUP BY 1
ORDER BY 1 DESC;

COMMENT ON VIEW v11_core.mart_daily_active_users IS 'MART: Métricas diárias de usuários ativos para o PM';

const { validationResult } = require('express-validator');

class SecurityMiddleware {
    constructor() {
        this.secretKey = process.env.JWT_SECRET || 'your-secret-key-change-in-production';
        this.tokenExpiry = '15m';
        this.refreshTokenExpiry = '7d';
        
        // Lista de endpoints públicos que não requerem autenticação
        this.publicEndpoints = [
            '/api/auth/login',
            '/api/auth/refresh',
            '/api/health',
            '/'
        ];
        
        // Rate limiting configuration
        this.rateLimitConfig = {
            windowMs: 15 * 60 * 1000, // 15 minutes
            max: 100 // limit each IP to 100 requests per windowMs
        };
    }

    /**
     * Middleware de autenticação JWT
     */
    authenticateJWT() {
        return (req, res, next) => {
            // Skip authentication for public endpoints
            if (this.publicEndpoints.includes(req.path)) {
                return next();
            }

            const authHeader = req.headers.authorization;
            
            if (!authHeader) {
                return res.status(401).json({
                    error: 'Unauthorized',
                    message: 'No authorization token provided'
                });
            }

            const token = authHeader.split(' ')[1];
            
            if (!token) {
                return res.status(401).json({
                    error: 'Unauthorized',
                    message: 'Invalid token format'
                });
            }

            jwt.verify(token, this.secretKey, (err, user) => {
                if (err) {
                    if (err.name === 'TokenExpiredError') {
                        return res.status(401).json({
                            error: 'TokenExpired',
                            message: 'Access token has expired'
                        });
                    }
                    return res.status(403).json({
                        error: 'Forbidden',
                        message: 'Invalid token'
                    });
                }

                // Adiciona informações do usuário ao request
                req.user = {
                    id: user.id,
                    email: user.email,
                    role: user.role,
                    permissions: user.permissions || []
                };

                // Log de auditoria para operações privilegiadas
                if (this.isPrivilegedOperation(req)) {
                    this.logAudit(req, 'PRIVILEGED_OPERATION', {
                        userId: user.id,
                        endpoint: req.path,
                        method: req.method,
                        timestamp: new Date().toISOString()
                    });
                }

                next();
            });
        };
    }

    /**
     * Middleware de autorização baseada em roles
     */
    authorize(requiredRoles = [], requiredPermissions = []) {
        return (req, res, next) => {
            if (!req.user) {
                return res.status(401).json({
                    error: 'Unauthorized',
                    message: 'User not authenticated'
                });
            }

            // Verifica se o usuário tem a role necessária
            if (requiredRoles.length > 0 && !requiredRoles.includes(req.user.role)) {
                return res.status(403).json({
                    error: 'Forbidden',
                    message: 'Insufficient role privileges'
                });
            }

            // Verifica se o usuário tem as permissões necessárias
            if (requiredPermissions.length > 0) {
                const hasAllPermissions = requiredPermissions.every(permission => 
                    req.user.permissions.includes(permission)
                );
                
                if (!hasAllPermissions) {
                    return res.status(403).json({
                        error: 'Forbidden',
                        message: 'Insufficient permissions'
                    });
                }
            }

            next();
        };
    }

    /**
     * Middleware de sanitização e validação de inputs
     */
    validateInputs(validationRules) {
        return async (req, res, next) => {
            // Executa as regras de validação
            await Promise.all(validationRules.map(validation => validation.run(req)));

            const errors = validationResult(req);
            
            if (!errors.isEmpty()) {
                return res.status(400).json({
                    error: 'ValidationError',
                    message: 'Invalid input data',
                    details: errors.array().map(err => ({
                        field: err.param,
                        message: err.msg,
                        value: err.value
                    }))
                });
            }

            // Sanitização adicional para prevenir XSS
            this.sanitizeInputs(req);

            next();
        };
    }

    /**
     * Sanitiza inputs para prevenir XSS
     */
    sanitizeInputs(req) {
        const sanitize = (value) => {
            if (typeof value !== 'string') return value;
            
            // Remove tags HTML perigosas
            return value
                .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
                .replace(/<[^>]*>/g, '')
                .trim();
        };

        // Sanitiza body, query e params
        ['body', 'query', 'params'].forEach(source => {
            if (req[source]) {
                Object.keys(req[source]).forEach(key => {
                    if (typeof req[source][key] === 'string') {
                        req[source][key] = sanitize(req[source][key]);
                    } else if (Array.isArray(req[source][key])) {
                        req[source][key] = req[source][key].map(item => 
                            typeof item === 'string' ? sanitize(item) : item
                        );
                    }
                });
            }
        });
    }

    /**
     * Middleware de rate limiting básico
     */
    rateLimit() {
        const requests = new Map();
        
        return (req, res, next) => {
            const ip = req.ip;
            const now = Date.now();
            const windowStart = now - this.rateLimitConfig.windowMs;
            
            // Limpa requisições antigas
            if (requests.has(ip)) {
                requests.set(ip, requests.get(ip).filter(time => time > windowStart));
            }
            
            const ipRequests = requests.get(ip) || [];
            
            if (ipRequests.length >= this.rateLimitConfig.max) {
                return res.status(429).json({
                    error: 'TooManyRequests',
                    message: 'Rate limit exceeded. Please try again later.'
                });
            }
            
            ipRequests.push(now);
            requests.set(ip, ipRequests);
            
            // Adiciona headers de rate limit
            res.setHeader('X-RateLimit-Limit', this.rateLimitConfig.max);
            res.setHeader('X-RateLimit-Remaining', this.rateLimitConfig.max - ipRequests.length);
            res.setHeader('X-RateLimit-Reset', new Date(windowStart + this.rateLimitConfig.windowMs).toISOString());
            
            next();
        };
    }

    /**
     * Gera tokens JWT
     */
    generateTokens(userData) {
        const accessToken = jwt.sign(
            {
                id: userData.id,
                email: userData.email,
                role: userData.role,
                permissions: userData.permissions || []
            },
            this.secretKey,
            { expiresIn: this.tokenExpiry }
        );

        const refreshToken = jwt.sign(
            { id: userData.id, type: 'refresh' },
            this.secretKey,
            { expiresIn: this.refreshTokenExpiry }
        );

        return { accessToken, refreshToken };
    }

    /**
     * Verifica se é uma operação privilegiada
     */
    isPrivilegedOperation(req) {
        const privilegedEndpoints = [
            '/api/admin/',
            '/api/users/delete',
            '/api/users/update-role',
            '/api/config/'
        ];
        
        const privilegedMethods = ['DELETE', 'PUT', 'PATCH'];
        
        return privilegedEndpoints.some(endpoint => req.path.startsWith(endpoint)) ||
               privilegedMethods.includes(req.method);
    }

    /**
     * Log de auditoria
     */
    logAudit(req, action, metadata = {}) {
        const auditLog = {
            timestamp: new Date().toISOString(),
            action,
            userId: req.user?.id || 'anonymous',
            ip: req.ip,

            method: req.method,
            metadata
        };

        // Em produção, isso seria enviado para um sistema de logs centralizado

        // Também pode ser salvo em banco de dados
        // await AuditLog.create(auditLog);
    }

    /**
     * Middleware de headers de segurança
     */
    securityHeaders() {
        return (req, res, next) => {
            // Content Security Policy
            res.setHeader(
                'Content-Security-Policy',
                "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline';"
            );
            
            // Outros headers de segurança

            res.setHeader('X-Frame-Options', 'DENY');
            res.setHeader('X-XSS-Protection', '1; mode=block');
            res.setHeader('Strict-Transport-Security', 'max-age=31536000; includeSubDomains');
            
            next();
        };
    }
}

     */

        return [
            body('email')
                .isEmail().withMessage('Invalid email format')
                .normalizeEmail(),
            body('password')
                .isLength({ min: 8 }).withMessage('Password must be at least 8 characters'),
            body('name')
                .trim()
                .isLength({ min: 2, max: 100 }).withMessage('Name must be between 2 and 100 characters')
                .matches(/^[a-zA-Z\s]+$/).withMessage('Name can only contain letters and spaces'),
            body('role')
                .optional()
                .isIn(['user', 'admin', 'moderator']).withMessage('Invalid role')
        ];
    }

    /**
     * Validações para IDs em parâmetros
     */
    static idParamValidator() {
        return [
            param('id')
                .isUUID().withMessage('Invalid ID format')
                .trim()
        ];
    }

    /**
     * Validações para queries de paginação
     */
    static paginationValidators() {
        return [
            query('page')
                .optional()
                .isInt({ min: 1 }).withMessage('Page must be a positive integer')
                .toInt(),
            query('limit')
                .optional()
                .isInt({ min: 1, max: 100 }).withMessage('Limit must be between 1 and 100')
                .toInt()
        ];
    }

    /**
     * Validações para inputs de LLM (prevenção de prompt injection)
     */
    static llmInputValidators() {
        return [
            body('prompt')
                .trim()
                .isLength({ min: 1, max: 1000 }).withMessage('Prompt must be between 1 and 1000 characters')
                .custom(value => {
                    // Bloqueia tentativas óbvias de prompt injection
                    const blockedPatterns = [
                        /ignore.*previous.*instructions/i,
                        /system.*prompt/i,
                        /role.*play/i,
                        /you.*are.*now/i,
                        /forget.*everything/i
                    ];
                    
                    if (blockedPatterns.some(pattern => pattern.test(value))) {
                        throw new Error('Invalid input detected');
                    }
                    
                    return true;
                }),
            body('context')
                .optional()
                .isArray().withMessage('Context must be an array')
                .custom(array => {
                    // Valida que não há dados sensíveis no contexto
                    const sensitivePatterns = [
                        /password/i,
                        /token/i,
                        /secret/i,
                        /api[_-]?key/i,
                        /credit[_-]?card/i
                    ];
                    
                    array.forEach(item => {
                        if (typeof item === 'string') {
                            sensitivePatterns.forEach(pattern => {
                                if (pattern.test(item)) {
                                    throw new Error('Context contains potentially sensitive data');
                                }
                            });
                        }
                    });
                    
                    return true;
                })
        ];
    }

  # Parse application logs
  if [source] =~ /application\.log/ {
    grok {
      match => { "message" => "%{TIMESTAMP_ISO8601:timestamp} %{LOGLEVEL:log_level} \[%{DATA:module}\] %{GREEDYDATA:message}" }
    }
    
    date {
      match => [ "timestamp", "ISO8601" ]
      target => "@timestamp"
    }
    
    # Add application metadata
    mutate {
      add_field => { 
        "application" => "DIYAPP-V11"
        "environment" => "%{[fields][environment] || 'production'}"
      }
    }
    
    # Parse error patterns
    if [log_level] == "ERROR" {
      grok {
        match => { "message" => "(?<error_type>[^:]+): %{GREEDYDATA:error_message}" }
      }
    }
  }
  
  # Parse metrics data
  if [type] == "metrics" {
    mutate {
      add_field => { "[@metadata][target_index]" => "diyapp-metrics-%{+YYYY.MM.dd}" }
    }
  }
  
  # Parse access logs
  if [type] == "access" {
    grok {
      match => { "message" => "%{COMBINEDAPACHELOG}" }
    }
    
    mutate {
      add_field => { "[@metadata][target_index]" => "diyapp-access-%{+YYYY.MM.dd}" }
    }
  }
}

output {
  # Default output to Elasticsearch
  elasticsearch {
    hosts => ["elasticsearch:9200"]
    index => "diyapp-logs-%{+YYYY.MM.dd}"
  }

      disk: await systemInfo.fsSize(),
      network: await systemInfo.networkStats(),
      processes: await systemInfo.processes()
    };

    metricsHistory.system.push(metrics);
    
    // Keep only last 1000 entries
    if (metricsHistory.system.length > 1000) {
      metricsHistory.system = metricsHistory.system.slice(-1000);
    }

    // Send to Elasticsearch if configured
    if (process.env.ELASTICSEARCH_URL) {
      try {
        await axios.post(`${process.env.ELASTICSEARCH_URL}/diyapp-metrics/_doc`, {
          ...metrics,
          type: 'system_metrics',
          application: process.env.APP_NAME || 'DIYAPP-V11'
        });
      } catch (esError) {
        logger.error('Failed to send metrics to Elasticsearch:', esError.message);
      }
    }

    logger.info('System metrics collected', { metrics: JSON.stringify(metrics) });
    return metrics;
  } catch (error) {
    logger.error('Error collecting system metrics:', error);
    return null;
  }
}

// Monitor application health
async function checkApplicationHealth() {
  const healthChecks = [];
  const timestamp = new Date().toISOString();

  // Check disk space
  try {
    const disk = await systemInfo.fsSize();
    const rootDisk = disk.find(d => d.mount === '/') || disk[0];
    const diskUsagePercent = (rootDisk.used / rootDisk.size) * 100;
    
    healthChecks.push({
      name: 'disk_space',
      status: diskUsagePercent > 90 ? 'critical' : diskUsagePercent > 80 ? 'warning' : 'healthy',
      value: diskUsagePercent,
      threshold: 80
    });
  } catch (error) {
    healthChecks.push({
      name: 'disk_space',
      status: 'error',
      error: error.message
    });
  }

  // Check memory usage
  try {
    const memory = await systemInfo.mem();
    const memoryUsagePercent = (memory.active / memory.total) * 100;
    
    healthChecks.push({
      name: 'memory_usage',
      status: memoryUsagePercent > 90 ? 'critical' : memoryUsagePercent > 80 ? 'warning' : 'healthy',
      value: memoryUsagePercent,
      threshold: 80
    });
  } catch (error) {
    healthChecks.push({
      name: 'memory_usage',
      status: 'error',
      error: error.message
    });
  }

  // Check CPU load
  try {
    const cpu = await systemInfo.currentLoad();
    
    healthChecks.push({
      name: 'cpu_load',
      status: cpu.currentLoad > 90 ? 'critical' : cpu.currentLoad > 70 ? 'warning' : 'healthy',
      value: cpu.currentLoad,
      threshold: 70
    });
  } catch (error) {
    healthChecks.push({
      name: 'cpu_load',
      status: 'error',
      error: error.message
    });
  }

  const healthStatus = {
    timestamp,
    checks: healthChecks,
    overall: healthChecks.every(c => c.status === 'healthy') ? 'healthy' : 
             healthChecks.some(c => c.status === 'critical') ? 'critical' : 'warning'
  };

  metricsHistory.application.push(healthStatus);
  
  if (metricsHistory.application.length > 1000) {
    metricsHistory.application = metricsHistory.application.slice(-1000);
  }

  // Log health status
  if (healthStatus.overall !== 'healthy') {
    logger.warn('Application health check failed', { healthStatus });
  }

  return healthStatus;
}

// Error pattern analysis
function analyzeErrorPatterns() {
  const recentErrors = metricsHistory.errors.slice(-100);
  const errorPatterns = {};

  recentErrors.forEach(error => {
    const errorKey = error.type || error.message?.substring(0, 50) || 'unknown';
    if (!errorPatterns[errorKey]) {
      errorPatterns[errorKey] = {
        count: 0,
        firstSeen: error.timestamp,
        lastSeen: error.timestamp,
        examples: []
      };
    }
    
    errorPatterns[errorKey].count++;
    errorPatterns[errorKey].lastSeen = error.timestamp;
    
    if (errorPatterns[errorKey].examples.length < 3) {
      errorPatterns[errorKey].examples.push({
        message: error.message,
        timestamp: error.timestamp
      });
    }
  });

  // Identify recurring errors
  const recurringErrors = Object.entries(errorPatterns)
    .filter(([_, pattern]) => pattern.count > 3)
    .map(([type, pattern]) => ({
      type,
      ...pattern,
      frequency: pattern.count / ((new Date(pattern.lastSeen) - new Date(pattern.firstSeen)) / (1000 * 60 * 60)) // errors per hour
    }));

  return {
    totalErrors: recentErrors.length,
    uniqueErrorTypes: Object.keys(errorPatterns).length,
    recurringErrors,
    patterns: errorPatterns
  };
}
