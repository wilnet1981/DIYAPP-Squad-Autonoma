import { useEffect, useRef, useState } from 'react';

export interface UseLazyLoadOptions {
  root?: Element | null;
  rootMargin?: string;
  threshold?: number | number[];
  enabled?: boolean;
}

export function useLazyLoad<T extends HTMLElement = HTMLElement>({
  root = null,
  rootMargin = '0px',
  threshold = 0,
  enabled = true,
}: UseLazyLoadOptions = {}) {
  const ref = useRef<T>(null);
  const [isVisible, setIsVisible] = useState(false);

  useEffect(() => {
    if (!enabled || !ref.current || isVisible) return;

    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsVisible(true);
          observer.disconnect();
        }
      },
      { root, rootMargin, threshold }
    );

    observer.observe(ref.current);

    return () => {
      observer.disconnect();
    };
  }, [enabled, root, rootMargin, threshold, isVisible]);

  return { ref, isVisible };
}

const requestLogger = (req, res, next) => {
  const startTime = Date.now();
  const correlationId = req.headers['x-correlation-id'] || require('crypto').randomUUID();
  
  // Adiciona correlation ID ao request
  req.correlationId = correlationId;
  res.setHeader('X-Correlation-ID', correlationId);
  
  // Loga o request
  logger.info('Incoming request', {
    correlation_id: correlationId,
    method: req.method,
    url: req.originalUrl,
    ip: req.ip,
    user_agent: req.get('user-agent'),
    user_id: req.user?.id || 'anonymous'
  });
  
  // Captura o response
  const originalSend = res.send;
  res.send = function(body) {
    const duration = Date.now() - startTime;
    
    logger.info('Request completed', {
      correlation_id: correlationId,
      method: req.method,
      url: req.originalUrl,
      status_code: res.statusCode,
      duration_ms: duration,
      user_id: req.user?.id || 'anonymous'
    });
    
    // Alerta se latência > 300ms (P95 threshold)
    if (duration > 300) {
      logger.warn('High latency detected', {
        correlation_id: correlationId,
        duration_ms: duration,
        threshold_ms: 300,
        endpoint: `${req.method} ${req.originalUrl}`
      });
    }
    
    return originalSend.call(this, body);
  };

        });

        correlation_id: correlationId,

      });

    const client = await this.pool.connect();

      const result = await callback(client);
      await client.query('COMMIT');
      return result;
    } catch (error) {
      await client.query('ROLLBACK');
      throw error;
    } finally {

WITH daily_events AS (
    SELECT
        event_date,
        COUNT(DISTINCT user_id) as daily_active_users,
        COUNT(DISTINCT session_id) as total_sessions,
        COUNT(*) as total_events,
        COUNT(DISTINCT CASE WHEN standardized_event_type = 'click' THEN session_id END) as sessions_with_clicks,
        COUNT(DISTINCT CASE WHEN page_section = 'builder' THEN user_id END) as builder_users,
        COUNT(DISTINCT CASE WHEN page_section = 'marketplace' THEN user_id END) as marketplace_users,
        AVG(CASE WHEN standardized_event_type = 'click' THEN 1 ELSE 0 END) as click_rate
    FROM {{ ref('stg_events') }}
    GROUP BY 1
),

feature_adoption AS (
    SELECT
        event_date,
        page_section as feature,
        COUNT(DISTINCT user_id) as daily_users,
        COUNT(*) as daily_interactions,
        COUNT(DISTINCT session_id) as daily_sessions
    FROM {{ ref('stg_events') }}
    WHERE page_section IS NOT NULL
    GROUP BY 1, 2
),

funnel_conversion AS (
    WITH user_journey AS (
        SELECT
            user_id,
            session_id,
            MIN(CASE WHEN page_section = 'landing' THEN event_timestamp END) as landing_time,
            MIN(CASE WHEN page_section = 'builder' THEN event_timestamp END) as builder_time,
            MIN(CASE WHEN page_section = 'checkout' THEN event_timestamp END) as checkout_time,
            MIN(CASE WHEN element_id = 'purchase_button' THEN event_timestamp END) as purchase_time
        FROM {{ ref('stg_events') }}
        GROUP BY 1, 2
    )
    
    SELECT
        event_date,
        COUNT(DISTINCT CASE WHEN landing_time IS NOT NULL THEN session_id END) as landing_sessions,
        COUNT(DISTINCT CASE WHEN builder_time IS NOT NULL THEN session_id END) as builder_sessions,
        COUNT(DISTINCT CASE WHEN checkout_time IS NOT NULL THEN session_id END) as checkout_sessions,
        COUNT(DISTINCT CASE WHEN purchase_time IS NOT NULL THEN session_id END) as purchase_sessions,
        ROUND(
            COUNT(DISTINCT CASE WHEN builder_time IS NOT NULL THEN session_id END) * 100.0 / 
            NULLIF(COUNT(DISTINCT CASE WHEN landing_time IS NOT NULL THEN session_id END), 0), 2
        ) as landing_to_builder_conversion,
        ROUND(
            COUNT(DISTINCT CASE WHEN purchase_time IS NOT NULL THEN session_id END) * 100.0 / 
            NULLIF(COUNT(DISTINCT CASE WHEN landing_time IS NOT NULL THEN session_id END), 0), 2
        ) as overall_conversion
    FROM {{ ref('stg_events') }} e
    LEFT JOIN user_journey j ON e.session_id = j.session_id
    GROUP BY 1
)

SELECT
    de.event_date,
    de.daily_active_users,
    de.total_sessions,
    de.total_events,
    de.sessions_with_clicks,
    de.builder_users,
    de.marketplace_users,
    de.click_rate,
    fa.feature,
    fa.daily_users as feature_daily_users,
    fa.daily_interactions as feature_interactions,
    fc.landing_sessions,
    fc.builder_sessions,
    fc.checkout_sessions,
    fc.purchase_sessions,
    fc.landing_to_builder_conversion,
    fc.overall_conversion,
    CURRENT_TIMESTAMP as calculated_at
FROM daily_events de
LEFT JOIN feature_adoption fa ON de.event_date = fa.event_date
LEFT JOIN funnel_conversion fc ON de.event_date = fc.event_date

            if metric in metrics_data.columns:
                features[f'latency_{metric}'] = metrics_data[metric]
        
        for metric in self.config['features']['error_metrics']:
            if metric in metrics_data.columns:
                features[f'error_{metric}'] = metrics_data[metric]
        
        for metric in self.config['features']['resource_metrics']:
            if metric in metrics_data.columns:
                features[f'resource_{metric}'] = metrics_data[metric]
        
        # Features derivadas
        if 'p95' in metrics_data.columns and 'p50' in metrics_data.columns:
            features['latency_spread'] = metrics_data['p95'] - metrics_data['p50']
        
        if 'error_rate' in metrics_data.columns:
            features['error_rate_trend'] = metrics_data['error_rate'].rolling(5).mean()
        
        # Features temporais
        if 'timestamp' in metrics_data.columns:
            metrics_data['timestamp'] = pd.to_datetime(metrics_data['timestamp'])
            features['hour_of_day'] = metrics_data['timestamp'].dt.hour
            features['day_of_week'] = metrics_data['timestamp'].dt.dayofweek
            features['is_weekend'] = features['day_of_week'].isin([5, 6]).astype(int)
        
        # Remover NaN
        features = features.fillna(method='ffill').fillna(0)
        
        self.feature_columns = features.columns.tolist()
        return features
    
    def train(self, historical_data: pd.DataFrame):
        """Treina o modelo com dados históricos"""
        print(f"Training anomaly detector with {len(historical_data)} samples")
        
        # Preparar features
        features = self.prepare_features(historical_data)
        
        # Normalizar features
        scaled_features = self.scaler.fit_transform(features)
        
        # Treinar modelo
        self.model.fit(scaled_features)
        
        # Calcular scores nos dados de treino
        train_scores = self.model.decision_function(scaled_features)
        self.anomaly_scores['train_mean'] = np.mean(train_scores)
        self.anomaly_scores['train_std'] = np.std(train_scores)
        
        self.last_training_time = datetime.now()
        
        # Salvar modelo
        self.save_model()
        
        print(f"Model trained successfully. Mean score: {self.anomaly_scores['train_mean']:.4f}")
    
    def detect(self, current_metrics: Dict) -> Tuple[bool, float, Dict]:
        """Detecta anomalias em métricas atuais"""
        
        # Verificar se precisa retreinar
        if self.should_retrain():
            print("Model retraining required")
            # Em produção, buscar dados históricos e retreinar
        
        # Converter para DataFrame
        metrics_df = pd.DataFrame([current_metrics])
        
        # Preparar features
        features = self.prepare_features(metrics_df)
        
        # Garantir que todas as colunas estão presentes
        missing_cols = set(self.feature_columns) - set(features.columns)
        for col in missing_cols:
            features[col] = 0
        
        features = features[self.feature_columns]
        
        # Normalizar
        scaled_features = self.scaler.transform(features)
        
        # Prever

            if feature in features:
                value = features[feature]
                
                # Calcular z-score (simplificado)
                if hasattr(self.scaler, 'mean_') and feature in self.scaler.mean_:
                    idx = list(self.scaler.mean_).index(self.scaler.mean_[feature])
                    mean = self.scaler.mean_[idx]
                    std = max(self.scaler.scale_[idx], 0.001)
                    z_score = abs((value - mean) / std)
                    
                    if z_score > 2.0:  # Significativamente fora do normal
                        explanation['top_contributors'].append({
                            'feature': feature,
                            'value': float(value),
                            'z_score': float(z_score),
                            'contribution': float(z_score * 0.1)  # Peso simplificado
                        })
                
                explanation['feature_values'][feature] = float(value)
        
        # Ordenar por contribuição
        explanation['top_contributors'].sort(key=lambda x: x['contribution'], reverse=True)
        explanation['top_contributors'] = explanation['top_contributors'][:5]  # Top 5
        
        return explanation
    
    def should_retrain(self) -> bool:
        """Verifica se o modelo precisa ser retreinado"""
        if not self.last_training_time:
            return True
        
        retrain_interval = timedelta(hours=self.config['anomaly_detection']['retrain_interval_hours'])
        return datetime.now() - self.last_training_time > retrain_interval
    
    def save_model(self):
        """Salva o modelo treinado"""
        model_data = {
            'model': self.model,
            'scaler': self.scaler,
            'feature_columns': self.feature_columns,
            'anomaly_scores': self.anomaly_scores,
            'last_training_time': self.last_training_time
        }
        
        with open('src/ml/models/anomaly_detector.pkl', 'wb') as f:
            pickle.dump(model_data, f)
    
    def load_model(self):
        """Carrega modelo salvo"""
        try:
            with open('src/ml/models/anomaly_detector.pkl', 'rb') as f:
                model_data = pickle.load(f)
            
            self.model = model_data['model']
            self.scaler = model_data['scaler']
            self.feature_columns = model_data['feature_columns']
            self.anomaly_scores = model_data['anomaly_scores']
            self.last_training_time = model_data['last_training_time']
            
            print("Anomaly detector model loaded successfully")
        except FileNotFoundError:
            print("No saved model found. Initializing new model.")
            self.initialize_model()

## 4. dbt/models/staging/stg_clicks.sql - Pipeline de Clicks

```sql
{{
    config(
        materialized='view',
        schema='staging'
    )
}}

WITH raw_clicks AS (
    SELECT
        click_id,
        user_id,
        session_id,
        page_url,
        element_id,
        event_timestamp,
        device_type,
        browser,
        country,
        ROW_NUMBER() OVER (PARTITION BY click_id ORDER BY event_timestamp) as rn
    FROM {{ source('raw', 'clicks') }}
    WHERE event_timestamp IS NOT NULL
        AND user_id IS NOT NULL
        AND click_id IS NOT NULL
),

cleaned_clicks AS (
    SELECT
        click_id,
        user_id,
        session_id,
        page_url,
        element_id,
        -- Standardize timestamp
        CAST(event_timestamp AS TIMESTAMP) as event_timestamp,
        -- Clean device type
        CASE 
            WHEN device_type IN ('mobile', 'Mobile', 'MOBILE') THEN 'mobile'
            WHEN device_type IN ('desktop', 'Desktop', 'DESKTOP') THEN 'desktop'
            WHEN device_type IN ('tablet', 'Tablet', 'TABLET') THEN 'tablet'
            ELSE 'unknown'
        END as device_type,
        -- Clean browser
        LOWER(browser) as browser,
        -- Clean country (uppercase)
        UPPER(TRIM(country)) as country
    FROM raw_clicks
    WHERE rn = 1  -- Remove duplicates
)

SELECT * FROM cleaned_clicks
```

## 5. dbt/models/marts/mart_product/daily_active_users.sql

```sql
{{
    config(
        materialized='table',
        schema='marts_product',
        tags=['product', 'daily', 'kpi']
    )
}}

WITH daily_sessions AS (
    SELECT
        DATE(event_timestamp) as activity_date,
        user_id,
        COUNT(DISTINCT session_id) as session_count,
        COUNT(*) as total_clicks
    FROM {{ ref('stg_clicks') }}
    WHERE event_timestamp >= DATEADD(day, -30, CURRENT_DATE())
    GROUP BY 1, 2
),

daily_features AS (
    SELECT
        DATE(event_timestamp) as activity_date,
        user_id,
        COUNT(DISTINCT page_url) as distinct_pages_visited,
        COUNT(DISTINCT element_id) as distinct_features_used
    FROM {{ ref('stg_clicks') }}
    WHERE element_id LIKE 'feature_%'
    GROUP BY 1, 2
),

user_segments AS (
    SELECT
        user_id,
        subscription_tier,
        country,
        CASE 
            WHEN last_login >= DATEADD(day, -7, CURRENT_DATE()) THEN 'active'
            WHEN last_login >= DATEADD(day, -30, CURRENT_DATE()) THEN 'dormant'
            ELSE 'inactive'
        END as engagement_segment
    FROM {{ ref('stg_users') }}
)

SELECT
    ds.activity_date,
    ds.user_id,
    us.subscription_tier,
    us.country,
    us.engagement_segment,
    ds.session_count,
    ds.total_clicks,
    COALESCE(df.distinct_pages_visited, 0) as distinct_pages_visited,
    COALESCE(df.distinct_features_used, 0) as distinct_features_used,
    CASE 
        WHEN ds.session_count >= 3 THEN 'high_engagement'
        WHEN ds.session_count >= 1 THEN 'medium_engagement'
        ELSE 'low_engagement'
    END as daily_engagement_level,
    -- Calculate retention cohort
    MIN(ds.activity_date) OVER (PARTITION BY ds.user_id) as first_activity_date,
    DATEDIFF(day, 
        MIN(ds.activity_date) OVER (PARTITION BY ds.user_id), 
        ds.activity_date
    ) as days_since_first_activity
FROM daily_sessions ds
LEFT JOIN daily_features df 
    ON ds.activity_date = df.activity_date 
    AND ds.user_id = df.user_id
LEFT JOIN user_segments us 
    ON ds.user_id = us.user_id
WHERE ds.activity_date >= DATEADD(day, -30, CURRENT_DATE())
```

## 2. Web Application Firewall (WAF)

**src/security/waf/middleware.js**
```javascript
const rateLimit = require('express-rate-limit');
const helmet = require('helmet');
const { body, validationResult } = require('express-validator');
const rules = require('./rules');
const SecurityMonitor = require('../monitoring/security-monitor');

class WAFMiddleware {
    constructor() {
        this.monitor = new SecurityMonitor();
        this.rules = rules;
    }

    // Middleware principal do WAF
    async protect(req, res, next) {
        const clientIP = req.ip || req.connection.remoteAddress;
        const userAgent = req.get('User-Agent') || '';
        const requestId = this.generateRequestId();

        // Log da requisição
        this.monitor.logRequest({
            requestId,
            ip: clientIP,
            method: req.method,
            url: req.url,
            userAgent,
            timestamp: new Date()
        });

        try {
            // 1. Verificação de Rate Limiting
            await this.checkRateLimit(req, res);

            // 2. Validação de Input Básica
            await this.validateInput(req);

            // 3. Verificação contra SQL Injection
            if (this.detectSQLInjection(req)) {
                throw new Error('SQL Injection attempt detected');
            }

            // 4. Verificação contra XSS
            if (this.detectXSS(req)) {
                throw new Error('XSS attempt detected');
            }

            // 5. Verificação de tamanho de payload
            if (this.checkPayloadSize(req)) {
                throw new Error('Payload size exceeds limit');
            }

            // 6. Verificação de User-Agent suspeito
            if (this.checkSuspiciousUserAgent(userAgent)) {
                this.monitor.logThreat('SUSPICIOUS_USER_AGENT', {
                    ip: clientIP,
                    userAgent,
                    requestId
                });
            }

            // Adiciona headers de segurança
            this.addSecurityHeaders(res);

            next();
        } catch (error) {
            this.monitor.logThreat('WAF_BLOCKED', {
                ip: clientIP,
                reason: error.message,
                requestId,
                url: req.url
            });

            return res.status(403).json({
                error: 'Access blocked by security policy',
                requestId,
                timestamp: new Date().toISOString()
            });
        }
    }

    // Rate Limiting por IP
    createRateLimiter() {
        return rateLimit({
            windowMs: 15 * 60 * 1000, // 15 minutos
            max: 100, // limite por IP
            message: {
                error: 'Too many requests from this IP',
                retryAfter: '15 minutes'
            },
            standardHeaders: true,
            legacyHeaders: false
        });
    }

    // Validação de input
    async validateInput(req) {
        const validationChain = [];

        // Validação para POST/PUT requests
        if (['POST', 'PUT', 'PATCH'].includes(req.method)) {
            Object.keys(req.body).forEach(key => {
                validationChain.push(
                    body(key)
                        .trim()
                        .escape()
                        .not().isEmpty()
                        .withMessage(`${key} cannot be empty`)
                );
            });

            const errors = validationResult(req);
            if (!errors.isEmpty()) {
                throw new Error(`Input validation failed: ${errors.array()[0].msg}`);
            }
        }
    }

    // Detecção de SQL Injection
    detectSQLInjection(req) {
        const sqlKeywords = [
            'SELECT', 'INSERT', 'UPDATE', 'DELETE', 'DROP', 'UNION',
            'OR', 'AND', 'WHERE', 'FROM', 'JOIN', 'HAVING', 'GROUP BY'
        ];
        
        const searchPatterns = [
            /(\%27)|(\')|(\-\-)|(\%23)|(#)/i,
            /((\%3D)|(=))[^\n]*((\%27)|(\')|(\-\-)|(\%3B)|(;))/i,
            /w*((\%27)|(\'))((\%6F)|o|(\%4F))((\%72)|r|(\%52))/i
        ];

        const checkString = (str) => {
            // Verifica keywords SQL
            const upperStr = str.toUpperCase();
            for (const keyword of sqlKeywords) {
                if (upperStr.includes(keyword) && 
                    (upperStr.includes('FROM') || upperStr.includes('WHERE'))) {
                    return true;
                }
            }

            // Verifica padrões de injeção
            for (const pattern of searchPatterns) {
                if (pattern.test(str)) {
                    return true;
                }
            }

            return false;
        };

        // Verifica query parameters
        if (req.query) {
            for (const [key, value] of Object.entries(req.query)) {
                if (checkString(value.toString())) {
                    return true;
                }
            }
        }

        // Verifica body parameters
        if (req.body) {
            for (const [key, value] of Object.entries(req.body)) {
                if (typeof value === 'string' && checkString(value)) {
                    return true;
                }
            }
        }

        return false;
    }

    // Detecção de XSS
    detectXSS(req) {
        const xssPatterns = [
            /<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi,
            /javascript:/gi,
            /on\w+\s*=/gi,
            /eval\s*\(/gi,
            /expression\s*\(/gi,
            /vbscript:/gi
        ];

        const checkString = (str) => {
            for (const pattern of xssPatterns) {
                if (pattern.test(str)) {
                    return true;
                }
            }
            return false;
        };

        // Verifica todos os inputs
        const inputs = { ...req.query, ...req.body, ...req.params };
        
        for (const [key, value] of Object.entries(inputs)) {
            if (typeof value === 'string' && checkString(value)) {
                return true;
            }
        }

        return false;
    }

    // Verificação de tamanho de payload
    checkPayloadSize(req) {
        const contentLength = parseInt(req.get('Content-Length') || '0');
        const maxSize = 10 * 1024 * 1024; // 10MB
        
        return contentLength > maxSize;
    }

    // Verificação de User-Agent suspeito
    checkSuspiciousUserAgent(userAgent) {
        const suspiciousPatterns = [
            /sqlmap/i,
            /nikto/i,
            /metasploit/i,
            /nmap/i,
            /wget/i,
            /curl/i,
            /python-requests/i,
            /libwww-perl/i
        ];

        return suspiciousPatterns.some(pattern => pattern.test(userAgent));
    }

    // Adiciona headers de segurança
    addSecurityHeaders(res) {
        res.setHeader('X-Content-Type-Options', 'nosniff');
        res.setHeader('X-Frame-Options', 'DENY');
        res.setHeader('X-XSS-Protection', '1; mode=block');
        res.setHeader('Referrer-Policy', 'strict-origin-when-cross-origin');
        res.setHeader('Permissions-Policy', 'geolocation=(), microphone=(), camera=()');
    }

    generateRequestId() {
        return crypto.randomBytes(16).toString('hex');
    }
}

module.exports = WAFMiddleware;
```

const speakeasy = require('speakeasy');
const QRCode = require('qrcode');
const crypto = require('crypto');
const config = require('../../../config/security-config');

class MFAService {
    constructor() {
        this.backupCodes = new Map();
    }

    // Gera segredo para um usuário
    generateSecret(userId) {
        const secret = speakeasy.generateSecret({
            name: `DIYAPP:${userId}`,
            length: 20
        });

        return {
            secret: secret.base32,
            otpauth_url: secret.otpauth_url
        };
    }

    // Gera QR Code para configuração
    async generateQRCode(otpauthUrl) {
        try {
            return await QRCode.toDataURL(otpauthUrl);
        } catch (error) {
            throw new Error('Failed to generate QR code');
        }
    }

    // Verifica token TOTP
    verifyToken(secret, token) {
        return speakeasy.totp.verify({
            secret: secret,
            encoding: 'base32',
            token: token,
            window: 1 // Permite 30 segundos de tolerância
        });
    }

    // Gera códigos de backup
    generateBackupCodes(userId, count = config.mfa.backupCodesCount) {
        const codes = [];
        
        for (let i = 0; i < count; i++) {
            const code = crypto.randomBytes(4).toString('hex').toUpperCase();
            codes.push(code);
        }

        // Armazena hash dos códigos
        const hashedCodes = codes.map(code => 
            crypto.createHash('sha256').update(code).digest('hex')
        );
        
        this.backupCodes.set(userId, hashedCodes);
        
        return codes;
    }

    // Verifica código de backup
    verifyBackupCode(userId, code) {
        const hashedCodes = this.backupCodes.get(userId);
        if (!hashedCodes) return false;

        const hashedCode = crypto.createHash('sha256').update(code).digest('hex');
        const index = hashedCodes.indexOf(hashedCode);

        if (index !== -1) {
            // Remove o código usado
            hashedCodes.splice(index, 1);
            this.backupCodes.set(userId, hashedCodes);
            return true;
        }

        return false;
    }

    // Verifica se usuário precisa de MFA
    requiresMFA(user, requestContext) {
        // Administradores sempre precisam de MFA
        if (user.role === 'admin' && config.mfa.requiredForAdmin) {
            return true;
        }

        // Verifica se é um dispositivo novo
        if (requestContext.isNewDevice) {
            return true;
        }

        // Verifica localização suspeita
        if (requestContext.suspiciousLocation) {
            return true;
        }

        return config.mfa.optionalForUsers ? user.mfaEnabled : false;
    }

    // Gera token de recuperação
    generateRecoveryToken(userId) {
        const token = crypto.randomBytes(32).toString('hex');
        const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000); // 24 horas
        
        return {
            token,
            expiresAt
        };
    }
}

module.exports = MFAService;

const crypto = require('crypto');
const config = require('../../../config/security-config');

class EncryptionService {

    generateKey() {
        return crypto.randomBytes(this.keyLength);
    }

    // Deriva chave de uma senha
    deriveKeyFromPassword(password, salt) {
        return crypto.pbkdf2Sync(
            password,
            salt,
            100000, // iterations
            this.keyLength,
            'sha256'
        );
    }

    // Criptografa dados
    encrypt(text, key) {
        const iv = crypto.randomBytes(this.ivLength);
        const cipher = crypto.createCipheriv(this.algorithm, key, iv);
        
        let encrypted = cipher.update(text, 'utf8', 'hex');
        encrypted += cipher.final('hex');
        
        const authTag = cipher.getAuthTag();
        
        return {
            iv: iv.toString('hex'),
            encryptedData: encrypted,
            authTag: authTag.toString('hex')
        };
    }

    // Descriptografa dados
    decrypt(encryptedObject, key) {
        const decipher = crypto.createDecipheriv(
            this.algorithm,
            key,
            Buffer.from(encryptedObject.iv, 'hex')
        );
        
        decipher.setAuthTag(Buffer.from(encryptedObject.authTag, 'hex'));
        
        let decrypted = decipher.update(encryptedObject.encryptedData, 'hex', 'utf8');
        decrypted += decipher.final('utf8');
        
        return decrypted;
    }

        if (typeof data !== 'object') {
