## 1. ADR-001: Arquitetura de Microsserviços V13

```markdown
# ADR-001: Arquitetura de Microsserviços para DIYAPP V13

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead + Especialista Infra

## CONTEXTO:
DIYAPP evoluiu de uma aplicação monolítica para uma plataforma distribuída com múltiplos times trabalhando em paralelo. Necessitamos de uma arquitetura que permita:
- Desenvolvimento independente por squads autônomas
- Escalabilidade horizontal por componente
- Deploy contínuo sem downtime
- Observabilidade granular
- Resiliência a falhas

## DECISÃO:
Adotar arquitetura de microsserviços com os seguintes princípios:
1. Cada serviço é responsável por um domínio de negócio específico
2. Comunicação via gRPC para serviços internos, REST para APIs públicas
3. Banco de dados por serviço (Database per Service)
4. Service Mesh (Istio) para comunicação entre serviços
5. Event-driven architecture com Apache Kafka para eventos de domínio

## OPÇÕES CONSIDERADAS:
- **Opção A:** Monólito modularizado - Prós: Simplicidade operacional. Contras: Acoplamento alto, escalabilidade limitada.
- **Opção B:** Microsserviços completos - Prós: Autonomia total, escalabilidade independente. Contras: Complexidade operacional.
- **Opção C:** Arquitetura hexagonal com módulos - Prós: Balanceamento entre simplicidade e autonomia. Contras: Compromisso que pode não atender crescimento.

**Opção escolhida:** B - Justificativa: A necessidade de squads totalmente autônomas e a previsão de crescimento rápido justificam a complexidade inicial.

## CONSEQUÊNCIAS:
**Positivas:**
- Desenvolvimento paralelo ilimitado
- Escalabilidade granular
- Tecnologia adequada por domínio
- Deploy independente

**Negativas:**
- Complexidade de operações distribuídas
- Overhead de comunicação
- Gerenciamento de dados distribuídos
- Monitoramento mais complexo

**Riscos:**
- Latência em chamadas entre serviços
- Consistência eventual
- Mitigação: Circuit breakers, caching estratégico, padrões SAGA

**REVISÃO:** 2024-04-15
```

```go
// packages/shared/telemetry/telemetry.go
package telemetry

import (
	"context"
	"go.opentelemetry.io/otel"
	"go.opentelemetry.io/otel/exporters/jaeger"
	"go.opentelemetry.io/otel/exporters/prometheus"
	"go.opentelemetry.io/otel/metric/global"
	"go.opentelemetry.io/otel/propagation"
	"go.opentelemetry.io/otel/sdk/metric"
	"go.opentelemetry.io/otel/sdk/resource"
	sdktrace "go.opentelemetry.io/otel/sdk/trace"
	semconv "go.opentelemetry.io/otel/semconv/v1.17.0"
	"log"
	"os"
)

var (
	Tracer  = otel.Tracer("diyapp")
	Meter   = global.Meter("diyapp")
	service = "unknown"
)

func InitTelemetry(serviceName string) (func(), error) {
	service = serviceName
	
	// Resource
	res, err := resource.New(context.Background(),
		resource.WithAttributes(
			semconv.ServiceName(serviceName),
			semconv.ServiceVersion("v13.0.0"),
			semconv.DeploymentEnvironment(os.Getenv("ENV")),
		),
	)
	if err != nil {
		return nil, err
	}

	// Trace Provider
	traceExporter, err := jaeger.New(jaeger.WithCollectorEndpoint(
		jaeger.WithEndpoint(os.Getenv("JAEGER_ENDPOINT")),
	))
	if err != nil {
		return nil, err
	}

	tp := sdktrace.NewTracerProvider(
		sdktrace.WithBatcher(traceExporter),
		sdktrace.WithResource(res),
		sdktrace.WithSampler(sdktrace.AlwaysSample()),
	)
	otel.SetTracerProvider(tp)
	otel.SetTextMapPropagator(propagation.NewCompositeTextMapPropagator(
		propagation.TraceContext{},
		propagation.Baggage{},
	))

	// Metric Provider
	metricExporter, err := prometheus.New()
	if err != nil {
		return nil, err
	}

	mp := metric.NewMeterProvider(
		metric.WithResource(res),
		metric.WithReader(metricExporter),
	)
	global.SetMeterProvider(mp)

	// Shutdown function
	shutdown := func() {
		if err := tp.Shutdown(context.Background()); err != nil {
			log.Printf("Error shutting down tracer provider: %v", err)
		}
		if err := mp.Shutdown(context.Background()); err != nil {
			log.Printf("Error shutting down meter provider: %v", err)
		}
	}

	return shutdown, nil
}

// Middleware para HTTP
func HTTPMiddleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		ctx, span := Tracer.Start(r.Context(), r.URL.Path)
		defer span.End()

		// Add attributes
		span.SetAttributes(
			semconv.HTTPMethod(r.Method),
			semconv.HTTPRoute(r.URL.Path),
			semconv.HTTPUserAgent(r.UserAgent()),
		)

		// Metrics
		requestCounter, _ := Meter.Int64Counter("http_requests_total")
		requestCounter.Add(ctx, 1,
			attribute.String("method", r.Method),
			attribute.String("path", r.URL.Path),
			attribute.String("service", service),
		)

		next.ServeHTTP(w, r.WithContext(ctx))
	})
}
```

## 5. CI/CD Pipeline (GitHub Actions)

```yaml
# .github/workflows/ci.yml
name: CI Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [auth, user, content, notification]
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Setup Go
      uses: actions/setup-go@v4
      with:
        go-version: '1.21'
    
    - name: Run unit tests
      run: |
        cd services/${{ matrix.service }}-service
        go test ./... -v -coverprofile=coverage.out
        go tool cover -func=coverage.out
    
    - name: Upload coverage
      uses: codecov/codecov-action@v3
      with:
        file: ./services/${{ matrix.service }}-service/coverage.out
        flags: unittests
    
    - name: Security scan
      uses: aquasecurity/trivy-action@master
      with:
        scan-type: 'fs'
        scan-ref: './services/${{ matrix.service }}-service'
        format: 'sarif'
        output: 'trivy-results.sarif'
    
    - name: Upload SARIF
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: 'trivy-results.sarif'

  build:
    needs: test
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [auth, user, content, notification]
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Set up Docker Buildx
      uses: docker/setup-buildx-action@v2
    
    - name: Login to DockerHub
      uses: docker/login-action@v2
      with:
        username: ${{ secrets.DOCKER_USERNAME }}
        password: ${{ secrets.DOCKER_PASSWORD }}
    
    - name: Build and push
      uses: docker/build-push-action@v4
      with:
        context: ./services/${{ matrix.service }}-service
        push: true
        tags: |
          diyapp/${{ matrix.service }}-service:${{ github.sha }}
          diyapp/${{ matrix.service }}-service:latest
        cache-from: type=gha
        cache-to: type=gha,mode=max
```

```yaml
# .github/workflows/cd.yml
name: CD Pipeline

on:
  workflow_run:
    workflows: ["CI Pipeline"]
    branches: [main]
    types:
      - completed

jobs:
  deploy-staging:
    if: ${{ github.event.workflow_run.conclusion == 'success' }}
    runs-on: ubuntu-latest
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Update ArgoCD manifests
      run: |
        # Update image tags in ArgoCD manifests
        for service in auth user content notification; do
          sed -i "s/{{ .Values.tag }}/${{ github.sha }}/g" argo/applications/$service-service.yaml
        done
        
        git config user.name "GitHub Actions"
        git config user.email "actions@github.com"
        git add argo/applications/*.yaml
        git commit -m "chore: update image tags to ${{ github.sha }}"
        git push
    
    - name: Sync ArgoCD
      uses: argoproj/actions-cli@v1
      with:
        command: app sync diyapp-v13
        version: '2.7'
        args: '--prune'
      env:
        ARGOCD_SERVER: ${{ secrets.ARGOCD_SERVER }}
        ARGOCD_AUTH_TOKEN: ${{ secrets.ARGOCD_AUTH_TOKEN }}

  canary-deploy:
    needs: deploy-staging
    runs-on: ubuntu-latest
    
    steps:
    - name: Run integration tests
      run: |
        # Run API tests against staging
        ./scripts/run-integration-tests.sh
    
    - name: Promote to production
      if: success()
      uses: argoproj/actions-cli@v1
      with:
        command: app sync diyapp-v13
        version: '2.7'
        args: '--prune --project diyapp-production'
      env:
        ARGOCD_SERVER: ${{ secrets.ARGOCD_SERVER }}
        ARGOCD_AUTH_TOKEN: ${{ secrets.ARGOCD_AUTH_TOKEN }}
    
    - name: Rollback on failure
      if: failure()
      uses: argoproj/actions-cli@v1
      with:
        command: app rollback diyapp-v13
        version: '2.7'
        args: '--prune'
      env:
        ARGOCD_SERVER: ${{ secrets.ARGOCD_SERVER }}
        ARGOCD_AUTH_TOKEN: ${{ secrets.ARGOCD_AUTH_TOKEN }}
```

```javascript
const winston = require('winston');
const config = require('../config');

const { combine, timestamp, json, errors, metadata } = winston.format;

// Formato para logs estruturados
const logFormat = combine(
  timestamp({ format: 'YYYY-MM-DD HH:mm:ss.SSS' }),
  errors({ stack: true }),
  metadata(),
  json()
);

// Filtro para mascarar dados sensíveis
const maskSensitiveData = winston.format((info) => {
  const maskedInfo = { ...info };
  
  // Campos sensíveis para mascarar
  const sensitiveFields = [
    'password', 'token', 'authorization', 'jwt', 
    'email', 'cpf', 'credit_card', 'card_number',
    'cvv', 'secret', 'api_key', 'private_key'
  ];
  
  // Mascara dados sensíveis no message
  if (maskedInfo.message && typeof maskedInfo.message === 'string') {
    sensitiveFields.forEach(field => {
      const regex = new RegExp(`(${field}[:=]\\s*)([^\\s,]+)`, 'gi');
      maskedInfo.message = maskedInfo.message.replace(regex, `$1[REDACTED]`);
    });
  }
  
  // Mascara dados sensíveis no metadata
  sensitiveFields.forEach(field => {
    if (maskedInfo[field]) {
      maskedInfo[field] = '[REDACTED]';
    }
    if (maskedInfo.metadata && maskedInfo.metadata[field]) {
      maskedInfo.metadata[field] = '[REDACTED]';
    }
  });
  
  return maskedInfo;
});

const logger = winston.createLogger({
  level: config.logLevel,
  format: combine(
    maskSensitiveData(),
    logFormat
  ),
  defaultMeta: { service: 'api-gateway' },
  transports: [
    new winston.transports.Console({
      format: combine(
        maskSensitiveData(),
        winston.format.colorize(),
        winston.format.simple()
      )
    }),
    new winston.transports.File({

// Middleware para logging de requests
logger.requestLogger = (req, res, next) => {
  const start = Date.now();
  const correlationId = req.headers['x-correlation-id'] || require('uuid').v4();
  
  // Adiciona correlation ID ao request
  req.correlationId = correlationId;
  res.setHeader('X-Correlation-ID', correlationId);
  
  // Log do request
  logger.info('Request received', {
    correlation_id: correlationId,
    method: req.method,
    url: req.originalUrl,
    ip: req.ip,
    user_agent: req.get('user-agent'),
    user_id: req.user ? req.user.id : null
  });
  
  // Intercepta o response para logging
  const originalSend = res.send;
  res.send = function(body) {
    const duration = Date.now() - start;
    
    logger.info('Response sent', {
      correlation_id: correlationId,

      content_length: res.get('Content-Length') || Buffer.byteLength(body || ''),
      user_id: req.user ? req.user.id : null
    });
    
    return originalSend.call(this, body);
  };
  
  next();
```

```javascript
const jwt = require('jsonwebtoken');
const { AppError, Errors } = require('../utils/error-handler');
const config = require('../config');
const logger = require('../utils/logger');

// Cache para tokens revogados (em produção usar Redis)
const tokenBlacklist = new Set();

/**
 * Middleware de autenticação JWT
 */
const authenticate = async (req, res, next) => {
  try {
    // Extrai token do header
    const authHeader = req.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      throw new AppError(
        Errors.UNAUTHORIZED.message,
        Errors.UNAUTHORIZED.status,
        Errors.UNAUTHORIZED.code
      );
    }
    
    const token = authHeader.split(' ')[1];
    
    // Verifica se token está na blacklist
    if (tokenBlacklist.has(token)) {
      throw new AppError(
        'Token revogado',
        401,
        'TOKEN_REVOKED'
      );
    }
    
    // Verifica e decodifica token
    const decoded = jwt.verify(token, config.jwtSecret);
    
    // Adiciona usuário ao request
    req.user = {
      id: decoded.userId,
      email: decoded.email,
      role: decoded.role,
      permissions: decoded.permissions || []
    };
    
    // Log de autenticação bem-sucedida
    logger.debug('Authentication successful', {
      correlation_id: req.correlationId,
      user_id: req.user.id,
      role: req.user.role
    });
    
    next();
  } catch (error) {
    if (error.name === 'JsonWebTokenError') {
      next(new AppError(
        'Token inválido',
        401,
        'INVALID_TOKEN'
      ));
    } else if (error.name === 'TokenExpiredError') {
      next(new AppError(
        'Token expirado',
        401,
        'TOKEN_EXPIRED'
      ));
    } else {
      next(error);
    }
  }
};

/**
 * Middleware de autorização baseada em roles
 */
const authorize = (...roles) => {
  return (req, res, next) => {
    if (!req.user) {
      return next(new AppError(
        Errors.UNAUTHORIZED.message,
        Errors.UNAUTHORIZED.status,
        Errors.UNAUTHORIZED.code
      ));
    }
    
    if (!roles.includes(req.user.role)) {
      logger.warn('Authorization failed', {
        correlation_id: req.correlationId,
        user_id: req.user.id,
        required_roles: roles,
        user_role: req.user.role
      });
      
      return next(new AppError(
        Errors.FORBIDDEN.message,
        Errors.FORBIDDEN.status,
        Errors.FORBIDDEN.code
      ));
    }
    
    next();
  };
};

/**
 * Middleware de autorização baseada em permissões
 */
const hasPermission = (...requiredPermissions) => {
  return (req, res, next) => {
    if (!req.user) {
      return next(new AppError(
        Errors.UNAUTHORIZED.message,
        Errors.UNAUTHORIZED.status,
        Errors.UNAUTHORIZED.code
      ));
    }
    
    const userPermissions = req.user.permissions || [];
    const hasAllPermissions = requiredPermissions.every(permission => 
      userPermissions.includes(permission)
    );
    
    if (!hasAllPermissions) {
      logger.warn('Permission check failed', {
        correlation_id: req.correlationId,
        user_id: req.user.id,
        required_permissions: requiredPermissions,
        user_permissions: userPermissions
      });
      
      return next(new AppError(
        Errors.FORBIDDEN.message,
        Errors.FORBIDDEN.status,
        Errors.FORBIDDEN.code
      ));
    }
    
    next();
  };
};

/**
 * Revoga um token JWT
 */
const revokeToken = (token) => {
  tokenBlacklist.add(token);
  
  // Em produção, armazenar no Redis com TTL baseado na expiração do token
  setTimeout(() => {
    tokenBlacklist.delete(token);
  }, 24 * 60 * 60 * 1000); // Limpa após 24h
};

module.exports = {
  authenticate,
  authorize,
  hasPermission,
  revokeToken
```

CREATE VIEW agent_performance_dashboard AS
SELECT 
    a.agent_id,
    a.agent_name,
    a.agent_type,
    COUNT(DISTINCT ae.execution_id) as total_executions,
    SUM(CASE WHEN ae.status = 'completed' THEN 1 ELSE 0 END) as successful_executions,
    SUM(CASE WHEN ae.status = 'failed' THEN 1 ELSE 0 END) as failed_executions,
    AVG(ae.duration_ms) as avg_duration_ms,
    PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY ae.duration_ms) as p95_duration_ms,

    MAX(ae.start_time) as last_execution
FROM agents a

WHERE a.is_active = TRUE
GROUP BY a.agent_id, a.agent_name, a.agent_type;

    pipeline_type,
    DATE(start_time) as execution_date,
    COUNT(*) as total_runs,
    SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END) as successful_runs,
    SUM(CASE WHEN status = 'failed' THEN 1 ELSE 0 END) as failed_runs,
    AVG(processing_time_ms) as avg_processing_time,
    MIN(start_time) as first_run,
    MAX(start_time) as last_run,
    ROUND(100.0 * SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END) / COUNT(*), 2) as success_rate
FROM pipeline_metrics
GROUP BY pipeline_name, pipeline_type, DATE(start_time);

-- View para qualidade de dados
CREATE VIEW data_quality_overview AS
SELECT 
    dqt.dataset_name,
    dqt.test_name,
    dqt.test_type,
    dqt.severity,
    COUNT(dqr.result_id) as total_runs,
    SUM(CASE WHEN dqr.status = 'passed' THEN 1 ELSE 0 END) as passed_runs,
    SUM(CASE WHEN dqr.status = 'failed' THEN 1 ELSE 0 END) as failed_runs,
    AVG(dqr.failure_rate) as avg_failure_rate,
    MAX(dqr.execution_time) as last_run
FROM data_quality_tests dqt
LEFT JOIN data_quality_results dqr ON dqt.test_id = dqr.test_id
WHERE dqt.is_active = TRUE
GROUP BY dqt.dataset_name, dqt.test_name, dqt.test_type, dqt.severity;

-- Procedure para registrar execução de agente
CREATE OR REPLACE PROCEDURE log_agent_execution(
    p_agent_id UUID,
    p_squad_id UUID,
    p_trigger_type VARCHAR,
    p_trigger_id VARCHAR DEFAULT NULL,
    p_input_data JSONB DEFAULT NULL,
    p_output_data JSONB DEFAULT NULL,
    p_status VARCHAR,
    p_error_message TEXT DEFAULT NULL,
    p_resource_usage JSONB DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_execution_id UUID;
    v_start_time TIMESTAMPTZ := NOW();
    v_end_time TIMESTAMPTZ;
    v_duration_ms INTEGER;
BEGIN
    -- Inserir execução
    INSERT INTO agent_executions (
        agent_id, squad_id, trigger_type, trigger_id, 
        input_data, output_data, status, error_message,
        start_time, end_time, duration_ms, resource_usage
    ) VALUES (
        p_agent_id, p_squad_id, p_trigger_type, p_trigger_id,
        p_input_data, p_output_data, p_status, p_error_message,
        v_start_time, 
        CASE WHEN p_status IN ('completed', 'failed', 'timeout') THEN NOW() ELSE NULL END,
        CASE WHEN p_status IN ('completed', 'failed', 'timeout') THEN 
            EXTRACT(EPOCH FROM (NOW() - v_start_time)) * 1000 
        ELSE NULL END,
        p_resource_usage
    ) RETURNING execution_id INTO v_execution_id;
    
    -- Se falhou, logar erro
    IF p_status = 'failed' AND p_error_message IS NOT NULL THEN
        INSERT INTO execution_logs (execution_id, log_level, component, message)
        VALUES (v_execution_id, 'ERROR', 'agent_execution', p_error_message);
    END IF;
    
    COMMIT;
END;
$$;

-- Procedure para backup automatizado
CREATE OR REPLACE PROCEDURE perform_automated_backup(
    p_backup_type VARCHAR,
    p_storage_location TEXT DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_backup_id UUID;
    v_start_time TIMESTAMPTZ := NOW();
BEGIN
    -- Registrar início do backup
    INSERT INTO backup_metadata (
        backup_type, status, start_time, storage_location
    ) VALUES (
        p_backup_type, 'started', v_start_time, p_storage_location
    ) RETURNING backup_id INTO v_backup_id;
    
    -- Aqui iria a lógica real de backup usando pg_dump/pg_basebackup
    -- Esta é uma implementação de placeholder
    
    -- Simular duração do backup
    PERFORM pg_sleep(1);
    
    -- Atualizar status para completado
    UPDATE backup_metadata 
    SET 
        status = 'completed',
        end_time = NOW(),
        duration_seconds = EXTRACT(EPOCH FROM (NOW() - v_start_time))::INTEGER,
        size_bytes = 1024 * 1024 * 100, -- 100MB placeholder
        checksum = MD5(v_backup_id::TEXT || NOW()::TEXT)
    WHERE backup_id = v_backup_id;
    
    -- Criar ponto de recuperação
    INSERT INTO recovery_points (point_name, recovery_time, backup_id, lsn)
    VALUES (
        'auto_backup_' || TO_CHAR(NOW(), 'YYYYMMDD_HH24MISS'),
        NOW(),
        v_backup_id,
        '0/0' -- LSN placeholder
    );
    
    COMMIT;
END;
$$;

-- Tabela para changelog do Flink (Change Data Capture)
CREATE TABLE flink_changelog (
    change_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    table_name VARCHAR(255) NOT NULL,
    operation CHAR(1) NOT NULL CHECK (operation IN ('I', 'U', 'D')),
    record_id VARCHAR(255) NOT NULL,
    before_state JSONB,
    after_state JSONB,
    change_time TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    processed BOOLEAN NOT NULL DEFAULT FALSE,
    
    -- Índices
    INDEX idx_flink_changelog_table (table_name),
    INDEX idx_flink_changelog_time (change_time),
    INDEX idx_flink_changelog_processed (processed)
);

-- Função para capturar changes (usada em triggers)
CREATE OR REPLACE FUNCTION capture_table_changes()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO flink_changelog (table_name, operation, record_id, after_state)
        VALUES (TG_TABLE_NAME, 'I', NEW.agent_id::TEXT, row_to_json(NEW)::jsonb);
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO flink_changelog (table_name, operation, record_id, before_state, after_state)
        VALUES (TG_TABLE_NAME, 'U', NEW.agent_id::TEXT, row_to_json(OLD)::jsonb, row_to_json(NEW)::jsonb);
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO flink_changelog (table_name, operation, record_id, before_state)
        VALUES (TG_TABLE_NAME, 'D', OLD.agent_id::TEXT, row_to_json(OLD)::jsonb);
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;
