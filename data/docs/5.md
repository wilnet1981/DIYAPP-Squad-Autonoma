```javascript
const { z } = require('zod');

const inputSchema = z.object({
  customerId: z.string().min(1, "Customer ID required"),
  amount: z.number().positive("Amount must be positive"),
  tokens: z.number().int("Tokens must be an integer"),
});

const outputSchema = z.object({
  success: z.boolean(),
  message: z.string(),
  tokens: z.number(),
});

function validateInput(req, res, next) {
  try {
    req.body = inputSchema.parse(req.body);
    next();
  } catch (e) {
    res.status(400).json({
      success: false,
      message: e.errors.map(err => err.message).join(", "),
    });
  }
}

function validateOutput(output) {
  try {
    return outputSchema.parse(output);
  } catch (e) {
    throw new Error("Invalid output format");
  }
}

module.exports = { validateInput, validateOutput };
```

### (2) Estratégia de Armazenamento do Tema

#### Análise
- **Opções:** Armazenar tema em cookies ou no banco de dados.
- **Decisão:** Utilizar `localStorage` para persistência no lado do cliente por simplicidade e performance, já que a seleção de tema não precisa ser sincronizada com o servidor.

#### Exemplo de Implementação
```javascript

import cookieParser from 'cookie-parser';

const app = express();

app.use(cookieParser());

// Esquema de validação usando Zod
const billingSchema = z.object({
  userId: z.string(),
  amount: z.number(),
  currency: z.string().length(3),
});

const responseSchema = z.object({
  tokens: z.number(),
  status: z.string(),
});

// Middleware para validação de payload usando Zod
function validateSchema(schema) {
  return (req, res, next) => {
    try {
      schema.parse(req.body);
      next();
    } catch (e) {
      return res.status(400).json({ error: e.errors });
    }
  };
}

// Endpoint de faturamento
app.post('/api/billing', validateSchema(billingSchema), (req, res) => {
  const { userId, amount, currency } = req.body;

  // Simulação do processamento de faturamento
  const tokens = Math.round(amount * 10); // Conversão de montante para tokens

  // Assegurando que tokens é um número
  const responseData = {
    tokens: Number(tokens),
    status: 'success',
  };

  // Validações de saída
  const validationResult = responseSchema.safeParse(responseData);
  if (!validationResult.success) {
    return res.status(500).json({ error: 'Invalid response structure' });
  }

  // Persistência de preferência de tema
  const themePreference = req.header('X-Theme-Preference');
  if (themePreference) {
    // Persistir em cookie ou banco de dados conforme necessidade
    res.cookie('theme-preference', themePreference, { httpOnly: true });
    // Se usar DB, insira lógica aqui para salvar em user settings
  }

  // Retornando resposta
  res.json(responseData);
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
```

```javascript
// Função para definir cookie
function setCookie(name, value, days) {
  const date = new Date();
  date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
  const expires = "expires=" + date.toUTCString();
  document.cookie = name + "=" + value + ";" + expires + ";path=/";
}

// Função para recuperar cookie
function getCookie(name) {
  const nameEQ = name + "=";
  const ca = document.cookie.split(';');
  for (let i = 0; i < ca.length; i++) {
    let c = ca[i];
    while (c.charAt(0) === ' ') c = c.substring(1, c.length);
    if (c.indexOf(nameEQ) === 0) return c.substring(nameEQ.length, c.length);
  }
  return null;
}

// Alternar tema
function toggleTheme() {
  const currentTheme = document.body.classList.contains('dark-theme') ? 'light' : 'dark';
  document.body.classList.toggle('dark-theme', currentTheme === 'dark');
  setCookie('theme', currentTheme, 30);
}

// Aplicar tema ao carregar
window.onload = function() {
  const savedTheme = getCookie('theme');
  if (savedTheme) {

  }
}
```

### **Atualizações no `app.js` (para suporte a temas)**
Adicionei um script para:
1. Detectar preferência do usuário (`prefers-color-scheme`).
2. Permitir toggle manual de tema.
3. Persistir a escolha no `localStorage`.

```javascript
// app.js
document.addEventListener('DOMContentLoaded', () => {
  const themeToggle = document.querySelector('.theme-toggle');
  const prefersDarkScheme = window.matchMedia('(prefers-color-scheme: dark)');

  // Check for saved theme preference or use system preference
  const currentTheme = localStorage.getItem('theme') ||
    (prefersDarkScheme.matches ? 'dark' : 'light');

  // Apply the theme
  document.documentElement.setAttribute('data-theme', currentTheme);

  // Toggle theme on button click
  themeToggle.addEventListener('click', () => {
    const newTheme = document.documentElement.getAttribute('data-theme') === 'dark'
      ? 'light'
      : 'dark';
    document.documentElement.setAttribute('data-theme', newTheme);
    localStorage.setItem('theme', newTheme);
  });

  // Listen for system theme changes
  prefersDarkScheme.addEventListener('change', (e) => {
    if (!localStorage.getItem('theme')) {
      document.documentElement.setAttribute('data-theme', e.matches ? 'dark' : 'light');
    }
  });
});
```

```javascript
/**
 * Renderiza a tabela de tokens garantindo que valores sejam numéricos
 * @param {Array<{name: string, value: string|number}>} tokenData
 */
function renderTokenColumns(tokenData) {
  const container = document.getElementById('token-table-body');
  if (!container) return;

  // Limpa o container antes de renderizar
  container.innerHTML = '';

  // Garante que os valores sejam numéricos (trata strings numéricas)
  const processedData = tokenData.map(token => ({
    ...token,
    value: typeof token.value === 'string' ? parseFloat(token.value) : token.value,
  }));

  // Validação: ignora tokens com valores não numéricos
  const validTokens = processedData.filter(token => !isNaN(token.value));

  validTokens.forEach(token => {
    const row = document.createElement('tr');
    row.className = 'token-row';

    const nameCell = document.createElement('td');
    nameCell.textContent = token.name;
    nameCell.className = 'token-name';

    const valueCell = document.createElement('td');
    valueCell.textContent = token.value;
    valueCell.className = 'token-value';

    row.appendChild(nameCell);
    row.appendChild(valueCell);
    container.appendChild(row);
  });

  // Adiciona transição suave para linhas novas
  requestAnimationFrame(() => {
    document.querySelectorAll('.token-row').forEach(row => {
      row.style.opacity = '1';
    });
  });
}

// Suporte a temas escuros (toggle)
function setupThemeToggle() {
  const themeToggle = document.getElementById('theme-toggle');
  if (!themeToggle) return;

  themeToggle.addEventListener('click', () => {
    document.body.classList.toggle('dark-theme');
    localStorage.setItem('theme', document.body.classList.contains('dark-theme') ? 'dark' : 'light');
  });

  // Carrega tema salvo
  const savedTheme = localStorage.getItem('theme') || 'light';
  if (savedTheme === 'dark') {
    document.body.classList.add('dark-theme');
  }
}
```

   ```javascript
   function toggleTheme() {
     const isDark = document.body.classList.toggle('dark-theme');
     localStorage.setItem('theme', isDark ? 'dark' : 'light');
     document.getElementById('theme-icon').textContent = isDark ? '☀️' : '🌙';
   }
   ```

2. **Renderização de Tokens:**
   - Função `renderTokens()` agora força conversão de strings para array de objetos.
   ```javascript
   function renderTokens(tokens) {
     if (typeof tokens === 'string') {
       tokens = JSON.parse(tokens); // Converte string para array
     }
     // Renderiza colunas estruturadas
     const columns = tokens.map(token => `
       <div class="token-column">
         <span>${token.id}</span>
         <span>${token.name}</span>
         <span>${token.value}</span>
       </div>
     `).join('');
     document.getElementById('tokens-container').innerHTML = columns;
   }
   ```

### **7. Runbook: Incidentes Relacionados ao Dashboard**
**Para:** SREs durante incidentes.

#### **Sintoma: Dashboard Premium não carrega**
1. **Verificar:**
   - Status da API `/api/premium-widgets` (código 200?).
   - Erros no console do navegador (`F12 > Console`).
2. **Ações:**
   - Se API falhar: Reiniciar serviço de backend.
   - Se erro de CORS: Verificar headers da API.
3. **Rollback:**
   - Reverter para versão anterior do `app.js` (tag `v1.9.0`).

#### **Sintoma: Tema Escuro não aplica**
1. **Verificar:**
   - `localStorage.getItem('theme')` retorna `dark`?
   - Classe `dark-theme` está no `<body>`?
2. **Ações:**
   - Limpar cache do navegador.
   - Verificar se o CSS está carregando corretamente (`<link rel="stylesheet" href="style.css">`).
