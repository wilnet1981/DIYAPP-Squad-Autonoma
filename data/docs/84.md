const EventEmitter = require('events');
const { performance } = require('perf_hooks');

class CircuitBreaker extends EventEmitter {
  constructor(options = {}) {
    super();
    
    this.config = {
      failureThreshold: options.failureThreshold || 5,
      successThreshold: options.successThreshold || 3,
      timeout: options.timeout || 30000,
      resetTimeout: options.resetTimeout || 60000,
      halfOpenMaxAttempts: options.halfOpenMaxAttempts || 2,
      fallback: options.fallback || this.defaultFallback,
      name: options.name || 'default'

    this.state = 'CLOSED';

    this.successCount = 0;
    this.nextAttempt = Date.now();
    this.halfOpenAttempts = 0;
    
    // Sliding window para tracking de falhas
    this.failureWindow = [];
    this.successWindow = [];
    this.windowSize = 100; // Mantém últimos 100 eventos
    
    // Métricas
    this.metrics = {
      totalRequests: 0,
      successfulRequests: 0,
      failedRequests: 0,
      circuitOpens: 0,
      stateChanges: []
    };

    this.initialize();
  }

  initialize() {
    // Limpa janelas a cada minuto
    setInterval(() => this.cleanWindows(), 60000);
  }

  cleanWindows() {
    const now = Date.now();
    const windowDuration = 60000; // 1 minuto
    
    this.failureWindow = this.failureWindow.filter(
      timestamp => now - timestamp < windowDuration
    );
    this.successWindow = this.successWindow.filter(
      timestamp => now - timestamp < windowDuration
    );
  }

  async execute(fn, ...args) {
    this.metrics.totalRequests++;
    
    // Verifica se pode fazer a requisição
    if (!this.canAttempt()) {
      this.emit('circuitOpen', {
        name: this.config.name,
        state: this.state,
        nextAttempt: this.nextAttempt
      });
      
      return this.config.fallback(...args);
    }

    try {
      const startTime = performance.now();
      const result = await fn(...args);
      const duration = performance.now() - startTime;
      
      this.onSuccess(duration);
      return result;
    } catch (error) {
      this.onFailure(error);
      throw error;
    }
  }

  canAttempt() {
    const now = Date.now();
    
    switch (this.state) {
      case 'CLOSED':
        return true;
      
      case 'OPEN':
        if (now >= this.nextAttempt) {
          this.state = 'HALF_OPEN';
          this.halfOpenAttempts = 0;
          this.emit('stateChange', {
            from: 'OPEN',
            to: 'HALF_OPEN',
            timestamp: now
          });
          return true;
        }
        return false;
      
      case 'HALF_OPEN':
        if (this.halfOpenAttempts < this.config.halfOpenMaxAttempts) {
          this.halfOpenAttempts++;
          return true;
        }
        return false;
      
      default:
        return false;
    }
  }

  onSuccess(duration) {
    const now = Date.now();
    this.successCount++;
    this.successWindow.push(now);
    
    if (this.state === 'HALF_OPEN') {
      this.halfOpenSuccess();
    } else {
      this.failureCount = 0;
    }
    
    this.metrics.successfulRequests++;
    
    this.emit('success', {
      name: this.config.name,
      duration,
      successCount: this.successCount,
      timestamp: now
    });
  }

  onFailure(error) {
    const now = Date.now();
    this.failureCount++;
    this.failureWindow.push(now);
    
    this.metrics.failedRequests++;
    
    // Calcula taxa de falha na janela deslizante
    const windowFailures = this.failureWindow.length;
    const windowTotal = this.failureWindow.length + this.successWindow.length;
    const failureRate = windowTotal > 0 ? windowFailures / windowTotal : 0;
    
    this.emit('failure', {
      name: this.config.name,
      error: error.message,
      failureCount: this.failureCount,
      failureRate,
      timestamp: now
    });
    
    if (this.state === 'HALF_OPEN') {
      this.halfOpenFailure();
    } else if (this.state === 'CLOSED') {
      // Verifica se deve abrir o circuito
      if (this.failureCount >= this.config.failureThreshold || 
          failureRate > 0.5) { // 50% de falha na janela
        this.openCircuit();
      }
    }
  }

  halfOpenSuccess() {
    if (++this.successCount >= this.config.successThreshold) {
      this.closeCircuit();
    }
  }

  halfOpenFailure() {
    this.openCircuit();
  }

  openCircuit() {
    const previousState = this.state;
    this.state = 'OPEN';
    this.nextAttempt = Date.now() + this.config.timeout;
    this.metrics.circuitOpens++;
    
    this.emit('stateChange', {
      from: previousState,
      to: 'OPEN',
      timestamp: Date.now(),
      failureCount: this.failureCount
    });
    
    this.emit('circuitOpened', {
      name: this.config.name,
      nextAttempt: this.nextAttempt,
      failureCount: this.failureCount
    });
    
    // Agenda tentativa de reset
    setTimeout(() => {
      if (this.state === 'OPEN') {
        this.state = 'HALF_OPEN';
        this.halfOpenAttempts = 0;
        this.emit('stateChange', {
          from: 'OPEN',
          to: 'HALF_OPEN',
          timestamp: Date.now()
        });
      }
    }, this.config.resetTimeout);
  }

  closeCircuit() {
    const previousState = this.state;
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.successCount = 0;
    this.halfOpenAttempts = 0;
    
    this.emit('stateChange', {
      from: previousState,
      to: 'CLOSED',
      timestamp: Date.now()
    });
    
    this.emit('circuitClosed', {
      name: this.config.name,
      timestamp: Date.now()
    });
  }

  defaultFallback(...args) {
    return {
      status: 'circuit_open',
      message: 'Service temporarily unavailable',
      timestamp: Date.now(),
      args: args
    };
  }

  getState() {
    return {
      state: this.state,
      failureCount: this.failureCount,
      successCount: this.successCount,
      nextAttempt: this.nextAttempt,
      halfOpenAttempts: this.halfOpenAttempts,
      metrics: { ...this.metrics },
      failureWindowSize: this.failureWindow.length,
      successWindowSize: this.successWindow.length
    };
  }

  reset() {
    const previousState = this.state;
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.successCount = 0;
    this.halfOpenAttempts = 0;
    this.nextAttempt = Date.now();
    this.failureWindow = [];
    this.successWindow = [];
    
    this.emit('reset', {
      from: previousState,
      to: 'CLOSED',
      timestamp: Date.now()
    });
  }
}

module.exports = CircuitBreaker;

    name: str
    service_type: ServiceType
    status: HealthStatus
    latency_ms: Optional[float] = None
    last_check: datetime = field(default_factory=datetime.now)
    error_message: Optional[str] = None
    details: Dict[str, Any] = field(default_factory=dict)
    
    def to_dict(self) -> Dict[str, Any]:
        """Converte para dicionário para serialização"""
        return {
            "name": self.name,
            "service_type": self.service_type.value,
            "status": self.status.value,
            "latency_ms": self.latency_ms,
            "last_check": self.last_check.isoformat(),
            "error_message": self.error_message,
            "details": self.details
        }

class HealthMetrics:
    """Gerenciador de métricas do sistema"""
    
    def __init__(self):
        self.start_time = time.time()
        
        # Inicializa métricas Prometheus se disponível
        if PROMETHEUS_AVAILABLE:
            self._init_prometheus_metrics()
        else:
            self._init_fallback_metrics()
    
    def _init_prometheus_metrics(self):
        """Inicializa métricas Prometheus"""
        # Contadores de requisições HTTP
        self.http_requests_total = Counter(
            'http_requests_total',
            'Total HTTP requests',
            ['method', 'endpoint', 'status']
        )
        
        # Latência de requisições
        self.http_request_duration_seconds = Histogram(
            'http_request_duration_seconds',
            'HTTP request duration in seconds',
            ['method', 'endpoint'],
            buckets=(0.01, 0.05, 0.1, 0.3, 0.5, 1.0, 2.0, 5.0)
        )
        
        # Status dos serviços
        self.service_status = Gauge(
            'service_status',
            'Status of external services (0=unhealthy, 1=degraded, 2=healthy)',
            ['service_name', 'service_type']
        )
        
        # Latência dos serviços
        self.service_latency_ms = Histogram(
            'service_latency_ms',
            'Service latency in milliseconds',
            ['service_name', 'service_type'],
            buckets=(10, 50, 100, 200, 500, 1000, 2000, 5000)
        )
        
        # Circuit breaker status
        self.circuit_breaker_state = Gauge(
            'circuit_breaker_state',
            'Circuit breaker state (0=closed, 1=half-open, 2=open)',
            ['service_name']
        )
        
        # Uptime do sistema
        self.system_uptime = Gauge(
            'system_uptime_seconds',
            'System uptime in seconds'
        )
        
        # Memória e CPU (seriam atualizados por outro serviço)
        self.system_memory_usage = Gauge(
            'system_memory_usage_bytes',
            'System memory usage in bytes'
        )
        
        self.system_cpu_usage = Gauge(
            'system_cpu_usage_percent',
            'System CPU usage percentage'
        )
        
        logger.info("Métricas Prometheus inicializadas")
    
    def _init_fallback_metrics(self):
        """Inicializa métricas fallback quando Prometheus não está disponível"""
        self.http_requests_total = Counter()
        self.http_request_duration_seconds = Histogram()
        self.service_status = Gauge()
        self.service_latency_ms = Histogram()
        self.circuit_breaker_state = Gauge()
        self.system_uptime = Gauge()
        self.system_memory_usage = Gauge()
        self.system_cpu_usage = Gauge()
        
        self._fallback_metrics = {
            'http_requests': {},
            'service_status': {},
            'circuit_breakers': {}
        }
        
        logger.warning("Prometheus não disponível, usando métricas fallback")
    
    def record_http_request(self, method: str, endpoint: str, status: int, duration: float):
        """Registra uma requisição HTTP"""
        if PROMETHEUS_AVAILABLE:
            self.http_requests_total.labels(
                method=method,
                endpoint=endpoint,
                status=str(status)
            ).inc()
            
            self.http_request_duration_seconds.labels(
                method=method,
                endpoint=endpoint
            ).observe(duration)
        else:
            # Fallback: armazena em dicionário
            key = f"{method}:{endpoint}:{status}"
            self._fallback_metrics['http_requests'][key] = \
                self._fallback_metrics['http_requests'].get(key, 0) + 1
    
    def update_service_status(self, service_name: str, service_type: str, 
                            status: HealthStatus, latency_ms: Optional[float] = None):
        """Atualiza status de um serviço"""
        status_value = {
            HealthStatus.HEALTHY: 2,
            HealthStatus.DEGRADED: 1,
            HealthStatus.UNHEALTHY: 0,
            HealthStatus.UNKNOWN: -1
        }.get(status, -1)
        
        if PROMETHEUS_AVAILABLE:
            self.service_status.labels(
                service_name=service_name,
                service_type=service_type
            ).set(status_value)
            
            if latency_ms is not None:
                self.service_latency_ms.labels(
                    service_name=service_name,
                    service_type=service_type
                ).observe(latency_ms / 1000.0)  # Converte para segundos
        else:
            # Fallback
            self._fallback_metrics['service_status'][service_name] = {
                'status': status_value,
                'latency_ms': latency_ms,
                'timestamp': datetime.now().isoformat()
            }
    
    def update_circuit_breaker(self, service_name: str, state: str):
        """Atualiza estado do circuit breaker"""
        state_value = {
            'closed': 0,
            'half_open': 1,
            'open': 2
        }.get(state, -1)
        
        if PROMETHEUS_AVAILABLE:
            self.circuit_breaker_state.labels(
                service_name=service_name
            ).set(state_value)
        else:
            # Fallback
            self._fallback_metrics['circuit_breakers'][service_name] = {
                'state': state,
                'state_value': state_value,
                'timestamp': datetime.now().isoformat()
            }
    
    def update_system_metrics(self):
        """Atualiza métricas do sistema"""
        if PROMETHEUS_AVAILABLE:
            self.system_uptime.set(time.time() - self.start_time)
            # Nota: Métricas de memória e CPU seriam atualizadas por outro serviço
    
    def get_metrics(self) -> bytes:
        """Retorna métricas no formato Prometheus"""
        if PROMETHEUS_AVAILABLE:
            self.update_system_metrics()
            return generate_latest(REGISTRY)
        else:
            # Retorna métricas fallback como JSON
            self._fallback_metrics['system'] = {
                'uptime_seconds': time.time() - self.start_time,
                'timestamp': datetime.now().isoformat()
            }
            return json.dumps(self._fallback_metrics, indent=2).encode('utf-8')
    
    def get_uptime(self) -> float:
        """Retorna uptime do sistema em segundos"""
        return time.time() - self.start_time

class CircuitBreakerManager:
    """Gerenciador de circuit breakers para serviços externos"""
    
    def __init__(self, metrics: HealthMetrics):
        self.metrics = metrics
        self.breakers: Dict[str, Any] = {}
        
        # Configurações padrão
        self.default_config = {
            'failure_threshold': 5,      # Número de falhas para abrir
            'recovery_timeout': 60,      # Segundos para tentar recuperar
            'expected_exception': (Exception,),  # Exceções que contam como falhas
            'fallback_function': None    # Função fallback
        }
        
        # Configurações específicas por tipo de serviço
        self.service_configs = {
            ServiceType.DATABASE: {
                'failure_threshold': 3,
                'recovery_timeout': 30,
                'name': 'database'
            },
            ServiceType.EXTERNAL_API: {
                'failure_threshold': 5,
                'recovery_timeout': 60,
                'name': 'external_api'
            },
            ServiceType.LLM_SERVICE: {
                'failure_threshold': 3,
                'recovery_timeout': 120,  # LLMs podem ser mais lentos
                'name': 'llm_service'
            },
            ServiceType.CACHE: {
                'failure_threshold': 10,  # Cache pode falhar mais frequentemente
                'recovery_timeout': 10,
                'name': 'cache'
            }
        }
    
    def get_circuit_breaker(self, service_name: str, service_type: ServiceType):
        """Obtém ou cria um circuit breaker para um serviço"""
        if service_name in self.breakers:
            return self.breakers[service_name]
        
        # Configuração específica do tipo de serviço
        config = self.default_config.copy()
        if service_type in self.service_configs:
            config.update(self.service_configs[service_type])
        
        # Cria decorator de circuit breaker
        if CIRCUIT_BREAKER_AVAILABLE:
            breaker = circuit(
                failure_threshold=config['failure_threshold'],
                recovery_timeout=config['recovery_timeout'],
                expected_exception=config['expected_exception'],
                name=f"{config['name']}_{service_name}"
            )
        else:
            # Fallback: decorator que apenas executa a função
            def breaker(func):
                return func
        
        self.breakers[service_name] = breaker
        
        # Registra métrica inicial
        self.metrics.update_circuit_breaker(service_name, 'closed')
        
        return breaker
    
    def get_breaker_status(self, service_name: str) -> Dict[str, Any]:
        """Retorna status de um circuit breaker específico"""
        if not CIRCUIT_BREAKER_AVAILABLE:
            return {'available': False, 'state': 'unknown'}
        
        try:
            # Tenta obter informações do circuit breaker real
            from circuitbreaker import CircuitBreakerMonitor
            
            for breaker in CircuitBreakerMonitor.get_circuits():
                if breaker.name == service_name or service_name in breaker.name:
                    return {
                        'available': True,
                        'state': breaker.state.name.lower(),
                        'failure_count': breaker.failure_count,
                        'last_failure': breaker.last_failure_time.isoformat() if breaker.last_failure_time else None,
                        'opened_until': breaker.opened_until.isoformat() if breaker.opened_until else None
                    }
        except Exception as e:
            logger.error(f"Erro ao obter status do circuit breaker {service_name}: {e}")
        
        return {'available': False, 'state': 'unknown'}
    
    def get_all_breakers_status(self) -> Dict[str, Dict[str, Any]]:
        """Retorna status de todos os circuit breakers"""
        status = {}
        for service_name in self.breakers:
            status[service_name] = self.get_breaker_status(service_name)
        return status

class EventTracker {
  constructor(config = {}) {
    this.endpoint = config.endpoint || '/api/events';
    this.batchSize = config.batchSize || 10;
    this.flushInterval = config.flushInterval || 5000; // 5 segundos
    this.eventsQueue = [];
    this.isFlushing = false;
    this.sessionId = this.generateSessionId();
    this.userId = this.getUserId();
    
    // Inicia flush automático
    this.startAutoFlush();
    
    // Captura eventos de page unload
    window.addEventListener('beforeunload', () => this.flushImmediately());
  }
  
  generateSessionId() {
    return 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
  }
  
  getUserId() {
    // Implementação real buscaria do auth system
    return localStorage.getItem('diyapp_user_id') || 'anonymous_' + Math.random().toString(36).substr(2, 9);
  }
  
  trackPremiumComponentView(eventData) {
    const event = {
      event_type: 'premium_component_view',
      event_id: this.generateEventId(),
      user_id: this.userId,
      session_id: this.sessionId,
      event_timestamp: new Date().toISOString(),
      ...eventData,
      app_version: 'v12',
      user_agent: navigator.userAgent,
      screen_resolution: `${window.screen.width}x${window.screen.height}`
    };
    
    this.validateEvent(event, 'premium_component_view');
    this.queueEvent(event);
    
    return event.event_id;
  }
  
  trackPremiumComponentInteraction(eventData) {
    const event = {
      event_type: 'premium_component_interaction',
      event_id: this.generateEventId(),
      user_id: this.userId,
      session_id: this.sessionId,
      event_timestamp: new Date().toISOString(),
      ...eventData,
      app_version: 'v12'
    };
    
    this.validateEvent(event, 'premium_component_interaction');
    this.queueEvent(event);
    
    return event.event_id;
  }
  
  trackPremiumFeatureAdoption(eventData) {
    const event = {
      event_type: 'premium_feature_adoption',
      event_id: this.generateEventId(),
      user_id: this.userId,
      session_id: this.sessionId,
      event_timestamp: new Date().toISOString(),
      ...eventData,
      app_version: 'v12'
    };
    
    this.validateEvent(event, 'premium_feature_adoption');
    this.queueEvent(event);
    
    return event.event_id;
  }
  
  trackLLMCall(eventData) {
    const event = {
      event_type: 'llm_api_call',
      event_id: this.generateEventId(),
      timestamp: new Date().toISOString(),
      ...eventData,
      app_version: 'v12'
    };
    
    this.queueEvent(event);
    return event.event_id;
  }
  
  validateEvent(event, eventType) {
    const schemas = {
      premium_component_view: {
        required: ['component_type', 'component_id', 'page_url']
      },
      premium_component_interaction: {
        required: ['component_type', 'component_id', 'interaction_type', 'interaction_value']
      },
      premium_feature_adoption: {
        required: ['feature_name', 'adoption_type']
      }
    };
    
    const schema = schemas[eventType];
    if (!schema) return;
    
    for (const field of schema.required) {
      if (!event[field]) {
        console.warn(`Event validation failed: Missing required field ${field}`, event);
        throw new Error(`Missing required field: ${field}`);
      }
    }
  }
  
  generateEventId() {
    return 'evt_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
  }
  
  queueEvent(event) {
    this.eventsQueue.push(event);
    
    // Flush se atingir batch size
    if (this.eventsQueue.length >= this.batchSize) {
      this.flushEvents();
    }
  }
  
  startAutoFlush() {
    setInterval(() => {
      if (this.eventsQueue.length > 0 && !this.isFlushing) {
        this.flushEvents();
      }
    }, this.flushInterval);
  }
  
  async flushEvents() {
    if (this.isFlushing || this.eventsQueue.length === 0) return;
    
    this.isFlushing = true;
    const eventsToSend = [...this.eventsQueue];
    this.eventsQueue = [];
    
    try {
      const response = await fetch(this.endpoint, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          events: eventsToSend,
          batch_id: 'batch_' + Date.now(),
          sent_at: new Date().toISOString()
        })
      });
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }
      
      console.log(`Successfully sent ${eventsToSend.length} events`);
    } catch (error) {
      console.error('Failed to send events:', error);
      // Requeue events on failure
      this.eventsQueue = [...eventsToSend, ...this.eventsQueue];
    } finally {
      this.isFlushing = false;
    }
  }
  
  async flushImmediately() {
    if (this.eventsQueue.length > 0) {
      // Usa sendBeacon para reliability durante page unload
      const blob = new Blob([JSON.stringify({
        events: this.eventsQueue,
        batch_id: 'unload_batch_' + Date.now(),
        sent_at: new Date().toISOString()
      })], { type: 'application/json' });
      
      navigator.sendBeacon(this.endpoint, blob);
      this.eventsQueue = [];
    }
  }
  
  // Métodos utilitários para analytics
  getSessionDuration() {
    const startTime = parseInt(this.sessionId.split('_')[1]);
    return Date.now() - startTime;
  }
  
  endSession(exitReason = 'user_navigation') {
    const sessionEvent = {
      event_type: 'user_engagement_session',
      session_id: this.sessionId,
      user_id: this.userId,
      start_timestamp: new Date(parseInt(this.sessionId.split('_')[1])).toISOString(),
      end_timestamp: new Date().toISOString(),
      total_events: this.eventsQueue.length,
      session_duration_seconds: Math.floor(this.getSessionDuration() / 1000),
      exit_reason: exitReason
    };
    
    this.queueEvent(sessionEvent);
    return this.flushImmediately();
  }
}

# DIYAPP Evolution - V12 Core - Threat Modeling e Reforço de Segurança

## 1. Threat Modeling da Arquitetura V12 (Framework STRIDE)

### Assets Críticos da V12:
1. **API Gateway V12** - Ponto único de entrada
2. **Serviço de Autenticação** - JWT, OAuth, SSO
3. **Serviço de Dados Pessoais** - PII dos usuários
4. **Serviço de LLM/IA** - Modelos e prompts
5. **Serviço de Pagamentos** - Dados financeiros
6. **Banco de Dados Central** - PostgreSQL com PII
7. **Cache Distribuído** - Redis com sessões
8. **Fila de Mensagens** - RabbitMQ/Kafka

### Análise STRIDE por Componente:

#### 1. API Gateway V12
```
Spoofing: Token JWT falsificado
  Controle: Assinatura HMAC-SHA256, validação de issuer, audience
  Risco: Médio → Aceito com validação rigorosa

Tampering: Manipulação de requisições
  Controle: Rate limiting por IP/usuario, WAF integrado
  Risco: Baixo → Mitigado

Repudiation: Negação de ações administrativas
  Controle: Logs imutáveis em CloudWatch/Splunk com hash chain
  Risco: Baixo → Mitigado

Info Disclosure: Vazamento de headers sensíveis
  Controle: CORS restritivo, remoção de headers de debug
  Risco: Médio → Monitoramento contínuo

Denial of Service: Ataque DDoS
  Controle: CloudFront + WAF, rate limiting em múltiplas camadas
  Risco: Alto → Mitigado com CDN

Elevation: Acesso a endpoints administrativos
  Controle: RBAC granular, JWT com claims específicos
  Risco: Médio → Aceito com auditoria
```

#### 2. Serviço de LLM/IA
```
Spoofing: Prompt injection no system prompt
  Controle: Separação física de system/user prompts, validação de input
  Risco: Alto → Mitigado com sanitização em 3 camadas

Tampering: Manipulação de output do modelo
  Controle: Validação de schema JSON, sanitização HTML
  Risco: Médio → Mitigado

Repudiation: Negação de geração de conteúdo
  Controle: Logs de prompts/responses com hash, user session tracking
  Risco: Baixo → Mitigado

Info Disclosure: Vazamento de dados via contexto LLM
  Controle: Mascaramento de PII antes do contexto, filtro de output
  Risco: Alto → Zero-tolerance policy

Denial of Service: Exaustão de tokens/computação

  Risco: Médio → Mitigado

  Controle: Sandboxing, nunca executar output diretamente
  Risco: Crítico → Zero-tolerance policy

    if (!origin) {
      logger.warn('Request without origin header');
      return callback(null, true);
    }

    // Verificar se está na whitelist
    if (ALLOWED_ORIGINS.indexOf(origin) !== -1) {
      logger.debug(`CORS allowed for origin: ${origin}`);
      return callback(null, true);
    }

    // Bloquear origem não autorizada
    logger.warn(`CORS blocked for origin: ${origin}`);
    return callback(new Error('Not allowed by CORS'), false);
  },
  methods: ALLOWED_METHODS,
  allowedHeaders: ALLOWED_HEADERS,
  exposedHeaders: EXPOSED_HEADERS,
  credentials: true, // Permitir cookies/auth headers
  maxAge: 86400, // Cache de preflight por 24h
  preflightContinue: false,
  optionsSuccessStatus: 204
};

const corsValidation = (req, res, next) => {
  const origin = req.headers.origin;
  
  // Log detalhado para auditoria
  if (origin) {
    logger.info('CORS Request', {
      origin,
      method: req.method,
      path: req.path,
      userAgent: req.headers['user-agent'],
      timestamp: new Date().toISOString()
    });
  }

  // Validar métodos HTTP

    logger.warn(`Method not allowed: ${req.method}`);
    return res.status(405).json({
      error: 'Method Not Allowed',
      message: `HTTP method ${req.method} is not supported`
    });
  }

  next();
};

// Middleware combinado

  corsValidation

    if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
      this.osPreference = 'dark';
    } else {
      this.osPreference = 'light';

  calculateTimeProgress() {
    const now = new Date();
    const hour = now.getHours();
    const minute = now.getMinutes();
    const totalMinutes = hour * 60 + minute;
    
    // Progresso durante o dia (6:00-18:00 = 0, 18:00-6:00 = 1)
    if (hour >= 6 && hour < 18) {
      // Durante o dia, progresso baseado em quanto perto do pôr do sol
      const dayStart = 6 * 60;
      const dayEnd = 18 * 60;
      const dayProgress = (totalMinutes - dayStart) / (dayEnd - dayStart);
      return Math.sin(dayProgress * Math.PI) * 0.3; // Suave variação
    } else {
      // Durante a noite
      if (hour < 6) {
        const nightProgress = (totalMinutes + (24 * 60 - 18 * 60)) / (12 * 60);
        return 0.7 + 0.3 * Math.sin(nightProgress * Math.PI);

  interpolateColor(color1, color2, factor) {
    if (color1.startsWith('rgba')) {
      return this.interpolateRGBA(color1, color2, factor);
    }
    
    const c1 = this.hexToRGB(color1);
    const c2 = this.hexToRGB(color2);
    
    const r = Math.round(c1.r + (c2.r - c1.r) * factor);
    const g = Math.round(c1.g + (c2.g - c1.g) * factor);
    const b = Math.round(c1.b + (c2.b - c1.b) * factor);
    
    return `rgb(${r}, ${g}, ${b})`;
  }

  interpolateRGBA(rgba1, rgba2, factor) {
    const match1 = rgba1.match(/rgba?\((\d+),\s*(\d+),\s*(\d+)(?:,\s*([\d.]+))?\)/);
    const match2 = rgba2.match(/rgba?\((\d+),\s*(\d+),\s*(\d+)(?:,\s*([\d.]+))?\)/);
    
    if (!match1 || !match2) return rgba1;
    
    const r1 = parseInt(match1[1]), g1 = parseInt(match1[2]), b1 = parseInt(match1[3]), a1 = match1[4] ? parseFloat(match1[4]) : 1;
    const r2 = parseInt(match2[1]), g2 = parseInt(match2[2]), b2 = parseInt(match2[3]), a2 = match2[4] ? parseFloat(match2[4]) : 1;
    
    const r = Math.round(r1 + (r2 - r1) * factor);
    const g = Math.round(g1 + (g2 - g1) * factor);
    const b = Math.round(b1 + (b2 - b1) * factor);
    const a = a1 + (a2 - a1) * factor;
    
    return `rgba(${r}, ${g}, ${b}, ${a.toFixed(2)})`;
  }

  hexToRGB(hex) {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ? {
      r: parseInt(result[1], 16),
      g: parseInt(result[2], 16),
      b: parseInt(result[3], 16)
    } : { r: 0, g: 0, b: 0 };
  }

   * Configura listeners de eventos
   */
  setupEventListeners() {
    // Monitora mudanças na preferência do OS
    if (window.matchMedia) {
      const darkModeMediaQuery = window.matchMedia('(prefers-color-scheme: dark)');
      darkModeMediaQuery.addEventListener('change', (e) => {
        this.osPreference = e.matches ? 'dark' : 'light';
        if (this.theme === 'auto') {
          this.applyTheme();
        }
      });
    }

    // Monitora mudanças de hora
    setInterval(() => {
      const oldTimeTheme = this.timeBasedTheme;
      this.calculateTimeBasedTheme();
      
      if (oldTimeTheme !== this.timeBasedTheme && this.theme === 'auto') {
        this.applyTheme();
      }
    }, 60000); // Atualiza a cada minuto
  }

  /**
   * Inicia atualização automática
   */
  startAutoUpdate() {
    // Atualiza tema automático a cada 5 minutos para transições suaves
    setInterval(() => {
      if (this.theme === 'auto') {
        this.applyAutoTheme();
      }
    }, 300000);
  }

  /**
   * Muda o tema manualmente
   */
  setTheme(theme) {
    if (['light', 'dark', 'auto'].includes(theme)) {
      this.theme = theme;
      this.applyTheme();
      return true;
    }
    return false;
  }

  /**
   * Obtém tema atual
   */
  getCurrentTheme() {
    if (this.theme === 'auto') {
      return {
        mode: 'auto',
        calculated: this.calculateAutoTheme(),
        osPreference: this.osPreference,
        timeBased: this.timeBasedTheme
      };
    }
    return {
      mode: this.theme,
      calculated: this.theme
    };
  }

  /**
   * Dispara evento de mudança de tema
   */
  dispatchThemeChangeEvent() {
    const event = new CustomEvent('themechange', {
      detail: this.getCurrentTheme()
    });
    document.dispatchEvent(event);
  }

  /**
   * Obtém estatísticas de uso
   */
  getStats() {
    const now = new Date();
    return {
      currentTime: now.toLocaleTimeString(),
      theme: this.theme,
      osPreference: this.osPreference,
      timeBasedTheme: this.timeBasedTheme,
      autoCalculated: this.theme === 'auto' ? this.calculateAutoTheme() : null,
      localStorage: localStorage.getItem('diyapp-theme')
    };
  }
}

// Validação de resposta
export const validateResponse = (res, expectedStatus = 200, checkName = 'API Response') => {
  const success = check(res, {
    [`${checkName} - Status ${expectedStatus}`]: (r) => r.status === expectedStatus,
    [`${checkName} - Response Time < 2s`]: (r) => r.timings.duration < 2000,
    [`${checkName} - Has Body`]: (r) => r.body && r.body.length > 0
  });
  
  if (!success) {
    console.error(`Validation failed for ${checkName}:`, {
      status: res.status,
      duration: res.timings.duration,
      body: res.body ? res.body.substring(0, 200) : 'No body'
    });
  }
  
  return success;
};

// Gera payload aleatório para testes

  const payloads = testPayloads[type] || testPayloads.default;
  return payloads[Math.floor(Math.random() * payloads.length)];
};

  sleep(Math.random() * (max - min) + min);

A arquitetura V12 implementa um sistema de auto-cura distribuído que detecta, diagnostica e corrige falhas automaticamente sem intervenção humana. O sistema opera em modo "Hive" com múltiplos nós independentes.

- Notificação de eventos de degradação

**Propósito**: Melhoria contínua baseada em incidentes
**Responsabilidades**:
- Análise de postmortems

- Sugestão de novas estratégias

// src/design-system/SelfHealingComponent.jsx
import React, { useState, useEffect } from 'react';
import PropTypes from 'prop-types';

class SelfHealingComponent extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      hasError: false,
      healingAttempts: 0,
      lastError: null
    };
    this.maxHealingAttempts = 3;
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, lastError: error };
  }

  componentDidCatch(error, errorInfo) {
    console.error('Component error:', error, errorInfo);
    this.attemptHealing();
  }

  async attemptHealing() {
    if (this.state.healingAttempts >= this.maxHealingAttempts) {
      console.warn('Max healing attempts reached, rendering fallback');
      return;
    }

    this.setState(prev => ({ 
      healingAttempts: prev.healingAttempts + 1 
    }));

    // Estratégias de cura
    const strategies = [
      this.reinitializeState,
      this.clearCaches,
      this.reloadDependencies
    ];

    for (const strategy of strategies) {
      try {
        await strategy.call(this);
        this.setState({ hasError: false });
        console.log('Component healed successfully');
        return;
      } catch (strategyError) {
        console.warn('Healing strategy failed:', strategyError);
      }
    }
  }

  reinitializeState() {
    return new Promise((resolve) => {
      this.setState({
        data: null,
        loading: true
      }, resolve);
    });
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="component-healing" data-healing-attempt={this.state.healingAttempts}>
          <div className="healing-indicator">
            <span className="healing-pulse"></span>
            Auto-repairing component...
          </div>
          {this.props.fallback || <DefaultFallback />}
        </div>
      );
    }

    return this.props.children;
  }
}

SelfHealingComponent.propTypes = {
  children: PropTypes.node.isRequired,
  fallback: PropTypes.node,
  componentId: PropTypes.string.isRequired
};

export default SelfHealingComponent;

// src/design-system/hooks/useSelfHealing.js
import { useState, useEffect, useCallback } from 'react';

export function useSelfHealing(initialState, options = {}) {
  const {
    maxAttempts = 3,
    healingStrategies = [],
    onHeal = () => {}
  } = options;

  const [state, setState] = useState(initialState);
  const [isHealing, setIsHealing] = useState(false);
  const [healingHistory, setHealingHistory] = useState([]);

  const heal = useCallback(async (error) => {
    if (healingHistory.length >= maxAttempts) {
      throw new Error('Max healing attempts reached');
    }

    setIsHealing(true);
    const attempt = {
      timestamp: new Date().toISOString(),
      error: error?.message,
      strategiesAttempted: []
    };

    for (const strategy of healingStrategies) {
      try {
        attempt.strategiesAttempted.push(strategy.name);
        const result = await strategy(state, error);
        
        setState(result);
        setHealingHistory(prev => [...prev, { ...attempt, success: true }]);
        onHeal({ strategy: strategy.name, success: true });
        
        return result;
      } catch (strategyError) {
        console.warn(`Strategy ${strategy.name} failed:`, strategyError);
      }
    }

    setHealingHistory(prev => [...prev, { ...attempt, success: false }]);
    throw new Error('All healing strategies failed');
  }, [state, healingStrategies, maxAttempts, healingHistory, onHeal]);

  return {
    state,
    setState,
    isHealing,
    healingHistory,
    heal,
    canHeal: healingHistory.length < maxAttempts
  };
}
