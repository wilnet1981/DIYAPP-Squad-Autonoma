    this._loading = this.hasAttribute('loading');
    this._disabled = this.hasAttribute('disabled');
  }

  static get observedAttributes() {
    return ['variant', 'size', 'loading', 'disabled', 'state'];
  }

  connectedCallback() {
    this.render();
    this.setupEventListeners();
  }

  attributeChangedCallback(name, oldValue, newValue) {
    if (oldValue !== newValue) {
      if (name === 'loading' || name === 'disabled') {
        this[`_${name}`] = this.hasAttribute(name);
      } else {
        this[`_${name}`] = newValue;
      }
      this.render();
    }
  }

  get variant() { return this._variant; }
  set variant(value) {
    this.setAttribute('variant', value);
  }

  get size() { return this._size; }
  set size(value) {
    this.setAttribute('size', value);
  }

  get loading() { return this._loading; }
  set loading(value) {
    if (value) {
      this.setAttribute('loading', '');
    } else {
      this.removeAttribute('loading');
    }
  }

  get disabled() { return this._disabled; }
  set disabled(value) {
    if (value) {
      this.setAttribute('disabled', '');
    } else {
      this.removeAttribute('disabled');
    }
  }

  setupEventListeners() {
    this.shadowRoot.addEventListener('click', (e) => {
      if (this._disabled || this._loading) {
        e.preventDefault();
        e.stopPropagation();
      }
    });

    // Estados de hover e focus
    const button = this.shadowRoot.querySelector('button');
    if (button) {
      button.addEventListener('mouseenter', () => {
        if (!this._disabled && !this._loading) {
          this._state = 'hover';
          this.updateState();
        }
      });

      button.addEventListener('mouseleave', () => {
        if (!this._disabled && !this._loading) {
          this._state = 'default';
          this.updateState();
        }
      });

      button.addEventListener('focus', () => {
        if (!this._disabled && !this._loading) {
          this._state = 'focus';
          this.updateState();
        }
      });

      button.addEventListener('blur', () => {
        if (!this._disabled && !this._loading) {
          this._state = 'default';
          this.updateState();
        }
      });

      button.addEventListener('mousedown', () => {
        if (!this._disabled && !this._loading) {
          this._state = 'active';
          this.updateState();
        }
      });

        if (!this._disabled && !this._loading) {
          this._state = 'hover';
          this.updateState();
        }
      });

 * 1. React/Next.js (status quo)
 *    - Prós: Conhecimento existente, ecossistema maduro
 *    - Contras: Bundle size grande, complexidade crescente, hydration overhead
 * 
 * 2. SvelteKit
 *    - Prós: Zero-runtime overhead, bundle 40-60% menor, DX excelente
 *    - Contras: Ecossistema menor, curva de aprendizado para time React
 * 
 * 3. SolidStart
 *    - Prós: Performance excelente, reatividade granular
 *    - Contras: Muito novo, documentação imatura
 *

 * - Alinhamento com meta de autonomia: menos código = menos bugs

// Arquivo: /v12/src/lib/cache/manager.js
export const CACHE_MANAGER = `
import { browser } from '$app/environment';
import { LRUCache } from 'lru-cache';
import { encrypt, decrypt } from './crypto';

export class CacheManager {
  constructor() {
    // Memory cache (LRU)
    this.memoryCache = new LRUCache({
      max: 100,
      ttl: 5 * 60 * 1000
    });
    
    // IndexedDB connection
    this.db = null;
    this.initPersistentCache();
  }
  
  async initPersistentCache() {
    if (!browser) return;
    
    this.db = await idb.openDB('diyapp-v12-cache', 1, {
      upgrade(db) {
        // Store para templates
        db.createObjectStore('templates', { keyPath: 'id' });
        
        // Store para componentes
        db.createObjectStore('components', { keyPath: 'id' });
        
        // Store para user data
        const userStore = db.createObjectStore('user-data', { keyPath: 'userId' });
        userStore.createIndex('by-timestamp', 'timestamp');
      }
    });
  }
  
  async get(key, options = {}) {
    const { useMemory = true, useSession = true, usePersistent = true } = options;
    
    // 1. Tentar memory cache
    if (useMemory) {
      const memoryHit = this.memoryCache.get(key);
      if (memoryHit) {
        console.log(\`[Cache] Memory hit: \${key}\`);
        return memoryHit;
      }
    }
    
    // 2. Tentar session storage
    if (useSession && browser) {
      try {
        const encrypted = sessionStorage.getItem(\`cache_\${key}\`);
        if (encrypted) {
          const sessionHit = JSON.parse(decrypt(encrypted));
          if (sessionHit && Date.now() < sessionHit.expires) {
            console.log(\`[Cache] Session hit: \${key}\`);
            // Popula memory cache
            this.memoryCache.set(key, sessionHit.data);
            return sessionHit.data;
          }
        }
      } catch (e) {
        console.warn(\`[Cache] Session storage error: \${e.message}\`);
      }
    }
    
    // 3. Tentar IndexedDB
    if (usePersistent && this.db) {
      try {
        // Determinar store baseado no prefixo da key
        const [store] = key.split(':');
        const tx = this.db.transaction(store, 'readonly');
        const persistentHit = await tx.store.get(key);
        
        if (persistentHit && Date.now() < persistentHit.expires) {
          console.log(\`[Cache] Persistent hit: \${key}\`);
          // Popula caches superiores
          this.memoryCache.set(key, persistentHit.data);
          this.setSessionCache(key, persistentHit.data);
          return persistentHit.data;
        }
      } catch (e) {
        console.warn(\`[Cache] Persistent storage error: \${e.message}\`);
      }
    }
    
    // Cache miss
    return null;
  }
  
  async set(key, data, options = {}) {
    const {
      ttl = 30 * 60 * 1000, // 30 minutos default
      store = 'templates'
    } = options;
    
    const expires = Date.now() + ttl;
    const cacheItem = { data, expires };
    
    // 1. Memory cache
    this.memoryCache.set(key, data);
    
    // 2. Session storage
    this.setSessionCache(key, data, expires);
    
    // 3. Persistent storage
    if (this.db) {
      try {
        const tx = this.db.transaction(store, 'readwrite');
        await tx.store.put({
          id: key,
          data,
          expires,
          timestamp: Date.now()
        });
        await tx.done;
      } catch (e) {
        console.error(\`[Cache] Failed to persist: \${e.message}\`);
      }
    }
    
    // 4. Invalidate related caches
    this.invalidateRelated(key);
  }
  
  setSessionCache(key, data, expires) {
    if (!browser) return;
    
    try {
      const encrypted = encrypt(JSON.stringify({
        data,
        expires: expires || (Date.now() + 30 * 60 * 1000)
      }));
      sessionStorage.setItem(\`cache_\${key}\`, encrypted);
    } catch (e) {
      // Session storage cheio ou bloqueado
      console.warn(\`[Cache] Cannot set session cache: \${e.message}\`);
    }
  }
  
  invalidateRelated(key) {
    // Implementação baseada em tags
    const tags = this.extractTags(key);
    tags.forEach(tag => {
      this.memoryCache.delete(tag);
      if (browser) {
        sessionStorage.removeItem(\`cache_\${tag}\`);
      }
    });
  }
  
  extractTags(key) {
    // Exemplo: 'template:123:preview' → ['template', 'template:123']
    const parts = key.split(':');
    return parts.reduce((acc, part, index) => {
      if (index > 0) {
        acc.push(parts.slice(0, index + 1).join(':'));
      }
      return acc;
    }, [parts[0]]);
  }
  
  async clear(pattern) {
    // Limpar caches matching pattern
    if (pattern === '*') {
      this.memoryCache.clear();
      if (browser) {
        Object.keys(sessionStorage)
          .filter(k => k.startsWith('cache_'))
          .forEach(k => sessionStorage.removeItem(k));
      }
      if (this.db) {
        const stores = await this.db.objectStoreNames;
        for (const store of stores) {
          await this.db.clear(store);
        }
      }
    }
  }
}

import React from 'react'
import ReactDOM from 'react-dom/client'
import { BrowserRouter } from 'react-router-dom'
import App from './App'
import './styles/globals.css'

// Medição de performance
const reportWebVitals = (metric: any) => {
  console.log('Web Vitals:', metric)
  
  // Envia para analytics em produção
  if (import.meta.env.PROD) {
    const body = JSON.stringify(metric)
    const url = '/api/web-vitals'
    
    // Usa sendBeacon se disponível, senão fetch
    if (navigator.sendBeacon) {
      navigator.sendBeacon(url, body)
    } else {
      fetch(url, { body, method: 'POST', keepalive: true })
    }
  }
}

// Observador para Core Web Vitals
const observer = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    reportWebVitals({
      name: entry.name,
      value: entry.startTime,
      rating: entry.startTime < 2500 ? 'good' : entry.startTime < 4000 ? 'needs-improvement' : 'poor'
    })
  }
})

observer.observe({ entryTypes: ['largest-contentful-paint'] })

ReactDOM.createRoot(document.getElementById('root')!).render(
  <React.StrictMode>
    <BrowserRouter>
      <App />
    </BrowserRouter>
  </React.StrictMode>
)
