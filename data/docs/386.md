class ArchitectureDecisionRecord:
    """ADR Template para decisões arquiteturais significativas"""
    
    def __init__(self, number: int, title: str, status: str = "Proposta"):
        self.number = number
        self.title = title
        self.status = status
        self.date = datetime.datetime.now().strftime("%Y-%m-%d")
        self.authors = ["Tech Lead"]
        self.context = ""
        self.decision = ""
        self.options = []
        self.consequences = {"positive": [], "negative": [], "risks": []}
        self.review_date = ""
    
    def to_markdown(self) -> str:
        return f"""# ADR-{self.number:03d}: {self.title}

**Data**: {self.date}
**Status**: {self.status}
**Autores**: {', '.join(self.authors)}

## CONTEXTO
{self.context}

## DECISÃO
{self.decision}

## OPÇÕES CONSIDERADAS
{self._format_options()}

## CONSEQUÊNCIAS
**Positivas:**
{self._format_list(self.consequences['positive'])}

**Negativas:**
{self._format_list(self.consequences['negative'])}

**Riscos:**
{self._format_list(self.consequences['risks'])}

## REVISÃO
{self.review_date if self.review_date else "Reavaliar em 6 meses"}
"""
    
    def _format_options(self) -> str:
        formatted = []
        for i, option in enumerate(self.options):
            formatted.append(f"- **Opção {chr(65+i)}**: {option['description']}")
            formatted.append(f"  - Prós: {', '.join(option['pros'])}")
            formatted.append(f"  - Contras: {', '.join(option['cons'])}")
        return "\n".join(formatted)
    
    def _format_list(self, items: List[str]) -> str:
        return "\n".join([f"- {item}" for item in items])

class CommunicationProtocol(Enum):
    GRPC = "gRPC"
    KAFKA = "Kafka"
    REST = "REST"
    GRAPHQL = "GraphQL"

@dataclass
class ServiceContract:
    """Contrato de serviço para comunicação entre microsserviços"""
    service_name: str
    version: str
    protocol: CommunicationProtocol
    endpoints: List[Dict[str, Any]]
    events_published: List[str]
    events_subscribed: List[str]
    dependencies: List[str]
    
    def to_proto(self) -> str:
        """Gera arquivo .proto para gRPC"""
        if self.protocol != CommunicationProtocol.GRPC:
            return ""
        
        proto_content = f"""syntax = "proto3";

package {self.service_name.lower()}.v{self.version.replace('.', '_')};

option go_package = "github.com/diyapp/{self.service_name}/proto";
option java_multiple_files = true;
option java_package = "com.diyapp.{self.service_name.lower()}.grpc";

service {self.service_name}Service {{
"""
        
        for endpoint in self.endpoints:
            method_name = endpoint['name'].replace('-', '_')
            request_type = f"{method_name.capitalize()}Request"
            response_type = f"{method_name.capitalize()}Response"
            
            proto_content += f"  rpc {method_name}({request_type}) returns ({response_type});\n"
        
        proto_content += "}\n\n"
        
        # Generate message definitions
        for endpoint in self.endpoints:
            method_name = endpoint['name'].replace('-', '_')
            request_type = f"{method_name.capitalize()}Request"
            response_type = f"{method_name.capitalize()}Response"
            
            proto_content += f"message {request_type} {{\n"
            for field in endpoint.get('request_fields', []):
                proto_content += f"  {field['type']} {field['name']} = {field['position']};\n"
            proto_content += "}\n\n"
            
            proto_content += f"message {response_type} {{\n"
            for field in endpoint.get('response_fields', []):
                proto_content += f"  {field['type']} {field['name']} = {field['position']};\n"
            proto_content += "}\n\n"
        
        return proto_content
    
    def to_kafka_schema(self) -> Dict:
        """Gera schema Avro para eventos Kafka"""
        return {
            "type": "record",
            "name": f"{self.service_name}Events",
            "namespace": f"com.diyapp.{self.service_name.lower()}",
            "fields": [
                {
                    "name": "event_id",
                    "type": "string"
                },
                {
                    "name": "event_type",
                    "type": {
                        "type": "enum",
                        "name": "EventType",
                        "symbols": self.events_published
                    }
                },
                {
                    "name": "timestamp",
                    "type": "long",
                    "logicalType": "timestamp-millis"
                },
                {
                    "name": "payload",
                    "type": "bytes"
                }
            ]
        }

// src/store/app-store.ts
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

export interface User {
  id: string
  name: string
  email: string
  avatar?: string
}

export interface Task {
  id: string
  title: string
  description: string
  status: 'pending' | 'in-progress' | 'completed'
  priority: 'low' | 'medium' | 'high'
  createdAt: Date
  updatedAt: Date
  assignedTo?: string
}

export interface AppState {
  // User state
  user: User | null
  isAuthenticated: boolean
  setUser: (user: User | null) => void
  
  // Tasks state
  tasks: Task[]
  isLoadingTasks: boolean
  error: string | null
  
  // Actions
  addTask: (task: Omit<Task, 'id' | 'createdAt' | 'updatedAt'>) => void
  updateTask: (id: string, updates: Partial<Task>) => void
  deleteTask: (id: string) => void
  fetchTasks: () => Promise<void>
  
  // UI state
  sidebarOpen: boolean
  toggleSidebar: () => void
  
  // Theme
  theme: 'light' | 'dark'
  toggleTheme: () => void
}

export const createAppStore = () =>
  create<AppState>()(
    persist(
      (set, get) => ({
        // Initial state
        user: null,
        isAuthenticated: false,
        tasks: [],
        isLoadingTasks: false,
        error: null,
        sidebarOpen: false,
        theme: 'light',
        
        // Actions
        setUser: (user) => set({ 
          user, 
          isAuthenticated: !!user 
        }),
        
        addTask: (taskData) => {
          const newTask: Task = {
            ...taskData,
            id: Math.random().toString(36).substr(2, 9),
            createdAt: new Date(),
            updatedAt: new Date(),
          }
          set((state) => ({
            tasks: [newTask, ...state.tasks]
          }))
        },
        
        updateTask: (id, updates) => {
          set((state) => ({
            tasks: state.tasks.map((task) =>
              task.id === id
                ? { ...task, ...updates, updatedAt: new Date() }
                : task
            )
          }))
        },
        
        deleteTask: (id) => {
          set((state) => ({
            tasks: state.tasks.filter((task) => task.id !== id)
          }))
        },
        
        fetchTasks: async () => {
          set({ isLoadingTasks: true, error: null })
          
          try {
            // Simulação de API call com delay
            await new Promise(resolve => setTimeout(resolve, 1000))
            
            const mockTasks: Task[] = Array.from({ length: 50 }, (_, i) => ({
              id: `task-${i}`,
              title: `Tarefa ${i + 1}`,
              description: `Descrição da tarefa ${i + 1}`,
              status: i % 3 === 0 ? 'pending' : i % 3 === 1 ? 'in-progress' : 'completed',
              priority: i % 3 === 0 ? 'high' : i % 3 === 1 ? 'medium' : 'low',
              createdAt: new Date(Date.now() - i * 3600000),
              updatedAt: new Date(Date.now() - i * 1800000),
            }))
            
            set({ tasks: mockTasks, isLoadingTasks: false })
          } catch (err) {
            set({ 
              error: 'Falha ao carregar tarefas', 
              isLoadingTasks: false 
            })
          }
        },
        
        toggleSidebar: () => {
          set((state) => ({ sidebarOpen: !state.sidebarOpen }))
        },
        
        toggleTheme: () => {
          set((state) => ({ 
            theme: state.theme === 'light' ? 'dark' : 'light' 
          }))
        },
      }),
      {
        name: 'app-storage',
        partialize: (state) => ({
          user: state.user,
          theme: state.theme,
        }),
      }
    )
  )

export type AppStore = ReturnType<typeof createAppStore>

import { ReactNode, useEffect } from 'react'
import { useAppStore } from './store-provider'

interface ThemeProviderProps {
  children: ReactNode
}

export function ThemeProvider({ children }: ThemeProviderProps) {
  const theme = useAppStore((state) => state.theme)
  
  useEffect(() => {
    const root = document.documentElement
    
    if (theme === 'dark') {
      root.classList.add('dark')
      root.style.setProperty('--foreground-rgb', '255, 255, 255')
      root.style.setProperty('--background-rgb', '0, 0, 0')
    } else {
      root.classList.remove('dark')
      root.style.setProperty('--foreground-rgb', '0, 0, 0')
      root.style.setProperty('--background-rgb', '255, 255, 255')
    }
  }, [theme])
  
  return <>{children}</>
}

```typescript
import { Injectable, Inject, Logger } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';
import { Redis } from 'ioredis';

@Injectable()
export class CacheService {
  private readonly logger = new Logger(CacheService.name);
  private readonly redisClient: Redis;

  constructor(@Inject(CACHE_MANAGER) private cacheManager: Cache) {
    // @ts-ignore - cache-manager-redis-store doesn't expose client properly
    this.redisClient = this.cacheManager.store.getClient();
    
    this.setupErrorHandling();
  }

  private setupErrorHandling(): void {
    this.redisClient.on('error', (error) => {
      this.logger.error(`Redis error: ${error.message}`, error.stack);
    });

    this.redisClient.on('connect', () => {
      this.logger.log('Redis connected successfully');
    });

    this.redisClient.on('reconnecting', () => {
      this.logger.warn('Redis reconnecting...');
    });
  }

  async get<T>(key: string): Promise<T | null> {
    try {
      const value = await this.cacheManager.get<T>(key);
      return value || null;
    } catch (error) {
      this.logger.error(`Failed to get key ${key}: ${error.message}`);
      return null;
    }
  }

  async set(key: string, value: any, ttl?: number): Promise<void> {
    try {
      await this.cacheManager.set(key, value, ttl);
    } catch (error) {
      this.logger.error(`Failed to set key ${key}: ${error.message}`);
      throw error;
    }
  }

  async del(key: string): Promise<void> {
    try {
      await this.cacheManager.del(key);
    } catch (error) {
      this.logger.error(`Failed to delete key ${key}: ${error.message}`);
      throw error;
    }
  }

  async reset(): Promise<void> {
    try {
      await this.cacheManager.reset();
    } catch (error) {
      this.logger.error(`Failed to reset cache: ${error.message}`);
      throw error;
    }
  }

  async keys(pattern: string): Promise<string[]> {
    try {
      return await this.redisClient.keys(pattern);
    } catch (error) {
      this.logger.error(`Failed to get keys with pattern ${pattern}: ${error.message}`);
      return [];
    }
  }

  async exists(key: string): Promise<boolean> {
    try {
      const result = await this.redisClient.exists(key);
      return result === 1;
    } catch (error) {
      this.logger.error(`Failed to check existence of key ${key}: ${error.message}`);
      return false;
    }
  }

  async incr(key: string): Promise<number> {
    try {
      return await this.redisClient.incr(key);
    } catch (error) {
      this.logger.error(`Failed to increment key ${key}: ${error.message}`);
      throw error;
    }
  }

  async decr(key: string): Promise<number> {
    try {
      return await this.redisClient.decr(key);
    } catch (error) {
      this.logger.error(`Failed to decrement key ${key}: ${error.message}`);
      throw error;
    }
  }

  async expire(key: string, seconds: number): Promise<boolean> {
    try {
      const result = await this.redisClient.expire(key, seconds);
      return result === 1;
    } catch (error) {
      this.logger.error(`Failed to set expire on key ${key}: ${error.message}`);
      return false;
    }
  }

  async ttl(key: string): Promise<number> {
    try {
      return await this.redisClient.ttl(key);
    } catch (error) {
      this.logger.error(`Failed to get TTL for key ${key}: ${error.message}`);
      return -2; // Key doesn't exist
    }
  }

"""
Configuração inicial do Data Warehouse no BigQuery para DIYAPP V13
Cria datasets, tabelas e configurações de segurança

import json
from google.cloud import bigquery
from google.cloud.exceptions import NotFound
import os
from datetime import datetime, timedelta
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class BigQuerySetup:
    def __init__(self, project_id, credentials_path):
        """Inicializa cliente BigQuery"""
        os.environ['GOOGLE_APPLICATION_CREDENTIALS'] = credentials_path
        self.client = bigquery.Client(project=project_id)
        self.project_id = project_id
        
    def create_dataset(self, dataset_id, location='US', description=None):
        """Cria dataset no BigQuery"""
        dataset_ref = self.client.dataset(dataset_id)
        
        try:
            self.client.get_dataset(dataset_ref)
            logger.info(f"Dataset {dataset_id} já existe")
            return dataset_ref
        except NotFound:
            dataset = bigquery.Dataset(dataset_ref)
            dataset.location = location
            dataset.description = description
            dataset.default_table_expiration_ms = 90 * 24 * 60 * 60 * 1000  # 90 dias
            
            dataset = self.client.create_dataset(dataset)
            logger.info(f"Dataset {dataset_id} criado com sucesso")
            return dataset
    
    def create_raw_tables(self, dataset_id):
        """Cria tabelas da camada RAW (bronze)"""
        dataset_ref = self.client.dataset(dataset_id)
        
        # Tabela de eventos da aplicação
        raw_events_schema = [
            bigquery.SchemaField("event_id", "STRING", mode="REQUIRED"),

# DIYAPP Evolution - V13 Core - Segurança

## 1. Threat Modeling STRIDE - V13

### Assets Críticos:
1. **Dados de usuário**: PII, credenciais, tokens de sessão
2. **Sistema de autenticação**: JWT, refresh tokens, endpoints de login
3. **API endpoints**: CRUD de dados sensíveis
4. **Integração com WhatsApp**: Webhooks e callbacks
5. **Banco de dados**: SQLite/PostgreSQL com dados sensíveis
6. **Sistema de logs**: Informações de auditoria

### Análise STRIDE:

#### **Spoofing (Falsificação)**
- **Ameaça**: Ataque brute force em endpoints de login
- **Controles**: Rate limiting, MFA, validação de complexidade de senha
- **Risco**: Alto - endpoints de auth expostos

#### **Tampering (Manipulação)**
- **Ameaça**: Manipulação de JWT tokens, SQL injection
- **Controles**: Assinatura HMAC de tokens, prepared statements
- **Risco**: Médio-Alto

#### **Repudiation (Repúdio)**
- **Ameaça**: Usuário nega ações realizadas
- **Controles**: Logs de auditoria imutáveis com user_id e timestamp
- **Risco**: Médio

#### **Information Disclosure (Divulgação)**
- **Ameaça**: Vazamento de tokens, dados sensíveis em logs
- **Controles**: Criptografia em repouso, mascaramento de logs
- **Risco**: Alto

#### **Denial of Service (Negação)**
- **Ameaça**: Ataque DDoS em endpoints públicos
- **Controles**: Rate limiting global, circuit breakers
- **Risco**: Médio

#### **Elevation of Privilege (Escalação)**
- **Ameaça**: Acesso não autorizado a recursos de admin

const jwt = require('jsonwebtoken');
const crypto = require('crypto');
const { promisify } = require('util');

class AuthManager {
    constructor() {
        this.accessTokenSecret = process.env.JWT_ACCESS_SECRET || crypto.randomBytes(32).toString('hex');
        this.refreshTokenSecret = process.env.JWT_REFRESH_SECRET || crypto.randomBytes(32).toString('hex');
        this.accessTokenExpiry = '15m';
        this.refreshTokenExpiry = '7d';
        
        // In-memory token blacklist (em produção usar Redis)
        this.tokenBlacklist = new Set();
    }

    /**
     * Gera tokens JWT seguros
     */
    async generateTokens(user) {
        const accessToken = jwt.sign(
            {
                userId: user.id,
                role: user.role,
                type: 'access'
            },
            this.accessTokenSecret,
            { expiresIn: this.accessTokenExpiry, algorithm: 'HS256' }
        );

        const refreshToken = jwt.sign(
            {
                userId: user.id,
                type: 'refresh',
                jti: crypto.randomBytes(16).toString('hex') // Unique token ID
            },
            this.refreshTokenSecret,
            { expiresIn: this.refreshTokenExpiry, algorithm: 'HS256' }
        );

        // Hash do refresh token para armazenamento seguro
        const refreshTokenHash = crypto
            .createHash('sha256')
            .update(refreshToken)
            .digest('hex');

        return {
            accessToken,
            refreshToken,
            refreshTokenHash,
            expiresIn: 900 // 15 minutos em segundos
        };
    }

    /**
     * Verifica token de acesso
     */
    async verifyAccessToken(token) {
        try {
            if (this.tokenBlacklist.has(token)) {
                throw new Error('Token revoked');
            }

            const decoded = jwt.verify(token, this.accessTokenSecret, { algorithms: ['HS256'] });
            
            if (decoded.type !== 'access') {
                throw new Error('Invalid token type');
            }

            return decoded;
        } catch (error) {
            if (error.name === 'TokenExpiredError') {
                throw new Error('Token expired');
            }
            throw new Error('Invalid token');
        }
    }

    /**
     * Verifica token de refresh
     */
    async verifyRefreshToken(token) {
        try {
            const decoded = jwt.verify(token, this.refreshTokenSecret, { algorithms: ['HS256'] });
            
            if (decoded.type !== 'refresh') {
                throw new Error('Invalid token type');
            }

            // Verificar se token não foi revogado
            const tokenHash = crypto
                .createHash('sha256')
                .update(token)
                .digest('hex');

            // Em produção, verificar no banco de dados
            // const isRevoked = await db.checkTokenRevocation(tokenHash);
            
            return decoded;
        } catch (error) {
            throw new Error('Invalid refresh token');
        }
    }

    /**
     * Revoga token (logout)
     */
    async revokeToken(token) {
        try {
            const decoded = jwt.decode(token);
            const expiry = decoded.exp * 1000; // Converter para ms
            
            // Adicionar à blacklist até expirar
            this.tokenBlacklist.add(token);
            
            // Remover após expiração
            setTimeout(() => {
                this.tokenBlacklist.delete(token);
            }, expiry - Date.now());
            
        } catch (error) {
            console.error('Error revoking token:', error);
        }
    }

    /**
     * Gera hash seguro para senhas
     */
    async hashPassword(password) {
        const salt = crypto.randomBytes(16).toString('hex');
        const hash = crypto
            .pbkdf2Sync(password, salt, 100000, 64, 'sha512')
            .toString('hex');
        
        return `${salt}:${hash}`;
    }

    /**
     * Verifica senha
     */
    async verifyPassword(password, storedHash) {
        const [salt, originalHash] = storedHash.split(':');
        const hash = crypto
            .pbkdf2Sync(password, salt, 100000, 64, 'sha512')
            .toString('hex');
        
        return hash === originalHash;
    }

    /**
     * Valida complexidade de senha
     */
    validatePasswordComplexity(password) {
        const minLength = 12;
        const hasUpperCase = /[A-Z]/.test(password);
        const hasLowerCase = /[a-z]/.test(password);
        const hasNumbers = /\d/.test(password);
        const hasSpecialChar = /[!@#$%^&*(),.?":{}|<>]/.test(password);
        
        if (password.length < minLength) {
            return { valid: false, reason: `Password must be at least ${minLength} characters` };
        }
        if (!hasUpperCase) {
            return { valid: false, reason: 'Password must contain at least one uppercase letter' };
        }
        if (!hasLowerCase) {
            return { valid: false, reason: 'Password must contain at least one lowercase letter' };
        }
        if (!hasNumbers) {
            return { valid: false, reason: 'Password must contain at least one number' };
        }
        if (!hasSpecialChar) {
            return { valid: false, reason: 'Password must contain at least one special character' };
        }
        
        return { valid: true };
    }
}

module.exports = new AuthManager();

const validator = require('validator');
const xss = require('xss');

class InputSanitizer {
    constructor() {
        this.xssOptions = {
            whiteList: {}, // Empty whitelist means no HTML allowed
            stripIgnoreTag: true,
            stripIgnoreTagBody: ['script', 'style', 'iframe']
        };
    }

    /**
     * Sanitiza string removendo XSS e validando
     */
    sanitizeString(input, options = {}) {
        if (typeof input !== 'string') {
            return input;
        }

        let sanitized = input.trim();
        
        // Remover XSS
        sanitized = xss(sanitized, this.xssOptions);
        
        // Validar comprimento
        if (options.maxLength && sanitized.length > options.maxLength) {
            sanitized = sanitized.substring(0, options.maxLength);
        }
        
        // Validar formato específico
        if (options.format) {
            switch (options.format) {
                case 'email':
                    if (!validator.isEmail(sanitized)) {
                        throw new Error('Invalid email format');
                    }
                    break;
                case 'url':
                    if (!validator.isURL(sanitized)) {
                        throw new Error('Invalid URL format');
                    }
                    break;
                case 'alpha':
                    if (!validator.isAlpha(sanitized.replace(/\s/g, ''))) {
                        throw new Error('Only alphabetic characters allowed');
                    }
                    break;
                case 'alphanumeric':
                    if (!validator.isAlphanumeric(sanitized.replace(/\s/g, ''))) {
                        throw new Error('Only alphanumeric characters allowed');
                    }
                    break;
            }
        }

        return sanitized;
    }

    /**
     * Sanitiza objeto completo recursivamente
     */
    sanitizeObject(obj, schema = {}) {
        if (typeof obj !== 'object' || obj === null) {
            return this.sanitizeString(obj);
        }

        const sanitized = Array.isArray(obj) ? [] : {};
        
        for (const [key, value] of Object.entries(obj)) {
            const fieldSchema = schema[key] || {};
            
            if (typeof value === 'string') {
                sanitized[key] = this.sanitizeString(value, fieldSchema);
            } else if (typeof value === 'object' && value !== null) {
                sanitized[key] = this.sanitizeObject(value, fieldSchema.nested || {});
            } else {
                sanitized[key] = value;
            }
        }
        
        return sanitized;
    }

    /**
     * Valida e sanitiza input de usuário para SQL
     */
    sanitizeForSQL(input) {
        if (typeof input !== 'string') {
            return input;
        }
        
        // Remover caracteres perigosos para SQL
        const dangerousPatterns = [
            /(\-\-)/g, // Comentários SQL
            /(\/\*)/g, // Comentários SQL
            /(\*\/)/g,
            /(;)/g,    // Separação de comandos
            /(\b(SELECT|INSERT|UPDATE|DELETE|DROP|UNION|EXEC)\b)/gi
        ];
        
        let sanitized = input;
        dangerousPatterns.forEach(pattern => {
            sanitized = sanitized.replace(pattern, '');
        });
        
        return sanitized.trim();
    }

    /**
     * Sanitiza output de LLM antes de renderizar
     */
    sanitizeLLMOutput(output, options = {}) {
        if (typeof output !== 'string') {
            return output;
        }

        let sanitized = output;
        
        // Remover qualquer tag HTML
        sanitized = sanitized.replace(/<[^>]*>/g, '');
        
        // Remover scripts e iframes mesmo sem tags
        sanitized = sanitized.replace(/javascript:/gi, '');
        sanitized = sanitized.replace(/data:/gi, '');
        sanitized = sanitized.replace(/on\w+=/gi, '');
        
        // Validar URLs se necessário
        if (options.validateUrls) {
            const urlRegex = /(https?:\/\/[^\s]+)/g;
            const urls = sanitized.match(urlRegex) || [];
            
            urls.forEach(url => {
                if (!validator.isURL(url)) {
                    sanitized = sanitized.replace(url, '');
                }
            });
        }
        
        // Limitar comprimento
        if (options.maxLength) {
            sanitized = sanitized.substring(0, options.maxLength);
        }
        
        return sanitized;
    }

    /**
     * Middleware de sanitização automática
     */
    sanitizeMiddleware(schema = {}) {
        return (req, res, next) => {
            try {
                if (req.body) {
                    req.body = this.sanitizeObject(req.body, schema.body || {});
                }
                
                if (req.query) {
                    req.query = this.sanitizeObject(req.query, schema.query || {});
                }
                
                if (req.params) {
                    req.params = this.sanitizeObject(req.params, schema.params || {});
                }
                
                next();
            } catch (error) {
                return res.status(400).json({
                    error: 'Invalid input data',
                    details: error.message,
                    code: 'VALIDATION_ERROR'
                });
            }
        };
    }
}

# DIYAPP V13 - Arquitetura de Sistema

**Data de Criação:** 2024-01-15  
**Última Atualização:** 2024-01-15  
**Owner:** Tech Lead  
**Status:** Ativo

## Visão Geral

A V13 do DIYAPP implementa uma arquitetura baseada em microserviços com foco em estabilidade 100% autônoma, utilizando o Modo Hive para refatoração paralela.

## Componentes Principais

### 1. API Gateway
- **Tecnologia:** Node.js + Express
- **Responsabilidade:** Roteamento, autenticação, rate limiting
- **Porta:** 3000
- **Health Check:** `/health`

### 2. Serviço de Usuários
- **Tecnologia:** Node.js + TypeScript
- **Banco de Dados:** PostgreSQL (users_db)
- **Endpoints Principais:**
  - `POST /api/v1/users` - Criar usuário
  - `GET /api/v1/users/:id` - Buscar usuário
  - `PUT /api/v1/users/:id` - Atualizar usuário

### 3. Serviço de Pedidos
- **Tecnologia:** Python + FastAPI
- **Banco de Dados:** MongoDB (orders_db)
- **Eventos:** Publica para Kafka tópico `order-events`

### 4. Serviço de Pagamentos
- **Tecnologia:** Go
- **Integrações:** Stripe API, PayPal API
- **Circuit Breaker:** Implementado com Hystrix

### 5. Message Broker
- **Tecnologia:** Apache Kafka
- **Tópicos:**
  - `user-events`
  - `order-events`
  - `payment-events`
  - `notification-events`

### 6. Banco de Dados
- **PostgreSQL:** Dados transacionais
- **MongoDB:** Dados de catálogo e logs
- **Redis:** Cache e sessões

## Modo Hive - Refatoração Paralela

### Estrutura de Deploy
```
v13-stable/      # Versão atual em produção
v13-hive-alpha/  # Refatoração do serviço A
v13-hive-beta/   # Refatoração do serviço B
v13-hive-gamma/  # Refatoração do serviço C
```

### Regras do Modo Hive:
1. Cada hive roda em namespace Kubernetes separado
2. Tráfego é gradualmente direcionado (5% → 100%)
3. Rollback automático se error rate > 1%
4. Métricas comparativas entre hives

## Monitoramento e Observabilidade

### Stack:
- **Prometheus:** Coleta de métricas
- **Grafana:** Dashboards
- **Jaeger:** Tracing distribuído
- **ELK Stack:** Logs centralizados

### Métricas Críticas:
- Latência p95 < 200ms
- Error rate < 0.1%
- Uptime > 99.95%
- CPU usage < 70%
- Memory usage < 80%
