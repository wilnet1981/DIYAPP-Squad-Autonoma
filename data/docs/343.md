# DIYAPP Evolution - V13 Core - Arquitetura Técnica

## ADR-001: Arquitetura V13 - Padrões e Estrutura

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

### CONTEXTO:
A versão V13 do DIYAPP precisa estabelecer uma arquitetura sólida para suportar:
1. Operação 100% autônoma com estabilidade garantida
2. Refatoração paralela usando Modo Hive
3. Performance consistente com P95 < 200ms
4. Manutenibilidade e escalabilidade do código

### DECISÃO:
Implementar arquitetura baseada em microserviços leves com os seguintes componentes:
- API Gateway central
- Serviços especializados (LLM, Processamento, Armazenamento)
- Banco de dados PostgreSQL com Redis para cache
- Sistema de monitoramento distribuído
- Interface web unificada

### OPÇÕES CONSIDERADAS:
- **Opção A**: Monolito tradicional - Prós: Simples de desenvolver. Contras: Dificulta refatoração paralela, acoplamento alto.
- **Opção B**: Microserviços completos - Prós: Isolamento total. Contras: Complexidade operacional alta.
- **Opção C**: Microserviços leves com gateway - Prós: Equilíbrio entre isolamento e simplicidade. Contras: Requer disciplina de comunicação.

**Opção escolhida: C** - Justificativa: Permite refatoração paralela (Modo Hive) enquanto mantém operabilidade simplificada.

### CONSEQUÊNCIAS:
**Positivas:**
- Isolamento de falhas por serviço
- Refatoração independente por squad
- Escalabilidade seletiva
- Deploy independente

**Negativas:**
- Complexidade de comunicação entre serviços
- Necessidade de gateway bem projetado
- Monitoramento distribuído essencial

**Riscos:**
- Latência adicional por chamadas entre serviços
- Mitigação: Cache agressivo e design assíncrono

```javascript
// src/shared/lib/logger.js
const winston = require('winston');
const { ElasticsearchTransport } = require('winston-elasticsearch');

class Logger {
  constructor(serviceName) {
    this.logger = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      defaultMeta: { service: serviceName },
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.errors({ stack: true }),
        winston.format.json()
      ),
      transports: [
        new winston.transports.File({ 
          filename: 'logs/error.log', 
          level: 'error',
          maxsize: 5242880, // 5MB
          maxFiles: 5
        }),
        new winston.transports.File({ 
          filename: 'logs/combined.log',
          maxsize: 5242880,
          maxFiles: 10
        }),
        ...(process.env.NODE_ENV !== 'production' ? [
          new winston.transports.Console({
            format: winston.format.simple()
          })
        ] : [])
      ]
    });

    // Elasticsearch para produção
    if (process.env.ELASTICSEARCH_HOST) {
      this.logger.add(new ElasticsearchTransport({
        level: 'info',
        clientOpts: { node: process.env.ELASTICSEARCH_HOST },
        index: `logs-${serviceName}-${process.env.NODE_ENV}`
      }));
    }
  }

  log(level, message, meta = {}) {
    this.logger.log(level, message, meta);
  }

  // Métodos helpers
  info(message, meta) { this.log('info', message, meta); }
  error(message, meta) { this.log('error', message, meta); }
  warn(message, meta) { this.log('warn', message, meta); }
  debug(message, meta) { this.log('debug', message, meta); }

  // Log estruturado para métricas de performance
  performance(endpoint, duration, status, meta = {}) {
    this.info('performance_metric', {
      endpoint,
      duration_ms: duration,
      status,
      timestamp: new Date().toISOString(),
      ...meta
    });
  }
}

module.exports = Logger;
```

### 3. Tratamento de Erros Centralizado

```javascript
// src/shared/lib/errorHandler.js
class AppError extends Error {
  constructor(message, statusCode, isOperational = true) {
    super(message);
    this.statusCode = statusCode;
    this.isOperational = isOperational;
    this.timestamp = new Date().toISOString();
    
    Error.captureStackTrace(this, this.constructor);
  }
}

class ErrorHandler {
  constructor(logger) {
    this.logger = logger;
  }

  handleError(error, response = null) {
    this.logger.error(error.message, {
      stack: error.stack,
      statusCode: error.statusCode,
      isOperational: error.isOperational
    });

    // Se for erro operacional e tiver response, enviar para cliente
    if (response && error.isOperational) {
      response.status(error.statusCode || 500).json({
        status: 'error',
        message: error.message,
        ...(process.env.NODE_ENV !== 'production' && { stack: error.stack })
      });
    }

    // Erros não operacionais requerem restart
    if (!error.isOperational) {
      this.logger.error('Aplicação encerrada devido a erro não operacional');
      process.exit(1);
    }
  }

  // Middleware para Express
  middleware() {
    return (error, req, res, next) => {
      this.handleError(error, res);
    };
  }

  // Handler para promises não tratadas
  setupUnhandledRejection() {
    process.on('unhandledRejection', (reason, promise) => {
      this.logger.error('Unhandled Rejection', { reason, promise });
      throw reason;
    });
  }

  setupUncaughtException() {
    process.on('uncaughtException', (error) => {
      this.handleError(error);
      process.exit(1);
    });
  }
}

module.exports = { AppError, ErrorHandler };
```

### 5. Configuração do Gateway

```javascript
// src/gateway/index.js
const express = require('express');
const helmet = require('helmet');
const cors = require('cors');
const rateLimit = require('express-rate-limit');
const Logger = require('../shared/lib/logger');
const { ErrorHandler } = require('../shared/lib/errorHandler');

class APIGateway {
  constructor() {
    this.app = express();
    this.logger = new Logger('api-gateway');
    this.errorHandler = new ErrorHandler(this.logger);
    
    this.setupMiddleware();
    this.setupRoutes();
    this.setupErrorHandling();
    this.setupMetrics();
  }

  setupMiddleware() {
    // Segurança
    this.app.use(helmet());
    this.app.use(cors({
      origin: process.env.ALLOWED_ORIGINS?.split(',') || '*',
      credentials: true
    }));

    // Rate limiting
    const limiter = rateLimit({
      windowMs: 15 * 60 * 1000, // 15 minutos
      max: 100, // limite por IP
      message: 'Muitas requisições deste IP'
    });
    this.app.use('/api/', limiter);

    // Body parsing
    this.app.use(express.json({ limit: '10mb' }));
    this.app.use(express.urlencoded({ extended: true }));

    // Logging de requests
    this.app.use((req, res, next) => {
      const start = Date.now();
      res.on('finish', () => {
        const duration = Date.now() - start;
        this.logger.performance(req.path, duration, res.statusCode, {
          method: req.method,
          userAgent: req.get('user-agent')
        });
      });
      next();
    });
  }

  setupRoutes() {
    // Health check
    this.app.get('/health', (req, res) => {
      res.json({
        status: 'healthy',
        timestamp: new Date().toISOString(),
        version: process.env.npm_package_version,
        services: this.checkServiceHealth()
      });
    });

    // Rotas de serviço
    this.app.use('/api/llm', require('./routes/llm-routes'));
    this.app.use('/api/process', require('./routes/process-routes'));
    this.app.use('/api/storage', require('./routes/storage-routes'));
  }

  setupErrorHandling() {
    // 404 handler
    this.app.use((req, res, next) => {
      res.status(404).json({
        status: 'error',
        message: `Rota não encontrada: ${req.method} ${req.path}`
      });
    });

    // Error handler global
    this.app.use(this.errorHandler.middleware());
  }

  setupMetrics() {
    // Endpoint de métricas para Prometheus
    this.app.get('/metrics', async (req, res) => {
      const metrics = await this.collectMetrics();
      res.set('Content-Type', 'text/plain');
      res.send(metrics);
    });
  }

  async checkServiceHealth() {
    // Implementar checks de saúde dos serviços
    return {
      llm_service: 'healthy',
      processing_service: 'healthy',
      storage_service: 'healthy',
      database: 'connected'
    };
  }

  async collectMetrics() {
    // Coletar métricas do sistema
    const os = require('os');
    
    return `
      # HELP node_memory_usage Memory usage in bytes
      # TYPE node_memory_usage gauge
      node_memory_usage ${process.memoryUsage().heapUsed}
      
      # HELP node_cpu_usage CPU usage percentage
      # TYPE node_cpu_usage gauge
      node_cpu_usage ${os.loadavg()[0]}
      
      # HELP node_uptime Uptime in seconds
      # TYPE node_uptime counter
      node_uptime ${process.uptime()}
    `;
  }

  start(port = process.env.PORT || 3000) {
    this.errorHandler.setupUnhandledRejection();
    this.errorHandler.setupUncaughtException();

    this.server = this.app.listen(port, () => {
      this.logger.info(`API Gateway iniciado na porta ${port}`);
      this.logger.info(`Ambiente: ${process.env.NODE_ENV}`);
    });

    return this.server;
  }

  async shutdown() {
    this.logger.info('Iniciando shutdown gracefull...');
    
    if (this.server) {
      this.server.close(() => {
        this.logger.info('Servidor HTTP fechado');
        process.exit(0);
      });

      // Force shutdown após 10s
      setTimeout(() => {
        this.logger.error('Shutdown forçado após timeout');
        process.exit(1);
      }, 10000);
    }
  }
}

module.exports = APIGateway;
```

const { Pool } = require('pg');
const sqlite3 = require('sqlite3').verbose();
const logger = require('./logger');

class DatabaseConnection {
  constructor() {
    this.pool = null;
    this.sqliteDb = null;
    this.type = process.env.DB_TYPE || 'sqlite';
    this.init();
  }

  async init() {
    try {
      if (this.type === 'postgres') {
        this.pool = new Pool({
          host: process.env.DB_HOST,
          port: process.env.DB_PORT,
          database: process.env.DB_NAME,
          user: process.env.DB_USER,
          password: process.env.DB_PASSWORD,
          max: 20,
          idleTimeoutMillis: 30000,
          connectionTimeoutMillis: 2000,
        });

        // Test connection
        const client = await this.pool.connect();
        await client.query('SELECT NOW()');
        client.release();
        
        logger.info('PostgreSQL connection established');
      } else {
        this.sqliteDb = new sqlite3.Database(
          process.env.DB_SQLITE_PATH || './data/diyapp.db',
          sqlite3.OPEN_READWRITE | sqlite3.OPEN_CREATE,
          (err) => {
            if (err) {
              logger.error('SQLite connection error:', err);
              throw err;
            }
            logger.info('SQLite connection established');
          }
        );
      }
    } catch (error) {
      logger.error('Database connection failed:', error);
      throw error;
    }
  }

  async query(text, params = []) {
    const correlationId = global.correlationId || 'system';
    const startTime = Date.now();
    
    try {
      let result;
      
      if (this.type === 'postgres') {
        const client = await this.pool.connect();
        try {
          result = await client.query(text, params);
        } finally {
          client.release();
        }
      } else {
        result = await new Promise((resolve, reject) => {
          this.sqliteDb.all(text, params, (err, rows) => {
            if (err) reject(err);
            else resolve({ rows, rowCount: rows.length });
          });
        });
      }
      
      const duration = Date.now() - startTime;
      logger.debug('Database query executed', {
        correlationId,
        query: text.substring(0, 200),
        params: this.maskSensitiveData(params),
        duration,
        rowCount: result.rowCount
      });
      
      return result;
    } catch (error) {
      const duration = Date.now() - startTime;
      logger.error('Database query failed', {
        correlationId,
        query: text.substring(0, 200),
        params: this.maskSensitiveData(params),
        duration,
        error: error.message
      });
      
      // Não exponha detalhes do banco ao cliente
      throw new Error('Database operation failed');
    }
  }

  maskSensitiveData(params) {
    return params.map(param => {
      if (typeof param === 'string' && (
        param.includes('@') || // email
        param.match(/\d{3}\.\d{3}\.\d{3}-\d{2}/) || // CPF
        param.match(/\d{16}/) // cartão
      )) {
        return '[MASKED]';
      }
      return param;
    });
  }

  async transaction(callback) {
    if (this.type === 'postgres') {
      const client = await this.pool.connect();
      try {
        await client.query('BEGIN');
        const result = await callback(client);
        await client.query('COMMIT');
        return result;
      } catch (error) {
        await client.query('ROLLBACK');
        throw error;
      } finally {
        client.release();
      }
    } else {
      return new Promise((resolve, reject) => {
        this.sqliteDb.run('BEGIN TRANSACTION', async (err) => {
          if (err) return reject(err);
          
          try {
            const result = await callback(this.sqliteDb);
            this.sqliteDb.run('COMMIT', (err) => {
              if (err) return reject(err);
              resolve(result);
            });
          } catch (error) {
            this.sqliteDb.run('ROLLBACK', () => reject(error));
          }
        });
      });
    }
  }

  async close() {
    if (this.type === 'postgres' && this.pool) {
      await this.pool.end();
    } else if (this.sqliteDb) {
      this.sqliteDb.close();
    }
  }
}

class BaseRepository {
  constructor(tableName) {
    this.tableName = tableName;
    this.db = db;
  }

  async findById(id, columns = ['*']) {
    const query = `SELECT ${columns.join(', ')} FROM ${this.tableName} WHERE id = $1`;
    const result = await this.db.query(query, [id]);
    
    if (result.rows.length === 0) {
      return null;
    }
    
    return result.rows[0];
  }

  async findAll(where = {}, options = {}) {
    const {
      columns = ['*'],
      limit = 100,
      offset = 0,
      orderBy = 'created_at',
      orderDir = 'DESC'
    } = options;
    
    let query = `SELECT ${columns.join(', ')} FROM ${this.tableName}`;
    const params = [];
    
    const conditions = Object.keys(where);
    if (conditions.length > 0) {
      query += ' WHERE ' + conditions.map((key, index) => {
        params.push(where[key]);
        return `${key} = $${index + 1}`;
      }).join(' AND ');
    }
    
    query += ` ORDER BY ${orderBy} ${orderDir} LIMIT $${params.length + 1} OFFSET $${params.length + 2}`;
    params.push(limit, offset);
    
    const result = await this.db.query(query, params);
    return result.rows;
  }

  async create(data) {
    const columns = Object.keys(data);
    const values = Object.values(data);
    
    const placeholders = columns.map((_, index) => `$${index + 1}`).join(', ');
    const query = `
      INSERT INTO ${this.tableName} (${columns.join(', ')})
      VALUES (${placeholders})
      RETURNING *
    `;
    
    const result = await this.db.query(query, values);
    return result.rows[0];
  }

  async update(id, data) {
    const columns = Object.keys(data);
    const values = Object.values(data);
    
    const setClause = columns.map((col, index) => `${col} = $${index + 1}`).join(', ');
    const query = `
      UPDATE ${this.tableName}
      SET ${setClause}, updated_at = CURRENT_TIMESTAMP
      WHERE id = $${columns.length + 1}
      RETURNING *
    `;
    
    values.push(id);
    const result = await this.db.query(query, values);
    
    if (result.rows.length === 0) {
      throw new Error(`${this.tableName} with id ${id} not found`);
    }
    
    return result.rows[0];
  }

  async delete(id) {
    const query = `DELETE FROM ${this.tableName} WHERE id = $1 RETURNING id`;
    const result = await this.db.query(query, [id]);
    
    if (result.rows.length === 0) {
      throw new Error(`${this.tableName} with id ${id} not found`);
    }
    
    return true;
  }

  async count(where = {}) {
    let query = `SELECT COUNT(*) as count FROM ${this.tableName}`;
    const params = [];
    
    const conditions = Object.keys(where);
    if (conditions.length > 0) {
      query += ' WHERE ' + conditions.map((key, index) => {
        params.push(where[key]);
        return `${key} = $${index + 1}`;
      }).join(' AND ');
    }
    
    const result = await this.db.query(query, params);
    return parseInt(result.rows[0].count, 10);
  }

  async exists(where) {
    const count = await this.count(where);
    return count > 0;
  }

  // Método para transações complexas
  async transaction(callback) {
    return this.db.transaction(callback);
  }
}

-- Daily Active Users View
CREATE VIEW daily_active_users AS
SELECT 
    DATE(created_at) as date,
    COUNT(DISTINCT user_id) as active_users,
    COUNT(DISTINCT organization_id) as active_organizations,
    COUNT(*) as total_events
FROM user_events_raw
WHERE created_at >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY DATE(created_at);

-- LLM Cost by Feature View
CREATE VIEW llm_cost_by_feature AS
SELECT 
    DATE(lac.created_at) as date,
    o.name as organization_name,
    p.name as provider_name,
    m.model_name,
    lac.endpoint,
    SUM(lac.prompt_tokens) as total_prompt_tokens,
    SUM(lac.completion_tokens) as total_completion_tokens,
    SUM(lac.total_tokens) as total_tokens,
    SUM(lac.cost_estimated) as total_cost,
    COUNT(*) as total_calls
FROM llm_api_calls_raw lac
LEFT JOIN organizations o ON lac.organization_id = o.id
LEFT JOIN llm_providers p ON lac.provider_id = p.id
LEFT JOIN llm_models m ON lac.model_id = m.id
WHERE lac.created_at >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY DATE(lac.created_at), o.name, p.name, m.model_name, lac.endpoint;

-- Feature Adoption View
CREATE VIEW feature_adoption AS
SELECT 
    f.name as feature_name,
    f.category,
    COUNT(DISTINCT fu.organization_id) as organizations_using,
    COUNT(DISTINCT fu.user_id) as users_using,
    SUM(fu.usage_count) as total_uses,
    MIN(fu.first_used_at) as first_usage,
    MAX(fu.last_used_at) as last_usage
FROM features f
LEFT JOIN feature_usage fu ON f.id = fu.feature_id
WHERE f.is_active = true
GROUP BY f.id, f.name, f.category;

-- Deployment Frequency View (for SRE)
CREATE VIEW deployment_frequency AS
SELECT 
    DATE(completed_at) as date,
    environment,
    service_name,
    COUNT(*) as deployments_count,
    AVG(duration_seconds) as avg_duration_seconds,
    SUM(CASE WHEN status = 'success' THEN 1 ELSE 0 END) as successful_deployments,
    SUM(CASE WHEN rollback = true THEN 1 ELSE 0 END) as rollbacks_count
FROM deployments
WHERE completed_at IS NOT NULL
  AND completed_at >= CURRENT_DATE - INTERVAL '90 days'
GROUP BY DATE(completed_at), environment, service_name;

-- Function to update updated_at timestamp
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';

-- Apply triggers to all tables with updated_at column
DO $$ 
DECLARE
    t text;
BEGIN
    FOR t IN 
        SELECT table_name 
        FROM information_schema.columns 
        WHERE column_name = 'updated_at' 
          AND table_schema = 'public'
    LOOP
        EXECUTE format('
            DROP TRIGGER IF EXISTS update_%s_updated_at ON %s;
            CREATE TRIGGER update_%s_updated_at
            BEFORE UPDATE ON %s
            FOR EACH ROW
            EXECUTE FUNCTION update_updated_at_column();
        ', t, t, t, t);
    END LOOP;
END $$;

-- Function for soft delete
CREATE OR REPLACE FUNCTION soft_delete_record()
RETURNS TRIGGER AS $$
BEGIN
    NEW.deleted_at = CURRENT_TIMESTAMP;
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';
