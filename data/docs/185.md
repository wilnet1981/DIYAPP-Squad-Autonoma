const winston = require('winston');
const { v4: uuidv4 } = require('uuid');

class Logger {
  constructor(serviceName, correlationId = null) {
    this.serviceName = serviceName;
    this.correlationId = correlationId || uuidv4();
    
    this.logger = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.json()
      ),
      defaultMeta: {
        service: serviceName,
        correlation_id: this.correlationId
      },
      transports: [
        new winston.transports.Console({
          format: winston.format.combine(
            winston.format.colorize(),
            winston.format.simple()
          )
        })
      ]
    });
  }

  // Métodos de log estruturados
  info(message, meta = {}) {
    this.logger.info(message, this._sanitizeMeta(meta));
  }

  error(message, error = null, meta = {}) {
    const logMeta = { ...meta };
    
    if (error) {
      logMeta.error = {
        message: error.message,
        stack: error.stack,
        code: error.code,
        name: error.name
      };
    }
    
    this.logger.error(message, this._sanitizeMeta(logMeta));
  }

  warn(message, meta = {}) {
    this.logger.warn(message, this._sanitizeMeta(meta));
  }

  debug(message, meta = {}) {
    this.logger.debug(message, this._sanitizeMeta(meta));
  }

  // Método para criar child logger com mesmo correlationId
  child(component) {
    return new Logger(`${this.serviceName}.${component}`, this.correlationId);
  }

  // Sanitiza dados sensíveis nos logs
  _sanitizeMeta(meta) {
    const sanitized = { ...meta };
    
    // Máscara dados sensíveis
    const sensitiveFields = ['password', 'token', 'secret', 'authorization', 'credit_card', 'cpf', 'email'];
    
    Object.keys(sanitized).forEach(key => {
      const keyLower = key.toLowerCase();
      
      if (sensitiveFields.some(field => keyLower.includes(field))) {
        sanitized[key] = '***MASKED***';
      }
      
      // Se for objeto, sanitiza recursivamente
      if (sanitized[key] && typeof sanitized[key] === 'object') {
        sanitized[key] = this._sanitizeMeta(sanitized[key]);
      }
    });
    
    return sanitized;
  }

  // Getter para correlationId
  getCorrelationId() {
    return this.correlationId;
  }
}

import json
import hashlib
from typing import Any, Optional, Union
import redis
from datetime import datetime, timedelta
from src.config import settings
from src.utils.logger import logger

class RedisCache:
    """Cache implementation using Redis with TTL support"""
    
    def __init__(self):
        self.client = None
        self.enabled = settings.cache_enabled
        self.ttl = settings.cache_ttl_seconds
        
        if self.enabled:
            try:
                self.client = redis.from_url(
                    settings.redis_url,
                    password=settings.redis_password,
                    decode_responses=True,
                    socket_connect_timeout=5,
                    socket_timeout=5,
                    retry_on_timeout=True
                )
                # Test connection
                self.client.ping()
                logger.info("Redis cache connected successfully")
            except Exception as e:
                logger.error(f"Failed to connect to Redis: {e}")
                self.enabled = False
                self.client = None
    
    def _generate_key(self, agent_name: str, prompt: str, parameters: Dict[str, Any]) -> str:
        """Generate a unique cache key from agent name, prompt, and parameters"""
        key_data = {
            "agent": agent_name,
            "prompt": prompt,
            "parameters": parameters
        }
        key_string = json.dumps(key_data, sort_keys=True)
        return f"llm_cache:{hashlib.sha256(key_string.encode()).hexdigest()}"
    
    def get(self, agent_name: str, prompt: str, parameters: Dict[str, Any]) -> Optional[str]:
        """Get cached response if exists"""
        if not self.enabled or not self.client:
            return None
            
        try:
            cache_key = self._generate_key(agent_name, prompt, parameters)
            cached = self.client.get(cache_key)
            
            if cached:
                logger.debug(f"Cache hit for agent: {agent_name}")
                return cached
                
            return None
        except Exception as e:
            logger.warning(f"Cache get error: {e}")
            return None
    
    def set(self, agent_name: str, prompt: str, parameters: Dict[str, Any], response: str) -> bool:
        """Cache response with TTL"""
        if not self.enabled or not self.client:
            return False
            
        try:
            cache_key = self._generate_key(agent_name, prompt, parameters)
            success = self.client.setex(cache_key, self.ttl, response)
            
            if success:
                logger.debug(f"Cached response for agent: {agent_name}")
                
            return bool(success)
        except Exception as e:
            logger.warning(f"Cache set error: {e}")
            return False
    
    def invalidate_agent(self, agent_name: str) -> int:
        """Invalidate all cache entries for a specific agent"""
        if not self.enabled or not self.client:
            return 0
            
        try:
            pattern = f"llm_cache:*agent*{agent_name}*"
            keys = self.client.keys(pattern)
            
            if keys:
                deleted = self.client.delete(*keys)
                logger.info(f"Invalidated {deleted} cache entries for agent: {agent_name}")
                return deleted
                
            return 0
        except Exception as e:
            logger.warning(f"Cache invalidation error: {e}")
            return 0
    
    def clear(self) -> bool:
        """Clear all cache entries"""
        if not self.enabled or not self.client:
            return False
            
        try:
            keys = self.client.keys("llm_cache:*")
            if keys:
                self.client.delete(*keys)
                logger.info(f"Cleared {len(keys)} cache entries")
            return True
        except Exception as e:
            logger.warning(f"Cache clear error: {e}")
            return False
    
    def get_stats(self) -> Dict[str, Any]:
        """Get cache statistics"""
        if not self.enabled or not self.client:
            return {"enabled": False}
            
        try:
            keys = self.client.keys("llm_cache:*")
            total_size = sum(len(self.client.get(key) or "") for key in keys)
            
            return {
                "enabled": True,
                "total_entries": len(keys),
                "total_size_bytes": total_size,
                "ttl_seconds": self.ttl,
                "connected": True
            }
        except Exception as e:
            logger.warning(f"Cache stats error: {e}")
            return {"enabled": False, "error": str(e)}

from typing import Dict, Any, Optional
import hashlib
import json
from datetime import datetime, timedelta
from src.utils.logger import logger

class MemoryCache:
    """In-memory cache fallback when Redis is not available"""
    
    def __init__(self, max_size: int = 1000):
        self.cache: Dict[str, Dict[str, Any]] = {}
        self.max_size = max_size
        self.hits = 0
        self.misses = 0
    
    def _generate_key(self, agent_name: str, prompt: str, parameters: Dict[str, Any]) -> str:
        """Generate a unique cache key"""
        key_data = {
            "agent": agent_name,
            "prompt": prompt,
            "parameters": parameters
        }
        key_string = json.dumps(key_data, sort_keys=True)
        return hashlib.sha256(key_string.encode()).hexdigest()
    
    def _clean_expired(self):
        """Remove expired cache entries"""
        now = datetime.now()
        expired_keys = [
            key for key, entry in self.cache.items()
            if entry["expires_at"] < now
        ]
        for key in expired_keys:
            del self.cache[key]
    
    def get(self, agent_name: str, prompt: str, parameters: Dict[str, Any]) -> Optional[str]:
        """Get cached response"""
        self._clean_expired()
        
        cache_key = self._generate_key(agent_name, prompt, parameters)
        
        if cache_key in self.cache:
            self.hits += 1
            logger.debug(f"Memory cache hit for agent: {agent_name}")
            return self.cache[cache_key]["response"]
        
        self.misses += 1
        return None
    
    def set(self, agent_name: str, prompt: str, parameters: Dict[str, Any], 
            response: str, ttl_seconds: int = 3600) -> bool:
        """Cache response with TTL"""
        self._clean_expired()
        
        # Evict if cache is full (LRU-like)
        if len(self.cache) >= self.max_size:
            # Remove oldest entry
            oldest_key = min(self.cache.keys(), key=lambda k: self.cache[k]["created_at"])
            del self.cache[oldest_key]
        
        cache_key = self._generate_key(agent_name, prompt, parameters)
        now = datetime.now()
        
        self.cache[cache_key] = {
            "response": response,
            "created_at": now,
            "expires_at": now + timedelta(seconds=ttl_seconds),
            "agent": agent_name
        }
        
        logger.debug(f"Memory cached response for agent: {agent_name}")
        return True
    
    def get_stats(self) -> Dict[str, Any]:
        """Get cache statistics"""
        self._clean_expired()
        
        return {
            "enabled": True,
            "type": "memory",
            "total_entries": len(self.cache),
            "hits": self.hits,
            "misses": self.misses,
            "hit_rate": self.hits / (self.hits + self.misses) if (self.hits + self.misses) > 0 else 0,
            "max_size": self.max_size
        }

class PromptTemplate(BaseModel):
    """Prompt template with variables and metadata"""
    
    name: str
    template: str
    description: str = ""
    version: str = "1.0.0"
    variables: List[str] = Field(default_factory=list)
    metadata: Dict[str, Any] = Field(default_factory=dict)
    created_at: datetime = Field(default_factory=datetime.now)
    updated_at: datetime = Field(default_factory=datetime.now)
    
    def render(self, **kwargs) -> str:
        """Render template with provided variables"""
        try:
            # Validate all required variables are provided
            missing_vars = [var for var in self.variables if var not in kwargs]
            if missing_vars:
                raise ValueError(f"Missing required variables: {missing_vars}")
            
            # Render template
            rendered = self.template
            for key, value in kwargs.items():
                placeholder = f"{{{key}}}"
                if placeholder in rendered:
                    rendered = rendered.replace(placeholder, str(value))
            
            # Check for any unreplaced placeholders
            import re
            remaining_placeholders = re.findall(r'\{(\w+)\}', rendered)
            if remaining_placeholders:
                logger.warning(f"Unreplaced placeholders in template '{self.name}': {remaining_placeholders}")
            
            return rendered
            
        except Exception as e:
            logger.error(f"Error rendering template '{self.name}': {e}")
            raise
    
    def add_variable(self, variable: str):
        """Add a variable to the template"""
        if variable not in self.variables:
            self.variables.append(variable)
            self.updated_at = datetime.now()
    
    def update_template(self, new_template: str, version: str = None):
        """Update template content and version"""
        self.template = new_template
        self.updated_at = datetime.now()
        
        # Extract variables from template
        import re
        self.variables = re.findall(r'\{(\w+)\}', new_template)
        
        if version:
            self.version = version

        """Convert to dictionary"""
        return {
            "name": self.name,
            "template": self.template,
            "description": self.description,
            "version": self.version,

            "created_at": self.created_at.isoformat(),

        updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
      )
    `);

    // Criar tabela de sessões/refresh tokens
    await this.run(`
      CREATE TABLE IF NOT EXISTS refresh_tokens (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id INTEGER NOT NULL,
        token TEXT NOT NULL UNIQUE,
        device_info TEXT,
        ip_address TEXT,
        expires_at TIMESTAMP NOT NULL,
        revoked BOOLEAN DEFAULT 0,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        FOREIGN KEY (user_id) REFERENCES users (id) ON DELETE CASCADE
      )
    `);

    // Criar tabela de auditoria
    await this.run(`
      CREATE TABLE IF NOT EXISTS audit_logs (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id INTEGER,
        action TEXT NOT NULL,
        resource_type TEXT,
        resource_id TEXT,
        ip_address TEXT,
        user_agent TEXT,
        details TEXT,
        status TEXT,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        FOREIGN KEY (user_id) REFERENCES users (id) ON DELETE SET NULL
      )
    `);

    // Criar índices para performance
    await this.run('CREATE INDEX IF NOT EXISTS idx_users_email ON users(email)');
    await this.run('CREATE INDEX IF NOT EXISTS idx_refresh_tokens_token ON refresh_tokens(token)');
    await this.run('CREATE INDEX IF NOT EXISTS idx_refresh_tokens_user ON refresh_tokens(user_id)');
    await this.run('CREATE INDEX IF NOT EXISTS idx_audit_logs_user ON audit_logs(user_id)');
    await this.run('CREATE INDEX IF NOT EXISTS idx_audit_logs_created ON audit_logs(created_at)');

    // Criar usuário admin padrão (apenas se não existir)
    const adminExists = await this.get('SELECT id FROM users WHERE email = ?', ['admin@diyapp.com']);
    if (!adminExists) {
      const hashedPassword = await bcrypt.hash('Admin@123', 12);
      await this.run(
        'INSERT INTO users (email, username, password_hash, full_name, role) VALUES (?, ?, ?, ?, ?)',
        ['admin@diyapp.com', 'admin', hashedPassword, 'Administrator', 'admin']
      );
    }
  }

  run(sql, params = []) {
    return new Promise((resolve, reject) => {
      this.db.run(sql, params, function(err) {
        if (err) reject(err);
        else resolve({ id: this.lastID, changes: this.changes });
      });
    });
  }

  get(sql, params = []) {
    return new Promise((resolve, reject) => {
      this.db.get(sql, params, (err, row) => {
        if (err) reject(err);
        else resolve(row);
      });
    });
  }

  all(sql, params = []) {
    return new Promise((resolve, reject) => {
      this.db.all(sql, params, (err, rows) => {
        if (err) reject(err);
        else resolve(rows);
      });
    });
  }

  close() {
    return new Promise((resolve, reject) => {
      this.db.close((err) => {
        if (err) reject(err);
        else resolve();
      });
    });
  }
}

module.exports = new Database();

```markdown
# ADR-001: Arquitetura Modular com Modo Hive para Refatoração Paralela

## Status
**Aceito** | 2024-01-15 | **Owner**: Tech Lead | **Última Atualização**: 2024-01-15

## Contexto
O DIYAPP precisa evoluir continuamente enquanto mantém 100% de disponibilidade. Sistemas monolíticos tradicionais criam risco durante deploys e refatorações. Precisamos de uma arquitetura que permita:

1. Refatoração de componentes sem downtime
2. Deploy paralelo de múltiplas versões
3. Rollback instantâneo em caso de falhas
4. Testes A/B em produção com tráfego controlado

## Decisão
Adotaremos uma **arquitetura modular com Modo Hive**, onde:

- Cada componente é um microsserviço independente
- Múltiplas versões de um componente podem rodar simultaneamente
- Um Service Mesh gerencia o roteamento entre versões
- Métricas automáticas determinam quando promover ou reverter versões

### Componentes da Arquitetura

1. **API Gateway**: Ponto único de entrada com load balancing inteligente
2. **Hive Workers**: Instâncias rodando diferentes versões do mesmo serviço
3. **Service Mesh (Consul)**: Descoberta de serviços e roteamento
4. **Control Plane**: Orquestração do Modo Hive e tomada de decisões
5. **Metrics Collector**: Coleta e análise de métricas em tempo real

## Alternativas Consideradas

### Alternativa A: Blue-Green Deployment
- **Prós**: Rollback simples, teste completo antes de switch
- **Contras**: Requer duplicação de infra, switch é "tudo ou nada"

### Alternativa B: Canary Releases
- **Prós**: Liberação gradual, minimiza impacto
- **Contras**: Complexo de gerenciar, não permite múltiplas versões paralelas

### Alternativa C: Feature Flags
- **Prós**: Controle granular por feature
- **Contras**: Código poluído com condicionais, não resolve problemas de infra

## Consequências

### Positivas
- ✅ **Zero downtime deployments**: Usuários não percebem atualizações
- ✅ **Refatoração segura**: Teste novas versões com tráfego real controlado
- ✅ **Rollback automático**: Sistema reverte se métricas piorarem
- ✅ **Experimentation seguro**: Teste A/B com isolamento completo

### Negativas
- ⚠️ **Complexidade aumentada**: Mais componentes para gerenciar
- ⚠️ **Overhead de rede**: Comunicação entre serviços adiciona latência
- ⚠️ **Custo de infra**: Múltiplas instâncias rodando simultaneamente

        for issue in architecture_issues:

        # Sugestões de performance

            if suggestion:

        # Sugestões de qualidade de código
        for issue in code_quality_issues:
            suggestion = self.refactoring_suggester.generate_suggestion(issue)
            if suggestion:
                suggestions.append(suggestion)

        prioritized_suggestions = self._prioritize_suggestions(suggestions)
        
        # Salvar no histórico
        self.optimization_history.extend(prioritized_suggestions)
        
        # Gerar relatório
        self._generate_optimization_report(prioritized_suggestions)
        
        self.logger.info(f"Análise completa concluída. {len(prioritized_suggestions)} sugestões geradas")
        return prioritized_suggestions
    
    def _prioritize_suggestions(self, suggestions: List[OptimizationSuggestion]) -> List[OptimizationSuggestion]:
        """Prioriza sugestões com base no impacto e esforço"""
        def priority_score(suggestion: OptimizationSuggestion) -> float:
            # Fórmula: impacto * peso_nível / esforço
            level_weights = {
                OptimizationLevel.CRITICAL: 4.0,
                OptimizationLevel.HIGH: 3.0,
                OptimizationLevel.MEDIUM: 2.0,
                OptimizationLevel.LOW: 1.0
            }
            
            impact = suggestion.estimated_impact / 100.0
            effort = max(1, suggestion.estimated_effort)  # evitar divisão por zero
            level_weight = level_weights.get(suggestion.level, 1.0)
            
            return (impact * level_weight) / effort
        
        return sorted(suggestions, key=priority_score, reverse=True)
    
    def _generate_optimization_report(self, suggestions: List[OptimizationSuggestion]):
        """Gera relatório de otimização"""
        report = {
            "timestamp": datetime.utcnow().isoformat(),
            "total_suggestions": len(suggestions),
            "suggestions_by_level": {},
            "estimated_total_impact": 0,
            "estimated_total_effort": 0
        }
        
        for level in OptimizationLevel:
            level_suggestions = [s for s in suggestions if s.level == level]
            report["suggestions_by_level"][level.value] = len(level_suggestions)
        
        report["estimated_total_impact"] = sum(s.estimated_impact for s in suggestions)
        report["estimated_total_effort"] = sum(s.estimated_effort for s in suggestions)
        
        # Salvar relatório
        report_path = f"logs/optimization_report_{datetime.utcnow().strftime('%Y%m%d_%H%M%S')}.json"
        with open(report_path, 'w') as f:
            json.dump(report, f, indent=2, default=str)
        
        self.logger.info(f"Relatório de otimização salvo em: {report_path}")
        return report
    
    async def apply_optimization(self, suggestion_id: str, dry_run: bool = True) -> Dict:
        """Aplica uma otimização específica"""
        suggestion = next((s for s in self.optimization_history if s.id == suggestion_id), None)

                    "estimated_improvement": suggestion.estimated_impact,

                }

                # TODO: Implementar aplicação real baseada no tipo de sugestão

                    "applied_at": datetime.utcnow().isoformat(),
                    "dry_run": False
                }
            
            return {"success": True, "result": result}
            
        except Exception as e:
            self.logger.error(f"Erro ao aplicar otimização: {str(e)}")
            return {"success": False, "error": str(e)}
    
    def get_optimization_history(self, limit: int = 50) -> List[Dict]:

## 3. Arquitetura de Microsserviços - ADR-001

### `docs/adr/001-microservices-architecture.md`
```markdown
# ADR-001: Arquitetura de Microsserviços para DIYAPP V12

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead, Especialista Infra

## CONTEXTO:
DIYAPP evoluiu de uma aplicação monolítica para uma plataforma complexa com múltiplos domínios:
- Autenticação e usuários
- Processamento de pagamentos
- Notificações em tempo real
- Processamento de IA/ML
- Gerenciamento de conteúdo

O monolito está atingindo limites de escalabilidade, deploy e manutenção.

## DECISÃO:
Adotar arquitetura de microsserviços com os seguintes princípios:

1. **Domínios de Negócio como Serviços:**
   - `auth-service`: Autenticação, autorização, perfis
   - `payment-service`: Processamento de pagamentos, assinaturas
   - `notification-service`: Email, SMS, push, WhatsApp
   - `ai-service`: Processamento de LLMs, embeddings, RAG
   - `content-service`: Gerenciamento de conteúdo, uploads

2. **Comunicação:**
   - Síncrona: REST/GraphQL para operações críticas
   - Assíncrona: Message Queue (RabbitMQ) para eventos de domínio
   - Service Mesh: Istio para observabilidade e resiliência

3. **Banco de Dados:**
   - Database per Service pattern
   - Event Sourcing para consistência eventual
   - CQRS para separação de leitura/escrita

## OPÇÕES CONSIDERADAS:

### Opção A: Monolito Modularizado
- **Prós:** Simplicidade inicial, transações ACID fáceis
- **Contras:** Escalabilidade limitada, deploy único, acoplamento alto

### Opção B: Microsserviços Completo
- **Prós:** Escalabilidade independente, deploy contínuo, resiliência
- **Contras:** Complexidade operacional, consistência eventual, latência

### Opção C: Arquitetura Híbrida (Escolhida)
- **Prós:** Balanceia benefícios dos microsserviços com pragmatismo
- **Contras:** Requer disciplina para evitar degeneração em monolito

## CONSEQUÊNCIAS:

### Positivas:
- Escalabilidade independente por serviço
- Deploy contínuo sem downtime
- Resiliência a falhas isoladas
- Evolução tecnológica independente

### Negativas:
- Complexidade de operação aumentada
- Consistência eventual em alguns casos
- Overhead de comunicação entre serviços
- Monitoramento distribuído necessário

### Riscos:
1. **Latência de rede:** Mitigar com cache, CDN e design assíncrono
2. **Consistência de dados:** Implementar SAGA pattern e compensação
3. **Observabilidade:** Implementar tracing distribuído (Jaeger) e métricas

## REVISÃO: 2024-04-15
```

// 4. VALIDAÇÃO DE ENTREGAS - TEMPLATE
class DeliveryValidator {
  constructor(storyId) {
    this.storyId = storyId;
    this.validationLog = [];
    this.timestamp = new Date().toISOString();
  }
  
  validateCriteria(criteria, implementation) {
    return criteria.every((criterion, index) => {
      const passed = this.checkCriterion(criterion, implementation);
      this.validationLog.push({
        criterion: `Cenário ${index + 1}: ${criterion.scenario}`,
        passed,
        details: passed ? "Atendido" : `Falhou: ${criterion.gherkin.split('\n')[0]}`
      });
      return passed;
    });
  }
  
  validateDoD(implementation) {
    const failedConditions = V12_DEFINITION_OF_DONE.universalConditions
      .filter(condition => !this.checkCondition(condition, implementation));
    
    failedConditions.forEach(condition => {
      this.validationLog.push({
        criterion: `DoD: ${condition}`,
        passed: false,
        details: "Condição não atendida"
      });
    });
    
    return failedConditions.length === 0;
  }
  
  generateVerdict() {
    const allPassed = this.validationLog.every(log => log.passed);
    
    return {
      storyId: this.storyId,
      timestamp: this.timestamp,
      verdict: allPassed ? "ACCEPTED" : "REJECTED",
      score: `${this.validationLog.filter(l => l.passed).length}/${this.validationLog.length}`,
      details: this.validationLog,
      nextSteps: allPassed ? 
        "Mover para Done. Atualizar burndown chart." : 
        "Reabrir card. Notificar dev com critérios específicos falhados."
    };
  }
  
  // Métodos auxiliares (simulação)
  checkCriterion() { return Math.random() > 0.3; }
  checkCondition() { return Math.random() > 0.2; }
}
