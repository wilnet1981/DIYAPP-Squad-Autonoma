package com.diyapp.user.resilience;

import io.github.resilience4j.circuitbreaker.CircuitBreaker;
import io.github.resilience4j.circuitbreaker.CircuitBreakerConfig;
import io.github.resilience4j.circuitbreaker.CircuitBreakerRegistry;
import io.github.resilience4j.retry.Retry;
import io.github.resilience4j.retry.RetryConfig;
import io.github.resilience4j.retry.RetryRegistry;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.time.Duration;

@Configuration
public class CircuitBreakerConfig {
    
    @Bean
    public CircuitBreakerRegistry circuitBreakerRegistry() {
        CircuitBreakerConfig config = CircuitBreakerConfig.custom()
                .failureRateThreshold(50) // Abre após 50% de falhas
                .slidingWindowSize(10) // Últimas 10 chamadas
                .minimumNumberOfCalls(5) // Mínimo 5 chamadas antes de avaliar
                .waitDurationInOpenState(Duration.ofSeconds(30)) // 30s em OPEN
                .permittedNumberOfCallsInHalfOpenState(3) // 3 chamadas em HALF_OPEN
                .recordExceptions(Exception.class) // Qualquer exceção conta como falha
                .build();
        
        return CircuitBreakerRegistry.of(config);
    }
    
    @Bean
    public CircuitBreaker userServiceCircuitBreaker(CircuitBreakerRegistry registry) {
        return registry.circuitBreaker("userService");
    }
    
    @Bean
    public CircuitBreaker externalApiCircuitBreaker(CircuitBreakerRegistry registry) {
        CircuitBreakerConfig externalConfig = CircuitBreakerConfig.custom()
                .failureRateThreshold(30)
                .slidingWindowSize(20)
                .minimumNumberOfCalls(10)
                .waitDurationInOpenState(Duration.ofMinutes(1))
                .permittedNumberOfCallsInHalfOpenState(5)
                .recordExceptions(Exception.class)
                .build();
        
        return registry.circuitBreaker("externalApi", externalConfig);
    }
    
    @Bean
    public RetryRegistry retryRegistry() {
        RetryConfig config = RetryConfig.custom()
                .maxAttempts(3) // Máximo 3 tentativas
                .waitDuration(Duration.ofMillis(500)) // Base delay
                .retryOnException(e -> e instanceof RuntimeException)
                .build();
        
        return RetryRegistry.of(config);
    }
}

// user-service/src/main/java/com/diyapp/user/resilience/ResilienceAspect.java
package com.diyapp.user.resilience;

import io.github.resilience4j.circuitbreaker.CircuitBreaker;
import io.github.resilience4j.circuitbreaker.CallNotPermittedException;
import io.github.resilience4j.retry.Retry;
import io.github.resilience4j.retry.RetryRegistry;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.concurrent.TimeoutException;

@Aspect
@Component
public class ResilienceAspect {
    
    private static final Logger logger = LoggerFactory.getLogger(ResilienceAspect.class);
    
    @Autowired
    private CircuitBreaker userServiceCircuitBreaker;
    
    @Autowired
    private RetryRegistry retryRegistry;
    
    @Around("@annotation(io.github.resilience4j.circuitbreaker.annotation.CircuitBreaker)")
    public Object circuitBreakerAround(ProceedingJoinPoint joinPoint) throws Throwable {
        return userServiceCircuitBreaker.executeSupplier(() -> {
            try {
                return joinPoint.proceed();
            } catch (Throwable e) {
                logger.error("Circuit breaker protected method failed: {}", e.getMessage());
                throw new RuntimeException(e);
            }
        });
    }
    
    @Around("@annotation(io.github.resilience4j.retry.annotation.Retry)")
    public Object retryAround(ProceedingJoinPoint joinPoint) throws Throwable {
        Retry retry = retryRegistry.retry("defaultRetry");
        
        return retry.executeSupplier(() -> {
            try {
                return joinPoint.proceed();
            } catch (Throwable e) {
                logger.warn("Retry attempt failed: {}", e.getMessage());
                throw new RuntimeException(e);
            }
        });
    }
}
