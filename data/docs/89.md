// ADR-001: V12 Microservices Architecture
// Date: 2024-01-15
// Status: ACCEPTED
// Authors: Tech Lead + Infra Specialist
// ============================================

// CONTEXT:
// V11 monolith has reached scaling limits (500+ concurrent users).
// Need independent scaling of AI, messaging, and core business logic.
// Team operates in Hive Mode with 5 autonomous squads.

// DECISION:
// Adopt microservices architecture with clear bounded contexts.
// Each service owns its database. Use event-driven communication.
// Implement API Gateway + Service Mesh pattern.

// OPTIONS CONSIDERED:
// Option A: Monolith with modules - Pros: Simpler devops. Cons: Scaling bottlenecks.
// Option B: Microservices with shared DB - Pros: Faster initial dev. Cons: Tight coupling.
// Option C: Microservices with DB per service - Pros: Independent scaling, clear boundaries. Cons: Complex devops.
// Chosen: Option C - Justification: Long-term autonomy worth initial complexity.

// CONSEQUENCES:
// Positive: Each squad can deploy independently, scale based on load.
// Negative: Distributed transactions harder, eventual consistency required.
// Risks: Network latency, service discovery complexity. Mitigation: Implement circuit breakers, centralized logging.

// REVIEW: 2024-04-15 (3 months)

const serviceTemplate = `
import Fastify from 'fastify';
import cors from '@fastify/cors';
import helmet from '@fastify/helmet';
import rateLimit from '@fastify/rate-limit';
import { ZodError } from 'zod';
import { logger } from './utils/logger';
import { connectDatabase } from './config/database';
import { connectNATS } from './config/nats';
import { registerRoutes } from './routes';

// Environment validation
const envSchema = z.object({
  NODE_ENV: z.enum(['development', 'staging', 'production']),
  PORT: z.string().transform(Number).default('3000'),
  DATABASE_URL: z.string().url(),
  REDIS_URL: z.string().url(),
  NATS_URL: z.string().url(),
  JWT_SECRET: z.string().min(32),
});

type Env = z.infer<typeof envSchema>;

// Load and validate environment
const env = envSchema.parse(process.env);

// Create Fastify instance
const app = Fastify({
  logger: logger[env.NODE_ENV],
  requestIdHeader: 'x-request-id',
  genReqId: (req) => req.headers['x-request-id'] || crypto.randomUUID(),
});

// Security middleware
await app.register(helmet);
await app.register(cors, {
  origin: env.NODE_ENV === 'production' 
    ? ['https://diyapp.com'] 
    : ['http://localhost:5173'],
  credentials: true,
});

// Rate limiting
await app.register(rateLimit, {
  max: 100,
  timeWindow: '1 minute',
  redis: new Redis(env.REDIS_URL),
  keyGenerator: (req) => req.headers['x-forwarded-for'] || req.ip,
});

// Error handling
app.setErrorHandler((error, request, reply) => {
  logger.error('Request error', {
    requestId: request.id,
    error: error.message,
    stack: error.stack,
    body: request.body,
    params: request.params,
  });

  if (error instanceof ZodError) {
    return reply.status(400).send({
      error: {
        code: 'VALIDATION_ERROR',
        message: 'Invalid request data',
        details: error.errors,
      },
      meta: {
        timestamp: new Date().toISOString(),
        requestId: request.id,
      },
    });
  }

  // Don't expose internal errors in production
  const isProduction = env.NODE_ENV === 'production';
  const message = isProduction && error.statusCode === 500
    ? 'Internal server error'
    : error.message;

  return reply.status(error.statusCode || 500).send({
    error: {
      code: error.code || 'INTERNAL_ERROR',
      message,
      details: isProduction ? null : error.details,
    },
    meta: {
      timestamp: new Date().toISOString(),
      requestId: request.id,
    },
  });
});

// Health check endpoint
app.get('/health', async () => {
  return {
    status: 'healthy',
    service: 'template-service',
    timestamp: new Date().toISOString(),
    uptime: process.uptime(),
  };
});

// Register routes
await registerRoutes(app);

// Start server
const start = async () => {
  try {
    // Connect to dependencies
    await connectDatabase();
    await connectNATS();
    
    await app.listen({ 
      port: env.PORT, 
      host: '0.0.0.0' 
    });
    
    logger.info(\`Server started on port \${env.PORT}\`);
  } catch (err) {
    logger.error('Failed to start server', err);
    process.exit(1);
  }
};

// Graceful shutdown
process.on('SIGTERM', async () => {
  logger.info('SIGTERM received, shutting down gracefully');
  await app.close();
  process.exit(0);
});

process.on('SIGINT', async () => {
  logger.info('SIGINT received, shutting down gracefully');
  await app.close();
  process.exit(0);
});

start();
`;

from circuitbreaker import circuit

from src.utils.logger import get_logger

logger = get_logger(__name__)

Base = declarative_base()

class Database:
    def __init__(self):
        self._engine: Optional[AsyncEngine] = None
        self._session_factory: Optional[async_sessionmaker] = None
        self._connection_retries = 0
        self._max_retries = 3
        
    async def connect(self) -> None:
        """Estabelece conexão com o banco de dados com retry e circuit breaker"""
        try:
            await self._connect_with_retry()
        except Exception as e:
            logger.error(
                "database_connection_failed",
                error=str(e),
                retries=self._connection_retries
            )
            raise
    
    @circuit(failure_threshold=5, recovery_timeout=60)
    async def _connect_with_retry(self) -> None:
        """Tenta conectar com retry exponencial"""
        for attempt in range(self._max_retries):
            try:
                # Usar asyncpg driver para PostgreSQL async
                database_url = str(settings.DATABASE_URL).replace(
                    "postgresql://", "postgresql+asyncpg://"
                )
                
                self._engine = create_async_engine(
                    database_url,
                    pool_size=settings.DATABASE_POOL_SIZE,
                    max_overflow=settings.DATABASE_MAX_OVERFLOW,
                    pool_recycle=settings.DATABASE_POOL_RECYCLE,
                    pool_pre_ping=True,  # Verifica conexão antes de usar
                    echo=settings.DEBUG,
                    future=True
                )
                
                # Testa a conexão
                async with self._engine.begin() as conn:
                    await conn.execute("SELECT 1")
                
                self._session_factory = async_sessionmaker(
                    self._engine,
                    class_=AsyncSession,
                    expire_on_commit=False,
                    autocommit=False,
                    autoflush=False
                )
                
                logger.info(
                    "database_connected",
                    pool_size=settings.DATABASE_POOL_SIZE,
                    max_overflow=settings.DATABASE_MAX_OVERFLOW
                )
                self._connection_retries = 0
                return
                
            except (OperationalError, SQLAlchemyError) as e:
                self._connection_retries += 1
                wait_time = 2 ** attempt  # Backoff exponencial
                
                logger.warning(
                    "database_connection_retry",
                    attempt=attempt + 1,
                    max_attempts=self._max_retries,
                    wait_time=wait_time,
                    error=str(e)
                )
                
                if attempt == self._max_retries - 1:
                    raise
                
                time.sleep(wait_time)
    
    async def disconnect(self) -> None:
        """Fecha conexão com o banco de dados"""
        if self._engine:
            await self._engine.dispose()
            logger.info("database_disconnected")
    
    @asynccontextmanager
    async def get_session(self) -> AsyncGenerator[AsyncSession, None]:
        """Context manager para sessões de banco de dados com tratamento de erros"""
        if not self._session_factory:
            raise RuntimeError("Database não está conectado")
        
        session: AsyncSession = self._session_factory()
        correlation_id = getattr(session.info, 'correlation_id', 'unknown')
        
        try:
            session.info['correlation_id'] = correlation_id
            yield session
            await session.commit()
            
        except SQLAlchemyError as e:
            await session.rollback()
            logger.error(
                "database_session_error",
                correlation_id=correlation_id,
                error_type=type(e).__name__,
                error=str(e)
            )
            raise
        except Exception as e:
            await session.rollback()
            logger.error(
                "database_unexpected_error",
                correlation_id=correlation_id,
                error_type=type(e).__name__,
                error=str(e)
            )
            raise
        finally:
            await session.close()
    
    async def health_check(self) -> dict:
        """Health check do banco de dados"""
        try:
            async with self._engine.connect() as conn:
                result = await conn.execute("SELECT 1")
                row = result.fetchone()
                return {
                    "status": "healthy" if row[0] == 1 else "unhealthy",
                    "connection_pool": {
                        "checked_out": self._engine.pool.checkedout(),
                        "checked_in": self._engine.pool.checkedin(),
                        "size": self._engine.pool.size()
                    }
                }
        except Exception as e:
            logger.error("database_health_check_failed", error=str(e))
            return {"status": "unhealthy", "error": str(e)}

    SENSITIVE_FIELDS = {'hashed_password', 'email'}
    
    def to_dict(self, mask_sensitive: bool = True) -> dict:
        """Converte para dict, mascarando campos sensíveis por padrão"""
        data = {
            'id': str(self.id),
            'email': self.email,
            'username': self.username,
            'full_name': self.full_name,
            'is_active': self.is_active,
            'is_superuser': self.is_superuser,
            'created_at': self.created_at.isoformat() if self.created_at else None,
            'updated_at': self.updated_at.isoformat() if self.updated_at else None
        }
        
        if mask_sensitive:
            data['email'] = self._mask_email(self.email)
        
        return data
    
    @staticmethod
    def _mask_email(email: str) -> str:
        """Mascara email para logs"""
        if '@' not in email:
            return '***@***'
        local, domain = email.split('@', 1)
        if len(local) <= 2:
            masked_local = '*' * len(local)
        else:
            masked_local = local[0] + '*' * (len(local) - 2) + local[-1]
        return f"{masked_local}@{domain}"

**src/store/index.ts**
```typescript
import { create } from 'zustand';
import { devtools, persist } from 'zustand/middleware';
import { AuthState, UIState, AppState, User } from '../types';

interface StoreState {
  // Auth
  auth: AuthState;
  login: (email: string, password: string) => Promise<void>;
  logout: () => void;
  setUser: (user: User) => void;
  
  // UI
  ui: UIState;
  toggleTheme: () => void;
  toggleSidebar: () => void;
  addNotification: (notification: Omit<Notification, 'id' | 'createdAt'>) => void;
  removeNotification: (id: string) => void;
  
  // App
  app: AppState;
  setLoading: (isLoading: boolean) => void;
  setError: (error: string | null) => void;
}

const initialAuthState: AuthState = {
  user: null,
  token: null,
  isLoading: false,
  error: null,
  isAuthenticated: false,
};

const initialUIState: UIState = {
  theme: 'light',
  sidebarOpen: true,
  notifications: [],
  modals: {},
};

const initialAppState: AppState = {
  isLoading: false,
  error: null,
  version: 'v12.0.0',
  lastUpdated: new Date().toISOString(),
};

export const useStore = create<StoreState>()(
  devtools(
    persist(
      (set, get) => ({
        auth: initialAuthState,
        ui: initialUIState,
        app: initialAppState,
        
        // Auth actions
        login: async (email: string, password: string) => {
          set(state => ({ 
            auth: { ...state.auth, isLoading: true, error: null } 
          }));
          
          try {
            // Simulate API call
            await new Promise(resolve => setTimeout(resolve, 1000));
            
            const mockUser: User = {
              id: '1',
              email,
              name: 'John Doe',
              role: 'user',
              createdAt: new Date().toISOString(),
            };
            
            set(state => ({
              auth: {
                ...state.auth,
                user: mockUser,
                token: 'mock-jwt-token',
                isLoading: false,
                isAuthenticated: true,
              },
            }));
          } catch (error) {
            set(state => ({
              auth: {
                ...state.auth,
                isLoading: false,
                error: error instanceof Error ? error.message : 'Login failed',
              },
            }));
          }
        },
        
        logout: () => {
          set(state => ({
            auth: initialAuthState,
          }));
        },
        
        setUser: (user: User) => {
          set(state => ({
            auth: { ...state.auth, user },
          }));
        },
        
        // UI actions
        toggleTheme: () => {
          set(state => ({
            ui: {
              ...state.ui,
              theme: state.ui.theme === 'light' ? 'dark' : 'light',
            },
          }));
        },
        
        toggleSidebar: () => {
          set(state => ({
            ui: {
              ...state.ui,
              sidebarOpen: !state.ui.sidebarOpen,
            },
          }));
        },
        
        addNotification: (notification) => {
          const newNotification = {
            ...notification,
            id: Math.random().toString(36).substr(2, 9),
            createdAt: new Date().toISOString(),
          };
          
          set(state => ({
            ui: {
              ...state.ui,
              notifications: [...state.ui.notifications, newNotification],
            },
          }));
        },
        
        removeNotification: (id: string) => {
          set(state => ({
            ui: {
              ...state.ui,
              notifications: state.ui.notifications.filter(n => n.id !== id),
            },
          }));
        },
        
        // App actions
        setLoading: (isLoading: boolean) => {
          set(state => ({
            app: { ...state.app, isLoading },
          }));
        },
        
        setError: (error: string | null) => {
          set(state => ({
            app: { ...state.app, error },
          }));
        },
      }),
      {
        name: 'diyapp-store',
        partialize: (state) => ({
          auth: {
            user: state.auth.user,
            token: state.auth.token,
            isAuthenticated: state.auth.isAuthenticated,
          },
          ui: {
            theme: state.ui.theme,
          },
        }),
      }
    )
  )
);

// Custom hooks for store slices
export const useAuth = () => useStore(state => state.auth);
export const useUI = () => useStore(state => state.ui);
export const useApp = () => useStore(state => state.app);
```

```typescript
import React, { forwardRef, ButtonHTMLAttributes } from 'react';
import { BaseComponentProps, ButtonVariant, ButtonSize } from '../../../types';
import './Button.css';

interface ButtonProps extends BaseComponentProps, ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: ButtonVariant;
  size?: ButtonSize;
  isLoading?: boolean;
  leftIcon?: React.ReactNode;
  rightIcon?: React.ReactNode;
  fullWidth?: boolean;
}

const Button = forwardRef<HTMLButtonElement, ButtonProps>(
  (
    {
      children,
      variant = 'primary',
      size = 'md',
      isLoading = false,
      disabled = false,
      leftIcon,
      rightIcon,
      fullWidth = false,
      className = '',
      'data-testid': testId = 'button',
      'aria-label': ariaLabel,
      role = 'button',
      ...props
    },
    ref
  ) => {
    const isDisabled = disabled || isLoading;
    
    const baseClasses = 'button';
    const variantClass = `button--${variant}`;
    const sizeClass = `button--${size}`;
    const widthClass = fullWidth ? 'button--full-width' : '';
    const loadingClass = isLoading ? 'button--loading' : '';
    const disabledClass = isDisabled ? 'button--disabled' : '';
    
    const buttonClasses = [
      baseClasses,
      variantClass,
      sizeClass,
      widthClass,
      loadingClass,
      disabledClass,
      className,
    ]
      .filter(Boolean)
      .join(' ');
    
    return (
      <button
        ref={ref}
        className={buttonClasses}
        disabled={isDisabled}
        data-testid={testId}
        aria-label={ariaLabel || (typeof children === 'string' ? children : undefined)}
        role={role}
        aria-busy={isLoading}
        {...props}
      >
        {isLoading && (
          <span className="button__loader" aria-hidden="true">
            <div className="button__loader-dot" />
            <div className="button__loader-dot" />
            <div className="button__loader-dot" />
          </span>
        )}
        
        {!isLoading && leftIcon && (
          <span className="button__icon button__icon--left" aria-hidden="true">
            {leftIcon}
          </span>
        )}
        
        <span className="button__content">{children}</span>
        
        {!isLoading && rightIcon && (
          <span className="button__icon button__icon--right" aria-hidden="true">
            {rightIcon}
          </span>
        )}
      </button>
    );
  }
);

Button.displayName = 'Button';

export default Button;
```

from datetime import datetime
from typing import List, Dict, Any
from dataclasses import dataclass
from google.cloud import bigquery
from google.cloud.exceptions import NotFound

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

@dataclass
class Migration:
    version: str
    description: str
    up_sql: str
    down_sql: str
    dependencies: List[str] = None
    
class MigrationSystem:
    def __init__(self, project_id: str, dataset_id: str):
        self.client = bigquery.Client(project=project_id)
        self.project_id = project_id
        self.dataset_id = dataset_id
        self.migrations_table = f"{project_id}.{dataset_id}.schema_migrations"
        
    def _ensure_migrations_table(self):
        """Cria tabela de controle de migrações se não existir"""
        schema = [
            bigquery.SchemaField("version", "STRING", mode="REQUIRED"),
            bigquery.SchemaField("description", "STRING", mode="REQUIRED"),
            bigquery.SchemaField("applied_at", "TIMESTAMP", mode="REQUIRED"),
            bigquery.SchemaField("execution_time_ms", "INTEGER"),
            bigquery.SchemaField("success", "BOOLEAN"),
            bigquery.SchemaField("checksum", "STRING")
        ]
        
        table_ref = self.client.dataset(self.dataset_id).table("schema_migrations")
        try:
            self.client.get_table(table_ref)
            logger.info("Tabela de migrações já existe")
        except NotFound:
            table = bigquery.Table(table_ref, schema=schema)
            table = self.client.create_table(table)
            logger.info(f"Tabela de migrações criada: {table.table_id}")
    
    def apply_migration(self, migration: Migration):
        """Aplica uma migração individual"""
        self._ensure_migrations_table()
        
        # Verifica se migração já foi aplicada
        query = f"""
        SELECT version FROM `{self.migrations_table}`
        WHERE version = '{migration.version}'
        AND success = TRUE
        """
        result = self.client.query(query).result()
        
        if result.total_rows > 0:
            logger.info(f"Migração {migration.version} já aplicada, pulando...")
            return
        
        # Verifica dependências
        if migration.dependencies:
            for dep in migration.dependencies:
                dep_query = f"""
                SELECT version FROM `{self.migrations_table}`
                WHERE version = '{dep}' AND success = TRUE
                """
                dep_result = self.client.query(dep_query).result()
                if dep_result.total_rows == 0:
                    raise Exception(f"Dependência não satisfeita: {dep}")
        
        # Aplica migração
        start_time = datetime.now()
        try:
            job = self.client.query(migration.up_sql)
            job.result()  # Aguarda conclusão
            
            execution_time = int((datetime.now() - start_time).total_seconds() * 1000)
            
            # Registra migração
            insert_query = f"""
            INSERT INTO `{self.migrations_table}`
            (version, description, applied_at, execution_time_ms, success, checksum)
            VALUES (
                '{migration.version}',
                '{migration.description}',
                CURRENT_TIMESTAMP(),
                {execution_time},
                TRUE,
                '{hash(migration.up_sql)}'
            )
            """
            self.client.query(insert_query).result()
            
            logger.info(f"Migração {migration.version} aplicada com sucesso em {execution_time}ms")
            
        except Exception as e:
            logger.error(f"Erro na migração {migration.version}: {str(e)}")
            
            # Tenta rollback
            try:
                if migration.down_sql:
                    self.client.query(migration.down_sql).result()
                    logger.info(f"Rollback da migração {migration.version} executado")
            except Exception as rollback_error:

```javascript
                code: 'AUTH_TOKEN_MISSING'

            };

                method: req.method,

            });

        } catch (error) {

                event: 'INVALID_TOKEN',

            });
```

```javascript
const xss = require('xss');
const validator = require('validator');

class InputSanitizer {
    constructor() {
        this.xssOptions = {
            whiteList: {}, // Nenhuma tag permitida por padrão
            stripIgnoreTag: true,
            stripIgnoreTagBody: ['script', 'style', 'iframe', 'object', 'embed']
        };
    }

    // Sanitização geral de strings
    sanitizeString(input, options = {}) {
        if (typeof input !== 'string') return input;
        
        let sanitized = input.trim();
        
        // Remove caracteres de controle
        sanitized = sanitized.replace(/[\x00-\x1F\x7F-\x9F]/g, '');
        
        // Sanitização XSS
        sanitized = xss(sanitized, this.xssOptions);
        
        // Limitação de tamanho se especificado
        if (options.maxLength && sanitized.length > options.maxLength) {
            sanitized = sanitized.substring(0, options.maxLength);
        }
        
        return sanitized;
    }

    // Validação de email
    sanitizeEmail(email) {
        if (!email || typeof email !== 'string') {
            throw new Error('Email inválido');
        }
        
        const sanitized = email.toLowerCase().trim();
        
        if (!validator.isEmail(sanitized)) {
            throw new Error('Formato de email inválido');
        }
        
        // Proteção contra email injection
        if (sanitized.includes('\n') || sanitized.includes('\r')) {
            throw new Error('Email contém caracteres inválidos');
        }
        
        return sanitized;
    }

    // Sanitização de números
    sanitizeNumber(input, options = {}) {
        if (typeof input === 'number') return input;
        
        const str = String(input).trim();
        
        // Remove caracteres não numéricos (exceto ponto decimal e sinal negativo)
        let sanitized = str.replace(/[^\d.-]/g, '');
        
        // Converte para número
        const num = parseFloat(sanitized);
        
        if (isNaN(num)) {
            throw new Error('Valor numérico inválido');
        }
        
        // Validação de range
        if (options.min !== undefined && num < options.min) {
            throw new Error(`Valor deve ser maior ou igual a ${options.min}`);
        }
        
        if (options.max !== undefined && num > options.max) {
            throw new Error(`Valor deve ser menor ou igual a ${options.max}`);
        }
        
        return num;
    }

    // Sanitização de objetos JSON
    sanitizeObject(obj, schema) {
        if (!obj || typeof obj !== 'object') {
            throw new Error('Objeto inválido');
        }
        
        const sanitized = {};
        
        for (const [key, config] of Object.entries(schema)) {
            if (config.required && !(key in obj)) {
                throw new Error(`Campo obrigatório faltando: ${key}`);
            }
            
            if (key in obj) {
                try {
                    sanitized[key] = this.sanitizeByType(obj[key], config);
                } catch (error) {
                    throw new Error(`Erro no campo ${key}: ${error.message}`);
                }
            } else if (config.default !== undefined) {
                sanitized[key] = config.default;
            }
        }
        
        return sanitized;
    }

    // Sanitização por tipo
    sanitizeByType(value, config) {
        switch (config.type) {
            case 'string':
                return this.sanitizeString(value, config);
            case 'email':
                return this.sanitizeEmail(value);
            case 'number':
                return this.sanitizeNumber(value, config);
            case 'boolean':
                return Boolean(value);
            case 'array':
                if (!Array.isArray(value)) {
                    throw new Error('Esperado array');
                }
                return value.map(item => 
                    this.sanitizeByType(item, config.items || { type: 'string' })
                );
            case 'object':
                if (config.schema) {
                    return this.sanitizeObject(value, config.schema);
                }
                return value;
            default:
                return this.sanitizeString(String(value));
        }
    }

    // Sanitização específica para SQL (prevenção de SQL Injection)
    sanitizeSQLInput(input) {
        if (typeof input !== 'string') return input;
        
        // Remove ou escapa caracteres perigosos
        return input
            .replace(/'/g, "''")
            .replace(/--/g, '')
            .replace(/;/g, '')
            .replace(/\/\*/g, '')
            .replace(/\*\//g, '');
    }

    // Sanitização para URLs
    sanitizeURL(url) {
        if (!url || typeof url !== 'string') {
            throw new Error('URL inválida');
        }
        
        const sanitized = url.trim();
        
        if (!validator.isURL(sanitized, {
            require_protocol: true,
            protocols: ['http', 'https'],
            require_valid_protocol: true
        })) {
            throw new Error('URL inválida');
        }
        
        return sanitized;
    }

    // Validação de senha forte
    validatePassword(password) {
        if (typeof password !== 'string' || password.length < 12) {
            throw new Error('Senha deve ter pelo menos 12 caracteres');
        }
        
        const checks = {
            hasLowercase: /[a-z]/.test(password),
            hasUppercase: /[A-Z]/.test(password),
            hasNumbers: /\d/.test(password),
            hasSpecial: /[!@#$%^&*(),.?":{}|<>]/.test(password)
        };
        
        const passedChecks = Object.values(checks).filter(Boolean).length;
        
        if (passedChecks < 3) {
            throw new Error('Senha muito fraca. Use letras maiúsculas, minúsculas, números e caracteres especiais');
        }
        
        // Verifica senhas comuns
        const commonPasswords = [
            'password', '123456', 'qwerty', 'admin', 'welcome'
        ];
        
        if (commonPasswords.includes(password.toLowerCase())) {
            throw new Error('Senha muito comum. Escolha uma senha mais segura');
        }
        
        return password;
    }
}
```

```python
class DriftDetector:
    def __init__(self, config: Dict):
        self.config = config
        self.drift_threshold = config['mlops']['monitoring']['drift_threshold']
        self.window_size = config['mlops']['monitoring']['window_size']
        self.baseline_data = {}
        self.drift_history = []
        
    def set_baseline(self, feature_name: str, metrics: Dict[str, List[float]]):
        """Define baseline para uma feature específica"""
        self.baseline_data[feature_name] = {
            'mean': {k: np.mean(v) for k, v in metrics.items()},
            'std': {k: np.std(v) for k, v in metrics.items()},
            'distribution': metrics,
            'timestamp': datetime.now(),
            'sample_size': len(next(iter(metrics.values())))
        }
        logger.info(f"Baseline set for {feature_name} with {self.baseline_data[feature_name]['sample_size']} samples")
        
    def detect_drift(self, feature_name: str, current_metrics: Dict[str, List[float]]) -> List[DriftMetrics]:
        """Detecta drift em múltiplas métricas"""
        if feature_name not in self.baseline_data:
            logger.warning(f"No baseline found for {feature_name}")
            return []
            
        baseline = self.baseline_data[feature_name]
        drift_results = []
        
        for metric_name, current_values in current_metrics.items():
            if metric_name not in baseline['distribution']:
                continue
                
            baseline_values = baseline['distribution'][metric_name]
            
            # Teste estatístico (Kolmogorov-Smirnov)
            ks_statistic, p_value = stats.ks_2samp(baseline_values, current_values)
            
            # Cálculo de diferença percentual
            current_mean = np.mean(current_values)
            baseline_mean = baseline['mean'][metric_name]
            
            if baseline_mean == 0:
                drift_percentage = 0
            else:
                drift_percentage = abs((current_mean - baseline_mean) / baseline_mean)
            
            is_significant = (p_value < 0.05) and (drift_percentage > self.drift_threshold)
            
            drift_metric = DriftMetrics(
                feature_name=feature_name,
                model_version="current",

                is_significant=is_significant,
                samples_analyzed=len(current_values)
            )
            
            drift_results.append(drift_metric)
            
            if is_significant:
                logger.warning(
                    f"Significant drift detected in {feature_name}.{metric_name}: "
                    f"{drift_percentage:.2%} change (p={p_value:.4f})"
                )
```

```python
import asyncio
from typing import Dict, List, Optional, Any
import time
from dataclasses import dataclass
from concurrent.futures import ThreadPoolExecutor
import hashlib
from functools import lru_cache
from src.utils.logger import get_logger

    latency_ms: float
    tokens_per_second: float
    cache_hit: bool
    batch_size: int
    timestamp: float

class InferenceOptimizer:
    def __init__(self, config: Dict):
        self.config = config
        self.cache = {}
        self.cache_ttl = config['mlops']['inference']['optimization']['cache_ttl']
        self.batch_size = config['mlops']['inference']['optimization']['batch_size']
        self.executor = ThreadPoolExecutor(max_workers=10)
        self.metrics_history = []
        
    async def optimize_inference(
        self, 
        model_name: str, 
        prompts: List[str],
        use_cache: bool = True
    ) -> List[str]:
        """Otimiza inferência com caching e batching"""
        start_time = time.time()
        
        # Verifica cache primeiro
        if use_cache:
            cached_results = self._check_cache(prompts)
            if cached_results and all(cached_results):
                logger.info(f"Cache hit for {len(prompts)} prompts")
                return cached_results
        
        # Agrupa prompts para batch processing
        batched_prompts = self._create_batches(prompts)
        
        # Processa batches em paralelo
        tasks = []
        for batch in batched_prompts:
            task = asyncio.create_task(
                self._process_batch(model_name, batch)
            )
            tasks.append(task)
        
        results = await asyncio.gather(*tasks)
        flat_results = [item for sublist in results for item in sublist]
        
        # Atualiza cache
        if use_cache:
            self._update_cache(prompts, flat_results)
        
        # Coleta métricas
        latency = (time.time() - start_time) * 1000
        self._record_metrics(model_name, latency, len(prompts))
        
        return flat_results
    
    def _create_batches(self, prompts: List[str]) -> List[List[str]]:
        """Cria batches otimizados"""
        batch_size = self.batch_size
        return [prompts[i:i + batch_size] for i in range(0, len(prompts), batch_size)]
    
    async def _process_batch(self, model_name: str, batch: List[str]) -> List[str]:
        """Processa um batch de prompts"""
        # Aqui viria a integração real com a API do modelo
        # Simulação para demonstração
        await asyncio.sleep(0.1 * len(batch))  # Simula latência
        
        # Em produção, chamaria a API do modelo
        results = [f"Response to: {prompt[:50]}..." for prompt in batch]
        return results
    
    def _check_cache(self, prompts: List[str]) -> Optional[List[str]]:
        """Verifica se prompts estão em cache"""
        cache_keys = [self._generate_cache_key(p) for p in prompts]
        results = []
        
        for key in cache_keys:
            if key in self.cache:
                cache_entry = self.cache[key]
                # Verifica TTL
                if time.time() - cache_entry['timestamp'] < self.cache_ttl:
                    results.append(cache_entry['response'])
                else:
                    results.append(None)
                    del self.cache[key]
            else:
                results.append(None)
        
        return results if all(results) else None
    
    def _update_cache(self, prompts: List[str], responses: List[str]):
        """Atualiza cache com novas respostas"""
        for prompt, response in zip(prompts, responses):
            key = self._generate_cache_key(prompt)
            self.cache[key] = {
                'response': response,
                'timestamp': time.time(),
                'prompt_hash': key
            }
    
    def _generate_cache_key(self, prompt: str) -> str:
        """Gera chave de cache única para um prompt"""
        return hashlib.md5(prompt.encode()).hexdigest()
    
    def _record_metrics(self, model_name: str, latency_ms: float, num_prompts: int):
        """Registra métricas de inferência"""
        metrics = InferenceMetrics(
            model_name=model_name,
            latency_ms=latency_ms,
            tokens_per_second=(num_prompts * 100) / (latency_ms / 1000),  # Estimativa
            cache_hit=False,  # Seria calculado
            batch_size=num_prompts,
            timestamp=time.time()
        )
        self.metrics_history.append(metrics)
        
        # Log de performance
        if latency_ms > 1000:  # Alerta para latência > 1s
            logger.warning(f"High latency for {model_name}: {latency_ms:.2f}ms")
    
    def get_performance_report(self) -> Dict:
        """Gera relatório de performance de inferência"""
        if not self.metrics_history:
            return {}
        
        recent_metrics = self.metrics_history[-100:]  # Últimas 100 inferências
        
        report = {
            "total_inferences": len(self.metrics_history),
            "avg_latency_ms": np.mean([m.latency_ms for m in recent_metrics]),
            "p95_latency_ms": np.percentile([m.latency_ms for m in recent_metrics], 95),
            "p99_latency_ms": np.percentile([m.latency_ms for m in recent_metrics], 99),
            "avg_tokens_per_second": np.mean([m.tokens_per_second for m in recent_metrics]),
            "cache_size": len(self.cache),
            "cache_hit_rate": self._calculate_cache_hit_rate(),
            "models_used": list(set(m.model_name for m in recent_metrics))
        }
        
        return report
    
    def _calculate_cache_hit_rate(self) -> float:
        """Calcula taxa de cache hit"""
        # Implementação simplificada
        return 0.0  # Em produção, rastrearia hits/misses
```

## 1. Princípios Arquiteturais

### 1.1 Autonomia Total
- Cada microserviço é 100% autônomo
- Zero dependências síncronas entre serviços
- Circuit breakers em todas as comunicações externas

### 1.2 Resiliência por Design
- Retry com backoff exponencial
- Dead letter queues para mensagens não processadas
- Health checks em três níveis (serviço, dependência, recurso)

### 1.3 Observabilidade
- Métricas: Prometheus + Grafana
- Logs: ELK Stack (Estrutura JSON padronizada)
- Tracing: Jaeger para distributed tracing

### 3. Middleware de Sanitização de Logs

**src/compliance/middleware/LogSanitizer.js**
```javascript
const ComplianceConfig = require('../config/compliance-config');

class LogSanitizer {
  constructor() {
    this.sanitizationPatterns = ComplianceConfig.logging.sanitization.patterns;
    this.replacement = ComplianceConfig.logging.sanitization.replacement;
  }

  /**
   * Sanitiza dados sensíveis em logs
   */
  sanitizeLogData(data) {
    if (!ComplianceConfig.logging.sanitization.enabled || !data) {
      return data;
    }

    let sanitizedData = JSON.stringify(data);
    
    this.sanitizationPatterns.forEach(pattern => {
      sanitizedData = sanitizedData.replace(pattern, this.replacement);
    });

    try {
      return JSON.parse(sanitizedData);
    } catch (error) {
      // Se não conseguir parsear, retorna como string sanitizada
      return sanitizedData;
    }
  }

  /**
   * Middleware para sanitizar logs automaticamente
   */
  middleware() {
    return (req, res, next) => {
      const originalSend = res.send;
      
      // Intercepta a resposta para sanitizar logs
      res.send = function(body) {
        // Sanitiza o body antes de logar
        if (process.env.NODE_ENV === 'production') {
          const sanitizedBody = this.sanitizeLogData(body);
          // Log sanitizado
          console.log({
            timestamp: new Date().toISOString(),
            method: req.method,
            url: req.url,
            statusCode: res.statusCode,
            responseBody: sanitizedBody,
            userId: req.user?.id || 'anonymous',
            ip: req.ip
          });
        }
        
        originalSend.call(this, body);
      }.bind(this);

      next();
    };
  }

  /**
   * Sanitiza objeto específico para logging
   */
  sanitizeForLogging(obj) {
    const sensitiveFields = [
      'password',
      'token',
      'secret',
      'creditCard',
      'cpf',
      'cnpj',
      'email',
      'phone',
      'address'
    ];

    const sanitized = { ...obj };
    
    sensitiveFields.forEach(field => {
      if (sanitized[field]) {
        sanitized[field] = this.replacement;
      }
    });

    return sanitized;
  }
}

module.exports = new LogSanitizer();
```

### 4. Serviço de Retenção de Dados

**src/compliance/services/DataRetentionService.js**
```javascript
const ComplianceConfig = require('../config/compliance-config');
const Database = require('../../database');
const logger = require('../../utils/logger');

class DataRetentionService {
  constructor() {
    this.retentionPeriods = ComplianceConfig.lgpd.retentionPeriods;
  }

  /**
   * Executa limpeza de dados expirados
   */
  async executeDataRetention() {
    try {
      logger.info('Iniciando processo de retenção de dados LGPD/GDPR');

      const results = {
        usersDeleted: 0,
        logsDeleted: 0,
        anonymizedRecords: 0,
        errors: []
      };

      // 1. Anonimizar dados de usuários inativos
      results.anonymizedRecords = await this.anonymizeInactiveUsers();
      
      // 2. Excluir logs antigos
      results.logsDeleted = await this.deleteOldLogs();
      
      // 3. Excluir backups temporários
      await this.cleanupTempBackups();
      
      // 4. Registrar execução
      await this.logRetentionExecution(results);

      logger.info(`Processo de retenção concluído: ${JSON.stringify(results)}`);
      return results;

    } catch (error) {
      logger.error('Erro no processo de retenção:', error);
      throw error;
    }
  }

  /**
   * Anonimiza dados de usuários inativos
   */
  async anonymizeInactiveUsers() {
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - this.retentionPeriods.userData);

    const query = `
      UPDATE users 
      SET 
        email = CONCAT('anon_', id, '@diyapp.com'),
        name = 'Usuário Anonimizado',
        phone = NULL,
        cpf = NULL,
        address = NULL,
        anonymized_at = NOW(),
        is_anonymized = TRUE
      WHERE 
        last_login < ? 
        AND is_anonymized = FALSE
        AND deleted_at IS NULL
    `;

    const result = await Database.execute(query, [cutoffDate]);
    return result.affectedRows;
  }

  /**
   * Exclui logs antigos
   */
  async deleteOldLogs() {
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - this.retentionPeriods.activityLogs);

    const queries = [
      // Logs de atividade
      `DELETE FROM activity_logs WHERE created_at < ?`,
      // Logs de acesso
      `DELETE FROM access_logs WHERE timestamp < ?`,
      // Logs de auditoria
      `DELETE FROM audit_logs WHERE created_at < ?`
    ];

    let totalDeleted = 0;
    
    for (const query of queries) {
      const result = await Database.execute(query, [cutoffDate]);
      totalDeleted += result.affectedRows;
    }

    return totalDeleted;
  }

  /**
   * Limpa backups temporários
   */
  async cleanupTempBackups() {
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - this.retentionPeriods.backupData);

    const query = `
      DELETE FROM data_backups 
      WHERE is_temporary = TRUE 
      AND created_at < ?
    `;

    await Database.execute(query, [cutoffDate]);
  }

  /**
   * Registra execução do processo
   */
  async logRetentionExecution(results) {
    const query = `
      INSERT INTO retention_execution_logs 
      (execution_date, users_anonymized, logs_deleted, errors) 
      VALUES (?, ?, ?, ?)
    `;

    await Database.execute(query, [
      new Date(),
      results.anonymizedRecords,
      results.logsDeleted,
      JSON.stringify(results.errors)
    ]);
  }

  /**
   * Agenda execução automática
   */
  scheduleRetention() {
    if (ComplianceConfig.autoRetention.enabled) {
      const cron = require('node-cron');
      
      cron.schedule(ComplianceConfig.autoRetention.schedule, async () => {
        try {
          await this.executeDataRetention();
        } catch (error) {
          logger.error('Erro na execução agendada de retenção:', error);
        }
      });

      logger.info('Agendamento de retenção configurado');
    }
  }
}

module.exports = new DataRetentionService();
```

   * Processa solicitação de acesso a dados
   */
  async handleAccessRequest(req, res) {
    try {
      const { userId, requestId } = req.body;
      
      // Valida prazo de resposta (15 dias LGPD)
      const deadline = new Date();
      deadline.setDate(deadline.getDate() + ComplianceConfig.lgpd.notificationThresholds.subjectRequest);

      // Registra a solicitação
      await this.logSubjectRequest({
        userId,
        requestType: 'ACCESS',
        requestId,
        deadline,
        status: 'PENDING'
      });

      // Coleta todos os dados do usuário
      const userData = await this.collectUserData(userId);

      // Gera relatório
      const report = this.generateDataReport(userData);

      // Atualiza status
      await this.updateRequestStatus(requestId, 'COMPLETED');

      // Registra na auditoria
      await this.auditDataAccess(userId, 'EXPORT', req.user.id);

      res.json({
        success: true,
        message: 'Solicitação de acesso processada',
        deadline: deadline.toISOString(),
        reportAvailable: true,
        downloadUrl: `/api/compliance/data-export/${requestId}`
      });

    } catch (error) {
      logger.error('Erro no processamento de solicitação de acesso:', error);
      res.status(500).json({
        success: false,
        error: 'Erro no processamento da solicitação'
      });
    }
  }

  /**
   * Processa solicitação de exclusão (direito ao esquecimento)
   */
  async handleDeletionRequest(req, res) {
    try {
      const { userId, requestId, reason } = req.body;

      // Verifica se há impedimentos legais
      const legalImpediments = await this.checkLegalImpediments(userId);
      
      if (legalImpediments.length > 0) {
        return res.status(400).json({
          success: false,
          error: 'Impedimentos legais para exclusão',
          impediments: legalImpediments
        });
      }

      // Executa exclusão/anonymização
      await this.executeUserDeletion(userId);

      // Registra a execução
      await this.logDeletionExecution(userId, requestId, reason, req.user.id);

      res.json({
        success: true,
        message: 'Dados excluídos/anonymizados conforme LGPD/GDPR',
        timestamp: new Date().toISOString(),
        referenceId: requestId
      });

    } catch (error) {
      logger.error('Erro no processamento de solicitação de exclusão:', error);
      res.status(500).json({
        success: false,
        error: 'Erro no processamento da solicitação'
      });
    }
  }

  /**
   * Coleta todos os dados de um usuário
   */
  async collectUserData(userId) {
    const queries = {
      userData: `SELECT * FROM users WHERE id = ?`,
      activityLogs: `SELECT * FROM activity_logs WHERE user_id = ?`,
      accessLogs: `SELECT * FROM access_logs WHERE user_id = ?`,
      consentRecords: `SELECT * FROM user_consents WHERE user_id = ?`,
      transactions: `SELECT * FROM transactions WHERE user_id = ?`
    };

    const results = {};

    for (const [key, query] of Object.entries(queries)) {
      results[key] = await Database.execute(query, [userId]);
    }

    return results;
  }

  /**
   * Executa exclusão/anonymização de usuário
   */
  async executeUserDeletion(userId) {
    // Primeiro anonymiza dados pessoais
    await Database.execute(`
      UPDATE users 
      SET 
        email = CONCAT('deleted_', id, '@diyapp.com'),
        name = 'Usuário Excluído',

        cpf = NULL,
        address = NULL,
        deleted_at = NOW(),
        deletion_reason = 'LGPD_RIGHT_TO_BE_FORGOTTEN'
      WHERE id = ?
    `, [userId]);

    // Depois exclui logs associados (após período de retenção)
    const retentionDate = new Date();
    retentionDate.setDate(retentionDate.getDate() - 30);

      `DELETE FROM activity_logs WHERE user_id = ? AND created_at < ?`,

  }

  async checkLegalImpediments(userId) {
    const impediments = [];

    const legalObligations = await Database.execute(`
      SELECT * FROM legal_obligations

      impediments.push('OBRIGAÇÕES_LEGAIS_PENDENTES');
    }

    // Verifica se há transações financeiras em disputa
    const financialDisputes = await Database.execute(`
      SELECT * FROM transactions 
      WHERE user_id = ? AND status IN ('DISPUTED', 'UNDER_REVIEW')
    `, [userId]);

    if (financialDisputes.length > 0) {
      impediments.push('TRANSAÇÕES_FINANCEIRAS_EM_DISPUTA');
    }

    return impediments;
  }
