```javascript
const { Sequelize } = require('sequelize');
const logger = require('../utils/logger');

class Database {
  constructor() {
    this.sequelize = null;
    this.isConnected = false;
  }

  async connect() {
    try {
      this.sequelize = new Sequelize(
        process.env.DB_NAME,
        process.env.DB_USER,
        process.env.DB_PASSWORD,
        {
          host: process.env.DB_HOST,
          port: process.env.DB_PORT,
          dialect: 'postgres',
          logging: process.env.NODE_ENV === 'development' ? 
            (msg) => logger.debug(`SQL: ${msg}`) : false,
          pool: {
            max: 10,
            min: 0,
            acquire: 30000,
            idle: 10000
          },
          retry: {
            match: [
              /SequelizeConnectionError/,
              /SequelizeConnectionRefusedError/,
              /SequelizeHostNotFoundError/,
              /SequelizeHostNotReachableError/,
              /SequelizeInvalidConnectionError/,
              /SequelizeConnectionTimedOutError/
            ],
            max: 3,
            backoffBase: 1000,
            backoffExponent: 1.5
          }
        }
      );

      await this.sequelize.authenticate();
      this.isConnected = true;
      
      logger.info('Database connection established successfully');
      
      // Test connection periodically
      setInterval(async () => {
        try {
          await this.sequelize.query('SELECT 1');
        } catch (error) {
          logger.error('Database connection lost', { error: error.message });
          this.isConnected = false;
          await this.reconnect();
        }
      }, 30000); // Check every 30 seconds
      
      return this.sequelize;
    } catch (error) {
      logger.error('Unable to connect to the database:', {
        error: error.message,
        host: process.env.DB_HOST,
        database: process.env.DB_NAME
      });
      throw error;
    }
  }

  async reconnect() {
    let attempts = 0;
    const maxAttempts = 5;
    
    while (attempts < maxAttempts && !this.isConnected) {
      attempts++;
      logger.info(`Attempting to reconnect to database (attempt ${attempts}/${maxAttempts})`);
      
      try {
        await this.sequelize.authenticate();
        this.isConnected = true;
        logger.info('Database reconnection successful');
        return;
      } catch (error) {
        logger.error(`Reconnection attempt ${attempts} failed:`, error.message);
        
        if (attempts < maxAttempts) {
          const delay = Math.min(1000 * Math.pow(2, attempts), 30000);
          await new Promise(resolve => setTimeout(resolve, delay));
        }
      }
    }
    
    if (!this.isConnected) {
      logger.error('Failed to reconnect to database after maximum attempts');
      process.exit(1);
    }
  }

  async disconnect() {
    if (this.sequelize) {
      await this.sequelize.close();
      this.isConnected = false;
      logger.info('Database connection closed');
    }
  }

  getConnection() {
    if (!this.isConnected) {
      throw new Error('Database not connected');
    }
    return this.sequelize;
  }
}

module.exports = new Database();
```

### src/config/jwt.js
```javascript
const jwt = require('jsonwebtoken');
const crypto = require('crypto');
const logger = require('../utils/logger');

class JWTService {
  constructor() {
    this.secret = process.env.JWT_SECRET;
    this.accessExpiry = process.env.JWT_ACCESS_EXPIRY || '15m';
    this.refreshExpiry = process.env.JWT_REFRESH_EXPIRY || '7d';
    
    if (!this.secret || this.secret === 'your_super_secret_jwt_key_change_in_production') {
      logger.warn('Using default JWT secret. Change JWT_SECRET in production!');
      this.secret = crypto.randomBytes(64).toString('hex');
    }
  }

  generateAccessToken(payload) {
    return jwt.sign(payload, this.secret, {
      expiresIn: this.accessExpiry,
      issuer: 'diyapp-v12',
      audience: 'diyapp-users'
    });
  }

  generateRefreshToken(payload) {
    return jwt.sign(payload, this.secret, {
      expiresIn: this.refreshExpiry,
      issuer: 'diyapp-v12',
      audience: 'diyapp-users'
    });
  }

  verifyToken(token) {
    try {
      return jwt.verify(token, this.secret, {
        issuer: 'diyapp-v12',
        audience: 'diyapp-users'
      });
    } catch (error) {
      logger.error('JWT verification failed:', {
        error: error.message,
        token: this.maskToken(token)
      });
      throw error;
    }
  }

  decodeToken(token) {
    try {
      return jwt.decode(token);
    } catch (error) {
      logger.error('JWT decode failed:', {
        error: error.message,
        token: this.maskToken(token)
      });
      return null;
    }
  }
```

CREATE OR REPLACE PROCEDURE audit.execute_data_retention(
    p_table_name VARCHAR(100),
    p_retention_days INTEGER,
    p_batch_size INTEGER DEFAULT 10000
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_cutoff_date DATE;
    v_deleted_count INTEGER := 0;
    v_total_deleted INTEGER := 0;
BEGIN
    -- Calculate cutoff date
    v_cutoff_date := CURRENT_DATE - p_retention_days;
    
    RAISE NOTICE 'Executing retention for %. Deleting records older than %', 
                 p_table_name, v_cutoff_date;
    
    -- Delete in batches to avoid long locks
    LOOP
        EXECUTE format(
            'DELETE FROM %I WHERE created_at < %L LIMIT %s',
            p_table_name, v_cutoff_date, p_batch_size
        );
        GET DIAGNOSTICS v_deleted_count = ROW_COUNT;
        
        v_total_deleted := v_total_deleted + v_deleted_count;
        COMMIT;
        
        EXIT WHEN v_deleted_count = 0;
        
        -- Small pause between batches
        PERFORM pg_sleep(0.1);
    END LOOP;
    
    -- Log the retention execution
    INSERT INTO audit.data_retention_logs (
        table_name,
        retention_policy,
        records_deleted_count,
        deletion_cutoff_date
    ) VALUES (
        p_table_name,
        format('%s days', p_retention_days),
        v_total_deleted,
        v_cutoff_date
    );
    
    RAISE NOTICE 'Deleted % records from %', v_total_deleted, p_table_name;
END;
$$;

```python
            )
        else:
            self.experiment_id = experiment.experiment_id
    
    def log_model_run(
        self,
        model,
        model_name: str,
        params: Dict[str, Any],
        metrics: Dict[str, float],
        training_data: pd.DataFrame,
        feature_names: List[str],
        run_tags: Optional[Dict[str, str]] = None
    ):
        """Log de uma execução de treinamento de modelo"""
        with mlflow.start_run(
            experiment_id=self.experiment_id,
            run_name=f"{model_name}_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
        ) as run:
            # Log de parâmetros
            mlflow.log_params(params)
            
            # Log de métricas
            mlflow.log_metrics(metrics)
            
            # Log do modelo
            mlflow.sklearn.log_model(
                model,
                "model",
                registered_model_name=model_name
            )
            
            # Log de tags
            if run_tags:
                mlflow.set_tags(run_tags)
            
            # Log de dataset sample
            sample_data = training_data.sample(min(100, len(training_data)))
            mlflow.log_table(sample_data, "training_sample.json")
            
            # Log de feature importance se disponível
            if hasattr(model, 'feature_importances_'):
                feature_importance = dict(zip(feature_names, model.feature_importances_))
                mlflow.log_dict(feature_importance, "feature_importance.json")
            
            return run.info.run_id
    
    def log_production_metrics(
        self,
        model_name: str,
        metrics: Dict[str, float],
        timestamp: datetime,
        metadata: Optional[Dict[str, Any]] = None
    ):
        """Log de métricas de produção"""

                "model": model_name,
                "monitoring": "true"
            })
```

```javascript
    SYSTEM: {

      LOGS_READ: 'system:logs:read',
      METRICS_READ: 'system:metrics:read'
    },

      PROMPT_BASIC: 'llm:prompt:basic',
      PROMPT_ADVANCED: 'llm:prompt:advanced',
      PROMPT_ADMIN: 'llm:prompt:admin',
      MODEL_UPDATE: 'llm:model:update'
    }
  },

  // Mapeamento Role → Permissões
  ROLE_PERMISSIONS: {
    guest: [
      'content:read:public'
    ],

    user: [
      'content:read:public',
      'content:create',
      'content:read:private',
      'content:update:own',
      'content:delete:own',
      'user:read:self',
      'user:update:self',
      'llm:prompt:basic'
    ],

    editor: [
      ...this.user,
      'content:update:any',
      'content:read:private',
      'llm:prompt:advanced'
    ],

    admin: [
      ...this.editor,
      'user:read:any',
      'user:update:any',
      'system:config:read',
      'system:logs:read',
      'system:metrics:read',
      'llm:prompt:admin'
    ],

    super_admin: [
      ...this.admin,
      'user:delete:any',
      'system:config:update',
      'llm:model:update'
    ],

    system: [
      '*'
    ]
  },

  // Validação de permissão
  hasPermission: function(userRole, requiredPermission, resourceOwnerId = null, userId = null) {
    const rolePermissions = this.ROLE_PERMISSIONS[userRole] || [];
    
    // Permissão curinga para system
    if (rolePermissions.includes('*')) {
      return true;
    }

    // Verifica permissão direta
    if (rolePermissions.includes(requiredPermission)) {
      // Verifica ownership para permissões self/own
      if (requiredPermission.includes(':self') || requiredPermission.includes(':own')) {
        return resourceOwnerId === userId;
      }
      return true;
    }

    return false;
  },

  // Middleware para Express
  middleware: function(requiredPermission) {
    return (req, res, next) => {
      const userRole = req.user?.role || 'guest';
      const userId = req.user?.id;
      const resourceOwnerId = req.params.userId || req.body.ownerId;

      if (this.hasPermission(userRole, requiredPermission, resourceOwnerId, userId)) {
        next();
      } else {
        res.status(403).json({
          error: 'Forbidden',
          message: 'Insufficient permissions',
          required: requiredPermission,
          has: this.ROLE_PERMISSIONS[userRole]
        });
      }
    };
  }
};
```

      IP: {
        windowMs: 60 * 1000, // 1 minuto
        maxRequests: 100,
        blockDuration: 300 // 5 minutos se exceder
      },
      USER: {
        windowMs: 60 * 1000,
        maxRequests: 50,
        blockDuration: 600
      },
      ORGANIZATION: {
        windowMs: 5 * 60 * 1000, // 5 minutos
        maxRequests: 1000,
        blockDuration: 3600
      },
      LLM_PROMPT: {
        windowMs: 60 * 1000,
        maxRequests: 10, // Limite baixo para prevenir abuso
        blockDuration: 1800
      }
    };
  }

  // Gera chave única para cada nível
  generateKey(level, identifier) {
    const hash = crypto.createHash('sha256').update(identifier).digest('hex');
    return `${level}:${hash}`;
  }

  // Verifica e incrementa contador
  async checkLimit(level, identifier, endpoint = null) {
    const config = this.configs[level];
    if (!config) throw new Error(`Invalid level: ${level}`);

    const key = this.generateKey(level, identifier);
    const now = Date.now();
    const windowStart = now - config.windowMs;

    // Verifica se está bloqueado
    const blockKey = `${key}:blocked`;
    const isBlocked = await this.redis.get(blockKey);
    if (isBlocked) {
      return {
        allowed: false,
        remaining: 0,
        reset: parseInt(isBlocked),
        retryAfter: Math.ceil((parseInt(isBlocked) - now) / 1000)
      };
    }

    // Remove requisições antigas
    await this.redis.zremrangebyscore(key, 0, windowStart);

    // Conta requisições atuais
    const requestCount = await this.redis.zcard(key);

    if (requestCount >= config.maxRequests) {
      // Bloqueia temporariamente
      const blockUntil = now + (config.blockDuration * 1000);
      await this.redis.setex(blockKey, config.blockDuration, blockUntil);
      
      // Log de bloqueio
      await this.logBlock(level, identifier, endpoint, requestCount);
      
      return {
        allowed: false,
        remaining: 0,
        reset: blockUntil,
        retryAfter: config.blockDuration
      };
    }

    // Adiciona nova requisição
    await this.redis.zadd(key, now, `${now}:${crypto.randomBytes(4).toString('hex')}`);
    await this.redis.expire(key, Math.ceil(config.windowMs / 1000) + 10);

    return {
      allowed: true,
      remaining: config.maxRequests - requestCount - 1,
      reset: now + config.windowMs,
      retryAfter: 0
    };
  }

  // Log de bloqueios para análise
  async logBlock(level, identifier, endpoint, count) {
    const logEntry = {
      timestamp: new Date().toISOString(),
      level,
      identifier: this.maskIdentifier(identifier),
      endpoint,
      requestCount: count,
      action: 'BLOCKED'
    };

    await this.redis.lpush('ratelimit:logs', JSON.stringify(logEntry));
    await this.redis.ltrim('ratelimit:logs', 0, 9999); // Mantém últimos 10k logs
  }

  // Mascara identificadores sensíveis
  maskIdentifier(identifier) {
    if (identifier.includes('@')) {
      // Email: mantém primeiro e último caractere local + domínio
      const [local, domain] = identifier.split('@');
      return `${local[0]}***${local.slice(-1)}@${domain}`;
    }
    if (identifier.match(/^\d+\.\d+\.\d+\.\d+$/)) {
      // IP: mascara últimos octetos
      const parts = identifier.split('.');
      return `${parts[0]}.${parts[1]}.***.***`;
    }
    return `${identifier.slice(0, 3)}***${identifier.slice(-3)}`;
  }

  // Middleware para Express

        }

```javascript
// Helper function para calcular similaridade
function calculateSimilarity(str1, str2) {
  const tokens1 = str1.split(/\s+/);
  const tokens2 = str2.split(/\s+/);
  const intersection = tokens1.filter(token => tokens2.includes(token));
  return intersection.length / Math.max(tokens1.length, tokens2.length);
}
```
