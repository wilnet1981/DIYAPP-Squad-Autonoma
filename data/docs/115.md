### 3.2. CI/CD Pipeline Configuration

**Arquivo: `.github/workflows/ci.yml`**
```yaml
name: CI Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  lint-and-test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [user-service, order-service, payment-service, notification-service]
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Setup Node.js
      uses: actions/setup-node@v3
      with:
        node-version: '18'
    
    - name: Install dependencies
      run: |
        cd services/${{ matrix.service }}
        npm ci
    
    - name: Lint
      run: |
        cd services/${{ matrix.service }}
        npm run lint
    
    - name: Run tests
      run: |
        cd services/${{ matrix.service }}
        npm test -- --coverage
    
    - name: Upload coverage
      uses: codecov/codecov-action@v3
      with:
        file: ./services/${{ matrix.service }}/coverage/lcov.info
  
  build-and-scan:
    runs-on: ubuntu-latest
    needs: lint-and-test
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Build Docker images
      run: |
        docker build -t diyapp/${{ matrix.service }}:${GITHUB_SHA} ./services/${{ matrix.service }}
    
    - name: Trivy vulnerability scanner
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: 'diyapp/${{ matrix.service }}:${GITHUB_SHA}'
        format: 'sarif'
        output: 'trivy-results.sarif'
    
    - name: Upload Trivy scan results
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: 'trivy-results.sarif'
  
  generate-protos:
    runs-on: ubuntu-latest
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Generate gRPC code
      run: |
        ./scripts/generate-protos.sh
    
    - name: Commit generated code
      run: |
        git config --local user.email "action@github.com"
        git config --local user.name "GitHub Action"
        git add -A
        git commit -m "Generated protobuf code" || echo "No changes to commit"
```

    res.end(metrics);

**api-gateway/src/gateway.js**
```javascript
const express = require('express');
const router = express.Router();
const axios = require('axios');
const { CircuitBreaker } = require('circuit-breaker-js');
const NodeCache = require('node-cache');

const authMiddleware = require('./middleware/auth');
const rateLimitMiddleware = require('./middleware/rate-limit');
const validationMiddleware = require('./middleware/validation');
const registry = require('./services/registry');
const logger = require('./utils/logger');
const { GatewayError, ServiceUnavailableError } = require('./utils/errors');

// Cache for service discovery (5 seconds TTL)
const serviceCache = new NodeCache({ stdTTL: 5, checkperiod: 1 });

// Circuit breaker configuration
const circuitBreakers = new Map();

const getCircuitBreaker = (serviceName) => {
  if (!circuitBreakers.has(serviceName)) {
    const breaker = new CircuitBreaker({
      timeoutDuration: 10000, // 10 seconds
      volumeThreshold: 10,
      errorThreshold: 50, // 50% error rate
      onCircuitOpen: (metrics) => {
        logger.warn(`Circuit breaker opened for ${serviceName}`, {
          serviceName,
          metrics,
        });
      },
      onCircuitClose: () => {
        logger.info(`Circuit breaker closed for ${serviceName}`, { serviceName });
      },
    });
    circuitBreakers.set(serviceName, breaker);
  }
  return circuitBreakers.get(serviceName);
};

// Service discovery with caching
const discoverService = async (serviceName) => {
  const cached = serviceCache.get(serviceName);
  if (cached) {
    return cached;
  }

  try {
    const service = await registry.getService(serviceName);
    if (!service) {
      throw new ServiceUnavailableError(`Service ${serviceName} not found`);
    }
    
    serviceCache.set(serviceName, service);
    return service;
  } catch (error) {
    logger.error('Service discovery failed', {
      serviceName,
      error: error.message,
    });
    throw error;
  }
};

// Proxy request to service
const proxyRequest = async (req, res, serviceName, path) => {
  const requestId = req.requestId;
  const startTime = Date.now();
  
  try {
    // Discover service
    const service = await discoverService(serviceName);
    
    // Get circuit breaker
    const breaker = getCircuitBreaker(serviceName);
    
    // Execute with circuit breaker
    const response = await breaker.run(async () => {
      const url = `http://${service.address}:${service.port}${path}`;
      
      const config = {
        method: req.method,
        url,
        headers: {
          ...req.headers,
          'x-request-id': requestId,
          'x-forwarded-for': req.ip,
          'x-original-url': req.originalUrl,
        },
        data: req.body,
        timeout: 10000, // 10 seconds timeout
        maxRedirects: 0,
        validateStatus: () => true, // We'll handle status codes
      };
      
      // Remove hop-by-hop headers
      delete config.headers['host'];
      delete config.headers['connection'];
      delete config.headers['keep-alive'];
      delete config.headers['proxy-authenticate'];
      delete config.headers['proxy-authorization'];
      delete config.headers['te'];
      delete config.headers['trailers'];
      delete config.headers['transfer-encoding'];
      delete config.headers['upgrade'];
      
      logger.debug('Proxying request', {
        requestId,
        serviceName,
        url,
        method: req.method,
      });
      
      return axios(config);
    });
    
    const duration = Date.now() - startTime;
    
    // Log slow requests
    if (duration > 1000) {
      logger.warn('Slow service response', {
        requestId,
        serviceName,
        durationMs: duration,
        statusCode: response.status,
      });
    }
    
    // Forward response
    res.status(response.status);
    
    // Copy headers
    Object.keys(response.headers).forEach(key => {
      // Don't forward hop-by-hop headers
      if (![
        'connection', 'keep-alive', 'proxy-authenticate',
        'proxy-authorization', 'te', 'trailers', 'transfer-encoding', 'upgrade'
      ].includes(key.toLowerCase())) {
        res.set(key, response.headers[key]);
      }
    });
    
    // Send response data
    res.send(response.data);
    
    logger.info('Request proxied successfully', {
      requestId,
      serviceName,
      durationMs: duration,
      statusCode: response.status,
    });
    
  } catch (error) {
    const duration = Date.now() - startTime;
    
    logger.error('Proxy request failed', {
      requestId,
      serviceName,
      error: error.message,
      durationMs: duration,
      stack: error.stack,
    });
    
    if (error.name === 'OpenCircuitError') {
      throw new ServiceUnavailableError(`Service ${serviceName} is temporarily unavailable`);
    }
    
    if (error.code === 'ECONNREFUSED') {
      throw new ServiceUnavailableError(`Cannot connect to service ${serviceName}`);
    }
    
    if (error.code === 'ETIMEDOUT') {
      throw new ServiceUnavailableError(`Service ${serviceName} timeout`);
    }
    
    throw new GatewayError(`Failed to proxy request to ${serviceName}`);
  }
};

// Dynamic route handler
const createServiceRoute = (serviceName, path = '') => {
  return async (req, res, next) => {
    try {
      const servicePath = req.params[0] || '';
      const fullPath = path + servicePath;
      await proxyRequest(req, res, serviceName, fullPath);
    } catch (error) {
      next(error);
    }
  };
};

    this.state = { hasError: false }
  }

    return { hasError: true }
  }

  }

  render() {
    if (this.state.hasError) {

        <div style={{ padding: '2rem', textAlign: 'center' }}>
          <h1>Something went wrong</h1>
          <p>Please refresh the page or contact support.</p>
        </div>
      )
    }

    return this.props.children
  }

"""
Gerenciador de senhas com Argon2
Implementa hash seguro com parГўmetros ajustГЎveis
"""
import argon2
import argon2.exceptions
import secrets
import base64
import json
from typing import Tuple, Optional
import logging

logger = logging.getLogger(__name__)

class PasswordManager:
    """
    Gerencia hash e verificaГ§ГЈo de senhas usando Argon2id
    ParГўmetros seguros conforme OWASP 2024
    """
    
    def __init__(self, 
                 time_cost: int = 2,
                 memory_cost: int = 65536,
                 parallelism: int = 4,
                 hash_len: int = 32,
                 salt_len: int = 16):
        """
        Inicializa Argon2 com parГўmetros seguros
        
        Args:
            time_cost: NГәmero de iteraГ§Гөes (1-3)
            memory_cost: MemГіria em KiB (64MB padrГЈo)
            parallelism: Grau de paralelismo
            hash_len: Tamanho do hash em bytes
            salt_len: Tamanho do salt em bytes
        """
        self.time_cost = max(1, min(time_cost, 3))
        self.memory_cost = max(32768, memory_cost)  # MГӯnimo 32MB
        self.parallelism = max(1, parallelism)
        self.hash_len = max(16, hash_len)
        self.salt_len = max(12, salt_len)
        
        self.hasher = argon2.PasswordHasher(
            time_cost=self.time_cost,
            memory_cost=self.memory_cost,
            parallelism=self.parallelism,
            hash_len=self.hash_len,
            salt_len=self.salt_len,
            type=argon2.Type.ID
        )
        
        logger.info(f"PasswordManager inicializado: time_cost={self.time_cost}, "
                   f"memory_cost={self.memory_cost}KiB")
    
    def hash_password(self, password: str) -> str:
        """
        Gera hash seguro da senha
        
        Args:
            password: Senha em texto claro
            
        Returns:
            Hash Argon2id codificado em string
            
        Raises:
            ValueError: Se senha for vazia ou muito curta
        """
        if not password or len(password) < 8:
            raise ValueError("Senha deve ter pelo menos 8 caracteres")
        
        try:
            # Verifica se senha Г© muito longa (prevenГ§ГЈo DoS)
            if len(password) > 1024:
                raise ValueError("Senha muito longa (max 1024 caracteres)")
            
            hash_result = self.hasher.hash(password)
            logger.debug("Hash de senha gerado com sucesso")
            return hash_result
            
        except Exception as e:
            logger.error(f"Erro ao gerar hash: {str(e)}")
            raise
    
    def verify_password(self, password: str, hashed_password: str) -> bool:
        """
        Verifica se senha corresponde ao hash
        
        Args:
            password: Senha em texto claro para verificar
            hashed_password: Hash armazenado
            
        Returns:
            True se senha vГЎlida, False caso contrГЎrio
        """
        try:
            return self.hasher.verify(hashed_password, password)
        except argon2.exceptions.VerifyMismatchError:
            logger.warning("Tentativa de login com senha invГЎlida")
            return False
        except argon2.exceptions.VerificationError as e:
            logger.error(f"Erro na verificaГ§ГЈo de hash: {str(e)}")
            return False
        except Exception as e:
            logger.error(f"Erro inesperado na verificaГ§ГЈo: {str(e)}")
            return False
    
    def needs_rehash(self, hashed_password: str) -> bool:
        """
        Verifica se hash precisa ser atualizado (parГўmetros mudaram)
        
        Args:
            hashed_password: Hash para verificar
            
        Returns:
            True se precisa rehash
        """
        try:
            return self.hasher.check_needs_rehash(hashed_password)
        except Exception as e:
            logger.error(f"Erro ao verificar rehash: {str(e)}")
            return True
    
    def generate_salt(self) -> str:
        """
        Gera salt criptograficamente seguro
        
        Returns:
            Salt em base64
        """
        salt = secrets.token_bytes(self.salt_len)
        return base64.b64encode(salt).decode('utf-8')
    
    def get_parameters(self) -> dict:
        """
        Retorna parГўmetros atuais para auditoria
        
        Returns:
            DicionГЎrio com parГўmetros de configuraГ§ГЈo
        """
        return {
            "algorithm": "Argon2id",
            "time_cost": self.time_cost,
            "memory_cost_kib": self.memory_cost,
            "parallelism": self.parallelism,
            "hash_length": self.hash_len,
            "salt_length": self.salt_len,
            "version": argon2.__version__
        }

            pattern: PadrГЈo regex para validaГ§ГЈo
            
        Returns:
            String sanitizada
            
        Raises:
            ValueError: Se validaГ§ГЈo falhar em strict_mode
        """
        if not isinstance(value, str):
            if self.strict_mode:
                raise ValueError(f"Valor deve ser string, recebido {type(value)}")
            value = str(value)
        
        # Remove espaГ§os em branco no inГӯcio/fim
        value = value.strip()
        
        # Trunca se muito longo
        if len(value) > max_length:
            if self.strict_mode:
                raise ValueError(f"String muito longa (max {max_length} caracteres)")
            value = value[:max_length]
            logger.warning(f"String truncada para {max_length} caracteres")
        
        # ValidaГ§ГЈo por regex
        if pattern:
            if pattern in self.PATTERNS:
                regex_pattern = self.PATTERNS[pattern]
            else:
                regex_pattern = pattern
            
            if not re.match(regex_pattern, value):
                if self.strict_mode:
                    raise ValueError(f"Valor nГЈo corresponde ao padrГЈo: {pattern}")
                # Tenta corrigir removendo caracteres invГЎlidos
                if pattern == 'no_sql_injection':
                    value = re.sub(r'[;\'\"\\-]', '', value)
                elif pattern == 'no_xss':
                    value = re.sub(r'[<>\"\']', '', value)
        
        # SanitizaГ§ГЈo HTML
        if allow_html:
            value = self.bleach_sanitizer.clean(value)
        else:
            # Escape HTML se nГЈo permitido
            value = html.escape(value)
        
        return value
    
    def sanitize_number(self, value: Any, 
                       min_value: float = None,
                       max_value: float = None,
                       integer_only: bool = False) -> Union[int, float]:
        """
        Sanitiza nГәmero
        
        Args:
            value: Valor numГ©rico
            min_value: Valor mГӯnimo permitido
            max_value: Valor mГЎximo permitido
            integer_only: Se True, converte para inteiro
            
        Returns:
            NГәmero sanitizado
        """
        try:
            if integer_only:
                num = int(value)
            else:
                num = float(value)
        except (ValueError, TypeError):
            if self.strict_mode:
                raise ValueError(f"Valor nГЈo Г© um nГәmero vГЎlido: {value}")
            return 0 if integer_only else 0.0
        
        # Valida limites
        if min_value is not None and num < min_value:
            if self.strict_mode:
                raise ValueError(f"Valor menor que mГӯnimo permitido ({min_value})")
            num = min_value
        
        if max_value is not None and num > max_value:
            if self.strict_mode:
                raise ValueError(f"Valor maior que mГЎximo permitido ({max_value})")
            num = max_value
        
        return num
    
    def sanitize_email(self, email: str) -> str:
        """
        Sanitiza e valida email
        
        Args:
            email: EndereГ§o de email
            
        Returns:
            Email sanitizado em lowercase
        """
        email = self.sanitize_string(email, max_length=254, pattern='email')
        email = email.lower()
        
        # ValidaГ§ГЈo adicional de domГӯnio
        parts = email.split('@')
        if len(parts) != 2:
            raise ValueError("Formato de email invГЎlido")
        
        domain = parts[1]
        if domain.startswith('.') or domain.endswith('.'):
            raise ValueError("DomГӯnio de email invГЎlido")
        
        return email
    
    def sanitize_url(self, url: str, 
                    allowed_schemes: list = None,
                    require_https: bool = True) -> str:
        """
        Sanitiza URL
        
        Args:
            url: URL para sanitizar
            allowed_schemes: Lista de esquemas permitidos
            require_https: Se True, forГ§a HTTPS em produГ§ГЈo
            
        Returns:
            URL sanitizada
        """
        if not url:
            return ""
        
        # Remove espaГ§os e caracteres invГЎlidos
        url = url.strip()
        
        # Adiciona esquema se nГЈo presente
        if not re.match(r'^[a-zA-Z][a-zA-Z0-9+\-.]*://', url):
            url = 'https://' + url if require_https else 'http://' + url
        
        # Valida esquema
        if allowed_schemes is None:
            allowed_schemes = ['http', 'https', 'ftp', 'mailto']
        
        scheme = url.split('://')[0].lower()
        if scheme not in allowed_schemes:
            if self.strict_mode:
                raise ValueError(f"Esquema nГЈo permitido: {scheme}")
            url = 'https://' + url.split('://')[1] if require_https else url
        
        # URL encode
        url = quote(url, safe=':/?#[]@!$&\'()*+,;=')
        
        return url
    
    def sanitize_json(self, json_str: str, 
                     schema: dict = None,
                     max_depth: int = 10) -> dict:
        """
        Sanitiza JSON string
        
        Args:
            json_str: String JSON
            schema: Schema para validaГ§ГЈo
            max_depth: Profundidade mГЎxima permitida
            
        Returns:
            DicionГЎrio Python sanitizado
        """
        try:
            data = json.loads(json_str)
        except json.JSONDecodeError as e:
            if self.strict_mode:
                raise ValueError(f"JSON invГЎlido: {str(e)}")
            return {}
        
        # Valida profundidade
        if self._get_depth(data) > max_depth:
            if self.strict_mode:
                raise ValueError(f"JSON muito profundo (max {max_depth} nГӯveis)")
            data = self._flatten_json(data, max_depth)
        
        # Sanitiza recursivamente
        return self._sanitize_json_recursive(data, schema)
    
    def _sanitize_json_recursive(self, data: Any, schema: dict = None) -> Any:
        """Sanitiza JSON recursivamente"""
        if isinstance(data, dict):
            result = {}
            for key, value in data.items():
                # Sanitiza chave
                safe_key = self.sanitize_string(str(key), max_length=100, pattern='no_xss')
                
                # Sanitiza valor
                if isinstance(value, (dict, list)):
                    safe_value = self._sanitize_json_recursive(value, schema)
                elif isinstance(value, str):
                    safe_value = self.sanitize_string(value, max_length=10000)
                else:
                    safe_value = value
                
                result[safe_key] = safe_value
            return result
        
        elif isinstance(data, list):
            return [self._sanitize_json_recursive(item, schema) for item in data]
        
        elif isinstance(data, str):
            return self.sanitize_string(data, max_length=10000)
        
        return data
    
    def _get_depth(self, data: Any, depth: int = 0) -> int:
        """Calcula profundidade do JSON"""
        if isinstance(data, dict):
            return max((self._get_depth(v, depth + 1) for v in data.values()), default=depth + 1)
        elif isinstance(data, list):
            return max((self._get_depth(v, depth + 1) for v in data), default=depth + 1)
        return depth
    
    def _flatten_json(self, data: Any, max_depth: int) -> Any:
        """Achata JSON se muito profundo"""
        if max_depth <= 0:
            return "[TRUNCATED]"
        
        if isinstance(data, dict):
            return {k: self._flatten_json(v, max_depth - 1) for k, v in data.items()}
        elif isinstance(data, list):
            return [self._flatten_json(v, max_depth - 1) for v in data]
        return data

                      validators: list = None,
                      custom_validator: Callable = None) -> bool:
        """
        Valida input com mГәltiplos validadores
        
        Args:
            value: Valor para validar
            validators: Lista de validadores (email, phone, etc.)

        if validators:
            for validator in validators:
                if validator in self.PATTERNS:
                    if not re.match(self.PATTERNS[validator], str(value)):
                        return False
        
        if custom_validator:
            try:
                return custom_validator(value)
            except Exception:
                return False
