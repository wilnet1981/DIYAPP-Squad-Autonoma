const { initializeDatabase } = require('./config/database');

class APIServer {
  constructor() {
    this.app = express();
    this.port = process.env.PORT || 3000;
    this.graphQLPath = '/graphql';
    this.restV1Path = '/api/v1';
    this.restV2Path = '/api/v2';
    this.docsPath = '/api-docs';
    
    this.initializeMiddleware();
    this.initializeServices();
    this.initializeRoutes();
    this.initializeErrorHandling();
  }

  async initializeMiddleware() {
    // Security headers
    this.app.use(helmet({
      contentSecurityPolicy: {
        directives: {
          defaultSrc: ["'self'"],
          styleSrc: ["'self'", "'unsafe-inline'"],
          scriptSrc: ["'self'"],
        },
      },
    }));

    // CORS configuration
    this.app.use(cors({
      origin: process.env.ALLOWED_ORIGINS ? process.env.ALLOWED_ORIGINS.split(',') : ['http://localhost:3000'],
      credentials: true,
      maxAge: 86400, // 24 hours
    }));

    // Compression
    this.app.use(compression());

    // Request logging
    this.app.use((req, res, next) => {
      const start = Date.now();
      const correlationId = req.headers['x-correlation-id'] || require('uuid').v4();
      
      res.setHeader('X-Correlation-ID', correlationId);
      
      res.on('finish', () => {
        const duration = Date.now() - start;
        logger.info({
          correlation_id: correlationId,
          method: req.method,
          url: req.url,
          status: res.statusCode,
          duration_ms: duration,
          user_agent: req.headers['user-agent'],
          ip: req.ip,
        });
        
        // Record metrics
        metrics.httpRequestDuration.observe(
          { method: req.method, route: req.route?.path || req.url, status_code: res.statusCode },
          duration / 1000
        );
      });
      
      next();
    });

    // Body parsing
    this.app.use(express.json({ limit: '10mb' }));
    this.app.use(express.urlencoded({ extended: true, limit: '10mb' }));
  }

  async initializeServices() {
    try {
      // Initialize Redis cache
      await initializeCache();
      logger.info('Redis cache initialized successfully');

      // Initialize database
      await initializeDatabase();
      logger.info('Database initialized successfully');

      // Initialize metrics endpoint
      this.app.get('/metrics', async (req, res) => {
        res.set('Content-Type', metrics.register.contentType);
        res.end(await metrics.register.metrics());
      });

    } catch (error) {
      logger.error('Failed to initialize services:', error);
      process.exit(1);
    }
  }

  async initializeRoutes() {
    // Health check endpoint
    this.app.get('/health', (req, res) => {
      res.status(200).json({
        status: 'healthy',
        timestamp: new Date().toISOString(),
        uptime: process.uptime(),
        version: '13.0.0',
      });
    });

    // API Documentation
    this.app.use(this.docsPath, swaggerUi.serve, swaggerUi.setup(openapiSpec));

    // REST API v1 (legacy)
    this.app.use(this.restV1Path, setupREST());

    // REST API v2 (current)
    this.app.use(this.restV2Path, setupRESTv2());

    // GraphQL API
    const apolloServer = await setupGraphQL();
    await apolloServer.start();
    apolloServer.applyMiddleware({ 
      app: this.app, 
      path: this.graphQLPath,
      cors: false // CORS already handled globally
    });

    // 404 handler
    this.app.use('*', (req, res) => {
      res.status(404).json({
        error: 'Not Found',
        message: `Cannot ${req.method} ${req.originalUrl}`,
        correlation_id: res.getHeader('X-Correlation-ID'),
      });
    });
  }

  initializeErrorHandling() {
    // Global error handler
    this.app.use((error, req, res, next) => {
      const correlationId = res.getHeader('X-Correlation-ID');
      
      logger.error({
        correlation_id: correlationId,
        error: error.message,
        stack: error.stack,
        url: req.url,
        method: req.method,
        user_id: req.user?.id,
      });

      // Don't expose internal errors to client
      const clientError = {
        error: 'Internal Server Error',
        message: 'An unexpected error occurred',
        correlation_id: correlationId,
      };

      // For validation errors, provide more details
      if (error.isJoi) {
        clientError.error = 'Validation Error';
        clientError.message = error.details.map(detail => detail.message);
        return res.status(422).json(clientError);
      }

      // For authentication errors
      if (error.name === 'JsonWebTokenError' || error.name === 'TokenExpiredError') {
        clientError.error = 'Authentication Error';
        clientError.message = 'Invalid or expired token';
        return res.status(401).json(clientError);
      }

      // For known business errors
      if (error.statusCode && error.statusCode < 500) {
        clientError.error = error.name || 'Client Error';
        clientError.message = error.message;
        return res.status(error.statusCode).json(clientError);
      }

      // Default to 500
      res.status(500).json(clientError);
    });
  }

  start() {
    return new Promise((resolve) => {
      this.server = this.app.listen(this.port, () => {
        logger.info(`Server running on port ${this.port}`);

        logger.info(`Metrics: http://localhost:${this.port}/metrics`);
        resolve();
      });
    });

      await new Promise((resolve) => {
        this.server.close(resolve);
      });
      logger.info('Server stopped');
    }
  }

const Redis = require('ioredis');
const logger = require('../../../utils/logger');

class RedisCache {
  constructor() {
    this.client = null;
    this.circuitBreaker = new circuitBreaker.CircuitBreaker({
      failureThreshold: 5,
      resetTimeout: 60000, // 1 minute
      timeout: 5000, // 5 seconds
    });
    this.localCache = new Map(); // Fallback local cache
    this.localCacheTTL = new Map();
  }

  async initialize() {
    try {
      this.client = new Redis({
        host: process.env.REDIS_HOST || 'localhost',
        port: process.env.REDIS_PORT || 6379,
        password: process.env.REDIS_PASSWORD,
        retryStrategy: (times) => {
          const delay = Math.min(times * 100, 3000);
          return delay;
        },
        maxRetriesPerRequest: 3,
        enableReadyCheck: true,
        connectTimeout: 10000,
      });

      this.client.on('error', (error) => {
        logger.error('Redis connection error:', error);
      });

      this.client.on('connect', () => {
        logger.info('Redis connected successfully');
      });

      await this.client.ping();
      return this;
    } catch (error) {
      logger.error('Failed to initialize Redis:', error);
      // Fallback to local cache
      logger.warn('Falling back to local cache');
      return this;
    }
  }

  async get(key) {
    return this.circuitBreaker.execute(async () => {
      if (!this.client || !this.client.status === 'ready') {
        // Fallback to local cache
        const cached = this.localCache.get(key);
        const ttl = this.localCacheTTL.get(key);
        
        if (cached && ttl && Date.now() < ttl) {
          logger.debug(`Cache hit (local): ${key}`);
          return JSON.parse(cached);
        }
        
        this.localCache.delete(key);
        this.localCacheTTL.delete(key);
        return null;
      }

      const value = await this.client.get(key);
      if (value) {
        logger.debug(`Cache hit (Redis): ${key}`);
        return JSON.parse(value);
      }
      return null;
    }).catch((error) => {
      logger.warn(`Cache get failed for key ${key}:`, error);
      return null;
    });
  }

  async set(key, value, ttlSeconds = 300) {
    return this.circuitBreaker.execute(async () => {
      const stringValue = JSON.stringify(value);
      
      // Always update local cache as fallback
      this.localCache.set(key, stringValue);
      this.localCacheTTL.set(key, Date.now() + (ttlSeconds * 1000));

      if (!this.client || this.client.status !== 'ready') {
        return;
      }

      if (ttlSeconds > 0) {
        await this.client.setex(key, ttlSeconds, stringValue);
      } else {
        await this.client.set(key, stringValue);
      }
      
      logger.debug(`Cache set: ${key} (TTL: ${ttlSeconds}s)`);
    }).catch((error) => {
      logger.warn(`Cache set failed for key ${key}:`, error);
    });
  }

  async delete(key) {
    return this.circuitBreaker.execute(async () => {
      this.localCache.delete(key);
      this.localCacheTTL.delete(key);

      if (!this.client || this.client.status !== 'ready') {
        return;
      }

      await this.client.del(key);
      logger.debug(`Cache deleted: ${key}`);
    }).catch((error) => {
      logger.warn(`Cache delete failed for key ${key}:`, error);
    });
  }

  async invalidatePattern(pattern) {
    return this.circuitBreaker.execute(async () => {

-- ==================== VIEW: stg_user_behavior (Silver Layer) ================
CREATE OR REPLACE VIEW stg_user_behavior AS
SELECT 
    -- Identificação
    ui.interaction_id,
    ui.user_id,
    ui.project_id,
    
    -- Dimensões
    ui.interaction_type,
    ui.rating_value,
    ui.device_type,
    
    -- Datas
    DATE(ui.created_at) AS interaction_date,
    EXTRACT(HOUR FROM ui.created_at) AS interaction_hour,
    ui.created_at AS interaction_timestamp,
    
    -- Contexto do projeto
    p.category_id,
    p.difficulty_level,
    p.estimated_hours,
    
    -- Contexto do usuário
    u.created_at AS user_created_date,
    DATE_PART('day', ui.created_at - u.created_at) AS user_tenure_days,
    
    -- Metadados
    ui.session_id
    
FROM diyapp_core.user_interactions ui
JOIN diyapp_core.projects p ON ui.project_id = p.project_id
JOIN diyapp_core.users u ON ui.user_id = u.user_id
WHERE p.status = 'published'
  AND u.is_blocked = FALSE;

-- ==================== VIEW: stg_project_performance (Silver Layer) ==========
CREATE OR REPLACE VIEW stg_project_performance AS
SELECT 
    -- Identificação
    p.project_id,
    p.user_id,
    p.category_id,
    
    -- Datas
    DATE(p.created_at) AS project_created_date,
    DATE(p.published_at) AS project_published_date,
    DATE(p.last_interaction_at) AS last_interaction_date,
    
    -- Métricas de engajamento
    p.view_count,
    p.like_count,
    p.save_count,
    p.comment_count,
    
    -- Taxas calculadas
    CASE 
        WHEN p.view_count > 0 
        THEN ROUND((p.like_count::DECIMAL / p.view_count) * 100, 2)
        ELSE 0 
    END AS like_rate_percent,
    
    CASE 
        WHEN p.view_count > 0 
        THEN ROUND((p.save_count::DECIMAL / p.view_count) * 100, 2)
        ELSE 0 
    END AS save_rate_percent,
    
    -- Dados do projeto
    p.difficulty_level,
    p.estimated_hours,
    p.estimated_cost,
    
    -- Status
    p.status,
    
    -- Timestamps
    p.created_at,
    p.updated_at,
    p.published_at
    
FROM diyapp_core.projects p
WHERE p.status IN ('published', 'archived');

-- Função para atualizar estatísticas denormalizadas
CREATE OR REPLACE FUNCTION update_project_stats()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' OR TG_OP = 'UPDATE' THEN
        -- Atualiza contagens no projeto
        UPDATE diyapp_core.projects p
        SET 
            view_count = (
                SELECT COUNT(*) 
                FROM diyapp_core.user_interactions 
                WHERE project_id = NEW.project_id 
                AND interaction_type = 'view'
            ),
            like_count = (
                SELECT COUNT(*) 
                FROM diyapp_core.user_interactions 
                WHERE project_id = NEW.project_id 
                AND interaction_type = 'like'
            ),
            save_count = (
                SELECT COUNT(*) 
                FROM diyapp_core.user_interactions 
                WHERE project_id = NEW.project_id 
                AND interaction_type = 'save'
            ),
            comment_count = (
                SELECT COUNT(*) 
                FROM diyapp_core.user_interactions 
                WHERE project_id = NEW.project_id 
                AND interaction_type = 'comment'
            ),
            last_interaction_at = NOW()
        WHERE p.project_id = NEW.project_id;
        
        -- Atualiza contagem de projetos do usuário
        UPDATE diyapp_core.users u
        SET project_count = (
            SELECT COUNT(*) 
            FROM diyapp_core.projects 
            WHERE user_id = NEW.user_id 
            AND status = 'published'
        )
        WHERE u.user_id = NEW.user_id;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger para atualizar estatísticas
DROP TRIGGER IF EXISTS trg_update_project_stats ON diyapp_core.user_interactions;
CREATE TRIGGER trg_update_project_stats
AFTER INSERT OR UPDATE ON diyapp_core.user_interactions
FOR EACH ROW
EXECUTE FUNCTION update_project_stats();

-- Função para limpar dados antigos (retention policy)
CREATE OR REPLACE PROCEDURE cleanup_old_interactions(
    retention_days INTEGER DEFAULT 365
)
LANGUAGE plpgsql
AS $$
BEGIN
    -- Arquiva interações antigas
    INSERT INTO diyapp_analytics.raw_events (
        event_type,
        event_data,
        event_timestamp,
        source_system,
        source_table,
        source_id
    )
    SELECT 
        'user_interaction_archived',
        jsonb_build_object(
            'interaction_id', interaction_id,
            'user_id', user_id,
            'project_id', project_id,
            'interaction_type', interaction_type,
            'rating_value', rating_value
        ),
        created_at,
        'diyapp_core',
        'user_interactions',
        interaction_id::TEXT
    FROM diyapp_core.user_interactions
    WHERE created_at < NOW() - (retention_days || ' days')::INTERVAL;
    
    --
