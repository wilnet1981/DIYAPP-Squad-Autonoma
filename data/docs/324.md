# DIYAPP Evolution - V13 Core - Arquitetura Técnica

## ADR-001: Arquitetura V13 - Padrões de Código e Estrutura

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

### CONTEXTO:
A V13 do DIYAPP precisa estabelecer uma arquitetura sólida que suporte:
1. Evolução contínua com múltiplas squads trabalhando em paralelo (Modo Hive)
2. Estabilidade 100% autônoma com auto-recuperação
3. Performance garantida (<200ms P95 para todas as APIs)
4. Manutenibilidade com padrões claros e documentados

### DECISÃO:
Adotar arquitetura em camadas com separação clara de responsabilidades, utilizando:
- **Estrutura hexagonal adaptada** para isolamento de domínio
- **Injeção de dependência** para testabilidade
- **Circuit Breaker pattern** para resiliência
- **Structured logging** com contexto de transação
- **Health checks** hierárquicos para monitoramento autônomo

### OPÇÕES CONSIDERADAS:
- **Opção A**: Arquitetura monolítica tradicional com controllers/services
  - Prós: Simples, baixa latência inicial
  - Contras: Acoplamento alto, difícil evolução paralela
- **Opção B**: Microserviços completos
  - Prós: Isolamento total, deploy independente
  - Contras: Overhead de rede, complexidade operacional
- **Opção C**: Arquitetura hexagonal com módulos independentes
  - Prós: Balance entre isolamento e performance, fácil refatoração paralela
  - Contras: Curva de aprendizado inicial

**Opção escolhida: C** - Justificativa: Permite trabalho em paralelo (Modo Hive) mantendo performance e facilitando migração futura para microserviços se necessário.

### CONSEQUÊNCIAS:
**Positivas:**
- Módulos podem evoluir independentemente
- Testes unitários mais fáceis
- Onboarding mais rápido para novas squads
- Refatoração paralela segura

**Negativas:**
- Overhead inicial de abstrações
- Necessidade de disciplina rigorosa nos contratos de interface

**Riscos:**
- Degradação de performance se injeção de dependência for mal implementada
- Mitigação: Benchmarking contínuo e code review rigoroso

export const errorHandler: ErrorRequestHandler = (err, req, res, _next) => {

  if (err instanceof AppError) {
    logger.warn('Handled application error', {
      code: err.code,
      context: err.context,
      path: req.path
    });
    
    return res.status(err.httpStatus).json({

        message: err.message,
        timestamp: err.timestamp.toISOString()

      code: 'INTERNAL_SERVER_ERROR',
      message: 'An unexpected error occurred',
      timestamp: new Date().toISOString()
    }
  });

// shared/logger/structured-logger.ts
export interface LogEntry {
  timestamp: string;
  level: 'debug' | 'info' | 'warn' | 'error' | 'fatal';
  message: string;
  context: Record<string, unknown>;
  transactionId?: string;
  userId?: string;
  durationMs?: number;
}

export class StructuredLogger implements ILogger {
  private readonly transactionId: string;
  
  constructor(private readonly baseLogger: winston.Logger, transactionId?: string) {
    this.transactionId = transactionId || crypto.randomUUID();
  }
  
  info(message: string, context: Record<string, unknown> = {}): void {
    this.baseLogger.info(message, {
      ...context,
      transactionId: this.transactionId,
      timestamp: new Date().toISOString()
    });
  }
  
  error(message: string, error?: Error, context: Record<string, unknown> = {}): void {
    this.baseLogger.error(message, {
      ...context,
      transactionId: this.transactionId,
      error: error?.message,
      stack: error?.stack,
      timestamp: new Date().toISOString()
    });
  }
  
  // Métodos para medir performance
  async measure<T>(
    operation: string,
    fn: () => Promise<T>,
    context: Record<string, unknown> = {}
  ): Promise<T> {
    const start = Date.now();
    this.info(`Starting ${operation}`, context);
    
    try {
      const result = await fn();
      const duration = Date.now() - start;
      
      this.info(`Completed ${operation}`, {
        ...context,
        durationMs: duration,
        success: true
      });
      
      // Alertar se operação exceder threshold
      if (duration > 200) {
        this.warn(`Operation ${operation} exceeded 200ms threshold`, {
          ...context,
          durationMs: duration
        });
      }
      
      return result;
    } catch (error) {
      const duration = Date.now() - start;
      this.error(`Failed ${operation}`, error as Error, {
        ...context,
        durationMs: duration,
        success: false
      });
      throw error;
    }
  }
}

// middleware/request-logger.ts
export const requestLogger = (logger: ILogger): RequestHandler => {
  return (req, res, next) => {
    const transactionId = req.headers['x-transaction-id'] as string || crypto.randomUUID();
    const requestLogger = new StructuredLogger(logger['baseLogger'], transactionId);
    
    // Injetar logger no request
    req.container.register('logger', requestLogger);
    
    // Log de request
    requestLogger.info('Incoming request', {
      method: req.method,
      path: req.path,
      ip: req.ip,
      userAgent: req.get('user-agent')
    });
    
    // Hook para log de response
    const originalSend = res.send;
    const startTime = Date.now();
    
    res.send = function(body: unknown) {
      const duration = Date.now() - startTime;
      
      requestLogger.info('Request completed', {
        method: req.method,
        path: req.path,
        statusCode: res.statusCode,
        durationMs: duration,
        responseSize: typeof body === 'string' ? body.length : JSON.stringify(body).length
      });
      
      // Alertar se response time > 200ms
      if (duration > 200) {
        requestLogger.warn('Slow request detected', {
          method: req.method,
          path: req.path,
          durationMs: duration,
          threshold: 200
        });
      }
      
      return originalSend.call(this, body);
    };
    
    next();
  };
};

// infrastructure/health/health-checker.ts
export interface HealthCheckResult {
  name: string;
  status: 'healthy' | 'unhealthy' | 'degraded';
  duration: number;
  details?: Record<string, unknown>;
  timestamp: string;
}

export interface HealthCheck {
  name: string;
  critical: boolean;
  check(): Promise<HealthCheckResult>;
}

export class HealthChecker {
  private checks: HealthCheck[] = [];
  
  register(check: HealthCheck): void {
    this.checks.push(check);
  }
  
  async checkAll(): Promise<{
    status: 'healthy' | 'unhealthy' | 'degraded';
    checks: HealthCheckResult[];
    timestamp: string;
    duration: number;
  }> {
    const startTime = Date.now();
    const results = await Promise.all(
      this.checks.map(async (check) => {
        const checkStart = Date.now();
        try {
          const result = await check.check();
          return {
            ...result,
            duration: Date.now() - checkStart
          };
        } catch (error) {
          return {
            name: check.name,
            status: 'unhealthy' as const,
            duration: Date.now() - checkStart,
            details: { error: error.message },
            timestamp: new Date().toISOString()
          };
        }
      })
    );
    
    const overallDuration = Date.now() - startTime;
    const criticalUnhealthy = results.some(
      r => r.status === 'unhealthy' && 
      this.checks.find(c => c.name === r.name)?.critical
    );
    
    const anyUnhealthy = results.some(r => r.status === 'unhealthy');
    
    return {
      status: criticalUnhealthy ? 'unhealthy' : anyUnhealthy ? 'degraded' : 'healthy',
      checks: results,
      timestamp: new Date().toISOString(),
      duration: overallDuration
    };
  }
}

  }

  name = 'redis';
  critical = false; // Cache não é crítico
  
  constructor(private redisClient: RedisClient) {}
  
  async check(): Promise<HealthCheckResult> {
    const start = Date.now();
    try {
      await this.redisClient.ping();
      return {
        name: this.name,
        status: 'healthy',
        duration: Date.now() - start,
        timestamp: new Date().toISOString()
      };
    } catch (error) {
      return {
        name: this.name,
        status: 'unhealthy',
        duration: Date.now() - start,
        details: { error: error.message },
        timestamp: new Date().toISOString()
      };
    }
  }
}

// Controller de health check
export class HealthController {
  constructor(private healthChecker: HealthChecker) {}
  
  async getHealth(req: Request, res: Response): Promise<void> {
    const health = await this.healthChecker.checkAll();
    
    const statusCode = health.status === 'healthy' ? 200 :
                      health.status === 'degraded' ? 207 : 503;
    
    res.status(statusCode).json({
      status: health.status,
      timestamp: health.timestamp,
      duration: health.duration,
      checks: health.checks.reduce((acc, check) => {
        acc[check.name] = {
          status: check.status,
          duration: check.duration,
          ...check.details
        };
        return acc;
      }, {} as Record<string, unknown>)
    });
  }
}

export class PerformanceMetrics {
  private static instance: PerformanceMetrics;
  private metrics: Map<string, number[]> = new Map();
  
  static getInstance(): PerformanceMetrics {
    if (!PerformanceMetrics.instance) {
      PerformanceMetrics.instance = new PerformanceMetrics();
    }
    return PerformanceMetrics.instance;
  }
  
  record(operation: string, durationMs: number): void {
    if (!this.metrics.has(operation)) {
      this.metrics.set(operation, []);
    }
    
    const values = this.metrics.get(operation)!;
    values.push(durationMs);
    
    // Manter apenas últimas 1000 medições
    if (values.length > 1000) {
      values.shift();
    }
  }
  
  getPercentile(operation: string, percentile: number): number {
    const values = this.metrics.get(operation);
    if (!values || values.length === 0) return 0;
    
    const sorted = [...values].sort((a, b) => a - b);
    const index = Math.ceil((percentile / 100) * sorted.length) - 1;
    return sorted[Math.max(0, index)];
  }
  
  getStats(operation: string): {
    p50: number;
    p95: number;
    p99: number;
    avg: number;
    min: number;
    max: number;
    count: number;
  } {
    const values = this.metrics.get(operation) || [];
    if (values.length === 0) {
      return { p50: 0, p95: 0, p99: 0, avg: 0, min: 0, max: 0, count: 0 };
    }
    
    const sorted = [...values].sort((a, b) => a - b);
    
    return {
      p50: this.getPercentile(operation, 50),
      p95: this.getPercentile(operation, 95),
      p99: this.getPercentile(operation, 99),
      avg: values.reduce((a, b) => a + b, 0) / values.length,
      min: sorted[0],
      max: sorted[sorted.length - 1],
      count: values.length
    };
  }
}

from prometheus_client import Counter, Histogram, Gauge

logger = structlog.get_logger()

# Métricas Prometheus
REQUEST_COUNT = Counter(
    'http_requests_total',
    'Total HTTP Requests',
    ['method', 'endpoint', 'status']
)

REQUEST_LATENCY = Histogram(
    'http_request_duration_seconds',
    'HTTP request latency in seconds',
    ['method', 'endpoint'],
    buckets=[0.01, 0.05, 0.1, 0.3, 0.5, 1.0, 2.0, 5.0]
)

ACTIVE_REQUESTS = Gauge(
    'http_requests_active',
    'Active HTTP requests'
)


class ObservabilityMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next: Callable) -> Response:
        # Correlation ID para tracing
        correlation_id = request.headers.get('X-Correlation-ID', str(uuid.uuid4()))
        
        # Atualizar contexto do logger
        structlog.contextvars.bind_contextvars(
            correlation_id=correlation_id,
            endpoint=request.url.path,
            method=request.method,
            user_agent=request.headers.get('user-agent')
        )
        
        # Métricas
        ACTIVE_REQUESTS.inc()
        start_time = time.time()
        
        try:
            response = await call_next(request)
            
            # Log da requisição
            duration = time.time() - start_time
            logger.info(
                "http_request",
                status_code=response.status_code,
                duration_ms=round(duration * 1000, 2),
                client_ip=request.client.host if request.client else None
            )
            
            # Métricas
            REQUEST_COUNT.labels(
                method=request.method,
                endpoint=request.url.path,
                status=response.status_code
            ).inc()
            
            REQUEST_LATENCY.labels(
                method=request.method,
                endpoint=request.url.path
            ).observe(duration)
            
            # Adicionar correlation ID ao header da resposta
            response.headers['X-Correlation-ID'] = correlation_id
            
            return response
            
        except Exception as e:
            duration = time.time() - start_time
            logger.error(
                "http_request_error",
                error=str(e),
                duration_ms=round(duration * 1000, 2)
            )
            raise
        finally:
            ACTIVE_REQUESTS.dec()
            structlog.contextvars.clear_contextvars()

from typing import Any, Callable, TypeVar, Optional
import asyncio
import random
from circuitbreaker import circuit
import structlog
from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
    before_sleep_log
)

logger = structlog.get_logger()

T = TypeVar('T')


class ResilienceManager:
    """Gerenciador de resiliência para chamadas externas"""
    
    @staticmethod
    def create_circuit_breaker(
        failure_threshold: int = 5,
        recovery_timeout: int = 60,
        expected_exception: tuple = (Exception,)
    ):
        """Cria um circuit breaker configurado"""
        return circuit(
            failure_threshold=failure_threshold,
            recovery_timeout=recovery_timeout,
            expected_exception=expected_exception
        )
    
    @staticmethod
    def create_retry_decorator(
        max_retries: int = 3,
        min_wait: float = 1,
        max_wait: float = 10,
        retry_on_exceptions: tuple = (TimeoutError, ConnectionError)
    ):
        """Cria decorator de retry com backoff exponencial"""
        return retry(
            stop=stop_after_attempt(max_retries),
            wait=wait_exponential(min=min_wait, max=max_wait),
            retry=retry_if_exception_type(retry_on_exceptions),
            before_sleep=before_sleep_log(logger, "WARNING"),
            reraise=True
        )
    
    @staticmethod
    async def execute_with_timeout(
        func: Callable[..., T],
        timeout: float,
        *args,
        **kwargs
    ) -> T:
        """Executa uma função com timeout explícito"""
        try:
            return await asyncio.wait_for(
                func(*args, **kwargs),
                timeout=timeout
            )
        except asyncio.TimeoutError:
            logger.warning(
                "operation_timeout",
                timeout=timeout,
                function=func.__name__
            )
            raise TimeoutError(f"Operation timed out after {timeout} seconds")


class ExternalServiceClient:
    """Cliente base para serviços externos com resiliência embutida"""
    
    def __init__(
        self,
        service_name: str,
        base_timeout: float = 10,
        max_retries: int = 3,
        circuit_breaker_failure_threshold: int = 5
    ):
        self.service_name = service_name
        self.base_timeout = base_timeout
        self.max_retries = max_retries
        
        # Circuit breaker específico para este serviço
        self.circuit_breaker = ResilienceManager.create_circuit_breaker(
            failure_threshold=circuit_breaker_failure_threshold,
            recovery_timeout=60,
            expected_exception=(Exception,)
        )
        
        # Decorator de retry
        self.retry_decorator = ResilienceManager.create_retry_decorator(
            max_retries=max_retries,
            min_wait=1,
            max_wait=10,
            retry_on_exceptions=(TimeoutError, ConnectionError)
        )
    
    async def call_with_resilience(
        self,
        func: Callable,
        timeout: Optional[float] = None,
        fallback: Optional[Callable] = None,
        *args,
        **kwargs
    ) -> Any:
        """Executa chamada com circuit breaker, retry e timeout"""
        timeout = timeout or self.base_timeout
        
        @self.circuit_breaker
        @self.retry_decorator
        async def _execute():
            return await ResilienceManager.execute_with_timeout(
                func,
                timeout,
                *args,
                **kwargs
            )
        
        try:
            return await _execute()
        except Exception as e:
            logger.error(
                "external_service_error",
                service=self.service_name,
                error=str(e),
                args=args,
                kwargs={k: v for k, v in kwargs.items() if not k.lower().contains('password')}  # Mascarar dados sensíveis
            )
            
            # Fallback gracioso
            if fallback:
                logger.info("executing_fallback", service=self.service_name)
                return await fallback(*args, **kwargs)
            
            raise

class DatabaseManager:
    """Gerenciador de banco de dados com tratamento de erros robusto"""
    
    @staticmethod
    @contextmanager
    def get_db():
        """Context manager para sessões do banco de dados"""
        db: Session = SessionLocal()
        try:
            yield db
            db.commit()
        except Exception as e:
            db.rollback()
            logger.error("database_transaction_error", error=str(e))
            raise
        finally:
            db.close()
    
    @staticmethod
    async def health_check() -> bool:
        """Health check do banco de dados"""
        try:
            with SessionLocal() as db:
                # Query simples para testar conexão
                result = db.execute(text("SELECT 1"))
                return result.scalar() == 1
        except Exception as e:
            logger.error("database_health_check_failed", error=str(e))
            return False
    
    @staticmethod
    def execute_safe_query(db: Session, query: str, params: dict = None):
        """Executa query com parâmetros preparados (SQL injection safe)"""
        try:
            # Usando text() com parâmetros bind para prevenir SQL injection
            stmt = text(query)
            if params:
                result = db.execute(stmt, params)
            else:
                result = db.execute(stmt)
            
            return result
        except Exception as e:
            logger.error(
                "database_query_error",
                query=query,
                params={k: '***' if 'password' in k.lower() else v for k, v in (params or {}).items()},  # Mascarar senhas
                error=str(e)
            )
            raise

-- Modelo otimizado para métricas de autonomia
{{ config(
    materialized='incremental',
    unique_key='autonomy_metric_id',
    partition_by={
      "field": "created_at",
      "data_type": "timestamp",
      "granularity": "day"
    },
    cluster_by=["squad_id", "metric_type"]
) }}

WITH raw_metrics AS (
    SELECT
        CAST(metric_id AS STRING) AS metric_id,
        CAST(squad_id AS STRING) AS squad_id,
        metric_type,
        metric_value,
        JSON_EXTRACT_SCALAR(metadata, '$.component') AS component,
        TIMESTAMP(created_at) AS created_at,
        TIMESTAMP(_loaded_at) AS loaded_at
    FROM {{ source('raw', 'autonomy_metrics') }}
    WHERE metric_type IS NOT NULL
      AND squad_id IS NOT NULL
      AND created_at IS NOT NULL
    {% if is_incremental() %}
        AND TIMESTAMP(created_at) > (
            SELECT MAX(created_at) 
            FROM {{ this }}
        )
    {% endif %}
),

enriched_metrics AS (
    SELECT
        GENERATE_UUID() AS autonomy_metric_id,
        rm.*,
        CASE 
            WHEN metric_type LIKE '%latency%' THEN 'performance'
            WHEN metric_type LIKE '%error%' THEN 'reliability'
            WHEN metric_type LIKE '%success%' THEN 'success_rate'
            WHEN metric_type LIKE '%autonomy%' THEN 'autonomy_score'
            ELSE 'other'
        END AS metric_category,
        CASE 
            WHEN metric_type = 'api_latency_p95' AND metric_value > 1000 THEN 'alert'
            WHEN metric_type = 'error_rate' AND metric_value > 0.05 THEN 'alert'
            WHEN metric_type = 'autonomy_score' AND metric_value < 0.8 THEN 'alert'
            ELSE 'normal'
        END AS health_status
    FROM raw_metrics rm
)

SELECT * FROM enriched_metrics

### `dbt/models/marts/mart_squad_autonomy_daily.sql`
```sql
-- Mart otimizado para dashboards de autonomia diária
{{ config(
    materialized='table',
    partition_by={
      "field": "metric_date",
      "data_type": "date",
      "granularity": "day"
    },
    cluster_by=["squad_id", "metric_category"],
    indexes=[
      {'columns': ['squad_id', 'metric_date'], 'type': 'clustered'},
      {'columns': ['health_status'], 'type': 'clustered'}
    ]
) }}

WITH daily_metrics AS (
    SELECT
        DATE(created_at) AS metric_date,
        squad_id,
        metric_category,
        COUNT(*) AS total_metrics,
        COUNT(DISTINCT metric_type) AS unique_metric_types,
        AVG(metric_value) AS avg_value,
        APPROX_QUANTILES(metric_value, 4)[OFFSET(2)] AS median_value,
        MIN(metric_value) AS min_value,
        MAX(metric_value) AS max_value,
        STDDEV(metric_value) AS stddev_value,
        SUM(CASE WHEN health_status = 'alert' THEN 1 ELSE 0 END) AS alert_count
    FROM {{ ref('stg_autonomy_metrics') }}
    WHERE created_at >= TIMESTAMP_SUB(CURRENT_TIMESTAMP(), INTERVAL 90 DAY)
    GROUP BY 1, 2, 3
),

squad_info AS (
    SELECT
        squad_id,
        squad_name,
        squad_type,
        created_at AS squad_created_at
    FROM {{ ref('stg_squads') }}
    WHERE is_active = true
),

autonomy_scores AS (
    SELECT
        DATE(created_at) AS score_date,
        squad_id,
        AVG(CASE 
            WHEN metric_type = 'autonomy_score' 
            THEN metric_value 
        END) AS autonomy_score,
        AVG(CASE 
            WHEN metric_type = 'decision_accuracy' 
            THEN metric_value 
        END) AS decision_accuracy,
        AVG(CASE 
            WHEN metric_type = 'self_healing_success_rate' 
            THEN metric_value 
        END) AS self_healing_rate
    FROM {{ ref('stg_autonomy_metrics') }}
    WHERE metric_type IN ('autonomy_score', 'decision_accuracy', 'self_healing_success_rate')
    GROUP BY 1, 2
)

SELECT
    dm.metric_date,
    dm.squad_id,
    si.squad_name,
    si.squad_type,
    dm.metric_category,
    dm.total_metrics,
    dm.unique_metric_types,
    dm.avg_value,
    dm.median_value,
    dm.min_value,
    dm.max_value,
    dm.stddev_value,
    dm.alert_count,
    dm.alert_count / NULLIF(dm.total_metrics, 0) AS alert_rate,
    ascores.autonomy_score,
    ascores.decision_accuracy,
    ascores.self_healing_rate,
    CASE 
        WHEN ascores.autonomy_score >= 0.9 THEN 'excellent'
        WHEN ascores.autonomy_score >= 0.8 THEN 'good'
        WHEN ascores.autonomy_score >= 0.7 THEN 'fair'
        ELSE 'needs_attention'
    END AS autonomy_health,
    CURRENT_TIMESTAMP() AS calculated_at
FROM daily_metrics dm
LEFT JOIN squad_info si ON dm.squad_id = si.squad_id
LEFT JOIN autonomy_scores ascores 
    ON dm.metric_date = ascores.score_date 
    AND dm.squad_id = ascores.squad_id
WHERE dm.metric_date >= DATE_SUB(CURRENT_DATE(), INTERVAL 30 DAY)
```

            "duration_ms": duration_ms,
            **additional_fields
        }
        
        if error:
            log_entry["error"] = {
                "type": error.__class__.__name__,
                "message": str(error),
                "traceback": self._get_traceback(error)
            }
        
        if level == LogLevel.DEBUG:
            self.logger.debug(json.dumps(log_entry))
        elif level == LogLevel.INFO:
            self.logger.info(json.dumps(log_entry))
        elif level == LogLevel.WARNING:
            self.logger.warning(json.dumps(log_entry))
        elif level == LogLevel.ERROR:
            self.logger.error(json.dumps(log_entry))
        elif level == LogLevel.CRITICAL:
            self.logger.critical(json.dumps(log_entry))
    
    @contextmanager
    def timed_operation(self, operation_name: str, context: LogContext):
        """Context manager para operações cronometradas"""
        start_time = time.time()
        operation_id = str(uuid.uuid4())
        
        try:
            self.log(
                LogLevel.INFO,
                f"Starting operation: {operation_name}",
                context,
                operation_name=operation_name,
                operation_id=operation_id
            )
            
            yield operation_id
            
            duration_ms = (time.time() - start_time) * 1000
            self.log(
                LogLevel.INFO,
                f"Completed operation: {operation_name}",
                context,
                operation_name=operation_name,
                operation_id=operation_id,
                duration_ms=duration_ms,
                success=True
            )
            
        except Exception as e:
            duration_ms = (time.time() - start_time) * 1000
            self.log(
                LogLevel.ERROR,
                f"Failed operation: {operation_name}",
                context,
                operation_name=operation_name,
                operation_id=operation_id,
                duration_ms=duration_ms,
                success=False,
                error=e
            )
            raise
    
    def _get_traceback(self, error: Exception) -> str:
        import traceback
        return traceback.format_exc()

class StructuredLogFormatter(logging.Formatter):
    def format(self, record):
        try:
            log_data = json.loads(record.getMessage())
            return json.dumps(log_data, ensure_ascii=False)
        except json.JSONDecodeError:
            return record.getMessage()
