function fetchUserOrders(userId) {
    // Original: Possível N+1 query problem
    const user = db.users.find(user => user.id === userId);
    const orders = db.orders.filter(order => order.userId === userId);

    // Refactoring: Using more efficient queries
    const [user, orders] = db.query(`
        SELECT * FROM users 
             JOIN orders ON users.id = orders.userId 
             WHERE users.id = ?
    `, [userId]);

    return { user, orders };
}

// Original: Missing error handling

### Exemplo de Estruturação do Processo no Código

Vou fornecer um exemplo de como poderíamos estruturar partes do código com controles de segurança aplicados. O exemplo a seguir é focalizado em endpoints de autenticação e manipulação de dados sensíveis:

```python
# security_model.py

from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from werkzeug.security import generate_password_hash, check_password_hash
from itsdangerous import URLSafeTimedSerializer

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///sec_audit.db'
app.config['SECRET_KEY'] = 'supersecretkey'
db = SQLAlchemy(app)
serializer = URLSafeTimedSerializer(app.config['SECRET_KEY'])

class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(150), nullable=False, unique=True)
    password_hash = db.Column(db.String(256), nullable=False)

# Endpoint: User Registration
@app.route('/register', methods=['POST'])
def register():
    data = request.get_json()
    username = data.get('username')
    password = data.get('password')

    if not username or not password:
        return jsonify({"error": "Missing fields"}), 400
    
    # Password hashing for secure storage
    password_hash = generate_password_hash(password)
    user = User(username=username, password_hash=password_hash)
    
    db.session.add(user)
    db.session.commit()

    return jsonify({"message": "User registered successfully"}), 201

# Endpoint: User Login
@app.route('/login', methods=['POST'])
def login():
    data = request.get_json()
    username = data.get('username')
    password = data.get('password')

    user = User.query.filter_by(username=username).first()

    if user and check_password_hash(user.password_hash, password):
        # Token generation for authenticated session
        token = serializer.dumps({'user_id': user.id})
        return jsonify({"token": token}), 200
    else:
        return jsonify({"error": "Invalid credentials"}), 401

# Security Headers
@app.after_request
def apply_security_headers(response):
    response.headers['Content-Security-Policy'] = "default-src 'self'"
    response.headers['X-Content-Type-Options'] = "nosniff"
    response.headers['X-Frame-Options'] = "DENY"
    return response

if __name__ == '__main__':
    db.create_all()
    app.run(debug=True)
```

### Key Security Practices in the Example

- **Password Hashing:** Utiliza `werkzeug.security` para hash seguro de senhas.
- **Tokenization:** Gera tokens temporários para sessões autenticadas, usando `itsdangerous.URLSafeTimedSerializer`.
- **Validation & Error Handling:** Inclui checagens de validade nas entradas de usuário e retorna códigos de resposta adequados.
- **Security Headers:** Configura cabeçalhos de segurança para mitigar ataques XSS, clickjacking e MIME sniffing.

Esse exemplo ilustra como integrar as práticas de segurança diretamente no processo de desenvolvimento ao invés de tratá-las como etapa posterior, garantindo um produto mais seguro desde o início.
