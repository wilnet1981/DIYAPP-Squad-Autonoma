# 2. DECISÕES ARQUITETURAIS PRINCIPAIS (ADRs)

# ADR-002: Service Mesh Selection
# ================================
# Contexto: Necessidade de observabilidade, segurança e controle de tráfego
# entre microserviços. Avaliamos Istio vs Linkerd vs Consul.
# Decisão: Implementar Istio como service mesh principal.
# Justificativa: 
# - Feature set mais completo (mTLS, rate limiting, circuit breaking)
# - Comunidade ativa e suporte da Google/IBM
# - Integração nativa com Kubernetes
# - Suporte a WebAssembly para extensibilidade
# Trade-offs aceitos:
# - Maior complexidade inicial
# - Overhead de recursos (~100MB por pod)
# - Curva de aprendizado mais íngreme

# ADR-003: Padrão de Comunicação
# ================================
# Contexto: Comunicação síncrona vs assíncrona entre serviços
# Decisão: Padrão híbrido:
# 1. gRPC para comunicação síncrona serviço-serviço
# 2. Apache Kafka para eventos assíncronos
# 3. REST/GraphQL para APIs públicas
# Justificativa:
# - gRPC: Performance superior, streaming bidirecional, contract-first
# - Kafka: Durabilidade de eventos, replay capability, alta throughput
# - REST/GraphQL: Compatibilidade com clientes externos

# ADR-004: Estratégia de Banco de Dados
# =====================================
# Contexto: Cada microserviço precisa de isolamento de dados
# Decisão: Database-per-service com PostgreSQL
# Justificativa:
# - Isolamento total entre serviços
# - Escalabilidade independente
# - Polyglot persistence quando necessário
# Implementação:
# - 1 instância PostgreSQL primária por serviço crítico
# - Read replicas para serviços de leitura intensiva
# - Redis para cache (padrão write-through)

alert_rules:
  - alert: "HighErrorRate"
    expr: 'rate(istio_requests_total{response_code!~"2.."}[5m]) / rate(istio_requests_total[5m]) > 0.05'
    for: "5m"
    labels:
      severity: "critical"
    annotations:
      summary: "High error rate detected"
      description: "Error rate for {{ $labels.destination_service }} is above 5%"

```typescript
import * as pulumi from "@pulumi/pulumi";
import * as aws from "@pulumi/aws";
import * as awsx from "@pulumi/awsx";

export async function createVPC(
    appName: string,
    region: string,
    tags: { [key: string]: string }
) {
    // VPC com 3 AZs para alta disponibilidade
    const vpc = new awsx.ec2.Vpc(`${appName}-vpc`, {
        cidrBlock: "10.0.0.0/16",
        numberOfAvailabilityZones: 3,
        numberOfNatGateways: 3, // NAT Gateway por AZ para resiliência
        subnets: [
            {
                type: "public",
                tags: { ...tags, SubnetType: "public" },
            },
            {
                type: "private",
                tags: { ...tags, SubnetType: "private" },
            },
        ],
        tags: { ...tags, Name: `${appName}-vpc` },
    });
    
    // Security Groups
    const clusterSG = new aws.ec2.SecurityGroup(`${appName}-cluster-sg`, {
        vpcId: vpc.vpcId,
        description: "Security group for EKS cluster",
        ingress: [
            // Comunicação entre nós
            {
                protocol: "-1",
                fromPort: 0,
                toPort: 0,
                cidrBlocks: [vpc.vpc.cidrBlock],
            },
            // API Server
            {
                protocol: "tcp",
                fromPort: 443,
                toPort: 443,
                cidrBlocks: ["0.0.0.0/0"], // Restringir em produção
            },
        ],
        egress: [
            {
                protocol: "-1",
                fromPort: 0,
                toPort: 0,
                cidrBlocks: ["0.0.0.0/0"],
            },
        ],
        tags: { ...tags, Name: `${appName}-cluster-sg` },
    });
    
    // Endpoint VPC para serviços AWS (mais seguro e barato)
    new aws.ec2.VpcEndpoint(`${appName}-s3-endpoint`, {
        vpcId: vpc.vpcId,
        serviceName: `com.amazonaws.${region}.s3`,
        vpcEndpointType: "Gateway",
        routeTableIds: vpc.privateSubnetIds,
        tags: { ...tags, Name: `${appName}-s3-endpoint` },
    });
    
    return {
        vpcId: vpc.vpcId,
        publicSubnetIds: vpc.publicSubnetIds,
        privateSubnetIds: vpc.privateSubnetIds,
        clusterSecurityGroupId: clusterSG.id,
    };
}
```

```typescript
import * as pulumi from "@pulumi/pulumi";
import * as aws from "@pulumi/aws";
import * as eks from "@pulumi/eks";
import * as k8s from "@pulumi/kubernetes";

export async function createEKSCluster(
    appName: string,
    vpc: any,
    region: string,
    tags: { [key: string]: string }
) {
    // IAM Role para nodes
    const nodeRole = new aws.iam.Role(`${appName}-node-role`, {
        assumeRolePolicy: aws.iam.assumeRolePolicyForPrincipal({
            Service: "ec2.amazonaws.com",
        }),
        tags,
    });
    
    // Políticas necessárias para nodes
    const nodePolicyArns = [
        "arn:aws:iam::aws:policy/AmazonEKSWorkerNodePolicy",
        "arn:aws:iam::aws:policy/AmazonEKS_CNI_Policy",
        "arn:aws:iam::aws:policy/AmazonEC2ContainerRegistryReadOnly",
        "arn:aws:iam::aws:policy/CloudWatchAgentServerPolicy",
    ];
    
    nodePolicyArns.forEach((policyArn, index) => {
        new aws.iam.RolePolicyAttachment(`${appName}-node-policy-${index}`, {
            role: nodeRole,
            policyArn,
        });
    });
    
    // Cluster EKS
    const cluster = new eks.Cluster(`${appName}-cluster`, {
        vpcId: vpc.vpcId,
        publicSubnetIds: vpc.publicSubnetIds,
        privateSubnetIds: vpc.privateSubnetIds,
        instanceType: "t3.medium",
        desiredCapacity: 3,
        minSize: 3,
        maxSize: 10,
        nodeAssociatePublicIpAddress: false,
        nodeRootVolumeSize: 100,
        createOidcProvider: true,
        serviceRole: {
            args: {
                tags,
            },
        },
        tags: { ...tags, Name: `${appName}-cluster` },
        
        // Configurações SRE
        version: "1.28",
        enabledClusterLogTypes: [
            "api",
            "audit",
            "authenticator",
            "controllerManager",
            "scheduler"
        ],
        
        // Node Groups para diferentes workloads
        nodeGroups: {
            "main": {
                instanceType: "t3.medium",
                desiredCapacity: 3,
                minSize: 3,
                maxSize: 6,
                labels: { "node-type": "main", "sre-monitored": "true" },
                taints: [],
                tags: { ...tags, "k8s.io/cluster-autoscaler/enabled": "true" },
            },
            "llm": {
                instanceType: "g4dn.xlarge", // GPU para LLMs
                desiredCapacity: 2,
                minSize: 2,
                maxSize: 4,
                labels: { "node-type": "llm", "gpu": "true" },
                taints: [{ key: "gpu", value: "true", effect: "NoSchedule" }],
                tags: { ...tags, "k8s.io/cluster-autoscaler/enabled": "true" },
            },
        },
        
        // Configurações de segurança
        encryptionConfigKeyArn: config.get("kmsKeyArn"),
        endpointPrivateAccess: true,
        endpointPublicAccess: true,
        publicAccessCidrs: ["0.0.0.0/0"], // Restringir em produção
    });
    
    // Auto Scaling para nodes
    const autoscalingPolicy = new aws.autoscaling.Policy(`${appName}-scaling-policy`, {
        autoscalingGroupName: cluster.nodeGroups.main.autoscalingGroup.groupName,
        policyType: "TargetTrackingScaling",
        targetTrackingConfiguration: {
            predefinedMetricSpecification: {
                predefinedMetricType: "ASGAverageCPUUtilization",
            },
            targetValue: 70.0,
        },
        tags,
    });
    
    return {
        eksCluster: cluster,
        kubeconfig: cluster.kubeconfig,
        provider: cluster.provider,
        oidcProvider: cluster.core.oidcProvider,
    };
}
```

```typescript
import * as pulumi from "@pulumi/pulumi";
import * as aws from "@pulumi/aws";
import * as k8s from "@pulumi/kubernetes";

export async function setupStorage(
    cluster: any,
    appName: string,
    env: string
) {
    const provider = cluster.provider;
    
    // Storage Class para EBS gp3 (mais performático)
    const gp3StorageClass = new k8s.storage.v1.StorageClass(
        `${appName}-gp3-sc`,
        {
            metadata: {
                name: "gp3",
                annotations: {
                    "storageclass.kubernetes.io/is-default-class": "true",
                },
            },
            provisioner: "ebs.csi.aws.com",
            parameters: {
                type: "gp3",
                fsType: "ext4",
                iops: "3000",
                throughput: "125",
            },
            allowVolumeExpansion: true,
            volumeBindingMode: "WaitForFirstConsumer",
        },
        { provider }
    );
    
    // Storage Class para alta performance (LLM models)
    const io2StorageClass = new k8s.storage.v1.StorageClass(
        `${appName}-io2-sc`,
        {
            metadata: {
                name: "io2",
            },
            provisioner: "ebs.csi.aws.com",
            parameters: {
                type: "io2",
                fsType: "ext4",
                iops: "10000",
            },
            allowVolumeExpansion: true,
        },
        { provider }
    );
    
    // EFS para shared storage (logs, modelos compartilhados)
    const efs = new aws.efs.FileSystem(`${appName}-efs`, {
        performanceMode: "generalPurpose",
        throughputMode: "bursting",
        encrypted: true,
        tags: {
            Name: `${appName}-efs`,
            Environment: env,
        },
    });
    
    // Mount targets em cada AZ
    const vpc = aws.ec2.getVpc({ default: true });
    const subnets = aws.ec2.getSubnets({
        filters: [{ name: "vpc-id", values: [vpc.then(v => v.id)] }],
    });
    
    subnets.then(s => {
        s.ids.forEach((subnetId, index) => {
            new aws.efs.MountTarget(`${appName}-mt-${index}`, {
                fileSystemId: efs.id,
                subnetId: subnetId,
                securityGroups: [cluster.eksCluster.clusterSecurityGroupId],
            });
        });
    });
    
    // CSI Driver para EFS
    const efsStorageClass = new k8s.storage.v1.StorageClass(
        `${appName}-efs-sc`,
        {
            metadata: {
                name: "efs",
            },
            provisioner: "efs.csi.aws.com",
            parameters: {
                provisioningMode: "efs-ap",
                fileSystemId: efs.id,
                directoryPerms: "700",
            },
        },
        { provider }
    );
    
    // S3 Bucket para backups e artefatos
    const backupBucket = new aws.s3.BucketV2(`${appName}-backups-${env}`, {
        forceDestroy: false,
        tags: {
            Name: `${appName}-backups`,
            Environment: env,
            "SRE-Backup": "daily",
        },
    });
    
    // Política de lifecycle para backups
    new aws.s3.BucketLifecycleConfigurationV2(
        `${appName}-backup-lifecycle`,
        {
            bucket: backupBucket.id,
            rules: [
                {
                    id: "backup-retention",
                    status: "Enabled",
                    transitions: [
                        {
                            days: 30,
                            storageClass: "STANDARD_IA",
                        },
                        {
                            days: 90,
                            storageClass: "GLACIER",
                        },
                    ],
                    expiration: {
                        days: 365,
                    },
                },
            ],
        }
    );
    
    return {
        gp3StorageClass: gp3StorageClass.metadata.name,
        io2StorageClass: io2StorageClass.metadata.name,
        efsStorageClass: efsStorageClass.metadata.name,
        efsId: efs.id,
        backupBucket: backupBucket.bucket,
    };
}
```

import time
from typing import Any, Callable, Optional
from circuitbreaker import circuit
from functools import wraps
import logging

logger = logging.getLogger(__name__)

class CircuitBreakerConfig:
    """Configuração do Circuit Breaker"""
    
    def __init__(
        self,
        failure_threshold: int = 5,
        recovery_timeout: int = 30,
        expected_exception: tuple = (Exception,),
        name: Optional[str] = None
    ):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.expected_exception = expected_exception
        self.name = name

class ResilientService:
    """Classe base para serviços com resiliência embutida"""
    
    @staticmethod
    def with_circuit_breaker(
        config: CircuitBreakerConfig
    ) -> Callable:
        """Decorator para aplicar circuit breaker a métodos"""
        def decorator(func: Callable) -> Callable:
            @wraps(func)
            @circuit(
                failure_threshold=config.failure_threshold,
                recovery_timeout=config.recovery_timeout,
                expected_exception=config.expected_exception,
                name=config.name or func.__name__
            )
            def wrapper(*args, **kwargs):
                return func(*args, **kwargs)
            return wrapper
        return decorator
    
    @staticmethod
    def with_retry(
        max_attempts: int = 3,
        delay: float = 1.0,
        backoff: float = 2.0,
        exceptions: tuple = (Exception,)
    ) -> Callable:
        """Decorator para aplicar retry com backoff exponencial"""
        from tenacity import (
            retry,
            stop_after_attempt,
            wait_exponential,
            retry_if_exception_type
        )
        
        def decorator(func: Callable) -> Callable:
            @wraps(func)
            @retry(
                stop=stop_after_attempt(max_attempts),
                wait=wait_exponential(multiplier=delay, exp_base=backoff),
                retry=retry_if_exception_type(exceptions),
                before_sleep=lambda retry_state: logger.warning(
                    f"Retrying {func.__name__} after failure: "
                    f"{retry_state.outcome.exception()}. "
                    f"Attempt {retry_state.attempt_number}/{max_attempts}"
                )
            )
            def wrapper(*args, **kwargs):
                return func(*args, **kwargs)
            return wrapper
        return decorator
    
    @staticmethod
    def with_timeout(
        timeout_seconds: int = 10
    ) -> Callable:
        """Decorator para aplicar timeout a métodos"""
        import signal
        from functools import wraps
        
        def timeout_handler(signum, frame):
            raise TimeoutError(f"Operation timed out after {timeout_seconds} seconds")
        
        def decorator(func: Callable) -> Callable:
            @wraps(func)
            def wrapper(*args, **kwargs):
                # Set the timeout signal
                signal.signal(signal.SIGALRM, timeout_handler)
                signal.alarm(timeout_seconds)
                
                try:
                    result = func(*args, **kwargs)
                finally:
                    # Disable the alarm
                    signal.alarm(0)
                
                return result
            return wrapper
        return decorator

from typing import Any, Dict, Optional
import httpx
from datetime import datetime
import logging
from .circuit_breaker import CircuitBreakerConfig, ResilientService

logger = logging.getLogger(__name__)

class ExternalServiceClient:
    """Cliente resiliente para serviços externos"""
    
    def __init__(
        self,
        base_url: str,
        service_name: str,
        timeout: int = 10,
        retry_attempts: int = 3
    ):
        self.base_url = base_url.rstrip("/")
        self.service_name = service_name
        self.timeout = timeout
        self.retry_attempts = retry_attempts
        
        # Configuração do circuit breaker específica para este serviço
        self.circuit_config = CircuitBreakerConfig(
            failure_threshold=5,
            recovery_timeout=60,
            expected_exception=(httpx.RequestError, httpx.HTTPStatusError),
            name=f"external_service_{service_name}"
        )
        
        # Cliente HTTP com configurações padrão
        self.client = httpx.AsyncClient(
            timeout=httpx.Timeout(timeout),
            limits=httpx.Limits(max_keepalive_connections=5, max_connections=10)
        )
    
    @ResilientService.with_circuit_breaker(CircuitBreakerConfig(
        failure_threshold=5,
        recovery_timeout=60,
        name="external_service_request"
    ))
    @ResilientService.with_retry(
        max_attempts=3,
        delay=1.0,
        backoff=2.0,
        exceptions=(httpx.RequestError, httpx.HTTPStatusError)
    )
    async def get(
        self,
        path: str,
        params: Optional[Dict] = None,
        headers: Optional[Dict] = None
    ) -> Dict[str, Any]:
        """GET request com resiliência"""
        url = f"{self.base_url}/{path.lstrip('/')}"
        
        try:
            response = await self.client.get(
                url,
                params=params,
                headers=headers
            )
            response.raise_for_status()
            return response.json()
        except httpx.HTTPStatusError as e:
            logger.error(
                f"HTTP error from {self.service_name}: {e.response.status_code}",
                extra={
                    "service": self.service_name,
                    "status_code": e.response.status_code,
                    "url": url
                }
            )
            raise
        except httpx.RequestError as e:
            logger.error(
                f"Request error from {self.service_name}: {str(e)}",
                extra={"service": self.service_name, "url": url}
            )
            raise
    
    @ResilientService.with_circuit_breaker(CircuitBreakerConfig(
        failure_threshold=5,
        recovery_timeout=60,
        name="external_service_post"
    ))
    async def post(
        self,
        path: str,
        data: Dict[str, Any],
        headers: Optional[Dict] = None
    ) -> Dict[str, Any]:
        """POST request com resiliência"""
        url = f"{self.base_url}/{path.lstrip('/')}"
        
        try:
            response = await self.client.post(
                url,
                json=data,
                headers=headers
            )
            response.raise_for_status()
            return response.json()
        except httpx.HTTPStatusError as e:
            logger.error(
                f"HTTP error from {self.service_name}: {e.response.status_code}",
                extra={
                    "service": self.service_name,
                    "status_code": e.response.status_code,
                    "url": url
                }
            )
            raise
        except httpx.RequestError as e:
            logger.error(
                f"Request error from {self.service_name}: {str(e)}",
                extra={"service": self.service_name, "url": url}
            )
            raise
    
    async def close(self):
        """Fecha o cliente HTTP"""
        await self.client.aclose()

from typing import Any, Dict, Optional
from kafka import KafkaProducer
from kafka.errors import KafkaError
import json
import logging
from ..resilience.circuit_breaker import ResilientService

logger = logging.getLogger(__name__)

class KafkaEventProducer:
    """Produtor resiliente de eventos Kafka"""
    
    def __init__(
        self,
        bootstrap_servers: str,
        client_id: str = "diyapp-evolution-producer"
    ):
        self.bootstrap_servers = bootstrap_servers
        self.client_id = client_id
        
        # Configuração do producer com serialização JSON
        self.producer = KafkaProducer(
            bootstrap_servers=bootstrap_servers,
            client_id=client_id,
            value_serializer=lambda v: json.dumps(v).encode('utf-8'),
            key_serializer=lambda k: k.encode('utf-8') if k else None,
            acks='all',  # Garantia de entrega
            retries=3,
            max_in_flight_requests_per_connection=1,
            request_timeout_ms=30000,
            retry_backoff_ms=1000
        )
    
    @ResilientService.with_circuit_breaker(
        ResilientService.CircuitBreakerConfig(
            failure_threshold=3,
            recovery_timeout=30,
            expected_exception=(KafkaError,),
            name="kafka_producer_send"
        )
    )
    def send_event(
        self,
        topic: str,
        event: Dict[str, Any],
        key: Optional[str] = None,
        headers: Optional[Dict] = None
    ) -> bool:
        """Envia evento para tópico Kafka com confirmação"""
        try:
            # Adiciona metadados ao evento
            enriched_event = {
                **event,
                "_metadata": {
                    "producer": self.client_id,
                    "timestamp": time.time(),
                    "version": "1.0"
                }
            }
            
            # Prepara headers
            kafka_headers = []
            if headers:
                kafka_headers = [(k, v.encode('utf-8')) 
                               for k, v in headers.items()]
            
            # Envia com callback para confirmação
            future = self.producer.send(
                topic=topic,
                value=enriched_event,
                key=key,
                headers=kafka_headers
            )
            
            # Aguarda confirmação (síncrono para garantir entrega)
            record_metadata = future.get(timeout=10)
            
            logger.info(
                f"Event sent to Kafka - Topic: {topic}, "
                f"Partition: {record_metadata.partition}, "
                f"Offset: {record_metadata.offset}",
                extra={
                    "topic": topic,
                    "partition": record_metadata.partition,
                    "offset": record_metadata.offset,
                    "event_type": event.get("type", "unknown")
                }
            )
            
            return True
            
        except KafkaError as e:
            logger.error(
                f"Failed to send event to Kafka: {str(e)}",
                extra={"topic": topic, "error": str(e)}
            )
            raise
    
    def flush(self):
        """Garante que todos os eventos foram enviados"""
        self.producer.flush()
    
    def close(self):
        """Fecha o producer"""
        self.producer.close()

        self.channel = self.connection.channel()

                exchange=self.exchange_name,
                exchange_type=self.exchange_type,

```jsx
          padding: ${spacing[2.5]} ${spacing[5]};
          gap: ${spacing[2]};
        `;
    }
  }}

  /* Variant styles */
  ${({ variant }) => {
    switch (variant) {
      case 'primary':
        return `
          background-color: ${colors.primary[600]};
          color: ${colors.text.inverse};
          &:hover:not(:disabled) {

          }
        `;
      case 'outline':

          color: ${colors.text.primary};
          &:hover:not(:disabled) {
            background-color: ${colors.neutral[50]};
            border-color: ${colors.neutral[400]};
          }
        `;
      case 'ghost':
        return `
          background-color: transparent;
          color: ${colors.text.primary};
          &:hover:not(:disabled) {
            background-color: ${colors.neutral[100]};
          }
        `;
      case 'danger':

          color: ${colors.text.inverse};
          &:hover:not(:disabled) {
            background-color: ${colors.error[600]};

        `;
      default:
        return '';
    }
  }}
```
