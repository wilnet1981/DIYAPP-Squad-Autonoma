@dataclass
class BaseEvent(ABC):
    """Classe base para todos os eventos do sistema"""
    event_id: str
    event_type: EventType
    timestamp: str
    source_service: ServiceName
    correlation_id: str
    priority: EventPriority = EventPriority.MEDIUM
    version: str = "1.0.0"
    
    def __post_init__(self):
        if not self.event_id:
            self.event_id = str(uuid.uuid4())
        if not self.timestamp:
            self.timestamp = datetime.utcnow().isoformat()
        if not self.correlation_id:
            self.correlation_id = str(uuid.uuid4())
    
    def to_dict(self) -> Dict[str, Any]:
        """Converte evento para dicionário"""
        data = asdict(self)
        data['event_type'] = self.event_type.value
        data['source_service'] = self.source_service.value
        data['priority'] = self.priority.value
        return data
    
    def to_json(self) -> str:
        """Converte evento para JSON"""
        return json.dumps(self.to_dict(), ensure_ascii=False)
    
    @classmethod
    def from_json(cls, json_str: str):
        """Cria evento a partir de JSON"""
        data = json.loads(json_str)
        return cls.from_dict(data)
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]):
        """Cria evento a partir de dicionário"""
        data['event_type'] = EventType(data['event_type'])
        data['source_service'] = ServiceName(data['source_service'])
        data['priority'] = EventPriority(data['priority'])
        return cls(**data)

class EventManager:
    """Gerenciador central de eventos para comunicação entre microsserviços"""
    
    def __init__(self, broker_type: str = "redis"):
        """
        Inicializa o gestor de eventos
        
        Args:
            broker_type: Tipo de broker (redis, rabbitmq, kafka)
        """
        self.broker_type = broker_type
        self.connected = False
        self.logger = logging.getLogger(f"{__name__}.EventManager")
        
    async def connect(self):
        """Conecta ao broker de mensagens"""
        try:
            if self.broker_type == "redis":
                import redis.asyncio as redis
                self.client = redis.Redis(
                    host="localhost",
                    port=6379,
                    password=None,
                    decode_responses=True
                )
                await self.client.ping()
            elif self.broker_type == "rabbitmq":
                import aio_pika
                self.connection = await aio_pika.connect_robust(
                    "amqp://guest:guest@localhost/"
                )
                self.channel = await self.connection.channel()
            
            self.connected = True
            self.logger.info(f"Conectado ao broker {self.broker_type}")
            
        except Exception as e:
            self.logger.error(f"Erro ao conectar ao broker: {e}")
            raise
    
    async def publish(self, event: BaseEvent, routing_key: str = None):
        """
        Publica um evento no broker
        
        Args:
            event: Instância do evento
            routing_key: Chave de roteamento (opcional)
        """
        if not self.connected:
            await self.connect()
        
        try:
            event_data = event.to_json()
            
            if self.broker_type == "redis":
                # Publica no canal específico do tipo de evento
                channel = routing_key or event.event_type.value
                await self.client.publish(channel, event_data)
                
                # Também publica em um canal geral para monitoramento
                await self.client.publish("diyapp.events.all", event_data)
                
            elif self.broker_type == "rabbitmq":
                exchange = await self.channel.declare_exchange(
                    "diyapp.events",
                    aio_pika.ExchangeType.TOPIC,
                    durable=True
                )
                
                routing_key = routing_key or f"{event.event_type.value}.{event.source_service.value}"
                message = aio_pika.Message(
                    body=event_data.encode(),
                    content_type="application/json",
                    headers={
                        "event_id": event.event_id,
                        "correlation_id": event.correlation_id,
                        "priority": event.priority.value
                    }
                )
                
                await exchange.publish(message, routing_key=routing_key)
            
            self.logger.info(f"Evento publicado: {event.event_type.value} - {event.event_id}")
            
        except Exception as e:
            self.logger.error(f"Erro ao publicar evento: {e}")
            raise
    
    async def subscribe(self, event_types: List[EventType], callback, queue_name: str = None):
        """
        Inscreve-se em tipos específicos de eventos
        
        Args:
            event_types: Lista de tipos de evento para escutar
            callback: Função callback para processar eventos
            queue_name: Nome da fila (opcional)
        """
        if not self.connected:
            await self.connect()
        
        try:
            if self.broker_type == "redis":
                pubsub = self.client.pubsub()
                
                channels = [et.value for et in event_types]
                channels.append("diyapp.events.all")  # Canal geral
                
                await pubsub.subscribe(*channels)
                
                async for message in pubsub.listen():
                    if message['type'] == 'message':
                        try:
                            event_dict = json.loads(message['data'])
                            event_type = EventType(event_dict['event_type'])
                            
                            # Filtra por tipo de evento
                            if event_type in event_types:
                                event_class = self._get_event_class(event_type)
                                event = event_class.from_dict(event_dict)
                                await callback(event)
                        except Exception as e:
                            self.logger.error(f"Erro ao processar mensagem: {e}")
            
            elif self.broker_type == "rabbitmq":
                queue_name = queue_name or f"{uuid.uuid4()}"
                queue = await self.channel.declare_queue(queue_name, durable=True)
                
                # Bind para cada tipo de evento
                for event_type in event_types:
                    routing_key = f"{event_type.value}.#"
                    await queue.bind("diyapp.events", routing_key)
                
                async for message in queue:
                    async with message.process():
                        try:
                            event_data = message.body.decode()
                            event_dict = json.loads(event_data)
                            event_type = EventType(event_dict['event_type'])
                            
                            event_class = self._get_event_class(event_type)
                            event = event_class.from_dict(event_dict)
                            await callback(event)
                        except Exception as e:
                            self.logger.error(f"Erro ao processar mensagem: {e}")
        
        except Exception as e:
            self.logger.error(f"Erro na inscrição: {e}")
            raise
    
    def _get_event_class(self, event_type: EventType):
        """Retorna a classe de evento baseada no tipo"""
        event_class_map = {
            EventType.USER_CREATED: UserEvent,
            EventType.USER_UPDATED: UserEvent,
            EventType.PROJECT_CREATED: ProjectEvent,
            EventType.PROJECT_UPDATED: ProjectEvent,
            EventType.TASK_ASSIGNED: TaskEvent,
            EventType.TASK_COMPLETED: TaskEvent,
            EventType.AI_AGENT_INVOKED: AIAgentEvent,
            EventType.FILE_UPLOADED: FileEvent,
            EventType.ERROR_OCCURRED: ErrorEvent,
            EventType.SYSTEM_HEALTH_CHECK: HealthEvent,
        }
        
        return event_class_map.get(event_type, BaseEvent)
    
    async def close(self):
        """Fecha a conexão com o broker"""
        if self.connected:
            if self.broker_type == "redis":
                await self.client.close()
            elif self.broker_type == "rabbitmq":
                await self.connection.close()
            self.connected = False

class HealthCheckService:
    """Serviço de monitoramento da saúde dos microsserviços"""
    
    def __init__(self, event_manager: EventManager):
        self.event_manager = event_manager
        self.services_status = {}
        self.logger = logging.getLogger(f"{__name__}.HealthCheckService")
    
    async def start_monitoring(self):
        """Inicia o monitoramento dos serviços"""
        await self.event_manager.subscribe(
            event_types=[EventType.SYSTEM_HEALTH_CHECK],
            callback=self._process_health_event
        )
        
        # Agenda health checks periódicos
        import asyncio
        asyncio.create_task(self._periodic_health_check())
    
    async def _process_health_event(self, event: HealthEvent):
        """Processa eventos de saúde recebidos"""
        self.services_status[event.source_service] = {
            "status": event.service_status,
            "timestamp": event.timestamp,
            "metrics": {
                "cpu_usage": event.cpu_usage,
                "memory_usage": event.memory_usage,
                "active_connections": event.active_connections
            }
        }
        
        # Verifica se algum serviço está com problemas
        self._check_service_health()
    
    async def _periodic_health_check(self):
        """Executa verificações periódicas de saúde"""
        import asyncio
        
        while True:
            await asyncio.sleep(60)  # A cada 60 segundos
            
            # Publica evento de health check do próprio serviço de monitoramento
            health_event = HealthEvent(
                event_id=str(uuid.uuid4()),
                event_type=EventType.SYSTEM_HEALTH_CHECK,
                timestamp=datetime.utcnow().isoformat(),
                source_service=ServiceName.MONITORING_SERVICE,
                correlation_id=str(uuid.uuid4()),
                service_status={"monitoring": "healthy"},
                cpu_usage=0.0,
                memory_usage=0.0,
                active_connections=1,
                queue_sizes={}
            )
            
            await self.event_manager.publish(health_event)
    
    def _check_service_health(self):
        """Verifica a saúde de todos os serviços"""
        unhealthy_services = []
        
        for service_name, status in self.services_status.items():
            if status.get("status", {}).get("overall") != "healthy":
                unhealthy_services.append(service_name)
        
        if unhealthy_services:
            self.logger.warning(f"Serviços não saudáveis: {unhealthy_services}")
            
            # Publica evento de erro para serviços não saudáveis
            error_event = ErrorEvent(
                event_id=str(uuid.uuid4()),
                event_type=EventType.ERROR_OCCURRED,
                timestamp=datetime.utcnow().isoformat(),
                source_service=ServiceName.MONITORING_SERVICE,
                correlation_id=str(uuid.uuid4()),
                priority=EventPriority.HIGH,
                error_code="SERVICE_UNHEALTHY",
                error_message=f"Serviços com problemas: {unhealthy_services}",
                context={"unhealthy_services": unhealthy_services}
            )
            
            # Publica assincronamente
            asyncio.create_task(self.event_manager.publish(error_event))
    
    def get_overall_health(self) -> Dict[str, Any]:
        """Retorna o status geral de saúde do sistema"""
        total_services = len(ServiceName)
        healthy_services = sum(
            1 for status in self.services_status.values()
            if status.get("status", {}).get("overall") == "healthy"
        )
        
        return {
            "overall_status": "healthy" if healthy_services == total_services else "degraded",
            "healthy_services": healthy_services,
            "total_services": total_services,
            "services": self.services_status
        }

"""
Circuit Breaker implementation for critical APIs
Follows the resilience patterns: timeout, retry, circuit breaker, fallback
"""
import time
import logging
from enum import Enum
from typing import Any, Callable, Optional, Dict
from dataclasses import dataclass, field
from datetime import datetime, timedelta
import asyncio
from functools import wraps
import json

logger = logging.getLogger(__name__)


class CircuitState(Enum):
    """Circuit breaker states"""
    CLOSED = "CLOSED"      # Normal operation, requests pass through
    OPEN = "OPEN"          # Circuit is open, requests fail fast
    HALF_OPEN = "HALF_OPEN"  # Testing if service is back


@dataclass
class CircuitBreakerConfig:
    """Configuration for circuit breaker"""
    failure_threshold: int = 5           # Number of failures before opening circuit
    reset_timeout: int = 30              # Seconds before trying half-open
    half_open_max_attempts: int = 3      # Max attempts in half-open state
    success_threshold: int = 2           # Successes needed to close circuit
    timeout: int = 10                    # Default timeout in seconds
    name: str = "default"                # Circuit breaker name for logging


@dataclass
class CircuitMetrics:
    """Metrics tracked by circuit breaker"""
    total_requests: int = 0
    successful_requests: int = 0
    failed_requests: int = 0
    consecutive_failures: int = 0
    consecutive_successes: int = 0
    last_failure_time: Optional[datetime] = None
    last_success_time: Optional[datetime] = None
    state_transitions: Dict[str, int] = field(default_factory=lambda: {
        "CLOSED_TO_OPEN": 0,
        "OPEN_TO_HALF_OPEN": 0,
        "HALF_OPEN_TO_CLOSED": 0,
        "HALF_OPEN_TO_OPEN": 0
    })


class CircuitBreaker:
    """Circuit breaker implementation for critical APIs"""
    
    def __init__(self, config: CircuitBreakerConfig):
        self.config = config
        self.state = CircuitState.CLOSED
        self.metrics = CircuitMetrics()
        self._lock = asyncio.Lock()
        self._half_open_attempts = 0
        
    async def execute(self, func: Callable, *args, **kwargs) -> Any:
        """
        Execute function with circuit breaker protection
        
        Args:
            func: Function to execute
            *args: Function arguments
            **kwargs: Function keyword arguments
            timeout: Override default timeout
            
        Returns:
            Function result
            
        Raises:
            CircuitBreakerError: If circuit is open
            TimeoutError: If operation times out
            Exception: Original function exception
        """
        async with self._lock:
            # Check circuit state
            if self.state == CircuitState.OPEN:
                await self._check_reset_timeout()
                
            if self.state == CircuitState.OPEN:
                self.metrics.total_requests += 1
                raise CircuitBreakerError(
                    f"Circuit '{self.config.name}' is OPEN. "
                    f"Last failure: {self.metrics.last_failure_time}"
                )
        
        # Execute with timeout
        timeout = kwargs.pop('timeout', self.config.timeout)
        
        try:
            if asyncio.iscoroutinefunction(func):
                result = await asyncio.wait_for(
                    func(*args, **kwargs), 
                    timeout=timeout
                )
            else:
                # Run sync function in thread pool
                loop = asyncio.get_event_loop()
                result = await asyncio.wait_for(
                    loop.run_in_executor(None, lambda: func(*args, **kwargs)),
                    timeout=timeout
                )
            
            await self._record_success()
            return result
            
        except asyncio.TimeoutError:
            await self._record_failure(TimeoutError(f"Timeout after {timeout}s"))
            raise TimeoutError(f"Operation timed out after {timeout} seconds")
            
        except Exception as e:
            await self._record_failure(e)
            raise
    
    async def _record_success(self):
        """Record successful execution"""
        async with self._lock:
            self.metrics.total_requests += 1
            self.metrics.successful_requests += 1
            self.metrics.consecutive_successes += 1
            self.metrics.consecutive_failures = 0
            self.metrics.last_success_time = datetime.utcnow()
            
            if self.state == CircuitState.HALF_OPEN:
                self._half_open_attempts += 1
                
                if self.metrics.consecutive_successes >= self.config.success_threshold:
                    self.state = CircuitState.CLOSED
                    self.metrics.state_transitions["HALF_OPEN_TO_CLOSED"] += 1
                    self._half_open_attempts = 0
                    logger.info(
                        f"Circuit '{self.config.name}' transitioned from HALF_OPEN to CLOSED",
                        extra={"circuit_name": self.config.name}
                    )
    
    async def _record_failure(self, error: Exception):
        """Record failed execution"""
        async with self._lock:
            self.metrics.total_requests += 1
            self.metrics.failed_requests += 1
            self.metrics.consecutive_failures += 1
            self.metrics.consecutive_successes = 0
            self.metrics.last_failure_time = datetime.utcnow()
            
            if self.state == CircuitState.CLOSED:
                if self.metrics.consecutive_failures >= self.config.failure_threshold:
                    self.state = CircuitState.OPEN
                    self.metrics.state_transitions["CLOSED_TO_OPEN"] += 1
                    logger.warning(
                        f"Circuit '{self.config.name}' transitioned from CLOSED to OPEN. "
                        f"Failure threshold ({self.config.failure_threshold}) reached.",
                        extra={
                            "circuit_name": self.config.name,
                            "error": str(error),
                            "consecutive_failures": self.metrics.consecutive_failures
                        }
                    )
            
            elif self.state == CircuitState.HALF_OPEN:
                self._half_open_attempts += 1
                self.state = CircuitState.OPEN
                self.metrics.state_transitions["HALF_OPEN_TO_OPEN"] += 1
                logger.warning(
                    f"Circuit '{self.config.name}' transitioned from HALF_OPEN to OPEN",
                    extra={"circuit_name": self.config.name, "error": str(error)}
                )
    
    async def _check_reset_timeout(self):
        """Check if reset timeout has elapsed and transition to HALF_OPEN"""
        if self.state == CircuitState.OPEN and self.metrics.last_failure_time:
            reset_time = self.metrics.last_failure_time + timedelta(
                seconds=self.config.reset_timeout
            )
            
            if datetime.utcnow() >= reset_time:
                self.state = CircuitState.HALF_OPEN
                self._half_open_attempts = 0
                self.metrics.state_transitions["OPEN_TO_HALF_OPEN"] += 1
                logger.info(
                    f"Circuit '{self.config.name}' transitioned from OPEN to HALF_OPEN",
                    extra={"circuit_name": self.config.name}
                )
    
    def get_status(self) -> Dict[str, Any]:
        """Get current circuit breaker status"""
        return {
            "name": self.config.name,
            "state": self.state.value,
            "metrics": {
                "total_requests": self.metrics.total_requests,
                "successful_requests": self.metrics.successful_requests,
                "failed_requests": self.metrics.failed_requests,
                "consecutive_failures": self.metrics.consecutive_failures,
                "consecutive_successes": self.metrics.consecutive_successes,
                "failure_rate": (
                    self.metrics.failed_requests / self.metrics.total_requests * 100
                    if self.metrics.total_requests > 0 else 0
                ),
                "state_transitions": self.metrics.state_transitions
            },
            "config": {
                "failure_threshold": self.config.failure_threshold,
                "reset_timeout": self.config.reset_timeout,
                "timeout": self.config.timeout
            },
            "timestamps": {
                "last_failure": (
                    self.metrics.last_failure_time.isoformat()
                    if self.metrics.last_failure_time else None
                ),
                "last_success": (
                    self.metrics.last_success_time.isoformat()
                    if self.metrics.last_success_time else None
                )
            }
        }


class CircuitBreakerError(Exception):
    """Exception raised when circuit breaker is open"""
    pass


# Circuit breaker registry for managing multiple breakers
class CircuitBreakerRegistry:
    """Registry for managing circuit breakers"""
    
    _instance = None
    _breakers: Dict[str, CircuitBreaker] = {}
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    @classmethod
    def register(cls, name: str, config: Optional[CircuitBreakerConfig] = None) -> CircuitBreaker:
        """Register or get a circuit breaker"""
        if name not in cls._breakers:
            config = config or CircuitBreakerConfig(name=name)
            cls._breakers[name] = CircuitBreaker(config)
        return cls._breakers[name]
    
    @classmethod
    def get(cls, name: str) -> Optional[CircuitBreaker]:
        """Get a circuit breaker by name"""
        return cls._breakers.get(name)
    
    @classmethod
    def get_all_status(cls) -> Dict[str, Dict[str, Any]]:
        """Get status of all circuit breakers"""
        return {
            name: breaker.get_status()
            for name, breaker in cls._breakers.items()
        }
    
    @classmethod
    def reset_all(cls):
        """Reset all circuit breakers to CLOSED state"""
        for breaker in cls._breakers.values():
            breaker.state = CircuitState.CLOSED
            breaker.metrics.consecutive_failures = 0
            breaker._half_open_attempts = 0


# Decorator for easy circuit breaker usage
def circuit_breaker(name: str, config: Optional[CircuitBreakerConfig] = None):
    """
    Decorator to apply circuit breaker to a function
    
    Usage:
        @circuit_breaker("database", CircuitBreakerConfig(timeout=5))
        async def query_database():
            ...
    """
    def decorator(func):
        breaker = CircuitBreakerRegistry.register(name, config)
        
        @wraps(func)
        async def async_wrapper(*args, **kwargs):
            return await breaker.execute(func, *args, **kwargs)
        
        @wraps(func)
        def sync_wrapper(*args, **kwargs):
            # For sync functions, we need to run in async context
            async def async_func():
                return func(*args, **kwargs)
            
            # Get current event loop or create new one
            try:
                loop = asyncio.get_event_loop()
            except RuntimeError:
                loop = asyncio.new_event_loop()
                asyncio.set_event_loop(loop)
            
            return loop.run_until_complete(breaker.execute(async_func))
        
        return async_wrapper if asyncio.iscoroutinefunction(func) else sync_wrapper
    
    return decorator

const Redis = require('ioredis');
const logger = require('../../utils/logger');
const metrics = require('../../utils/metrics');
const redisConfig = require('../../config/redis-config');

class RedisClient {
  constructor() {
    this.client = null;
    this.subscriber = null;
    this.isConnected = false;
    this.connectionAttempts = 0;
    this.maxConnectionAttempts = 5;
    this.circuitBreakerState = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
    this.failureCount = 0;
    this.failureThreshold = 3;
    this.resetTimeout = 10000; // 10 segundos
    this.lastFailureTime = null;
  }

  async connect() {
    if (this.isConnected && this.client?.status === 'ready') {
      return this.client;
    }

    // Circuit Breaker: Se OPEN, verifica se pode tentar reconexão
    if (this.circuitBreakerState === 'OPEN') {
      const now = Date.now();
      if (now - this.lastFailureTime > this.resetTimeout) {
        this.circuitBreakerState = 'HALF_OPEN';
        logger.info('Circuit breaker moving to HALF_OPEN state');
      } else {
        throw new Error('Circuit breaker OPEN - Redis não disponível');
      }
    }

    try {
      logger.info('Conectando ao Redis...', {
        host: redisConfig.getConfig().host,
        port: redisConfig.getConfig().port,
      });

      this.client = new Redis(redisConfig.getConfig());
      this.subscriber = new Redis(redisConfig.getConfig());

      // Configurar event listeners
      this.setupEventListeners(this.client, 'client');
      this.setupEventListeners(this.subscriber, 'subscriber');

      // Aguardar conexão
      await Promise.race([
        Promise.all([this.client.ping(), this.subscriber.ping()]),
        new Promise((_, reject) =>
          setTimeout(() => reject(new Error('Redis connection timeout')), 10000)
        ),
      ]);

      this.isConnected = true;
      this.connectionAttempts = 0;
      this.circuitBreakerState = 'CLOSED';
      this.failureCount = 0;

      logger.info('Redis conectado com sucesso');
      metrics.increment('redis.connection.success');

      return this.client;
    } catch (error) {
      this.handleConnectionError(error);
      throw error;
    }
  }

  setupEventListeners(redisInstance, instanceType) {
    redisInstance.on('connect', () => {
      logger.debug(`Redis ${instanceType} conectando`);
    });

    redisInstance.on('ready', () => {
      logger.info(`Redis ${instanceType} pronto`);
      metrics.gauge(`redis.${instanceType}.status`, 1);
    });

    redisInstance.on('error', (error) => {
      logger.error(`Redis ${instanceType} error:`, error);
      metrics.increment(`redis.${instanceType}.errors`);
      this.handleConnectionError(error);
    });

    redisInstance.on('close', () => {
      logger.warn(`Redis ${instanceType} conexão fechada`);
      this.isConnected = false;
      metrics.gauge(`redis.${instanceType}.status`, 0);
    });

    redisInstance.on('reconnecting', (delay) => {
      logger.warn(`Redis ${instanceType} reconectando em ${delay}ms`);
      metrics.increment(`redis.${instanceType}.reconnections`);
    });
  }

  handleConnectionError(error) {
    this.isConnected = false;
    this.connectionAttempts++;
    this.failureCount++;
    this.lastFailureTime = Date.now();

    metrics.increment('redis.connection.errors');

    // Atualizar estado do circuit breaker
    if (this.failureCount >= this.failureThreshold) {
      this.circuitBreakerState = 'OPEN';
      logger.error('Circuit breaker OPEN - Muitas falhas no Redis');
    }

    // Tentar reconexão automática se ainda não atingiu o limite
    if (this.connectionAttempts < this.maxConnectionAttempts) {
      setTimeout(() => {
        logger.info('Tentando reconexão automática ao Redis...');
        this.connect().catch(() => {
          // Ignorar erro na reconexão automática
        });
      }, Math.min(1000 * Math.pow(2, this.connectionAttempts), 10000));
    }
  }

  async getClient() {
    if (!this.isConnected) {
      await this.connect();
    }
    return this.client;
  }

  async getSubscriber() {
    if (!this.isConnected) {
      await this.connect();
    }
    return this.subscriber;
  }

  async disconnect() {
    if (this.client) {
      await this.client.quit();
    }
    if (this.subscriber) {
      await this.subscriber.quit();
    }
    this.isConnected = false;
    logger.info('Redis desconectado');
  }

  getStatus() {
    return {
      isConnected: this.isConnected,
      circuitBreakerState: this.circuitBreakerState,
      connectionAttempts: this.connectionAttempts,
      failureCount: this.failureCount,
      clientStatus: this.client?.status,
      subscriberStatus: this.subscriber?.status,
    };
  }
}

const redisClient = require('./client');
const logger = require('../../utils/logger');
const metrics = require('../../utils/metrics');

class RedisSubscriber {
  constructor() {
    this.subscriptions = new Map();
    this.channelPrefix = process.env.REDIS_CHANNEL_PREFIX || 'diyapp';
    this.isSubscribed = false;
  }

  async subscribe(channel, handler, options = {}) {
    const fullChannel = options.exactChannel ? channel : `${this.channelPrefix}:${channel}`;
    
    try {
      const subscriber = await redisClient.getSubscriber();
      
      if (!this.isSubscribed) {
        await this.setupGlobalListener(subscriber);
        this.isSubscribed = true;
      }

      await subscriber.subscribe(fullChannel);
      
      this.subscriptions.set(fullChannel, {
        handler,
        pattern: options.pattern || false,
        options,
      });

      logger.info(`Inscrito no canal: ${fullChannel}`);
      metrics.gauge('redis.subscriptions.count', this.subscriptions.size);

      return { success: true, channel: fullChannel };
    } catch (error) {
      logger.error(`Erro ao inscrever no canal ${fullChannel}:`, error);
      metrics.increment('redis.subscribe.errors');
      throw error;
    }
  }

  async subscribePattern(pattern, handler) {
    try {
      const subscriber = await redisClient.getSubscriber();
      
      if (!this.isSubscribed) {
        await this.setupGlobalListener(subscriber);
        this.isSubscribed = true;
      }

      await subscriber.psubscribe(pattern);
      
      this.subscriptions.set(pattern, {
        handler,
        pattern: true,
      });

      logger.info(`Inscrito no padrão: ${pattern}`);
      return { success: true, pattern };
    } catch (error) {
      logger.error(`Erro ao inscrever no padrão ${pattern}:`, error);
      throw error;
    }
  }

  async setupGlobalListener(subscriber) {
    subscriber.on('message', async (channel, message) => {
      await this.handleMessage(channel, message, false);
    });

    subscriber.on('pmessage', async (pattern, channel, message) => {
      await this.handleMessage(channel, message, true, pattern);
    });

    subscriber.on('error', (error) => {
      logger.error('Erro no subscriber:', error);
    });
  }

  async handleMessage(channel, message, isPattern = false, pattern = null) {
    const startTime = Date.now();
    
    try {
      const parsedMessage = JSON.parse(message);
      
      // Encontrar handler apropriado
      let handlerInfo;
      if (isPattern && pattern) {
        handlerInfo = this.subscriptions.get(pattern);
      } else {
        handlerInfo = this.subscriptions.get(channel);
      }

      if (!handlerInfo) {
        logger.warn(`Nenhum handler encontrado para canal: ${channel}`);
        return;
      }

      // Executar handler
      await handlerInfo.handler(parsedMessage, channel);

      const duration = Date.now() - startTime;
      metrics.timing('redis.message.processing.duration', duration);
      metrics.increment('redis.message.processed.success');

      logger.debug(`Mensagem processada: ${parsedMessage.type}`, {
        channel,
        messageId: parsedMessage.id,
        duration,
      });
    } catch (error) {
      const duration = Date.now() - startTime;
      metrics.timing('redis.message.processing.duration', duration);
      metrics.increment('redis.message.processed.errors');

      logger.error(`Erro ao processar mensagem do canal ${channel}:`, error, {
        rawMessage: message.substring(0, 200),
      });

      // Dead letter queue
      await this.sendToDLQ(channel, message, error);
    }
  }

  async sendToDLQ(channel, message, error) {
    try {
      const publisher = require('./publisher');
      await publisher.publish('dlq', {
        channel,
        message,
        error: error.message,
        timestamp: new Date().toISOString(),
      }, { channel: `${this.channelPrefix}:dlq` });
    } catch (dlqError) {
      logger.error('Erro ao enviar para DLQ:', dlqError);
    }
  }

  async unsubscribe(channel) {
    try {
      const subscriber = await redisClient.getSubscriber();
      await subscriber.unsubscribe(channel);
      
      this.subscriptions.delete(channel);
      logger.info(`Desinscrito do canal: ${channel}`);
      
      return { success: true, channel };
    } catch (error) {
      logger.error(`Erro ao desinscrever do canal ${channel}:`, error);
      throw error;
    }
  }

  async unsubscribeAll() {
    const channels = Array.from(this.subscriptions.keys());
    const results = [];
    
    for (const channel of channels) {
      try {
        await this.unsubscribe(channel);
        results.push({ channel, success: true });
      } catch (error) {
        results.push({ channel, success: false, error: error.message });
      }
    }
    
    return results;
  }

  getSubscriptions() {
    return Array.from(this.subscriptions.entries()).map(([channel, info]) => ({
      channel,
      pattern: info.pattern,
      options: info.options,
    }));
  }
}

// AI Ops - Health Monitoring System for DIYAPP V12 Core
// Monitors /health endpoint and implements alerting rules for Stories 1 and 5

const axios = require('axios');
const EventEmitter = require('events');
const winston = require('winston');
const fs = require('fs').promises;
const path = require('path');

class HealthMonitor extends EventEmitter {
    constructor(config) {
        super();
        this.config = {
            baseURL: 'http://localhost:3000',
            checkInterval: 30000, // 30 seconds
            alertThresholds: {
                responseTime: 1000, // 1 second
                errorRate: 0.05, // 5%
                consecutiveFailures: 3,
                memoryUsage: 0.8, // 80%
                cpuUsage: 0.7 // 70%
            },
            alertCooldown: 300000, // 5 minutes
            ...config
        };
        
        this.metrics = {
            checks: 0,
            successes: 0,
            failures: 0,
            responseTimes: [],
            lastAlertTime: {},
            healthHistory: []
        };
        
        this.setupLogger();
        this.initializeAlertRules();
    }
    
    setupLogger() {
        const logsDir = path.join(__dirname, '../../logs');
        try {
            require('fs').mkdirSync(logsDir, { recursive: true });
        } catch (e) {}
        
        this.logger = winston.createLogger({
            level: 'info',
            format: winston.format.combine(
                winston.format.timestamp(),
                winston.format.json()
            ),
            transports: [
                new winston.transports.File({ 
                    filename: path.join(logsDir, 'health-monitor.log'),
                    maxsize: 5242880, // 5MB
                    maxFiles: 5
                }),
                new winston.transports.Console({
                    format: winston.format.simple()
                })
            ]
        });
    }
    
    initializeAlertRules() {
        // Story 1: System Health Monitoring
        this.alertRules = {
            highResponseTime: {
                name: 'HIGH_RESPONSE_TIME',
                condition: (healthData) => healthData.responseTime > this.config.alertThresholds.responseTime,
                message: (healthData) => `Response time ${healthData.responseTime}ms exceeds threshold of ${this.config.alertThresholds.responseTime}ms`,
                severity: 'WARNING',
                story: 1
            },
            highErrorRate: {
                name: 'HIGH_ERROR_RATE',
                condition: (healthData) => {
                    const totalChecks = this.metrics.checks;
                    const failures = this.metrics.failures;
                    return totalChecks > 10 && (failures / totalChecks) > this.config.alertThresholds.errorRate;
                },
                message: () => `Error rate ${((this.metrics.failures / this.metrics.checks) * 100).toFixed(1)}% exceeds threshold of ${this.config.alertThresholds.errorRate * 100}%`,
                severity: 'ERROR',
                story: 1
            },
            consecutiveFailures: {
                name: 'CONSECUTIVE_FAILURES',
                condition: (healthData, consecutiveFails) => consecutiveFails >= this.config.alertThresholds.consecutiveFailures,
                message: (healthData, consecutiveFails) => `${consecutiveFails} consecutive health check failures`,
                severity: 'CRITICAL',
                story: 1
            },
            highMemoryUsage: {
                name: 'HIGH_MEMORY_USAGE',
                condition: (healthData) => healthData.memory && healthData.memory.usage > this.config.alertThresholds.memoryUsage,
                message: (healthData) => `Memory usage ${(healthData.memory.usage * 100).toFixed(1)}% exceeds threshold of ${this.config.alertThresholds.memoryUsage * 100}%`,
                severity: 'WARNING',
                story: 5
            },
            highCpuUsage: {
                name: 'HIGH_CPU_USAGE',
                condition: (healthData) => healthData.cpu && healthData.cpu.usage > this.config.alertThresholds.cpuUsage,
                message: (healthData) => `CPU usage ${(healthData.cpu.usage * 100).toFixed(1)}% exceeds threshold of ${this.config.alertThresholds.cpuUsage * 100}%`,
                severity: 'WARNING',
                story: 5
            },
            serviceUnavailable: {
                name: 'SERVICE_UNAVAILABLE',
                condition: (healthData) => !healthData.status || healthData.status !== 'healthy',
                message: (healthData) => `Service status: ${healthData.status || 'unknown'}`,
                severity: 'CRITICAL',
                story: 1
            }
        };
    }
    
    async start() {
        this.logger.info('Starting Health Monitor for DIYAPP V12 Core');
        this.logger.info(`Monitoring endpoint: ${this.config.baseURL}/health`);
        this.logger.info(`Check interval: ${this.config.checkInterval}ms`);
        
        this.monitoringInterval = setInterval(() => {
            this.checkHealth();
        }, this.config.checkInterval);
        
        // Initial check
        await this.checkHealth();
        
        return this;
    }
    
    stop() {
        if (this.monitoringInterval) {
            clearInterval(this.monitoringInterval);
            this.logger.info('Health Monitor stopped');
        }
    }
    
    async checkHealth() {
        const startTime = Date.now();
        let healthData = null;
        let consecutiveFails = 0;
        
        try {
            const response = await axios.get(`${this.config.baseURL}/health`, {
                timeout: 5000,
                headers: {
                    'User-Agent': 'DIYAPP-HealthMonitor/1.0'
                }
            });
            
            const responseTime = Date.now() - startTime;
            healthData = {
                ...response.data,
                responseTime,
                timestamp: new Date().toISOString(),
                success: true
            };
            
            this.metrics.successes++;
            this.metrics.responseTimes.push(responseTime);
            
            // Keep only last 100 response times for rolling average
            if (this.metrics.responseTimes.length > 100) {
                this.metrics.responseTimes.shift();
            }
            
            this.logger.debug('Health check successful', { responseTime, status: healthData.status });
            
        } catch (error) {
            const responseTime = Date.now() - startTime;
            healthData = {
                status: 'unhealthy',
                error: error.message,
                responseTime,
                timestamp: new Date().toISOString(),
                success: false
            };
            
            this.metrics.failures++;
            consecutiveFails = this.calculateConsecutiveFailures();
            
            this.logger.warn('Health check failed', { 
                error: error.message, 
                responseTime,
                consecutiveFails 
            });
        }
        
        this.metrics.checks++;
        this.metrics.healthHistory.push(healthData);
        
        // Keep only last 1000 health checks in history
        if (this.metrics.healthHistory.length > 1000) {
            this.metrics.healthHistory.shift();
        }
        
        // Evaluate alert rules
        this.evaluateAlertRules(healthData, consecutiveFails);
        
        // Emit health data for dashboard
        this.emit('healthUpdate', {
            ...healthData,
            metrics: this.getSummaryMetrics()
        });
        
        return healthData;
    }
    
    calculateConsecutiveFailures() {
        let consecutive = 0;
        for (let i = this.metrics.healthHistory.length - 1; i >= 0; i--) {
            if (this.metrics.healthHistory[i].success) break;
            consecutive++;
        }
        return consecutive;
    }
    
    evaluateAlertRules(healthData, consecutiveFails) {
        const now = Date.now();
        
        for (const [ruleName, rule] of Object.entries(this.alertRules)) {
            if (rule.condition(healthData, consecutiveFails)) {
                const lastAlertTime = this.metrics.lastAlertTime[ruleName] || 0;
                
                // Check cooldown period
                if (now - lastAlertTime > this.config.alertCooldown) {
                    const alertMessage = rule.message(healthData, consecutiveFails);
                    const alert = {
                        id: `${ruleName}_${now}`,
                        rule: ruleName,
                        name: rule.name,
                        message: alertMessage,
                        severity: rule.severity,
                        story: rule.story,
                        timestamp: new Date().toISOString(),
                        data: healthData,
                        metrics: this.getSummaryMetrics()
                    };
                    
                    this.metrics.lastAlertTime[ruleName] = now;
                    
                    // Log alert
                    this.logger[rule.severity.toLowerCase()](`ALERT: ${rule.name}`, alert);
                    
                    // Emit alert event
                    this.emit('alert', alert);
                    
                    // Story 5: Resource monitoring alerts
                    if (rule.story === 5) {
                        this.emit('resourceAlert', alert);
                    }
                }
            }
        }
    }
    
    getSummaryMetrics() {
        const responseTimes = this.metrics.responseTimes;
        const avgResponseTime = responseTimes.length > 0 
            ? responseTimes.reduce((a, b) => a + b, 0) / responseTimes.length 
            : 0;
        
        const sortedTimes = [...responseTimes].sort((a, b) => a - b);
        const p95Index = Math.floor(sortedTimes.length * 0.95);
        const p95ResponseTime = sortedTimes[p95Index] || 0;
        
        return {
            totalChecks: this.metrics.checks,
            successRate: this.metrics.checks > 0 ? (this.metrics.successes / this.metrics.checks) * 100 : 0,
            avgResponseTime: Math.round(avgResponseTime),
            p95ResponseTime,
            currentFailures: this.calculateConsecutiveFailures(),
            lastCheck: this.metrics.healthHistory[this.metrics.healthHistory.length - 1]
        };
    }
    
    async getHealthHistory(limit = 100) {
        return this.metrics.healthHistory.slice(-limit);
    }
    
    async getAlerts(limit = 50) {
        // In a real system, this would query from a database
        // For now, we'll simulate by checking recent health data
        const recentHistory = await this.getHealthHistory(limit * 10);
        const alerts = [];
        
        for (const healthData of recentHistory) {
            for (const [ruleName, rule] of Object.entries(this.alertRules)) {
                if (rule.condition(healthData, this.calculateConsecutiveFailures())) {
                    alerts.push({
                        rule: ruleName,
                        message: rule.message(healthData),
                        severity: rule.severity,
                        timestamp: healthData.timestamp,
                        data: healthData
                    });
                }
            }
        }
        
        return alerts.slice(-limit);
    }
}
