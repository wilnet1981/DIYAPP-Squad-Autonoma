```typescript
// Arquivo: services/template-service/src/index.ts
import { Server, ServerCredentials } from '@grpc/grpc-js';
import { Kafka } from 'kafkajs';
import { config } from './config';
import { logger } from './utils/logger';
import { healthCheck } from './health';
import { ServiceErrorHandler } from './middleware/error-handler';
import { MetricsCollector } from './monitoring/metrics';
import { initDatabase } from './database';
import { loadProtoDefinitions } from './grpc/proto-loader';

class TemplateService {
  private grpcServer: Server;
  private kafkaClient: Kafka;
  private metrics: MetricsCollector;
  private isShuttingDown = false;

  constructor() {
    this.grpcServer = new Server();
    this.kafkaClient = new Kafka({
      clientId: config.kafka.clientId,
      brokers: config.kafka.brokers,
      ssl: config.kafka.ssl,
      sasl: config.kafka.sasl,
    });
    this.metrics = new MetricsCollector();
  }

  async start(): Promise<void> {
    try {
      logger.info('Starting Template Service...');
      
      // 1. Initialize dependencies
      await this.initializeDependencies();
      
      // 2. Load gRPC proto definitions
      const protoPackage = await loadProtoDefinitions();
      
      // 3. Register gRPC services
      this.registerGrpcServices(protoPackage);
      
      // 4. Start gRPC server
      await this.startGrpcServer();
      
      // 5. Connect to Kafka
      await this.connectToKafka();
      
      // 6. Start health check endpoint
      await healthCheck.start();
      
      // 7. Start metrics server
      await this.metrics.start();
      
      logger.info(`Template Service started on port ${config.grpc.port}`);
      this.setupGracefulShutdown();
    } catch (error) {
      logger.error('Failed to start service:', error);
      process.exit(1);
    }
  }

  private async initializeDependencies(): Promise<void> {
    await initDatabase();
    // Add other dependency initializations here
  }

  private registerGrpcServices(protoPackage: any): void {
    // Implement service registration based on proto definitions
    // Example:
    // this.grpcServer.addService(protoPackage.TemplateService.service, {
    //   templateMethod: this.templateMethod.bind(this),

  private async startGrpcServer(): Promise<void> {
    return new Promise((resolve, reject) => {
      this.grpcServer.bindAsync(
        `0.0.0.0:${config.grpc.port}`,
        ServerCredentials.createInsecure(),
        (error, port) => {
          if (error) {
            reject(error);
          } else {
            this.grpcServer.start();
            resolve();
          }
        }
      );
    });
  }

  private async connectToKafka(): Promise<void> {
    const admin = this.kafkaClient.admin();
    await admin.connect();
    
    // Create topics if they don't exist
    const topics = config.kafka.topics;
    await admin.createTopics({
      topics: topics.map(topic => ({
        topic,
        numPartitions: 3,
        replicationFactor: 2,
      })),
    });
    
    await admin.disconnect();
  }

  private setupGracefulShutdown(): void {
    const signals: NodeJS.Signals[] = ['SIGTERM', 'SIGINT'];
    
    signals.forEach(signal => {
      process.on(signal, async () => {
        if (this.isShuttingDown) return;
        this.isShuttingDown = true;
        
        logger.info(`Received ${signal}, starting graceful shutdown...`);
        
        // 1. Stop accepting new connections
        this.grpcServer.tryShutdown(() => {
          logger.info('gRPC server stopped');
        });
        
        // 2. Stop health checks
        await healthCheck.stop();
        
        // 3. Stop metrics
        await this.metrics.stop();
        
        // 4. Close Kafka connections
        // Implement Kafka consumer/producer cleanup
        
        logger.info('Graceful shutdown completed');
        process.exit(0);
      });
    });
  }
}
```

```yaml
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

  # Job 3: Deploy to Staging

    needs: build-and-push

    environment: production
    
    steps:

          kubectl set image deployment/diyapp \
```

class SecurityManager:
    """Gerenciador de segurança com JWT e hash"""
    
    def __init__(self):
        self.secret_key = settings.SECRET_KEY
        self.algorithm = settings.ALGORITHM
    
    def verify_password(self, plain_password: str, hashed_password: str) -> bool:
        """Verifica senha com hash seguro"""
        return pwd_context.verify(plain_password, hashed_password)
    
    def get_password_hash(self, password: str) -> str:
        """Gera hash seguro para senha"""
        return pwd_context.hash(password)
    
    def create_access_token(
        self, 
        data: Dict[str, Any], 
        expires_delta: Optional[timedelta] = None
    ) -> str:
        """Cria token JWT de acesso"""
        to_encode = data.copy()
        if expires_delta:
            expire = datetime.utcnow() + expires_delta
        else:
            expire = datetime.utcnow() + timedelta(
                minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES
            )
        
        to_encode.update({"exp": expire, "type": "access"})
        encoded_jwt = jwt.encode(
            to_encode, 
            self.secret_key, 
            algorithm=self.algorithm
        )
        return encoded_jwt
    
    def create_refresh_token(
        self, 
        data: Dict[str, Any]
    ) -> str:
        """Cria token JWT de refresh"""
        to_encode = data.copy()
        expire = datetime.utcnow() + timedelta(
            days=settings.REFRESH_TOKEN_EXPIRE_DAYS
        )
        
        to_encode.update({"exp": expire, "type": "refresh"})
        encoded_jwt = jwt.encode(
            to_encode, 
            self.secret_key, 
            algorithm=self.algorithm
        )
        return encoded_jwt
    
    def verify_token(self, token: str) -> Dict[str, Any]:
        """Verifica e decodifica token JWT"""
        try:
            payload = jwt.decode(
                token, 
                self.secret_key, 
                algorithms=[self.algorithm]
            )
            return payload
        except JWTError as e:
            logger.warning(f"Token verification failed: {str(e)}")
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Could not validate credentials",
                headers={"WWW-Authenticate": "Bearer"},
            )
    
    def get_current_user(self, token: str) -> Dict[str, Any]:
        """Obtém usuário atual a partir do token"""
        payload = self.verify_token(token)
        
        # Verifica tipo do token
        if payload.get("type") != "access":
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid token type"
            )
        
        # Verifica se usuário existe (simulação - implementar com DB)
        user_id = payload.get("sub")
        if not user_id:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid token payload"
            )
        
        return {"user_id": user_id, "scopes": payload.get("scopes", [])}

import redis
from redis.exceptions import RedisError
import json
from typing import Optional, Any
import logging
from functools import wraps
import time

logger = logging.getLogger(__name__)

class CircuitBreaker:
    """Circuit breaker para serviços externos"""
    
    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 60):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.failures = 0
        self.last_failure_time = 0
        self.state = "CLOSED"  # CLOSED, OPEN, HALF_OPEN
    
    def call(self, func, *args, **kwargs):
        """Executa função com circuit breaker"""
        if self.state == "OPEN":
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = "HALF_OPEN"
            else:
                raise Exception("Circuit breaker is OPEN")
        
        try:
            result = func(*args, **kwargs)
            if self.state == "HALF_OPEN":
                self.state = "CLOSED"
                self.failures = 0
            return result
        except Exception as e:
            self.failures += 1
            self.last_failure_time = time.time()
            
            if self.failures >= self.failure_threshold:
                self.state = "OPEN"
            
            logger.error(f"Circuit breaker failure: {str(e)}")
            raise

class CacheManager:
    """Gerenciador de cache com Redis e circuit breaker"""
    
    def __init__(self, redis_url: str):
        self.redis_url = redis_url
        self.client = None
        self.circuit_breaker = CircuitBreaker()
        self._connect()
    
    def _connect(self):
        """Conecta ao Redis com retry"""
        try:
            self.client = redis.from_url(
                self.redis_url,
                socket_connect_timeout=2,
                socket_timeout=2,
                retry_on_timeout=True,
                max_connections=10
            )
            # Testa conexão
            self.client.ping()
            logger.info("Redis connected successfully")
        except RedisError as e:
            logger.error(f"Redis connection failed: {str(e)}")
            self.client = None
    
    def get(self, key: str) -> Optional[Any]:
        """Obtém valor do cache"""
        if not self.client:
            return None
        
        try:
            value = self.circuit_breaker.call(self.client.get, key)
            if value:
                return json.loads(value)
            return None
        except Exception as e:
            logger.warning(f"Cache get failed: {str(e)}")
            return None
    
    def set(self, key: str, value: Any, ttl: int = None) -> bool:
        """Define valor no cache"""
        if not self.client:
            return False
        
        try:
            serialized = json.dumps(value)
            ttl = ttl or settings.REDIS_CACHE_TTL
            return self.circuit_breaker.call(
                self.client.setex, key, ttl, serialized
            )
        except Exception as e:
            logger.warning(f"Cache set failed: {str(e)}")
            return False
    
    def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        if not self.client:
            return False
        
        try:
            return self.circuit_breaker.call(self.client.delete, key) > 0
        except Exception as e:
            logger.warning(f"Cache delete failed: {str(e)}")
            return False
    
    def cache_decorator(self, ttl: int = 300):
        """Decorator para cache automático"""
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                # Gera chave única baseada na função e argumentos
                key_parts = [func.__module__, func.__name__]
                key_parts.extend([str(arg) for arg in args])
                key_parts.extend([f"{k}:{v}" for k, v in kwargs.items()])
                cache_key = f"cache:{':'.join(key_parts)}"
                
                # Tenta obter do cache
                cached = self.get(cache_key)
                if cached is not None:
                    return cached
                
                # Executa função e armazena no cache
                result = func(*args, **kwargs)
                self.set(cache_key, result, ttl)
                return result
            return wrapper
        return decorator

## 1. Web Application Firewall (WAF) - Middleware Avançado

**src/security/waf/middleware.js**
```javascript
const { createHash } = require('crypto');
const rules = require('./rules');
const rateLimiter = require('./rate-limiter');

class WAFMiddleware {
    constructor(config = {}) {
        this.config = {
            blockThreshold: 10,
            ipWhitelist: [],
            enableLogging: true,
            ...config
        };
        
        this.offenseRegistry = new Map();
        this.requestCache = new Map();
    }

    async inspectRequest(req, res, next) {
        const clientIP = this._getClientIP(req);
        const requestId = createHash('sha256')
            .update(`${clientIP}-${Date.now()}-${Math.random()}`)
            .digest('hex');

        // Verificar whitelist
        if (this.config.ipWhitelist.includes(clientIP)) {
            return next();
        }

        // Rate limiting por IP
        const isRateLimited = await rateLimiter.check(clientIP, req.path);
        if (isRateLimited) {
            return this._blockRequest(req, res, 'RATE_LIMIT_EXCEEDED', 429);
        }

        // Análise STRIDE
        const threats = await this._analyzeThreats(req, clientIP);

        if (threats.length > 0) {
            this._logThreats(requestId, clientIP, threats, req);
            
            // Incrementar contador de ofensas
            const offenseCount = this.offenseRegistry.get(clientIP) || 0;
            this.offenseRegistry.set(clientIP, offenseCount + threats.length);

            // Bloquear se exceder threshold
            if (offenseCount + threats.length >= this.config.blockThreshold) {
                return this._blockRequest(req, res, 'BLOCK_THRESHOLD_EXCEEDED', 403);
            }

            // Para ameaças críticas, bloquear imediatamente
            const criticalThreats = threats.filter(t => t.severity === 'CRITICAL');
            if (criticalThreats.length > 0) {
                return this._blockRequest(req, res, 'CRITICAL_THREAT_DETECTED', 403);
            }
        }

        // Headers de segurança
        this._setSecurityHeaders(res);

        // Adicionar contexto de segurança à request
        req.securityContext = {
            requestId,
            clientIP,
            threatsDetected: threats.length,
            isSuspicious: threats.length > 0
        };

        next();
    }

    async _analyzeThreats(req, clientIP) {
        const threats = [];

        // 1. SQL Injection
        if (this._detectSQLInjection(req)) {
            threats.push({
                type: 'SQL_INJECTION',
                severity: 'CRITICAL',
                vector: 'INPUT',
                payload: JSON.stringify(req.body)
            });
        }

        // 2. XSS
        if (this._detectXSS(req)) {
            threats.push({
                type: 'XSS',
                severity: 'HIGH',
                vector: 'INPUT',
                payload: JSON.stringify(req.query)
            });
        }

        // 3. Path Traversal
        if (this._detectPathTraversal(req)) {
            threats.push({
                type: 'PATH_TRAVERSAL',
                severity: 'HIGH',
                vector: 'PATH',
                payload: req.path
            });
        }

        // 4. Command Injection
        if (this._detectCommandInjection(req)) {
            threats.push({
                type: 'COMMAND_INJECTION',
                severity: 'CRITICAL',
                vector: 'INPUT',
                payload: JSON.stringify(req.body)
            });
        }

        // 5. SSRF
        if (this._detectSSRF(req)) {
            threats.push({
                type: 'SSRF',
                severity: 'HIGH',
                vector: 'INPUT',
                payload: req.body?.url || ''
            });
        }

        return threats;
    }

    _detectSQLInjection(req) {
        const sqlPatterns = [
            /(\%27)|(\')|(\-\-)|(\%23)|(#)/i,
            /((\%3D)|(=))[^\n]*((\%27)|(\')|(\-\-)|(\%3B)|(;))/i,
            /\w*((\%27)|(\'))((\%6F)|o|(\%4F))((\%72)|r|(\%52))/i,
            /((\%27)|(\'))union/i,
            /exec(\s|\+)+(s|x)p\w+/i,
            /sleep\s*\(\s*[\d]+\s*\)/i,
            /benchmark\s*\(/i
        ];

        const checkPayload = (payload) => {
            if (typeof payload === 'string') {
                return sqlPatterns.some(pattern => pattern.test(payload));
            } else if (typeof payload === 'object') {
                return Object.values(payload).some(value => 
                    typeof value === 'string' && sqlPatterns.some(pattern => pattern.test(value))
                );
            }
            return false;
        };

        return checkPayload(req.query) || checkPayload(req.body) || checkPayload(req.params);
    }

    _detectXSS(req) {
        const xssPatterns = [
            /<script\b[^>]*>([\s\S]*?)<\/script>/i,
            /javascript:/i,
            /on\w+\s*=/i,
            /<\s*iframe/i,
            /<\s*object/i,
            /<\s*embed/i,
            /<\s*applet/i,
            /<\s*meta/i,
            /<\s*link/i,
            /expression\s*\(/i
        ];

        const checkPayload = (payload) => {
            if (typeof payload === 'string') {
                return xssPatterns.some(pattern => pattern.test(payload));
            } else if (typeof payload === 'object') {
                return Object.values(payload).some(value => 
                    typeof value === 'string' && xssPatterns.some(pattern => pattern.test(value))
                );
            }
            return false;
        };

        return checkPayload(req.query) || checkPayload(req.body) || checkPayload(req.params);
    }

    _detectPathTraversal(req) {
        const traversalPatterns = [
            /\.\.\//,
            /\.\.\\/,
            /\/etc\/passwd/,
            /\/proc\/self/,
            /\\windows\\system32/,
            /\.\.%2f/i,
            /\.\.%5c/i
        ];

        return traversalPatterns.some(pattern => 
            pattern.test(req.path) || 
            pattern.test(JSON.stringify(req.query)) ||
            pattern.test(JSON.stringify(req.body))
        );
    }

    _detectCommandInjection(req) {
        const commandPatterns = [
            /;\s*\w+/,
            /\|\s*\w+/,
            /&\s*\w+/,
            /\$\s*\(/,
            /`/,
            /\|\|/,
            /&&/,
            /nc\s+-/,
            /wget\s+/,
            /curl\s+/
        ];

        const checkPayload = (payload) => {
            if (typeof payload === 'string') {
                return commandPatterns.some(pattern => pattern.test(payload));
            } else if (typeof payload === 'object') {
                return Object.values(payload).some(value => 
                    typeof value === 'string' && commandPatterns.some(pattern => pattern.test(value))
                );
            }
            return false;
        };

        return checkPayload(req.query) || checkPayload(req.body);
    }

    _detectSSRF(req) {
        const ssrfPatterns = [
            /^http:\/\/localhost/,
            /^http:\/\/127\.0\.0\.1/,
            /^http:\/\/0\.0\.0\.0/,
            /^http:\/\/\[::1\]/,
            /^http:\/\/169\.254\.169\.254/, // AWS metadata
            /^http:\/\/metadata\.google\.internal/, // GCP metadata
            /^file:\/\//,
            /^gopher:\/\//,
            /^dict:\/\//
        ];

        const url = req.body?.url || req.query?.url;
        if (typeof url === 'string') {
            return ssrfPatterns.some(pattern => pattern.test(url.toLowerCase()));
        }
        return false;
    }

    _getClientIP(req) {
        return req.headers['x-forwarded-for'] || 
               req.connection.remoteAddress || 
               req.socket.remoteAddress || 
               req.connection.socket?.remoteAddress || 
               '0.0.0.0';
    }

    _setSecurityHeaders(res) {
        res.setHeader('X-Content-Type-Options', 'nosniff');
        res.setHeader('X-Frame-Options', 'DENY');
        res.setHeader('X-XSS-Protection', '1; mode=block');
        res.setHeader('Referrer-Policy', 'strict-origin-when-cross-origin');
        res.setHeader('Permissions-Policy', 'geolocation=(), microphone=(), camera=()');
    }

    _blockRequest(req, res, reason, statusCode = 403) {
        const clientIP = this._getClientIP(req);
        
        // Log do bloqueio
        console.warn(`[WAF BLOCK] IP: ${clientIP}, Path: ${req.path}, Reason: ${reason}`);
        
        // Enviar resposta de bloqueio
        res.status(statusCode).json({
            error: 'Access denied',
            reason: reason,
            requestId: createHash('sha256')
                .update(`${clientIP}-${Date.now()}`)
                .digest('hex').substring(0, 16),
            timestamp: new Date().toISOString()
        });
    }

    _logThreats(requestId, clientIP, threats, req) {
        if (this.config.enableLogging) {
            const logEntry = {
                timestamp: new Date().toISOString(),
                requestId,
                clientIP,
                method: req.method,
                path: req.path,
                threats: threats.map(t => ({
                    type: t.type,
                    severity: t.severity,
                    vector: t.vector
                })),
                userAgent: req.headers['user-agent'],
                referer: req.headers['referer']
            };

            console.warn('[WAF THREAT]', JSON.stringify(logEntry));
            
            // Em produção, enviar para sistema de logs centralizado
            // this._sendToSIEM(logEntry);
        }
    }

    getOffenseStats() {
        return {
            totalOffenses: Array.from(this.offenseRegistry.values()).reduce((a, b) => a + b, 0),
            uniqueOffenders: this.offenseRegistry.size,
            topOffenders: Array.from(this.offenseRegistry.entries())
                .sort((a, b) => b[1] - a[1])
                .slice(0, 10)
                .map(([ip, count]) => ({ ip, count }))
        };
    }

    resetOffenses(ip = null) {
        if (ip) {
            this.offenseRegistry.delete(ip);
        } else {
            this.offenseRegistry.clear();
        }
    }
}

module.exports = WAFMiddleware;
```

```javascript
const Redis = require('ioredis');

class RateLimiter {
    constructor(config = {}) {
        this.config = {
            redisUrl: process.env.REDIS_URL || 'redis://localhost:6379',
            windowMs: 15 * 60 * 1000, // 15 minutos
            maxRequests: 100,
            prefix: 'rate_limit:',
            ...config
        };

        this.redis = new Redis(this.config.redisUrl);
        this.redis.on('error', (err) => {
            console.error('Redis connection error:', err);
        });
    }

    async check(clientIP, endpoint) {
        const key = `${this.config.prefix}${clientIP}:${endpoint}`;
        const now = Date.now();
        const windowStart = now - this.config.windowMs;

        try {
            // Usar pipeline Redis para operações atômicas
            const pipeline = this.redis.pipeline();
            
            // Remover requisições antigas
            pipeline.zremrangebyscore(key, 0, windowStart);
            
            // Contar requisições na janela
            pipeline.zcard(key);
            
            // Adicionar requisição atual
            pipeline.zadd(key, now, `${now}:${Math.random()}`);
            
            // Expirar key após janela
            pipeline.expire(key, Math.ceil(this.config.windowMs / 1000));
            
            const results = await pipeline.exec();
            const requestCount = results[1][1]; // Resultado do zcard

            return requestCount >= this.config.maxRequests;
        } catch (error) {
            console.error('Rate limiter error:', error);
            return false; // Fail open em caso de erro
        }
    }

    async getStats(clientIP) {
        const pattern = `${this.config.prefix}${clientIP}:*`;
        const keys = await this.redis.keys(pattern);
        
        const stats = {};
        for (const key of keys) {
            const endpoint = key.split(':').pop();
            const count = await this.redis.zcard(key);
            stats[endpoint] = count;
        }
        
        return stats;
    }

    async reset(clientIP, endpoint = null) {
        if (endpoint) {
            const key = `${this.config.prefix}${clientIP}:${endpoint}`;
            await this.redis.del(key);
        } else {
            const pattern = `${this.config.prefix}${clientIP}:*`;
            const keys = await this.redis.keys(pattern);
            if (keys.length > 0) {
                await this.redis.del(...keys);
            }
        }
    }

    async close() {
        await this.redis.quit();
    }
}

module.exports = RateLimiter;
```
