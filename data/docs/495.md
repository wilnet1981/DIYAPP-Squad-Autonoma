```javascript
import redisClient from '../cache/redisClient.js';

async function getItem(itemId) {
  const cacheKey = `item:${itemId}`;
  let item = await redisClient.get(cacheKey);

  if (item) {
    return JSON.parse(item);
  }

  item = await itemRepository.getItemById(itemId);

  if (item) {
    redisClient.setEx(cacheKey, 3600, JSON.stringify(item)); // Cache for 1 hour

  return item;
}
```

### Como Funciona

1. **Cache Read**: O serviГ§o tenta buscar o item do Redis primeiro. Se encontrado, retorna o item rapidamente.

3. **Cache Write**: ApГіs buscar do banco, armazena o resultado no Redis por um tempo configurГЎvel (1 hora).

```javascript
const { InfluxDB } = require('@influxdata/influxdb-client');

const url = process.env.INFLUX_URL;
const token = process.env.INFLUX_TOKEN;
const org = process.env.INFLUX_ORG;
const bucket = process.env.INFLUX_BUCKET;

const client = new InfluxDB({ url, token });
const queryApi = client.getQueryApi(org);
const writeApi = client.getWriteApi(org, bucket);

function writeMetrics(measurement, fields) {
  const point = new Point(measurement);
  Object.keys(fields).forEach(key => point.floatField(key, fields[key]));
  writeApi.writePoint(point);
}

async function queryMetrics(measurement, condition) {
  const query = `from(bucket: "${bucket}")
                 |> range(start: -10m)
                 |> filter(fn: (r) => r._measurement == "${measurement}" and ${condition})`;

  try {
    const records = [];
    const data = await queryApi.collectRows(query);
    data.forEach(record => records.push(record));
    return records;
  } catch (error) {
    console.error('Error querying metrics:', error);
    return [];
  }
}

module.exports = { writeMetrics, queryMetrics };
```
