```tsx
// src/components/FeatureComponent.tsx
import React, { Suspense, useEffect } from 'react';
import { TOKEN_EXAMPLE } from '../utils/constants';
import useFetchData from '../hooks/useFetchData';

const LazyImage = React.lazy(() => import('./LazyImage'));

type FeatureComponentProps = {
  title: string;
};

const FeatureComponent: React.FC<FeatureComponentProps> = ({ title }) => {
  const { data, error, loading } = useFetchData('/api/data-endpoint');

  useEffect(() => {
    console.log('FeatureComponent mounted.');
  }, []);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error loading data.</div>;

  return (
    <div className="feature-component" style={{ padding: TOKEN_EXAMPLE }}>
      <h1>{title}</h1>
      <Suspense fallback={<div>Loading image...</div>}>
        {/* Lazy loading images */}
        {data.map((item: any) => (
          <LazyImage key={item.id} src={item.imageUrl} alt={item.name} />
        ))}
      </Suspense>
    </div>
  );
};

export default FeatureComponent;

// src/components/LazyImage.tsx
import React from 'react';

type LazyImageProps = {
  src: string;
  alt: string;
};

const LazyImage: React.FC<LazyImageProps> = ({ src, alt }) => {
  return <img src={src} alt={alt} loading="lazy" />;
};

export default LazyImage;

// src/hooks/useFetchData.ts
import { useState, useEffect } from 'react';
import axios from 'axios';

const useFetchData = (url: string) => {
  const [data, setData] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    axios.get(url)
      .then(response => {
        setData(response.data);
        setLoading(false);
      })
      .catch(error => {
        setError(error);
        setLoading(false);
      });
  }, [url]);

  return { data, loading, error };
};

export default useFetchData;

// src/utils/constants.ts
export const TOKEN_EXAMPLE = '16px';

// src/index.tsx

import ReactDOM from 'react-dom';
import FeatureComponent from './components/FeatureComponent';

ReactDOM.render(
  <React.StrictMode>
    <FeatureComponent title="Awesome Feature" />
  </React.StrictMode>,
  document.getElementById('root')
);
```

- **Lazy Loading**: Aqui, usamos o React.lazy para carregar componentes e imagens sob demanda, o que reduz o tempo de carregamento inicial da página e melhora a performance.
- **Suspense**: Utilizado para mostrar um fallback enquanto o componente está carregando, garantindo uma boa experiência de usuário.
- **useFetchData Hook**: Custom hook para abstrair a lógica de fetching de dados. Inclui tratamento de erro e controle de loading.
- **Constantes do Design System**: Usamos tokens de um arquivo de constantes para evitar valores hardcoded.
- **Estrutura Profissional de Código**: Organização por pastas e arquivos respeitando padrões de nomeação e estrutura.

Este código aborda as otimizações necessárias para melhorar a performance e a usabilidade da aplicação, em linha com os padrões descritos na sua especificação de frontend.

```javascript
collectDefaultMetrics();

// Define custom metric
const metricLLMLatency = new Prometheus.Gauge({
    name: 'llm_response_latency_seconds',
    help: 'Latency of LLM response',
    labelNames: ['model']
});

// Simula a coleta de latência de um modelo LLM
function collectLatencyData(model, latency) {
    metricLLMLatency.set({ model: model }, latency);
}

// Express app to expose metrics
const app = express();

app.get('/metrics', async (req, res) => {
    res.set('Content-Type', Prometheus.register.contentType);
    res.end(await Prometheus.register.metrics());
});

// Run collector
const PORT = 9150;
app.listen(PORT, () => {
    console.log(`Metrics server listening on port ${PORT}`);
});

// Simulated data collection
setInterval(() => {
    const simulatedLatency = Math.random() * 2; // Simulate a latency between 0 and 2 seconds
    collectLatencyData('example-model', simulatedLatency);
```

6. **Registro e Mitigação de Vulnerabilidades**
   - Documentar todas as vulnerabilidades encontradas com CVSS (Common Vulnerability Scoring System) e priorizar a correção baseada nos SLAs definidos.

### 2. Otimização de imagens
`src/components/Image/Image.tsx` (componente otimizado)
```typescript
import React, { useState, useEffect, useRef } from 'react';
import { getOptimizedImageSrc } from '../../utils/performance';
import { tokens } from '@diyapp/design-system';

interface OptimizedImageProps {
  src: string;
  alt: string;
  width: number;
  height?: number;
  priority?: boolean;
  className?: string;
}

export const OptimizedImage: React.FC<OptimizedImageProps> = ({
  src,
  alt,
  width,
  height,
  priority = false,
  className,
}) => {
  const [isLoaded, setIsLoaded] = useState(false);
  const [hasError, setHasError] = useState(false);
  const imgRef = useRef<HTMLImageElement>(null);

  useEffect(() => {
    if (priority) return;

    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            const img = entry.target as HTMLImageElement;
            img.src = img.dataset.src || '';
            observer.unobserve(img);
          }
        });
      },
      { rootMargin: '200px' }
    );

    if (imgRef.current) {
      observer.observe(imgRef.current);
    }

    return () => {
      if (imgRef.current) {
        observer.unobserve(imgRef.current);
      }
    };
  }, [priority]);

  const handleLoad = () => {
    setIsLoaded(true);
  };

  const handleError = () => {
    setHasError(true);
  };

  const optimizedSrc = getOptimizedImageSrc(src, width);

  return (
    <div
      className={className}
      style={{
        position: 'relative',
        width: '100%',
        height: height ? `${height}px` : 'auto',
        backgroundColor: tokens.color.neutral[100],
      }}
    >
      {!isLoaded && !hasError && (
        <div
          style={{
            position: 'absolute',
            inset: 0,
            backgroundColor: tokens.color.neutral[200],
            borderRadius: tokens.borderRadius.sm,
          }}
        />
      )}

      {hasError ? (
        <div
          style={{
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
            height: '100%',
            backgroundColor: tokens.color.neutral[100],
            color: tokens.color.neutral[500],
            borderRadius: tokens.borderRadius.sm,
          }}
        >
          {alt || 'Image not available'}
        </div>
      ) : (
        <img
          ref={imgRef}
          src={priority ? optimizedSrc : undefined}
          data-src={priority ? undefined : optimizedSrc}
          alt={alt}
          width={width}
          height={height}
          loading={priority ? 'eager' : 'lazy'}
          decoding="async"
          onLoad={handleLoad}
          onError={handleError}
          style={{
            opacity: isLoaded ? 1 : 0,
            transition: 'opacity 300ms ease',
            width: '100%',
            height: 'auto',
            objectFit: 'cover',
          }}
        />
      )}
    </div>
  );
};
```

```typescript
import { useState, useEffect } from 'react';

type AsyncResourceStatus = 'idle' | 'loading' | 'success' | 'error';

export const useAsyncResource = <T,>(loader: () => Promise<T>) => {
  const [status, setStatus] = useState<AsyncResourceStatus>('idle');
  const [resource, setResource] = useState<T | null>(null);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    let isMounted = true;

    const loadResource = async () => {
      try {
        setStatus('loading');
        const result = await loader();
        if (isMounted) {
          setResource(result);
          setStatus('success');
        }
      } catch (err) {
        if (isMounted) {
          setError(err instanceof Error ? err : new Error('Failed to load resource'));
          setStatus('error');
        }
      }
    };

    // Load after initial render to avoid blocking critical resources
    const timer = setTimeout(loadResource, 1000);

    return () => {
      isMounted = false;
      clearTimeout(timer);
    };
  }, [loader]);

  return { resource, status, error };
};
```

```typescript
import { useEffect } from 'react';
import { trackPerformance } from '../utils/performance';

export const usePerformanceMonitor = (componentName: string) => {
  useEffect(() => {
    const start = performance.now();

    return () => {
      const end = performance.now();
      trackPerformance(`${componentName} mount`, end - start);
    };
  }, [componentName]);

  useEffect(() => {
    const observer = new PerformanceObserver((list) => {
      list.getEntries().forEach((entry) => {
        if (entry.name.includes(componentName)) {
          trackPerformance(entry.name, entry.duration);
        }
      });
    });

    observer.observe({ type: 'measure', buffered: true });

    return () => observer.disconnect();
  }, [componentName]);
};
```

```javascript
// retry.js
const { setTimeout } = require('timers/promises');

class Retry {
  constructor(maxRetries = 3, baseDelay = 100) {
    this.maxRetries = maxRetries;
    this.baseDelay = baseDelay;
  }

  async withRetry(fn, shouldRetry = (err) => err.isRetryable) {
    let lastError;

    for (let i = 0; i < this.maxRetries; i++) {
      try {
        return await fn();
      } catch (err) {
        lastError = err;
        if (!shouldRetry(err)) {
          throw err;
        }

        const delay = this.calculateDelay(i);
        await setTimeout(delay);
      }
    }

    throw lastError;
  }

  calculateDelay(attempt) {
    // Exponential backoff with jitter
    const exponentialDelay = this.baseDelay * Math.pow(2, attempt);
    const jitter = exponentialDelay * 0.2 * Math.random();
    return exponentialDelay + jitter;
  }
}

module.exports = Retry;
```

```javascript
      setTimeout(() => {
```

```javascript
// sanitizer.js
const { BadRequestError } = require('../../utils/errors');
const validator = require('validator');

class Sanitizer {
  sanitizeInput(input, schema) {
    if (!input) {
      throw new BadRequestError('Input is required');
    }

    const sanitized = {};

    for (const [key, rules] of Object.entries(schema)) {
      if (input[key] === undefined && rules.required) {
        throw new BadRequestError(`${key} is required`);
      }

      if (input[key] !== undefined) {
        let value = input[key];

        // Type conversion
        if (rules.type === 'number') {
          value = Number(value);
          if (isNaN(value)) {
            throw new BadRequestError(`${key} must be a number`);
          }
        } else if (rules.type === 'boolean') {
          value = Boolean(value);
        } else if (rules.type === 'array') {
          if (!Array.isArray(value)) {
            value = [value];
          }
        }

        // Sanitization
        if (rules.type === 'string') {
          value = validator.escape(value.trim());
          if (rules.maxLength && value.length > rules.maxLength) {
            throw new BadRequestError(`${key} must be at most ${rules.maxLength} characters`);
          }
          if (rules.minLength && value.length < rules.minLength) {
            throw new BadRequestError(`${key} must be at least ${rules.minLength} characters`);
          }
          if (rules.regex && !new RegExp(rules.regex).test(value)) {
            throw new BadRequestError(`${key} format is invalid`);
          }
        }

        // Validation
        if (rules.validator && !rules.validator(value)) {
          throw new BadRequestError(`${key} is invalid`);
        }

        sanitized[key] = value;
      }
    }

    return sanitized;
  }

  sanitizeQueryParams(query, schema) {
    return this.sanitizeInput(query, schema);
  }

  sanitizeBody(body, schema) {
    return this.sanitizeInput(body, schema);
  }
}

module.exports = new Sanitizer();
```

### 6. Serviço de Usuários (services/userService.js)

```javascript
const userRepository = require('../../repositories/userRepository');
const { NotFoundError, ConflictError, UnprocessableEntityError } = require('../../utils/errors');
const logger = require('../../lib/observability/logger');
const { getCorrelationId } = require('../../lib/observability/tracing');
const { hashPassword, comparePasswords } = require('../../lib/security/password');
const authService = require('../../lib/security/auth');
const ResilientCircuitBreaker = require('../../lib/resilience/circuitBreaker');
const Retry = require('../../lib/resilience/retry');
const Timeout = require('../../lib/resilience/timeout');

class UserService {
  constructor() {
    // Circuit breaker para operações de banco de dados
    this.dbCircuitBreaker = new ResilientCircuitBreaker({
      timeout: 5000,
      errorThresholdPercentage: 50,
      resetTimeout: 30000
    });

    // Retry para operações transientes
    this.retry = new Retry(3, 100);

    // Timeout para operações de banco
    this.timeout = new Timeout(5000);
  }

  async getAllUsers() {
    const correlationId = getCorrelationId();
    logger.info('Fetching all users', { correlationId });

    try {
      return await this.dbCircuitBreaker.fire(async () => {
        return await this.retry.withRetry(async () => {
          return await this.timeout.withTimeout(async () => {
            return await userRepository.findAll();
          });
        });
      });
    } catch (err) {
      logger.error('Failed to fetch all users', { correlationId, error: err.message });
      throw err;
    }
  }

  async getUserById(id) {
    const correlationId = getCorrelationId();
    logger.info('Fetching user by ID', { correlationId, userId: id });

    try {
      return await this.dbCircuitBreaker.fire(async () => {
        return await this.retry.withRetry(async () => {
          return await this.timeout.withTimeout(async () => {
            const user = await userRepository.findById(id);
            if (!user) {
              throw new NotFoundError('User not found');
            }
            return user;
          });
        });
      });
    } catch (err) {
      logger.error('Failed to fetch user', { correlationId, userId: id, error: err.message });
      throw err;
    }
  }

  async createUser(userData) {
    const correlationId = getCorrelationId();
    logger.info('Creating new user', { correlationId, email: userData.email });

    try {
      return await this.dbCircuitBreaker.fire(async () => {
        return await this.retry.withRetry(async () => {
          return await this.timeout.withTimeout(async () => {
            // Check if user already exists
            const existingUser = await userRepository.findByEmail(userData.email);
            if (existingUser) {
              throw new ConflictError('Email already in use');
            }

            // Hash password
            const hashedPassword = await hashPassword(userData.password);
            const user = await userRepository.create({
              ...userData,
              password: hashedPassword
            });

            // Generate tokens
            const tokens = await authService.generateTokens(user);

            return {
              ...user,
              accessToken: tokens.accessToken,
              refreshToken: tokens.refreshToken
            };
          });
        });
      });
    } catch (err) {
      logger.error('Failed to create user', { correlationId, email: userData.email, error: err.message });
      throw err;
    }
  }

  async updateUser(id, userData) {
    const correlationId = getCorrelationId();
    logger.info('Updating user', { correlationId, userId: id });

    try {
      return await this.dbCircuitBreaker.fire(async () => {
        return await this.retry.withRetry(async () => {
          return await this.timeout.withTimeout(async () => {
            const existingUser = await userRepository.findById(id);
            if (!existingUser) {
              throw new NotFoundError('User not found');
            }

            // Check if email is being changed to an existing one
            if (userData.email && userData.email !== existingUser.email) {
              const userWithEmail = await userRepository.findByEmail(userData.email);
              if (userWithEmail) {
                throw new ConflictError('Email already in use');
              }
            }

            // If password is being updated, hash it
            if (userData.password) {
              userData.password = await hashPassword(userData.password);
            }

            const updatedUser = await userRepository.update(id, userData);
            return updatedUser;
          });
        });
      });
    } catch (err) {
      logger.error('Failed to update user', { correlationId, userId: id, error: err.message });
      throw err;
    }
  }

  async deleteUser(id) {
    const correlationId = getCorrelationId();
    logger.info('Deleting user', { correlationId, userId: id });

    try {
      return await this.dbCircuitBreaker.fire(async () => {
        return await this.retry.withRetry(async () => {
          return await this.timeout.withTimeout(async () => {
            const existingUser = await userRepository.findById(id);
            if (!existingUser) {
              throw new NotFoundError('User not found');
            }

            await userRepository.delete(id);
          });
        });
      });
    } catch (err) {
      logger.error('Failed to delete user', { correlationId, userId: id, error: err.message });
      throw err;
    }
  }

  async authenticate(email, password) {
    const correlationId = getCorrelationId();
    logger.info('Authenticating user', { correlationId, email });

    try {
      return await this.dbCircuitBreaker.fire(async () => {
        return await this.retry.withRetry(async () => {
          return await this.timeout.withTimeout(async () => {
            const user = await userRepository.findByEmail(email);
            if (!user) {
              throw new UnprocessableEntityError('Invalid email or password');
            }

            const isPasswordValid = await comparePasswords(password, user.password);
            if (!isPasswordValid) {
              throw new UnprocessableEntityError('Invalid email or password');
            }

            const tokens = await authService.generateTokens(user);
            return {
              ...user,
              accessToken: tokens.accessToken,
              refreshToken: tokens.refreshToken
            };
          });
        });
      });
    } catch (err) {
      logger.error('Authentication failed', { correlationId, email, error: err.message });
      throw err;
    }
  }
}

module.exports = new UserService();
```

```javascript
    });

    // Test the connection
    this.pool.query('SELECT 1')
      .then(() => logger.info('Database connection established'))
      .catch(err => logger.error('Database connection failed', { error: err.message }));
  }

    const correlationId = getCorrelationId();
    logger.debug('Executing findAll query', { correlationId });

    const query = 'SELECT id, email, name, roles, created_at, updated_at FROM users';
    const { rows } = await this.pool.query(query);
    return rows;
  }

  async findById(id) {
    const correlationId = getCorrelationId();
    logger.debug('Executing findById query', { correlationId, userId: id });

    const query = 'SELECT id, email, name, roles, created_at, updated_at FROM users WHERE id = $1';
    const { rows } = await this.pool.query(query, [id]);

    return rows[0] || null;
  }

  async findByEmail(email) {
    const correlationId = getCorrelationId();
    logger.debug('Executing findByEmail query', { correlationId, email });

    const query = 'SELECT id, email, name, password, roles, created_at, updated_at FROM users WHERE email = $1';
    const { rows } = await this.pool.query(query, [email]);

    return rows[0] || null;
  }

  async create(userData) {
    const correlationId = getCorrelationId();
    logger.debug('Executing create query', { correlationId, email: userData.email });

    const query = `
      INSERT INTO users (email, name, password, roles)
      VALUES ($1, $2, $3, $4)
      RETURNING id, email, name, roles, created_at, updated_at
    `;
    const values = [
      userData.email,
      userData.name,
      userData.password,
      userData.roles || ['user']
    ];

    const { rows } = await this.pool.query(query, values);
    return rows[0];
  }

  async update(id, userData) {
    const correlationId = getCorrelationId();
    logger.debug('Executing update query', { correlationId, userId: id });

    const fields = [];
    const values = [];
    let paramIndex = 1;

    for (const [key, value] of Object.entries(userData)) {
      if (key !== 'id' && value !== undefined) {
        fields.push(`${key} = $${paramIndex}`);
        values.push(value);
        paramIndex++;
      }
    }

    if (fields.length === 0) {
      const user = await this.findById(id);
      if (!user) {
        throw new NotFoundError('User not found');
      }
      return user;
    }

    values.push(id);
    const query = `
      UPDATE users
      SET ${fields.join(', ')}, updated_at = NOW()
      WHERE id = $${paramIndex}
      RETURNING id, email, name, roles, created_at, updated_at
    `;

    const { rows } = await this.pool.query(query, values);
    return rows[0];
  }

  async delete(id) {
    const correlationId = getCorrelationId();
    logger.debug('Executing delete query', { correlationId, userId: id });

    const query = 'DELETE FROM users WHERE id = $1';
    await this.pool.query(query, [id]);
  }
}
```

```javascript
// auth.js
const authService = require('../../lib/security/auth');
const { UnauthorizedError, ForbiddenError } = require('../../utils/errors');
const logger = require('../../lib/observability/logger');
const { getCorrelationId } = require('../../lib/observability/tracing');
const authorizationService = require('../../lib/security/authorization');

async function authMiddleware(req, res, next) {
  const correlationId = getCorrelationId();
  const authHeader = req.headers['authorization'];

  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    logger.warn('Missing or invalid authorization header', { correlationId });
    return next(new UnauthorizedError('Authentication required'));
  }

  const token = authHeader.split(' ')[1];

  try {
    const payload = await authService.verifyToken(token);
    req.user = payload;
    next();
  } catch (err) {
    logger.error('Authentication failed', { correlationId, error: err.message });
    next(new UnauthorizedError('Invalid or expired token'));
  }
}

function checkPermission(resource, action) {
  return (req, res, next) => {
    const correlationId = getCorrelationId();
    try {
      authorizationService.checkPermission(req.user, resource, action);
      next();
    } catch (err) {
      logger.warn('Authorization failed', {
        correlationId,
        userId: req.user?.id,
        resource,
        action,
        error: err.message
      });
      next(err);
    }
  };
}
```

### 6. Conector Customizado para LLM Gateway

**ingestion/custom_connectors/llm_gateway_connector.py**
```python
import requests
import json
from datetime import datetime, timedelta
from google.cloud import bigquery
from google.oauth2 import service_account
import os
from monitoring.alerts.slack_alerts import send_slack_alert

class LLMGatewayConnector:
    def __init__(self):
        self.gateway_url = os.getenv('LLM_GATEWAY_URL', 'https://api.diyapp.com/llm-gateway')
        self.api_key = os.getenv('LLM_GATEWAY_API_KEY')
        self.bq_project = os.getenv('BQ_PROJECT', 'diyapp-v9')
        self.bq_dataset = os.getenv('BQ_DATASET_RAW', 'raw')
        self.bq_table = 'llm_gateway_calls'
        self.credentials = service_account.Credentials.from_service_account_file(
            os.getenv('GOOGLE_APPLICATION_CREDENTIALS')
        )
        self.client = bigquery.Client(project=self.bq_project, credentials=self.credentials)

    def fetch_llm_calls(self, start_time, end_time):
        """Fetch LLM calls from gateway API"""
        headers = {
            'Authorization': f'Bearer {self.api_key}',
            'Content-Type': 'application/json'
        }

        params = {
            'start_time': start_time.isoformat(),
            'end_time': end_time.isoformat(),
            'limit': 10000
        }

        try:
            response = requests.get(
                f"{self.gateway_url}/calls",
                headers=headers,
                params=params
            )
            response.raise_for_status()
            return response.json()['data']
        except requests.exceptions.RequestException as e:
            send_slack_alert(f"Failed to fetch LLM calls: {str(e)}")
            raise

    def transform_data(self, calls):
        """Transform raw API data to match our schema"""
        transformed = []
        for call in calls:
            transformed.append({
                'call_id': call['id'],
                'user_id': call['user_id'],
                'feature_id': call['feature_id'],
                'model_id': call['model_id'],
                'provider': call['provider'],
                'input_tokens': call['usage']['input_tokens'],
                'output_tokens': call['usage']['output_tokens'],
                'start_time': call['timestamps']['start'],
                'end_time': call['timestamps']['end'],
                'status': call['status'],
                'metadata': json.dumps(call.get('metadata', {})),
                'received_at': datetime.utcnow().isoformat()
            })
        return transformed

    def load_to_bigquery(self, data):
        """Load data to BigQuery"""
        table_ref = f"{self.bq_project}.{self.bq_dataset}.{self.bq_table}"

        errors = self.client.insert_rows_json(
            table_ref,
            data
        )

        if errors:
            send_slack_alert(f"BigQuery insert errors: {errors}")
            raise Exception(f"BigQuery insert errors: {errors}")

    def run(self):
        """Run the ingestion pipeline"""
        end_time = datetime.utcnow()
        start_time = end_time - timedelta(hours=1)  # Last hour of data

        print(f"Fetching LLM calls from {start_time} to {end_time}")
        calls = self.fetch_llm_calls(start_time, end_time)

        if not calls:
            print("No new LLM calls found")
            return

        print(f"Transforming {len(calls)} LLM calls")
        transformed_data = self.transform_data(calls)

        print(f"Loading {len(transformed_data)} records to BigQuery")
        self.load_to_bigquery(transformed_data)

        print("LLM Gateway ingestion completed successfully")

if __name__ == "__main__":
    connector = LLMGatewayConnector()
    connector.run()
```

```html
        async function fetchDashboardData() {
            try {
                const response = await axios.get('/api/dashboard');
                const data = response.data;

                // Update last updated time
                document.getElementById('last-updated').textContent = new Date(data.lastUpdated).toLocaleString();

                // Update pipeline status
                updatePipelineStatus(data.pipelineStatus);

                // Update key metrics
                updateKeyMetrics(data.metrics);

                // Update alerts
                updateAlerts(data.alerts);

                // Update charts
                updateQualityTrendsChart(data.qualityTrends);
                updatePipelinePerformanceChart(data.pipelinePerformance);

                console.error('Error fetching dashboard data:', error);
                showErrorAlert();
            }
        }

        function updateKeyMetrics(metrics) {
            // DAU

            // LLM Cost
            document.getElementById('llm-cost-metric').textContent = '$' + metrics.llmCost.value.toFixed(2);
            const llmCostTrend = document.getElementById('llm-cost-trend');
            llmCostTrend.innerHTML = getTrendIcon(metrics.llmCost.trend) + ' ' + Math.abs(metrics.llmCost.trend) + '%';
            llmCostTrend.className = `badge bg-${getTrendColor(metrics.llmCost.trend)} text-dark`;

            // Pipeline Runs
            document.getElementById('pipeline-runs-metric').textContent = metrics.pipelineRuns.value;
            document.getElementById('success-rate').textContent = metrics.successRate.value + '% success';
            document.getElementById('success-rate').className = `badge bg-${metrics.successRate.value >= 95 ? 'success' : 'warning'}`;

        function updateAlerts(alerts) {
            const alertsContainer = document.getElementById('alerts-container');

                return;

            alerts.forEach(alert => {
                const alertEl = document.createElement('div');
                alertEl.className = `alert alert-${getAlertLevel(alert.level)} alert-dismissible fade show`;
                alertEl.role = 'alert';

                alertEl.innerHTML = `
```

### 9. Script de Retenção de Dados

**scripts/retention_policy.py**
```python
import argparse
from datetime import datetime, timedelta
from google.cloud import bigquery
from google.oauth2 import service_account
import logging
from monitoring.alerts.slack_alerts import send_slack_alert

# Configure logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

class DataRetentionManager:
    def __init__(self):
        self.project_id = 'diyapp-v9'
        self.credentials = service_account.Credentials.from_service_account_file(
            'path/to/service-account.json'
        )
        self.client = bigquery.Client(project=self.project_id, credentials=self.credentials)

        # Retention policies by dataset
        self.retention_policies = {
            'raw': {
                'retention_days': 90,
                'tables': [
                    'diyapp_events',
                    'llm_gateway_calls',
                    'infra_metrics',
                    'external_systems'
                ]
            },
            'staging': {
                'retention_days': 180,
                'tables': [
                    'diyapp_events',
                    'llm_gateway_calls',
                    'infra_metrics',
                    'external_systems'
                ]
            },
            'marts_product': {
                'retention_days': 365,
                'tables': [
                    'daily_active_users',
                    'feature_adoption',
                    'funnel_conversion'
                ]
            },
            'marts_ai': {
                'retention_days': 365,
                'tables': [
                    'llm_cost_by_feature',
                    'model_latency_daily',
                    'token_consumption_trend'
                ]
            },
            'marts_ops': {
                'retention_days': 365,
                'tables': [
                    'deploy_frequency',
                    'incident_metrics',
                    'sprint_velocity'
                ]
            }
        }

    def get_retention_cutoff(self, retention_days):
        """Calculate the cutoff date for retention"""
        return datetime.utcnow() - timedelta(days=retention_days)

    def delete_old_data(self, dataset_id, table_id, cutoff_date):
        """Delete data older than cutoff_date from the specified table"""
        table_ref = f"{self.project_id}.{dataset_id}.{table_id}"

        # For partitioned tables, we can delete partitions directly
        if self._is_partitioned_table(dataset_id, table_id):
            return self._delete_partitions(dataset_id, table_id, cutoff_date)

        # For non-partitioned tables, use a DELETE statement
        query = f"""
            DELETE FROM `{table_ref}`
            WHERE DATE(_PARTITIONTIME) < DATE('{cutoff_date.date()}')
            OR (DATE(_PARTITIONTIME) IS NULL AND DATE(created_at) < DATE('{cutoff_date.date()}'))
        """

        try:
            logger.info(f"Deleting data from {table_ref} older than {cutoff_date}")
            query_job = self.client.query(query)
            query_job.result()  # Wait for job to complete

            deleted_rows = query_job.num_dml_affected_rows
            logger.info(f"Deleted {deleted_rows} rows from {table_ref}")
            return deleted_rows
        except Exception as e:
            logger.error(f"Error deleting data from {table_ref}: {str(e)}")
            send_slack_alert(f"Data retention error: Failed to delete data from {table_ref}. Error: {str(e)}")
            raise

    def _is_partitioned_table(self, dataset_id, table_id):
        """Check if a table is partitioned"""
        table_ref = self.client.dataset(dataset_id).table(table_id)
        table = self.client.get_table(table_ref)
        return table.time_partitioning is not None

    def _delete_partitions(self, dataset_id, table_id, cutoff_date):
        """Delete entire partitions from a partitioned table"""
        table_ref = f"{self.project_id}.{dataset_id}.{table_id}"
        cutoff_str = cutoff_date.strftime('%Y%m%d')

        try:
            # Get the table to check partitioning
            table = self.client.get_table(table_ref)

            if table.time_partitioning.type_ == 'DAY':
                # For daily partitioned tables
                query = f"""
                    ALTER TABLE `{table_ref}`
                    DROP PARTITION IF EXISTS
                    DATE(_PARTITIONTIME) < DATE('{cutoff_date.date()}')
                """
            else:
                # For other partitioning types
                query = f"""
                    DELETE FROM `{table_ref}`
                    WHERE DATE(_PARTITIONTIME) < DATE('{cutoff_date.date()}')
                """

            logger.info(f"Deleting partitions from {table_ref} older than {cutoff_date}")
            query_job = self.client.query(query)
            query_job.result()

            logger.info(f"Deleted partitions from {table_ref} older than {cutoff_date}")
            return query_job.num_dml_affected_rows
        except Exception as e:
            logger.error(f"Error deleting partitions from {table_ref}: {str(e)}")
            send_slack_alert(f"Data retention error: Failed to delete partitions from {table_ref}. Error: {str(e)}")
            raise

    def apply_retention_policies(self):
        """Apply retention policies to all datasets"""
        logger.info("Starting data retention process")

        total_deleted = 0

        for dataset_id, policy in self.retention_policies.items():
            retention_days = policy['retention_days']
            cutoff_date = self.get_retention_cutoff(retention_days)

            logger.info(f"Processing dataset {dataset_id} with {retention_days} days retention")

            for table_id in policy['tables']:
                try:
                    deleted = self.delete_old_data(dataset_id, table_id, cutoff_date)
                    total_deleted += deleted
                except Exception as e:
                    logger.error(f"Failed to process {dataset_id}.{table_id}: {str(e)}")
                    continue

        logger.info(f"Data retention process completed. Total rows deleted: {total_deleted}")
        send_slack_alert(f"Data retention completed. Total rows deleted: {total_deleted}")
        return total_deleted

    def dry_run(self):
        """Perform a dry run to show what would be deleted"""
        logger.info("Starting data retention dry run")

        for dataset_id, policy in self.retention_policies.items():
            retention_days = policy['retention_days']
            cutoff_date = self.get_retention_cutoff(retention_days)

            logger.info(f"\nDataset: {dataset_id}")
            logger.info(f"Retention policy: {retention_days} days")
            logger.info(f"Cutoff date: {cutoff_date.date()}")

            for table_id in policy['tables']:
                table_ref = f"{self.project_id}.{dataset_id}.{table_id}"

                if self._is_partitioned_table(dataset_id, table_id):
                    query = f"""
                        SELECT COUNT(*) as count
                        FROM `{table_ref}`
                        WHERE DATE(_PARTITIONTIME) < DATE('{cutoff_date.date()}')
                    """
                else:
                    query = f"""
                        SELECT COUNT(*) as count
                        FROM `{table_ref}`
                        WHERE DATE(created_at) < DATE('{cutoff_date.date()}')
                    """

                try:
                    query_job = self.client.query(query)
                    results = query_job.result()
                    count = next(results).count

                    logger.info(f"  {table_id}: {count} rows would be deleted")
                except Exception as e:
                    logger.error(f"  {table_id}: Error counting rows - {str(e)}")

        logger.info("Data retention dry run completed")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description='DIYAPP Data Retention Manager')
    parser.add_argument('--dry-run', action='store_true', help='Perform a dry run without deleting data')
    args = parser.parse_args()

    retention_manager = DataRetentionManager()

    if args.dry_run:
        retention_manager.dry_run()
    else:
        retention_manager.apply_retention_policies()
```

```python
# auditor.py
import subprocess
import json

def run_bandit():
    try:
        # Executa a análise com o Bandit em todos os arquivos .py
        result = subprocess.run(
            ["bandit", "-r", "src/"],
            text=True, capture_output=True, check=True
        )
        print("Análise de Segurança (Bandit):")
        print(result.stdout)
    except subprocess.CalledProcessError as e:
        print("Erro ao executar o Bandit:")
        print(e.stderr)

def parse_bandit_output(output):
    # Código para processar o resultado em JSON (se Bandit oferecer tal opção) e
    # avaliar riscos detectados, sugerir correções e priorizar ações
    try:
        data = json.loads(output)
        for issue in data.get('results', []):
            print(f"Arquivo: {issue['filename']}")
            print(f"Problema: {issue['issue_text']}")
            print(f"Nível: {issue['issue_severity']}")
            print(f"Confiança: {issue['issue_confidence']}")
            print("---")
    except json.JSONDecodeError:
        print("Erro ao decodificar a saída do Bandit")

# Main
if __name__ == '__main__':
```
