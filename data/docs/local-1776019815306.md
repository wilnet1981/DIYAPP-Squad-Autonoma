# DIYAPP Evolution - V12 Core - Arquitetura Técnica

## 1. Documento de Arquitetura (ADR-001)

```
# ADR-001: Arquitetura de Microsserviços V12 Core

Data: 2024-01-15
Status: Aceita
Autores: Tech Lead, SRE Lead, Security Lead

CONTEXTO:
O DIYAPP precisa evoluir para uma arquitetura de microsserviços que suporte:
- Escalabilidade horizontal automática
- Deploy independente por serviço
- Isolamento de falhas
- Performance P95 < 200ms
- 100% de autonomia operacional
- Compliance com padrões de segurança

DECISÃO:
Adotar arquitetura baseada em microsserviços com os seguintes componentes:

1. **API Gateway**: Nginx + Lua (Kong) para roteamento e rate limiting
2. **Serviços Core** (stateless):
   - auth-service: Autenticação JWT + OAuth2
   - user-service: Gerenciamento de usuários
   - content-service: Conteúdo principal
   - notification-service: Notificações em tempo real
   - search-service: Busca indexada
3. **Serviços Stateful**:
   - session-store: Redis Cluster (sessões)
   - message-queue: RabbitMQ com HA
   - database: PostgreSQL com replicação
4. **Observabilidade**:
   - Prometheus + Grafana (métricas)
   - Jaeger (tracing distribuído)
   - ELK Stack (logs centralizados)

OPÇÕES CONSIDERADAS:
- Opção A: Monólito modularizado
  Prós: Simplicidade inicial, debugging fácil
  Contras: Escalabilidade limitada, deploy único, acoplamento alto
  
- Opção B: Microsserviços completos
  Prós: Escalabilidade independente, isolamento de falhas, deploy contínuo
  Contras: Complexidade operacional, latência de rede, consistência eventual
  
- Opção escolhida: B com mitigações
  Justificativa: Os benefícios de escalabilidade e resiliência superam a complexidade, que será gerenciada com automação.

CONSEQUÊNCIAS:
Positivas:
- Escalabilidade granular por serviço
- Deploy independente sem downtime
- Isolamento de falhas (circuit breaker)
- Tecnologia heterogênea por serviço

Negativas:
- Complexidade de debugging distribuído
- Latência adicional de rede
- Gerenciamento de transações distribuídas

Riscos:
1. Latência excessiva: Mitigado com caching agressivo e connection pooling
2. Consistência de dados: Mitigado com padrões SAGA e compensação
3. Observabilidade: Mitigado com tracing distribuído e métricas em tempo real

REVISÃO: 2024-04-15 (3 meses)
```

// Middleware de métricas
export const metricsMiddleware = (req: Request, res: Response, next: NextFunction) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    const route = req.route?.path || req.path;
    
    httpRequestDuration
      .labels(req.method, route, res.statusCode.toString())
      .observe(duration);
      
    httpRequestsTotal
      .labels(req.method, route, res.statusCode.toString())
      .inc();
    
    // Alerta se P95 > 200ms
    if (duration > 0.2) {
      console.warn(`Slow request detected: ${req.method} ${route} took ${duration}s`);
    }
  });
  
  next();
};

  validateContrast(foreground, background) {
    // Implementação simplificada - em produção usar biblioteca como chroma.js
    const luminance = (hex) => {
      const rgb = parseInt(hex.slice(1), 16);
      const r = (rgb >> 16) & 0xff;
      const g = (rgb >> 8) & 0xff;
      const b = (rgb >> 0) & 0xff;
      const [rs, gs, bs] = [r, g, b].map(c => {
        c = c / 255;
        return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
      });
      return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
    };
    
    const l1 = luminance(foreground) + 0.05;
    const l2 = luminance(background) + 0.05;
    const ratio = l1 > l2 ? l1 / l2 : l2 / l1;

        record.correlation_id = correlation_id_ctx.get()
        return True


class StructuredJsonFormatter(jsonlogger.JsonFormatter):
    def add_fields(self, log_record, record, message_dict):
        super().add_fields(log_record, record, message_dict)
        if not log_record.get('timestamp'):
            log_record['timestamp'] = datetime.utcnow().isoformat()
        if not log_record.get('level'):
            log_record['level'] = record.levelname
        if not log_record.get('correlation_id'):
            log_record['correlation_id'] = correlation_id_ctx.get()
        if hasattr(record, 'user_id'):
            log_record['user_id'] = record.user_id
        if hasattr(record, 'endpoint'):
            log_record['endpoint'] = record.endpoint
        if hasattr(record, 'duration_ms'):
            log_record['duration_ms'] = record.duration_ms
        if hasattr(record, 'status'):
            log_record['status'] = record.status

# src/core/security.py
from datetime import datetime, timedelta
from typing import Any, Dict, Optional, Union
from jose import JWTError, jwt
from passlib.context import CryptContext
from fastapi import HTTPException, status
import secrets

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


def create_access_token(
    subject: Union[str, Any], 
    expires_delta: Optional[timedelta] = None,
    additional_claims: Optional[Dict] = None
) -> str:
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES)
    
    to_encode = {"exp": expire, "sub": str(subject)}
    if additional_claims:
        to_encode.update(additional_claims)
    
    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    return encoded_jwt


def create_refresh_token(subject: Union[str, Any]) -> str:
    expire = datetime.utcnow() + timedelta(days=settings.REFRESH_TOKEN_EXPIRE_DAYS)
    to_encode = {"exp": expire, "sub": str(subject), "type": "refresh"}
    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    return encoded_jwt


def verify_password(plain_password: str, hashed_password: str) -> bool:
    return pwd_context.verify(plain_password, hashed_password)


def get_password_hash(password: str) -> str:
    return pwd_context.hash(password)


def decode_token(token: str) -> Dict[str, Any]:
    try:
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        return payload
    except JWTError as e:
        logger.error(f"JWT decode error: {str(e)}", extra={"error_type": "jwt_decode"})
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Could not validate credentials",
            headers={"WWW-Authenticate": "Bearer"},
        )


def generate_session_token() -> str:
    """Generate a secure random token for session IDs"""
    return secrets.token_urlsafe(32)

# src/core/cache.py
import redis
from redis.exceptions import RedisError, ConnectionError, TimeoutError
from circuitbreaker import circuit
from functools import wraps
import pickle
from typing import Any, Optional, Callable, TypeVar
import time

T = TypeVar('T')


class RedisCache:
    def __init__(self):
        self._client: Optional[redis.Redis] = None
        self._circuit_open = False
        self._circuit_last_failure = 0
        self._circuit_reset_timeout = 60  # seconds
    
    @property
    def client(self) -> redis.Redis:
        if self._client is None:
            self._connect()
        return self._client
    
    def _connect(self):
        try:
            self._client = redis.Redis.from_url(
                str(settings.REDIS_URL),
                password=settings.REDIS_PASSWORD,
                socket_timeout=5,  # Timeout explícito
                socket_connect_timeout=5,
                retry_on_timeout=True,
                max_connections=10
            )
            # Test connection
            self._client.ping()
            self._circuit_open = False
            logger.info("Redis connection established")
        except (ConnectionError, TimeoutError) as e:
            logger.error(f"Redis connection failed: {str(e)}", extra={"error_type": "redis_connection"})
            self._circuit_open = True
            self._circuit_last_failure = time.time()
            raise
    
    def _check_circuit(self):
        if self._circuit_open:
            if time.time() - self._circuit_last_failure > self._circuit_reset_timeout:
                # Try to reset circuit
                try:
                    self._connect()
                except RedisError:
                    self._circuit_last_failure = time.time()
                    raise
            else:
                raise RedisError("Circuit breaker is open")
    
    @circuit(failure_threshold=5, expected_exception=RedisError, recovery_timeout=60)
    def get(self, key: str) -> Optional[Any]:
        """Get value from cache with circuit breaker"""
        self._check_circuit()
        try:
            value = self.client.get(key)
            if value is None:
                return None
            return pickle.loads(value)
        except (RedisError, pickle.PickleError) as e:
            logger.error(f"Redis get error: {str(e)}", extra={"key": key, "error_type": "redis_get"})
            raise
    
    @circuit(failure_threshold=5, expected_exception=RedisError, recovery_timeout=60)
    def set(self, key: str, value: Any, expire: int = 3600) -> bool:
        """Set value in cache with circuit breaker"""
        self._check_circuit()
        try:
            serialized = pickle.dumps(value)
            return self.client.setex(key, expire, serialized)
        except (RedisError, pickle.PickleError) as e:
            logger.error(f"Redis set error: {str(e)}", extra={"key": key, "error_type": "redis_set"})
            raise
    
    @circuit(failure_threshold=5, expected_exception=RedisError, recovery_timeout=60)
    def delete(self, key: str) -> bool:
        """Delete key from cache with circuit breaker"""
        self._check_circuit()
        try:
            return bool(self.client.delete(key))
        except RedisError as e:
            logger.error(f"Redis delete error: {str(e)}", extra={"key": key, "error_type": "redis_delete"})
            raise
    
    @circuit(failure_threshold=5, expected_exception=RedisError, recovery_timeout=60)
    def exists(self, key: str) -> bool:
        """Check if key exists with circuit breaker"""
        self._check_circuit()
        try:
            return bool(self.client.exists(key))
        except RedisError as e:
            logger.error(f"Redis exists error: {str(e)}", extra={"key": key, "error_type": "redis_exists"})
            raise
    
    def get_with_fallback(self, key: str, fallback_func: Callable[[], T], expire: int = 3600) -> T:
        """Get from cache, fallback to function if cache fails"""
        try:
            cached = self.get(key)
            if cached is not None:
                logger.debug(f"Cache hit for key: {key}")
                return cached
        except RedisError:
            logger.warning(f"Cache failure, using fallback for key: {key}")
        
        # Execute fallback function
        result = fallback_func()
        
        # Try to cache result (non-blocking)
        try:
            self.set(key, result, expire)
        except RedisError:
            logger.warning(f"Failed to cache result for key: {key}")
        
        return result

// src/components/error-boundary.tsx
'use client';

import { Component, ErrorInfo, ReactNode } from 'react';

interface Props {
  children: ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('ErrorBoundary caught an error:', error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="min-h-screen flex items-center justify-center bg-red-50">
          <div className="text-center p-8 max-w-md">
            <h1 className="text-2xl font-bold text-red-600 mb-4">
              Something went wrong
            </h1>
            <p className="text-gray-600 mb-6">
              {this.state.error?.message || 'An unexpected error occurred'}
            </p>
            <button
              onClick={() => window.location.reload()}
              className="px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700 transition-colors"
            >
              Reload Page
            </button>
          </div>
        </div>
      );
    }

    return this.props.children;
  }
}

        const lastEntry = entries[entries.length - 1];

      });

      lcpObserver.observe({ type: 'largest-contentful-paint', buffered: true });

      // Monitor CLS
      const clsObserver = new PerformanceObserver((entryList) => {
        for (const entry of entryList.getEntries()) {
          console.log('CLS:', entry.value);
        }
      });

      clsObserver.observe({ type: 'layout-shift', buffered: true });

      // Monitor INP
      const inpObserver = new PerformanceObserver((entryList) => {
        for (const entry of entryList.getEntries()) {
          console.log('INP:', entry.duration);
        }
      });

      inpObserver.observe({ type: 'first-input', buffered: true });

      return () => {
        lcpObserver.disconnect();
        clsObserver.disconnect();
        inpObserver.disconnect();
      };
    }
  }, []);

  return null;
}
