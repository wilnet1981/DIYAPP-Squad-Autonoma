### 4. Ponto de Entrada (`src/index.tsx`)

```tsx
import React from 'react';
import ReactDOM from 'react-dom/client';
import { ErrorBoundary } from './components/ui/ErrorBoundary';
import App from './App';
import './styles/globals.css';

// Performance monitoring
import { initPerformanceMonitoring } from './utils/performance';

// Initialize performance monitoring
initPerformanceMonitoring();

// Preload critical resources
const preloadCriticalResources = () => {
  // Preload critical fonts
  const link = document.createElement('link');
  link.rel = 'preload';
  link.href = 'https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700&display=swap';
  link.as = 'style';
  link.crossOrigin = 'anonymous';
  document.head.appendChild(link);
  
  // Preload critical images
  const criticalImages = [
    '/logo.svg',
    '/hero-image.webp'
  ];
  
  criticalImages.forEach(src => {
    const img = new Image();
    img.src = src;
  });
};

// Execute after DOM is ready
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', preloadCriticalResources);
} else {
  preloadCriticalResources();
}

const root = ReactDOM.createRoot(
  document.getElementById('root') as HTMLElement
);

root.render(
  <React.StrictMode>
    <ErrorBoundary>
      <App />
    </ErrorBoundary>
  </React.StrictMode>
);
```

const { Pool } = require('pg');
const CircuitBreaker = require('opossum');
const logger = require('../logging/logger');
const metrics = require('../monitoring/metrics');

class ResilientPostgresPool {
  constructor(config) {
    this.config = {
      host: config.host || 'localhost',
      port: config.port || 5432,
      user: config.user,
      password: config.password,
      database: config.database,
      max: config.max || 20,
      min: config.min || 5,
      idleTimeoutMillis: config.idleTimeoutMillis || 30000,
      connectionTimeoutMillis: config.connectionTimeoutMillis || 2000,
      ssl: config.ssl ? { rejectUnauthorized: false } : false
    };

    this.pool = new Pool(this.config);
    this.setupEventListeners();
    this.setupCircuitBreaker();
    this.healthCheckInterval = null;
  }

  setupEventListeners() {
    this.pool.on('connect', (client) => {
      logger.debug('New PostgreSQL client connected');
      metrics.increment('postgres.connections.active');
    });

    this.pool.on('remove', (client) => {
      logger.debug('PostgreSQL client removed from pool');
      metrics.decrement('postgres.connections.active');
    });

    this.pool.on('error', (err, client) => {
      logger.error('Unexpected PostgreSQL pool error', {
        error: err.message,
        stack: err.stack
      });
      metrics.increment('postgres.errors.pool');
    });
  }

  setupCircuitBreaker() {
    const breakerOptions = {
      timeout: 10000, // 10 seconds timeout for queries
      errorThresholdPercentage: 50,
      resetTimeout: 30000,
      rollingCountTimeout: 60000,
      rollingCountBuckets: 10,
      name: 'postgres-query'
    };

    this.circuitBreaker = new CircuitBreaker(
      async (query, params) => {
        const startTime = Date.now();
        try {
          const result = await this.pool.query(query, params);
          const duration = Date.now() - startTime;
          
          metrics.histogram('postgres.query.duration', duration);
          metrics.increment('postgres.query.success');
          
          // Log slow queries
          if (duration > 1000) {
            logger.warn('Slow PostgreSQL query detected', {
              query: query.substring(0, 200),
              duration,
              params: this.maskSensitiveData(params)
            });
          }
          
          return result;
        } catch (error) {
          const duration = Date.now() - startTime;
          metrics.histogram('postgres.query.duration', duration);
          metrics.increment('postgres.query.error');
          throw error;
        }
      },
      breakerOptions
    );

    this.circuitBreaker.on('open', () => {
      logger.error('PostgreSQL circuit breaker opened - database unavailable');
      metrics.increment('postgres.circuit_breaker.open');
    });

    this.circuitBreaker.on('halfOpen', () => {
      logger.info('PostgreSQL circuit breaker half-open - testing connection');
      metrics.increment('postgres.circuit_breaker.half_open');
    });

    this.circuitBreaker.on('close', () => {
      logger.info('PostgreSQL circuit breaker closed - database available');
      metrics.increment('postgres.circuit_breaker.close');
    });

    this.circuitBreaker.fallback(async (query, params, err) => {
      logger.error('PostgreSQL query failed, using fallback', {
        error: err.message,
        query: query.substring(0, 200)
      });
      
      // Return empty result set as fallback
      return { rows: [], rowCount: 0 };
    });
  }

  async query(text, params) {
    return this.circuitBreaker.fire(text, params);
  }

  async connect() {
    const client = await this.pool.connect();
    
    // Wrap client query with metrics
    const originalQuery = client.query;
    client.query = async (text, params) => {
      const startTime = Date.now();
      try {
        const result = await originalQuery.call(client, text, params);
        const duration = Date.now() - startTime;
        metrics.histogram('postgres.client.query.duration', duration);
        return result;
      } catch (error) {
        const duration = Date.now() - startTime;
        metrics.histogram('postgres.client.query.duration', duration);
        metrics.increment('postgres.client.query.error');
        throw error;
      }
    };
    
    return client;
  }

  async healthCheck() {
    try {
      const startTime = Date.now();
      await this.pool.query('SELECT 1');
      const duration = Date.now() - startTime;
      
      metrics.gauge('postgres.health.check.duration', duration);
      metrics.increment('postgres.health.check.success');
      
      return {
        healthy: true,
        latency: duration,
        poolSize: this.pool.totalCount,
        idleCount: this.pool.idleCount,
        waitingCount: this.pool.waitingCount
      };
    } catch (error) {
      metrics.increment('postgres.health.check.failure');
      return {
        healthy: false,
        error: error.message
      };
    }
  }

  maskSensitiveData(params) {
    if (!params || !Array.isArray(params)) return params;
    
    return params.map(param => {
      if (typeof param === 'string') {
        // Mask emails
        if (param.includes('@')) {
          const [local, domain] = param.split('@');
          return `${local.substring(0, 2)}***@${domain}`;
        }
        // Mask tokens
        if (param.length > 32 && /^[A-Za-z0-9]+$/.test(param)) {
          return `${param.substring(0, 8)}...`;
        }
      }
      return param;
    });
  }

  async close() {
    if (this.healthCheckInterval) {
      clearInterval(this.healthCheckInterval);
    }
    await this.pool.end();
    logger.info('PostgreSQL pool closed');
  }
}

module.exports = ResilientPostgresPool;

class ResilientRedisCache {

      enableReadyCheck: true,

      commandTimeout: 3000

    this.client = new Redis(this.config);
    this.defaultTTL = config.defaultTTL || 300; // 5 minutes
    this.setupEventListeners();
    this.setupCircuitBreaker();
  }

  setupEventListeners() {
    this.client.on('connect', () => {
      logger.info('Redis connected successfully');
      metrics.increment('redis.connections.active');
    });

    this.client.on('ready', () => {
      logger.info('Redis ready for commands');
      metrics.gauge('redis.status', 1);
    });

    this.client.on('error', (error) => {
      logger.error('Redis connection error', {
        error: error.message,
        stack: error.stack
      });
      metrics.increment('redis.errors.connection');
      metrics.gauge('redis.status', 0);
    });

    this.client.on('close', () => {
      logger.warn('Redis connection closed');
      metrics.decrement('redis.connections.active');
    });

    this.client.on('reconnecting', (delay) => {
      logger.info(`Redis reconnecting in ${delay}ms`);
      metrics.increment('redis.reconnections');
    });
  }

  setupCircuitBreaker() {
    const breakerOptions = {
      timeout: 5000, // 5 seconds timeout for cache operations
      errorThresholdPercentage: 50,
      resetTimeout: 30000,
      rollingCountTimeout: 60000,
      rollingCountBuckets: 10,
      name: 'redis-cache'
    };

    this.circuitBreaker = new CircuitBreaker(
      async (operation, ...args) => {
        const startTime = Date.now();
        const operationName = operation.name || 'unknown';
        
        try {
          const result = await operation(...args);
          const duration = Date.now() - startTime;
          
          metrics.histogram('redis.operation.duration', duration, { operation: operationName });
          metrics.increment('redis.operation.success', { operation: operationName });
          
          if (duration > 100) {
            logger.warn('Slow Redis operation detected', {
              operation: operationName,
              duration,
              args: this.maskCacheKeys(args)
            });
          }
          
          return result;
        } catch (error) {
          const duration = Date.now() - startTime;
          metrics.histogram('redis.operation.duration', duration, { operation: operationName });
          metrics.increment('redis.operation.error', { operation: operationName });
          throw error;
        }
      },
      breakerOptions

      logger.error('Redis circuit breaker opened - cache unavailable');
      metrics.increment('redis.circuit_breaker.open');
    });

    this.circuitBreaker.on('close', () => {
      logger.info('Redis circuit breaker closed - cache available');
      metrics.increment('redis.circuit_breaker.close');
    });

    this.circuitBreaker.fallback(async (operation, args, err) => {
      logger.error('Redis operation failed, using fallback', {
        error: err.message,
        operation: operation.name
      });

      return null;
    });

            const items = fs.readdirSync(dir);
            
            for (const item of items) {
                const fullPath = path.join(dir, item);
                const stat = fs.statSync(fullPath);
                
                if (stat.isDirectory()) {
                    // Ignorar node_modules e pastas ocultas
                    if (!item.includes('node_modules') && !item.startsWith('.')) {
                        scanDirectory(fullPath);
                    }
                } else if (extensions.some(ext => item.endsWith(ext))) {

## Decisões de Design (ADRs)

### ADR-001: Stateless por Design
**Contexto:** Necessidade de escalar horizontalmente sem complexidade
**Decisão:** Todos os serviços são stateless, estado externalizado
**Consequências:** + Escalabilidade, - Latência em alguns casos

### ADR-002: Event-Driven para Background Jobs
**Contexto:** Processamento assíncrono de tarefas pesadas
**Decisão:** RabbitMQ com dead letter queues
**Consequências:** + Resiliência, - Complexidade operacional

### ADR-003: Health Checks Hierárquicos
**Contexto:** Detecção precoce de falhas em cascata
**Decisão:** Health checks em 3 níveis (app, service, dependency)
**Consequências:** + Estabilidade, - Overhead de monitoramento

const axios = require('axios');
const { execSync } = require('child_process');
const fs = require('fs');
const path = require('path');

class VulnerabilityScanner {
  constructor(config) {
    this.config = config;
    this.snykApiKey = process.env.SNYK_API_KEY;
    this.owaspDependencyCheckPath = process.env.OWASP_DEPENDENCY_CHECK_PATH;
  }

  /**
   * Executa scanner completo de vulnerabilidades
   */
  async runFullScan() {
    const results = {
      timestamp: new Date().toISOString(),
      dependencies: await this.scanDependencies(),
      code: await this.scanCodeQuality(),
      secrets: await this.scanForSecrets(),
      configuration: await this.scanConfiguration(),
      summary: {}
    };

    // Classifica vulnerabilidades por severidade
    results.summary = this.classifyVulnerabilities(results);
    
    return results;
  }

  /**
   * Scanner de dependências usando Snyk
   */
  async scanDependencies() {
    const results = {
      snyk: null,
      npmAudit: null,
      owasp: null
    };

    try {
      // 1. Snyk Scan
      if (this.snykApiKey) {
        results.snyk = await this.runSnykScan();
      }

      // 2. NPM Audit
      results.npmAudit = this.runNpmAudit();

      // 3. OWASP Dependency Check
      if (this.owaspDependencyCheckPath) {
        results.owasp = this.runOwaspDependencyCheck();
      }

    } catch (error) {
      console.error('Erro no scan de dependências:', error);
    }

    return results;
  }

  /**
   * Executa Snyk via API
   */
  async runSnykScan() {
    try {
      const packageJson = JSON.parse(fs.readFileSync('package.json', 'utf8'));
      
      const response = await axios.post(
        'https://snyk.io/api/v1/test/npm',
        {
          package: packageJson
        },
        {
          headers: {
            'Authorization': `token ${this.snykApiKey}`,
            'Content-Type': 'application/json'
          }
        }
      );

      return this.parseSnykResults(response.data);
    } catch (error) {
      throw new Error(`Snyk scan failed: ${error.message}`);
    }
  }

  parseSnykResults(data) {
    const vulnerabilities = {
      critical: [],
      high: [],
      medium: [],
      low: []
    };

    if (data.vulnerabilities) {
      data.vulnerabilities.forEach(vuln => {
        const severity = vuln.severity.toLowerCase();
        vulnerabilities[severity].push({
          id: vuln.id,
          package: vuln.package,
          version: vuln.version,
          title: vuln.title,
          cvssScore: vuln.cvssScore,
          description: vuln.description,
          exploitMaturity: vuln.exploitMaturity,
          fixedIn: vuln.fixedIn,
          publicationTime: vuln.publicationTime
        });
      });
    }

    return {
      vulnerabilities,
      summary: {
        total: data.vulnerabilities?.length || 0,
        critical: vulnerabilities.critical.length,
        high: vulnerabilities.high.length,
        medium: vulnerabilities.medium.length,
        low: vulnerabilities.low.length
      }
    };
  }

  /**
   * Executa npm audit
   */
  runNpmAudit() {
    try {
      const output = execSync('npm audit --json', { encoding: 'utf8' });
      const auditData = JSON.parse(output);
      
      return {
        advisories: auditData.advisories || {},
        metadata: auditData.metadata || {},
        actions: auditData.actions || []
      };
    } catch (error) {
      return { error: error.message };
    }
  }

  /**
   * Scanner de qualidade de código
   */
  async scanCodeQuality() {
    const results = {
      eslint: this.runEslint(),
      sonarqube: await this.runSonarQubeScan(),
      securityPatterns: this.checkSecurityPatterns()
    };

    return results;
  }

  /**
   * Verifica padrões de segurança no código
   */
  checkSecurityPatterns() {
    const patterns = {
      sqlInjection: this.checkForSqlInjection(),
      xss: this.checkForXSS(),
      commandInjection: this.checkForCommandInjection(),
      pathTraversal: this.checkForPathTraversal(),
      insecureRandomness: this.checkForInsecureRandomness(),
      hardcodedSecrets: this.checkForHardcodedSecrets()
    };

    return patterns;
  }

  checkForSqlInjection() {
    const issues = [];
    const sourceDir = path.join(__dirname, '..', '..', 'src');
    
    this.walkDirectory(sourceDir, (filePath) => {
      if (filePath.endsWith('.js')) {
        const content = fs.readFileSync(filePath, 'utf8');
        
        // Padrões suspeitos
        const dangerousPatterns = [
          /query\s*\(\s*['"`].*?\$\{.*?\}.*?['"`]\)/gs,
          /execute\s*\(\s*['"`].*?\$\{.*?\}.*?['"`]\)/gs,
          /\.query\s*\(\s*['"`].*?\+\s*.*?['"`]\)/gs
        ];

        dangerousPatterns.forEach((pattern, index) => {
          const matches = content.match(pattern);
          if (matches) {
            issues.push({
              file: path.relative(sourceDir, filePath),
              pattern: `SQL Injection Pattern ${index + 1}`,
              matches: matches.slice(0, 3) // Mostra apenas os primeiros 3 matches
            });
          }
        });
      }
    });

    return {
      found: issues.length > 0,
      issues: issues,
      recommendation: issues.length > 0 
        ? 'Use prepared statements ou ORM para todas as queries SQL'
        : 'Nenhum padrão de SQL Injection detectado'
    };
  }

  checkForXSS() {
    const issues = [];
    const sourceDir = path.join(__dirname, '..', '..', 'src');
    
    this.walkDirectory(sourceDir, (filePath) => {
      if (filePath.endsWith('.js') || filePath.endsWith('.html')) {
        const content = fs.readFileSync(filePath, 'utf8');
        
        // Padrões suspeitos de XSS
        const dangerousPatterns = [
          /innerHTML\s*=\s*.*?\$\{.*?\}/gs,
          /\.html\s*\(\s*.*?\$\{.*?\}.*?\)/gs,
          /document\.write\s*\(\s*.*?\$\{.*?\}.*?\)/gs,
          /eval\s*\(\s*.*?\$\{.*?\}.*?\)/gs
        ];

        dangerousPatterns.forEach((pattern, index) => {
          const matches = content.match(pattern);
          if (matches) {
            issues.push({
              file: path.relative(sourceDir, filePath),
              pattern: `XSS Pattern ${index + 1}`,
              matches: matches.slice(0, 3)
            });
          }
        });
      }
    });

    return {
      found: issues.length > 0,
      issues: issues,
      recommendation: issues.length > 0
        ? 'Use textContent em vez de innerHTML e sanitize todos os inputs do usuário'
        : 'Nenhum padrão de XSS detectado'
    };
  }

  /**
   * Scanner de secrets no código
   */
  async scanForSecrets() {
    const secrets = [];
    const sourceDir = path.join(__dirname, '..', '..');
    
    // Padrões de secrets comuns
    const secretPatterns = [
      {
        name: 'API Key',
        pattern: /(api[_-]?key|apikey)\s*[:=]\s*['"`]([a-zA-Z0-9_\-]{20,})['"`]/gi
      },
      {
        name: 'Secret Key',
        pattern: /(secret[_-]?key|secretkey)\s*[:=]\s*['"`]([a-zA-Z0-9_\-]{20,})['"`]/gi
      },
      {
        name: 'Password',
        pattern: /(password|passwd|pwd)\s*[:=]\s*['"`]([^\s'"]{6,})['"`]/gi
      },
      {
        name: 'JWT Secret',
        pattern: /(jwt[_-]?secret)\s*[:=]\s*['"`]([a-zA-Z0-9_\-]{20,})['"`]/gi
      },
      {
        name: 'Database URL',
        pattern: /(postgresql|mysql|mongodb):\/\/[^:]+:[^@]+@/gi
      }
    ];

    this.walkDirectory(sourceDir, (filePath) => {
      // Ignora node_modules e outros diretórios
      if (filePath.includes('node_modules') || 
          filePath.includes('.git') ||
          filePath.endsWith('.log')) {
        return;
      }

      try {
        const content = fs.readFileSync(filePath, 'utf8');
        
        secretPatterns.forEach(patternConfig => {
          const matches = [...content.matchAll(patternConfig.pattern)];
          matches.forEach(match => {
            secrets.push({
              file: path.relative(sourceDir, filePath),
              type: patternConfig.name,
              match: match[0].substring(0, 50) + '...', // Mostra apenas parte do secret
              line: this.getLineNumber(content, match.index)
            });
          });
        });
      } catch (error) {
        // Ignora arquivos binários
      }
    });

    return {
      found: secrets.length > 0,
      secrets: secrets,
      recommendation: secrets.length > 0
        ? 'Mova todos os secrets para variáveis de ambiente ou gerenciador de secrets'
        : 'Nenhum secret hardcoded detectado'
    };
  }

  /**
   * Métodos auxiliares
   */
  walkDirectory(dir, callback) {
    const files = fs.readdirSync(dir);
    
    files.forEach(file => {
      const filePath = path.join(dir, file);
      const stat = fs.statSync(filePath);
      
      if (stat.isDirectory()) {
        this.walkDirectory(filePath, callback);
      } else {
        callback(filePath);
      }
    });
  }

  getLineNumber(content, index) {
    const lines = content.substring(0, index).split('\n');
    return lines.length;
  }

  classifyVulnerabilities(results) {
    const summary = {
      critical: 0,
      high: 0,
      medium: 0,
      low: 0,
      total: 0,
      slaCompliance: {
        critical: { withinSLA: true, maxHours: 24 },
        high: { withinSLA: true, maxHours: 72 },
        medium: { withinSLA: true, maxDays: 14 },
        low: { withinSLA: true, maxDays: 90 }
      }
    };

    // Conta vulnerabilidades de dependências
    if (results.dependencies?.snyk?.summary) {
      summary.critical += results.dependencies.snyk.summary.critical;
      summary.high += results.dependencies.snyk.summary.high;
      summary.medium += results.dependencies.snyk.summary.medium;
      summary.low += results.dependencies.snyk.summary.low;
    }

    // Conta issues de código
    if (results.code?.securityPatterns) {
      const patterns = results.code.securityPatterns;
      Object.values(patterns).forEach(pattern => {
        if (pattern.found && pattern.issues.length > 0) {
          summary.high += pattern.issues.length;
        }
      });
    }

    // Conta secrets
    if (results.secrets?.found) {
      summary.critical += results.secrets.secrets.length;
    }

    summary.total = summary.critical + summary.high + summary.medium + summary.low;

    return summary;
  }
}
