Architecture Decision Record — ADR-001: Microsserviços com Comunicação Híbrida
Data: 2024-01-15
Status: Aceita
Autores: Tech Lead + Especialista Infra

CONTEXTO:
A V11 do DIYAPP está enfrentando problemas de escalabilidade devido à arquitetura monolítica.
Necessitamos de:
1. Deploy independente por funcionalidade
2. Escala horizontal seletiva
3. Stack tecnológica heterogênea (Python/Node/Go)
4. Observabilidade granular por serviço

DECISÃO:
Adotar arquitetura de microsserviços com comunicação híbrida:
- Comunicação síncrona: gRPC para chamadas diretas serviço-serviço
- Comunicação assíncrona: Apache Kafka para eventos de domínio
- API Gateway: Kong para roteamento e rate limiting
- Service Mesh: Linkerd para comunicação segura entre serviços

OPÇÕES CONSIDERADAS:
- Opção A: REST + RabbitMQ — Prós: Simples, amplo suporte. Contras: Overhead HTTP, schemas fracos
- Opção B: gRPC + Kafka — Prós: Alta performance, schemas fortes, streaming. Contras: Curva de aprendizado
- Opção escolhida: B — Justificativa: Performance crítica para fluxos de IA, necessidade de schemas fortes para evolução

CONSEQUÊNCIAS:
Positivas: 
- Deploy independente por serviço
- Escala granular (ex: notification-service pode escalar separadamente)
- Stack tecnológica flexível por serviço
- Observabilidade por serviço

Negativas:
- Complexidade operacional aumentada
- Latência de rede entre serviços
- Necessidade de gerenciamento de dados distribuídos

Riscos:
- Falhas em cascata: Mitigação com circuit breakers e retries com backoff exponencial
- Consistência eventual: Mitigação com padrão Saga e compensações
- Debug distribuído: Mitigação com OpenTelemetry e correlation IDs

REVISÃO: 2024-04-15 (3 meses para avaliação de performance)
"""

# shared/tracing/otel_config.py
import os
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.resources import Resource, SERVICE_NAME, SERVICE_VERSION
from opentelemetry.instrumentation.grpc import GrpcInstrumentorClient, GrpcInstrumentorServer
from opentelemetry.instrumentation.kafka import KafkaInstrumentor
import logging
import json

class StructuredLogger:
    """Logger estruturado para observabilidade"""
    
    def __init__(self, service_name):
        self.service_name = service_name
        self.logger = logging.getLogger(service_name)
        self.tracer = trace.get_tracer(service_name)
        
    def info(self, message, **kwargs):
        span = trace.get_current_span()
        log_data = {
            "timestamp": self._iso_now(),
            "service": self.service_name,
            "level": "INFO",
            "message": message,
            "trace_id": format(span.get_span_context().trace_id, '032x') if span else None,
            "span_id": format(span.get_span_context().span_id, '016x') if span else None,
            **kwargs
        }
        self.logger.info(json.dumps(log_data))
    
    def error(self, message, error=None, **kwargs):
        span = trace.get_current_span()
        log_data = {
            "timestamp": self._iso_now(),
            "service": self.service_name,
            "level": "ERROR",
            "message": message,
            "error_type": error.__class__.__name__ if error else None,
            "error_message": str(error) if error else None,
            "trace_id": format(span.get_span_context().trace_id, '032x') if span else None,
            "span_id": format(span.get_span_context().span_id, '016x') if span else None,
            **kwargs
        }
        self.logger.error(json.dumps(log_data))
    
    def _iso_now(self):
        from datetime import datetime
        return datetime.utcnow().isoformat() + "Z"

def init_tracing(service_name, service_version="1.0.0"):
    """Inicializa OpenTelemetry para o serviço"""
    
    # Configurar recurso
    resource = Resource.create({
        SERVICE_NAME: service_name,
        SERVICE_VERSION: service_version,
        "deployment.environment": os.getenv("ENVIRONMENT", "development")
    })
    
    # Configurar provider
    trace.set_tracer_provider(TracerProvider(resource=resource))
    
    # Configurar exportador (Jaeger via OTLP)
    otlp_exporter = OTLPSpanExporter(
        endpoint=os.getenv("OTLP_ENDPOINT", "http://jaeger:4317"),
        insecure=True
    )
    
    # Adicionar processador
    span_processor = BatchSpanProcessor(otlp_exporter)
    trace.get_tracer_provider().add_span_processor(span_processor)
    
    # Instrumentar gRPC
    GrpcInstrumentorClient().instrument()
    GrpcInstrumentorServer().instrument()
    
    # Instrumentar Kafka
    KafkaInstrumentor().instrument()
    
    return StructuredLogger(service_name)

# Importações geradas pelo gRPC
from shared.protos import auth_pb2, auth_pb2_grpc

class BaseService:
    """Classe base para todos os microsserviços"""
    
    def __init__(self, name: str, port: int):
        self.name = name
        self.port = port
        self.logger = init_tracing(name)
        self.tracer = trace.get_tracer(name)
        
        # Configurações do Kafka
        self.kafka_bootstrap = os.getenv("KAFKA_BOOTSTRAP_SERVERS", "kafka:9092")
        self.consumer = None
        self.producer = None
        
        # Health check
        self.healthy = True
        
    async def start(self):
        """Inicia o serviço"""
        self.logger.info(f"Starting {self.name} service on port {self.port}")
        
        # Iniciar consumidor Kafka
        await self._start_kafka_consumer()
        
        # Iniciar servidor gRPC
        await self._start_grpc_server()
        
        self.logger.info(f"{self.name} service started successfully")
        
    async def _start_kafka_consumer(self):
        """Inicia consumidor de eventos Kafka"""
        try:
            self.consumer = aiokafka.AIOKafkaConsumer(
                f"{self.name}-events",
                bootstrap_servers=self.kafka_bootstrap,
                group_id=f"{self.name}-group",
                enable_auto_commit=True
            )
            await self.consumer.start()
            
            # Task para consumir eventos
            asyncio.create_task(self._consume_events())
            
            self.logger.info("Kafka consumer started")
        except Exception as e:
            self.logger.error("Failed to start Kafka consumer", error=e)
            
    async def _consume_events(self):
        """Consome eventos do Kafka"""
        async for msg in self.consumer:
            with self.tracer.start_as_current_span("process_kafka_event"):
                try:
                    await self.handle_event(msg.value)
                except Exception as e:
                    self.logger.error("Error processing event", error=e)
                    
    async def handle_event(self, event: Dict[str, Any]):
        """Método para ser sobrescrito pelos serviços"""
        pass
    
    async def _start_grpc_server(self):
        """Inicia servidor gRPC"""
        server = grpc.aio.server(
            interceptors=[server_interceptor()],
            options=[
                ('grpc.max_send_message_length', 50 * 1024 * 1024),
                ('grpc.max_receive_message_length', 50 * 1024 * 1024),
                ('grpc.keepalive_time_ms', 30000),
            ]
        )
        
        # Adicionar serviços (será sobrescrito)
        self.add_grpc_services(server)
        
        server.add_insecure_port(f'[::]:{self.port}')
        await server.start()
        
        # Graceful shutdown
        async def shutdown():
            await server.stop(5)
            
        asyncio.get_event_loop().add_signal_handler(
            signal.SIGTERM, 
            lambda: asyncio.create_task(shutdown())
        )
        
        await server.wait_for_termination()
        
    def add_grpc_services(self, server):
        """Método para ser sobrescrito pelos serviços"""
        pass
    
    async def produce_event(self, topic: str, event: Dict[str, Any]):
        """Produz evento no Kafka"""
        if not self.producer:
            self.producer = aiokafka.AIOKafkaProducer(
                bootstrap_servers=self.kafka_bootstrap
            )
            await self.producer.start()
            
        with self.tracer.start_as_current_span("produce_kafka_event"):
            try:
                await self.producer.send_and_wait(topic, json.dumps(event).encode())
                self.logger.info(f"Event produced to {topic}", event_type=event.get('event_type'))
            except Exception as e:
                self.logger.error(f"Failed to produce event to {topic}", error=e)
                
    def health_check(self):
        """Endpoint de health check"""
        return {
            "service": self.name,
            "status": "healthy" if self.healthy else "unhealthy",
            "timestamp": self.logger._iso_now(),
            "dependencies": {
                "kafka": self.consumer is not None,
                "grpc": True
            }
        }

```typescript
export class RedisClient {
  private static instance: Redis;
  private static connection: Redis;

  static getInstance(): Redis {
    if (!RedisClient.instance) {
      RedisClient.instance = new Redis({
        host: process.env.REDIS_HOST || 'localhost',
        port: parseInt(process.env.REDIS_PORT || '6379'),

        retryStrategy: (times: number) => {

          return delay;
        },

        enableReadyCheck: true,

      });

      RedisClient.instance.on('error', (error) => {

        console.log('✅ Redis connection established');
      });

      RedisClient.instance.on('ready', () => {
        console.log('✅ Redis client ready');
      });
    }

    return RedisClient.instance;
  }

  static async disconnect(): Promise<void> {

export class BullMQService {
  private static queues: Map<string, Queue> = new Map();
  private static workers: Map<string, Worker> = new Map();
  private static schedulers: Map<string, QueueScheduler> = new Map();

  static createQueue(name: string): Queue {
    if (this.queues.has(name)) {
      return this.queues.get(name)!;
    }

    const queue = new Queue(name, {
      connection: RedisClient.getInstance(),
      defaultJobOptions: {
        attempts: parseInt(process.env.QUEUE_ATTEMPTS || '3'),
        backoff: {
          type: 'exponential',
          delay: parseInt(process.env.QUEUE_BACKOFF_DELAY || '5000'),
        },
        removeOnComplete: 100,
        removeOnFail: 1000,
      },
    });

    this.queues.set(name, queue);
    return queue;
  }

  static createWorker<T = any>(
    name: string,
    processor: (job: any) => Promise<T>,
    concurrency: number = parseInt(process.env.QUEUE_CONCURRENCY || '5')
  ): Worker {
    if (this.workers.has(name)) {
      return this.workers.get(name)!;
    }

    const worker = new Worker(
      name,
      async (job) => {
        try {
          console.log(`🎯 Processing job ${job.id} from queue ${name}`);
          return await processor(job);
        } catch (error) {
          console.error(`❌ Job ${job.id} failed:`, error);
          throw error;
        }
      },
      {
        connection: RedisClient.getInstance(),
        concurrency,
      }
    );

    worker.on('completed', (job) => {
      console.log(`✅ Job ${job.id} completed successfully`);
    });

    worker.on('failed', (job, error) => {
      console.error(`❌ Job ${job?.id} failed:`, error);
    });

    this.workers.set(name, worker);
    return worker;
  }

  static createScheduler(name: string): QueueScheduler {
    if (this.schedulers.has(name)) {
      return this.schedulers.get(name)!;
    }

    const scheduler = new QueueScheduler(name, {
      connection: RedisClient.getInstance(),
    });

    this.schedulers.set(name, scheduler);
    return scheduler;
  }

  static async shutdown(): Promise<void> {
    for (const [name, worker] of this.workers) {
      await worker.close();
      console.log(`✅ Worker ${name} closed`);
    }

    for (const [name, queue] of this.queues) {
      await queue.close();
      console.log(`✅ Queue ${name} closed`);
    }

    for (const [name, scheduler] of this.schedulers) {
      await scheduler.close();
      console.log(`✅ Scheduler ${name} closed`);
    }
  }
}
```

```typescript
import { z } from 'zod';
import { Request, Response, NextFunction } from 'express';

export class ValidationError extends Error {
  constructor(
    public errors: z.ZodError,
    message: string = 'Validation failed'
  ) {
    super(message);
    this.name = 'ValidationError';
  }
}

export const validate = <T extends z.ZodType<any>>(schema: T) => {
  return (req: Request, res: Response, next: NextFunction): void => {
    try {
      const result = schema.safeParse(req.body);
      
      if (!result.success) {
        throw new ValidationError(result.error);
      }
      
      req.body = result.data;
      next();
    } catch (error) {
      if (error instanceof ValidationError) {
        res.status(400).json({
          error: 'Validation Error',
          details: error.errors.errors.map(err => ({
            field: err.path.join('.'),
            message: err.message,
          })),
          timestamp: new Date().toISOString(),
        });
      } else {
        next(error);
      }
    }
  };
};

export const validateQuery = <T extends z.ZodType<any>>(schema: T) => {
  return (req: Request, res: Response, next: NextFunction): void => {
    try {
      const result = schema.safeParse(req.query);
      
      if (!result.success) {
        throw new ValidationError(result.error);
      }
      
      req.query = result.data as any;
      next();
    } catch (error) {
      if (error instanceof ValidationError) {
        res.status(400).json({
          error: 'Query Validation Error',
          details: error.errors.errors.map(err => ({
            field: err.path.join('.'),
            message: err.message,
          })),
          timestamp: new Date().toISOString(),
        });
      } else {
        next(error);
      }
    }
  };
};
```

```typescript
import { create } from 'zustand'
import { persist } from 'zustand/middleware'
import { AuthState, User } from '@/types'
import { TOKEN_KEY, USER_KEY } from '@/constants'

interface AuthStore extends AuthState {
  login: (email: string, password: string) => Promise<void>
  logout: () => void
  setUser: (user: User) => void
  setToken: (token: string) => void
  clearError: () => void
  hydrateFromStorage: () => void
}

const initialState: AuthState = {
  user: null,
  token: null,
  isAuthenticated: false,
  isLoading: false,
  error: null,
}

export const useAuthStore = create<AuthStore>()(
  persist(
    (set, get) => ({
      ...initialState,

      login: async (email: string, password: string) => {
        set({ isLoading: true, error: null })
        
        try {
          // Simulação de login - substituir por chamada GraphQL real
          await new Promise(resolve => setTimeout(resolve, 1000))
          
          const mockUser: User = {
            id: '1',
            email,
            name: 'John Doe',
            role: 'admin',
            createdAt: new Date().toISOString(),
          }
          
          const mockToken = 'mock_jwt_token'
          
          set({
            user: mockUser,
            token: mockToken,
            isAuthenticated: true,
            isLoading: false,
          })
          
          // Armazenar no localStorage
          localStorage.setItem(TOKEN_KEY, mockToken)
          localStorage.setItem(USER_KEY, JSON.stringify(mockUser))
        } catch (error) {
          set({
            error: error instanceof Error ? error.message : 'Login failed',
            isLoading: false,
          })
        }
      },

      logout: () => {
        set(initialState)
        localStorage.removeItem(TOKEN_KEY)
        localStorage.removeItem(USER_KEY)
      },

      setUser: (user: User) => {
        set({ user })
        localStorage.setItem(USER_KEY, JSON.stringify(user))
      },

      setToken: (token: string) => {
        set({ token, isAuthenticated: true })
        localStorage.setItem(TOKEN_KEY, token)
      },

      clearError: () => set({ error: null }),

      hydrateFromStorage: () => {
        const token = localStorage.getItem(TOKEN_KEY)
        const userStr = localStorage.getItem(USER_KEY)
        
        if (token && userStr) {
          try {
            const user = JSON.parse(userStr)
            set({
              user,
              token,
              isAuthenticated: true,
            })
          } catch {
            // Limpar dados inválidos
            localStorage.removeItem(TOKEN_KEY)
            localStorage.removeItem(USER_KEY)
          }
        }
      },
    }),
    {
      name: 'auth-storage',
      partialize: (state) => ({
        user: state.user,
        token: state.token,
        isAuthenticated: state.isAuthenticated,
      }),
    }
  )
)
```

```typescript
import { create } from 'zustand'
import { UIState, Notification } from '@/types'

interface UIStore extends UIState {
  toggleTheme: () => void
  toggleSidebar: () => void
  addNotification: (notification: Omit<Notification, 'id' | 'createdAt' | 'read'>) => void
  markNotificationAsRead: (id: string) => void
  clearNotifications: () => void
  setLoading: (isLoading: boolean) => void
}

const initialState: UIState = {
  theme: 'light',
  sidebarOpen: true,
  notifications: [],
  isLoading: false,
}

export const useUIStore = create<UIStore>((set, get) => ({
  ...initialState,

  toggleTheme: () => {
    set((state) => ({
      theme: state.theme === 'light' ? 'dark' : 'light',
    }))
    
    // Aplicar tema ao documento
    const theme = get().theme
    document.documentElement.classList.toggle('dark', theme === 'dark')
  },

  toggleSidebar: () => {
    set((state) => ({
      sidebarOpen: !state.sidebarOpen,
    }))
  },

  addNotification: (notification) => {
    const newNotification: Notification = {
      ...notification,
      id: Date.now().toString(),
      createdAt: new Date().toISOString(),
      read: false,
    }
    
    set((state) => ({
      notifications: [newNotification, ...state.notifications].slice(0, 50), // Limitar a 50 notificações
    }))
  },

  markNotificationAsRead: (id: string) => {
    set((state) => ({
      notifications: state.notifications.map((n) =>
        n.id === id ? { ...n, read: true } : n
      ),
    }))
  },

  clearNotifications: () => {
    set({ notifications: [] })
  },

  setLoading: (isLoading: boolean) => {
    set({ isLoading })
  },
}))
```

```typescript
import { ApolloClient, InMemoryCache, createHttpLink, split } from '@apollo/client'
import { setContext } from '@apollo/client/link/context'
import { GraphQLWsLink } from '@apollo/client/link/subscriptions'
import { createClient } from 'graphql-ws'
import { getMainDefinition } from '@apollo/client/utilities'
import { API_URL, WS_URL } from '@/constants'
import { useAuthStore } from '@/store/auth.store'

// Link HTTP para queries/mutations
const httpLink = createHttpLink({
  uri: API_URL,
})

// Link WebSocket para subscriptions
const wsLink = new GraphQLWsLink(
  createClient({
    url: WS_URL,
    connectionParams: () => {
      const token = useAuthStore.getState().token
      return {
        authorization: token ? `Bearer ${token}` : '',
      }
    },
  })
)

// Link de autenticação
const authLink = setContext((_, { headers }) => {
  const token = useAuthStore.getState().token
  
  return {
    headers: {
      ...headers,
      authorization: token ? `Bearer ${token}` : '',
    },
  }
})

// Split link: WebSocket para subscriptions, HTTP para o resto
const splitLink = split(
  ({ query }) => {
    const definition = getMainDefinition(query)
    return (
      definition.kind === 'OperationDefinition' &&
      definition.operation === 'subscription'
    )
  },
  wsLink,
  authLink.concat(httpLink)
)

// Cache configuration
const cache = new InMemoryCache({
  typePolicies: {
    Query: {
      fields: {
        projects: {
          merge(existing = [], incoming: any[]) {
            return incoming
          },
        },
        users: {
          merge(existing = [], incoming: any[]) {
            return incoming
          },
        },
      },
    },
    Project: {
      fields: {
        owner: {
          merge(existing, incoming) {
            return incoming || existing
          },
        },
      },
    },
  },
})

// Apollo Client instance
export const apolloClient = new ApolloClient({
  link: splitLink,
  cache,
  defaultOptions: {
    watchQuery: {
      fetchPolicy: 'cache-and-network',
      errorPolicy: 'all',
    },
    query: {
      fetchPolicy: 'network-only',
      errorPolicy: 'all',
    },
    mutate: {
      errorPolicy: 'all',
    },
  },
})
```

```python
#!/usr/bin/env python3
"""
Setup script for BigQuery data warehouse for DIYAPP V12
Creates datasets, tables, and sets up proper IAM permissions
"""

import os
from google.cloud import bigquery
from google.cloud.bigquery import Dataset, Table, TimePartitioning, ClusteringField
import json
from datetime import datetime
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class BigQuerySetup:
    def __init__(self, project_id):
        self.project_id = project_id
        self.client = bigquery.Client(project=project_id)
        
    def create_datasets(self):
        """Create the three-layer dataset structure"""
        datasets = {
            'diyapp_raw': {
                'description': 'Raw data layer - never deleted',
                'default_table_expiration_ms': None,  # Never expire
                'labels': {'layer': 'raw', 'environment': 'production'}
            },
            'diyapp_staging': {
                'description': 'Staging layer - cleaned and standardized data',
                'default_table_expiration_ms': 30 * 24 * 60 * 60 * 1000,  # 30 days
                'labels': {'layer': 'staging', 'environment': 'production'}
            },
            'diyapp_marts': {
                'description': 'Business marts layer - ready for consumption',
                'default_table_expiration_ms': 90 * 24 * 60 * 60 * 1000,  # 90 days
                'labels': {'layer': 'marts', 'environment': 'production'}
            }
        }
        
        for dataset_id, config in datasets.items():
            dataset_ref = self.client.dataset(dataset_id)
            dataset = Dataset(dataset_ref)
            dataset.description = config['description']
            dataset.default_table_expiration_ms = config['default_table_expiration_ms']
            dataset.labels = config['labels']
            
            try:
                dataset = self.client.create_dataset(dataset)
                logger.info(f"Created dataset {dataset_id}")
            except Exception as e:
                logger.info(f"Dataset {dataset_id} already exists or error: {e}")
    
    def create_raw_tables(self):
        """Create raw tables with proper partitioning"""
        raw_tables = {
            'events': {
                'schema': [
                    bigquery.SchemaField('event_id', 'STRING', mode='REQUIRED'),
                    bigquery.SchemaField('user_id', 'STRING', mode='REQUIRED'),
                    bigquery.SchemaField('session_id', 'STRING'),
                    bigquery.SchemaField('event_name', 'STRING', mode='REQUIRED'),
                    bigquery.SchemaField('event_properties', 'STRING'),  # JSON as string
                    bigquery.SchemaField('event_timestamp', 'TIMESTAMP', mode='REQUIRED'),
                    bigquery.SchemaField('app_version', 'STRING'),
                    bigquery.SchemaField('platform', 'STRING'),
                    bigquery.SchemaField('source', 'STRING'),
                    bigquery.SchemaField('ingestion_timestamp', 'TIMESTAMP', mode='REQUIRED'),

                'partitioning': TimePartitioning(

                    field='event_timestamp'
                ),
                'clustering': ['user_id', 'event_name']
            },
            'llm_calls': {

                    bigquery.SchemaField('call_id', 'STRING', mode='REQUIRED'),

                    bigquery.SchemaField('session_id', 'STRING'),
                    bigquery.SchemaField('model', 'STRING', mode='REQUIRED'),
                    bigquery.SchemaField('provider', 'STRING', mode='REQUIRED'),
                    bigquery.SchemaField('feature', 'STRING', mode='REQUIRED'),
                    bigquery.SchemaField('input_tokens', 'INTEGER'),
                    bigquery.SchemaField('output_tokens', 'INTEGER'),
                    bigquery.SchemaField('total_tokens', 'INTEGER'),
                    bigquery.SchemaField('cost_usd', 'FLOAT'),
                    bigquery.SchemaField('latency_ms', 'INTEGER'),
                    bigquery.SchemaField('success', 'BOOLEAN'),
                    bigquery.SchemaField('error_message', 'STRING'),
                    bigquery.SchemaField('request_timestamp', 'TIMESTAMP', mode='REQUIRED'),
                    bigquery.SchemaField('ingestion_timestamp', 'TIMESTAMP', mode='REQUIRED'),
                ],
                'partitioning': TimePartitioning(
                    type_=TimePartitioningType.DAY,
                    field='request_timestamp'
                ),
                'clustering': ['feature', 'model', 'provider']
            },
            'user_sessions': {
                'schema': [
                    bigquery.SchemaField('session_id', 'STRING', mode='REQUIRED'),

                    bigquery.SchemaField('country', 'STRING'),
                    bigquery.SchemaField('ingestion_timestamp', 'TIMESTAMP', mode='REQUIRED'),

                'partitioning': TimePartitioning(
                    type_=TimePartitioningType.DAY,
                    field='start_timestamp'
                ),
                'clustering': ['user_id', 'country']
            }
        }
        
        for table_name, config in raw_tables.items():
            table_ref = self.client.dataset('diyapp_raw').table(table_name)
            table = Table(table_ref, schema=config['schema'])
            
            if 'partitioning' in config:
                table.time_partitioning = config['partitioning']

                    COUNT(*) as total_events,
                    ARRAY_AGG(DISTINCT event_name) as event_types
                FROM `{project_id}.diyapp_raw.events`

                        WHEN TIMESTAMP_DIFF(CURRENT_TIMESTAMP(), creation_time, DAY) > 700 THEN 'WARNING'

                    END as retention_status
```

      if (this.tokenBlacklist.has(decoded.jti)) {

    }
  }

    try {
      const decoded = jwt.verify(token, this.config.refresh.secret, {
        algorithms: [this.config.refresh.algorithm]
      });
      
      if (decoded.type !== 'refresh') {
        throw new Error('Not a refresh token');
      }
      
      return decoded;
    } catch (error) {
      throw new Error(`Invalid refresh token: ${error.message}`);
    }
  }

    try {
      const decoded = jwt.decode(token);
      if (decoded && decoded.jti) {
        this.tokenBlacklist.add(decoded.jti);
        
        // Schedule cleanup (in production, set TTL in Redis)
        setTimeout(() => {
          this.tokenBlacklist.delete(decoded.jti);
        }, 24 * 60 * 60 * 1000); // 24 hours
      }
    } catch (error) {
      console.error('Error revoking token:', error);
    }
  }

  rotateTokens(user) {
    // Generate new tokens and revoke old ones
    const accessToken = this.generateAccessToken({
      userId: user.id,
      role: user.role
    });
    
    const refreshToken = this.generateRefreshToken({
      userId: user.id
    });

    def detect_cost_anomalies(self, cost_data: pd.DataFrame, threshold_pct: float = 20.0) -> List[Dict]:
        """Detect cost anomalies (>20% increase without new features)"""
        anomalies = []
        
        # Group by feature and day
        daily_costs = cost_data.groupby(['feature', pd.Grouper(key='date', freq='D')])['total_cost'].sum().reset_index()
        
        for feature in daily_costs['feature'].unique():
            feature_data = daily_costs[daily_costs['feature'] == feature].sort_values('date')
            
            if len(feature_data) < 7:  # Need at least a week of data
                continue
                
            # Calculate 7-day moving average
            feature_data['7d_avg'] = feature_data['total_cost'].rolling(window=7, min_periods=1).mean()
            feature_data['pct_change'] = ((feature_data['total_cost'] - feature_data['7d_avg']) / feature_data['7d_avg']) * 100
            
            # Detect anomalies
            recent_data = feature_data.tail(3)  # Last 3 days
            for _, row in recent_data.iterrows():
                if row['pct_change'] > threshold_pct:
                    anomaly = {
                        'timestamp': row['date'],
                        'anomaly_type': 'cost',
                        'severity': 'high' if row['pct_change'] > 50 else 'medium',
                        'feature': feature,
                        'description': f"Cost increased by {row['pct_change']:.1f}% for feature '{feature}'",
                        'metrics': {
                            'current_cost': row['total_cost'],
                            '7d_avg': row['7d_avg'],
                            'pct_increase': row['pct_change']
                        }
                    }
                    anomalies.append(anomaly)
                    
        return anomalies
    
    def detect_latency_anomalies(self, latency_data: pd.DataFrame, slo_ms: float = 8000) -> List[Dict]:
        """Detect latency anomalies (P95 > SLO)"""
        anomalies = []
        
        # Group by provider and hour
        latency_data['hour'] = latency_data['timestamp'].dt.floor('H')
        hourly_latency = latency_data.groupby(['provider', 'hour'])['latency_ms'].agg(['count', 'mean', lambda x: np.percentile(x, 95)]).reset_index()
        hourly_latency.columns = ['provider', 'hour', 'count', 'mean', 'p95']
        
        for _, row in hourly_latency.iterrows():
            if row['p95'] > slo_ms and row['count'] > 10:  # Only if we have enough samples
                anomaly = {
                    'timestamp': row['hour'],
                    'anomaly_type': 'latency',
                    'severity': 'high' if row['p95'] > slo_ms * 2 else 'medium',
                    'provider': row['provider'],
                    'description': f"P95 latency {row['p95']:.0f}ms exceeds SLO ({slo_ms}ms) for {row['provider']}",
                    'metrics': {
                        'p95_latency': row['p95'],
                        'mean_latency': row['mean'],
                        'call_count': row['count'],
                        'slo': slo_ms
                    }
                }
                anomalies.append(anomaly)
                
        return anomalies
    
    def detect_quality_drift(self, quality_data: pd.DataFrame, threshold_pct: float = 10.0) -> List[Dict]:
        """Detect quality drift (>10% degradation)"""
        anomalies = []
        
        for feature in quality_data['feature'].unique():
            feature_data = quality_data[quality_data['feature'] == feature].sort_values('timestamp')
            
            if len(feature_data) < 2:
                continue
                
            baseline = feature_data['baseline_score'].iloc[0]

```javascript
// Helper para calcular similaridade semântica
function calculateSemanticSimilarity(texts) {
  // Implementação simplificada - em produção usar embedding + cosine similarity
  const words = texts.map(text => 
    new Set(text.toLowerCase().split(/\W+/).filter(w => w.length > 3))
  );
  
  let totalSimilarity = 0;
  let comparisons = 0;
  
  for (let i = 0; i < words.length; i++) {
    for (let j = i + 1; j < words.length; j++) {
      const intersection = new Set([...words[i]].filter(x => words[j].has(x)));
      const union = new Set([...words[i], ...words[j]]);
      const similarity = intersection.size / union.size;
      totalSimilarity += similarity;
      comparisons++;
    }
  }
  
  return totalSimilarity / comparisons;
}
```

## 2. server.js (Servidor Express)

```javascript
const express = require('express');
const cors = require('cors');
const helmet = require('helmet');
const path = require('path');
const { createLogger, format, transports } = require('winston');

// Configuração do logger
const logger = createLogger({
  level: 'info',
  format: format.combine(
    format.timestamp(),
    format.errors({ stack: true }),
    format.json()
  ),
  transports: [
    new transports.File({ filename: 'logs/error.log', level: 'error' }),
    new transports.File({ filename: 'logs/combined.log' }),
    new transports.Console({
      format: format.combine(
        format.colorize(),
        format.simple()
      )
    })
  ]
});

const app = express();
const PORT = process.env.PORT || 3000;

// Middleware de segurança
app.use(helmet({
  contentSecurityPolicy: {
    directives: {
      defaultSrc: ["'self'"],
      scriptSrc: ["'self'", "'unsafe-inline'", "https://cdn.jsdelivr.net"],
      styleSrc: ["'self'", "'unsafe-inline'", "https://fonts.googleapis.com"],
      imgSrc: ["'self'", "data:", "https:"],
      connectSrc: ["'self'", "ws:", "wss:"],
      mediaSrc: ["'self'"],
      fontSrc: ["'self'", "https://fonts.gstatic.com"]
    }
  }
}));

app.use(cors());
app.use(express.json());
app.use(express.static(path.join(__dirname, 'public')));
app.use('/src', express.static(path.join(__dirname, 'src')));

// Rotas
app.get('/', (req, res) => {
  res.sendFile(path.join(__dirname, 'index.html'));
});

app.get('/api/health', (req, res) => {
  res.json({ status: 'healthy', timestamp: new Date().toISOString() });
});

// Rota para modelos 3D
app.get('/api/models', (req, res) => {
  const models = [
    { id: 'chair', name: 'Simple Chair', file: 'simple-chair.glb', category: 'furniture' },
    { id: 'shelf', name: 'Bookshelf', file: 'shelf.glb', category: 'furniture' },
    { id: 'table', name: 'Coffee Table', file: 'table.glb', category: 'furniture' }
  ];
  res.json(models);
});

// Middleware de erro
app.use((err, req, res, next) => {
  logger.error(err.stack);
  res.status(500).json({ error: 'Something went wrong!' });
});

app.listen(PORT, () => {
  logger.info(`DIYAPP WebXR Server running on port ${PORT}`);
  logger.info(`Access: http://localhost:${PORT}`);
});
```
