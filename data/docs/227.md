class ArchitectureDecisionRecord:
    """Template para documentação de decisões arquiteturais"""
    
    def __init__(self, number: int, title: str, status: str = "Proposta"):
        self.number = number
        self.title = title
        self.status = status
        self.date = datetime.now().strftime("%Y-%m-%d")
        self.authors = ["Tech Lead", "Especialista Infra", "Especialista Backend"]
        self.context = ""
        self.decision = ""
        self.options = []
        self.consequences = {"positive": [], "negative": [], "risks": []}
        self.review_date = ""
        
    def to_markdown(self) -> str:
        """Converte ADR para formato Markdown"""
        options_text = "\n".join([f"- {opt}" for opt in self.options])
        consequences_text = f"""
Positivas:
{chr(10).join([f'- {c}' for c in self.consequences['positive']])}

Negativas:
{chr(10).join([f'- {c}' for c in self.consequences['negative']])}

Riscos:
{chr(10).join([f'- {c}' for c in self.consequences['risks']])}
        """
        
        return f"""# ADR-{self.number:03d}: {self.title}

**Data**: {self.date}
**Status**: {self.status}
**Autores**: {', '.join(self.authors)}

## CONTEXTO
{self.context}

## DECISÃO
{self.decision}

## OPÇÕES CONSIDERADAS
{options_text}

## CONSEQUÊNCIAS
{consequences_text}

## REVISÃO
{self.review_date}
"""

class CommunicationPattern(Enum):
    """Padrões de comunicação entre serviços"""
    GRPC_SYNC = "gRPC Síncrono"

    REST_HTTP = "REST HTTP"
    GRPC_STREAMING = "gRPC Streaming"
    WEBHOOK = "Webhook"

@dataclass
class ServiceContract:

    target: str
    pattern: CommunicationPattern
    protocol: str
    endpoint: str
    timeout_ms: int
    retry_policy: Dict[str, Any]
    circuit_breaker: Dict[str, Any]
    
    def to_dict(self) -> Dict:
        return asdict(self)

```typescript
import React from 'react';
import { colors, spacing, typography } from '../../tokens';
import { ButtonProps } from './Button.types';
import './Button.css';

export const Button: React.FC<ButtonProps> = ({
  children,
  variant = 'primary',
  size = 'medium',
  isLoading = false,
  disabled = false,
  fullWidth = false,
  startIcon,
  endIcon,
  onClick,
  type = 'button',
  className = '',
  ...rest
}) => {
  const baseClasses = 'button';
  const variantClasses = `button--${variant}`;
  const sizeClasses = `button--${size}`;
  const widthClass = fullWidth ? 'button--full-width' : '';
  const loadingClass = isLoading ? 'button--loading' : '';
  const disabledClass = disabled ? 'button--disabled' : '';
  
  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    if (!disabled && !isLoading && onClick) {
      onClick(e);
    }
  };
  
  return (
    <button
      type={type}
      className={`${baseClasses} ${variantClasses} ${sizeClasses} ${widthClass} ${loadingClass} ${disabledClass} ${className}`}
      onClick={handleClick}
      disabled={disabled || isLoading}
      aria-busy={isLoading}
      {...rest}
    >
      {isLoading && (
        <span className="button__loader" aria-hidden="true">
          <div className="button__loader-dot" />
          <div className="button__loader-dot" />
          <div className="button__loader-dot" />
        </span>
      )}
      
      {!isLoading && startIcon && (
        <span className="button__icon button__icon--start">
          {startIcon}
        </span>
      )}
      
      <span className="button__content">
        {children}
      </span>
      
      {!isLoading && endIcon && (
        <span className="button__icon button__icon--end">
          {endIcon}
        </span>
      )}
    </button>
  );
};
```

```sql
    event_id UUID DEFAULT generateUUIDv4(),
    session_id String,
    user_id Nullable(String),
    event_type Enum8(
        'page_view' = 1,
        'button_click' = 2,
```

# DIYAPP Evolution - V12 Core - Threat Modeling e Segurança

## 1. Threat Modeling da Nova Arquitetura (STRIDE)

### Documento de Threat Modeling - Arquitetura V12

**Data:** 2024-01-15  
**Responsável:** Especialista em Segurança  
**Arquitetura:** DIYAPP V12 Core  
**Status:** Ativo  

---

### 1.1 Ativos Críticos

| Ativo | Descrição | Classificação |
|-------|-----------|---------------|
| Dados de usuário | PII, credenciais, histórico de uso | Crítico |
| Tokens de API | Chaves de integração com serviços externos | Crítico |
| Infraestrutura | Servidores, containers, rede | Alto |
| Código fonte | Lógica de negócio, configurações | Médio |
| Logs de auditoria | Registros de acesso e operações | Médio |

### 1.2 Análise STRIDE

#### **Spoofing (Falsificação)**
- **Ameaça:** Ataque de credential stuffing em endpoints de autenticação
- **Impacto:** Acesso não autorizado a contas de usuário
- **Controles:** 
  - MFA obrigatório para contas administrativas
  - Rate limiting por IP (100 req/min)
  - Monitoramento de tentativas falhas
  - Tokens JWT com expiração curta (15min)

#### **Tampering (Manipulação)**
- **Ameaça:** Manipulação de dados em trânsito ou em repouso
- **Impacto:** Integridade dos dados comprometida
- **Controles:**
  - TLS 1.3 em todas as comunicações
  - Assinatura digital de logs de auditoria
  - Validação de checksum em uploads de arquivos
  - WAF com proteção contra SQL injection

#### **Repudiation (Repúdio)**
- **Ameaça:** Usuários negam ações realizadas
- **Impacto:** Falta de accountability
- **Controles:**
  - Logs imutáveis em SIEM centralizado
  - Timestamp assinado digitalmente
  - Auditoria completa de operações privilegiadas

#### **Information Disclosure (Divulgação)**
- **Ameaça:** Vazamento de dados sensíveis
- **Impacto:** Violação de LGPD, perda de confiança
- **Controles:**
  - Criptografia AES-256 para PII em repouso
  - Mascaramento de dados em logs
  - Política de least privilege para acesso a dados
  - DLP (Data Loss Prevention) em endpoints

#### **Denial of Service (Negação)**
- **Ameaça:** Ataques DDoS contra APIs públicas
- **Impacto:** Indisponibilidade do serviço
- **Controles:**
  - CDN com proteção DDoS (Cloudflare)
  - Rate limiting por usuário/IP
  - Auto-scaling com limites de custo
  - Circuit breakers em chamadas externas

#### **Elevation of Privilege (Elevação)**
- **Ameaça:** Exploração de vulnerabilidades para acesso privilegiado
- **Impacto:** Controle total do sistema
- **Controles:**
  - RBAC granular (nível de recurso)
  - Revisão periódica de permissões
  - Segregação de ambientes (dev/staging/prod)
  - Scanning contínuo de vulnerabilidades

### 1.3 Riscos Aceitos

| Risco | Justificativa | Proprietário |
|-------|--------------|--------------|
| Vulnerabilidades de dia zero em dependências | Mitigado por WAF e runtime protection | Especialista em Segurança |
| Ataques DDoS de grande escala | Mitigado por CDN, custo aceitável para proteção | SRE |
| Social engineering contra administradores | Mitigado por treinamento e MFA | Squad Leader |

### 2.2 Implementação do WAF Controller - `waf-controller.js`

```javascript
// src/security/waf-controller.js
const crypto = require('crypto');
const logger = require('../utils/logger');

class WAFController {
  constructor(config) {
    this.config = config;
    this.rules = this.loadRules(config.rules);
    this.requestCache = new Map();
    this.blockedIPs = new Set();
    
    // Inicializar limpeza periódica do cache
    setInterval(() => this.cleanCache(), 60000); // 1 minuto
  }

  loadRules(rulesConfig) {
    return rulesConfig.map(rule => ({
      ...rule,
      compiledPattern: rule.conditions?.map(cond => 
        cond.operator === 'rx' ? new RegExp(cond.value, 'i') : null
      ).filter(Boolean)
    }));
  }

  async inspectRequest(req) {
    const requestId = crypto.randomUUID();
    const clientIP = req.headers['x-forwarded-for'] || req.socket.remoteAddress;
    const timestamp = Date.now();
    
    // Verificar IP bloqueado
    if (this.blockedIPs.has(clientIP)) {
      logger.warn(`Request from blocked IP: ${clientIP}`, { requestId });
      return { action: 'block', reason: 'IP_BLOCKED' };
    }

    // Aplicar regras
    for (const rule of this.rules) {
      const match = this.evaluateRule(rule, req);
      if (match) {
        logger.security(`WAF Rule triggered: ${rule.name}`, {
          requestId,
          clientIP,
          ruleId: rule.id,
          path: req.url,
          method: req.method
        });

        // Registrar métrica
        this.recordMetric('waf_rule_triggered', {
          rule_id: rule.id,
          action: rule.action
        });

        return {
          action: rule.action,
          rule: rule.name,
          requestId
        };
      }
    }

    // Rate limiting
    const rateLimitResult = this.checkRateLimit(clientIP, req.url);
    if (rateLimitResult.blocked) {
      return {
        action: 'challenge',
        reason: 'RATE_LIMIT_EXCEEDED',
        retryAfter: rateLimitResult.retryAfter
      };
    }

    return { action: 'allow' };
  }

  evaluateRule(rule, req) {
    for (const condition of rule.conditions) {
      let value;
      
      switch (condition.field) {
        case 'ARGS':
          value = JSON.stringify(req.query) + JSON.stringify(req.body);
          break;
        case 'REQUEST_URI':
          value = req.url;
          break;
        case 'USER_AGENT':
          value = req.headers['user-agent'] || '';
          break;
        case 'ARGS_COUNT':
          value = Object.keys(req.query).length + Object.keys(req.body || {}).length;
          break;
        default:
          continue;
      }

      if (!this.evaluateCondition(condition, value)) {
        return false;
      }
    }
    return true;
  }

  evaluateCondition(condition, value) {
    switch (condition.operator) {
      case 'rx':
        return condition.compiledPattern?.some(pattern => pattern.test(value));
      case 'gt':
        return Number(value) > Number(condition.value);
      case 'lt':
        return Number(value) < Number(condition.value);
      case 'eq':
        return value === condition.value;
      default:
        return false;
    }
  }

  checkRateLimit(clientIP, endpoint) {
    const key = `${clientIP}:${endpoint}`;
    const now = Date.now();
    const windowMs = 60000; // 1 minuto
    
    if (!this.requestCache.has(key)) {
      this.requestCache.set(key, {
        count: 1,
        firstRequest: now,
        lastRequest: now
      });
      return { blocked: false };
    }
    
    const data = this.requestCache.get(key);
    
    // Resetar se a janela expirou
    if (now - data.firstRequest > windowMs) {
      data.count = 1;
      data.firstRequest = now;
    } else {
      data.count++;
    }
    
    data.lastRequest = now;
    
    // Limite: 100 requests por minuto
    if (data.count > 100) {
      const retryAfter = Math.ceil((windowMs - (now - data.firstRequest)) / 1000);
      return { blocked: true, retryAfter };
    }
    
    return { blocked: false };
  }

  blockIP(ip, reason, duration = 3600000) { // 1 hora padrão
    this.blockedIPs.add(ip);
    logger.security(`IP blocked: ${ip}`, { reason, duration });
    
    setTimeout(() => {
      this.blockedIPs.delete(ip);
      logger.security(`IP unblocked: ${ip}`);
    }, duration);
  }

  cleanCache() {
    const now = Date.now();
    const maxAge = 300000; // 5 minutos
    
    for (const [key, data] of this.requestCache.entries()) {
      if (now - data.lastRequest > maxAge) {
        this.requestCache.delete(key);
      }
    }
  }

  recordMetric(name, labels) {
    // Integrar com sistema de métricas (Prometheus)
    console.log(`[METRIC] ${name}`, labels);
  }

  getStats() {
    return {
      totalRules: this.rules.length,
      blockedIPs: this.blockedIPs.size,
      cacheSize: this.requestCache.size,
      activeMode: this.config.mode
    };
  }
}

module.exports = WAFController;
```

### 2.3 Middleware Express - `waf-middleware.js`

```javascript
// src/middleware/waf-middleware.js
const WAFController = require('../security/waf-controller');
const config = require('../../waf-config.yaml');

const wafController = new WAFController(config);

function wafMiddleware(req, res, next) {
  // Pular WAF para health checks
  if (req.path === '/api/health' || req.path === '/health') {
    return next();
  }

  wafController.inspectRequest(req)
    .then(result => {
      switch (result.action) {
        case 'allow':
          req.wafResult = result;
          next();
          break;
          
        case 'block':
          res.status(403).json({
            error: 'Access denied',
            requestId: result.requestId,
            message: 'Request blocked by security policy'
          });
          break;
          
        case 'challenge':
          res.status(429).json({
            error: 'Rate limit exceeded',
            retryAfter: result.retryAfter,
```
