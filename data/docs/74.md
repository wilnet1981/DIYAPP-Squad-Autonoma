# ============================================================================
# 1. ARCHITECTURE DECISION RECORDS (ADRs) SYSTEM
# ============================================================================

class ADRStatus(Enum):
    PROPOSED = "proposed"
    ACCEPTED = "accepted"
    DEPRECATED = "deprecated"
    SUPERSEDED = "superseded"

@dataclass
class ArchitectureDecision:
    id: str
    title: str
    date: str
    status: ADRStatus
    authors: List[str]
    context: str
    decision: str
    options: List[Dict]
    consequences: Dict
    review_date: Optional[str]
    superseded_by: Optional[str]
    
    def to_markdown(self) -> str:
        """Convert ADR to markdown format"""
        options_str = "\n".join([
            f"- **{opt['name']}**: {opt['description']}\n"
            f"  - Prós: {', '.join(opt['pros'])}\n"
            f"  - Contras: {', '.join(opt['cons'])}"
            for opt in self.options
        ])
        
        return f"""# ADR-{self.id}: {self.title}

**Data**: {self.date}
**Status**: {self.status.value.upper()}
**Autores**: {', '.join(self.authors)}
{"**Substituído por**: " + self.superseded_by if self.superseded_by else ""}

## CONTEXTO
{self.context}

## DECISÃO
{self.decision}

## OPÇÕES CONSIDERADAS
{options_str}

## CONSEQUÊNCIAS
**Positivas**:
{self.consequences['positive']}

**Negativas**:
{self.consequences['negative']}

**Riscos**:
{self.consequences['risks']}

{"**Revisão programada**: " + self.review_date if self.review_date else ""}
"""

class ADRManager:
    """Gerencia ADRs - Architecture Decision Records"""
    
    def __init__(self, base_path: str = "src/architecture/adrs"):
        self.base_path = Path(base_path)
        self.base_path.mkdir(parents=True, exist_ok=True)
        
    def create_adr(self, title: str, context: str, decision: str, 
                   options: List[Dict], authors: List[str]) -> str:
        """Cria um novo ADR"""
        adr_id = self._get_next_id()
        date = datetime.datetime.now().strftime("%Y-%m-%d")
        
        adr = ArchitectureDecision(
            id=adr_id,
            title=title,
            date=date,
            status=ADRStatus.PROPOSED,
            authors=authors,
            context=context,
            decision=decision,
            options=options,
            consequences={
                "positive": "",
                "negative": "",
                "risks": ""
            },
            review_date=None,
            superseded_by=None
        )
        
        # Salva o ADR
        file_path = self.base_path / f"ADR-{adr_id}.md"
        file_path.write_text(adr.to_markdown())
        
        # Salva metadados
        meta_path = self.base_path / f"ADR-{adr_id}.json"
        meta_path.write_text(json.dumps(asdict(adr), indent=2, default=str))
        
        return adr_id
    
    def approve_adr(self, adr_id: str, consequences: Dict) -> None:
        """Aprova um ADR"""
        meta_path = self.base_path / f"ADR-{adr_id}.json"
        
        if not meta_path.exists():
            raise ValueError(f"ADR {adr_id} não encontrado")
        
        data = json.loads(meta_path.read_text())
        data['status'] = ADRStatus.ACCEPTED.value
        data['consequences'] = consequences
        
        # Atualiza arquivos
        meta_path.write_text(json.dumps(data, indent=2))
        
        # Atualiza markdown
        adr = ArchitectureDecision(**data)
        md_path = self.base_path / f"ADR-{adr_id}.md"
        md_path.write_text(adr.to_markdown())
    
    def _get_next_id(self) -> str:
        """Obtém próximo ID sequencial"""
        existing = list(self.base_path.glob("ADR-*.json"))
        if not existing:
            return "001"
        
        ids = [int(f.stem.split('-')[1]) for f in existing]
        return f"{max(ids) + 1:03d}"

import { useEffect, useRef, useState, useCallback } from 'react';

interface UseLazyLoadOptions {
  root?: Element | Document | null;
  rootMargin?: string;
  threshold?: number | number[];
  enabled?: boolean;
}

export function useLazyLoad<T extends HTMLElement = HTMLElement>(
  options: UseLazyLoadOptions = {}
) {
  const {
    root = null,
    rootMargin = '0px',
    threshold = 0.1,
    enabled = true,
  } = options;

  const ref = useRef<T>(null);
  const [isVisible, setIsVisible] = useState(false);
  const [hasLoaded, setHasLoaded] = useState(false);

  const handleIntersection = useCallback(
    (entries: IntersectionObserverEntry[]) => {
      const [entry] = entries;
      if (entry.isIntersecting) {
        setIsVisible(true);
        setHasLoaded(true);
      }
    },
    []
  );

  useEffect(() => {
    if (!enabled || hasLoaded || !ref.current) return;

    const observer = new IntersectionObserver(handleIntersection, {
      root,
      rootMargin,
      threshold,
    });

    observer.observe(ref.current);

    return () => {
      observer.disconnect();
    };
  }, [enabled, hasLoaded, root, rootMargin, threshold, handleIntersection]);

  return { ref, isVisible, hasLoaded };
}

// Hook para lazy loading de imagens com placeholder
export function useLazyImage(
  src: string,
  options: UseLazyLoadOptions = {}
) {
  const [imageSrc, setImageSrc] = useState<string>('');
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  const { ref, isVisible } = useLazyLoad<HTMLImageElement>(options);

  useEffect(() => {
    if (!isVisible || !src) return;

    setIsLoading(true);
    setError(null);

    const img = new Image();
    img.src = src;

    img.onload = () => {
      setImageSrc(src);
      setIsLoading(false);
    };

    img.onerror = (e) => {
      setError(new Error('Failed to load image'));
      setIsLoading(false);
    };

    return () => {
      img.onload = null;
      img.onerror = null;
    };
  }, [src, isVisible]);

  return {
    ref,
    src: imageSrc,
    isLoading,
    error,
    isVisible,
  };
}

                    raise
            return wrapper
        return decorator
    
    @staticmethod
    async def retry_with_backoff(func, max_retries=3, base_delay=1, 
                                 retry_on=(TimeoutError, ConnectionError)):
        """Retry com backoff exponencial + jitter"""
        last_exception = None
        
        for attempt in range(max_retries + 1):
            try:
                return await func()
            except retry_on as e:
                last_exception = e
                if attempt == max_retries:
                    break
                
                # Backoff exponencial com jitter
                delay = base_delay * (2 ** attempt)
                jitter = delay * 0.1  # 10% jitter
                await asyncio.sleep(delay + jitter)
        
        raise last_exception

            logger = StructuredLogger()

    async def get(self, key: str) -> Optional[Any]:
        """Obter valor do cache com fallback gracioso"""
        if not self.is_connected or not self.redis_client:
            REDIS_OPS.labels(operation="get", cache_hit="connection_error").inc()
            return None
        
        try:
            value = await self.redis_client.get(key)
            cache_hit = "hit" if value else "miss"
            REDIS_OPS.labels(operation="get", cache_hit=cache_hit).inc()
            
            if value:
                try:
                    return json.loads(value)
                except json.JSONDecodeError:
                    return value
            return None
        except RedisError as e:
            REDIS_OPS.labels(operation="get", cache_hit="error").inc()
            logger = StructuredLogger()
            logger.log("WARNING", f"Redis get error: {str(e)}")
            return None
    
    async def set(self, key: str, value: Any, ttl: int = 3600) -> bool:
        """Definir valor no cache com fallback gracioso"""
        if not self.is_connected or not self.redis_client:
            REDIS_OPS.labels(operation="set", cache_hit="connection_error").inc()
            return False
        
        try:
            serialized = json.dumps(value) if not isinstance(value, str) else value
            await self.redis_client.setex(key, ttl, serialized)
            REDIS_OPS.labels(operation="set", cache_hit="success").inc()
            return True
        except RedisError as e:
            REDIS_OPS.labels(operation="set", cache_hit="error").inc()
            logger = StructuredLogger()
            logger.log("WARNING", f"Redis set error: {str(e)}")
            return False

        """Executar query com métricas e tratamento de erro"""
        start_time = time.time()
        
        try:
            async with self.pool.acquire() as connection:
                # SEMPRE usar parâmetros preparados - ZERO concatenação
                if query_type.lower() == "select":
                    result = await connection.fetch(query, *params)
                else:
                    result = await connection.execute(query, *params)
                
                duration = time.time() - start_time
                DB_QUERIES.labels(query_type=query_type, table=table).observe(duration)
                
                return result
        except PostgresError as e:
            DB_QUERIES.labels(query_type=query_type + "_error", table=table).observe(
                time.time() - start_time
            )
            logger = StructuredLogger()
            logger.log("ERROR", f"Database query failed: {str(e)}", extra={"query": query[:100]})
            raise
    
    async def get_user_by_id(self, user_id: str):
        """Exemplo de query parametrizada"""
        query = """

        return await self.execute_query(query, user_id, query_type="select", table="users")

class ArchitectureDecisionRecord:
    """ADR Manager for V11 Migration"""
    
    def __init__(self, repo_path: str = "./adrs"):
        self.repo_path = Path(repo_path)
        self.repo_path.mkdir(exist_ok=True)
        
    def create_adr(self, title: str, context: str, decision: str, 
                   options: List[Dict], consequences: Dict) -> str:
        """Create a new ADR following the squad template"""
        
        adr_id = len(list(self.repo_path.glob("ADR-*.md"))) + 1
        filename = self.repo_path / f"ADR-{adr_id:03d}-{title.lower().replace(' ', '-')}.md"
        
        adr_content = f"""# ADR-{adr_id:03d}: {title}

**Date**: {datetime.now().strftime('%Y-%m-%d')}
**Status**: Proposed
**Authors**: Tech Lead + Architecture Team

## CONTEXT
{context}

## DECISION
{decision}

## OPTIONS CONSIDERED
"""
        
        for i, option in enumerate(options, 1):
            adr_content += f"""
### Option {i}: {option['name']}
**Description**: {option['description']}
**Pros**: {', '.join(option['pros'])}
**Cons**: {', '.join(option['cons'])}
"""
        
        chosen_idx = next((i for i, opt in enumerate(options, 1) 
                          if opt.get('chosen', False)), 1)
        
        adr_content += f"""
## CHOSEN OPTION: Option {chosen_idx}
**Justification**: {options[chosen_idx-1]['justification']}

## CONSEQUENCES
**Positive**: {', '.join(consequences['positive'])}
**Negative**: {', '.join(consequences['negative'])}
**Risks**: {', '.join(consequences['risks'])}

## REVIEW DATE
{datetime.now().replace(year=datetime.now().year + 1).strftime('%Y-%m-%d')}
"""
        
        filename.write_text(adr_content)
        return str(filename)

    def _section_to_markdown(self, content: Any, depth: int) -> str:
        """Convert section to markdown recursively"""
        
        if isinstance(content, dict):
            result = ""
            for key, value in content.items():
                indent = "  " * depth
                result += f"{indent}- **{key}**: "
                if isinstance(value, (dict, list)):
                    result += "\n" + self._section_to_markdown(value, depth + 1)
                else:
                    result += f"{value}\n"
            return result
        elif isinstance(content, list):
            result = ""
            for item in content:
                indent = "  " * depth
                result += f"{indent}- {item}\n"
            return result
        else:
            return f"{content}\n"

class JSONFormatter(logging.Formatter):
    def format(self, record: logging.LogRecord) -> str:
        log_data: Dict[str, Any] = {
            "timestamp": self.formatTime(record),
            "level": record.levelname,
            "correlation_id": correlation_id.get(),
            "service": record.name,
            "message": record.getMessage(),
            "location": f"{record.filename}:{record.lineno}",
        }
        
        # Adiciona campos extras se existirem
        if hasattr(record, 'extra_fields'):
            log_data.update(record.extra_fields)
        
        # Mascara dados sensíveis
        self._mask_sensitive_data(log_data)
        
        return json.dumps(log_data)
    
    def _mask_sensitive_data(self, data: Dict[str, Any]) -> None:
        sensitive_fields = ['email', 'password', 'token', 'cpf', 'credit_card', 'api_key']
        for field in sensitive_fields:
            if field in data:
                value = str(data[field])
                if len(value) > 4:
                    data[field] = value[:2] + '*' * (len(value) - 4) + value[-2:]
                else:
                    data[field] = '***'

def setup_logging():
    logger = logging.getLogger()
    logger.setLevel(logging.INFO)
    
    handler = logging.StreamHandler()
    handler.setFormatter(JSONFormatter())
    logger.addHandler(handler)
    
    # Remove handlers duplicados
    logger.propagate = False
    
    return logger

# shared/circuit_breaker.py
import time
from typing import Callable, Any, Optional
from enum import Enum
import asyncio

class CircuitState(Enum):
    CLOSED = "CLOSED"
    OPEN = "OPEN"
    HALF_OPEN = "HALF_OPEN"

class CircuitBreaker:
    def __init__(
        self,
        name: str,
        failure_threshold: int = 5,
        recovery_timeout: int = 30,
        expected_exceptions: tuple = (Exception,)
    ):
        self.name = name
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.expected_exceptions = expected_exceptions
        
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.last_failure_time: Optional[float] = None
        self.logger = logging.getLogger(f"circuit_breaker.{name}")
    
    async def call(self, func: Callable, *args, **kwargs) -> Any:
        if self.state == CircuitState.OPEN:
            # Verifica se pode tentar recuperar
            if self._can_attempt_recovery():
                self.state = CircuitState.HALF_OPEN
                self.logger.info(
                    f"Circuit {self.name} moving to HALF_OPEN",
                    extra={"circuit_name": self.name, "state": "HALF_OPEN"}
                )
            else:
                self.logger.warning(
                    f"Circuit {self.name} is OPEN, failing fast",
                    extra={"circuit_name": self.name, "state": "OPEN"}
                )
                raise CircuitOpenError(f"Circuit {self.name} is OPEN")
        
        try:
            result = await func(*args, **kwargs) if asyncio.iscoroutinefunction(func) else func(*args, **kwargs)
            
            # Sucesso - reseta contador se estava em HALF_OPEN
            if self.state == CircuitState.HALF_OPEN:
                self._reset()
                self.logger.info(
                    f"Circuit {self.name} recovered, moving to CLOSED",
                    extra={"circuit_name": self.name, "state": "CLOSED"}
                )
            elif self.state == CircuitState.CLOSED:
                self.failure_count = 0
            
            return result
            
        except self.expected_exceptions as e:
            self._record_failure()
            self.logger.error(
                f"Circuit {self.name} recorded failure: {str(e)}",
                extra={
                    "circuit_name": self.name,
                    "error": str(e),
                    "failure_count": self.failure_count,
                    "state": self.state.value
                }
            )
            raise
    
    def _record_failure(self):
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        if self.failure_count >= self.failure_threshold:
            self.state = CircuitState.OPEN
            self.logger.error(
                f"Circuit {self.name} opened due to {self.failure_count} failures",
                extra={
                    "circuit_name": self.name,
                    "failure_count": self.failure_count,
                    "state": "OPEN"
                }
            )
    
    def _can_attempt_recovery(self) -> bool:
        if not self.last_failure_time:
            return False
        
        elapsed = time.time() - self.last_failure_time
        return elapsed >= self.recovery_timeout
    
    def _reset(self):
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.last_failure_time = None

class CircuitOpenError(Exception):
    pass

                f"Failed to connect to Redis: {str(e)}",
                extra={"host": self.host, "port": self.port, "error": str(e)}
            )
            # Fallback: cache em memória para desenvolvimento
            self._client = None
    
    async def _create_client(self) -> redis.Redis:
        """Cria cliente Redis com timeout configurado"""
        return redis.Redis(
            host=self.host,
            port=self.port,
            password=self.password,
            socket_connect_timeout=5,  # 5s timeout de conexão
            socket_timeout=3,          # 3s timeout de operação
            retry_on_timeout=True,
            max_connections=10,
            decode_responses=False
        )
    
    async def get(self, key: str) -> Optional[Any]:
        """Obtém valor do cache com fallback"""
        if not self._client:
            return None
        
        try:
            data = await self.circuit_breaker.call(
                self._client.get, key
            )
            
            if data:
                try:
                    return json.loads(data)
                except json.JSONDecodeError:
                    return pickle.loads(data)
            return None
            
        except Exception as e:
            self.logger.warning(
                f"Cache get failed for key {key}: {str(e)}",
                extra={"key": key, "error": str(e)}
            )
            return None
    
    async def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Define valor no cache com fallback silencioso"""
        if not self._client:
            return False
        
        try:
            ttl = ttl or self.default_ttl
            
            # Tenta serializar como JSON primeiro
            try:
                serialized = json.dumps(value)
            except (TypeError, ValueError):
                serialized = pickle.dumps(value)
            
            await self.circuit_breaker.call(
                self._client.setex, key, ttl, serialized
            )
            return True
            
        except Exception as e:
            self.logger.warning(
                f"Cache set failed for key {key}: {str(e)}",
                extra={"key": key, "error": str(e)}
            )
            return False
    
    async def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        if not self._client:
            return False
        
        try:
            await self.circuit_breaker.call(self._client.delete, key)
            return True
        except Exception as e:
            self.logger.warning(
                f"Cache delete failed for key {key}: {str(e)}",
                extra={"key": key, "error": str(e)}
            )
            return False
    
    async def clear_pattern(self, pattern: str) -> int:
        """Limpa cache por pattern (ex: 'user:*')"""
        if not self._client:
            return 0
        
        try:
            keys = await self.circuit_breaker.call(self._client.keys, pattern)
            if keys:
                await self.circuit_breaker.call(self._client.delete, *keys)
                return len(keys)
            return 0
        except Exception as e:
            self.logger.warning(
                f"Cache clear pattern failed for {pattern}: {str(e)}",
                extra={"pattern": pattern, "error": str(e)}
            )
            return 0
    
    async def health_check(self) -> bool:
        """Verifica saúde do Redis"""
        if not self._client:
            return False
        
        try:
            await self.circuit_breaker.call(self._client.ping)
            return True
        except Exception:
            return False

# DIYAPP Evolution - V11 Core - Threat Modeling e Segurança

## 1. Threat Modeling da Nova Arquitetura (STRIDE)

### Documento de Threat Modeling - Arquitetura V11

**Data:** 2024-03-15
**Responsável:** Especialista em Segurança
**Arquitetura:** DIYAPP V11 - Sistema Autônomo com IA
**Status:** Análise Completa

---

### 1.1 Ativos Críticos

| Ativo | Descrição | Classificação |
|-------|-----------|---------------|
| Dados de Usuário | PII, credenciais, histórico de ações | Crítico |
| Tokens de Autenticação | JWT, refresh tokens | Crítico |
| Modelos de IA | LLMs, embeddings, prompts do sistema | Alto |
| Configurações do Sistema | Secrets, chaves de API, configurações | Crítico |
| Banco de Dados | SQLite/PostgreSQL com dados sensíveis | Crítico |
| API Gateway | Ponto de entrada para todas as requisições | Alto |

### 1.2 Análise STRIDE por Componente

#### **Componente: API Gateway**
- **Spoofing**: Autenticação JWT com assinatura HMAC-SHA256
- **Tampering**: Validação de assinatura em todos os tokens
- **Repudiation**: Logs de auditoria com ID de usuário e timestamp
- **Info Disclosure**: Rate limiting para prevenir enumeração
- **Denial of Service**: Circuit breaker + rate limiting por IP
- **Elevation of Privilege**: Validação de escopo em cada endpoint

#### **Componente: Serviço de Autenticação**
- **Spoofing**: MFA obrigatório para admin, bcrypt para senhas
- **Tampering**: Tokens com expiração curta (15min access, 7d refresh)
- **Repudiation**: Logs de login/logout imutáveis
- **Info Disclosure**: Senhas nunca em logs, apenas hash
- **Denial of Service**: Bloqueio após 5 tentativas falhas
- **Elevation of Privilege**: RBAC com princípio do menor privilégio

#### **Componente: Serviço de IA/LLM**
- **Spoofing**: Validação de contexto de usuário em cada request
- **Tampering**: Sanitização de input/output do modelo
- **Repudiation**: Logs de prompts e respostas (sem dados sensíveis)
- **Info Disclosure**: Não incluir PII no contexto do LLM
- **Denial of Service**: Rate limiting por usuário no modelo
- **Elevation of Privilege**: Validação de permissão antes de processar

#### **Componente: Banco de Dados**
- **Spoofing**: Credenciais separadas por serviço
- **Tampering**: Transações ACID, backups criptografados
- **Repudiation**: Audit trail para operações DML
- **Info Disclosure**: Criptografia em repouso para PII
- **Denial of Service**: Connection pooling, query timeout
- **Elevation of Privilege**: Schema permissions granulares

### 1.3 Riscos Identificados e Controles

| Risco | Impacto | Probabilidade | Controle |
|-------|---------|---------------|----------|
| Prompt Injection no LLM | Alto | Médio | Sanitização de input, separação system/user prompt |
| Token JWT Roubado | Crítico | Baixo | Expiração curta, refresh token rotation |
| SQL Injection | Crítico | Baixo | Prepared statements, ORM com escaping |
| XSS via Output LLM | Alto | Médio | Sanitização HTML, CSP headers |
| Enumeração de Usuários | Médio | Alto | Rate limiting, mensagens genéricas |
| DDoS no API Gateway | Alto | Médio | Cloudflare/WAF, auto-scaling |

### 1.4 Riscos Aceitos (Documentados)

1. **Risco**: Modelo de IA pode gerar conteúdo não confiável
   **Justificativa**: Natureza probabilística dos LLMs
   **Mitigação**: Sanitização de output, não usar em operações críticas
   **Responsável**: Especialista LLM

2. **Risco**: Cookies de sessão podem ser roubados via XSS
   **Justificativa**: Necessidade de sessões persistentes
   **Mitigação**: HttpOnly, Secure flags, SameSite=Strict
   **Responsável**: Dev Frontend

#### **src/security/scanners/dependency-checker.js**
```javascript
const { exec } = require('child_process');
const fs = require('fs');
const path = require('path');
const axios = require('axios');

class DependencySecurityScanner {
  constructor() {
    this.vulnerabilityDB = [];
    this.criticalCVEs = [];
    this.scanResults = {
      timestamp: new Date().toISOString(),
      dependencies: [],
      vulnerabilities: [],
      summary: {}
    };
  }

  async scan() {
    console.log('🔍 Iniciando scanner de dependências...');
    
    try {
      // 1. Ler package.json
      const packageJson = JSON.parse(fs.readFileSync('package.json', 'utf8'));
      
      // 2. Verificar dependências com Snyk (se configurado)
      await this.checkWithSnyk(packageJson);
      
      // 3. Verificar NVD (National Vulnerability Database)
      await this.checkNVD(packageJson);
      
      // 4. Verificar licenças
      await this.checkLicenses(packageJson);
      
      // 5. Gerar relatório
      await this.generateReport();
      
      // 6. Alertar se houver CVEs críticos
      await this.alertCriticalIssues();
      
      return this.scanResults;
      
    } catch (error) {
      console.error('❌ Erro no scanner:', error);
      throw error;
    }
  }

  async checkWithSnyk(packageJson) {
    console.log('📦 Verificando com Snyk...');
    
    // Método 1: Usar CLI do Snyk (se instalado)
    try {
      const { stdout } = await this.execPromise('npx snyk test --json');
      const snykResults = JSON.parse(stdout);
      
      snykResults.vulnerabilities?.forEach(vuln => {
        this.addVulnerability({
          package: vuln.packageName,
          version: vuln.version,
          severity: vuln.severity,
          cvssScore: vuln.cvssScore,
          title: vuln.title,
          description: vuln.description,
          identifiers: vuln.identifiers,
          fixedIn: vuln.fixedIn,
          source: 'snyk'
        });
      });
      
    } catch (error) {
      console.log('⚠️ Snyk não disponível, usando verificação básica...');
      await this.basicDependencyCheck(packageJson);
    }
  }

  async checkNVD(packageJson) {
    console.log('🛡️ Consultando NVD...');
    
    const dependencies = {
      ...packageJson.dependencies,
      ...packageJson.devDependencies
    };
    
    for (const [pkg, version] of Object.entries(dependencies)) {
      try {
        // API pública do NVD
        const response = await axios.get(
          `https://services.nvd.nist.gov/rest/json/cves/2.0`,
          {
            params: {
              keywordSearch: pkg,
              resultsPerPage: 10
            },
            timeout: 5000
          }
        );
        
        if (response.data.vulnerabilities) {
          response.data.vulnerabilities.forEach(cve => {
            const metrics = cve.cve.metrics.cvssMetricV31?.[0] || 
                           cve.cve.metrics.cvssMetricV30?.[0] ||
                           cve.cve.metrics.cvssMetricV2?.[0];
            
            if (metrics) {
              this.addVulnerability({
                package: pkg,
                version: version,
                severity: this.cvssToSeverity(metrics.cvssData.baseScore),
                cvssScore: metrics.cvssData.baseScore,
                title: cve.cve.descriptions[0]?.value || 'CVE encontrado',
                cveId: cve.cve.id,
                description: cve.cve.descriptions[0]?.value,
                source: 'nvd'
              });
            }
          });
        }
        
        // Rate limiting
        await new Promise(resolve => setTimeout(resolve, 100));
        
      } catch (error) {
        console.log(`⚠️ Não foi possível verificar ${pkg} no NVD:`, error.message);
      }
    }
  }

  async checkLicenses(packageJson) {
    console.log('📄 Verificando licenças...');
    
    const dependencies = {
      ...packageJson.dependencies,
      ...packageJson.devDependencies
    };
    
    const prohibitedLicenses = [
      'GPL-3.0',
      'AGPL-3.0',
      'SSPL-1.0'
    ];
    
    for (const [pkg, version] of Object.entries(dependencies)) {
      try {
        // Tentar obter license do npm registry
        const response = await axios.get(
          `https://registry.npmjs.org/${pkg}/${version}`,
          { timeout: 3000 }
        );
        
        const license = response.data.license;
        
        if (license && prohibitedLicenses.includes(license)) {
          this.scanResults.dependencies.push({
            package: pkg,
            version: version,
            license: license,
            status: 'BLOCKED',
            issue: `Licença ${license} não permitida`
          });
        } else {
          this.scanResults.dependencies.push({
            package: pkg,
            version: version,
            license: license || 'Unknown',
            status: 'OK'
          });
        }
        
      } catch (error) {
        this.scanResults.dependencies.push({
          package: pkg,
          version: version,
          license: 'Unknown',

## Breaking Changes

### 1. Autenticação
**V10:**
```javascript

  apiKey: 'your-key',
  endpoint: 'https://api.diyapp.com/v10'

**V11:**

const client = new DIYAppClient({
  clientId: 'your-client-id',
  clientSecret: 'your-client-secret',
  authEndpoint: 'https://auth.diyapp.com/oauth/token',
  apiEndpoint: 'https://api.diyapp.com/v11'
});

// Primeiro obtenha o token
const token = await client.authenticate();
```

### 2. Endpoints Modificados

#### Upload de Arquivos
**V10 (DEPRECATED):**
```
POST /api/v10/files/upload
Content-Type: multipart/form-data
```

**V11:**
```
POST /api/v11/storage/upload
Content-Type: multipart/form-data
X-Upload-Type: direct|chunked
```

Exemplo de migração:
```javascript
// Antigo
const formData = new FormData();
formData.append('file', file);
const response = await fetch('/api/v10/files/upload', {
  method: 'POST',
  body: formData
});

// Novo
const uploadSession = await client.storage.createUploadSession({
  fileName: file.name,
  fileSize: file.size,
  chunkSize: 1024 * 1024 // 1MB chunks
});

// Upload em chunks para arquivos grandes
await uploadSession.uploadChunks(file);
```

### 3. Webhooks
**V10:** Configuração via painel administrativo apenas
**V11:** API completa para gerenciamento de webhooks

```javascript
// Criar webhook
const webhook = await client.webhooks.create({
  url: 'https://your-server.com/webhooks',
  events: ['content.created', 'user.updated'],
  secret: 'your-signing-secret'
});

// Verificar assinatura
function verifyWebhookSignature(payload, signature, secret) {
  const hmac = crypto.createHmac('sha256', secret);
  const computed = hmac.update(payload).digest('hex');
  return crypto.timingSafeEqual(
    Buffer.from(computed),
    Buffer.from(signature)
  );
}

### 2. Circuit Breaker Pattern
Implementar circuit breakers para todas as chamadas entre serviços:

```javascript
// services/CircuitBreaker.js
class CircuitBreaker {
  constructor(failureThreshold = 5, resetTimeout = 60000) {
    this.failureThreshold = failureThreshold;
    this.resetTimeout = resetTimeout;
    this.failureCount = 0;
    this.lastFailureTime = null;
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
  }

  async execute(serviceCall) {
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.resetTimeout) {
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('Service unavailable (Circuit Breaker OPEN)');
      }
    }

    try {
      const result = await serviceCall();
      if (this.state === 'HALF_OPEN') {
        this.reset();
      }
      return result;
    } catch (error) {
      this.recordFailure();
      throw error;
    }
  }

  recordFailure() {
    this.failureCount++;
    this.lastFailureTime = Date.now();
    
    if (this.failureCount >= this.failureThreshold) {
      this.state = 'OPEN';
      setTimeout(() => {
        this.state = 'HALF_OPEN';
      }, this.resetTimeout);
    }
  }
}

// scripts/auto-rollback.js
class AutoRollback {
  constructor(deploymentId, maxErrorRate = 0.05) {
    this.deploymentId = deploymentId;
    this.maxErrorRate = maxErrorRate;
    this.metrics = {
      totalRequests: 0,
      errors: 0,
      startTime: Date.now()
    };
  }

  monitorRequest(success) {
    this.metrics.totalRequests++;
    if (!success) this.metrics.errors++;
    
    const errorRate = this.metrics.errors / this.metrics.totalRequests;
    
    if (this.metrics.totalRequests > 100 && errorRate > this.maxErrorRate) {
      this.triggerRollback();
    }
  }

  async triggerRollback() {
    console.error(`🚨 Auto-rollback triggered for ${this.deploymentId}`);
    console.error(`Error rate: ${(errorRate * 100).toFixed(2)}%`);
    
    // Reverter para versão anterior
    await exec(`kubectl rollout undo deployment/${this.deploymentId}`);
    
    // Notificar equipe
    await this.sendAlert({
      type: 'AUTO_ROLLBACK',
      deploymentId: this.deploymentId,
      errorRate,
      timestamp: new Date().toISOString()
    });
  }
}
