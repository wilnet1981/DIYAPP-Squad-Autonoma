logger = get_logger(__name__)
router = APIRouter(prefix="/api/v1/auth", tags=["authentication"])

# Models
class LoginRequest(BaseModel):
    email: EmailStr
    password: str
    device_id: Optional[str] = None

class TokenResponse(BaseModel):
    access_token: str
    refresh_token: str
    token_type: str = "bearer"
    expires_in: int

class UserResponse(BaseModel):
    id: str
    email: str
    name: str
    is_active: bool
    created_at: datetime

# Dependencies
def get_auth_service() -> AuthService:
    """Dependency injection for auth service"""
    return AuthService(UserRepository())

# Routes
@router.post("/login", response_model=TokenResponse)
async def login(
    request: LoginRequest,
    auth_service: AuthService = Depends(get_auth_service)
) -> TokenResponse:
    """
    Authenticate user and return JWT tokens
    """
    try:
        logger.info("Login attempt", extra={"email": request.email})
        
        # Rate limiting check
        await auth_service.check_rate_limit(request.email)
        
        # Authenticate
        user = await auth_service.authenticate(
            email=request.email,
            password=request.password
        )
        
        # Generate tokens
        tokens = await auth_service.generate_tokens(
            user_id=user.id,
            device_id=request.device_id
        )
        
        # Audit log
        await auth_service.log_auth_event(
            user_id=user.id,
            event_type="login_success",
            device_id=request.device_id
        )
        
        logger.info("Login successful", extra={"user_id": user.id})
        return tokens
        
    except AuthenticationError as e:
        logger.warning("Authentication failed", extra={"email": request.email})
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid credentials"
        )
    except ValidationError as e:
        logger.error("Validation error", extra={"error": str(e)})
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=str(e)
        )

@router.post("/refresh", response_model=TokenResponse)
async def refresh_token(
    refresh_token: str,
    auth_service: AuthService = Depends(get_auth_service)
) -> TokenResponse:
    """
    Refresh access token using refresh token
    """
    try:
        tokens = await auth_service.refresh_access_token(refresh_token)
        return tokens
    except AuthenticationError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid refresh token"
        )

@router.get("/me", response_model=UserResponse)
async def get_current_user(
    current_user: Dict = Depends(auth_service.get_current_user)
) -> UserResponse:
    """
    Get current authenticated user profile
    """
    return UserResponse(**current_user)

        """Authenticate user with email and password"""
        user = await self.user_repo.find_by_email(email)
        
        if not user or not user.is_active:
            raise AuthenticationError("Invalid credentials")
        
        # Verify password

            raise AuthenticationError("Invalid credentials")
        
        return user.to_dict()

export interface DashboardData {
  metrics: {
    uptime: number;
    activeUsers: number;
    responseTime: number;
    errorRate: number;
    cpuUsage: number;
    memoryUsage: number;
  };
  activities: Array<{
    id: string;
    type: 'info' | 'warning' | 'error' | 'success';
    message: string;
    timestamp: Date;
    source: string;
  }>;
  chartData: Array<{
    timestamp: number;
    value: number;
    label: string;
  }>;
}

interface DashboardContainerProps {
  autoRefresh?: boolean;
  refreshInterval?: number;
  onError?: (error: Error) => void;
}

export const DashboardContainer: React.FC<DashboardContainerProps> = ({
  autoRefresh = true,
  refreshInterval = 30000, // 30 segundos
  onError,
}) => {
  const [dashboardData, setDashboardData] = useState<DashboardData | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [lastUpdated, setLastUpdated] = useState<Date>(new Date());
  const [error, setError] = useState<string | null>(null);

  // Monitor de performance
  const { measureRender, reportMetric } = usePerformanceMonitor('DashboardContainer');

  // WebSocket para atualizações em tempo real
  const { connect, disconnect, isConnected } = useWebSocket(
    process.env.REACT_APP_WS_URL || 'wss://api.diyapp.com/ws/dashboard',
    {
      onMessage: handleWebSocketMessage,
      onError: handleWebSocketError,
      reconnectAttempts: 5,
      reconnectDelay: 1000,
    }
  );

  // Fetch inicial dos dados
  useEffect(() => {
    measureRender('initial-load');
    fetchDashboardData();
    
    if (autoRefresh) {
      const interval = setInterval(fetchDashboardData, refreshInterval);
      return () => clearInterval(interval);
    }
  }, [autoRefresh, refreshInterval]);

  // Conectar WebSocket quando componente montar
  useEffect(() => {
    connect();
    return () => disconnect();
  }, []);

  async function fetchDashboardData() {
    try {
      setIsLoading(true);
      const response = await fetch('/api/dashboard/metrics', {
        headers: {
          'Authorization': `Bearer ${localStorage.getItem('auth_token')}`,
          'Content-Type': 'application/json',
        },
        signal: AbortSignal.timeout(10000), // Timeout de 10 segundos
      });

      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }

      const data = await response.json();
      setDashboardData(data);
      setLastUpdated(new Date());
      setError(null);
      
      reportMetric('dashboard_fetch_success', { duration: performance.now() });
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Unknown error');
      setError(error.message);
      onError?.(error);
      reportMetric('dashboard_fetch_error', { error: error.message });
    } finally {
      setIsLoading(false);
    }
  }

  function handleWebSocketMessage(event: MessageEvent) {
    try {
      const data = JSON.parse(event.data);
      
      // Atualizar dados em tempo real
      if (data.type === 'metrics_update') {
        setDashboardData(prev => prev ? {
          ...prev,
          metrics: { ...prev.metrics, ...data.payload },
        } : null);
      } else if (data.type === 'activity') {
        setDashboardData(prev => prev ? {
          ...prev,
          activities: [data.payload, ...prev.activities.slice(0, 49)], // Mantém últimos 50
        } : null);
      }
      
      setLastUpdated(new Date());
    } catch (err) {
      console.error('WebSocket message error:', err);
    }
  }

  function handleWebSocketError(error: Event) {
    console.error('WebSocket error:', error);
    reportMetric('websocket_error', { error: error.type });
  }

  const handleRefresh = useCallback(() => {
    fetchDashboardData();
    reportMetric('manual_refresh');
  }, []);

  if (isLoading && !dashboardData) {
    return <DashboardSkeleton />;
  }

  if (error) {
    return (
      <div style={styles.errorContainer}>
        <div style={styles.errorIcon}>⚠️</div>
        <h3 style={styles.errorTitle}>Erro ao carregar dashboard</h3>
        <p style={styles.errorMessage}>{error}</p>
        <button
          onClick={handleRefresh}
          style={styles.retryButton}
          aria-label="Tentar novamente"
        >
          Tentar novamente
        </button>
      </div>
    );
  }

  return (
    <ErrorBoundary onError={onError}>
      <div style={styles.container}>
        {/* Header do Dashboard */}
        <header style={styles.header}>
          <div>
            <h1 style={styles.title}>DIYAPP Monitor</h1>
            <p style={styles.subtitle}>
              Status do sistema em tempo real
              {isConnected && (
                <span style={styles.connectionStatus}>
                  <span style={styles.connectionDot} />
                  Conectado
                </span>
              )}
            </p>
          </div>
          <div style={styles.headerActions}>
            <button
              onClick={handleRefresh}
              style={styles.refreshButton}
              aria-label="Atualizar dados"
              disabled={isLoading}
            >
              {isLoading ? '🔄' : '↻'}
              <span style={styles.buttonText}>Atualizar</span>
            </button>
            <div style={styles.lastUpdated}>
              Última atualização: {lastUpdated.toLocaleTimeString()}
            </div>
          </div>
        </header>

        {/* Conteúdo principal com lazy loading */}
        <main style={styles.mainContent}>
          <Suspense fallback={<div style={styles.loadingSection}>Carregando métricas...</div>}>

              <>
                <section style={styles.metricsSection}>
                  <MetricsGrid metrics={dashboardData.metrics} />
                </section>

                <div style={styles.chartsGrid}>
                  <section style={styles.chartSection}>
                    <RealTimeChart data={dashboardData.chartData} />

                    <SystemHealth metrics={dashboardData.metrics} />
                  </section>
                </div>

                <section style={styles.activitySection}>

            <a href="/settings" style={styles.footerLink}>Configurações</a>
            <a href="/export" style={styles.footerLink}>Exportar dados</a>
          </div>

import time
import logging
from enum import Enum
from typing import Any, Callable, Optional, Dict
from dataclasses import dataclass, field
from threading import Lock
import json

logger = logging.getLogger(__name__)

class CircuitState(Enum):
    CLOSED = "CLOSED"
    OPEN = "OPEN"
    HALF_OPEN = "HALF_OPEN"

@dataclass
class CircuitBreakerConfig:
    failure_threshold: int = 5
    reset_timeout: int = 60
    half_open_max_calls: int = 3
    excluded_exceptions: tuple = ()
    name: str = "default"

@dataclass
class CircuitMetrics:
    failures: int = 0
    successes: int = 0
    last_failure_time: Optional[float] = None
    state_changes: list = field(default_factory=list)

class CircuitBreaker:
    """Circuit breaker pattern implementation with thread safety"""
    
    def __init__(self, config: CircuitBreakerConfig):
        self.config = config
        self.state = CircuitState.CLOSED
        self.metrics = CircuitMetrics()
        self._lock = Lock()
        self._half_open_calls = 0
        
    def execute(self, func: Callable, *args, **kwargs) -> Any:
        with self._lock:
            if self.state == CircuitState.OPEN:
                if self._should_try_half_open():
                    self.state = CircuitState.HALF_OPEN
                    self._half_open_calls = 0
                    logger.info(f"Circuit {self.config.name} transitioning to HALF_OPEN")
                else:
                    raise CircuitOpenException(
                        f"Circuit {self.config.name} is OPEN. "
                        f"Next retry in {self._time_until_retry():.1f}s"
                    )
        
        try:
            result = func(*args, **kwargs)
            self._record_success()
            return result
        except Exception as e:
            if isinstance(e, self.config.excluded_exceptions):
                raise e
            self._record_failure()
            raise
    
    def _should_try_half_open(self) -> bool:
        if not self.metrics.last_failure_time:
            return True
        elapsed = time.time() - self.metrics.last_failure_time
        return elapsed > self.config.reset_timeout
    
    def _time_until_retry(self) -> float:
        if not self.metrics.last_failure_time:
            return 0
        elapsed = time.time() - self.metrics.last_failure_time
        return max(0, self.config.reset_timeout - elapsed)
    
    def _record_success(self):
        with self._lock:
            if self.state == CircuitState.HALF_OPEN:
                self._half_open_calls += 1
                if self._half_open_calls >= self.config.half_open_max_calls:
                    self.state = CircuitState.CLOSED
                    self.metrics.failures = 0
                    self.metrics.state_changes.append({
                        "timestamp": time.time(),
                        "from": "HALF_OPEN",
                        "to": "CLOSED"
                    })
                    logger.info(f"Circuit {self.config.name} reset to CLOSED")
            
            self.metrics.successes += 1
    
    def _record_failure(self):
        with self._lock:
            self.metrics.failures += 1
            self.metrics.last_failure_time = time.time()
            
            if self.state == CircuitState.HALF_OPEN:
                self.state = CircuitState.OPEN
                self.metrics.state_changes.append({
                    "timestamp": time.time(),
                    "from": "HALF_OPEN",
                    "to": "OPEN"
                })
                logger.warning(f"Circuit {self.config.name} failed in HALF_OPEN, returning to OPEN")
            
            elif (self.state == CircuitState.CLOSED and 
                  self.metrics.failures >= self.config.failure_threshold):
                self.state = CircuitState.OPEN
                self.metrics.state_changes.append({
                    "timestamp": time.time(),
                    "from": "CLOSED",
                    "to": "OPEN"
                })
                logger.error(f"Circuit {self.config.name} tripped to OPEN")
    
    def get_health(self) -> Dict:
        with self._lock:
            return {
                "name": self.config.name,
                "state": self.state.value,
                "metrics": {
                    "failures": self.metrics.failures,
                    "successes": self.metrics.successes,
                    "failure_rate": (
                        self.metrics.failures / (self.metrics.failures + self.metrics.successes)
                        if (self.metrics.failures + self.metrics.successes) > 0 else 0
                    )
                },
                "last_failure_time": self.metrics.last_failure_time,
                "state_changes": len(self.metrics.state_changes)
            }

import random
import asyncio
from typing import Callable, Any, Optional
from functools import wraps

class RetryPolicy:
    """Configurable retry policy with exponential backoff and jitter"""
    
    def __init__(
        self,
        max_attempts: int = 3,
        base_delay: float = 0.1,
        max_delay: float = 10.0,
        jitter: bool = True,
        retry_on_exceptions: tuple = (Exception,)
    ):
        self.max_attempts = max_attempts
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.jitter = jitter
        self.retry_on_exceptions = retry_on_exceptions
    
    def __call__(self, func: Callable) -> Callable:
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            
            for attempt in range(1, self.max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except self.retry_on_exceptions as e:
                    last_exception = e
                    
                    # Don't retry on 4xx errors
                    if hasattr(e, 'status_code') and 400 <= getattr(e, 'status_code') < 500:
                        raise
                    
                    if attempt == self.max_attempts:
                        break
                    
                    delay = self._calculate_delay(attempt)
                    logger.warning(
                        f"Retry attempt {attempt}/{self.max_attempts} for {func.__name__}. "
                        f"Waiting {delay:.2f}s. Error: {str(e)}"
                    )
                    time.sleep(delay)
            
            raise last_exception or Exception("Max retry attempts exceeded")
        
        return wrapper
    
    def async_retry(self, func: Callable) -> Callable:
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_exception = None
            
            for attempt in range(1, self.max_attempts + 1):
                try:
                    return await func(*args, **kwargs)
                except self.retry_on_exceptions as e:
                    last_exception = e
                    
                    if hasattr(e, 'status_code') and 400 <= getattr(e, 'status_code') < 500:
                        raise
                    
                    if attempt == self.max_attempts:
                        break
                    
                    delay = self._calculate_delay(attempt)
                    logger.warning(
                        f"Async retry attempt {attempt}/{self.max_attempts} for {func.__name__}. "
                        f"Waiting {delay:.2f}s. Error: {str(e)}"
                    )
                    await asyncio.sleep(delay)
            
            raise last_exception or Exception("Max retry attempts exceeded")
        
        return wrapper
    
    def _calculate_delay(self, attempt: int) -> float:
        delay = min(self.max_delay, self.base_delay * (2 ** (attempt - 1)))
        if self.jitter:
            delay = random.uniform(0.5 * delay, 1.5 * delay)
        return delay

import queue
import threading
from concurrent.futures import ThreadPoolExecutor
from typing import Callable, Any, Dict
import uuid

class AsyncTaskQueue:
    """Asynchronous task queue for non-blocking operations"""
    
    def __init__(self, max_workers: int = 4, max_queue_size: int = 1000):
        self.task_queue = queue.Queue(maxsize=max_queue_size)
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self.running = True
        self.worker_threads = []
        self.pending_tasks: Dict[str, Dict] = {}
        
        # Start worker threads
        for i in range(max_workers):
            thread = threading.Thread(target=self._worker, daemon=True, name=f"AsyncQueueWorker-{i}")
            thread.start()
            self.worker_threads.append(thread)
    
    def submit(self, task_func: Callable, *args, **kwargs) -> str:
        """Submit a task and return task ID"""
        if not self.running:
            raise RuntimeError("AsyncTaskQueue is shutdown")
        
        task_id = str(uuid.uuid4())
        task_data = {
            'id': task_id,
            'func': task_func,
            'args': args,
            'kwargs': kwargs,
            'submitted_at': time.time(),
            'status': 'queued'
        }
        
        self.pending_tasks[task_id] = task_data
        self.task_queue.put(task_data)
        
        logger.info(f"Task {task_id} submitted to async queue")
        return task_id
    
    def _worker(self):
        while self.running:
            try:
                task_data = self.task_queue.get(timeout=1)
                task_id = task_data['id']
                
                try:
                    self.pending_tasks[task_id]['status'] = 'processing'
                    self.pending_tasks[task_id]['started_at'] = time.time()
                    
                    result = task_data['func'](*task_data['args'], **task_data['kwargs'])
                    
                    self.pending_tasks[task_id]['status'] = 'completed'
                    self.pending_tasks[task_id]['completed_at'] = time.time()
                    self.pending_tasks[task_id]['result'] = result
                    self.pending_tasks[task_id]['error'] = None
                    
                    logger.info(f"Task {task_id} completed successfully")
                    
                except Exception as e:
                    self.pending_tasks[task_id]['status'] = 'failed'
                    self.pending_tasks[task_id]['completed_at'] = time.time()
                    self.pending_tasks[task_id]['error'] = str(e)
                    self.pending_tasks[task_id]['result'] = None
                    
                    logger.error(f"Task {task_id} failed: {str(e)}")
                
                self.task_queue.task_done()
                
            except queue.Empty:
                continue
            except Exception as e:
                logger.error(f"Worker thread error: {str(e)}")
    
    def get_task_status(self, task_id: str) -> Optional[Dict]:
        """Get status of a specific task"""
        return self.pending_tasks.get(task_id)
    
    def shutdown(self, wait: bool = True):
        """Shutdown the queue gracefully"""
        self.running = False
        if wait:
            self.task_queue.join()
        self.executor.shutdown(wait=wait)
    
    def get_queue_metrics(self) -> Dict:
        """Get queue metrics for monitoring"""
        return {
            'queue_size': self.task_queue.qsize(),
            'pending_tasks': len(self.pending_tasks),
            'running': self.running,
            'workers_alive': sum(1 for t in self.worker_threads if t.is_alive()),
            'tasks_by_status': {
                status: sum(1 for t in self.pending_tasks.values() if t['status'] == status)
                for status in ['queued', 'processing', 'completed', 'failed']
            }
        }

import psutil

from datetime import datetime

class HealthCheckRegistry:
    """Centralized health check system"""
    
    def __init__(self):
        self.checks: List[Dict] = []
        self.system_metrics = {}
    
    def register_check(self, name: str, check_func: Callable, interval: int = 30):
        """Register a health check"""
        self.checks.append({
            'name': name,
            'func': check_func,
            'interval': interval,
            'last_run': None,
            'last_status': None,
            'last_error': None
        })
    
    def run_checks(self) -> Dict:
        """Run all health checks"""
        results = {
            'timestamp': datetime.utcnow().isoformat(),
            'status': 'healthy',
            'checks': [],
            'system_metrics': self._get_system_metrics()
        }
        
        for check in self.checks:
            try:
                start_time = time.time()
                check_result = check['func']()
                duration = time.time() - start_time
                
                check['last_run'] = datetime.utcnow()
                check['last_status'] = 'healthy'
                check['last_error'] = None
                
                results['checks'].append({
                    'name': check['name'],
                    'status': 'healthy',
                    'duration_ms': round(duration * 1000, 2),
                    'details': check_result
                })
                
                if not self._is_healthy(check_result):
                    results['status'] = 'degraded'

                check['last_run'] = datetime.utcnow()
                check['last_status'] = 'unhealthy'
                check['last_error'] = str(e)
                
                results['checks'].append({
                    'name': check['name'],
                    'status': 'unhealthy',
                    'error': str(e),
                    'duration_ms': None

                results['status'] = 'unhealthy'
