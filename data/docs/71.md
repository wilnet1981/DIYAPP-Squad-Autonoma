```jsx
import React from 'react';
import PropTypes from 'prop-types';
import './Button.css';

const Button = React.forwardRef(({
  children,
  variant = 'primary',
  size = 'medium',
  isLoading = false,
  disabled = false,
  fullWidth = false,
  leftIcon,
  rightIcon,
  onClick,
  type = 'button',
  className = '',
  ...props
}, ref) => {
  const baseClasses = 'btn';
  const variantClasses = {
    primary: 'btn-primary',
    secondary: 'btn-secondary',
    outline: 'btn-outline',
    ghost: 'btn-ghost',
    danger: 'btn-danger',
  };
  
  const sizeClasses = {
    small: 'btn-sm',
    medium: 'btn-md',
    large: 'btn-lg',
  };
  
  const classes = [
    baseClasses,
    variantClasses[variant],
    sizeClasses[size],
    isLoading ? 'btn-loading' : '',
    disabled ? 'btn-disabled' : '',
    fullWidth ? 'btn-full-width' : '',
    className,
  ].filter(Boolean).join(' ');
  
  return (
    <button
      ref={ref}
      type={type}
      className={classes}
      disabled={disabled || isLoading}
      onClick={onClick}
      aria-busy={isLoading}
      {...props}
    >
      {isLoading && (
        <span className="btn-spinner" aria-hidden="true">
          <svg className="spinner" viewBox="0 0 50 50">
            <circle cx="25" cy="25" r="20" fill="none" strokeWidth="5" />
          </svg>
        </span>
      )}
      
      {!isLoading && leftIcon && (
```

```javascript
const { Pool } = require('pg');
const sqlite3 = require('sqlite3').verbose();
const path = require('path');
const logger = require('../utils/logger');

class Database {
  constructor() {
    this.dbType = process.env.DB_TYPE || 'sqlite';
    this.connection = null;
  }

  async connect() {
    try {
      if (this.dbType === 'postgres') {
        this.connection = new Pool({
          host: process.env.DB_HOST,
          port: process.env.DB_PORT,
          database: process.env.DB_NAME,
          user: process.env.DB_USER,
          password: process.env.DB_PASSWORD,
          max: 20,
          idleTimeoutMillis: 30000,
          connectionTimeoutMillis: 2000,
        });

        // Test connection
        await this.connection.query('SELECT NOW()');
        logger.info('PostgreSQL connected successfully');
      } else {
        // SQLite
        const dbPath = process.env.SQLITE_PATH || './data/diyapp.db';
        this.connection = new sqlite3.Database(dbPath, (err) => {
          if (err) {
            logger.error('SQLite connection error:', err);
            throw err;
          }
          logger.info('SQLite connected successfully');
          
          // Enable foreign keys
          this.connection.run('PRAGMA foreign_keys = ON');
        });
      }
    } catch (error) {
      logger.error('Database connection failed:', error);
      throw error;
    }
  }

  async query(sql, params = []) {
    const correlationId = global.correlationId || 'system';
    const startTime = Date.now();
    
    try {
      logger.debug({
        correlation_id: correlationId,
        query: sql,
        params: this.maskSensitiveData(params),
        action: 'database_query_start'
      });

      let result;
      
      if (this.dbType === 'postgres') {
        result = await this.connection.query(sql, params);
      } else {
        // SQLite
        result = await new Promise((resolve, reject) => {
          this.connection.all(sql, params, (err, rows) => {
            if (err) {
              reject(err);
            } else {
              resolve({ rows });
            }
          });
        });
      }

      const duration = Date.now() - startTime;
      
      logger.debug({
        correlation_id: correlationId,
        query: sql,
        duration_ms: duration,
        row_count: result.rows?.length || 0,
        action: 'database_query_complete'
      });

      return result.rows || [];
    } catch (error) {
      const duration = Date.now() - startTime;
      
      logger.error({
        correlation_id: correlationId,
        query: sql,
        params: this.maskSensitiveData(params),
        duration_ms: duration,
        error: error.message,
        action: 'database_query_error'
      });

      // Never expose SQL errors to client
      throw new Error('Database operation failed');
    }
  }

  async execute(sql, params = []) {
    const correlationId = global.correlationId || 'system';
    const startTime = Date.now();
    
    try {
      logger.debug({
        correlation_id: correlationId,
        query: sql,
        params: this.maskSensitiveData(params),
        action: 'database_execute_start'
      });

      if (this.dbType === 'postgres') {
        await this.connection.query(sql, params);
      } else {
        // SQLite
        await new Promise((resolve, reject) => {
          this.connection.run(sql, params, function(err) {
            if (err) {
              reject(err);
            } else {
              resolve({ lastID: this.lastID, changes: this.changes });
            }
          });
        });
      }

      const duration = Date.now() - startTime;
      
      logger.debug({
        correlation_id: correlationId,
        query: sql,
        duration_ms: duration,
        action: 'database_execute_complete'
      });
    } catch (error) {
      const duration = Date.now() - startTime;
      
      logger.error({
        correlation_id: correlationId,
        query: sql,
        params: this.maskSensitiveData(params),
        duration_ms: duration,
        error: error.message,
        action: 'database_execute_error'
      });

      throw new Error('Database operation failed');
    }
  }

  maskSensitiveData(params) {
    // Mask sensitive data in logs
    return params.map(param => {
      if (typeof param === 'string') {
        // Mask emails
        if (param.includes('@')) {
          return '[EMAIL_MASKED]';
        }
        // Mask tokens
        if (param.length > 32 && /^[A-Za-z0-9-_]+\.[A-Za-z0-9-_]+\.[A-Za-z0-9-_]+$/.test(param)) {
          return '[JWT_MASKED]';
        }
      }
      return param;
    });
  }

  async disconnect() {
    if (this.connection) {
      if (this.dbType === 'postgres') {
        await this.connection.end();
      } else {
        this.connection.close();
      }
      logger.info('Database disconnected');
    }
  }
}

module.exports = new Database();
```

```javascript
const jwt = require('jsonwebtoken');
const logger = require('../utils/logger');

const authenticate = (req, res, next) => {
  const startTime = Date.now();
  const correlationId = req.headers['x-correlation-id'] || req.id || 'unknown';
  
  try {
    const authHeader = req.headers.authorization;
    
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      logger.warn({
        correlation_id: correlationId,
        endpoint: req.originalUrl,
        action: 'authentication_failed',
        reason: 'no_bearer_token'
      });
      
      return res.status(401).json({
        error: 'Unauthorized',
        message: 'Authentication token required',
        correlation_id: correlationId
      });
    }

    const token = authHeader.split(' ')[1];
    
    // Verify token
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    
    // Attach user to request
    req.user = {
      id: decoded.userId,
      email: decoded.email,
      role: decoded.role
    };
    
    // Set correlation ID in global context for logging
    global.correlationId = correlationId;
    
    const duration = Date.now() - startTime;
    
    logger.debug({
      correlation_id: correlationId,
      user_id: req.user.id,
      endpoint: req.originalUrl,
      duration_ms: duration,
      action: 'authentication_success'
    });
    
    next();
  } catch (error) {
    const duration = Date.now() - startTime;
    
    logger.error({
      correlation_id: correlationId,
      endpoint: req.originalUrl,
      duration_ms: duration,
      error: error.message,
      action: 'authentication_error'
    });
    
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({
        error: 'TokenExpired',
        message: 'Authentication token has expired',
        correlation_id: correlationId
      });
    }
    
    if (error.name === 'JsonWebTokenError') {
      return res.status(401).json({
        error: 'InvalidToken',
        message: 'Invalid authentication token',
        correlation_id: correlationId
      });
    }
    
    return res.status(401).json({
      error: 'Unauthorized',
      message: 'Authentication failed',
      correlation_id: correlationId
    });
  }
};
```

```sql
RETURNS VARCHAR AS $$
DECLARE
    slug VARCHAR := LOWER(REGEXP_REPLACE(base_slug, '[^a-zA-Z0-9]+', '-', 'g'));
    counter INTEGER := 1;
    test_slug VARCHAR;
    exists BOOLEAN;
BEGIN
    test_slug := slug;
    
    LOOP
        EXECUTE format('SELECT EXISTS(SELECT 1 FROM %I WHERE %I = $1 AND id != $2)', table_name, id_column)
        INTO exists
        USING test_slug, id_value;
        
        IF NOT exists THEN
            RETURN test_slug;
        END IF;
        
        test_slug := slug || '-' || counter;
        counter := counter + 1;
    END LOOP;
END;
```

```javascript
class Database {
  constructor() {
    this.type = process.env.DB_TYPE || 'sqlite';
    this.client = null;
    this.isConnected = false;
  }

  async connect() {
    try {
      if (this.type === 'postgres') {
        this.client = new Pool({
          host: process.env.DB_HOST,
          port: process.env.DB_PORT,

          (err) => {
            if (err) {

              throw err;

  async query(text, params) {
    if (!this.isConnected) {
      throw new Error('Database not connected');
    }

    const start = Date.now();
    try {
      let result;
      
      if (this.type === 'postgres') {
        result = await this.client.query(text, params);
      } else {
        // SQLite parameter binding
        const stmt = await this.client.prepare(text);
        result = await stmt.all(...(params || []));
        await stmt.finalize();
      }

      logger.debug('Query executed', { 
        text, 
        params: this.maskSensitiveData(params),
        duration,
        rows: result?.rows?.length || result?.length || 0 
      });

        text, 
        params: this.maskSensitiveData(params),
        error: error.message

      throw error;
    }

        const lowerParam = param.toLowerCase();
        if (sensitiveKeys.some(key => lowerParam.includes(key))) {
          return '***MASKED***';
        }
      }
      return param;
    });
  }

  async disconnect() {
    if (this.client) {
      if (this.type === 'postgres') {
        await this.client.end();
      } else {
        await new Promise((resolve, reject) => {
          this.client.close((err) => {
            if (err) reject(err);
            else resolve();
          });
        });
      }
      this.isConnected = false;
      logger.info('Database disconnected');
    }
  }
}
```

```javascript
              version: version,
              safeVersion: safeVersion,
              description: `Dependência ${dep} v${version} possui vulnerabilidades conhecidas`,
              remediation: `Atualizar para v${safeVersion} ou superior`
            });
          }
        }
      }
    } catch (error) {
      console.error('Erro ao analisar dependências:', error);
    }
  }

  async scanCodebase() {
    const scanPaths = ['src', 'routes', 'controllers', 'models', 'middleware'];
    
    for (const scanPath of scanPaths) {
      if (fs.existsSync(scanPath)) {
        await this.scanDirectory(scanPath);
      }
    }
  }

  async scanDirectory(dirPath) {
    const files = fs.readdirSync(dirPath);
    
    for (const file of files) {
      const fullPath = path.join(dirPath, file);
      const stat = fs.statSync(fullPath);
      
      if (stat.isDirectory()) {
        await this.scanDirectory(fullPath);
      } else if (file.endsWith('.js') || file.endsWith('.ts')) {
        await this.scanFile(fullPath);
      }
    }
  }

  async scanFile(filePath) {
    const content = fs.readFileSync(filePath, 'utf8');
    
    // Padrões de vulnerabilidade a serem detectados
    const vulnerabilityPatterns = [
      {
        pattern: /eval\s*\(/g,
        type: 'CODE_INJECTION',
        severity: 'CRITICAL',
        description: 'Uso de eval() pode permitir injeção de código'
      },
      {
        pattern: /child_process\.exec\s*\(/g,
        type: 'COMMAND_INJECTION',
        severity: 'CRITICAL',
        description: 'Uso de exec() sem sanitização pode permitir injeção de comandos'
      },
      {
        pattern: /sql\s*=\s*["'].*?\$\{.*?\}.*?["']/gs,
        type: 'SQL_INJECTION',
        severity: 'HIGH',
        description: 'Template strings em queries SQL podem permitir SQL Injection'
      },
      {
        pattern: /innerHTML\s*=\s*.*?\$\{.*?\}/gs,
        type: 'XSS',
        severity: 'HIGH',
        description: 'Atribuição direta a innerHTML com variáveis pode permitir XSS'
      },
      {
        pattern: /jwt\.sign.*?{.*?secret.*?:.*?["'].*?["']/gs,
        type: 'INSECURE_JWT',
        severity: 'HIGH',
        description: 'Secret do JWT hardcoded no código'
      },
      {
        pattern: /password.*?=.*?["'].*?["']/g,
        type: 'HARDCODED_CREDENTIALS',
        severity: 'CRITICAL',
        description: 'Credenciais hardcoded no código'
      }
    ];

    for (const pattern of vulnerabilityPatterns) {
      const matches = content.match(pattern.pattern);
      if (matches) {
        this.vulnerabilities.push({
          type: pattern.type,
          severity: pattern.severity,
          file: filePath,
          matches: matches.length,
          description: pattern.description,
          remediation: this.getRemediation(pattern.type)
        });
      }
    }
  }

  getRemediation(vulnType) {
    const remediations = {
      'CODE_INJECTION': 'Substituir eval() por JSON.parse() ou Function() com validação',
      'COMMAND_INJECTION': 'Usar execFile() com argumentos sanitizados ou child_process.spawn()',
      'SQL_INJECTION': 'Usar prepared statements ou ORM com parametrização',
      'XSS': 'Usar textContent ou sanitizar com DOMPurify antes de innerHTML',
      'INSECURE_JWT': 'Armazenar secret em variáveis de ambiente',
      'HARDCODED_CREDENTIALS': 'Mover para variáveis de ambiente ou secret manager'
    };
    return remediations[vulnType] || 'Revisar implementação de segurança';
  }

  async scanConfigurations() {
    // Verificar arquivos de configuração
    const configFiles = ['.env', 'config.js', 'config.json'];
    
    for (const configFile of configFiles) {
      if (fs.existsSync(configFile)) {
        await this.scanConfigFile(configFile);
      }
    }
  }

  async scanConfigFile(configFile) {
    const content = fs.readFileSync(configFile, 'utf8');
    
    // Verificar secrets expostos
    const secretPatterns = [
      /password\s*=\s*["'].*?["']/g,
      /secret\s*=\s*["'].*?["']/g,
      /api[_-]?key\s*=\s*["'].*?["']/g,
      /token\s*=\s*["'].*?["']/g
    ];

    for (const pattern of secretPatterns) {
      const matches = content.match(pattern);
      if (matches) {
        this.vulnerabilities.push({
          type: 'EXPOSED_SECRET',
          severity: 'CRITICAL',
          file: configFile,
          description: 'Secret exposto em arquivo de configuração',
          remediation: 'Mover para variáveis de ambiente ou arquivo .env (não commitado)'
        });
      }
    }
  }

  async scanAIComponents() {
    // Verificar componentes específicos de IA/LLM
    const aiFiles = this.findAIFiles();
    
    for (const file of aiFiles) {
      await this.scanAIFile(file);
    }
  }

  findAIFiles() {
    const aiFiles = [];
    const searchTerms = ['llm', 'openai', 'chatgpt', 'gpt', 'ai', 'model', 'prompt'];
    
    const scanDir = (dir) => {
      if (!fs.existsSync(dir)) return;
      
      const files = fs.readdirSync(dir);
      for (const file of files) {
        const fullPath = path.join(dir, file);
        const stat = fs.statSync(fullPath);
        
        if (stat.isDirectory()) {
          scanDir(fullPath);
        } else if (file.endsWith('.js') || file.endsWith('.ts')) {
          const content = fs.readFileSync(fullPath, 'utf8').toLowerCase();
          if (searchTerms.some(term => content.includes(term))) {
            aiFiles.push(fullPath);
          }
        }
      }
    };

    ['src', 'routes', 'controllers'].forEach(dir => scanDir(dir));
    return aiFiles;
  }

  async scanAIFile(filePath) {
    const content = fs.readFileSync(filePath, 'utf8');
    
    // Padrões de vulnerabilidade específicos de IA
    const aiVulnerabilities = [
      {
        pattern: /systemPrompt.*?=.*?`.*?\$\{.*?\}.*?`/gs,
        type: 'PROMPT_INJECTION',
        severity: 'HIGH',
        description: 'System prompt concatenado com input do usuário sem sanitização'
      },
      {
        pattern: /llm\.call.*?\{.*?prompt.*?:.*?\$\{.*?\}.*?\}/gs,
        type: 'LLM_INJECTION',
        severity: 'MEDIUM',
        description: 'Input do usuário passado diretamente para LLM sem sanitização'
      },
      {
        pattern: /JSON\.parse.*?llmResponse/gs,
        type: 'UNSAFE_LLM_OUTPUT',
        severity: 'MEDIUM',
        description: 'Output do LLM processado sem validação'
      }
    ];

    for (const vuln of aiVulnerabilities) {
      const matches = content.match(vuln.pattern);
      if (matches) {
        this.vulnerabilities.push({
          type: vuln.type,
          severity: vuln.severity,
          file: filePath,
          description: vuln.description,
          remediation: 'Implementar sanitização de input/output e validação de schema'
        });
      }
    }
  }

  compareVersions(v1, v2) {
    const parts1 = v1.replace(/[^0-9.]/g, '').split('.').map(Number);
    const parts2 = v2.replace(/[^0-9.]/g, '').split('.').map(Number);
    
    for (let i = 0; i < Math.max(parts1.length, parts2.length); i++) {
      const part1 = parts1[i] || 0;
      const part2 = parts2[i] || 0;
      if (part1 !== part2) {
        return part1 - part2;
      }
    }
    return 0;
  }

  generateReport() {
    const critical = this.vulnerabilities.filter(v => v.severity === 'CRITICAL');
    const high = this.vulnerabilities.filter(v => v.severity === 'HIGH');
    const medium = this.vulnerabilities.filter(v => v.severity === 'MEDIUM');
    const low = this.vulnerabilities.filter(v => v.severity === 'LOW');

    console.log('\n📊 RELATÓRIO DE SEGURANÇA SAST');
    console.log('=' .repeat(50));
    console.log(`🔴 CRÍTICAS: ${critical.length}`);
    console.log(`🟠 ALTAS: ${high.length}`);
    console.log(`🟡 MÉDIAS: ${medium.length}`);
    console.log(`🟢 BAIXAS: ${low.length}`);
    console.log('=' .repeat(50));

    if (critical.length > 0) {
      console.log('\n🚨 VULNERABILIDADES CRÍTICAS:');
      critical.forEach((vuln, i) => {
        console.log(`${i+1}. ${vuln.type} em ${vuln.file}`);
        console.log(`   Descrição: ${vuln.description}`);
        console.log(`   Correção: ${vuln.remediation}`);
      });
    }

    return {
      timestamp: new Date().toISOString(),
      summary: {
        total: this.vulnerabilities.length,
        critical: critical.length,
        high: high.length,
        medium: medium.length,
        low: low.length
      },
      vulnerabilities: this.vulnerabilities,
      compliance: this.checkCompliance()
    };
  }

  checkCompliance() {
    const checks = {
      authentication: this.vulnerabilities.filter(v => 
        ['INSECURE_JWT', 'HARDCODED_CREDENTIALS'].includes(v.type)
      ).length === 0,
      authorization: true, // Será verificado no código
      dataProtection: this.vulnerabilities.filter(v => 
        ['EXPOSED_SECRET'].includes(v.type)
      ).length === 0,
      inputValidation: this.vulnerabilities.filter(v => 
        ['SQL_INJECTION', 'XSS', 'CODE_INJECTION'].includes(v.type)
      ).length === 0
    };

    return {
      passed: Object.values(checks).every(v => v),
      details: checks
    };
  }
}

module.exports = SASTScanner;
```

- **Modo Hive**: Refatoração paralela com múltiplos agentes

```python
"""
Sistema de Retenção e Exclusão Automatizada de Dados
"""
import asyncio
import logging
from datetime import datetime, timedelta
from typing import Dict, List, Optional
from sqlalchemy import and_, or_
from sqlalchemy.ext.asyncio import AsyncSession

from src.models.user_data import User, UserSession, AccessLog
from src.models.audit_log import DataDeletionLog
from config.compliance_config import RETENTION_POLICY, DataCategory

logger = logging.getLogger(__name__)

class DataRetentionManager:
    """Gerencia retenção e exclusão automática de dados"""
    
    def __init__(self, db_session: AsyncSession):
        self.db = db_session
        self.deletion_logs = []
    
    async def apply_retention_policy(self) -> Dict[str, int]:
        """
        Aplica política de retenção para todas as categorias
        Retorna contagem de registros excluídos por categoria
        """
        results = {}
        
        # 1. Logs de Acesso (6 meses)
        access_logs_deleted = await self._delete_expired_access_logs()
        results["access_logs"] = access_logs_deleted
        
        # 2. Logs de Sistema (3 meses)
        system_logs_deleted = await self._delete_expired_system_logs()
        results["system_logs"] = system_logs_deleted
        
        # 3. Sessões Inativas (30 dias)
        sessions_deleted = await self._delete_inactive_sessions()
        results["sessions"] = sessions_deleted
        
        # 4. Usuários Inativos (2 anos)
        users_deleted = await self._delete_inactive_users()
        results["users"] = users_deleted
        
        # Log da execução
        await self._log_deletion_activity(results)
        
        return results
    
    async def _delete_expired_access_logs(self) -> int:
        """Exclui logs de acesso com mais de 6 meses"""
        cutoff_date = datetime.utcnow() - timedelta(
            days=RETENTION_POLICY[DataCategory.LOG_ACESSO]["retention_days"]
        )
        
        result = await self.db.execute(
            AccessLog.__table__.delete().where(
                AccessLog.created_at < cutoff_date
            )
        )
        deleted_count = result.rowcount
        
        if deleted_count > 0:
            logger.info(f"Excluídos {deleted_count} logs de acesso expirados")
        
        return deleted_count
    
    async def _delete_expired_system_logs(self) -> int:
        """Exclui logs de sistema com mais de 3 meses"""
        # Implementação específica para logs do sistema
        # Esta é uma implementação genérica - ajustar conforme estrutura real
        from src.models.audit_log import SystemLog
        
        cutoff_date = datetime.utcnow() - timedelta(
            days=RETENTION_POLICY[DataCategory.LOG_SISTEMA]["retention_days"]
        )
        
        result = await self.db.execute(
            SystemLog.__table__.delete().where(
                SystemLog.timestamp < cutoff_date
            )
        )
        deleted_count = result.rowcount
        
        if deleted_count > 0:
            logger.info(f"Excluídos {deleted_count} logs de sistema expirados")
        
        return deleted_count
    
    async def _delete_inactive_sessions(self) -> int:
        """Exclui sessões inativas por mais de 30 dias"""
        cutoff_date = datetime.utcnow() - timedelta(days=30)
        
        result = await self.db.execute(
            UserSession.__table__.delete().where(
                UserSession.last_activity < cutoff_date
            )
        )
        deleted_count = result.rowcount
        
        if deleted_count > 0:
            logger.info(f"Excluídas {deleted_count} sessões inativas")
        
        return deleted_count
    
    async def _delete_inactive_users(self) -> int:
        """
        Exclui usuários inativos por mais de 2 anos
        Aplica exclusão em cascata para dados relacionados
        """
        cutoff_date = datetime.utcnow() - timedelta(days=365 * 2)
        
        # Primeiro, identifica usuários inativos
        inactive_users = await self.db.execute(
            User.__table__.select().where(
                and_(
                    User.last_login < cutoff_date,
                    User.is_active == False
                )
            )
        )
        users = inactive_users.fetchall()
        
        deleted_count = 0
        for user in users:
            # Exclui dados relacionados antes do usuário
            await self._delete_user_related_data(user.id)
            
            # Exclui o usuário
            await self.db.execute(
                User.__table__.delete().where(User.id == user.id)
            )
            deleted_count += 1
        
        if deleted_count > 0:
            logger.info(f"Excluídos {deleted_count} usuários inativos")
        
        return deleted_count
    
    async def _delete_user_related_data(self, user_id: int):
        """Exclui dados relacionados ao usuário (em cascata)"""
        # Exclui logs de acesso
        await self.db.execute(
            AccessLog.__table__.delete().where(AccessLog.user_id == user_id)
        )
        
        # Exclui sessões
        await self.db.execute(
            UserSession.__table__.delete().where(UserSession.user_id == user_id)
        )
        
        # Nota: Projetos não são excluídos automaticamente
        # Requer decisão explícita do usuário
    
    async def _log_deletion_activity(self, results: Dict[str, int]):
        """Registra atividade de exclusão para auditoria"""
        for category, count in results.items():
            if count > 0:
                log = DataDeletionLog(
                    data_category=category,
                    records_deleted=count,
                    retention_policy_applied=True,
                    executed_at=datetime.utcnow(),
                    executor="retention_scheduler"
                )
                self.db.add(log)
        
        await self.db.commit()
    
    async def process_data_subject_request(self, user_id: int, request_type: str) -> bool:
        """
        Processa solicitação de titular de dados
        Tipos: 'access', 'rectification', 'erasure', 'portability'
        """
        if request_type == "erasure":
            return await self._execute_right_to_be_forgotten(user_id)
        elif request_type == "access":

        elif request_type == "portability":
            return await self._provide_data_portability(user_id)
        
        return False
    
    async def _execute_right_to_be_forgotten(self, user_id: int) -> bool:
        """
        Executa direito ao esquecimento (LGPD Art. 18, VI)
        Exclui todos os dados pessoais do usuário
        """
        try:
            # 1. Anonimiza dados do usuário
            await self.db.execute(
                User.__table__.update().where(User.id == user_id).values(
                    email=f"deleted_{user_id}@anonymized.diyapp",
                    nome="Usuário Excluído",
                    telefone=None,
                    is_active=False,
                    deleted_at=datetime.utcnow()
                )
            )
            
            # 2. Exclui logs e sessões
            await self._delete_user_related_data(user_id)
            
            # 3. Registra a exclusão
            log = DataDeletionLog(
                data_category="user_all",
                records_deleted=1,
                retention_policy_applied=False,
                subject_request=True,
                user_id=user_id,
                executed_at=datetime.utcnow(),
                executor="data_subject"
            )
            self.db.add(log)
            
            await self.db.commit()
            logger.info(f"Direito ao esquecimento executado para usuário {user_id}")
            return True
```

    """

    Tipos: access, rectification, erasure, restriction, portability, objection
    """

        user_id=current_user.id,
        request_type=request_type,
        description=description,
        status="received",
        received_at=datetime.utcnow(),
        deadline=datetime.utcnow() + timedelta(
            days=NOTIFICATION_SETTINGS["subject_request_response_days"]
        )
    )
    
    db.add(ds_request)
    await db.commit()
    await db.refresh(ds_request)
    
    # Inicia processamento assíncrono
    manager = DataSubjectRequestManager(db)
    
    if request_type == "erasure":
        # Direito ao esquecimento - requer confirmação adicional
        return {
            "request_id": ds_request.id,
            "message": "Solicitação de exclusão recebida. Será processada em até 15 dias.",
            "confirmation_required": True,
            "deadline": ds_request.deadline.isoformat()
        }
    else:
        # Processa outras solicitações
        success = await manager.process_request(ds_request.id)
        
        return {
            "request_id": ds_request.id,
            "message": "Solicitação recebida e em processamento",
            "status": "processing",
            "deadline": ds_request.deadline.isoformat()
        }
