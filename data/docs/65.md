# DIYAPP Evolution - V11 Core - Arquitetura de Microsserviços

## 1. Architecture Decision Record - ADR-001: Estratégia de Microsserviços V11

**ADR-001: Arquitetura de Microsserviços Baseada em Eventos para V11**
**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead V11

### CONTEXTO:
A V10 do DIYAPP evoluiu para uma arquitetura monolítica que está limitando:
1. Escalabilidade independente de módulos
2. Deploy contínuo sem downtime total
3. Isolamento de falhas entre funcionalidades
4. Adoção de diferentes stacks tecnológicas por domínio
5. Manutenção paralela por múltiplas squads

### DECISÃO:
Adotar arquitetura de microsserviços baseada em eventos com os seguintes princípios:
1. Cada domínio de negócio vira um microsserviço independente
2. Comunicação via eventos assíncronos (80%) e APIs síncronas (20%)
3. Cada serviço possui seu próprio banco de dados
4. API Gateway único para clientes externos
5. Service Discovery dinâmico com health checks

### OPÇÕES CONSIDERADAS:
- **Opção A: Monólito Modularizado** - Prós: Simplicidade inicial, transações ACID fáceis. Contras: Escalabilidade limitada, acoplamento alto.
- **Opção B: Microsserviços REST** - Prós: Desacoplamento, deploy independente. Contras: Latência em cadeias de chamadas, complexidade de transações.
- **Opção C: Microsserviços Baseados em Eventos** - Prós: Desacoplamento máximo, resiliência, escalabilidade. Contras: Complexidade eventual consistency, debugging distribuído.
- **Opção escolhida: C** - Justificativa: Melhor alinhamento com visão de squads autônomas, tolerância a falhas nativa e escalabilidade horizontal.

### CONSEQUÊNCIAS:
**Positivas:**
- Deploy independente por squad
- Escalabilidade granular por domínio
- Isolamento de falhas
- Stack tecnológica adequada por domínio
- Desenvolvimento paralelo acelerado

**Negativas:**
- Complexidade de transações distribuídas
- Eventual consistency em alguns cenários
- Overhead de infraestrutura
- Monitoramento distribuído necessário

**Riscos:**
1. Latência em cadeias síncronas - Mitigação: Circuit breakers e timeouts
2. Perda de eventos - Mitigação: Persistent event log com retry
3. Debugging distribuído - Mitigação: Distributed tracing (Jaeger)

import { Redis } from 'ioredis';
import { v4 as uuidv4 } from 'uuid';

export class IdempotencyManager {
  private redis: Redis;
  private readonly TTL = 24 * 60 * 60; // 24 hours

  constructor(redisClient: Redis) {
    this.redis = redisClient;
  }

  async generateKey(operation: string, params: Record<string, any>): Promise<string> {
    const sortedParams = Object.keys(params)
      .sort()
      .map(key => `${key}=${JSON.stringify(params[key])}`)
      .join('&');
    
    return `idempotency:${operation}:${sortedParams}`;
  }

  async checkAndSet(
    idempotencyKey: string,
    operation: string,
    context: { userId: string; correlationId: string }
  ): Promise<{ isDuplicate: boolean; existingResult?: any }> {
    const redisKey = `idempotency:${idempotencyKey}`;
    
    // Check if operation already completed
    const existing = await this.redis.get(redisKey);
    if (existing) {
      return { isDuplicate: true, existingResult: JSON.parse(existing) };
    }

    // Set lock to prevent concurrent execution
    const lockKey = `idempotency_lock:${idempotencyKey}`;
    const lockId = uuidv4();
    const lockAcquired = await this.redis.set(
      lockKey,
      lockId,
      'EX',
      30, // 30 second lock
      'NX'
    );

    if (!lockAcquired) {
      // Wait and retry
      await new Promise(resolve => setTimeout(resolve, 100));
      return this.checkAndSet(idempotencyKey, operation, context);
    }

    // Store operation metadata
    const metadata = {
      operation,
      userId: context.userId,
      correlationId: context.correlationId,
      startedAt: new Date().toISOString(),
      lockId
    };

    await this.redis.set(
      `idempotency_meta:${idempotencyKey}`,
      JSON.stringify(metadata),
      'EX',
      this.TTL
    );

    return { isDuplicate: false };
  }

  async complete(
    idempotencyKey: string,
    result: any,
    error?: Error
  ): Promise<void> {
    const lockKey = `idempotency_lock:${idempotencyKey}`;
    const metaKey = `idempotency_meta:${idempotencyKey}`;
    const resultKey = `idempotency:${idempotencyKey}`;

    // Get lock ID to ensure we own the lock
    const lockId = await this.redis.get(lockKey);
    const metadata = await this.redis.get(metaKey);
    
    if (!metadata) {
      throw new Error('Idempotency metadata not found');
    }

    const meta = JSON.parse(metadata);
    
    // Store result
    const resultData = {
      success: !error,
      data: result,
      error: error ? {
        message: error.message,
        code: (error as any).code || 'INTERNAL_ERROR'
      } : null,
      completedAt: new Date().toISOString(),
      correlationId: meta.correlationId
    };

    await this.redis.set(
      resultKey,
      JSON.stringify(resultData),
      'EX',
      this.TTL
    );

    // Cleanup
    await this.redis.del(lockKey);
    await this.redis.del(metaKey);
  }

  async cleanupExpired(): Promise<number> {
    // This would be run as a scheduled job
    // For now, just return 0
    return 0;
  }
}

export interface RetryConfig {
  maxAttempts: number;
  baseDelay: number;
  maxDelay: number;
  jitter: boolean;
  retryableErrors: string[];
}

export class RetryManager {
  static async executeWithRetry<T>(
    operation: () => Promise<T>,
    config: Partial<RetryConfig> = {},
    context: { operationName: string; correlationId: string }
  ): Promise<T> {
    const defaultConfig: RetryConfig = {
      maxAttempts: 3,
      baseDelay: 100,
      maxDelay: 10000,
      jitter: true,
      retryableErrors: ['ECONNRESET', 'ETIMEDOUT', 'ECONNABORTED', 'ESOCKETTIMEDOUT']
    };

    const finalConfig = { ...defaultConfig, ...config };
    let lastError: Error;
    
    for (let attempt = 1; attempt <= finalConfig.maxAttempts; attempt++) {
      try {
        return await operation();
      } catch (error) {
        lastError = error as Error;
        
        // Check if error is retryable
        const isRetryable = this.isRetryableError(error, finalConfig.retryableErrors);
        
        if (!isRetryable || attempt === finalConfig.maxAttempts) {
          throw this.enhanceError(error as Error, attempt, context);
        }

        // Calculate delay with exponential backoff and jitter
        const delay = this.calculateDelay(attempt, finalConfig);
        
        console.warn(`Retry attempt ${attempt}/${finalConfig.maxAttempts} for ${context.operationName}`, {
          correlationId: context.correlationId,
          delay,
          error: error.message
        });

        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }

    throw lastError!;
  }

  private static isRetryableError(error: any, retryableErrors: string[]): boolean {
    if (error.code && retryableErrors.includes(error.code)) {
      return true;
    }
    
    // Check for timeout errors
    if (error.message?.includes('timeout') || error.message?.includes('Timeout')) {
      return true;
    }
    
    // Check for network errors
    if (error.message?.includes('network') || error.message?.includes('Network')) {
      return true;
    }
    
    // 5xx errors are retryable
    if (error.response?.status >= 500 && error.response?.status < 600) {
      return true;
    }
    
    return false;
  }

  private static calculateDelay(attempt: number, config: RetryConfig): number {
    const exponentialDelay = config.baseDelay * Math.pow(2, attempt - 1);
    const cappedDelay = Math.min(exponentialDelay, config.maxDelay);
    
    if (!config.jitter) {
      return cappedDelay;
    }
    
    // Add jitter (±20%)
    const jitter = cappedDelay * 0.2;
    return cappedDelay - jitter + Math.random() * 2 * jitter;
  }

  private static enhanceError(error: Error, attempt: number, context: any): Error {
    const enhancedError = new Error(
      `Operation failed after ${attempt} attempts: ${error.message}`
    );
    
    (enhancedError as any).originalError = error;
    (enhancedError as any).attempts = attempt;
    (enhancedError as any).correlationId = context.correlationId;
    (enhancedError as any).operationName = context.operationName;
    
    return enhancedError;
  }
}

export class CircuitBreaker {
  private state: 'CLOSED' | 'OPEN' | 'HALF_OPEN' = 'CLOSED';
  private failureCount = 0;
  private lastFailureTime: number | null = null;
  private nextAttemptTime: number | null = null;
  
  constructor(
    private readonly config: {
      failureThreshold: number;
      resetTimeout: number;
      halfOpenSuccessThreshold: number;
    }
  ) {}

  async execute<T>(operation: () => Promise<T>): Promise<T> {
    this.checkState();
    
    if (this.state === 'OPEN') {
      throw new Error('Circuit breaker is OPEN');
    }
    
    try {
      const result = await operation();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private checkState(): void {
    if (this.state === 'OPEN' && this.nextAttemptTime) {
      if (Date.now() >= this.nextAttemptTime) {
        this.state = 'HALF_OPEN';
        this.failureCount = 0;
      }
    }
  }

  private onSuccess(): void {
    if (this.state === 'HALF_OPEN') {
      this.failureCount++;
      
      if (this.failureCount >= this.config.halfOpenSuccessThreshold) {
        this.state = 'CLOSED';
        this.failureCount = 0;
        this.lastFailureTime = null;
        this.nextAttemptTime = null;
      }
    } else {
      this.failureCount = 0;
    }
  }

  private onFailure(): void {
    this.failureCount++;
    this.lastFailureTime = Date.now();
    
    if (this.state === 'HALF_OPEN') {
      this.state = 'OPEN';
      this.nextAttemptTime = Date.now() + this.config.resetTimeout;
    } else if (this.failureCount >= this.config.failureThreshold) {
      this.state = 'OPEN';
      this.nextAttemptTime = Date.now() + this.config.resetTimeout;
    }
  }

  getState(): { state: string; failureCount: number; nextAttempt: number | null } {
    return {
      state: this.state,
      failureCount: this.failureCount,
      nextAttempt: this.nextAttemptTime
    };
  }
}

```tsx
// src/components/providers/error-boundary.tsx
'use client'

import { Component, ReactNode } from 'react'
import { AlertTriangle } from 'lucide-react'

interface Props {
  children: ReactNode
}

interface State {
  hasError: boolean
  error?: Error
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props)
    this.state = { hasError: false }
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    console.error('ErrorBoundary caught an error:', error, errorInfo)
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="flex min-h-screen items-center justify-center p-6">
          <div className="card max-w-md text-center">
            <AlertTriangle className="mx-auto h-12 w-12 text-error-500" />
            <h2 className="mt-4 text-xl font-semibold">Something went wrong</h2>
            <p className="mt-2 text-sm text-neutral-600">
              {this.state.error?.message || 'An unexpected error occurred'}
            </p>
            <button
              onClick={() => this.setState({ hasError: false })}
              className="btn-primary mt-6"
            >
              Try again
            </button>
          </div>
        </div>
      )
    }

    return this.props.children
  }
}
```

class EventBus:
    """Event Bus for Service Communication"""
    
    def __init__(self):
        self.topics = {}
        self.subscribers = {}
    
    def publish(self, topic: str, event: Dict):
        """Publish event to topic"""
        if topic not in self.topics:
            self.topics[topic] = []
        
        standardized_event = {
            "event_id": event.get("event_id") or os.urandom(16).hex(),
            "event_type": event["event_type"],
            "timestamp": datetime.now().isoformat(),
            "source": event["source"],
            "correlation_id": event.get("correlation_id"),
            "payload": event.get("payload", {}),
            "metadata": event.get("metadata", {})
        }
        
        self.topics[topic].append(standardized_event)
        
        # Notify subscribers
        for subscriber in self.subscribers.get(topic, []):
            try:
                subscriber(standardized_event)
            except Exception as e:
                print(f"Error notifying subscriber for topic {topic}: {e}")
    
    def subscribe(self, topic: str, callback):
        """Subscribe to topic events"""
        if topic not in self.subscribers:
            self.subscribers[topic] = []
        self.subscribers[topic].append(callback)
    
    def get_events(self, topic: str, limit: int = 100) -> List[Dict]:
        """Get recent events from topic"""
        return self.topics.get(topic, [])[-limit:]

        self.next_id += 1
        return debt_item
    
    def get_pending_debt(self) -> List[Dict]:
        """Get all pending technical debt"""
        return [item for item in self.debt_items if item["status"] == "pending"]

        """Schedule debt payment for specific sprint"""
        for item in self.debt_items:

                item["assigned_sprint"] = sprint_number
                item["status"] = "scheduled"
                break

        """Calculate capacity to allocate for debt payment (max 20%)"""

import { createHash } from 'crypto';

export class IdempotencyUtil {
  private static readonly IDEMPOTENCY_KEY_PREFIX = 'idempotency:';
  private static readonly DEFAULT_TTL = 24 * 60 * 60; // 24 horas

  static generateKey(request: {
    userId: string;
    action: string;
    payload: Record<string, any>;
  }): string {
    const hash = createHash('sha256')
      .update(`${request.userId}:${request.action}:${JSON.stringify(request.payload)}`)
      .digest('hex');
    return `${this.IDEMPOTENCY_KEY_PREFIX}${hash}`;
  }

  static async checkAndStore(
    cache: CacheService,
    key: string,
    ttl: number = this.DEFAULT_TTL
  ): Promise<{ exists: boolean; storedValue?: any }> {
    const existing = await cache.get(key);
    if (existing) {
      return { exists: true, storedValue: existing };
    }
    
    // Marca como em processamento
    await cache.set(key, { status: 'processing', timestamp: new Date() }, ttl);
    return { exists: false };
  }

  static async storeResult(
    cache: CacheService,
    key: string,
    result: any,
    ttl: number = this.DEFAULT_TTL
  ): Promise<void> {
    await cache.set(key, {
      status: 'completed',
      result,
      timestamp: new Date()
    }, ttl);
  }
}

// src/core/utils/retry.util.ts
export class RetryUtil {
  static async withExponentialBackoff<T>(
    operation: () => Promise<T>,
    maxRetries: number = 3,
    baseDelay: number = 100,
    shouldRetry?: (error: any) => boolean
  ): Promise<T> {
    let lastError: any;
    
    for (let attempt = 0; attempt <= maxRetries; attempt++) {
      try {
        return await operation();
      } catch (error) {
        lastError = error;
        
        // Não retenta erros 4xx (exceto 429)
        if (error.statusCode && error.statusCode >= 400 && error.statusCode < 500 && error.statusCode !== 429) {
          throw error;
        }
        
        // Verifica se deve retentar
        if (shouldRetry && !shouldRetry(error)) {
          throw error;
        }
        
        if (attempt < maxRetries) {
          const delay = baseDelay * Math.pow(2, attempt) + Math.random() * 100; // + jitter
          await new Promise(resolve => setTimeout(resolve, delay));
        }
      }
    }
    
    throw lastError;
  }
}

import { Pool, QueryResult } from 'pg';
import { User } from '../entities/user.entity';
import { Logger } from '../utils/logger.util';

export class UserRepository {
  constructor(private pool: Pool) {}

  async create(user: Omit<User, 'id' | 'createdAt' | 'updatedAt'>): Promise<User> {
    const query = `
      INSERT INTO users (email, name, hashed_password, api_key)
      VALUES ($1, $2, $3, $4)
      RETURNING *
    `;
    
    const values = [user.email, user.name, user.hashedPassword, user.apiKey];
    
    try {
      const result = await this.pool.query(query, values);
      return this.mapRowToUser(result.rows[0]);
    } catch (error) {
      Logger.error('UserRepository.create', error as Error, { email: user.email });
      throw error;
    }
  }

  async findById(id: string): Promise<User | null> {
    const query = 'SELECT * FROM users WHERE id = $1 AND deleted_at IS NULL';
    
    try {
      const result = await this.pool.query(query, [id]);
      return result.rows.length > 0 ? this.mapRowToUser(result.rows[0]) : null;
    } catch (error) {
      Logger.error('UserRepository.findById', error as Error, { id });
      throw error;
    }
  }

  async findByEmail(email: string): Promise<User | null> {
    const query = 'SELECT * FROM users WHERE email = $1 AND deleted_at IS NULL';
    
    try {
      const result = await this.pool.query(query, [email]);
      return result.rows.length > 0 ? this.mapRowToUser(result.rows[0]) : null;
    } catch (error) {
      Logger.error('UserRepository.findByEmail', error as Error, { email });
      throw error;
    }
  }

  async update(id: string, updates: Partial<User>): Promise<User | null> {
    const fields = [];
    const values = [];
    let paramIndex = 1;

    if (updates.name !== undefined) {
      fields.push(`name = $${paramIndex++}`);
      values.push(updates.name);
    }
    if (updates.hashedPassword !== undefined) {
      fields.push(`hashed_password = $${paramIndex++}`);
      values.push(updates.hashedPassword);
    }
    if (updates.apiKey !== undefined) {
      fields.push(`api_key = $${paramIndex++}`);
      values.push(updates.apiKey);
    }

    if (fields.length === 0) {
      return this.findById(id);
    }

    fields.push(`updated_at = $${paramIndex++}`);
    values.push(new Date());

    values.push(id);
    const query = `
      UPDATE users 
      SET ${fields.join(', ')}
      WHERE id = $${paramIndex} AND deleted_at IS NULL
      RETURNING *
    `;

    try {
      const result = await this.pool.query(query, values);
      return result.rows.length > 0 ? this.mapRowToUser(result.rows[0]) : null;
    } catch (error) {
      Logger.error('UserRepository.update', error as Error, { id });
      throw error;
    }
  }

  async softDelete(id: string): Promise<boolean> {
    const query = `
      UPDATE users 
      SET deleted_at = $1, updated_at = $1
      WHERE id = $2 AND deleted_at IS NULL
    `;
    
    try {
      const result = await this.pool.query(query, [new Date(), id]);
      return result.rowCount > 0;
    } catch (error) {
      Logger.error('UserRepository.softDelete', error as Error, { id });
      throw error;
    }
  }

  private mapRowToUser(row: any): User {
    return {
      id: row.id,
      email: row.email,
      name: row.name,
      hashedPassword: row.hashed_password,
      apiKey: row.api_key,
      createdAt: row.created_at,
      updatedAt: row.updated_at,
      deletedAt: row.deleted_at
    };
  }
}

## 3. Cache Semântico com Redis

**src/ai_ops/cache/semantic_cache.py**
```python
import json
import hashlib
from typing import Any, Dict, Optional, Tuple
import numpy as np
from sentence_transformers import SentenceTransformer
import redis
import logging
from datetime import datetime, timedelta

logger = logging.getLogger(__name__)

class SemanticCache:
    """Cache semântico para prompts similares usando embeddings"""
    
    def __init__(self, redis_client: redis.Redis, 
                 embedding_model_name: str = 'all-MiniLM-L6-v2',
                 similarity_threshold: float = 0.85,
                 ttl_hours: int = 24):
        """
        Inicializa o cache semântico
        
        Args:
            redis_client: Cliente Redis configurado
            embedding_model_name: Nome do modelo de embeddings
            similarity_threshold: Limiar de similaridade para cache hit
            ttl_hours: Tempo de vida do cache em horas
        """
        self.redis = redis_client
        self.similarity_threshold = similarity_threshold
        self.ttl = ttl_hours * 3600
        
        # Carregar modelo de embeddings (cache local)
        logger.info(f"Carregando modelo de embeddings: {embedding_model_name}")
        self.embedding_model = SentenceTransformer(embedding_model_name)
        
        # Estatísticas
        self.hits = 0
        self.misses = 0
        
    def _generate_key(self, prompt: str, model: str) -> str:
        """Gera chave única para o prompt"""
        prompt_hash = hashlib.md5(prompt.encode()).hexdigest()
        return f"cache:{model}:{prompt_hash}"
    
    def _generate_embedding_key(self, embedding: np.ndarray) -> str:
        """Gera chave para embedding"""
        embedding_bytes = embedding.tobytes()
        return f"embedding:{hashlib.md5(embedding_bytes).hexdigest()}"
    
    def get_embedding(self, text: str) -> np.ndarray:
        """Gera embedding para o texto"""
        return self.embedding_model.encode(text)
    
    def find_similar(self, prompt: str, model: str) -> Optional[Tuple[str, float]]:
        """
        Encontra prompt similar no cache
        
        Returns:
            Tuple (resposta_cache, similaridade) ou None
        """
        # Gerar embedding do prompt atual
        current_embedding = self.get_embedding(prompt)
        
        # Buscar embeddings similares no cache
        cache_key = f"similarity_index:{model}"
        cached_embeddings = self.redis.hgetall(cache_key)
        
        best_similarity = 0.0
        best_response_key = None
        
        for embedding_key_bytes, response_key_bytes in cached_embeddings.items():
            embedding_key = embedding_key_bytes.decode()
            response_key = response_key_bytes.decode()
            
            # Recuperar embedding armazenado
            cached_embedding_bytes = self.redis.get(embedding_key)
            if not cached_embedding_bytes:
                continue
                
            cached_embedding = np.frombuffer(cached_embedding_bytes, dtype=np.float32)
            
            # Calcular similaridade de cosseno
            similarity = np.dot(current_embedding, cached_embedding) / (
                np.linalg.norm(current_embedding) * np.linalg.norm(cached_embedding)
            )
            
            if similarity > best_similarity and similarity >= self.similarity_threshold:
                best_similarity = similarity
                best_response_key = response_key
        
        if best_response_key:
            # Recuperar resposta do cache
            cached_data = self.redis.get(best_response_key)
            if cached_data:
                self.hits += 1
                logger.debug(f"Cache hit: similaridade={best_similarity:.3f}")
                return json.loads(cached_data), best_similarity
        
        self.misses += 1
        return None
    
    def store(self, prompt: str, model: str, response: Dict[str, Any], 
              tokens_used: int, cost: float):
        """
        Armazena resposta no cache
        
        Args:
            prompt: Prompt original
            model: Modelo usado
            response: Resposta do modelo
            tokens_used: Tokens consumidos
            cost: Custo da chamada
        """
        cache_key = self._generate_key(prompt, model)
        
        # Dados para cache
        cache_data = {
            'response': response,
            'tokens_used': tokens_used,
            'cost': cost,
            'model': model,
            'cached_at': datetime.utcnow().isoformat(),
            'prompt_hash': hashlib.md5(prompt.encode()).hexdigest()
        }
        
        # Armazenar resposta
        self.redis.setex(
            cache_key,
            self.ttl,
            json.dumps(cache_data)
        )
        
        # Armazenar embedding para busca semântica
        embedding = self.get_embedding(prompt)
        embedding_key = self._generate_embedding_key(embedding)
        
        # Armazenar embedding
        self.redis.setex(
            embedding_key,
            self.ttl,
            embedding.tobytes()
        )
        
        # Atualizar índice de similaridade
        similarity_index_key = f"similarity_index:{model}"
        self.redis.hset(
            similarity_index_key,
            embedding_key,
            cache_key
        )
        self.redis.expire(similarity_index_key, self.ttl)
        
        logger.debug(f"Resposta armazenada no cache: {cache_key}")
    
    def get_stats(self) -> Dict[str, Any]:
        """Retorna estatísticas do cache"""
        total = self.hits + self.misses
        hit_rate = self.hits / total if total > 0 else 0
        
        return {
            'hits': self.hits,
            'misses': self.misses,
            'hit_rate': hit_rate,
            'total_entries': self.redis.dbsize(),
            'similarity_threshold': self.similarity_threshold
        }
    
    def invalidate_model(self, model: str):
        """Invalidar todas as entradas de cache para um modelo"""
        pattern = f"cache:{model}:*"
        keys = self.redis.keys(pattern)
        
        if keys:
            self.redis.delete(*keys)
            logger.info(f"Invalidado {len(keys)} entradas para modelo {model}")
        
        # Limpar índice de similaridade
        similarity_index_key = f"similarity_index:{model}"
        self.redis.delete(similarity_index_key)
    
    def cleanup_expired(self):
        """Limpar entradas expiradas (Redis faz automaticamente com TTL)"""
        logger.info("Cleanup de cache expirado realizado")
```

# DIYAPP Evolution - V11 Core - Threat Modeling & Security Implementation

## 1. Threat Modeling STRIDE Analysis

### Assets Identificados:
1. **Dados de Usuário**: Credenciais, tokens, dados pessoais
2. **Dados da Aplicação**: Conteúdo gerado pelo usuário, configurações
3. **Sistema de Autenticação**: JWT tokens, refresh tokens
4. **APIs**: Endpoints de mutação de dados
5. **Logs de Auditoria**: Registros de operações sensíveis

### Análise STRIDE:

#### **Spoofing (Falsificação)**
- **Ameaça**: Ataque rouba JWT token para se passar por usuário
- **Controles**: Tokens com expiração curta (15min), refresh tokens rotativos, invalidação no logout
- **Risco**: ALTO - Mitigado por controles implementados

#### **Tampering (Manipulação)**
- **Ameaça**: Manipulação de dados em trânsito ou requests
- **Controles**: Validação de schema em todas as entradas, assinatura JWT, sanitização de inputs
- **Risco**: MÉDIO - Mitigado por validação rigorosa

#### **Repudiation (Repúdio)**
- **Ameaça**: Usuário nega ter realizado operação
- **Controles**: Logs de auditoria imutáveis para todas as mutações
- **Risco**: BAIXO - Mitigado por sistema de logs completo

#### **Information Disclosure (Divulgação)**
- **Ameaça**: Vazamento de dados sensíveis em logs ou responses
- **Controles**: Mascaramento de PII em logs, criptografia em repouso
- **Risco**: MÉDIO - Mitigado por políticas de mascaramento

#### **Denial of Service (Negação)**
- **Ameaça**: Ataque de sobrecarga nos endpoints de autenticação
- **Controles**: Rate limiting por IP e usuário, circuit breakers
- **Risco**: ALTO - Mitigado por rate limiting agressivo

#### **Elevation of Privilege (Elevação)**
- **Ameaça**: Usuário acessa recursos além das permissões
- **Controles**: Verificação de autorização por recurso, princípio do menor privilégio
- **Risco**: MÉDIO - Mitigado por verificações granulares

## 1. Configuração do Banco de Dados (SQLite com suporte a PostgreSQL)

```javascript:src/config/database.js
const sqlite3 = require('sqlite3').verbose();
const { open } = require('sqlite');
const path = require('path');
const fs = require('fs');

class ComplianceDatabase {
    constructor() {
        this.db = null;
        this.init();
    }

    async init() {
        // Criar diretório de dados se não existir
        const dataDir = path.join(__dirname, '..', '..', 'data');
        if (!fs.existsSync(dataDir)) {
            fs.mkdirSync(dataDir, { recursive: true });
        }

        this.db = await open({
            filename: path.join(dataDir, 'compliance.db'),
            driver: sqlite3.Database
        });

        await this.createTables();
        await this.seedInitialData();
    }

    async createTables() {
        // Tabela de Atividades de Processamento (ROPA - LGPD Art. 37)
        await this.db.exec(`
            CREATE TABLE IF NOT EXISTS data_processing_activities (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                activity_name TEXT NOT NULL,
                purpose TEXT NOT NULL,
                legal_basis TEXT NOT NULL CHECK(legal_basis IN ('consent', 'contract', 'legal_obligation', 'vital_interest', 'public_interest', 'legitimate_interest')),
                data_categories TEXT NOT NULL,
                data_subjects TEXT NOT NULL,
                retention_period_days INTEGER NOT NULL,
                processors TEXT,
                international_transfers BOOLEAN DEFAULT 0,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                status TEXT DEFAULT 'active' CHECK(status IN ('active', 'inactive', 'under_review'))
            )
        `);

        // Tabela de Consentimentos (LGPD Art. 8)
        await this.db.exec(`
            CREATE TABLE IF NOT EXISTS consents (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id TEXT NOT NULL,
                consent_type TEXT NOT NULL,
                purpose TEXT NOT NULL,
                granted BOOLEAN NOT NULL,
                granted_at TIMESTAMP NOT NULL,
                expires_at TIMESTAMP,
                version TEXT NOT NULL,
                ip_address TEXT,
                user_agent TEXT,
                consent_text TEXT,
                withdrawal_at TIMESTAMP,
                metadata TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                UNIQUE(user_id, consent_type, version)
            )
        `);

        // Tabela de Solicitações de Titulares (LGPD Art. 18)
        await this.db.exec(`
            CREATE TABLE IF NOT EXISTS data_subject_requests (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                request_id TEXT UNIQUE NOT NULL,
                user_id TEXT NOT NULL,
                request_type TEXT NOT NULL CHECK(request_type IN ('access', 'rectification', 'erasure', 'restriction', 'portability', 'objection')),
                status TEXT DEFAULT 'pending' CHECK(status IN ('pending', 'in_progress', 'completed', 'rejected', 'cancelled')),
                description TEXT,
                requested_data TEXT,
                response_data TEXT,
                deadline_date TIMESTAMP,
                completed_at TIMESTAMP,
                verified_by TEXT,
                verification_method TEXT CHECK(verification_method IN ('email', 'phone', 'document', 'account')),
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        `);

        // Tabela de DPIAs (Avaliação de Impacto - LGPD Art. 38)
        await this.db.exec(`
            CREATE TABLE IF NOT EXISTS dpias (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                dpia_id TEXT UNIQUE NOT NULL,
                feature_name TEXT NOT NULL,
                description TEXT NOT NULL,
                data_categories TEXT NOT NULL,
                legal_basis TEXT NOT NULL,
                risk_assessment TEXT NOT NULL,
                mitigation_measures TEXT NOT NULL,
                residual_risk TEXT CHECK(residual_risk IN ('low', 'medium', 'high')),
                approved BOOLEAN DEFAULT 0,
                approved_by TEXT,
                approved_at TIMESTAMP,
                review_date TIMESTAMP,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        `);

        // Tabela de Incidentes (LGPD Art. 48)
        await this.db.exec(`
            CREATE TABLE IF NOT EXISTS security_incidents (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                incident_id TEXT UNIQUE NOT NULL,
                incident_type TEXT NOT NULL CHECK(incident_type IN ('breach', 'leak', 'unauthorized_access', 'loss', 'other')),
                severity TEXT NOT NULL CHECK(severity IN ('low', 'medium', 'high', 'critical')),
                description TEXT NOT NULL,
                affected_data TEXT NOT NULL,
                affected_users_count INTEGER,
                detected_at TIMESTAMP NOT NULL,
                contained_at TIMESTAMP,
                resolved_at TIMESTAMP,
                notified_authority BOOLEAN DEFAULT 0,
                notified_authority_at TIMESTAMP,
                notified_users BOOLEAN DEFAULT 0,
                notified_users_at TIMESTAMP,
                measures_taken TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        `);

        // Tabela de Logs de Auditoria
        await this.db.exec(`
            CREATE TABLE IF NOT EXISTS audit_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                event_type TEXT NOT NULL,
                entity_type TEXT NOT NULL,
                entity_id TEXT,
                user_id TEXT,
                ip_address TEXT,
                user_agent TEXT,
                old_values TEXT,
                new_values TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        `);

        // Índices para performance
        await this.db.exec('CREATE INDEX IF NOT EXISTS idx_consents_user_id ON consents(user_id)');
        await this.db.exec('CREATE INDEX IF NOT EXISTS idx_consents_type ON consents(consent_type)');
        await this.db.exec('CREATE INDEX IF NOT EXISTS idx_requests_user_id ON data_subject_requests(user_id)');
        await this.db.exec('CREATE INDEX IF NOT EXISTS idx_requests_status ON data_subject_requests(status)');
        await this.db.exec('CREATE INDEX IF NOT EXISTS idx_audit_entity ON audit_logs(entity_type, entity_id)');
    }

    async seedInitialData() {
        // Inserir políticas de retenção padrão
        const retentionPolicies = [
            {
                data_category: 'user_account',
                purpose: 'authentication_and_service',
                retention_days: 730, // 2 anos
                legal_basis: 'contract'
            },
            {
                data_category: 'usage_analytics',
                purpose: 'service_improvement',
                retention_days: 365, // 1 ano
                legal_basis: 'legitimate_interest'
            },
            {
                data_category: 'marketing_communications',
                purpose: 'direct_marketing',
                retention_days: 180, // 6 meses após último contato
                legal_basis: 'consent'
            }
        ];

        for (const policy of retentionPolicies) {
            await this.db.run(
                `INSERT OR IGNORE INTO data_retention_policies 
                 (data_category, purpose, retention_days, legal_basis) 
                 VALUES (?, ?, ?, ?)`,
                [policy.data_category, policy.purpose, policy.retention_days, policy.legal_basis]
            );
        }
    }

    async getConnection() {
        if (!this.db) {
            await this.init();
        }
        return this.db;
    }
}

module.exports = new ComplianceDatabase();
```

## 3. Modelos de Dados

```javascript:src/compliance/models/Consent.js
class Consent {
    constructor(db) {
        this.db = db;
    }

    async createConsent(consentData) {
        const {
            user_id,
            consent_type,
            purpose,
            granted,
            version,
            ip_address,
            user_agent,
            consent_text,
            expires_at
        } = consentData;

        const consentId = `CONS-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;

        await this.db.run(
            `INSERT INTO consents 
             (user_id, consent_type, purpose, granted, granted_at, expires_at, version, 
              ip_address, user_agent, consent_text) 
             VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
            [
                user_id,
                consent_type,
                purpose,
                granted,
                new Date().toISOString(),
                expires_at,
                version,
                ip_address,
                user_agent,
                consent_text
            ]
        );

        return consentId;
    }

    async getUserConsents(userId) {
        return await this.db.all(
            `SELECT * FROM consents 
             WHERE user_id = ? 
             ORDER BY granted_at DESC`,
            [userId]
        );
    }

    async withdrawConsent(userId, consentType, version) {
        await this.db.run(
            `UPDATE consents 
             SET withdrawal_at = ?, granted = 0 
             WHERE user_id = ? AND consent_type = ? AND version = ? AND withdrawal_at IS NULL`,
            [new Date().toISOString(), userId, consentType, version]
        );

        return true;
    }

    async getActiveConsent(userId, consentType) {
        return await this.db.get(
```

```javascript:src/compliance/models/DataSubjectRequest.js
        const updates = {
            status,
            updated_at: new Date().toISOString()
```

```javascript
const fs = require('fs').promises;
const path = require('path');
const logger = require('../../utils/logger');

class CodeAnalyzer {
    constructor() {
        this.metrics = {
            complexity: 0,
            maintainability: 0,
            testCoverage: 0,
            performance: 0,
            security: 0
        };
    }

    async analyzeProject(projectPath) {
        logger.info(`Iniciando análise do projeto: ${projectPath}`);
        
        const analysis = {
            files: [],
            metrics: {},
            issues: [],
            recommendations: []
        };

        try {
            // Analisar estrutura do projeto
            const projectStructure = await this.scanProjectStructure(projectPath);
            analysis.structure = projectStructure;

            // Coletar métricas de código
            analysis.metrics = await this.collectCodeMetrics(projectPath);

            // Identificar padrões de código
            analysis.patterns = await this.identifyCodePatterns(projectPath);

            // Detectar problemas comuns
            analysis.issues = await this.detectCommonIssues(projectPath);

            logger.info(`Análise concluída: ${analysis.files.length} arquivos analisados`);
            return analysis;
        } catch (error) {
            logger.error(`Erro na análise do projeto: ${error.message}`);
            throw error;
        }
    }

    async scanProjectStructure(projectPath) {
        const structure = {
            directories: [],
            filesByType: {},
            dependencies: [],
            entryPoints: []
        };

        async function scanDir(currentPath, depth = 0) {
            if (depth > 10) return; // Limitar profundidade

            const items = await fs.readdir(currentPath, { withFileTypes: true });
            
            for (const item of items) {
                const fullPath = path.join(currentPath, item.name);
                
                if (item.isDirectory()) {
                    if (!item.name.startsWith('.') && item.name !== 'node_modules') {
                        structure.directories.push(fullPath);
                        await scanDir(fullPath, depth + 1);
                    }
                } else if (item.isFile()) {
                    const ext = path.extname(item.name);
                    if (!structure.filesByType[ext]) {
                        structure.filesByType[ext] = [];
                    }
                    structure.filesByType[ext].push(fullPath);

                    // Identificar pontos de entrada
                    if (item.name === 'index.js' || item.name === 'server.js' || 
                        item.name === 'app.js' || item.name === 'main.js') {
                        structure.entryPoints.push(fullPath);
                    }

                    // Identificar dependências
                    if (item.name === 'package.json') {
                        try {
                            const content = await fs.readFile(fullPath, 'utf8');
                            const pkg = JSON.parse(content);
                            structure.dependencies = Object.keys(pkg.dependencies || {});
                        } catch (e) {
                            logger.warn(`Não foi possível ler package.json: ${e.message}`);
                        }
                    }
                }
```

        // Dividir arquivos entre workers
        const chunks = this.chunkArray(files, Math.ceil(files.length / this.maxWorkers));

        }
