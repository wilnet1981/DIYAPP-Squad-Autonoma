### Uso de Workers para Tarefas Assíncronas

**workers.js**
```javascript
import { Worker } from 'worker_threads';

export function runWorkerTask(workerData) {
    return new Promise((resolve, reject) => {
        const worker = new Worker('./path/to/workerScript.js', { workerData });
        worker.on('message', resolve);
        worker.on('error', reject);
        worker.on('exit', (code) => {
            if (code !== 0) {
                reject(new Error(`Worker stopped with exit code ${code}`));
            }
        });
    });
}
```

### Solução Proposta: Autorrecuperação de Falhas
Para permitir a autorrecuperação de falhas, usaremos um padrão comum conhecido como "Circuit Breaker". Esse padrão monitora as interações entre serviços e abre um "circuito" (interrompe temporariamente a comunicação) quando uma falha é detectada várias vezes, permitindo que o sistema se recupere antes de tentar novamente. Junto com isso, implementaremos tentativas automáticas com backoff exponencial.

**Exemplo de Código:**
```javascript
class CircuitBreaker {
    constructor(requestFunction, options) {
        this.requestFunction = requestFunction;
        this.failureThreshold = options.failureThreshold || 5;
        this.cooldownPeriod = options.cooldownPeriod || 10000; // 10 seconds
        this.retryTimeout = options.retryTimeout || 5000; // 5 seconds
        this.state = 'CLOSED';
        this.failureCount = 0;
        this.cooldownTimer = null;
    }

    async call(...args) {
        if (this.state === 'OPEN') {
            console.log('Circuit is open, skipping request...');
            return Promise.reject(new Error('Circuit is open'));
        }

        try {
            const result = await this.requestFunction(...args);
            this.resetBreaker();
            return result;
        } catch (error) {
            this.failureCount++;
            if (this.failureCount >= this.failureThreshold) {
                this.tripBreaker();
            }
            throw error;
        }
    }

    tripBreaker() {
        this.state = 'OPEN';
        this.cooldownTimer = setTimeout(() => this.state = 'HALF-OPEN', this.cooldownPeriod);
        console.log(`Circuit tripped. Entering cooldown for ${this.cooldownPeriod / 1000} seconds.`);
    }

    resetBreaker() {
        this.state = 'CLOSED';
        this.failureCount = 0;
        clearTimeout(this.cooldownTimer);
    }
}

// Example usage
const faultyRequest = async () => {
    // Simulate a request that fails sometimes
    if (Math.random() > 0.7) {
        return 'Success';
    } else {
        throw new Error('Failed request');
    }
};

const breaker = new CircuitBreaker(faultyRequest, {
    failureThreshold: 3, 
    cooldownPeriod: 10000, 
    retryTimeout: 2000
});

(async () => {
    // Attempt to make requests using the circuit breaker
    for (let i = 0; i < 10; i++) {
        try {
            const result = await breaker.call();
            console.log('Request succeeded:', result);
        } catch (error) {
            console.log('Request failed:', error.message);
        }
        await new Promise(resolve => setTimeout(resolve, 1000)); // Wait 1 second between calls
    }
})();
```

```javascript
const authenticateToken = (req, res, next) => {
```

#### Airflow - Orquestração

Defina um DAG no Airflow para agendar e monitorar os pipelines de transformação e carga:

```python
from airflow import DAG
from airflow.operators.dummy import DummyOperator
from airflow.operators.bash import BashOperator
from datetime import datetime, timedelta

default_args = {
    'owner': 'data-engineer',
    'depends_on_past': False,
    'start_date': datetime(2023, 10, 1),
    'email_on_failure': False,
    'retries': 1,
    'retry_delay': timedelta(minutes=5),
}

dag = DAG(
    'logs_pipeline',
    default_args=default_args,
    description='ETL pipeline for processing logs',
    schedule_interval='@hourly',
```

### 5. Serviço com Lógica de Negócio (services/projectService.js)

```javascript
const projectRepository = require('../repositories/projectRepository');
const cacheService = require('./cacheService');
const logger = require('../utils/logger');
const { ValidationError } = require('../utils/errors');

class ProjectService {
  async listProjects(userId, page, limit) {
    try {
      const cacheKey = `projects:${userId}:${page}:${limit}`;
      const cachedData = await cacheService.get(cacheKey);

      if (cachedData) {
        logger.debug({
          message: 'Returning cached projects list',
          userId,
          page,
          limit
        });
        return JSON.parse(cachedData);
      }

      const { projects, total } = await projectRepository.findByUserId(userId, page, limit);

      const result = {
        projects,
        total
      };

      // Cache por 5 minutos
      await cacheService.set(cacheKey, JSON.stringify(result), 300);

      return result;
    } catch (err) {
      logger.error({
        message: 'Failed to list projects',
        userId,
        error: err.message,
        stack: process.env.NODE_ENV === 'development' ? err.stack : undefined
      });
      throw err;
    }
  }

  async createProject(userId, projectData) {
    // Validação de negócio
    if (!projectData.name || projectData.name.trim().length < 3) {
      throw new ValidationError('Project name must be at least 3 characters long', {
        field: 'name',
        value: projectData.name
      });
    }

    if (projectData.name.length > 100) {
      throw new ValidationError('Project name must be less than 100 characters', {
        field: 'name',
        value: projectData.name
      });
    }

    // Verificar duplicidade
    const existingProject = await projectRepository.findByName(userId, projectData.name);
    if (existingProject) {
      const error = new Error('Project name already exists');
      error.code = 'PROJECT_NAME_DUPLICATE';
      throw error;
    }

    try {
      const project = await projectRepository.create({
        ...projectData,
        userId,
        createdAt: new Date(),
        updatedAt: new Date(),
        status: 'active'
      });

      // Invalidar cache de listagem
      await cacheService.delPattern(`projects:${userId}:*`);

      logger.info({
        message: 'Project created successfully',
        projectId: project.id,
        userId
      });

      return project;
    } catch (err) {
      logger.error({
        message: 'Failed to create project',
        userId,
        error: err.message,
        stack: process.env.NODE_ENV === 'development' ? err.stack : undefined
      });
      throw err;
    }
  }

  async getProject(userId, projectId) {
    try {
      const cacheKey = `project:${userId}:${projectId}`;
      const cachedProject = await cacheService.get(cacheKey);

      if (cachedProject) {
        logger.debug({
          message: 'Returning cached project',
          projectId,
          userId
        });
        return JSON.parse(cachedProject);
      }

      const project = await projectRepository.findById(projectId);

      if (!project || project.userId !== userId) {
        return null;
      }

      // Cache por 1 minuto
      await cacheService.set(cacheKey, JSON.stringify(project), 60);

      return project;
    } catch (err) {
      logger.error({
        message: 'Failed to get project',
        projectId,
        userId,
        error: err.message,
        stack: process.env.NODE_ENV === 'development' ? err.stack : undefined
      });
      throw err;
    }
  }

  async getCachedProject(userId, projectId) {
    try {
      const cacheKey = `project:${userId}:${projectId}`;
      const cachedProject = await cacheService.get(cacheKey);

      if (cachedProject) {
        const project = JSON.parse(cachedProject);
        project.isFallback = true;
        return project;
      }

      return null;
    } catch (err) {
      logger.warn({
        message: 'Failed to get cached project',
        projectId,
        userId,
        error: err.message
      });
      return null;
    }
  }
```

### 6. Implementação do Circuit Breaker (utils/circuitBreaker.js)

```javascript
const logger = require('./logger');

class CircuitBreaker {
  constructor(options) {
    this.name = options.name;
    this.threshold = options.threshold || 5;
    this.resetTimeout = options.resetTimeout || 30000;
    this.fallback = options.fallback;
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.nextAttempt = 0;
  }

  async exec(fn) {
    if (this.state === 'OPEN') {
      if (this.nextAttempt <= Date.now()) {
        this.state = 'HALF_OPEN';
      } else {
        if (this.fallback) {
          try {
            return await this.fallback();
          } catch (fallbackErr) {
            logger.error({
              message: 'Fallback function failed',
              circuitBreaker: this.name,
              error: fallbackErr.message
            });
            throw fallbackErr;
          }
        }
        const error = new Error('Circuit breaker is open');
        error.isCircuitBreakerOpen = true;
        throw error;
      }
    }

    try {
      const result = await fn();
      this.reset();
      return result;
    } catch (err) {
      this.failureCount++;
      logger.warn({
        message: 'Circuit breaker failure',
        circuitBreaker: this.name,
        failureCount: this.failureCount,
        error: err.message
      });

      if (this.failureCount >= this.threshold) {
        this.trip();
      }
      throw err;
    }
  }

  trip() {
    this.state = 'OPEN';
    this.nextAttempt = Date.now() + this.resetTimeout;
    logger.warn({
      message: 'Circuit breaker tripped',
      circuitBreaker: this.name,
      nextAttempt: new Date(this.nextAttempt).toISOString()
    });
  }

  reset() {
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.nextAttempt = 0;
  }

  halfOpen() {
    this.state = 'HALF_OPEN';
  }
}

module.exports = (options) => new CircuitBreaker(options);
```

### 7. Middleware de Tratamento de Erros (middlewares/errorHandler.js)

```javascript
const logger = require('../utils/logger');

const errorHandler = (err, req, res, next) => {
  const correlationId = req.correlationId || 'unknown';
  const userId = req.user?.id || 'anonymous';

  // Log detalhado do erro
  logger.error({
    correlation_id: correlationId,
    user_id: userId,
    endpoint: `${req.method} ${req.path}`,
    error: err.message,
    stack: process.env.NODE_ENV === 'development' ? err.stack : undefined,
    input: req.body || req.query || req.params
  });

  // Determinar o status code apropriado
  let statusCode = 500;
  let errorResponse = {
    success: false,
    error: 'Internal Server Error',
    message: 'An unexpected error occurred'
  };

  if (err.isCircuitBreakerOpen) {
    statusCode = 503;
    errorResponse = {
      success: false,
      error: 'Service Unavailable',
      message: 'The service is temporarily unavailable. Please try again later.'
    };
  } else if (err.name === 'ValidationError') {
    statusCode = 400;
    errorResponse = {
      success: false,
      error: 'Validation Error',
      message: err.message,
      details: err.details
    };
  } else if (err.name === 'UnauthorizedError') {
    statusCode = 401;
    errorResponse = {
      success: false,
      error: 'Unauthorized',
      message: err.message || 'Authentication required'
    };
  } else if (err.name === 'ForbiddenError') {
    statusCode = 403;
    errorResponse = {
      success: false,
      error: 'Forbidden',
      message: err.message || 'You do not have permission to access this resource'
    };
  } else if (err.name === 'NotFoundError') {
    statusCode = 404;
    errorResponse = {
      success: false,
      error: 'Not Found',
      message: err.message || 'Resource not found'
    };
  } else if (err.code === 'PROJECT_NAME_DUPLICATE') {
    statusCode = 409;
    errorResponse = {
      success: false,
      error: 'Conflict',
      message: err.message
    };
  } else if (err.name === 'RateLimitError') {
    statusCode = 429;
    errorResponse = {
      success: false,
      error: 'Too Many Requests',
      message: err.message || 'You have exceeded the request limit'
    };
  }

  // Adicionar correlation ID na resposta
  errorResponse.correlation_id = correlationId;

  res.status(statusCode).json(errorResponse);
};

module.exports = errorHandler;
```

### 11. Configuração do Cache (services/cacheService.js)

```javascript
const { redisClient } = require('../app');
const logger = require('../utils/logger');

class CacheService {
  async get(key) {
    try {
      const value = await redisClient.get(key);
      return value;
    } catch (err) {
      logger.error({
        message: 'Cache get failed',
        key,
        error: err.message
      });
      return null;
    }
  }

  async set(key, value, ttl) {
    try {
      if (ttl) {
        await redisClient.set(key, value, 'EX', ttl);
      } else {
        await redisClient.set(key, value);
      }
      return true;
    } catch (err) {
      logger.error({
        message: 'Cache set failed',
        key,
        error: err.message
      });
      return false;
    }
  }

  async del(key) {
    try {
      await redisClient.del(key);
      return true;
    } catch (err) {
      logger.error({
        message: 'Cache delete failed',
        key,
        error: err.message
      });
      return false;
    }
  }

  async delPattern(pattern) {
    try {
      const keys = await redisClient.keys(pattern);
      if (keys.length > 0) {
        await redisClient.del(keys);
      }
      return true;
    } catch (err) {
      logger.error({
        message: 'Cache delete pattern failed',
        pattern,
        error: err.message
      });
      return false;
    }
  }

  async getWithFallback(key, fallbackFn, ttl) {
    try {
      const cachedValue = await this.get(key);
      if (cachedValue) {
        return JSON.parse(cachedValue);
      }

      const value = await fallbackFn();
      if (value) {
        await this.set(key, JSON.stringify(value), ttl);
      }
      return value;
    } catch (err) {
      logger.error({
        message: 'Cache get with fallback failed',
        key,
        error: err.message
      });
      return fallbackFn();
    }
  }
}

module.exports = new CacheService();
```

```javascript
// src/autorecovery/index.js
const Docker = require('dockerode');
const axios = require('axios');
const { recoveryRules } = require('./rules');

class AutoRecovery {
  constructor() {
    this.docker = new Docker();
    this.containers = [];
    this.apiEndpoints = [
      'http://localhost:3000/api/health',
      'http://localhost:3001/api/status'
    ];
  }

  async monitor() {
    // Monitorar containers Docker
    this.containers = await this.docker.listContainers({ all: true });
    for (const container of this.containers) {
      const stats = await this.docker.getContainer(container.Id).stats({ stream: false });
      const cpuUsage = this.calculateCPUUsage(stats);
      const memoryUsage = stats.memory_stats.usage / stats.memory_stats.limit;

      // Verificar regras de recuperação
      for (const rule of recoveryRules) {
        if (rule.condition(cpuUsage, memoryUsage)) {
          await rule.action(container.Id);
        }
      }
    }

    // Monitorar APIs
    for (const endpoint of this.apiEndpoints) {
      try {
        const response = await axios.get(endpoint);
        if (response.status !== 200) {
          await this.recoverAPI(endpoint);
        }
      } catch (error) {
        await this.recoverAPI(endpoint);
      }
    }
  }

  calculateCPUUsage(stats) {
    const cpuDelta = stats.cpu_stats.cpu_usage.total_usage - stats.precpu_stats.cpu_usage.total_usage;
    const systemDelta = stats.cpu_stats.system_cpu_usage - stats.precpu_stats.system_cpu_usage;
    return (cpuDelta / systemDelta) * 100;
  }

  async recoverAPI(endpoint) {
    console.log(`[AutoRecovery] Recuperando API: ${endpoint}`);
    // Exemplo: Reiniciar container associado à API
    const container = this.containers.find(c => c.Names.includes('api-service'));
    if (container) {
      await this.docker.getContainer(container.Id).restart();
    }
  }
}

module.exports = AutoRecovery;
```

```javascript
// src/optimization/autoscaler.js
const Docker = require('dockerode');
const docker = new Docker();

class AutoScaler {
  constructor() {
    this.scaleThresholds = {
      cpu: 70,    // Escalar se CPU > 70%
      memory: 0.7 // Escalar se memória > 70%
    };
    this.minReplicas = 1;
    this.maxReplicas = 5;
  }

  async checkAndScale() {
    const containers = await docker.listContainers({ all: true });
    const serviceContainers = containers.filter(c => c.Names.includes('api-service'));

    for (const container of serviceContainers) {
      const stats = await docker.getContainer(container.Id).stats({ stream: false });
      const cpuUsage = this.calculateCPUUsage(stats);
      const memoryUsage = stats.memory_stats.usage / stats.memory_stats.limit;

      if (cpuUsage > this.scaleThresholds.cpu || memoryUsage > this.scaleThresholds.memory) {
        await this.scaleUp(serviceContainers.length);
      } else if (cpuUsage < 30 && memoryUsage < 0.3 && serviceContainers.length > this.minReplicas) {
        await this.scaleDown(serviceContainers.length);
      }
    }
  }

  async scaleUp(currentReplicas) {
    if (currentReplicas < this.maxReplicas) {
      console.log(`[AutoScaler] Escalando para ${currentReplicas + 1} réplicas`);
      await docker.compose.upOne('api-service', { cwd: '/path/to/docker-compose' });
    }
  }

  async scaleDown(currentReplicas) {
    if (currentReplicas > this.minReplicas) {
      console.log(`[AutoScaler] Reduzindo para ${currentReplicas - 1} réplicas`);
      const containers = await docker.listContainers({ all: true });
      const apiContainers = containers.filter(c => c.Names.includes('api-service'));
      await docker.getContainer(apiContainers[0].Id).stop();
    }
  }

  calculateCPUUsage(stats) {
    const cpuDelta = stats.cpu_stats.cpu_usage.total_usage - stats.precpu_stats.cpu_usage.total_usage;
    const systemDelta = stats.cpu_stats.system_cpu_usage - stats.precpu_stats.system_cpu_usage;
    return (cpuDelta / systemDelta) * 100;
  }
}

module.exports = AutoScaler;
```

## **4. Código de Correção para Vulnerabilidades Críticas**
### **4.1. SQL Injection (Prepared Statements)**
**Antes (Vulnerável):**
```javascript
// src/controllers/user.js
const getUser = async (req, res) => {
  const userId = req.params.id;
  const query = `SELECT * FROM users WHERE id = ${userId}`; // ❌ Vulnerável
  const result = await db.query(query);
  res.json(result.rows[0]);
};
```

**Depois (Corrigido):**
```javascript
// src/controllers/user.js
const getUser = async (req, res) => {
  const userId = req.params.id;
  const query = 'SELECT * FROM users WHERE id = $1'; // ✅ Prepared statement
  const result = await db.query(query, [userId]);
  res.json(result.rows[0]);
};
```

### **4.2. XSS (Sanitização com DOMPurify + CSP)**
**Antes (Vulnerável):**
```html

```

**Depois (Corrigido):**
```javascript
// src/middleware/sanitize.js
const DOMPurify = require('dompurify');
const { JSDOM } = require('jsdom');
const window = new JSDOM('').window;
const purify = DOMPurify(window);

const sanitizeInput = (input) => purify.sanitize(input);
```

```html
<!-- src/views/profile.ejs -->

```
