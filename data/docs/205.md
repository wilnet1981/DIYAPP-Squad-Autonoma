 * DIYAPP V12 - Sistema de Métricas de Estabilidade e Performance
 * KPIs definidos para operação 100% autônoma

class V12MetricsCollector {
    constructor() {
        this.metrics = {
            // KPIs de Estabilidade Autônoma
            autonomy: {
                rate: 0,          // % de execuções sem intervenção humana
                mttr: 0,          // Mean Time To Recovery (minutos)
                errorRate: 0,     // Erros por 1000 execuções
                escalationRate: 0 // % de tarefas escaladas para humano
            },
            
            // KPIs de Performance Premium
            performance: {
                latencyP50: 0,    // Latência mediana (ms)
                latencyP95: 0,    // Latência percentil 95 (ms)
                latencyP99: 0,    // Latência percentil 99 (ms)
                throughput: 0,    // Requisições por segundo
                concurrency: 0,   // Conexões simultâneas
                uptime: 100       // % de disponibilidade
            },
            
            // KPIs de Qualidade
            quality: {
                successRate: 0,   // % de tarefas concluídas com sucesso
                accuracy: 0,      // Precisão nas respostas
                consistency: 0    // Consistência entre execuções similares
            }
        };
        
        this.thresholds = {
            critical: {
                autonomyRate: 99.0,    // Abaixo disso = intervenção necessária
                mttr: 10,              // Acima de 10min = crítico
                errorRate: 5,          // Acima de 5 erros/1k = crítico
                latencyP95: 500        // Acima de 500ms = degradação
            },
            warning: {
                autonomyRate: 99.5,
                mttr: 5,
                errorRate: 2,
                latencyP95: 200
            }
        };
        
        this.logs = [];
    }
    
    /**
     * Registra uma execução autônoma
     * @param {Object} execution - Dados da execução
     */
    recordExecution(execution) {
        const timestamp = new Date().toISOString();
        
        this.logs.push({
            timestamp,
            type: execution.type || 'autonomous',
            duration: execution.duration || 0,
            success: execution.success || false,
            escalated: execution.escalated || false,
            error: execution.error || null,
            metadata: execution.metadata || {}
        });
        
        // Mantém apenas últimas 10000 execuções
        if (this.logs.length > 10000) {
            this.logs = this.logs.slice(-10000);
        }
        
        this.calculateMetrics();
    }
    
    /**
     * Calcula todas as métricas baseadas nos logs
     */
    calculateMetrics() {
        if (this.logs.length === 0) return;
        
        const recentLogs = this.getLastHours(24); // Últimas 24h
        
        // Cálculo de autonomia
        const totalExecutions = recentLogs.length;
        const autonomousExecutions = recentLogs.filter(log => !log.escalated).length;
        const successfulExecutions = recentLogs.filter(log => log.success).length;
        const escalatedExecutions = recentLogs.filter(log => log.escalated).length;
        const errorExecutions = recentLogs.filter(log => log.error).length;
        
        // Tempo de recuperação (simulado - em produção viria do sistema)
        const recoveryTimes = recentLogs
            .filter(log => log.error && !log.escalated)
            .map(log => log.metadata.recoveryTime || 2.5);
        
        // Cálculo de latência
        const latencies = recentLogs
            .map(log => log.duration)
            .filter(d => d > 0);
        
        // Atualiza métricas
        this.metrics.autonomy.rate = totalExecutions > 0 ? 
            (autonomousExecutions / totalExecutions * 100) : 0;
        
        this.metrics.autonomy.mttr = recoveryTimes.length > 0 ?
            recoveryTimes.reduce((a, b) => a + b, 0) / recoveryTimes.length : 0;
        
        this.metrics.autonomy.errorRate = totalExecutions > 0 ?
            (errorExecutions / totalExecutions * 1000) : 0;
        
        this.metrics.autonomy.escalationRate = totalExecutions > 0 ?
            (escalatedExecutions / totalExecutions * 100) : 0;
        
        this.metrics.performance.successRate = totalExecutions > 0 ?
            (successfulExecutions / totalExecutions * 100) : 0;
        
        if (latencies.length > 0) {
            const sorted = [...latencies].sort((a, b) => a - b);
            this.metrics.performance.latencyP50 = this.percentile(sorted, 50);
            this.metrics.performance.latencyP95 = this.percentile(sorted, 95);
            this.metrics.performance.latencyP99 = this.percentile(sorted, 99);
        }
        
        // Throughput (execuções por segundo nas últimas 5min)
        const last5min = this.getLastMinutes(5);
        this.metrics.performance.throughput = last5min.length > 0 ?
            last5min.length / 300 : 0; // 5min = 300s
    }

     * Verifica se algum KPI está fora dos limites
     * @returns {Array} Alertas gerados
     */

        const alerts = [];
        
        // Verifica autonomia
        if (this.metrics.autonomy.rate < this.thresholds.critical.autonomyRate) {
            alerts.push({
                level: 'CRITICAL',
                metric: 'autonomyRate',
                value: this.metrics.autonomy.rate,
                threshold: this.thresholds.critical.autonomyRate,
                message: `Autonomia abaixo do limite crítico: ${this.metrics.autonomy.rate.toFixed(1)}%`
            });
        } else if (this.metrics.autonomy.rate < this.thresholds.warning.autonomyRate) {
            alerts.push({
                level: 'WARNING',
                metric: 'autonomyRate',
                value: this.metrics.autonomy.rate,
                threshold: this.thresholds.warning.autonomyRate,
                message: `Autonomia abaixo do limite de warning: ${this.metrics.autonomy.rate.toFixed(1)}%`
            });
        }

            alerts.push({

## ADR-001: Arquitetura Modular com Monorepo

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

**CONTEXTO:**
O DIYAPP precisa evoluir para uma arquitetura mais sustentável que permita:
1. Desenvolvimento paralelo por múltiplas squads
2. Deploy independente de módulos
3. Reutilização máxima de código
4. Manutenção simplificada de dependências

**DECISÃO:**
Adotar estrutura de monorepo com TurboRepo para gerenciamento de builds e deploys. 
Cada módulo será um pacote independente com suas próprias dependências e configurações.

**OPÇÕES CONSIDERADAS:**
- **Opção A:** Monorepo com Lerna + Yarn Workspaces
  - Prós: Maturidade, comunidade grande
  - Contras: Builds mais lentas, configuração complexa
- **Opção B:** Monorepo com TurboRepo
  - Prós: Builds incrementais cacheadas, integração com Vercel, DX excelente
  - Contras: Relativamente novo, dependência da Vercel
- **Opção C:** Multi-repo tradicional
  - Prós: Isolamento total, deploy independente
  - Contras: Sincronização complexa, duplicação de código, overhead de CI/CD

**Opção escolhida:** B - Justificativa: TurboRepo oferece o melhor balanceamento entre performance de build e simplicidade de configuração. Suporte nativo a deploy incremental é crucial para nossa estratégia de CI/CD.

**CONSEQUÊNCIAS:**
- Positivas: Builds 70% mais rápidas, deploy seletivo, cache distribuído
- Negativas: Curva de aprendizado, lock-in parcial com Vercel
- Riscos: Falhas no cache podem causar builds incorretas. Mitigação: Implementar validação de checksum

**REVISÃO:** 2024-04-15

      await this.setupLogging();

      console.error(`❌ Failed to initialize ${this.config.name}:`, error);
      throw error;
    }

        message: `Memory usage: ${Math.round(process.memoryUsage().heapUsed / 1024 / 1024)}MB`,
      },
      uptime: {
        status: 'pass',
        latency: process.uptime(),
      },
    };

    return {
      status: Object.values(checks).every(c => c.status === 'pass') ? 'healthy' : 'degraded',
      checks,
      timestamp: new Date().toISOString(),
    };
  }

  registerModule(name: string, module: any): void {
    if (this.modules.has(name)) {
      throw new Error(`Module ${name} already registered`);
    }
    this.modules.set(name, module);
  }

    const module = this.modules.get(name);
    if (!module) {
      throw new Error(`Module ${name} not found`);
    }
    return module as T;
  }

export enum LogLevel {
  DEBUG = 0,
  INFO = 1,
  WARN = 2,
  ERROR = 3,
}

export interface LogEntry {
  timestamp: string;
  level: LogLevel;
  module: string;
  message: string;
  data?: Record<string, unknown>;
  correlationId?: string;
}

export class Logger {
  private static instance: Logger;
  private level: LogLevel;
  private transports: Array<(entry: LogEntry) => void> = [];

  private constructor(level: LogLevel = LogLevel.INFO) {
    this.level = level;
    this.setupDefaultTransports();
  }

  static getInstance(): Logger {
    if (!Logger.instance) {
      Logger.instance = new Logger();
    }
    return Logger.instance;
  }

  debug(module: string, message: string, data?: Record<string, unknown>): void {
    this.log(LogLevel.DEBUG, module, message, data);
  }

  info(module: string, message: string, data?: Record<string, unknown>): void {
    this.log(LogLevel.INFO, module, message, data);
  }

  warn(module: string, message: string, data?: Record<string, unknown>): void {
    this.log(LogLevel.WARN, module, message, data);
  }

  error(module: string, message: string, error?: Error, data?: Record<string, unknown>): void {
    const errorData = {
      ...data,
      error: {
        name: error?.name,
        message: error?.message,
        stack: process.env.NODE_ENV === 'development' ? error?.stack : undefined,
      },
    };
    this.log(LogLevel.ERROR, module, message, errorData);
  }

  addTransport(transport: (entry: LogEntry) => void): void {
    this.transports.push(transport);
  }

  private log(
    level: LogLevel,
    module: string,
    message: string,
    data?: Record<string, unknown>
  ): void {
    if (level < this.level) return;

    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level,
      module,
      message,
      data,
      correlationId: this.getCorrelationId(),
    };

    this.transports.forEach(transport => {
      try {
        transport(entry);
      } catch (error) {
        // Não quebrar o app se um transport falhar
        console.error('Logger transport failed:', error);
      }
    });
  }

  private setupDefaultTransports(): void {
    // Transporte para console
    this.addTransport((entry) => {
      const prefix = `[${entry.timestamp}] [${entry.module}]`;
      const levelStr = LogLevel[entry.level];
      
      switch (entry.level) {
        case LogLevel.DEBUG:
          console.debug(`${prefix} 🔍 ${entry.message}`, entry.data || '');
          break;
        case LogLevel.INFO:
          console.info(`${prefix} ℹ️ ${entry.message}`, entry.data || '');
          break;
        case LogLevel.WARN:
          console.warn(`${prefix} ⚠️ ${entry.message}`, entry.data || '');
          break;
        case LogLevel.ERROR:
          console.error(`${prefix} ❌ ${entry.message}`, entry.data || '');
          break;
      }
    });

    // Transporte para arquivo (em produção)
    if (process.env.NODE_ENV === 'production') {
      this.addTransport((entry) => {
        // Implementar escrita em arquivo/Elasticsearch
      });
    }
  }

  private getCorrelationId(): string {
    // Implementar tracing distribuído
    return process.pid.toString();
  }
}

// packages/core/src/utils/health-check.ts

import { HealthStatus, CheckResult } from '@diyapp/shared-types';

export class HealthChecker {
  private checks: Map<string, () => Promise<CheckResult>> = new Map();
  private interval?: NodeJS.Timeout;

  registerCheck(name: string, check: () => Promise<CheckResult>): void {
    this.checks.set(name, check);
  }

  async runChecks(): Promise<HealthStatus> {
    const results: Record<string, CheckResult> = {};
    const startTime = Date.now();

    // Executar checks em paralelo com timeout
    const checkPromises = Array.from(this.checks.entries()).map(
      async ([name, check]) => {
        try {
          const result = await Promise.race([
            check(),
            new Promise<CheckResult>((_, reject) =>
              setTimeout(() => reject(new Error('Check timeout')), 5000)
            ),
          ]);
          results[name] = result;
        } catch (error) {
          results[name] = {
            status: 'fail',
            message: error instanceof Error ? error.message : 'Unknown error',
          };
        }
      }
    );

    await Promise.allSettled(checkPromises);

    const overallStatus = Object.values(results).every(r => r.status === 'pass')
      ? 'healthy'
      : Object.values(results).some(r => r.status === 'fail')
      ? 'unhealthy'
      : 'degraded';

    return {
      status: overallStatus,
      checks: results,
      timestamp: new Date().toISOString(),
    };
  }

  startPeriodicChecks(intervalMs: number = 30000): void {
    if (this.interval) {
      clearInterval(this.interval);
    }

    this.interval = setInterval(async () => {
      const health = await this.runChecks();
      
      if (health.status === 'unhealthy') {
        console.error('❌ System unhealthy:', health);
        // Notificar sistemas de monitoramento
      } else if (health.status === 'degraded') {
        console.warn('⚠️ System degraded:', health);
      }
    }, intervalMs);
  }

  stop(): void {
    if (this.interval) {
      clearInterval(this.interval);
      this.interval = undefined;
    }
  }
}

    this.components.forEach(meta => {
      total += meta.bundleSize;
    });
    return total;
  }

  observeLoading(componentId: string, callback: (loading: boolean) => void): () => void {
    if (!this.loadingObservers.has(componentId)) {
      this.loadingObservers.set(componentId, []);
    }
    const observers = this.loadingObservers.get(componentId)!;
    observers.push(callback);

    return () => {
      const index = observers.indexOf(callback);
      if (index > -1) {
        observers.splice(index, 1);
      }
    };
  }

  notifyLoading(componentId: string, loading: boolean): void {
    const observers = this.loadingObservers.get(componentId);
    if (observers) {
      observers.forEach(callback => callback(loading));
    }
  }

import asyncio
import logging
from typing import Optional, Any
from datetime import datetime, timedelta
from redis.asyncio import Redis, ConnectionError, TimeoutError
from redis.asyncio.retry import Retry
from redis.backoff import ExponentialBackoff

logger = logging.getLogger(__name__)

class CircuitBreaker:
    """Circuit breaker pattern para Redis"""
    
    def __init__(
        self,
        failure_threshold: int = 5,
        recovery_timeout: int = 30,
        half_open_max_attempts: int = 3
    ):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_max_attempts = half_open_max_attempts
        self.state = "CLOSED"  # CLOSED, OPEN, HALF_OPEN
        self.failure_count = 0
        self.last_failure_time: Optional[datetime] = None
        self.half_open_attempts = 0
        self.lock = asyncio.Lock()
    
    async def execute(self, func, *args, **kwargs) -> Any:
        """Executa função com circuit breaker"""
        async with self.lock:
            if self.state == "OPEN":
                # Verifica se pode tentar recovery
                if self.last_failure_time and \
                   datetime.now() - self.last_failure_time > timedelta(seconds=self.recovery_timeout):
                    self.state = "HALF_OPEN"
                    self.half_open_attempts = 0
                else:
                    raise ConnectionError("Circuit breaker is OPEN")
            
            try:
                result = await func(*args, **kwargs)
                
                # Sucesso - reset circuit breaker
                if self.state == "HALF_OPEN":
                    self.half_open_attempts += 1
                    if self.half_open_attempts >= self.half_open_max_attempts:
                        self.state = "CLOSED"
                        self.failure_count = 0
                        logger.info("Circuit breaker reset to CLOSED")
                elif self.state == "CLOSED":
                    self.failure_count = 0
                
                return result
                
            except (ConnectionError, TimeoutError) as e:
                await self._record_failure()
                logger.error(f"Circuit breaker failure: {e}")
                raise
            except Exception as e:
                # Outros erros não afetam o circuit breaker
                raise
    
    async def _record_failure(self):
        """Registra falha e atualiza estado"""
        self.failure_count += 1
        self.last_failure_time = datetime.now()
        
        if self.state == "HALF_OPEN":
            self.state = "OPEN"
            logger.warning("Circuit breaker moved from HALF_OPEN to OPEN")
        elif self.failure_count >= self.failure_threshold:
            self.state = "OPEN"
            logger.warning("Circuit breaker moved from CLOSED to OPEN")

class ResilientRedis:
    """Cliente Redis com resiliência embutida"""
    
    def __init__(self, redis_url: str, timeout: int = 3):
        self.redis_url = redis_url
        self.timeout = timeout
        self.circuit_breaker = CircuitBreaker()
        self._client: Optional[Redis] = None
    
    async def get_client(self) -> Redis:
        """Retorna cliente Redis com retry configurado"""
        if not self._client:
            retry = Retry(ExponentialBackoff(), 3)
            self._client = Redis.from_url(
                self.redis_url,
                socket_timeout=self.timeout,
                socket_connect_timeout=self.timeout,
                retry=retry,
                retry_on_error=[ConnectionError, TimeoutError],
                decode_responses=True
            )
        return self._client
    
    async def get(self, key: str) -> Optional[str]:
        """GET com circuit breaker"""
        async def _get():
            client = await self.get_client()
            return await client.get(key)
        
        return await self.circuit_breaker.execute(_get)
    
    async def set(
        self,
        key: str,
        value: str,
        ex: Optional[int] = None
    ) -> bool:
        """SET com circuit breaker"""
        async def _set():
            client = await self.get_client()
            return await client.set(key, value, ex=ex)
        
        return await self.circuit_breaker.execute(_set)
    
    async def delete(self, key: str) -> int:
        """DELETE com circuit breaker"""
        async def _delete():
            client = await self.get_client()
            return await client.delete(key)
        
        return await self.circuit_breaker.execute(_delete)
    
    async def close(self):
        """Fecha conexão"""
        if self._client:
            await self._client.close()

        for key in sorted(kwargs.keys()):
            key_parts.append(f"{key}:{kwargs[key]}")
        
        # Cria hash para chaves longas
        key_str = ":".join(key_parts)
        if len(key_str) > 100:
            key_str = f"{prefix}:{hashlib.md5(key_str.encode()).hexdigest()}"
        
        return key_str
    
    async def get_or_set(
        self,
        key_prefix: str,
        fetch_func: Callable[..., T],
        ttl: Optional[int] = None,
        *args,
        **kwargs
    ) -> T:
        """
        Pega do cache ou executa função e cacheia resultado
        Implementa cache-aside pattern
        """
        if not settings.REDIS_CACHE_ENABLED:
            # Cache desabilitado - executa diretamente
            return await fetch_func(*args, **kwargs)
        
        cache_key = self._generate_key(key_prefix, *args, **kwargs)
        
        try:
            # Tenta pegar do cache
            cached = await self.redis.get(cache_key)
            if cached:
                logger.debug(f"Cache hit for key: {cache_key}")
                return json.loads(cached)
            
            # Cache miss - executa função
            logger.debug(f"Cache miss for key: {cache_key}")
            result = await fetch_func(*args, **kwargs)
            
            # Cacheia resultado
            if result is not None:
                cache_ttl = ttl or self.default_ttl
                await self.redis.set(
                    cache_key,
                    json.dumps(result, default=str),
                    ex=cache_ttl
                )
            
            return result
            
        except Exception as e:
            # Se cache falhar, executa função sem cache
            logger.error(f"Cache error for key {cache_key}: {e}")
            return await fetch_func(*args, **kwargs)
    
    async def invalidate_pattern(self, pattern: str) -> int:
        """Invalida todas as chaves que correspondem ao pattern"""
        try:
            client = await self.redis.get_client()
            keys = await client.keys(pattern)
            if keys:
                return await client.delete(*keys)
            return 0
        except Exception as e:
            logger.error(f"Error invalidating cache pattern {pattern}: {e}")
            return 0

def cached(prefix: str, ttl: Optional[int] = None):
    """Decorator para cache automático"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # Remove self/cls dos args para chave de cache
            cache_args = args[1:] if args else args
            cache_service = args[0].cache_service if args else None
            
            if not cache_service:
                # Se não tem cache_service, executa sem cache
                return await func(*args, **kwargs)
            
            return await cache_service.get_or_set(
                prefix,
                lambda: func(*args, **kwargs),
                ttl,
                *cache_args,
                **kwargs
            )
        return wrapper
    return decorator

# src/core/queue/redis_queue.py
import asyncio
import uuid
import json
from typing import Dict, Any, Optional, Callable, List
from datetime import datetime
from enum import Enum
import logging

logger = logging.getLogger(__name__)

class JobStatus(Enum):
    """Status dos jobs"""
    PENDING = "pending"
    PROCESSING = "processing"
    COMPLETED = "completed"
    FAILED = "failed"
    RETRY = "retry"

class AsyncQueue:
    """Sistema de filas assíncronas baseado em Redis"""
    
    def __init__(self, redis_client: ResilientRedis, queue_name: str):
        self.redis = redis_client
        self.queue_name = queue_name
        self.processing_queue = f"{queue_name}:processing"
        self.failed_queue = f"{queue_name}:failed"
        self.completed_queue = f"{queue_name}:completed"
    
    async def enqueue(
        self,
        job_type: str,
        data: Dict[str, Any],
        delay: int = 0,
        priority: int = 1
    ) -> str:
        """
        Adiciona job à fila
        """
        job_id = str(uuid.uuid4())
        job_data = {
            "id": job_id,
            "type": job_type,
            "data": data,
            "status": JobStatus.PENDING.value,
            "created_at": datetime.utcnow().isoformat(),
            "priority": priority,
            "attempts": 0,
            "max_attempts": settings.QUEUE_RETRY_ATTEMPTS
        }
        
        try:
            client = await self.redis.get_client()
            
            # Armazena job details
            job_key = f"job:{queue_name}:{job_id}"
            await client.set(
                job_key,
                json.dumps(job_data),
                ex=86400  # 24 horas
            )
            
            # Adiciona à fila principal
            if delay > 0:
                # Job delayed
                score = datetime.utcnow().timestamp() + delay
                await client.zadd(
                    f"{self.queue_name}:delayed",
                    {job_id: score}
                )
            else:
                # Job imediato
                await client.lpush(
                    self.queue_name,
                    job_id
                )
            
            logger.info(f"Job enqueued: {job_id} ({job_type})")
            return job_id
            
        except Exception as e:
            logger.error(f"Error enqueuing job: {e}")
            raise
    
    async def dequeue(self) -> Optional[Dict[str, Any]]:
        """
        Pega próximo job da fila
        Implementa BRPOPLPUSH para atomicidade
        """
        try:
            client = await self.redis.get_client()
            
            # Move job da fila principal para processing
            job_id = await client.brpoplpush(
                self.queue_name,
                self.processing_queue,
                timeout=5
            )
            
            if not job_id:
                return None
            
            # Pega dados do job
            job_key = f"job:{self.queue_name}:{job_id}"
            job_data_json = await client.get(job_key)
            
            if not job_data_json:
                # Job não encontrado - remove da processing queue
                await client.lrem(self.processing_queue, 0, job_id)
                return None
            
            job_data = json.loads(job_data_json)
            job_data["status"] = JobStatus.PROCESSING.value
            job_data["started_at"] = datetime.utcnow().isoformat()
            job_data["attempts"] += 1
            
            # Atualiza status
            await client.set(job_key, json.dumps(job_data))
            
            return job_data
            
        except Exception as e:
            logger.error(f"Error dequeuing job: {e}")
            return None
    
    async def complete_job(self, job_id: str, result: Dict[str, Any]):
        """Marca job como completado"""
        await self._update_job_status(
            job_id,
            JobStatus.COMPLETED,
            result=result
        )
        
        # Move para completed queue
        client = await self.redis.get_client()
        await client.lrem(self.processing_queue, 0, job_id)
        await client.lpush(self.completed_queue, job_id)
    
    async def fail_job(self, job_id: str, error: str):
        """Marca job como falho"""
        job_data = await self._get_job(job_id)
        
        if job_data and job_data["attempts"] < job_data["max_attempts"]:
            # Retry
            await self._update_job_status(
                job_id,
                JobStatus.RETRY,
                error=error
