### 1.2 Débitos Técnicos Críticos (ADR-001)
**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

**CONTEXTO:**
A V11 apresenta acoplamento excessivo entre camadas, falta de testes automatizados, inconsistência em padrões de código e ausência de métricas de performance.

**DECISÃO:**
Refatorar para arquitetura em camadas com contratos bem definidos, implementar testes automatizados em todos os níveis, estabelecer padrões de código rigorosos e adicionar observabilidade completa.

**OPÇÕES CONSIDERADAS:**
- **Opção A:** Refatoração incremental mantendo compatibilidade
- **Opção B:** Reescrita completa com nova arquitetura
- **Opção escolhida:** A - Justificativa: Menor risco, entrega contínua de valor

**CONSEQUÊNCIAS:**
- Positivas: Estabilidade melhorada, manutenibilidade, performance
- Negativas: Esforço inicial maior, curva de aprendizado
- Riscos: Regressões durante refatoração

### 3.3 Estrutura de Controller Padrão
```typescript
// src/api/controllers/base.controller.ts
import { Request, Response, NextFunction } from 'express';
import { Logger } from '../../shared/utils/logger';
import { ApiResponse } from '../../shared/types/api.types';
import { ValidationError } from '../../shared/errors/validation.error';

export abstract class BaseController {
  protected readonly logger: Logger;

  constructor(controllerName: string) {
    this.logger = new Logger(controllerName);
  }

  protected async execute(
    req: Request,
    res: Response,
    next: NextFunction,
    handler: () => Promise<any>
  ): Promise<void> {
    const startTime = Date.now();
    const requestId = req.headers['x-request-id'] as string || crypto.randomUUID();

    try {
      this.logger.info('Request started', { 
        requestId, 
        path: req.path,
        method: req.method 
      });

      const result = await handler();
      
      const response: ApiResponse = {
        success: true,
        data: result,
        metadata: {
          requestId,
          timestamp: new Date().toISOString(),
          duration: Date.now() - startTime
        }
      };

      res.status(200).json(response);
      
      this.logger.info('Request completed', {
        requestId,
        duration: Date.now() - startTime,
        status: 200
      });
    } catch (error) {
      this.handleError(error, req, res, next, requestId, startTime);
    }
  }

  private handleError(
    error: any,
    req: Request,
    res: Response,
    next: NextFunction,
    requestId: string,
    startTime: number
  ): void {
    if (error instanceof ValidationError) {
      res.status(400).json({
        success: false,
        error: {
          code: 'VALIDATION_ERROR',
          message: error.message,
          details: error.details
        },
        metadata: { requestId }
      });
    } else {
      this.logger.error('Unhandled error', {
        requestId,
        error: error.message,
        stack: error.stack
      });
      
      res.status(500).json({
        success: false,
        error: {
          code: 'INTERNAL_ERROR',
          message: 'Internal server error'
        },
        metadata: { requestId }
      });
    }

    this.logger.error('Request failed', {
      requestId,
      duration: Date.now() - startTime,
      status: res.statusCode,
      error: error.message
    });
  }
}
```

### 3.4 Padrão de Repository com Prisma
```typescript
// src/core/infrastructure/database/base.repository.ts
import { PrismaClient, Prisma } from '@prisma/client';
import { Logger } from '../../../shared/utils/logger';

export abstract class BaseRepository<T> {
  protected readonly prisma: PrismaClient;
  protected readonly logger: Logger;
  protected readonly model: keyof PrismaClient;

  constructor(model: keyof PrismaClient) {
    this.prisma = new PrismaClient({
      log: [
        { level: 'query', emit: 'event' },
        { level: 'error', emit: 'event' },
        { level: 'info', emit: 'event' },
        { level: 'warn', emit: 'event' }
      ]
    });
    
    this.logger = new Logger(`${model}Repository`);
    this.model = model;
    
    this.setupPrismaLogging();
  }

  private setupPrismaLogging(): void {
    this.prisma.$on('query', (e) => {
      this.logger.debug('Database query', {
        query: e.query,
        duration: e.duration,
        params: e.params
      });
    });

    this.prisma.$on('error', (e) => {
      this.logger.error('Database error', { error: e.message });
    });
  }

  protected async withTransaction<T>(
    operation: (tx: Prisma.TransactionClient) => Promise<T>
  ): Promise<T> {
    return this.prisma.$transaction(async (tx) => {
      try {
        return await operation(tx);
      } catch (error) {
        this.logger.error('Transaction failed', { error });
        throw error;
      }
    });
  }

  async disconnect(): Promise<void> {
    await this.prisma.$disconnect();
  }
}
```

// src/services/redis.js
const Redis = require('ioredis');
const config = require('../config');
const logger = require('../utils/logger');

class RedisService {
  constructor() {
    this.client = null;
    this.connected = false;
  }

  async initialize() {
    try {
      this.client = new Redis({
        host: config.redis.host,
        port: config.redis.port,
        password: config.redis.password,
        tls: config.redis.tls ? {} : undefined,
        keyPrefix: config.redis.keyPrefix,
        retryStrategy: (times) => {
          const delay = Math.min(times * 50, 2000);
          return delay;
        },
        maxRetriesPerRequest: 3
      });

      this.client.on('connect', () => {
        this.connected = true;
        logger.info('Redis conectado');
      });

      this.client.on('error', (error) => {
        this.connected = false;
        logger.error('Redis error:', { error: error.message });
      });

      this.client.on('close', () => {
        this.connected = false;
        logger.warn('Redis desconectado');
      });

      // Testa conexão
      await this.client.ping();
      return this.client;
    } catch (error) {
      logger.error('Falha ao conectar ao Redis:', { error: error.message });
      throw error;
    }
  }

  async get(key) {
    if (!this.connected) {
      throw new Error('Redis não conectado');
    }

    try {
      const value = await this.client.get(key);
      return value ? JSON.parse(value) : null;
    } catch (error) {
      logger.error('Erro ao buscar do Redis:', { key, error: error.message });
      return null;
    }
  }

  async set(key, value, ttl = config.cache.ttl) {
    if (!this.connected) {
      throw new Error('Redis não conectado');
    }

    try {
      const serialized = JSON.stringify(value);
      if (ttl) {
        await this.client.setex(key, ttl, serialized);
      } else {
        await this.client.set(key, serialized);
      }
      return true;
    } catch (error) {
      logger.error('Erro ao salvar no Redis:', { key, error: error.message });
      return false;
    }
  }

  async del(key) {
    if (!this.connected) {
      throw new Error('Redis não conectado');
    }

    try {
      await this.client.del(key);
      return true;
    } catch (error) {
      logger.error('Erro ao deletar do Redis:', { key, error: error.message });
      return false;
    }
  }

  async exists(key) {
    if (!this.connected) {
      throw new Error('Redis não conectado');
    }

    try {
      const result = await this.client.exists(key);
      return result === 1;
    } catch (error) {
      logger.error('Erro ao verificar existência no Redis:', { key, error: error.message });
      return false;
    }
  }

  async flushPattern(pattern) {
    if (!this.connected) {
      throw new Error('Redis não conectado');
    }

    try {
      const keys = await this.client.keys(`${config.redis.keyPrefix}${pattern}`);
      if (keys.length > 0) {
        await this.client.del(...keys);
      }
      return keys.length;
    } catch (error) {
      logger.error('Erro ao limpar padrão no Redis:', { pattern, error: error.message });
      return 0;
    }
  }

  getClient() {
    return this.client;
  }

  isConnected() {
    return this.connected;
  }
}

module.exports = new RedisService();

      });

      throw error;

          tls: config.redis.tls ? {} : undefined

        });

import { Routes, Route, Navigate } from 'react-router-dom'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { ErrorBoundary } from 'react-error-boundary'
import Layout from './components/Layout/Layout'
import LoadingSpinner from './components/Feedback/LoadingSpinner'
import ErrorFallback from './components/Feedback/ErrorFallback'

// Lazy load pages for code splitting
const Dashboard = lazy(() => import('./pages/Dashboard'))
const Projects = lazy(() => import('./pages/Projects'))
const AIStudio = lazy(() => import('./pages/AIStudio'))
const Settings = lazy(() => import('./pages/Settings'))
const Documentation = lazy(() => import('./pages/Documentation'))

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 5 * 60 * 1000, // 5 minutes
      retry: 2,
      retryDelay: attemptIndex => Math.min(1000 * 2 ** attemptIndex, 30000),
    },
  },
})

function App() {
  return (
    <ErrorBoundary FallbackComponent={ErrorFallback}>
      <QueryClientProvider client={queryClient}>
        <Layout>
          <Suspense fallback={<LoadingSpinner fullScreen />}>
            <Routes>
              <Route path="/" element={<Navigate to="/dashboard" replace />} />
              <Route path="/dashboard" element={<Dashboard />} />
              <Route path="/projects" element={<Projects />} />
              <Route path="/ai-studio" element={<AIStudio />} />
              <Route path="/settings" element={<Settings />} />
              <Route path="/docs" element={<Documentation />} />
              <Route path="*" element={<Navigate to="/dashboard" replace />} />
            </Routes>
          </Suspense>
        </Layout>
      </QueryClientProvider>
    </ErrorBoundary>
  )
}

export default App

```sql
-- =============================================
-- DIYAPP V12 - SCHEMA OTIMIZADO
-- =============================================

-- Enable essential extensions
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pg_stat_statements";
CREATE EXTENSION IF NOT EXISTS "btree_gin";

-- =============================================
-- CORE TABLES WITH PARTITIONING
-- =============================================

-- Users table with partitioning by created_date
CREATE TABLE users (
    user_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    email VARCHAR(255) NOT NULL,
    username VARCHAR(100) NOT NULL,
    full_name VARCHAR(255),
    status VARCHAR(20) DEFAULT 'active' CHECK (status IN ('active', 'inactive', 'suspended', 'pending')),
    tier VARCHAR(20) DEFAULT 'free' CHECK (tier IN ('free', 'pro', 'enterprise')),
    
    -- Profile data (JSONB for flexibility)
    profile_data JSONB DEFAULT '{}',
    
    -- Timestamps with timezone
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    last_login_at TIMESTAMPTZ,
    
    -- Audit fields
    created_by UUID,
    updated_by UUID,
    
    -- Soft delete
    deleted_at TIMESTAMPTZ,
    deleted_by UUID,
    
    -- Index-only fields for performance
    email_domain VARCHAR(100) GENERATED ALWAYS AS (SUBSTRING(email FROM '@(.*)$')) STORED,
    account_age_days INTEGER GENERATED ALWAYS AS (EXTRACT(DAY FROM NOW() - created_at)) STORED
) PARTITION BY RANGE (created_at);

-- Create monthly partitions for users
CREATE TABLE users_y2024m01 PARTITION OF users
    FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

CREATE TABLE users_y2024m02 PARTITION OF users
    FOR VALUES FROM ('2024-02-01') TO ('2024-03-01');

-- Default partition for future dates
CREATE TABLE users_default PARTITION OF users DEFAULT;

-- =============================================
-- CONTENT TABLES WITH SHARDING STRATEGY
-- =============================================

-- Projects table with tenant-based partitioning
CREATE TABLE projects (
    project_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL,
    title VARCHAR(500) NOT NULL,
    description TEXT,
    project_type VARCHAR(50) NOT NULL CHECK (project_type IN ('web_app', 'mobile', 'api', 'data_pipeline', 'ai_model')),
    
    -- Metadata
    tags TEXT[] DEFAULT '{}',
    category VARCHAR(100),
    complexity_level INTEGER CHECK (complexity_level BETWEEN 1 AND 5),
    
    -- Statistics (denormalized for performance)
    view_count BIGINT DEFAULT 0,
    like_count BIGINT DEFAULT 0,
    fork_count BIGINT DEFAULT 0,
    comment_count BIGINT DEFAULT 0,
    
    -- Versioning
    version INTEGER DEFAULT 1,
    parent_project_id UUID,
    is_latest_version BOOLEAN DEFAULT TRUE,
    
    -- Content (large objects in separate table)
    content_hash VARCHAR(64),
    
    -- Timestamps
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    published_at TIMESTAMPTZ,
    
    -- Status
    visibility VARCHAR(20) DEFAULT 'private' CHECK (visibility IN ('private', 'public', 'unlisted', 'team')),
    status VARCHAR(20) DEFAULT 'draft' CHECK (status IN ('draft', 'published', 'archived', 'deleted')),
    
    -- Index-only fields
    title_vector TSVECTOR GENERATED ALWAYS AS (to_tsvector('english', title)) STORED,
    word_count INTEGER GENERATED ALWAYS AS (array_length(regexp_split_to_array(description, '\s+'), 1)) STORED
) PARTITION BY HASH (user_id);

-- Create 8 hash partitions for projects
CREATE TABLE projects_p0 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 0);
CREATE TABLE projects_p1 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 1);
CREATE TABLE projects_p2 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 2);
CREATE TABLE projects_p3 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 3);
CREATE TABLE projects_p4 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 4);
CREATE TABLE projects_p5 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 5);
CREATE TABLE projects_p6 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 6);
CREATE TABLE projects_p7 PARTITION OF projects FOR VALUES WITH (MODULUS 8, REMAINDER 7);

-- =============================================
-- ANALYTICS TABLES WITH TIME-SERIES OPTIMIZATION
-- =============================================

-- User activity events (time-series optimized)
CREATE TABLE user_events (
    event_id BIGSERIAL,
    user_id UUID NOT NULL,
    event_type VARCHAR(100) NOT NULL,
    event_source VARCHAR(50) CHECK (event_source IN ('web', 'mobile', 'api', 'cli')),
    
    -- Event data (structured JSON)
    event_data JSONB NOT NULL DEFAULT '{}',
    session_id UUID,
    
    -- Device/context info
    user_agent TEXT,
    ip_address INET,
    country_code CHAR(2),
    
    -- Performance metrics
    response_time_ms INTEGER,
    error_code VARCHAR(50),
    
    -- Timestamp with microsecond precision
    event_time TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    ingested_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    
    -- Partition key
    event_date DATE NOT NULL DEFAULT CURRENT_DATE,
    
    PRIMARY KEY (event_date, event_id)
) PARTITION BY RANGE (event_date);

-- Create daily partitions for events (rolling 90 days)
DO $$
DECLARE
    start_date DATE := CURRENT_DATE - INTERVAL '90 days';
    end_date DATE := CURRENT_DATE + INTERVAL '30 days';
    current_date_var DATE := start_date;
BEGIN
    WHILE current_date_var <= end_date LOOP
        EXECUTE format(
            'CREATE TABLE IF NOT EXISTS user_events_%s PARTITION OF user_events '
            'FOR VALUES FROM (%L) TO (%L)',
            to_char(current_date_var, 'YYYYMMDD'),
            current_date_var,
            current_date_var + INTERVAL '1 day'
        );
        current_date_var := current_date_var + INTERVAL '1 day';
    END LOOP;
END $$;

-- =============================================
-- RELATIONSHIP TABLES (MANY-TO-MANY)
-- =============================================

-- Project collaborations
CREATE TABLE project_collaborators (
    project_id UUID NOT NULL,
    user_id UUID NOT NULL,

-- User activity summary (refreshed hourly)
CREATE MATERIALIZED VIEW user_activity_summary AS
SELECT 
    user_id,
    DATE(event_time) AS activity_date,
    COUNT(*) AS total_events,
    COUNT(DISTINCT event_type) AS unique_event_types,
    COUNT(DISTINCT session_id) AS sessions_count,
    SUM(CASE WHEN event_type LIKE 'project_%' THEN 1 ELSE 0 END) AS project_events,
    SUM(CASE WHEN event_type LIKE 'search_%' THEN 1 ELSE 0 END) AS search_events,
    MIN(event_time) AS first_event_time,
    MAX(event_time) AS last_event_time,
    AVG(response_time_ms) AS avg_response_time
FROM user_events
WHERE event_time >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY user_id, DATE(event_time)
WITH DATA;

-- Project popularity ranking (refreshed every 6 hours)
CREATE MATERIALIZED VIEW project_popularity AS
SELECT 
    p.project_id,
    p.title,
    p.user_id,
    p.project_type,
    p.view_count,
    p.like_count,
    p.fork_count,
    p.comment_count,
    p.created_at,
    p.updated_at,
    
    -- Engagement score (weighted formula)
    (p.view_count * 0.1 + 
     p.like_count * 0.3 + 
     p.fork_count * 0.5 + 
     p.comment_count * 0.2) AS engagement_score,
    
    -- Recent activity (last 7 days)
    COALESCE(ue.recent_views, 0) AS recent_views,
    COALESCE(ue.recent_likes, 0) AS recent_likes,
    
    -- Trending score (acceleration)
    CASE 
        WHEN p.created_at >= NOW() - INTERVAL '7 days' THEN 2.0
        WHEN p.updated_at >= NOW() - INTERVAL '7 days' THEN 1.5
        ELSE 1.0
    END AS trending_multiplier
    
FROM projects p
LEFT JOIN LATERAL (
    SELECT 
        COUNT(CASE WHEN event_type = 'project_view' THEN 1 END) AS recent_views,
        COUNT(CASE WHEN event_type = 'project_like' THEN 1 END) AS recent_likes
    FROM user_events ue
    WHERE ue.event_data->>'project_id' = p.project_id::text
      AND ue.event_time >= NOW() - INTERVAL '7 days'
) ue ON TRUE
WHERE p.visibility = 'public' 
  AND p.status = 'published'
WITH DATA;

# DIYAPP Evolution - V12 Core - Análise de Segurança e Implementação

## 1. Análise de Segurança da V11

### Vulnerabilidades Identificadas (STRIDE Framework)

**Spoofing (S):**
- Autenticação baseada apenas em sessão sem tokens JWT
- Sem verificação de origem de requisições (CORS permissivo)
- Sem MFA para contas administrativas

**Tampering (T):**
- Inputs de usuário não sanitizados adequadamente
- Ausência de validação de schema em APIs
- Cookies sem flag HttpOnly e Secure

**Repudiation (R):**
- Logs de auditoria insuficientes para operações críticas
- Sem rastreamento de mudanças em dados sensíveis
- Timestamps não sincronizados

**Information Disclosure (I):**
- Headers de resposta expõem versões de software
- Mensagens de erro detalhadas em produção
- Dados sensíveis em logs não mascarados

**Denial of Service (D):**
- Sem rate limiting em endpoints críticos
- Sem circuit breakers para integrações externas
- Uploads de arquivos sem limites de tamanho

**Elevation of Privilege (E):**
- Autorização baseada apenas em roles, não em recursos
- Sem verificação de permissão por recurso (IDOR)

```javascript
const jwt = require('jsonwebtoken');
const crypto = require('crypto');
const { promisify } = require('util');

class JWTSecurity {
  constructor() {
    this.accessTokenSecret = process.env.JWT_ACCESS_SECRET || crypto.randomBytes(32).toString('hex');
    this.refreshTokenSecret = process.env.JWT_REFRESH_SECRET || crypto.randomBytes(32).toString('hex');
    this.accessTokenExpiry = '15m'; // 15 minutos
    this.refreshTokenExpiry = '7d'; // 7 dias
    this.algorithm = 'HS256';
    
    // Blacklist para tokens invalidados
    this.tokenBlacklist = new Set();
  }

  /**
   * Gera par de tokens (access + refresh)
   */
  async generateTokenPair(user) {
    const accessToken = await this.generateAccessToken(user);
    const refreshToken = await this.generateRefreshToken(user);
    
    return {
      accessToken,
      refreshToken,
      expiresIn: 15 * 60 // 15 minutos em segundos
    };
  }

  /**
   * Gera access token de curta duração
   */
  async generateAccessToken(user) {
    const payload = {
      sub: user.id,
      email: user.email,
      roles: user.roles,
      permissions: user.permissions,
      type: 'access'
    };

    return jwt.sign(payload, this.accessTokenSecret, {
      expiresIn: this.accessTokenExpiry,
      algorithm: this.algorithm,
      issuer: 'diyapp-v12',
      audience: 'diyapp-users'
    });
  }

  /**
   * Gera refresh token de longa duração
   */
  async generateRefreshToken(user) {
    const payload = {
      sub: user.id,
      type: 'refresh',
      jti: crypto.randomBytes(16).toString('hex') // ID único do token
    };

    return jwt.sign(payload, this.refreshTokenSecret, {
      expiresIn: this.refreshTokenExpiry,
      algorithm: this.algorithm,
      issuer: 'diyapp-v12',
      audience: 'diyapp-users'
    });
  }

  /**
   * Verifica access token
   */
  async verifyAccessToken(token) {
    if (this.tokenBlacklist.has(token)) {
      throw new Error('Token revoked');
    }

    try {
      const decoded = await promisify(jwt.verify)(token, this.accessTokenSecret, {
        algorithms: [this.algorithm],
        issuer: 'diyapp-v12',
        audience: 'diyapp-users'
      });

      if (decoded.type !== 'access') {
        throw new Error('Invalid token type');
      }

      return decoded;
    } catch (error) {
      if (error.name === 'TokenExpiredError') {
        throw new Error('Token expired');
      }
      throw new Error('Invalid token');
    }
  }

  /**
   * Verifica refresh token
   */
  async verifyRefreshToken(token) {
    try {
      const decoded = await promisify(jwt.verify)(token, this.refreshTokenSecret, {
        algorithms: [this.algorithm],
        issuer: 'diyapp-v12',
        audience: 'diyapp-users'
      });

      if (decoded.type !== 'refresh') {
        throw new Error('Invalid token type');
      }

      return decoded;
    } catch (error) {
      throw new Error('Invalid refresh token');
    }
  }

  /**
   * Revoga token (logout)
   */
  revokeToken(token) {
    this.tokenBlacklist.add(token);
    
    // Limpa tokens expirados da blacklist periodicamente
    setTimeout(() => {
      this.tokenBlacklist.delete(token);
    }, 15 * 60 * 1000); // 15 minutos
  }

  /**
   * Refresh token rotation
   */
  async refreshTokens(refreshToken) {
    const decoded = await this.verifyRefreshToken(refreshToken);
    
    // Busca usuário do banco
    const user = await this.getUserById(decoded.sub);
    
    if (!user) {
      throw new Error('User not found');
    }

    // Invalida o refresh token antigo
    this.revokeToken(refreshToken);

    // Gera novo par de tokens
    return this.generateTokenPair(user);
  }

  async getUserById(userId) {
    // Implementação real buscaria do banco de dados
    return {
      id: userId,
      email: 'user@example.com',
      roles: ['user'],
      permissions: ['read:own', 'write:own']
    };
  }
}

module.exports = new JWTSecurity();

      }
    });
  }

  /**
   * Rate limiting para API geral
   */
  getAPILimiter() {
    return rateLimit({
      windowMs: this.windows.api,
      max: 1000, // 1000 requisições por hora
      message: {
        error: 'Too many requests',
        code: 'RATE_LIMITED',
        retryAfter: '1 hour'
      },
      standardHeaders: true,
      legacyHeaders: false,
      skip: (req) => {
        // Não aplica rate limiting a endpoints de saúde
        return req.path === '/health' || req.path === '/metrics';
      }
    });
  }

  /**
   * Rate limiting estrito para endpoints críticos
   */
  getStrictLimiter() {
    return rateLimit({
      windowMs: this.windows.strict,
      max: 100, // 100 requisições por 5 minutos
      message: {
        error: 'Too many requests to critical endpoint',
        code: 'RATE_LIMITED_STRICT',
        retryAfter: '5 minutes'
      },
      standardHeaders: true,
      legacyHeaders: false
    });
  }

  /**
   * Rate limiting baseado em usuário autenticado
   */
  getUserBasedLimiter() {
    return rateLimit({
      windowMs: this.windows.api,
      max: (req) => {
        // Usuários premium têm limites mais altos
        if (req.user?.roles?.includes('premium')) {
          return 5000;
        }
        return 1000;
      },
      keyGenerator: (req) => {
        return req.user?.id || req.ip;
      },
      message: {
        error: 'Rate limit exceeded for your account',
        code: 'USER_RATE_LIMITED'
      }
    });
  }

  /**
   * Circuit breaker para integrações externas
   */
  createCircuitBreaker(name, options = {}) {
    const defaultOptions = {
      timeout: 10000, // 10 segundos
      errorThresholdPercentage: 50,
      resetTimeout: 30000, // 30 segundos
      rollingCountTimeout: 10000,
      rollingCountBuckets: 10
    };

    const config = { ...defaultOptions, ...options };

    let state = 'CLOSED';
    let failureCount = 0;
    let lastFailureTime = null;
    let nextAttempt = null;

    return async (fn) => {
      const now = Date.now();

      // Verifica se está em estado OPEN
      if (state === 'OPEN') {
        if (now < nextAttempt) {
          throw new Error(`Circuit breaker OPEN for ${name}`);
        }
        
        // Tenta mudar para HALF-OPEN
        state = 'HALF_OPEN';
      }

      try {
        // Executa a função com timeout
        const result = await Promise.race([
          fn(),
          new Promise((_, reject) => {
            setTimeout(() => reject(new Error('Timeout')), config.timeout);
          })
        ]);

        // Sucesso - reseta contador
        if (state === 'HALF_OPEN') {
          state = 'CLOSED';
          failureCount = 0;
          lastFailureTime = null;
        }

        return result;
      } catch (error) {
        failureCount++;
        lastFailureTime = now;

        // Verifica se deve abrir o circuito
        if (failureCount >= config.errorThresholdPercentage) {
          state = 'OPEN';
          nextAttempt = now + config.resetTimeout;
          
          console.error(`Circuit breaker OPEN for ${name}: ${error.message}`);
        }

        throw error;
      }
    };
  }
}

        """Configura logging estruturado com structlog"""
        structlog.configure(
            processors=[
                structlog.stdlib.filter_by_level,
                structlog.stdlib.add_logger_name,
                structlog.stdlib.add_log_level,
                structlog.stdlib.PositionalArgumentsFormatter(),
                TimeStamper(fmt="iso"),
                structlog.processors.StackInfoRenderer(),
                structlog.processors.format_exc_info,
                structlog.processors.UnicodeDecoder(),

                self._enrich_with_context,

            ],
            context_class=dict,
            logger_factory=structlog.stdlib.LoggerFactory(),
            wrapper_class=structlog.stdlib.BoundLogger,
            cache_logger_on_first_use=True,
        )
    
    def _add_correlation_id(self, logger, method_name, event_dict):
        """Adiciona correlation ID para tracing distribuído"""
        from opentelemetry import trace
        span = trace.get_current_span()
        if span and span.get_span_context().trace_id:
            trace_id = format(span.get_span_context().trace_id, '032x')
            event_dict['correlation_id'] = trace_id
            event_dict['span_id'] = format(span.get_span_context().span_id, '016x')
        return event_dict
    
    def _enrich_with_context(self, logger, method_name, event_dict):
        """Enriquece logs com contexto da aplicação"""
        event_dict['service'] = self.service_name
        event_dict['environment'] = os.getenv("ENVIRONMENT", "production")
        event_dict['hostname'] = os.getenv("HOSTNAME", "unknown")
        event_dict['timestamp'] = datetime.utcnow().isoformat()
        return event_dict
    
    def get_logger(self, name: str = None) -> BoundLogger:
        """Retorna logger configurado"""
        return get_logger(name or self.service_name)
    
    def log_to_elasticsearch(self, log_data: Dict[str, Any]):
        """Envia log para Elasticsearch para análise centralizada"""
        if not self.elasticsearch_client:
            return
        
        try:

            )
        except Exception as e:

            print(f"Failed to send log to Elasticsearch: {e}")
            print(json.dumps(log_data, indent=2))

// tests/e2e/helpers/api.helper.js
export class ApiHelper {
  constructor(page) {
    this.page = page;
  }

  async getAuthToken() {
    const cookies = await this.page.context().cookies();
    const authCookie = cookies.find(c => c.name === 'auth_token');
    return authCookie ? authCookie.value : null;
  }

  async makeApiRequest(endpoint, method = 'GET', body = null) {
    const token = await this.getAuthToken();
    const response = await this.page.evaluate(async ({ endpoint, method, body, token }) => {
      const options = {
        method,
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${token}`
        }
      };
      
      if (body) {
        options.body = JSON.stringify(body);
      }
      
      const response = await fetch(`/api/v1${endpoint}`, options);
      return {
        status: response.status,
        headers: Object.fromEntries(response.headers.entries()),
        body: await response.json()
      };
    }, { endpoint, method, body, token });
    
    return response;
  }
}

"""

class DataFlowMapper:
    """Mapeador de fluxos de dados da aplicação"""
    
    def __init__(self, db_path: str = "compliance.db"):
        self.db_path = db_path
        self._init_database()
        self.activities: Dict[str, DataProcessingActivity] = {}
        self.load_activities()
    
    def _init_database(self):
        """Inicializa banco de dados de compliance"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS ropa_activities (
                id TEXT PRIMARY KEY,
                name TEXT NOT NULL,
                description TEXT,
                controller TEXT NOT NULL,
                data_categories TEXT,
                purposes TEXT,
                legal_basis TEXT,
                data_subjects TEXT,

                security_measures TEXT,

                updated_at TIMESTAMP,
                dpia_required INTEGER,
                dpia_id TEXT,
                status TEXT

        ''')
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS data_inventory (

                data_category TEXT,
                processing_activity_id TEXT,
                source TEXT,
                collection_method TEXT,
                mandatory INTEGER,
                sensitive INTEGER,

                FOREIGN KEY (processing_activity_id) REFERENCES ropa_activities(id)

                status TEXT

        ''')

        conn.close()

        """Registra uma nova atividade de processamento"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        activity.updated_at = datetime.now()
        
        cursor.execute('''
            INSERT OR REPLACE INTO ropa_activities 
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            activity.id,
            activity.name,
            activity.description,

            activity.updated_at.isoformat(),
            1 if activity.dpia_required else 0,
            activity.dpia_id,
            activity.status
        ))
        
        conn.commit()
        conn.close()
        
        self.activities[activity.id] = activity
        logger.info(f"Atividade registrada: {activity.name}")
        return activity.id

        """Adiciona campo de dados ao inventário"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        field_id = hashlib.md5(field_data['field_name'].encode()).hexdigest()[:8]
        
        cursor.execute('''
            INSERT INTO data_inventory 
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            field_id,
            field_data['field_name'],
            field_data.get('data_category'),
            field_data.get('processing_activity_id'),
            field_data.get('source'),
            field_data.get('collection_method'),
            1 if field_data.get('mandatory', False) else 0,
            1 if field_data.get('sensitive', False) else 0,
            datetime.now().isoformat()
        ))
        
        conn.commit()
        conn.close()
        return field_id
    
    def generate_ropa_report(self) -> Dict[str, Any]:
        """Gera relatório ROPA completo"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # Atividades de processamento
        cursor.execute('SELECT COUNT(*) FROM ropa_activities WHERE status = "active"')
        active_count = cursor.fetchone()[0]
        
        cursor.execute('SELECT legal_basis, COUNT(*) FROM ropa_activities GROUP BY legal_basis')
        basis_distribution = dict(cursor.fetchall())
        
        # Inventário de dados
        cursor.execute('''
            SELECT data_category, COUNT(*) 
            FROM data_inventory 
            GROUP BY data_category
        ''')
        data_categories = dict(cursor.fetchall())
        
        # Transferências internacionais
        cursor.execute('SELECT COUNT(*) FROM ropa_activities WHERE third_country_transfers = 1')
        intl_transfers = cursor.fetchone()[0]
        
        conn.close()
        
        return {
            "report_date": datetime.now().isoformat(),
            "active_processing_activities": active_count,
            "legal_basis_distribution": basis_distribution,
            "data_categories_inventory": data_categories,
            "international_transfers": intl_transfers,
            "requires_dpia": sum(1 for a in self.activities.values() if a.dpia_required)
        }
    
    def load_activities(self):
        """Carrega atividades do banco de dados"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute('SELECT * FROM ropa_activities')
        rows = cursor.fetchall()
        
        for row in rows:
            activity = DataProcessingActivity(
                id=row[0],
                name=row[1],
                description=row[2],
                controller=row[3],
                data_categories=[DataCategory(cat) for cat in json.loads(row[4])],
                purposes=json.loads(row[5]),
                legal_basis=LegalBasis(row[6]),
                data_subjects=row[7],
                recipients=json.loads(row[8]),
                third_country_transfers=bool(row[9]),
                retention_period=row[10],
                security_measures=json.loads(row[11]),
                created_at=datetime.fromisoformat(row[12]),
                updated_at=datetime.fromisoformat(row[13]),
                dpia_required=bool(row[14]),
                dpia_id=row[15],
                status=row[16]
            )
            self.activities[activity.id] = activity
        
        conn.close()
    
    def export_ropa_json(self) -> str:
        """Exporta ROPA em formato JSON para auditoria"""
        data = {
            "export_date": datetime.now().isoformat(),
            "version": "1.0",
            "activities": [asdict(activity) for activity in self.activities.values()],
            "metadata": {
                "total_activities": len(self.activities),
                "last_updated": max(a.updated_at for a in self.activities.values()).isoformat()
            }
        }
        
        # Converter enums para strings
        for activity in data["activities"]:
            activity["data_categories"] = [cat.value for cat in activity["data_categories"]]
            activity["legal_basis"] = activity["legal_basis"].value
            activity["created_at"] = activity["created_at"].isoformat()
            activity["updated_at"] = activity["updated_at"].isoformat()
        
        return json.dumps(data, indent=2, ensure_ascii=False)
