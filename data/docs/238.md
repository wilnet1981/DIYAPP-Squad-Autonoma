# DIYAPP Evolution - V12 Core - Arquitetura Técnica

## ADR-001: Arquitetura V12 - Padrões e Estrutura

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

### CONTEXTO:
A V12 do DIYAPP precisa estabelecer uma arquitetura que suporte:
1. 100% de autonomia operacional
2. Refatoração paralela via Modo Hive
3. Monitoramento completo sem intervenção humana
4. Integração fluida entre camadas (IA, backend, frontend)
5. Sustentabilidade técnica a longo prazo

### DECISÃO:
Adotar arquitetura em camadas com separação clara de responsabilidades, padrões de código rigorosos e sistema de observabilidade nativo.

### OPÇÕES CONSIDERADAS:

**Opção A: Microserviços distribuídos**
- Prós: Escalabilidade independente, isolamento de falhas
- Contras: Complexidade de deploy, overhead de comunicação, difícil monitoramento unificado

**Opção B: Monólito modular**
- Prós: Simplicidade de deploy, debugging facilitado, transações ACID fáceis
- Contras: Acoplamento potencial, escalabilidade vertical limitada

**Opção C: Arquitetura em camadas com módulos autônomos (ESCOLHIDA)**
- Prós: Separação de responsabilidades, refatoração paralela possível, observabilidade unificada
- Contras: Requer disciplina de equipe, interface entre módulos bem definidas

**Justificativa:** Para o Modo Hive, precisamos de módulos que possam ser refatorados independentemente mas com interfaces estáveis. A arquitetura em camadas com contratos bem definidos permite isso.

### CONSEQUÊNCIAS:
**Positivas:**
- Refatoração paralela possível sem quebrar o sistema
- Monitoramento unificado de todas as camadas
- Onboarding de novos devs facilitado por padrões claros
- Testabilidade aumentada por separação de responsabilidades

**Negativas:**
- Overhead inicial para definir todos os contratos
- Necessidade de disciplina rigorosa para manter separação
- Complexidade de configuração inicial

**Riscos:**
- Módulos podem desenvolver acoplamento silencioso
- Mitigação: Revisões de PR focadas em dependências entre módulos
- Monitoramento pode gerar volume excessivo de logs
- Mitigação: Sistema de log levels e amostragem inteligente

// src/core/logging/logger.js
const winston = require('winston');
const { ElasticsearchTransport } = require('winston-elasticsearch');
const { v4: uuidv4 } = require('uuid');

class StructuredLogger {
  constructor() {
    this.correlationId = uuidv4();
    this.moduleName = process.env.MODULE_NAME || 'unknown';
    
    this.logger = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.errors({ stack: true }),
        winston.format.json()
      ),
      defaultMeta: {
        service: 'diyapp-v12',
        module: this.moduleName,
        correlationId: this.correlationId,
        version: '12.0.0'
      },
      transports: [
        // Console para desenvolvimento
        new winston.transports.Console({
          format: winston.format.combine(
            winston.format.colorize(),
            winston.format.simple()
          )
        }),
        
        // Arquivo para produção
        new winston.transports.File({ 
          filename: `logs/${this.moduleName}-error.log`, 
          level: 'error' 
        }),
        new winston.transports.File({ 
          filename: `logs/${this.moduleName}-combined.log` 
        }),
        
        // Elasticsearch para monitoramento (se configurado)
        ...(process.env.ELASTICSEARCH_HOST ? [
          new ElasticsearchTransport({
            level: 'info',
            clientOpts: {
              node: process.env.ELASTICSEARCH_HOST,
              auth: {
                username: process.env.ELASTICSEARCH_USER,
                password: process.env.ELASTICSEARCH_PASS
              }
            },
            index: `diyapp-logs-${new Date().toISOString().split('T')[0]}`
          })
        ] : [])
      ]
    });
  }
  
  // Métodos de log estruturados
  info(message, meta = {}) {
    this.logger.info(message, { ...meta, correlationId: this.correlationId });
  }
  
  error(message, error = null, meta = {}) {
    const logMeta = {
      ...meta,
      correlationId: this.correlationId,
      stack: error?.stack,
      errorCode: error?.code,
      errorName: error?.name
    };
    this.logger.error(message, logMeta);
  }
  
  warn(message, meta = {}) {
    this.logger.warn(message, { ...meta, correlationId: this.correlationId });
  }
  
  debug(message, meta = {}) {
    this.logger.debug(message, { ...meta, correlationId: this.correlationId });
  }
  
  // Log de métricas de performance
  metric(name, value, unit = 'ms', meta = {}) {
    this.logger.info(`METRIC: ${name}`, {
      ...meta,
      metricName: name,
      metricValue: value,
      metricUnit: unit,
      correlationId: this.correlationId,
      timestamp: new Date().toISOString()
    });
  }
  
  // Log de transação entre módulos
  transaction(fromModule, toModule, action, duration, status, meta = {}) {
    this.logger.info(`TRANSACTION: ${fromModule} -> ${toModule}`, {
      ...meta,
      transaction: {
        from: fromModule,
        to: toModule,
        action,
        duration,
        status,
        timestamp: new Date().toISOString()
      },
      correlationId: this.correlationId
    });
  }
}

module.exports = StructuredLogger;

        name: 'diyapp_llm_requests_total',

      }),

    data: Dict[str, Any],
    expires_delta: Optional[timedelta] = None,
    settings: Settings = Depends(get_settings)
) -> str:
    """Create JWT access token."""
    to_encode = data.copy()
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(
            minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES
        )

    encoded_jwt = jwt.encode(
        to_encode,
        settings.SECRET_KEY,
        algorithm=settings.ALGORITHM
    )
    return encoded_jwt

class RedisCache:
    """Redis cache wrapper with circuit breaker pattern."""
    
    def __init__(self):
        self._client = None
        self._circuit_open = False
        self._failure_count = 0
        self._circuit_reset_timeout = 60  # seconds
        
    @property
    def client(self) -> Optional[redis.Redis]:
        """Lazy initialization of Redis client with circuit breaker."""
        if self._circuit_open:
            logger.warning("Circuit breaker open - Redis unavailable")
            return None
            
        if self._client is None:
            try:
                self._client = redis.Redis(
                    host=settings.REDIS_HOST,
                    port=settings.REDIS_PORT,
                    password=settings.REDIS_PASSWORD,
                    db=settings.REDIS_DB,
                    socket_timeout=5,  # 5 second timeout
                    socket_connect_timeout=5,
                    retry_on_timeout=True,
                    decode_responses=True
                )
                # Test connection
                self._client.ping()
                self._failure_count = 0
                logger.info("Redis connection established")
            except Exception as e:
                self._failure_count += 1
                logger.error("Redis connection failed", error=str(e), failure_count=self._failure_count)
                
                if self._failure_count >= 3:
                    self._circuit_open = True
                    logger.critical("Circuit breaker opened - Redis failures exceeded threshold")
                    # Schedule circuit reset
                    import asyncio
                    asyncio.create_task(self._reset_circuit())
                
                return None
        
        return self._client
    
    async def _reset_circuit(self):
        """Reset circuit breaker after timeout."""
        import asyncio
        await asyncio.sleep(self._circuit_reset_timeout)
        self._circuit_open = False
        self._failure_count = 0
        logger.info("Circuit breaker reset")
    
    def get(self, key: str) -> Optional[Any]:
        """Get value from cache."""
        client = self.client
        if not client:
            return None
            
        try:
            value = client.get(key)
            if value:
                try:
                    return json.loads(value)
                except:
                    return pickle.loads(value)
        except Exception as e:
            logger.error("Redis get failed", key=key, error=str(e))
            return None
    
    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Set value in cache."""
        client = self.client
        if not client:
            return False
            
        try:
            if isinstance(value, (dict, list, tuple, int, float, str, bool, type(None))):
                serialized = json.dumps(value)
            else:
                serialized = pickle.dumps(value)
            
            if ttl:
                client.setex(key, ttl, serialized)
            else:
                client.set(key, serialized)
            return True
        except Exception as e:
            logger.error("Redis set failed", key=key, error=str(e))
            return False
    
    def delete(self, key: str) -> bool:
        """Delete key from cache."""
        client = self.client
        if not client:
            return False
            
        try:
            client.delete(key)
            return True
        except Exception as e:
            logger.error("Redis delete failed", key=key, error=str(e))
            return False
    
    def increment(self, key: str, amount: int = 1) -> Optional[int]:
        """Increment counter in cache."""
        client = self.client
        if not client:
            return None
            
        try:
            return client.incrby(key, amount)
        except Exception as e:
            logger.error("Redis increment failed", key=key, error=str(e))
            return None
    
    def get_or_set(self, key: str, func, ttl: int = 300) -> Any:
        """Get from cache or set using function."""
        cached = self.get(key)
        if cached is not None:
            return cached
        
        value = func()
        self.set(key, value, ttl)
        return value


# Cache decorator with timeout and fallback
def cached(key_prefix: str, ttl: int = 300):
    """Decorator for caching function results."""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            cache = RedisCache()
            
            # Generate cache key
            cache_key = f"{key_prefix}:{str(args)}:{str(kwargs)}"
            
            # Try to get from cache
            cached_result = cache.get(cache_key)
            if cached_result is not None:
                logger.debug("Cache hit", key=cache_key)
                return cached_result
            
            # Cache miss - execute function
            logger.debug("Cache miss", key=cache_key)
            result = await func(*args, **kwargs)
            
            # Store in cache
            cache.set(cache_key, result, ttl)
            
            return result
        return wrapper
    return decorator

@contextmanager
def get_db() -> Generator[Session, None, None]:
    """Database session context manager with error handling."""
    db = SessionLocal()
    try:
        yield db
        db.commit()
    except SQLAlchemyError as e:
        db.rollback()
        logger.error("Database error", error=str(e), error_type=type(e).__name__)
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail="Database error occurred"
        )
    finally:
        db.close()

import { 
  persistStore,

  PERSIST,
  PURGE,
  REGISTER,

import { setupListeners } from '@reduxjs/toolkit/query';

// Slices
import authReducer from './slices/authSlice';
import uiReducer from './slices/uiSlice';
import appReducer from './slices/appSlice';

// API Services
import { api } from '../services/api';

// Config persist
const persistConfig = {
  key: 'root',
  version: 1,
  storage,
  whitelist: ['auth', 'ui'], // O que persistir
  blacklist: [api.reducerPath], // O que não persistir
};

const rootReducer = combineReducers({
  auth: authReducer,
  ui: uiReducer,
  app: appReducer,
  [api.reducerPath]: api.reducer,
});

const persistedReducer = persistReducer(persistConfig, rootReducer);

export const store = configureStore({
  reducer: persistedReducer,
  middleware: (getDefaultMiddleware) =>
    getDefaultMiddleware({
      serializableCheck: {
        ignoredActions: [FLUSH, REHYDRATE, PAUSE, PERSIST, PURGE, REGISTER],
      },
    }).concat(api.middleware),
  devTools: process.env.NODE_ENV !== 'production',
});

export const persistor = persistStore(store);

// Setup listeners para refetch on focus/reconnect
setupListeners(store.dispatch);

// Tipos
export type RootState = ReturnType<typeof store.getState>;
export type AppDispatch = typeof store.dispatch;

## 6. Slice de Autenticação (src/store/slices/authSlice.ts)

```typescript
import { createSlice, PayloadAction } from '@reduxjs/toolkit';

export interface User {
  id: string;
  email: string;
  name: string;
  avatar?: string;
  role: 'user' | 'admin' | 'moderator';
}

export interface AuthState {
  user: User | null;
  token: string | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  error: string | null;
}

const initialState: AuthState = {
  user: null,
  token: null,
  isAuthenticated: false,
  isLoading: false,
  error: null,
};

const authSlice = createSlice({
  name: 'auth',
  initialState,
  reducers: {
    setCredentials: (
      state,
      action: PayloadAction<{ user: User; token: string }>
    ) => {
      state.user = action.payload.user;
      state.token = action.payload.token;
      state.isAuthenticated = true;
      state.error = null;
    },
    clearCredentials: (state) => {
      state.user = null;
      state.token = null;
      state.isAuthenticated = false;
      state.error = null;
    },
    setLoading: (state, action: PayloadAction<boolean>) => {
      state.isLoading = action.payload;
    },
    setError: (state, action: PayloadAction<string | null>) => {
      state.error = action.payload;
    },
    updateUser: (state, action: PayloadAction<Partial<User>>) => {
      if (state.user) {
        state.user = { ...state.user, ...action.payload };
      }
    },
  },
});

export const {
  setCredentials,
  clearCredentials,
  setLoading,
  setError,
  updateUser,
} = authSlice.actions;

export default authSlice.reducer;

// Selectors
export const selectCurrentUser = (state: { auth: AuthState }) => state.auth.user;
export const selectIsAuthenticated = (state: { auth: AuthState }) => 
  state.auth.isAuthenticated;
export const selectAuthLoading = (state: { auth: AuthState }) => 
  state.auth.isLoading;
export const selectAuthError = (state: { auth: AuthState }) => 
  state.auth.error;
```

-- ==================== FUNCTIONS E PROCEDURES ====================

-- Função para limpeza automática de dados antigos
CREATE OR REPLACE FUNCTION telemetry.cleanup_old_data(retention_days INTEGER)
RETURNS INTEGER AS $$
DECLARE
    deleted_count INTEGER;
BEGIN
    -- Remove dados raw com mais de X dias (mantém staging e marts)
    DELETE FROM telemetry.raw_events 
    WHERE event_timestamp < NOW() - (retention_days || ' days')::INTERVAL;
    
    GET DIAGNOSTICS deleted_count = ROW_COUNT;
    
    -- Log da operação
    INSERT INTO mart_ops.data_retention_log 
    VALUES (NOW(), 'raw_events', retention_days, deleted_count);
    
    RETURN deleted_count;
END;
$$ LANGUAGE plpgsql;

-- Procedure para atualização diária de métricas
CREATE OR REPLACE PROCEDURE analytics.refresh_daily_metrics(target_date DATE)
LANGUAGE plpgsql
AS $$
BEGIN
    -- Atualiza DAU
    INSERT INTO mart_product.daily_active_users
    SELECT 
        target_date,
        COUNT(DISTINCT user_id),
        COUNT(DISTINCT CASE WHEN is_first_day THEN user_id END),
        COUNT(DISTINCT CASE WHEN NOT is_first_day THEN user_id END),
        -- ... cálculos completos
    FROM staging.stg_events
    WHERE event_date = target_date
    ON CONFLICT (date) DO UPDATE SET
        total_users = EXCLUDED.total_users,
        updated_at = NOW();
    
    -- Atualiza custo LLM
    INSERT INTO mart_ai.llm_cost_by_feature
    SELECT 
        target_date,
        feature_name,
        model_provider,
        model_name,
        COUNT(*) as total_calls,
        SUM(input_tokens) as total_input_tokens,
        -- ... cálculos completos
    FROM staging.stg_llm_calls
    WHERE request_date = target_date
    GROUP BY feature_name, model_provider, model_name
    ON CONFLICT (date, feature_name, model_provider, model_name) DO UPDATE SET
        total_cost_usd = EXCLUDED.total_cost_usd,
        updated_at = NOW();
    
    COMMIT;
END;
$$;

-- ==================== TRIGGERS ====================

-- Trigger para atualizar updated_at
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Aplica trigger às tabelas de staging
CREATE TRIGGER update_stg_events_updated_at
    BEFORE UPDATE ON staging.stg_events
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_stg_llm_calls_updated_at
    BEFORE UPDATE ON staging.stg_llm_calls
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();
