client.connect();

const cacheMiddleware = async (req, res, next) => {
  const key = req.originalUrl;
  const cachedData = await client.get(key);
  if (cachedData) {
    res.send(JSON.parse(cachedData));

import pandas as pd
from sklearn.linear_model import LinearRegression
import numpy as np

class AutoScaler:
    def __init__(self):
        self.model = LinearRegression()

    def train(self, data):
        # Expecting data with ['timestamp', 'load']
        data['timestamp'] = pd.to_datetime(data['timestamp']).astype(int) / 10**9
        X = data[['timestamp']]
        y = data['load']
        self.model.fit(X, y)

    def predict_scale(self, future_timestamps):
        future_timestamps = pd.to_datetime(future_timestamps).astype(int) / 10**9
        X_future = np.array(future_timestamps).reshape(-1, 1)

from sklearn.ensemble import IsolationForest

class AnomalyDetector:
    def __init__(self):
        self.model = IsolationForest(contamination=0.1)

    def train(self, data):
        # Expecting data with ['timestamp', 'metric']
        X = data[['metric']]
        self.model.fit(X)

    def detect(self, data):
        # Expecting data with ['timestamp', 'metric']
        X = data[['metric']]
        predictions = self.model.predict(X)
        anomalies = data[predictions == -1]
        return anomalies

### Criptografia em Repouso e Trânsito

#### Em Repouso

Utilizando AES para criptografar dados sensíveis:

```javascript
const crypto = require('crypto');

const algorithm = 'aes-256-cbc';
const key = crypto.randomBytes(32);
const iv = crypto.randomBytes(16);

function encrypt(text) {
  let cipher = crypto.createCipheriv(algorithm, Buffer.from(key), iv);
  let encrypted = cipher.update(text);
  encrypted = Buffer.concat([encrypted, cipher.final()]);
  return iv.toString('hex') + ':' + encrypted.toString('hex');
}

function decrypt(text) {
  let textParts = text.split(':');
  let iv = Buffer.from(textParts.shift(), 'hex');
  let encryptedText = Buffer.from(textParts.join(':'), 'hex');
  let decipher = crypto.createDecipheriv(algorithm, Buffer.from(key), iv);
  let decrypted = decipher.update(encryptedText);
  decrypted = Buffer.concat([decrypted, decipher.final()]);
  return decrypted.toString();
}
```

exports.giveConsent = async (req, res) => {

  try {
    const consent = new UserConsent({ userId, consentGiven: true });
    await consent.save();
    res.status(201).json({ message: 'Consent given successfully', consent });

    res.status(500).json({ message: 'Error giving consent', error });
  }

exports.revokeConsent = async (req, res) => {
  const { userId } = req.body;
  try {
    const result = await UserConsent.findOneAndUpdate({ userId }, { consentGiven: false }, { new: true });
    if (result) {
      res.status(200).json({ message: 'Consent revoked successfully', result });
    } else {
      res.status(404).json({ message: 'User not found' });
    }
  } catch (error) {
    res.status(500).json({ message: 'Error revoking consent', error });
  }
};

##### 3. `src/utils/mathUtils.js`

```javascript
function calculateLinearRegression(data) {
    const n = data.length;
    const sumX = data.reduce((acc, point) => acc + point.x, 0);
    const sumY = data.reduce((acc, point) => acc + point.y, 0);
    const sumXY = data.reduce((acc, point) => acc + point.x * point.y, 0);
    const sumX2 = data.reduce((acc, point) => acc + point.x * point.x, 0);

    const slope = (n * sumXY - sumX * sumY) / (n * sumX2 - sumX * sumX);
    const intercept = (sumY - slope * sumX) / n;

    return { slope, intercept };
}

function predict(data, xValue) {
    const { slope, intercept } = calculateLinearRegression(data);
    return (slope * xValue) + intercept;
}

export { calculateLinearRegression, predict };
```
