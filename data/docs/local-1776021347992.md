## 1. Documento de Arquitetura V12

### 1.1 Visão Geral
```
DIYAPP V12 adota uma arquitetura de microsserviços baseada em eventos com 
autonomia total por squad. Cada microsserviço é propriedade de uma squad 
específica com controle completo sobre seu ciclo de vida.

### 1.4 Padrões de Comunicação
```typescript
// communication-standards.ts
export interface ServiceContract {
  version: string;
  endpoints: EndpointDefinition[];
  events: EventDefinition[];
  fallback_strategy: FallbackStrategy;
}

export interface EndpointDefinition {
  method: 'GET' | 'POST' | 'PUT' | 'DELETE';
  path: string;
  request_schema: object;
  response_schema: object;
  timeout_ms: number;
  circuit_breaker: {
    failure_threshold: number;
    reset_timeout: number;
  };
}

export interface EventDefinition {
  name: string;
  schema: object;
  version: string;
  guaranteed_delivery: boolean;
}

// Padrão de implementação REST
export class RESTClient {
  private readonly circuitBreaker: CircuitBreaker;
  
  constructor(private serviceName: string) {
    this.circuitBreaker = new CircuitBreaker({
      failureThreshold: 5,
      resetTimeout: 30000
    });
  }
  
  async call<T>(endpoint: string, data?: any): Promise<T> {
    return this.circuitBreaker.execute(async () => {
      const response = await fetch(`${this.getServiceUrl()}/${endpoint}`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Request-ID': generateRequestId(),
          'X-Service-Name': this.serviceName
        },
        body: JSON.stringify(data),
        timeout: 5000 // 5s timeout padrão
      });
      
      if (!response.ok) {
        throw new ServiceError(this.serviceName, endpoint, response.status);
      }
      
      return response.json();
    });
  }
}

```typescript
// error-handling.ts
export class AppError extends Error {
  constructor(
    public code: string,
    message: string,
    public statusCode: number = 500,
    public context?: any
  ) {
    super(message);
  }
}

export class ValidationError extends AppError {
  constructor(message: string, validationErrors: any[]) {
    super('VALIDATION_ERROR', message, 400, { validationErrors });
  }
}

// Middleware de tratamento de erros
app.use((err: Error, req: Request, res: Response, next: NextFunction) => {
  if (err instanceof AppError) {
    res.status(err.statusCode).json({
      error: {
        code: err.code,
        message: err.message,
        ...(process.env.NODE_ENV === 'development' && { stack: err.stack }),
        ...(err.context && { context: err.context })
      }
    });
  } else {
    // Erro não esperado
    logger.error('Unhandled error', {
      error: err.message,
      stack: err.stack,
      path: req.path,
      method: req.method
    });
    
    res.status(500).json({
      error: {
        code: 'INTERNAL_ERROR',
        message: 'Internal server error'
      }
    });
  }
});

      - name: Lint
        run: npm run lint
        
      - name: Type check
        run: npm run type-check
        
      - name: Unit tests
        run: npm test -- --coverage
        
      - name: Check coverage
        run: |
          COVERAGE=$(node -p "require('./coverage/coverage-summary.json').total.lines.pct")
          if (( $(echo "$COVERAGE < 80" | bc -l) )); then
            echo "Coverage $COVERAGE% is below 80% threshold"
            exit 1
          fi
          
      - name: Security scan
        uses: snyk/actions/node@master
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
          
      - name: Build Docker image
        run: docker build -t ${{ secrets.DOCKER_REGISTRY }}/${{ github.event.repository.name }}:${{ github.sha }} .
        
      - name: Integration tests
        run: docker-compose -f docker-compose.test.yml up --exit-code-from tests
        
      - name: Performance tests
        run: k6 run --out influxdb=http://localhost:8086/k6 tests/performance.js
        
      - name: Deploy to Staging
        if: github.ref == 'refs/heads/main'
        run: |
          echo "Deploying to staging..."
          # Squad-specific deployment logic
          ./deploy.sh staging
          
      - name: Canary Deployment
        if: github.ref == 'refs/heads/main'
        run: |
          # 10% traffic initially
          kubectl apply -f k8s/canary.yaml
          sleep 300 # Wait 5 minutes
          
          # Check metrics
          if ./check-metrics.sh p95 < 200; then
            # Roll out to 100%
            kubectl apply -f k8s/full-deployment.yaml
          else:
            # Rollback
            kubectl rollout undo deployment/${{ github.event.repository.name }}
            
      - name: Notify Squad
        uses: 8398a7/action-slack@v3
        with:
          status: ${{ job.status }}
          channel: '#squad-alerts'

### 1.9 Script de Deploy Autônomo por Squad
```bash
#!/bin/bash
# deploy.sh - Squad Autonomous Deployment

set -e

SERVICE_NAME=$(basename $(pwd))
SQUAD_NAME="${SERVICE_NAME%-service}"
ENVIRONMENT=${1:-staging}
VERSION=$(git describe --tags --always)

echo "🚀 Deploying $SERVICE_NAME ($VERSION) to $ENVIRONMENT"

# Validar pré-requisitos
validate_prerequisites() {
    echo "🔍 Validating prerequisites..."
    
    # Testes passaram
    if [ ! -f "coverage/coverage-summary.json" ]; then
        echo "❌ Coverage report not found"
        exit 1
    fi
    
    COVERAGE=$(node -p "require('./coverage/coverage-summary.json').total.lines.pct")
    if (( $(echo "$COVERAGE < 80" | bc -l) )); then
        echo "❌ Coverage $COVERAGE% is below 80% threshold"
        exit 1
    fi
    
    # Security scan
    if [ -f "snyk-report.json" ]; then
        VULNERABILITIES=$(jq '.vulnerabilities | length' snyk-report.json)
        if [ "$VULNERABILITIES" -gt 0 ]; then
            echo "❌ $VULNERABILITIES vulnerabilities found"
            exit 1
        fi
    fi
    
    echo "✅ Prerequisites validated"
}

# Build e push da imagem
build_and_push() {
    echo "🏗️ Building Docker image..."
    
    docker build \
        --build-arg BUILD_DATE=$(date -u +'%Y-%m-%dT%H:%M:%SZ') \
        --build-arg VERSION=$VERSION \
        --build-arg SQUAD=$SQUAD_NAME \
        -t $DOCKER_REGISTRY/$SERVICE_NAME:$VERSION \
        -t $DOCKER_REGISTRY/$SERVICE_NAME:latest \
        .
        
    echo "📤 Pushing to registry..."
    docker push $DOCKER_REGISTRY/$SERVICE_NAME:$VERSION
    docker push $DOCKER_REGISTRY/$SERVICE_NAME:latest
    
    echo "✅ Image pushed"
}

# Deploy no Kubernetes
deploy_to_kubernetes() {
    echo "🎯 Deploying to Kubernetes..."
    
    # Atualizar deployment
    kubectl set image deployment/$SERVICE_NAME \
        $SERVICE_NAME=$DOCKER_REGISTRY/$SERVICE_NAME:$VERSION \
        -n diyapp-$ENVIRONMENT
        
    # Aguardar rollout
    echo "⏳ Waiting for rollout..."

```typescript
// src/design-system/theme/ThemeProvider.tsx
import React, { createContext, useContext, ReactNode } from 'react';
import { tokens, DesignTokens } from '../tokens';

interface ThemeContextType {
  tokens: DesignTokens;
  mode: 'light' | 'dark';
  toggleMode: () => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

interface ThemeProviderProps {
  children: ReactNode;
  defaultMode?: 'light' | 'dark';
}

export const ThemeProvider: React.FC<ThemeProviderProps> = ({ 
  children, 
  defaultMode = 'light' 
}) => {
  const [mode, setMode] = React.useState<'light' | 'dark'>(defaultMode);

  const toggleMode = () => {
    setMode(prev => prev === 'light' ? 'dark' : 'light');
  };

  const value = {
    tokens,
    mode,
    toggleMode,
  };

  return (
    <ThemeContext.Provider value={value}>
      <div data-theme={mode} style={{ height: '100%' }}>
        {children}
      </div>
    </ThemeContext.Provider>
  );
};

export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within a ThemeProvider');
  }
  return context;
};

# DIYAPP V12 - Arquitetura de Microsserviços

## Visão Geral
Arquitetura baseada em microsserviços para o DIYAPP V12, projetada para 100% de autonomia, escalabilidade horizontal e estabilidade de produção.

## Princípios Arquiteturais
1. **Autonomia por Domínio**: Cada microsserviço possui seu próprio banco de dados e é responsável por um domínio específico
2. **Resiliência**: Circuit breakers, retry policies e fallbacks em todos os serviços
3. **Observabilidade**: Logs estruturados, métricas e tracing distribuído
4. **Segurança por Design**: Autenticação centralizada, autorização descentralizada
5. **Deploy Independente**: Cada serviço pode ser deployado sem afetar outros

DO $$
DECLARE
    month_start DATE;
    month_end DATE;
    partition_name TEXT;
BEGIN
    FOR i IN 0..11 LOOP
        month_start := DATE_TRUNC('month', CURRENT_DATE + (i || ' months')::INTERVAL);
        month_end := DATE_TRUNC('month', CURRENT_DATE + ((i + 1) || ' months')::INTERVAL);
        partition_name := 'ai_sessions_' || TO_CHAR(month_start, 'YYYY_MM');
        
        EXECUTE format('
            CREATE TABLE IF NOT EXISTS %I 
            PARTITION OF ai_sessions
            FOR VALUES FROM (%L) TO (%L)',
            partition_name, month_start, month_end
        );
    END LOOP;
END $$;

DECLARE
    day_date DATE;
    next_date DATE;
    partition_name TEXT;
BEGIN
    FOR i IN 0..29 LOOP
        day_date := CURRENT_DATE + i;
        next_date := day_date + 1;
        partition_name := 'events_' || TO_CHAR(day_date, 'YYYY_MM_DD');
        
        EXECUTE format('
            CREATE TABLE IF NOT EXISTS %I 
            PARTITION OF events
            FOR VALUES FROM (%L) TO (%L)',
            partition_name, day_date, next_date
        );
    END LOOP;
END $$;

  static async generateSecurePassword(password) {
    const saltRounds = SecurityPolicyV12.EncryptionPolicy.atRest.saltRounds;
    return await bcrypt.hash(password, saltRounds);
  }

  static async verifyPassword(password, hash) {
    return await bcrypt.compare(password, hash);
  }

  static generateJWT(payload, type = 'access') {
    const config = SecurityPolicyV12.AuthPolicy.jwt;
    const options = {
      issuer: config.validation.issuer,
      audience: config.validation.audience,
      expiresIn: config[type === 'access' ? 'accessToken' : 'refreshToken'].expiresIn
    };
    
    return jwt.sign(
      payload,
      config[type === 'access' ? 'accessToken' : 'refreshToken'].secret,
      options
    );
  }

  static verifyJWT(token, type = 'access') {
    const config = SecurityPolicyV12.AuthPolicy.jwt;
    
    return jwt.verify(token, config[type === 'access' ? 'accessToken' : 'refreshToken'].secret, {
      issuer: config.validation.issuer,
      audience: config.validation.audience,
      clockTolerance: config.validation.clockTolerance
    });
  }

  static encryptData(text, key) {
    const iv = crypto.randomBytes(SecurityPolicyV12.EncryptionPolicy.atRest.ivLength);
    const cipher = crypto.createCipheriv(
      SecurityPolicyV12.EncryptionPolicy.atRest.algorithm,
      key,
      iv
    );
    
    let encrypted = cipher.update(text, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    return {
      iv: iv.toString('hex'),
      content: encrypted,
      tag: cipher.getAuthTag().toString('hex')
    };
  }

  static decryptData(encryptedData, key) {
    const decipher = crypto.createDecipheriv(
      SecurityPolicyV12.EncryptionPolicy.atRest.algorithm,
      key,
      Buffer.from(encryptedData.iv, 'hex')
    );
    
    decipher.setAuthTag(Buffer.from(encryptedData.tag, 'hex'));
    
    let decrypted = decipher.update(encryptedData.content, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }
