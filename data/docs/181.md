Data: 2024-01-15
Status: Aceita
Autores: Tech Lead + Especialista Infra + Especialista Backend

CONTEXTO:
A V11 do DIYAPP atingiu limites de escalabilidade devido à arquitetura monolítica.
Precisamos evoluir para microsserviços que permitam:
1. Deploy independente por squad
2. Escala horizontal por componente crítico
3. Isolamento de falhas
4. Autonomia técnica por domínio de negócio

DECISÃO:
Adotar arquitetura de microsserviços baseada em eventos com os seguintes princípios:
- Cada microsserviço é dono de seu banco de dados
- Comunicação síncrona via gRPC para operações críticas
- Comunicação assíncrona via Apache Kafka para eventos de domínio
- API Gateway único com rate limiting e autenticação centralizada
- Service Mesh (Istio) para observabilidade e resiliência

OPÇÕES CONSIDERADAS:
- Opção A: Monolito modularizado - Prós: Simples, baixa latência. Contras: Acoplamento, deploy único.
- Opção B: Microsserviços REST - Prós: Desacoplamento. Contras: Latência, versionamento complexo.
- Opção C: Microsserviços gRPC + Eventos - Prós: Performance, resiliência. Contras: Complexidade operacional.

Opção escolhida: C - Justificativa: Balanceia performance com resiliência, alinha com necessidades de IA.

CONSEQUÊNCIAS:
Positivas: Escalabilidade ilimitada, deploy independente, isolamento de falhas.
Negativas: Complexidade distribuída, necessidade de observabilidade robusta.
Riscos: Latência em chamadas cross-service, consistência eventual. Mitigação: Circuit breakers, sagas.

REVISÃO: 2024-04-15
"""

```typescript
import React, { createContext, useContext } from 'react'
import { tokens } from './tokens'

interface ThemeContextType {
  tokens: typeof tokens
  mode: 'light' | 'dark'
  toggleMode: () => void
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined)

export const useTheme = () => {
  const context = useContext(ThemeContext)
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider')
  }
  return context
}

interface ThemeProviderProps {
  children: React.ReactNode
}

export const ThemeProvider: React.FC<ThemeProviderProps> = ({ children }) => {
  const [mode, setMode] = React.useState<'light' | 'dark'>('light')

  const toggleMode = () => {
    setMode(prev => {
      const newMode = prev === 'light' ? 'dark' : 'light'
      document.documentElement.classList.toggle('dark', newMode === 'dark')
      return newMode
    })
  }

  React.useEffect(() => {
    const isDark = window.matchMedia('(prefers-color-scheme: dark)').matches
    setMode(isDark ? 'dark' : 'light')
    document.documentElement.classList.toggle('dark', isDark)
  }, [])

  return (
    <ThemeContext.Provider value={{ tokens, mode, toggleMode }}>
      {children}
    </ThemeContext.Provider>
  )
}
```

const amqp = require('amqplib');
const logger = require('../utils/logger');

let connection = null;
let channel = null;

const QUEUES = {
  TASKS: 'tasks_queue',
  TASK_RESULTS: 'task_results_queue',
  WORKFLOWS: 'workflows_queue',
  DEAD_LETTER: 'dead_letter_queue',
};

const EXCHANGES = {
  TASKS: 'tasks_exchange',
  WORKFLOWS: 'workflows_exchange',
};

async function connectRabbitMQ() {
  try {
    const url = process.env.RABBITMQ_URL || 'amqp://localhost:5672';
    connection = await amqp.connect(url);
    
    channel = await connection.createChannel();
    
    // Declarar exchanges
    await channel.assertExchange(EXCHANGES.TASKS, 'direct', { durable: true });
    await channel.assertExchange(EXCHANGES.WORKFLOWS, 'direct', { durable: true });
    
    // Declarar filas com DLQ
    const queueOptions = {
      durable: true,
      deadLetterExchange: '',
      deadLetterRoutingKey: QUEUES.DEAD_LETTER,
    };
    
    await channel.assertQueue(QUEUES.TASKS, queueOptions);
    await channel.assertQueue(QUEUES.TASK_RESULTS, queueOptions);
    await channel.assertQueue(QUEUES.WORKFLOWS, queueOptions);
    await channel.assertQueue(QUEUES.DEAD_LETTER, { durable: true });
    
    // Bind filas aos exchanges
    await channel.bindQueue(QUEUES.TASKS, EXCHANGES.TASKS, 'task');
    await channel.bindQueue(QUEUES.WORKFLOWS, EXCHANGES.WORKFLOWS, 'workflow');
    
    logger.info('RabbitMQ connected and queues configured');
    
    // Tratamento de erros
    connection.on('error', (err) => {
      logger.error('RabbitMQ connection error:', err);
    });
    
    connection.on('close', () => {
      logger.warn('RabbitMQ connection closed');
    });
  } catch (error) {
    logger.error('Failed to connect to RabbitMQ:', error);
    throw error;
  }
}

  if (!channel) {
    throw new Error('RabbitMQ channel not initialized');
  }
  return channel;

```sql
-- Função para calcular hash de stack trace para agrupamento de erros
CREATE OR REPLACE FUNCTION calculate_stack_trace_hash(stack_trace TEXT)
RETURNS VARCHAR(64) AS $$
BEGIN
    RETURN encode(digest(
        regexp_replace(
            regexp_replace(stack_trace, '\s+', ' ', 'g'),
            'at\s+[^\s]+\s+\([^)]+\)',
            'at FUNCTION',
            'g'
        ),
        'sha256'
    ), 'hex');
END;
$$ LANGUAGE plpgsql IMMUTABLE;

-- Procedure para limpeza automática de dados antigos
CREATE OR REPLACE PROCEDURE cleanup_old_telemetry_data(
    retention_days INTEGER DEFAULT 90,
    batch_size INTEGER DEFAULT 10000
)
LANGUAGE plpgsql
AS $$
DECLARE
    rows_deleted INTEGER;
    total_deleted INTEGER := 0;
BEGIN
    RAISE NOTICE 'Iniciando limpeza de dados com mais de % dias', retention_days;
    
    -- Limpar raw_performance_metrics
    LOOP
        DELETE FROM raw_performance_metrics
        WHERE event_timestamp < NOW() - (retention_days || ' days')::INTERVAL
        AND id IN (
            SELECT id FROM raw_performance_metrics
            WHERE event_timestamp < NOW() - (retention_days || ' days')::INTERVAL
            LIMIT batch_size
        );
        
        GET DIAGNOSTICS rows_deleted = ROW_COUNT;
        total_deleted := total_deleted + rows_deleted;
        
        RAISE NOTICE 'Deletados % registros de raw_performance_metrics', rows_deleted;
        
        EXIT WHEN rows_deleted = 0;
        COMMIT;
    END LOOP;
    
    -- Limpar raw_structured_logs
    LOOP
        DELETE FROM raw_structured_logs
        WHERE timestamp < NOW() - (retention_days || ' days')::INTERVAL
        AND id IN (
            SELECT id FROM raw_structured_logs
            WHERE timestamp < NOW() - (retention_days || ' days')::INTERVAL
            LIMIT batch_size
        );
        
        GET DIAGNOSTICS rows_deleted = ROW_COUNT;
        total_deleted := total_deleted + rows_deleted;
        
        RAISE NOTICE 'Deletados % registros de raw_structured_logs', rows_deleted;
        
        EXIT WHEN rows_deleted = 0;
        COMMIT;
    END LOOP;
    
    RAISE NOTICE 'Limpeza concluída. Total deletado: % registros', total_deleted;
END;
$$;

-- Função de trigger para auditoria
CREATE OR REPLACE FUNCTION audit_data_changes()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO data_audit_log (
            operation, table_name, record_id, new_values
        ) VALUES (
            'INSERT', TG_TABLE_NAME, NEW.id::TEXT, to_jsonb(NEW)
        );
        RETURN NEW;
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO data_audit_log (
            operation, table_name, record_id, old_values, new_values
        ) VALUES (
            'UPDATE', TG_TABLE_NAME, NEW.id::TEXT, to_jsonb(OLD), to_jsonb(NEW)
        );
        RETURN NEW;
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO data_audit_log (
            operation, table_name, record_id, old_values
        ) VALUES (
            'DELETE', TG_TABLE_NAME, OLD.id::TEXT, to_jsonb(OLD)
        );
        RETURN OLD;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;
```

# DIYAPP Evolution - V12 Core - Segurança

## 1. Threat Modeling da Nova Arquitetura (STRIDE)

### Documento de Threat Modeling - Arquitetura V12

**Data:** 2024-01-15  
**Responsável:** Especialista em Segurança  
**Feature:** Arquitetura Core V12  
**Status:** Ativo  

---

### 1.1 Ativos Críticos
- **A1:** Credenciais de usuário (JWT tokens, refresh tokens)
- **A2:** Dados pessoais (PII) dos usuários
- **A3:** Configurações do sistema e secrets
- **A4:** Modelos de IA e prompts do sistema
- **A5:** Logs de auditoria e operações

### 1.2 Análise STRIDE

#### **Spoofing (Falsificação)**
- **T1:** Ataque em endpoints de autenticação
- **Controles:** Rate limiting, MFA, validação de tokens JWT com assinatura forte
- **Risco:** Alto (CVSS 8.2)

#### **Tampering (Manipulação)**
- **T2:** Manipulação de dados em trânsito (MITM)
- **Controles:** TLS 1.3 obrigatório, HSTS headers
- **T3:** Manipulação de dados em repouso
- **Controles:** Criptografia AES-256-GCM para PII, checksums de integridade
- **Risco:** Médio (CVSS 6.5)

#### **Repudiation (Repúdio)**
- **T4:** Negação de operações críticas
- **Controles:** Logs imutáveis com WAL, assinatura digital de logs críticos
- **Risco:** Baixo (CVSS 3.1)

#### **Information Disclosure (Divulgação)**
- **T5:** Vazamento de dados sensíveis via API
- **Controles:** Mascaramento de PII em responses, controle de acesso granular
- **T6:** Vazamento de secrets no código
- **Controles:** Scanner de secrets no pipeline, uso de vault
- **Risco:** Alto (CVSS 7.8)

#### **Denial of Service (Negação)**
- **T7:** Ataque DDoS em endpoints públicos
- **Controles:** Rate limiting por IP/user, circuit breakers, CDN/WAF
- **Risco:** Médio (CVSS 5.9)

#### **Elevation of Privilege (Elevação)**
- **T8:** Escalação de privilégios via vulnerabilidade
- **Controles:** Princípio do menor privilégio, RBAC granular, revisão de código
- **Risco:** Alto (CVSS 8.5)

### 1.3 Riscos Aceitos
- **RA1:** Ataques de força bruta em login (mitigado por rate limiting e MFA)
- **RA2:** Vulnerabilidades de dia zero em dependências (mitigado por scanner contínuo)

```javascript
const jwt = require('jsonwebtoken');
const crypto = require('crypto');
const securityConfig = require('../../config/security');

class JWTManager {
  constructor() {
    this.accessSecret = securityConfig.jwt.accessToken.secret;
    this.refreshSecret = securityConfig.jwt.refreshToken.secret;
    this.accessExpiresIn = securityConfig.jwt.accessToken.expiresIn;
    this.refreshExpiresIn = securityConfig.jwt.refreshToken.expiresIn;
  }

  /**
   * Gera um par de tokens (access + refresh)
   */
  generateTokenPair(user) {
    const accessToken = jwt.sign(
      {
        userId: user.id,
        email: user.email,
        role: user.role,
        type: 'access'
      },
      this.accessSecret,
      { 
        expiresIn: this.accessExpiresIn,
        algorithm: securityConfig.jwt.accessToken.algorithm 
      }
    );

    const refreshToken = jwt.sign(
      {
        userId: user.id,
        type: 'refresh',
        jti: crypto.randomBytes(16).toString('hex') // JWT ID único
      },
      this.refreshSecret,
      { 
        expiresIn: this.refreshExpiresIn,
        algorithm: securityConfig.jwt.refreshToken.algorithm 
      }
    );

    return { accessToken, refreshToken };
  }

  /**
   * Verifica um access token
   */
  verifyAccessToken(token) {
    try {
      return jwt.verify(token, this.accessSecret, {
        algorithms: [securityConfig.jwt.accessToken.algorithm]
      });
    } catch (error) {
      throw new Error(`Token inválido: ${error.message}`);
    }
  }

  /**
   * Verifica um refresh token
   */
  verifyRefreshToken(token) {
    try {
      return jwt.verify(token, this.refreshSecret, {
        algorithms: [securityConfig.jwt.refreshToken.algorithm]
      });
    } catch (error) {
      throw new Error(`Refresh token inválido: ${error.message}`);
    }
  }

  /**
   * Decodifica token sem verificar (apenas para logging)
   */
  decodeToken(token) {
    return jwt.decode(token);
  }

  /**
   * Invalida refresh token (simulação - em produção usar blacklist)
   */
  async invalidateRefreshToken(tokenId) {
    // Em produção: adicionar à blacklist no Redis
    console.log(`Token ${tokenId} invalidado`);
    return true;
  }
}

module.exports = new JWTManager();
```

```javascript
class AuthMiddleware {
  /**
   * Middleware de autenticação JWT
   */
  authenticate(req, res, next) {
    const authHeader = req.headers.authorization;
    
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      return res.status(401).json({
        error: 'Não autorizado',
        message: 'Token de acesso não fornecido'
      });
    }

    const token = authHeader.substring(7);

    try {
      const decoded = jwtManager.verifyAccessToken(token);
      req.user = decoded;
      next();
    } catch (error) {
      return res.status(401).json({
        error: 'Não autorizado',
        message: 'Token inválido ou expirado'
      });
    }
  }

  /**
   * Middleware de autorização baseada em roles
   */
  authorize(roles = []) {
    return (req, res, next) => {
      if (!req.user) {
        return res.status(401).json({
          error: 'Não autorizado',
          message: 'Usuário não autenticado'
        });
      }

      if (roles.length && !roles.includes(req.user.role)) {
        return res.status(403).json({
          error: 'Proibido',
          message: 'Permissão insuficiente'
        });
      }

      next();
    };
  }

  /**
   * Middleware de rate limiting para autenticação
   */
  authRateLimit(req, res, next) {
    const ip = req.ip || req.connection.remoteAddress;
    const key = `auth_${ip}`;
    
    return rateLimiter.auth(key, req, res, next);
  }

  /**
   * Middleware para refresh token
   */
  async handleRefreshToken(req, res, next) {
    const { refreshToken } = req.body;

    if (!refreshToken) {
      return res.status(400).json({
        error: 'Bad Request',
        message: 'Refresh token não fornecido'
      });
    }

    try {
      const decoded = jwtManager.verifyRefreshToken(refreshToken);
      
      if (decoded.type !== 'refresh') {
        throw new Error('Tipo de token inválido');
      }

      // Em produção: verificar se token está na blacklist
      // const isBlacklisted = await checkTokenBlacklist(decoded.jti);
      
      // Gerar novo par de tokens
      const user = {
        id: decoded.userId,
        email: decoded.email,
        role: decoded.role
      };
      
      const tokens = jwtManager.generateTokenPair(user);
      
      // Invalidação do refresh token antigo (rotatividade)
      await jwtManager.invalidateRefreshToken(decoded.jti);
      
      req.newTokens = tokens;
      next();
    } catch (error) {
      return res.status(401).json({
        error: 'Não autorizado',
        message: 'Refresh token inválido'
      });
    }
  }

  /**
   * Middleware para logging de auditoria
   */
  auditLog(req, res, next) {
    const startTime = Date.now();
    
    // Intercepta a resposta
    const originalSend = res.send;
    res.send = function(data) {
      const duration = Date.now() - startTime;
      
      // Log de auditoria
      console.log(JSON.stringify({
        timestamp: new Date().toISOString(),
        method: req.method,
        url: req.url,
        statusCode: res.statusCode,
        duration: `${duration}ms`,
        userId: req.user?.userId || 'anonymous',
        userAgent: req.get('User-Agent'),
        ip: req.ip
      }));
      
      originalSend.call(this, data);
    };
    
    next();
  }
}
```

```javascript
const securityConfig = require('../../config/security');

class RateLimiter {
  constructor() {
    this.authLimits = new Map();
    this.apiLimits = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60000); // Limpeza a cada minuto
  }

  /**
   * Rate limiting para autenticação
   */
  auth(key, req, res, next) {
    const now = Date.now();
    const windowMs = securityConfig.rateLimiting.auth.windowMs;
    const maxAttempts = securityConfig.rateLimiting.auth.max;

    if (!this.authLimits.has(key)) {
      this.authLimits.set(key, {
        count: 1,
        resetTime: now + windowMs,
        firstAttempt: now
      });
      return next();
    }

    const limit = this.authLimits.get(key);

    if (now > limit.resetTime) {
      // Reset do limite
      this.authLimits.set(key, {
        count: 1,
        resetTime: now + windowMs,
        firstAttempt: now
      });
      return next();
    }

    if (limit.count >= maxAttempts) {
      const retryAfter = Math.ceil((limit.resetTime - now) / 1000);
      
      res.setHeader('Retry-After', retryAfter);
      return res.status(429).json({
        error: 'Muitas requisições',
        message: securityConfig.rateLimiting.auth.message,
        retryAfter: `${retryAfter} segundos`
      });
    }

    limit.count++;
    return next();
  }

  /**
   * Rate limiting para API geral
   */
  api(key, req, res, next) {
    const now = Date.now();
    const windowMs = securityConfig.rateLimiting.api.windowMs;
    const maxRequests = securityConfig.rateLimiting.api.max;

    if (!this.apiLimits.has(key)) {
      this.apiLimits.set(key, {
        count: 1,
        resetTime: now + windowMs
      });
      return next();
    }

    const limit = this.apiLimits.get(key);

    if (now > limit.resetTime) {
      this.apiLimits.set(key, {
        count: 1,
        resetTime: now + windowMs
      });
      return next();
    }

    if (limit.count >= maxRequests) {
      return res.status(429).json({
        error: 'Muitas requisições',
        message: 'Limite de requisições excedido'
      });
    }

    limit.count++;
    return next();
  }

  /**
   * Limpeza de entradas expiradas
   */
  cleanup() {
    const now = Date.now();
    
    // Limpa auth limits
    for (const [key, limit] of this.authLimits.entries()) {
      if (now > limit.resetTime) {
        this.authLimits.delete(key);
      }
    }
    
    // Limpa api limits
    for (const [key, limit] of this.apiLimits.entries()) {
      if (now > limit.resetTime) {
        this.apiLimits.delete(key);
      }
    }
  }

  /**
   * Obtém status do rate limiting para um IP
   */
  getStatus(key) {
    const authLimit = this.authLimits.get(key);
    const apiLimit = this.apiLimits.get(key);
    
    return {
      auth: authLimit ? {
        attempts: authLimit.count,
        resetIn: Math.max(0, authLimit.resetTime - Date.now()),
        blocked: authLimit.count >= securityConfig.rateLimiting.auth.max
      } : null,
      api: apiLimit ? {
        requests: apiLimit.count,
        resetIn: Math.max(0, apiLimit.resetTime - Date.now()),
        blocked: apiLimit.count >= securityConfig.rateLimiting.api.max
      } : null
    };
  }
}

module.exports = new RateLimiter();
```

```python
"""
Sistema de Previsão de Falhas baseado em ML para telemetria de pods Kubernetes
"""
import numpy as np
import pandas as pd
from datetime import datetime, timedelta
from typing import Dict, List, Optional, Tuple
import joblib
import json
import logging
from dataclasses import dataclass
from enum import Enum
import asyncio
from prometheus_client import CollectorRegistry, Gauge, Counter

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class FailureType(Enum):
    MEMORY_LEAK = "memory_leak"
    CPU_SATURATION = "cpu_saturation"
    NETWORK_LATENCY = "network_latency"
    DISK_IO = "disk_io"
    APPLICATION_ERROR = "application_error"
    UNKNOWN = "unknown"

@dataclass
class PodMetrics:
    """Estrutura para métricas de telemetria de pods"""
    pod_name: str
    namespace: str
    timestamp: datetime
    cpu_usage: float  # percentual
    memory_usage: float  # percentual
    memory_rss: float  # MB
    network_rx: float  # KB/s
    network_tx: float  # KB/s
    disk_read: float  # IOPS
    disk_write: float  # IOPS
    restart_count: int
    http_errors: int
    latency_p95: float  # ms
    error_rate: float  # percentual
    
@dataclass
class FailurePrediction:
    """Resultado da previsão de falha"""
    pod_name: str
    namespace: str
    failure_type: FailureType
    confidence: float  # 0-1
    estimated_time_to_failure: timedelta
    severity: str  # LOW, MEDIUM, HIGH, CRITICAL
    recommended_action: str
    features: Dict[str, float]

class FailurePredictor:
    """Modelo de ML para previsão de falhas em tempo real"""
    
    def __init__(self, model_path: Optional[str] = None):
        self.model = None
        self.scaler = None
        self.feature_columns = [
            'cpu_usage', 'memory_usage', 'memory_rss',
            'network_rx_rate', 'network_tx_rate',
            'disk_io_ratio', 'restart_count_24h',
            'error_rate_5m', 'latency_increase',
            'memory_growth_rate', 'cpu_spike_count'
        ]
        
        # Métricas Prometheus
        self.registry = CollectorRegistry()
        self.prediction_confidence = Gauge(
            'failure_prediction_confidence',
            'Confidence level of failure predictions',
            ['pod', 'namespace', 'failure_type'],
            registry=self.registry
        )
        self.prediction_count = Counter(
            'failure_predictions_total',
            'Total number of failure predictions made',
            ['severity'],
            registry=self.registry
        )
        
        if model_path:
            self.load_model(model_path)
        else:
            self.initialize_default_model()
    
    def initialize_default_model(self):
        """Inicializa um modelo padrão baseado em Isolation Forest"""
        from sklearn.ensemble import IsolationForest
        from sklearn.preprocessing import StandardScaler
        
        self.model = IsolationForest(
            n_estimators=100,
            contamination=0.1,
            random_state=42,
            n_jobs=-1
        )
        self.scaler = StandardScaler()
        logger.info("Modelo padrão Isolation Forest inicializado")
    
    def load_model(self, model_path: str):
        """Carrega modelo treinado"""
        try:
            model_data = joblib.load(model_path)
            self.model = model_data['model']
            self.scaler = model_data['scaler']
            logger.info(f"Modelo carregado de {model_path}")
        except Exception as e:
            logger.error(f"Erro ao carregar modelo: {e}")
            self.initialize_default_model()
    
    def save_model(self, model_path: str):
        """Salva o modelo atual"""
        model_data = {
            'model': self.model,
            'scaler': self.scaler,
            'feature_columns': self.feature_columns,
            'timestamp': datetime.now()
        }
        joblib.dump(model_data, model_path)
        logger.info(f"Modelo salvo em {model_path}")
    
    def extract_features(self, metrics_history: List[PodMetrics]) -> np.ndarray:
        """Extrai features para predição a partir do histórico de métricas"""
        if len(metrics_history) < 10:
            return None
        
        df = pd.DataFrame([m.__dict__ for m in metrics_history])
        df.set_index('timestamp', inplace=True)
        
        # Calcula features derivadas
        features = {}
        
        # Métricas atuais
        latest = metrics_history[-1]
        features['cpu_usage'] = latest.cpu_usage
        features['memory_usage'] = latest.memory_usage
        features['memory_rss'] = latest.memory_rss
        
        # Taxas de rede (últimos 5 minutos)
        recent_metrics = [m for m in metrics_history[-5:] if len(metrics_history) >= 5]
        if recent_metrics:
            features['network_rx_rate'] = np.mean([m.network_rx for m in recent_metrics])
            features['network_tx_rate'] = np.mean([m.network_tx for m in recent_metrics])
        
        # Razão de IO de disco
        features['disk_io_ratio'] = latest.disk_write / (latest.disk_read + 1e-6)
        
        # Contagem de reinícios nas últimas 24h
        day_ago = datetime.now() - timedelta(hours=24)
        features['restart_count_24h'] = len([m for m in metrics_history 
                                           if m.timestamp > day_ago and m.restart_count > 0])
        
        # Taxa de erro nos últimos 5 minutos
        features['error_rate_5m'] = latest.error_rate
        
        # Aumento de latência (comparação com baseline)
        if len(metrics_history) > 30:
            baseline_latency = np.percentile([m.latency_p95 for m in metrics_history[:30]], 50)
            features['latency_increase'] = latest.latency_p95 / (baseline_latency + 1e-6)
        
        # Taxa de crescimento de memória
        if len(metrics_history) > 10:
            mem_values = [m.memory_usage for m in metrics_history[-10:]]
            features['memory_growth_rate'] = np.polyfit(range(10), mem_values, 1)[0]
        
        # Contagem de picos de CPU
        cpu_values = [m.cpu_usage for m in metrics_history[-20:]]
        cpu_mean = np.mean(cpu_values)
        cpu_std = np.std(cpu_values)
        features['cpu_spike_count'] = sum(1 for c in cpu_values 
                                         if c > cpu_mean + 2 * cpu_std)
        
        # Converte para array no formato correto
        feature_array = np.array([[features.get(col, 0) for col in self.feature_columns]])
        return feature_array
    
    def predict_failure(self, metrics_history: List[PodMetrics]) -> Optional[FailurePrediction]:
        """Faz previsão de falha baseada no histórico de métricas"""
        try:
            features = self.extract_features(metrics_history)
            if features is None:
                return None
            
            # Normaliza features
            if self.scaler:
                features_scaled = self.scaler.transform(features)
            else:
                features_scaled = features
            
            # Faz predição
            anomaly_score = self.model.decision_function(features_scaled)[0]
            is_anomaly = self.model.predict(features_scaled)[0] == -1
            
            # Converte score para confiança (0-1)
            confidence = max(0, min(1, 1 - abs(anomaly_score)))
            
            if not is_anomaly or confidence < 0.3:
                return None
            
            # Determina tipo de falha baseado nas features
            failure_type = self._classify_failure_type(features[0])
            
            # Estima tempo até falha
            ttf = self._estimate_time_to_failure(features[0], confidence)
            
            # Determina severidade
            severity = self._determine_severity(failure_type, confidence, features[0])
            
            # Recomenda ação
            action = self._recommend_action(failure_type, severity)
            
            latest_metric = metrics_history[-1]
            prediction = FailurePrediction(
                pod_name=latest_metric.pod_name,
                namespace=latest_metric.namespace,
                failure_type=failure_type,
                confidence=confidence,
                estimated_time_to_failure=ttf,
                severity=severity,
                recommended_action=action,
                features={col: float(features[0][i]) 
                         for i, col in enumerate(self.feature_columns)}
            )
            
            # Atualiza métricas Prometheus
            self.prediction_confidence.labels(
                pod=latest_metric.pod_name,
                namespace=latest_metric.namespace,
                failure_type=failure_type.value
            ).set(confidence)
            
            self.prediction_count.labels(severity=severity).inc()
            
            logger.info(f"Previsão de falha detectada: {prediction}")
            return prediction
            
        except Exception as e:
            logger.error(f"Erro na previsão de falha: {e}")
            return None
    
    def _classify_failure_type(self, features: np.ndarray) -> FailureType:
        """Classifica o tipo de falha baseado nas features"""
        feature_dict = {col: features[i] for i, col in enumerate(self.feature_columns)}
        
        if feature_dict['memory_usage'] > 90 and feature_dict['memory_growth_rate'] > 0.5:
            return FailureType.MEMORY_LEAK
        elif feature_dict['cpu_usage'] > 85 and feature_dict['cpu_spike_count'] > 3:
            return FailureType.CPU_SATURATION
        elif feature_dict['latency_increase'] > 2.0:
            return FailureType.NETWORK_LATENCY
        elif feature_dict['disk_io_ratio'] > 10 or feature_dict['disk_io_ratio'] < 0.1:
            return FailureType.DISK_IO
        elif feature_dict['error_rate_5m'] > 5.0:
            return FailureType.APPLICATION_ERROR
        else:
            return FailureType.UNKNOWN
    
    def _estimate_time_to_failure(self, features: np.ndarray, confidence: float) -> timedelta:
        """Estima tempo até falha baseado na confiança e features"""
        # Modelo simplificado - em produção usar regressão
        if confidence > 0.8:
            return timedelta(minutes=5)
        elif confidence > 0.6:
            return timedelta(minutes=15)
        elif confidence > 0.4:
            return timedelta(minutes=30)
        else:
            return timedelta(hours=1)
    
    def _determine_severity(self, failure_type: FailureType, 
                           confidence: float, features: np.ndarray) -> str:
        """Determina severidade da falha prevista"""
        if confidence > 0.8:
            return "CRITICAL"
        elif confidence > 0.6:
            return "HIGH"
        elif confidence > 0.4:
            return "MEDIUM"
        else:
            return "LOW"
    
    def _recommend_action(self, failure_type: FailureType, severity: str) -> str:
        """Recomenda ação baseada no tipo e severidade da falha"""
        actions = {
            FailureType.MEMORY_LEAK: {
                "CRITICAL": "Reinicializar pod imediatamente e aumentar memory limits",
                "HIGH": "Reinicializar pod e analisar heap dump",
                "MEDIUM": "Monitorar e coletar memory profiles",
                "LOW": "Ajustar memory limits e monitorar"
            },
            FailureType.CPU_SATURATION: {
                "CRITICAL": "Escalar horizontalmente e redistribuir carga",
                "HIGH": "Aumentar CPU limits e otimizar código",
                "MEDIUM": "Analisar profiling de CPU",
                "LOW": "Ajustar CPU requests/limits"
            },
            FailureType.NETWORK_LATENCY: {
                "CRITICAL": "Redistribuir pods entre nodes e verificar network policies",
                "HIGH": "Verificar network plugins e DNS",
                "MEDIUM": "Monitorar network metrics",
                "LOW": "Ajustar timeouts e retries"
            },
            FailureType.DISK_IO: {
                "CRITICAL": "Migrar para storage de maior performance",
                "HIGH": "Otimizar queries e cache",
                "MEDIUM": "Monitorar disk IOPS",
                "LOW": "Ajustar volume size"
            },
            FailureType.APPLICATION_ERROR: {
                "CRITICAL": "Rollback para versão estável",
                "HIGH": "Reinicializar pod e analisar logs",
                "MEDIUM": "Aumentar logging e monitorar",
                "LOW": "Ajustar error handling"
            }
        }
        
        return actions.get(failure_type, {}).get(severity, 
               "Monitorar e coletar mais métricas para análise")
    
    async def continuous_monitoring(self, telemetry_source, check_interval: int = 30):
        """Monitoramento contínuo de pods"""
        logger.info(f"Iniciando monitoramento contínuo (intervalo: {check_interval}s)")
        
        pod_histories = {}  # {pod_id: List[PodMetrics]}
        
        while True:
            try:
                # Coleta métricas atuais
                current_metrics = await telemetry_source.get_current_metrics()
                
                for metric in current_metrics:
                    pod_id = f"{metric.namespace}/{metric.pod_name}"
                    
                    if pod_id not in pod_histories:
                        pod_histories[pod_id] = []
                    
                    # Adiciona nova métrica ao histórico
                    pod_histories[pod_id].append(metric)
                    
                    # Mantém apenas últimas 100 métricas
                    if len(pod_histories[pod_id]) > 100:
                        pod_histories[pod_id] = pod_histories[pod_id][-100:]
                    
                    # Faz previsão se tiver histórico suficiente
                    if len(pod_histories[pod_id]) >= 20:
                        prediction = self.predict_failure(pod_histories[pod_id])
                        
                        if prediction:
                            # Publica evento de previsão
                            await self._publish_prediction_event(prediction)
                
                # Limpa históricos de pods não mais existentes
                current_pod_ids = {f"{m.namespace}/{m.pod_name}" for m in current_metrics}
                pod_histories = {k: v for k, v in pod_histories.items() 
                               if k in current_pod_ids}
                
                await asyncio.sleep(check_interval)
                
            except Exception as e:
                logger.error(f"Erro no monitoramento contínuo: {e}")
                await asyncio.sleep(check_interval)
    
    async def _publish_prediction_event(self, prediction: FailurePrediction):
        """Publica evento de previsão para outros componentes"""
        # Em produção, integrar com message queue (Kafka, RabbitMQ)
        event = {
            "type": "failure_prediction",
            "timestamp": datetime.now().isoformat(),
            "prediction": {
                "pod": prediction.pod_name,
                "namespace": prediction.namespace,
                "failure_type": prediction.failure_type.value,
                "confidence": prediction.confidence,
                "estimated_ttf": str(prediction.estimated_time_to_failure),
                "severity": prediction.severity,
                "action": prediction.recommended_action
            }
        }
        logger.info(f"Evento publicado: {json.dumps(event, default=str)}")
```

- Invalidação por evento

## Decisões Arquiteturais

### ADR-001: Escolha de Message Broker
**Contexto:** Necessidade de comunicação assíncrona confiável entre serviços
**Decisão:** RabbitMQ sobre Kafka
**Justificativa:** 
- Menor latência para nosso caso de uso
- Setup mais simples
- Suporte nativo a DLQ
- Menor overhead operacional

### ADR-002: Estratégia de Banco de Dados
**Contexto:** Dados transacionais com necessidade de alta disponibilidade
**Decisão:** PostgreSQL com replicação síncrona
**Justificativa:**
- ACID compliance
- Replicação nativa
- Comunidade ativa
- Ferramentas de backup maduras

```javascript
// src/services/AuditService.js
const { createLogger, transports, format } = require('winston');
const { combine, timestamp, json, metadata } = format;
const PrivacyLog = require('../models/PrivacyLog');
const crypto = require('crypto');

class AuditService {
  constructor() {
    this.logger = createLogger({
      level: 'info',
      format: combine(
        timestamp(),
        json(),
        metadata()
      ),
      transports: [
        new transports.File({ 
          filename: 'logs/compliance-audit.log',
          maxsize: 5242880, // 5MB
          maxFiles: 5
        }),
        new transports.Console({
          format: format.combine(
            format.colorize(),
            format.simple()
          )
        })
      ]
    });
  }

  async logDataAccess(userId, action, resource, details) {
    const auditLog = {
      timestamp: new Date(),
      userId,
      action,
      resource,
      ipAddress: details.ipAddress || 'unknown',
      userAgent: details.userAgent || 'unknown',
      dataBefore: this.hashSensitiveData(details.dataBefore),
      dataAfter: this.hashSensitiveData(details.dataAfter),
      complianceCheck: this.performComplianceCheck(action, details),
      checksum: this.generateChecksum({
        userId, action, resource, timestamp: new Date()
      })
    };

    // Log no banco de dados
    await PrivacyLog.create(auditLog);

    // Log no arquivo
    this.logger.info('Data access audit', auditLog);

    return auditLog;
  }

  async logConsentChange(userId, consentType, previousValue, newValue, context) {
    const logEntry = {
      timestamp: new Date(),
      userId,
      action: 'consent.change',
      resource: 'user.consent',
      consentType,
      previousValue,
      newValue,
      context: JSON.stringify(context),
      ipAddress: context.ipAddress,
      userAgent: context.userAgent,
      legalBasis: 'consentimento',
      checksum: this.generateChecksum({
        userId, consentType, timestamp: new Date()
      })
    };

    await PrivacyLog.create(logEntry);
    this.logger.info('Consent change recorded', logEntry);
  }

  async logDataSubjectRequest(requestId, requestType, status, processorId) {
    const logEntry = {
      timestamp: new Date(),
      requestId,
      action: `data.subject.${requestType}`,
      resource: 'user.data',
      status,
      processorId,
      slaDeadline: this.calculateSLADeadline(),
      checksum: this.generateChecksum({ requestId, timestamp: new Date() })
    };

    await PrivacyLog.create(logEntry);
    this.logger.info('Data subject request processed', logEntry);
  }

  async getAuditTrail(userId, startDate, endDate) {
    const whereClause = {
      timestamp: {
        [Op.between]: [startDate, endDate]
      }
    };

    if (userId) {
      whereClause.userId = userId;
    }

    return await PrivacyLog.findAll({
      where: whereClause,
      order: [['timestamp', 'DESC']],
      limit: 1000
    });
  }

  async generateComplianceReport(period) {
    const report = {
      period,
      generatedAt: new Date(),
      statistics: {},
      violations: [],
      recommendations: []
    };

    // Estatísticas de acesso
    report.statistics.dataAccess = await this.getAccessStatistics(period);
    
    // Solicitações de titulares
    report.statistics.dataSubjectRequests = await this.getDSRStatistics(period);
    
    // Mudanças de consentimento
    report.statistics.consentChanges = await this.getConsentStatistics(period);
    
    // Verificação de conformidade
    report.violations = await this.detectComplianceViolations(period);
    
    // Recomendações
    report.recommendations = await this.generateRecommendations(report);

    return report;
  }

  hashSensitiveData(data) {
    if (!data || typeof data !== 'object') return data;
    
    const sensitiveFields = ['cpf', 'cnpj', 'password', 'creditCard', 'token'];
    const hashedData = { ...data };
    
    sensitiveFields.forEach(field => {
      if (hashedData[field]) {
        hashedData[field] = '***HASHED***';
      }
    });
    
    return hashedData;
  }

  generateChecksum(data) {
    const str = JSON.stringify(data);
    return crypto.createHash('sha256').update(str).digest('hex');
  }

  calculateSLADeadline() {
    const deadline = new Date();
    deadline.setDate(deadline.getDate() + 15); // SLA de 15 dias da LGPD
    return deadline;
  }

  performComplianceCheck(action, details) {
    const checks = {
      legalBasis: true,
      purposeLimitation: true,
      dataMinimization: true,
      storageLimitation: true,
      integrityConfidentiality: true
    };

    // Verificações específicas por ação
    if (action.includes('export')) {
      checks.dataMinimization = this.checkDataMinimization(details.dataExported);
    }

    if (action.includes('delete')) {
      checks.storageLimitation = this.checkRetentionCompliance(details);
    }

    return checks;
  }

  async getAccessStatistics(period) {
    const { startDate, endDate } = this.parsePeriod(period);
    
    const result = await PrivacyLog.sequelize.query(`
      SELECT 
        DATE(timestamp) as date,
        action,
        COUNT(*) as count,
        COUNT(DISTINCT userId) as uniqueUsers
      FROM privacy_logs
      WHERE timestamp BETWEEN :startDate AND :endDate
        AND action LIKE 'data.access.%'
      GROUP BY DATE(timestamp), action
      ORDER BY date DESC
    `, {
      replacements: { startDate, endDate },
      type: sequelize.QueryTypes.SELECT
    });

    return result;
  }

  async detectComplianceViolations(period) {
    const { startDate, endDate } = this.parsePeriod(period);
    
    const violations = [];

    // Verificar acessos fora do horário comercial
    const unusualAccess = await PrivacyLog.findAll({
      where: {
        timestamp: {
          [Op.between]: [startDate, endDate]
        },
        [Op.and]: [
          sequelize.where(
            sequelize.fn('HOUR', sequelize.col('timestamp')),
            { [Op.notBetween]: [9, 18] }
          ),
          {
            action: {
              [Op.in]: ['data.access.sensitive', 'data.export', 'user.delete']
            }
          }
        ]
      }
    });

    if (unusualAccess.length > 0) {
      violations.push({
        type: 'unusual_access_pattern',
        count: unusualAccess.length,
        details: unusualAccess.map(access => ({
          timestamp: access.timestamp,
          userId: access.userId,
          action: access.action
        }))
      });
    }

    // Verificar solicitações de titulares fora do SLA
    const overdueRequests = await PrivacyLog.findAll({
      where: {
        timestamp: {
          [Op.between]: [startDate, endDate]
        },
        action: {
          [Op.like]: 'data.subject.%'
        },
        status: 'pending',
        slaDeadline: {
          [Op.lt]: new Date()
        }
      }
    });

    if (overdueRequests.length > 0) {
      violations.push({
        type: 'sla_violation',
        count: overdueRequests.length,
        details: overdueRequests.map(req => ({
          requestId: req.requestId,
          deadline: req.slaDeadline,
          daysOverdue: Math.floor((new Date() - req.slaDeadline) / (1000 * 60 * 60 * 24))
        }))
      });
    }

    return violations;
  }

  parsePeriod(period) {
    const endDate = new Date();
    const startDate = new Date();
    
    switch (period) {
      case 'daily':
        startDate.setDate(startDate.getDate() - 1);
        break;
      case 'weekly':
        startDate.setDate(startDate.getDate() - 7);
        break;
      case 'monthly':
        startDate.setMonth(startDate.getMonth() - 1);
        break;
      case 'quarterly':
        startDate.setMonth(startDate.getMonth() - 3);
        break;
      default:
        startDate.setDate(startDate.getDate() - 1);
    }

    return { startDate, endDate };
  }
}
```
