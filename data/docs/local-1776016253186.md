## 1. ADR-001: Arquitetura de Microsserviços

**ADR-001: Arquitetura Baseada em Microsserviços para DIYAPP V11**

**Data:** 2024-01-15
**Status:** Aceita
**Autores:** Tech Lead

**CONTEXTO:**
DIYAPP V10 opera como uma aplicação monolítica que enfrenta desafios de escalabilidade, deploy independente e manutenção. A necessidade de evolução rápida de funcionalidades específicas (IA, notificações, conteúdo) requer isolamento de domínios e capacidade de deploy independente.

**DECISÃO:**
Adotar arquitetura de microsserviços com os seguintes serviços:
1. API Gateway (NestJS) - Roteamento e rate limiting
2. Auth Service (NestJS) - Autenticação JWT e autorização
3. User Service (NestJS) - Gestão de usuários e perfis
4. Content Service (NestJS) - Gestão de conteúdo e IA
5. Notification Service (Node.js + BullMQ) - Filas de notificação
6. Web Dashboard (React + Vite) - Interface administrativa

**OPÇÕES CONSIDERADAS:**
- **Opção A:** Monolito com módulos - Prós: Simplicidade inicial. Contras: Acoplamento alto, deploy único.
- **Opção B:** Microsserviços completos - Prós: Isolamento, deploy independente, escalabilidade. Contras: Complexidade operacional.
- **Opção C:** Arquitetura hexagonal - Prós: Testabilidade. Contras: Over-engineering para MVP.

**Opção escolhida:** B - Justificativa: Balanceia isolamento de domínios com capacidade de evolução independente, alinhado com roadmap de IA.

**CONSEQUÊNCIAS:**
**Positivas:** Deploy independente, escalabilidade por serviço, isolamento de falhas.
**Negativas:** Complexidade de comunicação, necessidade de orquestração, overhead operacional.
**Riscos:** Latência em chamadas entre serviços, consistência eventual. Mitigação: Circuit breakers, caching estratégico.

### `packages/shared-utils/src/auth.ts`
```typescript
import jwt from 'jsonwebtoken';
import bcrypt from 'bcryptjs';
import { config } from '@diyapp/shared-config';

export class AuthUtils {
  static async hashPassword(password: string): Promise<string> {
    return bcrypt.hash(password, 12);
  }

  static async comparePassword(password: string, hash: string): Promise<boolean> {
    return bcrypt.compare(password, hash);
  }

  static generateToken(payload: Record<string, any>, expiresIn: string = '1h'): string {
    return jwt.sign(payload, config.jwtSecret, { expiresIn });
  }

  static verifyToken(token: string): any {
    try {
      return jwt.verify(token, config.jwtSecret);
    } catch (error) {
      throw new Error('Invalid token');
    }
  }

  static generateRefreshToken(userId: string): string {
    return jwt.sign({ userId, type: 'refresh' }, config.jwtSecret, { expiresIn: '7d' });
  }
}
```

import React, { useState, useMemo } from 'react';
import {
  Table,
  TableHeader,
  TableBody,
  TableRow,
  TableHead,
  TableCell,
  Pagination,
  EmptyState,
  LoadingState,
  SortButton,
} from './styles';

export interface Column<T> {
  key: keyof T;
  header: string;
  sortable?: boolean;
  render?: (value: T[keyof T], row: T) => React.ReactNode;
  width?: string;
}

export interface DataGridProps<T> {
  columns: Column<T>[];
  data: T[];
  loading?: boolean;
  pageSize?: number;
  onSort?: (key: keyof T, direction: 'asc' | 'desc') => void;
  onRowClick?: (row: T) => void;
}

export function DataGrid<T extends Record<string, any>>({
  columns,
  data,
  loading = false,
  pageSize = 10,
  onSort,
  onRowClick,
}: DataGridProps<T>) {
  const [currentPage, setCurrentPage] = useState(1);
  const [sortConfig, setSortConfig] = useState<{
    key: keyof T;
    direction: 'asc' | 'desc';
  } | null>(null);

  const sortedData = useMemo(() => {
    if (!sortConfig || !data.length) return data;

    return [...data].sort((a, b) => {
      const aValue = a[sortConfig.key];
      const bValue = b[sortConfig.key];

      if (aValue < bValue) {
        return sortConfig.direction === 'asc' ? -1 : 1;
      }
      if (aValue > bValue) {
        return sortConfig.direction === 'asc' ? 1 : -1;
      }
      return 0;
    });
  }, [data, sortConfig]);

  const paginatedData = useMemo(() => {
    const startIndex = (currentPage - 1) * pageSize;
    return sortedData.slice(startIndex, startIndex + pageSize);
  }, [sortedData, currentPage, pageSize]);

  const totalPages = Math.ceil(sortedData.length / pageSize);

  const handleSort = (key: keyof T) => {
    if (!columns.find((col) => col.key === key)?.sortable) return;

    let direction: 'asc' | 'desc' = 'asc';
    if (sortConfig && sortConfig.key === key) {
      direction = sortConfig.direction === 'asc' ? 'desc' : 'asc';
    }

    setSortConfig({ key, direction });
    onSort?.(key, direction);
  };

  const handleRowClick = (row: T) => {
    onRowClick?.(row);
  };

  if (loading) {
    return <LoadingState>Loading data...</LoadingState>;
  }

  if (!data.length) {
    return <EmptyState>No data available</EmptyState>;
  }

  return (
    <div>
      <Table>
        <TableHeader>
          <TableRow>
            {columns.map((column) => (
              <TableHead key={column.key as string} style={{ width: column.width }}>
                {column.sortable ? (
                  <SortButton
                    onClick={() => handleSort(column.key)}
                    $active={sortConfig?.key === column.key}
                    $direction={sortConfig?.direction}
                  >
                    {column.header}
                  </SortButton>
                ) : (
                  column.header
                )}
              </TableHead>
            ))}
          </TableRow>
        </TableHeader>
        <TableBody>
          {paginatedData.map((row, index) => (
            <TableRow
              key={index}
              $clickable={!!onRowClick}
              onClick={() => handleRowClick(row)}
            >
              {columns.map((column) => (
                <TableCell key={column.key as string}>
                  {column.render
                    ? column.render(row[column.key], row)
                    : String(row[column.key])}
                </TableCell>
              ))}

-- Criação do schema raw (bronze layer)
CREATE SCHEMA IF NOT EXISTS raw;
COMMENT ON SCHEMA raw IS 'Camada bronze: dados brutos das fontes, sem transformação';

-- Tabela para logs de execução
CREATE TABLE IF NOT EXISTS raw.execution_logs (
    log_id UUID PRIMARY KEY,
    execution_id VARCHAR(255) NOT NULL,
    agent_id VARCHAR(100) NOT NULL,
    agent_type VARCHAR(50) NOT NULL,
    task_type VARCHAR(100),
    input_tokens INTEGER,
    output_tokens INTEGER,
    total_tokens INTEGER,
    latency_ms INTEGER NOT NULL,
    success BOOLEAN NOT NULL,
    error_message TEXT,
    start_time TIMESTAMP WITH TIME ZONE NOT NULL,
    end_time TIMESTAMP WITH TIME ZONE NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    source_system VARCHAR(50) NOT NULL,
    raw_data JSONB
);

CREATE INDEX idx_raw_execution_logs_agent_id ON raw.execution_logs(agent_id);
CREATE INDEX idx_raw_execution_logs_start_time ON raw.execution_logs(start_time);
CREATE INDEX idx_raw_execution_logs_success ON raw.execution_logs(success);

-- Tabela para uso de agentes
CREATE TABLE IF NOT EXISTS raw.agent_usage (
    usage_id UUID PRIMARY KEY,
    user_id VARCHAR(100),
    session_id VARCHAR(255) NOT NULL,
    agent_id VARCHAR(100) NOT NULL,
    agent_type VARCHAR(50) NOT NULL,
    feature_name VARCHAR(100),
    action_type VARCHAR(50),
    duration_seconds INTEGER,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    metadata JSONB
);

CREATE INDEX idx_raw_agent_usage_agent_id ON raw.agent_usage(agent_id);
CREATE INDEX idx_raw_agent_usage_created_at ON raw.agent_usage(created_at);

-- Tabela para chamadas LLM
CREATE TABLE IF NOT EXISTS raw.llm_calls (
    call_id UUID PRIMARY KEY,
    execution_id VARCHAR(255),
    provider VARCHAR(50) NOT NULL,
    model VARCHAR(100) NOT NULL,
    prompt_tokens INTEGER NOT NULL,
    completion_tokens INTEGER NOT NULL,
    total_tokens INTEGER NOT NULL,
    cost_usd DECIMAL(10,6),
    latency_ms INTEGER,
    success BOOLEAN NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    request_metadata JSONB,
    response_metadata JSONB
);

CREATE INDEX idx_raw_llm_calls_provider ON raw.llm_calls(provider);
CREATE INDEX idx_raw_llm_calls_created_at ON raw.llm_calls(created_at);

-- Tabela para dados de custo
CREATE TABLE IF NOT EXISTS raw.cost_data (
    cost_id UUID PRIMARY KEY,
    provider VARCHAR(50) NOT NULL,
    service_type VARCHAR(50) NOT NULL,
    resource_id VARCHAR(255),
    quantity DECIMAL(10,4),
    unit VARCHAR(20),
    cost_usd DECIMAL(10,6) NOT NULL,
    period_start TIMESTAMP WITH TIME ZONE NOT NULL,
    period_end TIMESTAMP WITH TIME ZONE NOT NULL,
    tags JSONB,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_raw_cost_data_provider ON raw.cost_data(provider);
CREATE INDEX idx_raw_cost_data_period ON raw.cost_data(period_start);

-- Criação do schema staging (silver layer)
CREATE SCHEMA IF NOT EXISTS staging;

-- Criação do schema marts (gold layer)

CREATE SCHEMA IF NOT EXISTS product;

            ELSE created_at

const jwt = require('jsonwebtoken');
const crypto = require('crypto');
const logger = require('../../utils/logger');

class JWTSecurity {
  constructor() {
    // Em produção, usar variáveis de ambiente com segredos diferentes
    this.accessSecret = process.env.JWT_ACCESS_SECRET || crypto.randomBytes(64).toString('hex');
    this.refreshSecret = process.env.JWT_REFRESH_SECRET || crypto.randomBytes(64).toString('hex');
    this.accessExpiry = '15m'; // 15 minutos - padrão OWASP
    this.refreshExpiry = '7d'; // 7 dias
    this.tokenBlacklist = new Set(); // Em produção, usar Redis
  }

  // Gerar tokens
  generateTokens(user) {
    const accessToken = jwt.sign(
      {
        userId: user.id,
        email: user.email,
        role: user.role,
        type: 'access'
      },
      this.accessSecret,
      { expiresIn: this.accessExpiry }
    );

    const refreshToken = jwt.sign(
      {
        userId: user.id,
        type: 'refresh'
      },
      this.refreshSecret,
      { expiresIn: this.refreshExpiry }
    );

    // Hash do refresh token para armazenamento seguro
    const refreshTokenHash = crypto
      .createHash('sha256')
      .update(refreshToken)
      .digest('hex');

    logger.info(`Tokens generated for user ${user.id}`);

    return {
      accessToken,
      refreshToken,
      refreshTokenHash,
      expiresIn: 15 * 60 // 15 minutos em segundos
    };
  }

  // Verificar access token
  verifyAccessToken(token) {
    try {
      const decoded = jwt.verify(token, this.accessSecret);
      
      // Verificar se token está na blacklist
      if (this.tokenBlacklist.has(token)) {
        throw new Error('Token revoked');
      }

      // Verificar tipo do token
      if (decoded.type !== 'access') {
        throw new Error('Invalid token type');
      }

      return decoded;
    } catch (error) {
      logger.warn(`Access token verification failed: ${error.message}`);
      throw error;
    }
  }

  // Verificar refresh token
  verifyRefreshToken(token) {
    try {
      const decoded = jwt.verify(token, this.refreshSecret);
      
      if (decoded.type !== 'refresh') {
        throw new Error('Invalid token type');
      }

      return decoded;
    } catch (error) {
      logger.warn(`Refresh token verification failed: ${error.message}`);
      throw error;
    }
  }

  // Revogar token (logout)
  revokeToken(token) {
    try {
      const decoded = jwt.decode(token);
      const timeToExpiry = decoded.exp * 1000 - Date.now();
      
      if (timeToExpiry > 0) {
        this.tokenBlacklist.add(token);
        
        // Remover da blacklist após expiração
        setTimeout(() => {
          this.tokenBlacklist.delete(token);
        }, timeToExpiry);
      }
      
      logger.info(`Token revoked for user ${decoded.userId}`);
      return true;
    } catch (error) {
      logger.error(`Token revocation failed: ${error.message}`);
      return false;
    }
  }

  // Rotação de refresh tokens
  rotateRefreshToken(oldRefreshToken, user) {
    // Verificar token antigo
    const decoded = this.verifyRefreshToken(oldRefreshToken);
    
    // Revogar token antigo
    this.revokeToken(oldRefreshToken);
    
    // Gerar novos tokens
    return this.generateTokens(user);
  }
}

const rateLimit = require('express-rate-limit');
const logger = require('../../utils/logger');

class RateLimiter {
  constructor() {
    this.limits = {
      // Limites por tipo de endpoint
      auth: this.createLimiter({
        windowMs: 15 * 60 * 1000, // 15 minutos
        max: 5, // 5 tentativas por IP
        message: 'Too many login attempts, please try again later',
        skipSuccessfulRequests: true
      }),

      api: this.createLimiter({
        windowMs: 15 * 60 * 1000, // 15 minutos
        max: 100, // 100 requests por IP
        message: 'Too many requests from this IP'
      }),

      critical: this.createLimiter({
        windowMs: 60 * 60 * 1000, // 1 hora
        max: 10, // 10 requests por IP
        message: 'Too many critical operations'
      })
    };
  }

  createLimiter(options) {
    return rateLimit({
      ...options,
      standardHeaders: true,
      legacyHeaders: false,
      handler: (req, res) => {
        logger.warn(`Rate limit exceeded for IP: ${req.ip}, Path: ${req.path}`);
        res.status(429).json({
          error: options.message,
          retryAfter: Math.ceil(options.windowMs / 1000)
        });
      },
      skip: (req) => {
        // Pular rate limiting para IPs confiáveis (ex: serviços internos)
        const trustedIPs = process.env.TRUSTED_IPS ? 
          process.env.TRUSTED_IPS.split(',') : [];
        return trustedIPs.includes(req.ip);
      }
    });
  }

  // Middleware específico para endpoints
  forAuth() {
    return this.limits.auth;
  }

  forAPI() {
    return this.limits.api;
  }

  forCritical() {
    return this.limits.critical;
  }

  // Rate limiting dinâmico baseado em risco
  dynamicLimiter(userRiskScore) {
    const baseWindow = 15 * 60 * 1000; // 15 minutos
    const baseMax = 100;
    
    // Reduzir limite para usuários de alto risco
    const maxRequests = Math.max(10, baseMax - (userRiskScore * 10));
    
    return this.createLimiter({
      windowMs: baseWindow,
      max: maxRequests,
      message: 'Rate limit exceeded based on risk assessment'
    });
  }
}

const crypto = require('crypto');
const CryptoJS = require('crypto-js');
const logger = require('../../utils/logger');

class EncryptionService {
  constructor() {
    // Chaves devem vir de variáveis de ambiente em produção
    this.encryptionKey = process.env.ENCRYPTION_KEY || 
      crypto.randomBytes(32).toString('hex'); // AES-256
    this.ivLength = 16; // Para AES
  }

  // Criptografar dados sensíveis (PII)
  encryptSensitiveData(data) {
    try {
      if (typeof data !== 'string') {
        data = JSON.stringify(data);
      }

      const iv = crypto.randomBytes(this.ivLength);
      const cipher = crypto.createCipheriv(
        'aes-256-gcm',
        Buffer.from(this.encryptionKey, 'hex'),
        iv
      );

      let encrypted = cipher.update(data, 'utf8', 'hex');
      encrypted += cipher.final('hex');
      const authTag = cipher.getAuthTag();

      return {
        encryptedData: encrypted,
        iv: iv.toString('hex'),
        authTag: authTag.toString('hex'),
        algorithm: 'aes-256-gcm'
      };
    } catch (error) {
      logger.error(`Encryption failed: ${error.message}`);
      throw new Error('Data encryption failed');
    }
  }

  // Descriptografar dados
  decryptSensitiveData(encryptedObject) {
    try {
      const iv = Buffer.from(encryptedObject.iv, 'hex');
      const authTag = Buffer.from(encryptedObject.authTag, 'hex');
      const encryptedText = encryptedObject.encryptedData;

      const decipher = crypto.createDecipheriv(
        'aes-256-gcm',
        Buffer.from(this.encryptionKey, 'hex'),
        iv
      );

      decipher.setAuthTag(authTag);

      let decrypted = decipher.update(encryptedText, 'hex', 'utf8');
      decrypted += decipher.final('utf8');

      // Tentar parsear como JSON, senão retornar string
      try {
        return JSON.parse(decrypted);
      } catch {
        return decrypted;
      }
    } catch (error) {
      logger.error(`Decryption failed: ${error.message}`);
      throw new Error('Data decryption failed');
    }
  }

  // Hash para dados que não precisam ser descriptografados (ex: senhas)
  async hashPassword(password) {
    try {
      const saltRounds = 12; // OWASP recomenda no mínimo 10
      const bcrypt = require('bcrypt');
      return await bcrypt.hash(password, saltRounds);
    } catch (error) {
      logger.error(`Password hashing failed: ${error.message}`);
      throw new Error('Password hashing failed');
    }
  }

  // Verificar senha
  async verifyPassword(password, hash) {
    try {
      const bcrypt = require('bcrypt');
      return await bcrypt.compare(password, hash);
    } catch (error) {
      logger.error(`Password verification failed: ${error.message}`);
      throw new Error('Password verification failed');
    }
  }

  // Mascarar dados para logs (ex: CPF, email)
  maskSensitiveData(data, type) {
    switch (type) {
      case 'email':
        const [local, domain] = data.split('@');
        return `${local.substring(0, 2)}***@${domain}`;
      
      case 'cpf':
        return data.replace(/(\d{3})(\d{3})(\d{3})(\d{2})/, '$1.***.***-**');
      
      case 'creditCard':
        return data.replace(/(\d{4})(\d{4})(\d{4})(\d{4})/, '**** **** **** $4');
      
      case 'phone':
        return data.replace(/(\d{2})(\d{4,5})(\d{4})/, '($1) *****-$3');
      
      default:
        // Mascarar genericamente mantendo primeiro e último caractere
        if (data.length <= 2) return '**';
        return `${data[0]}${'*'.repeat(data.length - 2)}${data[data.length - 1]}`;
    }
  }

Cypress.Commands.add('getWithRetry', (selector, options = {}) => {
  const { timeout = 10000, interval = 1000 } = options
  
  return cy.get('body', { log: false }).then(($body) => {
    const startTime = Date.now()
    
    const tryGet = () => {
      const element = $body.find(selector)
      if (element.length > 0) {
        return cy.wrap(element)
      }
      
      if (Date.now() - startTime > timeout) {
        throw new Error(`Elemento ${selector} não encontrado após ${timeout}ms`)
      }
      
      return cy.wait(interval, { log: false }).then(tryGet)
    }
    
    return tryGet()
  })
})

const express = require('express');
const cors = require('cors');
const bodyParser = require('body-parser');
const { config, validateConfig } = require('./config');
const logger = require('./logger');
const { initializeIntegrations } = require('../integrations');
const { initializeAI } = require('../ai/flow-suggester');

class DIYAPP {
  constructor() {
    this.app = express();
    this.server = null;
    this.port = config.app.port;
    
    this.setupMiddleware();
    this.setupRoutes();
    this.setupErrorHandling();
  }
  
  setupMiddleware() {
    this.app.use(cors());
    this.app.use(bodyParser.json());
    this.app.use(bodyParser.urlencoded({ extended: true }));
    this.app.use(express.static('src/public/dashboard'));
    
    // Logging middleware
    this.app.use((req, res, next) => {
      logger.info(`${req.method} ${req.url}`, {
        ip: req.ip,
        userAgent: req.get('User-Agent')
      });
      next();
    });
  }
  
  setupRoutes() {
    // Health check
    this.app.get('/health', (req, res) => {
      res.json({ 
        status: 'healthy', 
        timestamp: new Date().toISOString(),
        version: 'v11-core'
      });
    });
    
    // API routes
    this.app.use('/api/integrations', require('../integrations/routes'));
    this.app.use('/api/ai', require('../ai/routes'));
    
    // Dashboard route
    this.app.get('/dashboard', (req, res) => {
      res.sendFile('index.html', { root: 'src/public/dashboard' });
    });
  }
  
  setupErrorHandling() {
    this.app.use((err, req, res, next) => {
      logger.error('Unhandled error:', err);
      res.status(err.status || 500).json({
        error: {
          message: err.message || 'Internal server error',
          ...(config.app.env === 'development' && { stack: err.stack })
        }
      });
    });
  }
  
  async initialize() {
    try {
      validateConfig();
      logger.info('Starting DIYAPP V11 Core...');
      
      // Initialize AI module
      await initializeAI();
      logger.info('AI module initialized');
      
      // Initialize integrations
      await initializeIntegrations();
      logger.info('Integrations initialized');
      
      // Start server
      this.server = this.app.listen(this.port, () => {
        logger.info(`DIYAPP V11 Core running on port ${this.port}`);
        logger.info(`Dashboard available at http://localhost:${this.port}/dashboard`);
      });
      
      // Graceful shutdown
      process.on('SIGTERM', () => this.shutdown());
      process.on('SIGINT', () => this.shutdown());
      
    } catch (error) {
      logger.error('Failed to initialize DIYAPP:', error);
      process.exit(1);
    }
  }
  
  async shutdown() {
    logger.info('Shutting down DIYAPP V11 Core...');
    
    if (this.server) {
      this.server.close(() => {
        logger.info('HTTP server closed');
        process.exit(0);
      });
    }
    
    // Force exit after 10 seconds
    setTimeout(() => {
      logger.error('Forced shutdown after timeout');
      process.exit(1);
    }, 10000);
  }
}

// Start application if run directly
if (require.main === module) {
  const app = new DIYAPP();

}

          debug: () => {},
          info: (message) => logger.info(`GitHub: ${message}`),
          warn: (message) => logger.warn(`GitHub: ${message}`),
          error: (message) => logger.error(`GitHub: ${message}`)
        }
      });
      
      // Test connection
      const { data: user } = await this.octokit.users.getAuthenticated();
      logger.info(`GitHub integration initialized for user: ${user.login}`);
      
      this.initialized = true;
      
    } catch (error) {
      logger.error('Failed to initialize GitHub integration:', error);
      throw error;
    }
  }
  
  async createRepository(name, description = '', isPrivate = true) {
    if (!this.initialized) {
      throw new Error('GitHub integration not initialized');
    }
    
    try {
      const response = await this.octokit.repos.createForAuthenticatedUser({
        name,
        description,
        private: isPrivate,
        auto_init: true,
        license_template: 'mit'
      });
      
      logger.info(`GitHub repository created: ${name}`);
      return response.data;
    } catch (error) {
      logger.error('Error creating GitHub repository:', error);
      throw error;
    }
  }
  
  async createIssue(repoOwner, repoName, title, body, labels = []) {
    if (!this.initialized) {
      throw new Error('GitHub integration not initialized');
    }
    
    try {
      const response = await this.octokit.issues.create({
        owner: repoOwner,
        repo: repoName,
        title,
        body,
        labels
      });
      
      logger.info(`Issue created in ${repoOwner}/${repoName}: ${title}`);
      return response.data;
    } catch (error) {
      logger.error('Error creating GitHub issue:', error);
      throw error;
    }
  }
  
  async createWebhook(repoOwner, repoName, webhookUrl, events = ['push', 'issues', 'pull_request']) {
    if (!this.initialized) {
      throw new Error('GitHub integration not initialized');
    }
    
    try {
      const response = await this.octokit.repos.createWebhook({
        owner: repoOwner,
        repo: repoName,
        config: {
          url: webhookUrl,
          content_type: 'json',
          secret: config.github.webhookSecret
        },
        events,
        active: true
      });
      
      logger.info(`Webhook created for ${repoOwner}/${repoName}`);
      return response.data;
    } catch (error) {
      logger.error('Error creating GitHub webhook:', error);
      throw error;
    }
  }

    if (!this.initialized) {
      throw new Error('GitHub integration not initialized');
    }
    
    try {
      const response = await this.octokit.repos.getContent({
        owner: repoOwner,
        repo: repoName,
        path
      });
      
      return response.data;
    } catch (error) {
      logger.error('Error getting repository contents:', error);
      throw error;
    }
  }
}
