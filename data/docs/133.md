const fs = require('fs').promises;
const path = require('path');

class StabilityMonitor {
    constructor() {
        this.kpiPath = path.join(__dirname, '../data/kpi_log.json');
        this.init();
    }

    async init() {
        try {
            await fs.mkdir(path.dirname(this.kpiPath), { recursive: true });
            const exists = await fs.access(this.kpiPath).then(() => true).catch(() => false);
            if (!exists) {
                await this.resetKPIs();
            }
        } catch (error) {
            console.error('Erro inicializando monitor:', error);
        }
    }

    async resetKPIs() {
        const baseKPIs = {
            uptime_30d: 99.99,
            daily_interventions: 0,
            auto_heal_success_rate: 95,
            mttr_minutes: 5,
            last_updated: new Date().toISOString(),
            daily_log: []
        };
        await this.saveKPIs(baseKPIs);
    }

    async recordIncident(type, resolved = true, duration_min = 0) {
        const kpis = await this.loadKPIs();
        
        // Atualiza MTTR
        if (resolved && duration_min > 0) {
            kpis.mttr_minutes = (kpis.mttr_minutes * 0.9 + duration_min * 0.1).toFixed(1);
        }
        
        // Conta intervenГ§Гµes
        if (!resolved) {
            kpis.daily_interventions += 1;
        }
        
        // Log do evento
        kpis.daily_log.unshift({
            timestamp: new Date().toISOString(),
            type,
            resolved,
            duration_min
        });
        
        // MantГ©m apenas Гєltimos 100 eventos
        kpis.daily_log = kpis.daily_log.slice(0, 100);
        
        kpis.last_updated = new Date().toISOString();
        await this.saveKPIs(kpis);
        
        // Dispara auto-cura se necessГЎrio
        if (!resolved) {
            this.triggerAutoHeal(type);
        }
        
        return kpis;
    }

    async triggerAutoHeal(incidentType) {
        const healActions = {
            'high_memory': 'Reiniciando container afetado...',
            'db_timeout': 'Aumentando timeout e recriando conexГЈo...',
            'api_slow': 'Habilitando cache e escalando instГўncia...',
            'disk_full': 'Limpando arquivos temporГЎrios...'
        };
        
        const action = healActions[incidentType] || 'Executando diagnГіstico genГ©rico...';
        console.log(`рџљ‘ Auto-cura acionada: ${action}`);
        
        // Simula aГ§ГЈo de cura
        await new Promise(resolve => setTimeout(resolve, 2000));
        
        // Registra como resolvido
        await this.recordIncident(`healed_${incidentType}`, true, 2);
        
        return { success: true, action };
    }

    async getDashboardData() {
        const kpis = await this.loadKPIs();
        
        // Calcula uptime baseado em incidentes dos Гєltimos 30 dias
        const thirtyDaysAgo = new Date();
        thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
        
        const recentIncidents = kpis.daily_log.filter(entry => 
            new Date(entry.timestamp) > thirtyDaysAgo && !entry.resolved
        ).length;
        
        const uptime = 100 - (recentIncidents * 0.001); // SimulaГ§ГЈo
        kpis.uptime_30d = Math.max(99.9, uptime).toFixed(3);
        
        // Taxa de sucesso de auto-cura
        const healEvents = kpis.daily_log.filter(e => e.type.startsWith('healed_'));
        kpis.auto_heal_success_rate = healEvents.length > 0 ? 
            Math.min(99, (healEvents.length / kpis.daily_log.length) * 100).toFixed(1) : 95;
        
        return kpis;
    }

    async loadKPIs() {
        try {
            const data = await fs.readFile(this.kpiPath, 'utf8');
            return JSON.parse(data);
        } catch {
            return await this.resetKPIs();
        }
    }

    async saveKPIs(data) {
        await fs.writeFile(this.kpiPath, JSON.stringify(data, null, 2));
    }
}

// Health Check AutomГЎtico
setInterval(async () => {
    const monitor = new StabilityMonitor();
    const health = await monitor.getDashboardData();

        // Aqui integraria com sistemas reais de monitoramento
    }
}, 300000); // A cada 5 minutos

module.exports = StabilityMonitor;

    errorRate: number;

  target: string;
  reason: string;
  severity: 'low' | 'medium' | 'high' | 'critical';

  private kafka: Kafka;
  private producer: Producer;
  private consumer: Consumer;
  private circuitBreakers: Map<string, CircuitBreaker>;
  private healthStatus: Map<string, HealthStatus>;

    super();
    
    this.kafka = new Kafka({
      clientId: 'auto-healing-orchestrator',
      brokers: kafkaConfig.brokers,
      ssl: kafkaConfig.ssl,
      sasl: kafkaConfig.sasl,
    });
    
    this.producer = this.kafka.producer();
    this.consumer = this.kafka.consumer({ groupId: 'healing-group' });
    this.circuitBreakers = new Map();
    this.healthStatus = new Map();
    
    // Initialize metrics
    const meter = metrics.getMeter('auto-healing');
    this.healingActionsCounter = meter.createCounter('healing_actions_total', {
      description: 'Total number of healing actions executed',
    });
    
    this.serviceHealthGauge = meter.createGauge('service_health_score', {
      description: 'Health score of services (0-100)',
    });
    
    this.healingLatencyHistogram = meter.createHistogram('healing_action_latency_ms', {
      description: 'Latency of healing actions in milliseconds',
      unit: 'ms',
    });
  }
  
  async initialize(): Promise<void> {
    await this.producer.connect();
    await this.consumer.connect();
    
    // Subscribe to health check topics
    await this.consumer.subscribe({ 
      topic: 'service-health', 
      fromBeginning: false 
    });
    
    await this.consumer.subscribe({ 
      topic: 'system-alerts', 
      fromBeginning: false 
    });
    
    // Start consuming messages
    await this.consumer.run({
      eachMessage: async ({ topic, partition, message }) => {
        await this.processMessage(topic, message);
      },
    });
    
    // Start periodic health evaluation
    setInterval(() => this.evaluateSystemHealth(), 30000); // Every 30 seconds
    
    console.log('AutoHealingOrchestrator initialized');
  }
  
  private async processMessage(topic: string, message: any): Promise<void> {
    const startTime = Date.now();
    
    try {
      const data = JSON.parse(message.value.toString());
      
      switch (topic) {
        case 'service-health':
          await this.updateHealthStatus(data);
          break;
          
        case 'system-alerts':
          await this.handleAlert(data);
          break;
          
        default:
          console.warn(`Unknown topic: ${topic}`);
      }
      
      // Record latency
      this.healingLatencyHistogram.record(Date.now() - startTime, {
        topic,
        success: 'true',
      });
      
    } catch (error) {
      console.error('Error processing message:', error);
      this.healingLatencyHistogram.record(Date.now() - startTime, {
        topic,
        success: 'false',
        error: error.message,
      });
    }
  }
  
  private async updateHealthStatus(status: HealthStatus): Promise<void> {
    this.healthStatus.set(status.service, status);
    
    // Update metrics
    const healthScore = this.calculateHealthScore(status);
    this.serviceHealthGauge.record(healthScore, {
      service: status.service,
      status: status.status,
    });

    if (status.status === 'unhealthy' || status.metrics.errorRate > 0.1) {

        target: status.service,

        severity: 'high',

      });
    }
  }
  
  private async handleAlert(alert: any): Promise<void> {
    const { type, service, message, severity } = alert;

      case 'HIGH_LATENCY':
        healingAction = {
          action: 'scale',
          target: service,
          reason: `High latency detected: ${message}`,
          severity: severity || 'medium',

          action: 'restart',
          target: service,
          reason: 'Out of memory error',
          severity: 'critical',
          timestamp: new Date(),
        };
        break;

        healingAction = {
          action: 'fallback',
          target: service,
          reason: 'Database connection failure',
          severity: 'high',
          timestamp: new Date(),
        };
        break;
        
      case 'DEPENDENCY_FAILURE':
        healingAction = {
          action: 'circuit_break',
          target: service,
          reason: `Dependency failure: ${message}`,
          severity: 'high',
          timestamp: new Date(),
        };
        break;
    }
    
    if (healingAction) {
      await this.triggerHealingAction(healingAction);
    }
  }

              'x-timestamp': new Date().toISOString(),
            },

        action: action.action,
        target: action.target,
        severity: action.severity,
      });

        action: action.action,
        success: 'true',
      });
      
    } catch (error) {
      console.error('Failed to trigger healing action:', error);
      
      // Record failure latency
      this.healingLatencyHistogram.record(Date.now() - startTime, {
        action: action.action,
        success: 'false',
        error: error.message,
      });
    }

    if (status.metrics.latency > 5000) score -= 30;
    
    return Math.max(0, score);

from sqlalchemy import create_engine, text
from sqlalchemy.orm import sessionmaker, declarative_base, Session
from sqlalchemy.exc import SQLAlchemyError, OperationalError
from contextlib import contextmanager
import os
from typing import Generator, Optional
from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type
import time
from circuitbreaker import circuit

Base = declarative_base()

class DatabaseConnection:
    def __init__(self):
        self._engine = None
        self._session_factory = None
        self._connection_pool = {}
        
    def initialize(self, database_url: str, pool_size: int = 20, max_overflow: int = 10):
        """Initialize database connection with connection pooling"""
        try:
            self._engine = create_engine(
                database_url,
                pool_size=pool_size,
                max_overflow=max_overflow,
                pool_pre_ping=True,
                pool_recycle=3600,
                echo=False,
                connect_args={
                    "connect_timeout": 10,
                    "application_name": "diyapp_v12"
                }
            )
            
            # Test connection
            with self._engine.connect() as conn:
                conn.execute(text("SELECT 1"))
            
            self._session_factory = sessionmaker(
                bind=self._engine,
                autocommit=False,
                autoflush=False,
                expire_on_commit=False
            )
            
            logger.info("database_initialized", 
                       pool_size=pool_size, 
                       max_overflow=max_overflow)
            
        except Exception as e:
            logger.error("database_initialization_failed", error=str(e))
            raise
    
    @circuit(failure_threshold=5, expected_exception=OperationalError)
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=4, max=10),
        retry=retry_if_exception_type(OperationalError)
    )
    def get_session(self) -> Session:
        """Get database session with circuit breaker and retry logic"""
        if not self._session_factory:
            raise RuntimeError("Database not initialized")
        
        session = self._session_factory()
        session.connection(execution_options={"isolation_level": "READ COMMITTED"})
        return session
    
    @contextmanager
    def session_scope(self, correlation_id: Optional[str] = None) -> Generator[Session, None, None]:
        """Context manager for database sessions with automatic rollback on error"""
        session = self.get_session()
        try:
            if correlation_id:
                session.info["correlation_id"] = correlation_id
            
            yield session
            session.commit()
            logger.info("database_session_committed", 
                       correlation_id=correlation_id or "unknown")
            
        except SQLAlchemyError as e:
            session.rollback()
            logger.error("database_session_rolled_back",
                        error=str(e),
                        correlation_id=correlation_id or "unknown")
            raise
        finally:
            session.close()
    
    def health_check(self) -> Dict[str, Any]:
        """Health check for database connectivity"""
        start_time = time.time()
        try:
            with self._engine.connect() as conn:
                result = conn.execute(text("SELECT 1 as status, pg_database_size(current_database()) as db_size"))
                row = result.fetchone()
                
                latency_ms = (time.time() - start_time) * 1000
                
                return {
                    "status": "healthy",
                    "latency_ms": round(latency_ms, 2),
                    "database_size_bytes": row.db_size if row else 0,
                    "pool_status": {
                        "checked_out": self._engine.pool.checkedout(),
                        "checked_in": self._engine.pool.checkedin(),
                        "overflow": self._engine.pool.overflow()
                    }
                }
        except Exception as e:
            latency_ms = (time.time() - start_time) * 1000
            return {
                "status": "unhealthy",
                "latency_ms": round(latency_ms, 2),
                "error": str(e)
            }

        except Exception as e:

            raise
    
    @circuit(failure_threshold=3, expected_exception=(redis.RedisError, TimeoutError))
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=2, max=10)
    )
    def enqueue_task(self, task_name: str, args: tuple = None, kwargs: Dict[str, Any] = None,
                    correlation_id: Optional[str] = None, priority: int = 0) -> str:
        """Enqueue a task with retry and circuit breaker logic"""
        start_time = time.time()
        
        if not self._celery_app:
            raise RuntimeError("Queue system not initialized")
        
        try:
            task_kwargs = kwargs or {}
            if correlation_id:
                task_kwargs['correlation_id'] = correlation_id
            
            # Add to Redis for tracking
            task_id = f"task_{int(time.time())}_{correlation_id or 'unknown'}"
            
            task_info = {
                "task_name": task_name,
                "args": args or [],
                "kwargs": task_kwargs,
                "enqueued_at": time.time(),
                "correlation_id": correlation_id,
                "priority": priority
            }
            
            # Store task metadata in Redis with expiry
            self._redis_client.setex(
                f"task:meta:{task_id}",
                86400,  # 24 hours
                json.dumps(task_info)
            )
            
            # Send to Celery
            result = self._celery_app.send_task(
                task_name,
                args=args,
                kwargs=task_kwargs,
                task_id=task_id,
                priority=priority
            )
            
            latency_ms = (time.time() - start_time) * 1000
            
            logger.info("task_enqueued",
                       task_id=task_id,
                       task_name=task_name,
                       correlation_id=correlation_id,
                       latency_ms=round(latency_ms, 2))
            
            return task_id
            
        except Exception as e:
            latency_ms = (time.time() - start_time) * 1000
            logger.error("task_enqueue_failed",
                        task_name=task_name,
                        correlation_id=correlation_id,
                        error=str(e),
                        latency_ms=round(latency_ms, 2))
            raise
    
    def get_task_status(self, task_id: str) -> Dict[str, Any]:
        """Get status of a task"""
        try:
            if not self._celery_app:
                return {"status": "error", "message": "Queue system not initialized"}
            
            # Get from Redis metadata
            meta_data = self._redis_client.get(f"task:meta:{task_id}")
            meta = json.loads(meta_data) if meta_data else {}
            
            # Get Celery result
            result = self._celery_app.AsyncResult(task_id)
            
            response = {
                "task_id": task_id,
                "status": result.state,
                "meta": meta
            }
            
            if result.ready():
                if result.successful():
                    response["result"] = result.result
                else:
                    response["error"] = str(result.info)
            
            return response
            
        except Exception as e:
            logger.error("get_task_status_failed", task_id=task_id, error=str(e))
            return {"status": "error", "message": str(e)}
    
    def health_check(self) -> Dict[str, Any]:
        """Health check for queue system"""
        checks = {}
        
        # Check Redis
        redis_start = time.time()
        try:
            redis_ping = self._redis_client.ping()
            redis_latency = (time.time() - redis_start) * 1000
            checks["redis"] = {
                "status": "healthy" if redis_ping else "unhealthy",
                "latency_ms": round(redis_latency, 2)
            }
        except Exception as e:
            checks["redis"] = {
                "status": "unhealthy",
                "error": str(e),
                "latency_ms": round((time.time() - redis_start) * 1000, 2)
            }
        
        # Check Celery worker (via control inspect)
        celery_start = time.time()
        try:
            inspector = self._celery_app.control.inspect()
            workers = inspector.ping()
            celery_latency = (time.time() - celery_start) * 1000
            
            checks["celery"] = {
                "status": "healthy" if workers else "unhealthy",
                "latency_ms": round(celery_latency, 2),
                "active_workers": len(workers) if workers else 0
            }
        except Exception as e:
            checks["celery"] = {
                "status": "unhealthy",
                "error": str(e),
                "latency_ms": round((time.time() - celery_start) * 1000, 2)
            }
        
        # Overall status
        all_healthy = all(check["status"] == "healthy" for check in checks.values())
        
        return {
            "status": "healthy" if all_healthy else "degraded",
            "components": checks,
            "metrics": self._queue_metrics
        }
    
    def register_task(self, func: Callable) -> Callable:
        """Decorator to register a function as a Celery task"""
        if not self._celery_app:
            raise RuntimeError("Queue system not initialized")
        
        @wraps(func)
        def wrapper(*args, **kwargs):
            start_time = time.time()
            correlation_id = kwargs.pop('correlation_id', None)
            
            try:
                logger.info("task_started",
                           task_name=func.__name__,
                           correlation_id=correlation_id)
                
                result = func(*args, **kwargs)
                
                processing_time = time.time() - start_time
                self._queue_metrics["tasks_processed"] += 1
                
                # Update average processing time (exponential moving average)
                old_avg = self._queue_metrics["average_processing_time"]
                self._queue_metrics["average_processing_time"] = (
                    0.9 * old_avg + 0.1 * processing_time
                )
                
                logger.info("task_completed",
                           task_name=func.__name__,
                           correlation_id=correlation_id,
                           processing_time_seconds=round(processing_time, 2))
                
                return result
                
            except Exception as e:
                self._queue_metrics["tasks_failed"] += 1
                logger.error("task_failed",
                            task_name=func.__name__,
                            correlation_id=correlation_id,
                            error=str(e))
                raise
        
        # Register with Celery
        return self._celery_app.task(wrapper)
