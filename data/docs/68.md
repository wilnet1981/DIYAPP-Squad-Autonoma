### 2.2. Template de Serviço (services/template-service/src/index.ts)
```typescript
import express from 'express';
import helmet from 'helmet';
import cors from 'cors';
import rateLimit from 'express-rate-limit';
import { createLogger, format, transports } from 'winston';
import { errorHandler } from './middleware/error-handler';
import { requestLogger } from './middleware/request-logger';
import { healthRouter } from './routes/health';
import { apiRouter } from './routes/api';

// Configuração do logger
const logger = createLogger({
  level: process.env.LOG_LEVEL || 'info',
  format: format.combine(
    format.timestamp(),
    format.errors({ stack: true }),
    format.json()
  ),
  transports: [
    new transports.Console(),
    new transports.File({ filename: 'logs/error.log', level: 'error' }),
    new transports.File({ filename: 'logs/combined.log' })
  ]
});

const app = express();
const PORT = process.env.PORT || 3000;

// Middleware de segurança
app.use(helmet());
app.use(cors({
  origin: process.env.ALLOWED_ORIGINS?.split(',') || [],
  credentials: true
}));

// Rate limiting
const limiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 minutos
  max: 100, // limite por IP
  message: 'Too many requests from this IP'
});
app.use('/api', limiter);

// Logging
app.use(requestLogger(logger));

// Parsers
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true }));

// Rotas
app.use('/health', healthRouter);
app.use('/api/v1', apiRouter);

// 404 handler
app.use((req, res) => {
  res.status(404).json({
    error: 'Not Found',
    message: `Route ${req.method} ${req.path} not found`
  });
});

// Error handler
app.use(errorHandler(logger));

// Inicialização
if (require.main === module) {
  app.listen(PORT, () => {
    logger.info(`Service running on port ${PORT}`);
  });
}

export { app, logger };
```

### 2.3. Health Check Controller (services/template-service/src/controllers/health.controller.ts)
```typescript
import { Request, Response } from 'express';
import { Redis } from 'ioredis';
import { Pool } from 'pg';

export class HealthController {
  constructor(
    private redisClient: Redis,
    private dbPool: Pool
  ) {}

  async getHealth(req: Request, res: Response) {
    const checks = {
      api: 'healthy',
      database: await this.checkDatabase(),
      redis: await this.checkRedis(),
      timestamp: new Date().toISOString(),
      uptime: process.uptime(),
      memory: process.memoryUsage()
    };

    const isHealthy = checks.database === 'healthy' && checks.redis === 'healthy';
    const statusCode = isHealthy ? 200 : 503;

    res.status(statusCode).json({
      status: isHealthy ? 'healthy' : 'unhealthy',
      checks,
      version: process.env.npm_package_version || '1.0.0'
    });
  }

  private async checkDatabase(): Promise<'healthy' | 'unhealthy'> {
    try {
      const result = await this.dbPool.query('SELECT 1 as health');
      return result.rows[0]?.health === 1 ? 'healthy' : 'unhealthy';
    } catch (error) {
      return 'unhealthy';
    }
  }

  private async checkRedis(): Promise<'healthy' | 'unhealthy'> {
    try {
      await this.redisClient.ping();
      return 'healthy';
    } catch (error) {
      return 'unhealthy';
    }
  }
}
```

### 3.1. GitHub Actions (.github/workflows/ci-cd.yml)
```yaml
name: CI/CD Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        node-version: [18.x, 20.x]
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Use Node.js ${{ matrix.node-version }}
      uses: actions/setup-node@v3
      with:
        node-version: ${{ matrix.node-version }}
        cache: 'npm'
    
    - name: Install dependencies
      run: npm ci
    
    - name: Run linting
      run: npm run lint
    
    - name: Run tests
      run: npm test -- --coverage
    
    - name: Upload coverage
      uses: codecov/codecov-action@v3
      with:
        file: ./coverage/lcov.info
        fail_ci_if_error: true

  build:
    needs: test
    runs-on: ubuntu-latest
    if: github.event_name == 'push' && github.ref == 'refs/heads/main'
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Set up Docker Buildx
      uses: docker/setup-buildx-action@v2
    
    - name: Log in to Container Registry
      uses: docker/login-action@v2
      with:
        registry: ${{ env.REGISTRY }}
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}
    
    - name: Extract metadata
      id: meta
      uses: docker/metadata-action@v4
      with:
        images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
        tags: |
          type=sha,prefix={{branch}}-
          type=ref,event=branch
          type=ref,event=pr
          type=semver,pattern={{version}}
          type=semver,pattern={{major}}.{{minor}}
    
    - name: Build and push
      uses: docker/build-push-action@v4
      with:
        context: .
        push: true
        tags: ${{ steps.meta.outputs.tags }}
        labels: ${{ steps.meta.outputs.labels }}
        cache-from: type=gha
        cache-to: type=gha,mode=max

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.event_name == 'push' && github.ref == 'refs/heads/main'
    
    steps:
    - name: Deploy to Kubernetes
      uses: appleboy/ssh-action@v0.1.5
      with:
        host: ${{ secrets.K8S_HOST }}
        username: ${{ secrets.K8S_USER }}
        key: ${{ secrets.K8S_SSH_KEY }}
        script: |
          cd /opt/diyapp
          kubectl set image deployment/api-gateway api-gateway=${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.sha }}
          kubectl rollout status deployment/api-gateway --timeout=300s
```

### 3.2. Dockerfile Multistage
```dockerfile
# Build stage
FROM node:18-alpine AS builder

WORKDIR /app

# Copy package files
COPY package*.json ./
COPY tsconfig.json ./

# Install dependencies
RUN npm ci --only=production

# Copy source code
COPY src/ ./src/

# Build TypeScript
RUN npm run build

# Runtime stage
FROM node:18-alpine

WORKDIR /app

# Install runtime dependencies
RUN apk add --no-cache curl

# Copy built application
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/package.json ./

# Create non-root user
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nodejs -u 1001

USER nodejs

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:3000/health || exit 1

EXPOSE 3000

CMD ["node", "dist/index.js"]
```

import json

from typing import Any, Dict
from uuid import uuid4
import logging
from pythonjsonlogger import jsonlogger

class StructuredLogger:
    """Logger estruturado em JSON com correlation ID"""
    
    def __init__(self):
        self.logger = logging.getLogger(__name__)
        self.correlation_id = str(uuid4())
        
        # Configuração do handler
        log_handler = logging.StreamHandler()
        formatter = jsonlogger.JsonFormatter(
            '%(asctime)s %(levelname)s %(name)s %(message)s',
            json_ensure_ascii=False
        )
        log_handler.setFormatter(formatter)
        self.logger.addHandler(log_handler)
        self.logger.setLevel(getattr(logging, settings.LOG_LEVEL))
    
    def _create_log_record(self, level: str, message: str, **extra: Any) -> Dict[str, Any]:
        """Cria registro de log estruturado"""
        return {
            "level": level,
            "correlation_id": self.correlation_id,
            "timestamp": time.time(),
            "message": message,
            **extra
        }
    
    def info(self, message: str, **extra: Any):
        record = self._create_log_record("INFO", message, **extra)
        self.logger.info(json.dumps(record))
    
    def error(self, message: str, **extra: Any):
        record = self._create_log_record("ERROR", message, **extra)
        self.logger.error(json.dumps(record))
    
    def warning(self, message: str, **extra: Any):
        record = self._create_log_record("WARNING", message, **extra)
        self.logger.warning(json.dumps(record))
    
    def debug(self, message: str, **extra: Any):
        record = self._create_log_record("DEBUG", message, **extra)
        self.logger.debug(json.dumps(record))

security = HTTPBearer()
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

class AuthService:
    """Serviço de autenticação JWT"""
    
    def __init__(self):
        self.secret_key = settings.SECRET_KEY
        self.algorithm = settings.ALGORITHM
        self.logger = StructuredLogger()
    
    def verify_password(self, plain_password: str, hashed_password: str) -> bool:
        """Verifica senha com hash"""
        return pwd_context.verify(plain_password, hashed_password)
    
    def get_password_hash(self, password: str) -> str:
        """Gera hash da senha"""
        return pwd_context.hash(password)
    
    def create_access_token(
        self,
        data: dict,
        expires_delta: Optional[timedelta] = None
    ) -> str:
        """Cria token JWT de acesso"""
        to_encode = data.copy()
        if expires_delta:
            expire = datetime.utcnow() + expires_delta
        else:
            expire = datetime.utcnow() + timedelta(
                minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES
            )
        
        to_encode.update({"exp": expire, "type": "access"})
        encoded_jwt = jwt.encode(
            to_encode,
            self.secret_key,
            algorithm=self.algorithm
        )
        
        self.logger.info(
            "Token de acesso criado",
            user_id=data.get("sub"),
            expires_at=expire.isoformat()
        )
        
        return encoded_jwt
    
    def create_refresh_token(
        self,
        data: dict,
        expires_delta: Optional[timedelta] = None
    ) -> str:
        """Cria token JWT de refresh"""
        to_encode = data.copy()
        if expires_delta:
            expire = datetime.utcnow() + expires_delta
        else:
            expire = datetime.utcnow() + timedelta(
                days=settings.REFRESH_TOKEN_EXPIRE_DAYS
            )
        
        to_encode.update({"exp": expire, "type": "refresh"})
        encoded_jwt = jwt.encode(
            to_encode,
            self.secret_key,
            algorithm=self.algorithm
        )
        
        return encoded_jwt
    
    def verify_token(
        self,
        token: str,
        token_type: str = "access"
    ) -> dict:
        """Verifica e decodifica token JWT"""
        try:
            payload = jwt.decode(
                token,
                self.secret_key,
                algorithms=[self.algorithm]
            )
            
            if payload.get("type") != token_type:
                raise JWTError("Tipo de token inválido")
            
            return payload
            
        except JWTError as e:
            self.logger.error(
                "Falha na verificação do token",
                error=str(e),
                token_type=token_type
            )
            raise UnauthorizedException("Token inválido ou expirado")

import redis
from redis.exceptions import RedisError
from typing import Any, Optional, Union
import pickle
import json
from functools import wraps
import time

class CircuitBreaker:
    """Circuit breaker para serviços externos"""
    
    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 30):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.failure_count = 0
        self.last_failure_time = 0
        self.state = "CLOSED"  # CLOSED, OPEN, HALF_OPEN
        
    def can_execute(self) -> bool:
        """Verifica se pode executar operação"""
        if self.state == "OPEN":
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = "HALF_OPEN"
                return True
            return False
        return True
    
    def record_success(self):
        """Registra sucesso"""
        if self.state == "HALF_OPEN":
            self.state = "CLOSED"
        self.failure_count = 0
    
    def record_failure(self):
        """Registra falha"""
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        if self.failure_count >= self.failure_threshold:
            self.state = "OPEN"

class RedisCache:
    """Cliente Redis com circuit breaker e fallback"""
    
    def __init__(self):
        self.circuit_breaker = CircuitBreaker()
        self.logger = StructuredLogger()
        self._client = None
        self._fallback_cache = {}  # Cache local como fallback
        
    @property
    def client(self) -> redis.Redis:
        """Lazy initialization do cliente Redis"""
        if self._client is None:
            try:
                self._client = redis.Redis(
                    host=settings.REDIS_HOST,
                    port=settings.REDIS_PORT,
                    password=settings.REDIS_PASSWORD,
                    db=settings.REDIS_DB,
                    socket_timeout=5,  # Timeout explícito
                    socket_connect_timeout=5,
                    retry_on_timeout=True,
                    max_connections=10
                )
                # Testa conexão
                self._client.ping()
            except RedisError as e:
                self.logger.error("Falha na conexão com Redis", error=str(e))
                self._client = None
                raise
        return self._client
    
    def get(self, key: str) -> Optional[Any]:
        """Obtém valor do cache com fallback"""
        if not self.circuit_breaker.can_execute():
            # Circuito aberto - usa fallback local
            self.logger.warning("Circuit breaker aberto - usando fallback local", key=key)
            return self._fallback_cache.get(key)
        
        try:
            value = self.client.get(key)
            if value:
                self.circuit_breaker.record_success()
                return pickle.loads(value)
            return None
            
        except RedisError as e:
            self.circuit_breaker.record_failure()
            self.logger.error("Erro ao obter do Redis", key=key, error=str(e))
            
            # Fallback para cache local
            return self._fallback_cache.get(key)
    
    def set(
        self,
        key: str,
        value: Any,
        ttl: Optional[int] = None
    ) -> bool:
        """Define valor no cache com fallback"""
        if not self.circuit_breaker.can_execute():
            # Circuito aberto - usa fallback local
            self._fallback_cache[key] = value
            return True
        
        try:
            serialized = pickle.dumps(value)
            actual_ttl = ttl or settings.REDIS_CACHE_TTL
            
            result = self.client.setex(key, actual_ttl, serialized)
            self.circuit_breaker.record_success()
            
            # Também armazena no fallback local
            self._fallback_cache[key] = value
            
            return result
            
        except RedisError as e:
            self.circuit_breaker.record_failure()
            self.logger.error("Erro ao definir no Redis", key=key, error=str(e))
            
            # Fallback para cache local
            self._fallback_cache[key] = value
            return True
    
    def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        try:
            result = self.client.delete(key)
            # Remove do fallback também
            self._fallback_cache.pop(key, None)
            return result > 0
            
        except RedisError as e:
            self.logger.error("Erro ao deletar do Redis", key=key, error=str(e))
            return False
    
    def clear_pattern(self, pattern: str) -> int:
        """Limpa cache por pattern"""
        try:
            keys = self.client.keys(pattern)
            if keys:
                return self.client.delete(*keys)
            return 0
        except RedisError as e:
            self.logger.error("Erro ao limpar pattern", pattern=pattern, error=str(e))
            return 0

const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_URL || 'https://api.diyapp.com/v1',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }

// Add request interceptor for auth
apiClient.interceptors.request.use(
  config => {
    const token = localStorage.getItem('auth_token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  error => Promise.reject(error)
)

// Add response interceptor for error handling
apiClient.interceptors.response.use(
  response => response,
  error => {
    if (error.response?.status === 401) {
      // Handle unauthorized
      localStorage.removeItem('auth_token')
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

// Configure router
const router = createRouter({
  history: createWebHistory(),
  routes,
  scrollBehavior(to, from, savedPosition) {
    if (savedPosition) {
      return savedPosition
    } else {
      return { top: 0 }
    }
  }
})

// Use plugins

app.use(router)
app.use(VueAxios, apiClient)

// Mount app
app.mount('#app')

import { defineStore } from 'pinia'

export const useAuthStore = defineStore('auth', {
  state: () => ({
    user: null,
    token: localStorage.getItem('auth_token') || null,
    isAuthenticated: !!localStorage.getItem('auth_token')
  }),
  actions: {
    setAuth(user, token) {
      this.user = user
      this.token = token
      this.isAuthenticated = true
      localStorage.setItem('auth_token', token)
    },
    clearAuth() {
      this.user = null
      this.token = null
      this.isAuthenticated = false
      localStorage.removeItem('auth_token')
    },
    async login(credentials) {
      try {
        const response = await this.axios.post('/auth/login', credentials)
        this.setAuth(response.data.user, response.data.token)
        return { success: true }
      } catch (error) {
        return { 
          success: false, 
          error: error.response?.data?.message || 'Login failed' 
        }
      }
    },
    async logout() {
      try {
        await this.axios.post('/auth/logout')
      } finally {
        this.clearAuth()
      }
    }
  }
})

export const useAppStore = defineStore('app', {

-- Função para atualizar métricas diárias
CREATE OR REPLACE FUNCTION refresh_daily_metrics(p_date DATE DEFAULT CURRENT_DATE)
RETURNS VOID AS $$
BEGIN
    -- Limpar métricas do dia (idempotente)
    DELETE FROM mart_user_metrics WHERE metric_date = p_date;
    DELETE FROM mart_project_metrics WHERE metric_date = p_date;
    DELETE FROM mart_event_metrics WHERE metric_date = p_date;
    DELETE FROM mart_performance_metrics WHERE metric_date = p_date;
    
    -- 1. Métricas de usuário
    INSERT INTO mart_user_metrics (
        metric_date,
        total_users,
        active_users,
        new_users,
        churned_users,
        free_users,
        pro_users,
        enterprise_users,
        avg_logins_per_user
    )
    SELECT 
        p_date,
        COUNT(DISTINCT user_id) as total_users,
        COUNT(DISTINCT CASE WHEN last_login_at >= p_date - INTERVAL '30 days' THEN user_id END) as active_users,
        COUNT(DISTINCT CASE WHEN created_at::DATE = p_date THEN user_id END) as new_users,
        COUNT(DISTINCT CASE WHEN last_login_at < p_date - INTERVAL '90 days' AND is_active = TRUE THEN user_id END) as churned_users,
        COUNT(DISTINCT CASE WHEN tier = 'free' THEN user_id END) as free_users,
        COUNT(DISTINCT CASE WHEN tier = 'pro' THEN user_id END) as pro_users,
        COUNT(DISTINCT CASE WHEN tier = 'enterprise' THEN user_id END) as enterprise_users,
        COALESCE(
            AVG(CASE WHEN login_count > 0 THEN login_count END),
            0
        ) as avg_logins_per_user
    FROM diyapp_prod.users u
    LEFT JOIN (
        SELECT 
            user_id,
            COUNT(*) as login_count
        FROM diyapp_prod.user_events
        WHERE event_type = 'login'
            AND event_date = p_date
        GROUP BY user_id
    ) logins ON u.user_id = logins.user_id;
    
    -- 2. Métricas de projeto
    INSERT INTO mart_project_metrics (
        metric_date,
        project_type,
        total_projects,
        active_projects,
        projects_created,
        avg_components_per_project,
        avg_activity_days
    )
    SELECT 
        p_date,
        p.project_type,
        COUNT(DISTINCT p.project_id) as total_projects,
        COUNT(DISTINCT CASE WHEN p.status = 'active' THEN p.project_id END) as active_projects,
        COUNT(DISTINCT CASE WHEN p.created_at::DATE = p_date THEN p.project_id END) as projects_created,
        COALESCE(AVG(comp_count), 0) as avg_components_per_project,
        COALESCE(AVG(activity_days), 0) as avg_activity_days
    FROM diyapp_prod.projects p
    LEFT JOIN (
        SELECT 
            project_id,
            COUNT(DISTINCT component_id) as comp_count
        FROM diyapp_prod.project_components
        GROUP BY project_id
    ) comps ON p.project_id = comps.project_id
    LEFT JOIN (
        SELECT 
            project_id,
            COUNT(DISTINCT event_date) as activity_days
        FROM diyapp_prod.user_events
        WHERE event_date >= p_date - INTERVAL '30 days'
        GROUP BY project_id
    ) activity ON p.project_id = activity.project_id
    GROUP BY p.project_type;
    
    -- 3. Métricas de evento
    INSERT INTO mart_event_metrics (
        metric_date,
        event_type,
        total_events,
        unique_users,
        unique_sessions,
        avg_events_per_user
    )
    SELECT 
        p_date,
        event_type,
        COUNT(*) as total_events,
        COUNT(DISTINCT user_id) as unique_users,
        COUNT(DISTINCT session_id) as unique_sessions,
        COUNT(*)::DECIMAL / NULLIF(COUNT(DISTINCT user_id), 0) as avg_events_per_user
    FROM diyapp_prod.user_events
    WHERE event_date = p_date
    GROUP BY event_type;
    
    -- 4. Métricas de performance
    INSERT INTO mart_performance_metrics (
        metric_date,
        metric_type,
        project_type,
        p50_value,
        p90_value,
        p95_value,
        max_value,
        sample_size
    )
    SELECT 
        p_date,
        pm.metric_type,
        p.project_type,
        PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY pm.value) as p50_value,
        PERCENTILE_CONT(0.9) WITHIN GROUP (ORDER BY pm.value) as p90_value,
        PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY pm.value) as p95_value,
        MAX(pm.value) as max_value,
        COUNT(*) as sample_size
    FROM diyapp_prod.performance_metrics pm
    JOIN diyapp_prod.projects p ON pm.project_id = p.project_id
    WHERE pm.metric_date = p_date
    GROUP BY pm.metric_type, p.project_type;
    
END;
$$ LANGUAGE plpgsql;

# DIYAPP Evolution - V11 Core - Segurança

## 1. THREAT MODELING STRIDE - DIYAPP V11

### Ativos Críticos:
1. **Dados de Usuário**: PII (nome, email, telefone), tokens de autenticação
2. **Sistema de IA**: LLM prompts, contextos de conversa, chaves de API
3. **Infraestrutura**: Servidores, bancos de dados, filas de mensagens
4. **Integrações**: WhatsApp API, serviços externos

### Análise STRIDE:

#### **Spoofing (Falsificação)**
- **Ameaça**: Ataque de força bruta em login, token JWT roubado
- **Controles**: MFA obrigatório, rate limiting (10 tentativas/hora), tokens JWT com expiração curta (15min), refresh tokens rotativos
- **Risco**: Alto - Mitigado com MFA

#### **Tampering (Manipulação)**
- **Ameaça**: SQL injection, manipulação de parâmetros de API
- **Controles**: Prepared statements, validação de schema JSON, assinatura digital em webhooks
- **Risco**: Médio - Mitigado com WAF

#### **Repudiation (Repúdio)**
- **Ameaça**: Usuário nega ação realizada
- **Controles**: Logs imutáveis com auditoria completa, assinatura digital em logs críticos
- **Risco**: Baixo - Mitigado com auditoria

#### **Information Disclosure (Divulgação)**
- **Ameaça**: Vazamento de dados sensíveis, erro de configuração S3
- **Controles**: Criptografia AES-256 em repouso, mascaramento de PII em logs, política de CORS restritiva
- **Risco**: Alto - Mitigado com criptografia

#### **Denial of Service (Negação)**
- **Ameaça**: Ataque DDoS, sobrecarga de API
- **Controles**: WAF com rate limiting, circuit breakers, auto-scaling
- **Risco**: Médio - Mitigado com WAF

#### **Elevation of Privilege (Elevação)**
- **Ameaça**: Acesso não autorizado a recursos admin
- **Controles**: RBAC granular, princípio do menor privilégio, verificação de permissão por recurso
- **Risco**: Alto - Mitigado com RBAC

```javascript
// src/security/waf.js
const rateLimit = require('express-rate-limit');
const helmet = require('helmet');
const { body, validationResult } = require('express-validator');
const mongoSanitize = require('express-mongo-sanitize');
const xss = require('xss-clean');
const hpp = require('hpp');

class WAF {
  constructor() {
    this.rules = [];
    this.blockedIPs = new Set();
    this.init();
  }

  init() {
    // Regras de rate limiting por tipo de endpoint
    this.rules = [
      {
        name: 'auth-limiter',
        windowMs: 15 * 60 * 1000, // 15 minutos
        max: 10, // 10 tentativas por IP
        message: 'Muitas tentativas de login, tente novamente mais tarde',
        skipSuccessfulRequests: true
      },
      {
        name: 'api-limiter',
        windowMs: 5 * 60 * 1000, // 5 minutos
        max: 100, // 100 requisições por IP
        message: 'Muitas requisições, tente novamente mais tarde'
      },
      {
        name: 'whatsapp-limiter',
        windowMs: 60 * 1000, // 1 minuto
        max: 30, // 30 mensagens por minuto
        message: 'Limite de mensagens excedido'
      }
    ];
  }

  getMiddleware() {
    const middlewares = [
      // Headers de segurança
      helmet({
        contentSecurityPolicy: {
          directives: {
            defaultSrc: ["'self'"],
            styleSrc: ["'self'", "'unsafe-inline'"],
            scriptSrc: ["'self'"],
            imgSrc: ["'self'", "data:", "https:"],
            connectSrc: ["'self'"],
            fontSrc: ["'self'"],
            objectSrc: ["'none'"],
            mediaSrc: ["'self'"],
            frameSrc: ["'none'"]
          }
        },
        hsts: {
          maxAge: 31536000,
          includeSubDomains: true,
          preload: true
        },
        referrerPolicy: { policy: 'strict-origin-when-cross-origin' }
      }),

      // Sanitização
      mongoSanitize(),
      xss(),
      hpp(),

      // CORS seguro
      (req, res, next) => {
        res.header('Access-Control-Allow-Origin', process.env.ALLOWED_ORIGINS || 'https://app.diyapp.com');
        res.header('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS');
        res.header('Access-Control-Allow-Headers', 'Content-Type, Authorization, X-Requested-With');
        res.header('Access-Control-Allow-Credentials', 'true');
        
        if (req.method === 'OPTIONS') {
          return res.status(200).end();
        }
        next();
      },

      // Rate limiting dinâmico
      (req, res, next) => {
        const clientIP = req.ip || req.connection.remoteAddress;
        
        // Verifica IP bloqueado
        if (this.blockedIPs.has(clientIP)) {
          return res.status(403).json({
            error: 'IP bloqueado por atividade suspeita',
            code: 'IP_BLOCKED'
          });
        }

        // Verifica padrões suspeitos
        if (this.isSuspiciousRequest(req)) {
          this.blockedIPs.add(clientIP);
          console.warn(`IP ${clientIP} bloqueado por atividade suspeita`);
          return res.status(403).json({
            error: 'Atividade suspeita detectada',
            code: 'SUSPICIOUS_ACTIVITY'
          });
        }

        next();
      }
    ];

    // Adiciona rate limiters específicos
    this.rules.forEach(rule => {
      middlewares.push(rateLimit(rule));
    });

    return middlewares;
  }

  isSuspiciousRequest(req) {
    // Detecta SQL injection patterns
    const sqlInjectionPatterns = [
      /(\%27)|(\')|(\-\-)|(\%23)|(#)/i,
      /((\%3D)|(=))[^\n]*((\%27)|(\')|(\-\-)|(\%3B)|(;))/i,
      /\w*((\%27)|(\'))((\%6F)|o|(\%4F))((\%72)|r|(\%52))/i,
      /((\%27)|(\'))union/i
    ];

    // Detecta XSS patterns
    const xssPatterns = [
      /<script\b[^>]*>([\s\S]*?)<\/script>/i,
      /javascript:/i,
      /on\w+\s*=/i,
      /eval\s*\(/i
    ];

    const checkPatterns = (input) => {
      if (!input) return false;
      
      // Verifica SQL injection
      for (const pattern of sqlInjectionPatterns) {
        if (pattern.test(input)) return true;
      }
      
      // Verifica XSS
      for (const pattern of xssPatterns) {
        if (pattern.test(input)) return true;
      }
      
      return false;
    };

    // Verifica query params
    for (const key in req.query) {
      if (checkPatterns(req.query[key])) return true;
    }

    // Verifica body params
    if (req.body) {
      const bodyStr = JSON.stringify(req.body);
      if (checkPatterns(bodyStr)) return true;
    }

    // Verifica headers suspeitos
    const suspiciousHeaders = ['x-forwarded-for', 'user-agent'];
    for (const header of suspiciousHeaders) {
      if (req.headers[header] && checkPatterns(req.headers[header])) {
        return true;
      }
    }

    return false;
  }

  async logSecurityEvent(event) {
    const SecurityLog = require('../models/SecurityLog');
    
    await SecurityLog.create({
      type: event.type,
      severity: event.severity,
      ip: event.ip,
      userAgent: event.userAgent,
      path: event.path,
      method: event.method,
      details: event.details,
      timestamp: new Date()
    });

    // Alerta para Slack se for crítico
    if (event.severity === 'CRITICAL') {
      await this.sendSecurityAlert(event);
    }
  }

  async sendSecurityAlert(event) {
    // Implementação de alerta para Slack/Teams
    const webhookUrl = process.env.SECURITY_WEBHOOK_URL;
    
    if (webhookUrl) {
      const payload = {
        text: `🚨 ALERTA DE SEGURANÇA CRÍTICA`,
        attachments: [{
          color: 'danger',
          fields: [
            { title: 'Tipo', value: event.type, short: true },
            { title: 'Severidade', value: event.severity, short: true },
            { title: 'IP', value: event.ip || 'N/A', short: true },
            { title: 'Path', value: event.path, short: true },
            { title: 'Detalhes', value: event.details || 'N/A' }
          ],
          timestamp: new Date()
        }]
      };

      try {
        await fetch(webhookUrl, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload)
        });
      } catch (error) {
        console.error('Erro ao enviar alerta de segurança:', error);
      }
    }
  }
}
```

```javascript
// src/security/secrets.js
const crypto = require('crypto');
const AWS = require('aws-sdk');
const NodeCache = require('node-cache');

class SecretManager {
  constructor() {
    this.cache = new NodeCache({ stdTTL: 300, checkperiod: 60 }); // 5 minutos cache
    this.encryptionKey = process.env.ENCRYPTION_KEY;
    
    // Configura AWS Secrets Manager se disponível
    if (process.env.AWS_REGION) {
      this.secretsManager = new AWS.SecretsManager({
        region: process.env.AWS_REGION
      });
    }
  }

  async getSecret(secretName, useCache = true) {
    // Tenta cache primeiro
    if (useCache) {
      const cached = this.cache.get(secretName);
      if (cached) return cached;
    }

    let secret;
    
    // Prioridade: AWS Secrets Manager > Variáveis de ambiente > Arquivo .env
    if (this.secretsManager) {
      try {
        const data = await this.secretsManager.getSecretValue({ SecretId: secretName }).promise();
        secret = data.SecretString;
      } catch (error) {
        console.warn(`AWS Secrets Manager falhou para ${secretName}:`, error.message);
      }
    }

    // Fallback para variáveis de ambiente
    if (!secret) {
      secret = process.env[secretName];
    }

    // Decrypt se necessário
    if (secret && secret.startsWith('enc:')) {
      secret = this.decrypt(secret.substring(4));
    }

    // Cache o resultado
    if (secret && useCache) {
      this.cache.set(secretName, secret);
    }

    return secret;
  }

  encrypt(text) {
    if (!this.encryptionKey) {
      throw new Error('ENCRYPTION_KEY não configurada');
    }

    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv('aes-256-gcm', 
      Buffer.from(this.encryptionKey, 'hex'), 
      iv
    );

    let encrypted = cipher.update(text, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    return `enc:${iv.toString('hex')}:${encrypted}:${authTag.toString('hex')}`;
  }

  decrypt(encryptedText) {
    if (!this.encryptionKey) {
      throw new Error('ENCRYPTION_KEY não configurada');
    }

    const parts = encryptedText.split(':');
    if (parts.length !== 3) {
      throw new Error('Texto criptografado mal formatado');
    }

    const [ivHex, encrypted, authTagHex] = parts;
    
    const decipher = crypto.createDecipheriv('aes-256-gcm',
      Buffer.from(this.encryptionKey, 'hex'),
      Buffer.from(ivHex, 'hex')
    );

    decipher.setAuthTag(Buffer.from(authTagHex, 'hex'));

    let decrypted = decipher.update(encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');

    return decrypted;
  }

  async rotateSecret(secretName, newSecret) {
    // Rotação segura de secrets
    const oldSecret = await this.getSecret(secretName, false);
    
    // Atualiza no AWS Secrets Manager
    if (this.secretsManager) {
      await this.secretsManager.updateSecret({
        SecretId: secretName,
        SecretString: newSecret
      }).promise();
    }
    
    // Atualiza cache
    this.cache.set(secretName, newSecret);
    
    // Log da rotação
    await this.logRotation(secretName);
    
    return { success: true, message: 'Secret rotacionado com sucesso' };
  }

  async logRotation(secretName) {
    const SecurityLog = require('../models/SecurityLog');
    
    await SecurityLog.create({
      type: 'SECRET_ROTATION',
      severity: 'INFO',
      details: `Secret ${secretName} rotacionado`,
      timestamp: new Date()
    });
  }

  // Geração segura de tokens
  generateSecureToken(length = 32) {
    return crypto.randomBytes(length).toString('hex');
  }

  // Hash seguro de senhas
  async hashPassword(password) {
    const salt = crypto.randomBytes(16).toString('hex');
    const hash = crypto.pbkdf2Sync(password, salt, 10000, 64, 'sha512').toString('hex');
    return `${salt}:${hash}`;
  }

  async verifyPassword(password, storedHash) {
    const [salt, originalHash] = storedHash.split(':');
    const hash = crypto.pbkdf2Sync(password, salt, 10000, 64, 'sha512').toString('hex');
    return hash === originalHash;
  }
}

module.exports = new SecretManager();
```

## 1. Threat Modeling (STRIDE) para DIYAPP V11

### Assets Críticos:
1. **Dados de usuário**: Credenciais, tokens, dados pessoais
2. **Sistema de pagamento**: Chaves de API, transações
3. **LLM Integration**: Prompts do sistema, contextos de conversa
4. **Infraestrutura**: Secrets, configurações, logs de auditoria

### Análise STRIDE:

#### Spoofing (S):
- **Ameaça**: Ataque de força bruta em login
- **Controle**: MFA obrigatório, rate limiting, bloqueio após tentativas
- **Ameaça**: Token JWT roubado
- **Controle**: Tokens de curta duração (15min), refresh tokens rotativos

#### Tampering (T):
- **Ameaça**: Manipulação de dados em trânsito
- **Controle**: TLS 1.3 obrigatório, assinatura de payloads
- **Ameaça**: SQL Injection
- **Controle**: Prepared statements, ORM com escaping

#### Repudiation (R):
- **Ameaça**: Negação de ações administrativas
- **Controle**: Logs imutáveis com assinatura digital, auditoria completa

#### Information Disclosure (I):
- **Ameaça**: Vazamento de dados sensíveis em logs
- **Controle**: Mascaramento de PII, criptografia em repouso
- **Ameaça**: Prompt injection em LLM
- **Controle**: Sanitização de input, separação de contexto

#### Denial of Service (D):
- **Ameaça**: Ataque DDoS na API
- **Controle**: WAF com rate limiting, circuit breakers
- **Ameaça**: Exaustão de recursos por LLM
- **Controle**: Limite de tokens por usuário, quotas

#### Elevation of Privilege (E):
- **Ameaça**: IDOR em endpoints de API
- **Controle**: Autorização baseada em recursos, RBAC granular
- **Ameaça**: Manipulação de parâmetros de URL
- **Controle**: Validação de input estrita

```javascript
            max: 100, // Limit each IP to 100 requests per windowMs
            message: 'Too many requests from this IP, please try again later.',
            standardHeaders: true,
            legacyHeaders: false
        });

        const authLimiter = rateLimit({
            windowMs: 60 * 60 * 1000, // 1 hour
            max: 5, // 5 failed attempts per hour
            message: 'Too many failed login attempts, please try again later.',
            skipSuccessfulRequests: true

            }
        }
```

```javascript
                });

                console.log('Connected to HashiCorp Vault');
                this.initialized = true;
            } else {
                console.log('Using local secret management (Vault not configured)');
                this.initialized = true;
            }
        } catch (error) {
            console.warn('Vault connection failed, using local secrets:', error.message);
            this.initialized = true;
        }
    }

    async getSecret(key, defaultValue = null) {
        if (!this.initialized) await this.init();

        // Check environment variables first
        if (process.env[key]) {
            return process.env[key];
        }

        // Try Vault
        if (this.vault) {
            try {
                const secret = await this.vault.read(`secret/data/${key}`);
                return secret.data.data.value;
            } catch (error) {
                console.warn(`Secret ${key} not found in Vault:`, error.message);
            }
        }

        // Check local cache
        if (this.localSecrets.has(key)) {
            return this.localSecrets.get(key);
        }

        // Return default or throw
        if (defaultValue !== null) {
            return defaultValue;
        }

        throw new Error(`Secret ${key} not found`);
    }

    async setSecret(key, value, description = '') {
        if (!this.initialized) await this.init();

        // Encrypt locally before storing
        const encrypted = this.encryptSecret(value);

        if (this.vault) {
            await this.vault.write(`secret/data/${key}`, {
                data: { 
                    value: encrypted,
                    description,
                    created: new Date().toISOString()
                }
            });
        } else {
            this.localSecrets.set(key, encrypted);
            // Also store in encrypted file for persistence
            await this.saveToEncryptedFile(key, encrypted);
        }

        console.log(`Secret ${key} stored securely`);
    }

    encryptSecret(plaintext) {
        const algorithm = 'aes-256-gcm';
        const key = crypto.scryptSync(process.env.ENCRYPTION_KEY || 'default-key-change-in-production', 'salt', 32);
        const iv = crypto.randomBytes(16);

        let encrypted = cipher.update(plaintext, 'utf8', 'hex');
        encrypted += cipher.final('hex');
        
        const authTag = cipher.getAuthTag();
        
        return JSON.stringify({
            iv: iv.toString('hex'),
            encrypted,
            authTag: authTag.toString('hex'),
            algorithm
        });
    }

    decryptSecret(ciphertext) {
        const { iv, encrypted, authTag, algorithm } = JSON.parse(ciphertext);
        const key = crypto.scryptSync(process.env.ENCRYPTION_KEY || 'default-key-change-in-production', 'salt', 32);
        
        const decipher = crypto.createDecipheriv(algorithm, key, Buffer.from(iv, 'hex'));
        decipher.setAuthTag(Buffer.from(authTag, 'hex'));
        
        let decrypted = decipher.update(encrypted, 'hex', 'utf8');
        decrypted += decipher.final('utf8');
        
        return decrypted;
    }
```

```javascript
class Consent {
  constructor(db) {
    this.db = db;
  }

  async createTable() {
    await this.db.exec(`
      CREATE TABLE IF NOT EXISTS consents (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id VARCHAR(255) NOT NULL,
        consent_type VARCHAR(100) NOT NULL,
        legal_basis VARCHAR(50) NOT NULL,
        purpose TEXT NOT NULL,
        data_categories JSON NOT NULL,
        granted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        expires_at TIMESTAMP,
        revoked_at TIMESTAMP,
        version VARCHAR(20) NOT NULL,
        ip_address VARCHAR(45),
        user_agent TEXT,
        metadata JSON,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
      );

      CREATE INDEX IF NOT EXISTS idx_consents_user_id ON consents(user_id);
      CREATE INDEX IF NOT EXISTS idx_consents_consent_type ON consents(consent_type);
      CREATE INDEX IF NOT EXISTS idx_consents_granted_at ON consents(granted_at);
      CREATE INDEX IF NOT EXISTS idx_consents_revoked_at ON consents(revoked_at);
    `);
  }

  async recordConsent(consentData) {
    const {
      userId,
      consentType,
      legalBasis,
      purpose,
      dataCategories,
      expiresAt,
      ipAddress,
      userAgent,
      metadata = {}
    } = consentData;

    const result = await this.db.run(
      `INSERT INTO consents 
       (user_id, consent_type, legal_basis, purpose, data_categories, expires_at, ip_address, user_agent, metadata, version)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [
        userId,
        consentType,
        legalBasis,
        purpose,
        JSON.stringify(dataCategories),
        expiresAt,
        ipAddress,
        userAgent,
        JSON.stringify(metadata),
        '1.0.0'
      ]
    );

    return result.lastID;
  }

  async revokeConsent(userId, consentType) {
    await this.db.run(
      `UPDATE consents 
       SET revoked_at = CURRENT_TIMESTAMP, updated_at = CURRENT_TIMESTAMP
       WHERE user_id = ? AND consent_type = ? AND revoked_at IS NULL`,
      [userId, consentType]
    );
  }

  async getUserConsents(userId) {
    return await this.db.all(
      `SELECT * FROM consents 
       WHERE user_id = ? 
       ORDER BY granted_at DESC`,
      [userId]
    );
  }

  async getActiveConsent(userId, consentType) {
    return await this.db.get(
      `SELECT * FROM consents 
       WHERE user_id = ? AND consent_type = ? 
       AND revoked_at IS NULL 
       AND (expires_at IS NULL OR expires_at > CURRENT_TIMESTAMP)`,
      [userId, consentType]
    );
  }
}

module.exports = Consent;
```

```javascript
    const deadline = new Date();
    deadline.setDate(deadline.getDate() + 15); // LGPD/GDPR: 15 dias
    return deadline;
};

DataSubjectRequest.prototype.isOverdue = function() {
    return new Date() > new Date(this.deadlineDate) && this.status !== 'completed';
};

DataSubjectRequest.prototype.canExtend = function() {
    return this.status === 'processing' && !this.isOverdue();
};

DataSubjectRequest.prototype.extendDeadline = function() {
    if (this.canExtend()) {
        const newDeadline = new Date(this.deadlineDate);
        newDeadline.setDate(newDeadline.getDate() + 15); // +15 dias prorrogáveis
        this.deadlineDate = newDeadline;
```
