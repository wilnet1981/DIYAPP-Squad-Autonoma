syntax = "proto3";

package diyapp.v12;

option go_package = "github.com/diyapp/v12/template-service/gen";

import "google/protobuf/timestamp.proto";

// Serviço de exemplo com padrões V12
service TemplateService {
  // Operação síncrona padrão
  rpc CreateItem(CreateItemRequest) returns (ItemResponse);
  
  // Operação com streaming para atualizações em tempo real
  rpc StreamItems(StreamItemsRequest) returns (stream ItemResponse);
  
  // Health check para Kubernetes
  rpc HealthCheck(HealthRequest) returns (HealthResponse);
}

message CreateItemRequest {
  string name = 1;
  string description = 2;
  map<string, string> metadata = 3;
}

message ItemResponse {
  string id = 1;
  string name = 2;
  google.protobuf.Timestamp created_at = 3;
  Status status = 4;
  
  enum Status {
    PENDING = 0;
    ACTIVE = 1;
    ARCHIVED = 2;
  }
}

message StreamItemsRequest {
  string filter = 1;
  int32 batch_size = 2;
}

message HealthRequest {}

message HealthResponse {
  enum ServingStatus {
    UNKNOWN = 0;
    SERVING = 1;
    NOT_SERVING = 2;
  }
  ServingStatus status = 1;
  string version = 2;
  int64 timestamp = 3;
}

#!/bin/bash
set -e

# Configurações
SERVICE_NAME=$1
NAMESPACE="diyapp-v12"
ERROR_THRESHOLD=0.05  # 5%
CHECK_INTERVAL=30     # segundos
MAX_CHECKS=10         # 5 minutos total

echo "🔍 Iniciando monitoramento pós-deploy para $SERVICE_NAME"

for i in $(seq 1 $MAX_CHECKS); do
  echo "Verificação $i/$MAX_CHECKS..."
  
  # Coleta métricas do Prometheus via Istio
  ERROR_RATE=$(curl -s "http://prometheus.istio-system:9090/api/v1/query" \
    --data-urlencode "query=rate(istio_requests_total{destination_service=\"$SERVICE_NAME.$NAMESPACE.svc.cluster.local\",response_code!=\"200\"}[1m]) / rate(istio_requests_total{destination_service=\"$SERVICE_NAME.$NAMESPACE.svc.cluster.local\"}[1m])" \
    | jq -r '.data.result[0].value[1] // "0"')
  
  LATENCY_P95=$(curl -s "http://prometheus.istio-system:9090/api/v1/query" \
    --data-urlencode "query=histogram_quantile(0.95, rate(istio_request_duration_milliseconds_bucket{destination_service=\"$SERVICE_NAME.$NAMESPACE.svc.cluster.local\"}[1m]))" \
    | jq -r '.data.result[0].value[1] // "0"')
  
  echo "📊 Métricas - Error Rate: ${ERROR_RATE}, Latência P95: ${LATENCY_P95}ms"
  
  # Condições de rollback
  if (( $(echo "$ERROR_RATE > $ERROR_THRESHOLD" | bc -l) )); then
    echo "🚨 ALERTA: Error rate ($ERROR_RATE) acima do threshold ($ERROR_THRESHOLD)"
    echo "🔄 Iniciando rollback automático..."
    
    # Rollback para versão anterior
    kubectl rollout undo deployment/$SERVICE_NAME -n $NAMESPACE
    
    # Notificação
    curl -X POST -H 'Content-type: application/json' \
      --data "{\"text\":\"🚨 Rollback automático para $SERVICE_NAME. Error rate: $ERROR_RATE\"}" \
      $SLACK_WEBHOOK
    
    exit 1
  fi
  
  # Verifica se latência está aceitável (< 500ms)
  if (( $(echo "$LATENCY_P95 > 500" | bc -l) )); then
    echo "⚠️  ALERTA: Latência P95 ($LATENCY_P95 ms) acima do limite"
  fi
  
  sleep $CHECK_INTERVAL
done

echo "✅ Deploy estável. Monitoramento concluído."

```typescript
import { logger } from './logger';

export interface CircuitBreakerOptions {
  failureThreshold: number;
  recoveryTimeout: number;
  fallback?: () => Promise<any>;
}

export class CircuitBreaker {
  private state: 'CLOSED' | 'OPEN' | 'HALF_OPEN' = 'CLOSED';
  private failureCount = 0;
  private lastFailureTime: number | null = null;
  private readonly failureThreshold: number;
  private readonly recoveryTimeout: number;
  private readonly fallback?: () => Promise<any>;

  constructor(options: CircuitBreakerOptions) {
    this.failureThreshold = options.failureThreshold;
    this.recoveryTimeout = options.recoveryTimeout;
    this.fallback = options.fallback;
  }

  async execute<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === 'OPEN') {
      const now = Date.now();
      if (this.lastFailureTime && (now - this.lastFailureTime) > this.recoveryTimeout) {
        this.state = 'HALF_OPEN';
        logger.info('Circuit breaker transitioning to HALF_OPEN');
      } else {
        logger.warn('Circuit breaker is OPEN, using fallback');
        return this.executeFallback();
      }
    }

    try {
      const result = await fn();
      
      if (this.state === 'HALF_OPEN') {
        this.reset();
      }
      
      return result;
    } catch (error) {
      this.recordFailure();
      
      if (this.state === 'OPEN') {
        return this.executeFallback();
      }
      
      throw error;
    }
  }

  private recordFailure(): void {
    this.failureCount++;
    this.lastFailureTime = Date.now();

    if (this.failureCount >= this.failureThreshold) {
      this.state = 'OPEN';
      logger.error(`Circuit breaker opened after ${this.failureCount} failures`);
    }
  }

  private reset(): void {
    this.state = 'CLOSED';
    this.failureCount = 0;
    this.lastFailureTime = null;
    logger.info('Circuit breaker reset to CLOSED');
  }

  private async executeFallback<T>(): Promise<T> {
    if (this.fallback) {
      try {
        return await this.fallback();
      } catch (fallbackError) {
        logger.error('Fallback also failed', { error: fallbackError });
        throw new Error('Service unavailable and fallback failed');
      }
    }
    throw new Error('Service unavailable');
  }

  getState(): string {
    return this.state;
  }

  getFailureCount(): number {
    return this.failureCount;
  }
}

import { logger } from './logger';

export interface RetryOptions {
  maxAttempts: number;
  baseDelay: number;
  maxDelay: number;
  shouldRetry?: (error: any) => boolean;
}

export class Retry {
  static async execute<T>(
    fn: () => Promise<T>,
    options: RetryOptions
  ): Promise<T> {
    const {
      maxAttempts,
      baseDelay,
      maxDelay,
      shouldRetry = (error) => {
        // Retry on network errors, timeouts, and 5xx status codes
        return (
          error.code === 'ECONNREFUSED' ||
          error.code === 'ETIMEDOUT' ||
          error.code === 'ECONNRESET' ||
          (error.response && error.response.status >= 500)
        );
      },
    } = options;

    let lastError: any;
    
    for (let attempt = 1; attempt <= maxAttempts; attempt++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;
        
        if (!shouldRetry(error) || attempt === maxAttempts) {
          break;
        }
        
        const delay = Math.min(
          baseDelay * Math.pow(2, attempt - 1) + Math.random() * 100,
          maxDelay
        );
        
        logger.warn(`Retry attempt ${attempt}/${maxAttempts} after ${delay}ms`, {
          error: error.message,
          nextAttemptInMs: delay,
        });
        
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
    
    throw lastError;
  }
}

**lib/store/index.ts**
```typescript
import { create } from 'zustand'
import { persist, createJSONStorage } from 'zustand/middleware'
import { ThemeMode } from '@/lib/design-system/theme'

interface AppState {
  // Theme
  theme: ThemeMode
  setTheme: (theme: ThemeMode) => void
  toggleTheme: () => void
  
  // User
  user: {
    id: string | null
    name: string | null
    email: string | null
    avatar: string | null
  } | null
  setUser: (user: AppState['user']) => void
  clearUser: () => void
  
  // UI State
  sidebarOpen: boolean
  setSidebarOpen: (open: boolean) => void
  
  // Notifications
  notifications: Array<{
    id: string
    type: 'success' | 'error' | 'warning' | 'info'
    title: string
    message: string
    timestamp: number
  }>
  addNotification: (notification: Omit<AppState['notifications'][0], 'id' | 'timestamp'>) => void
  removeNotification: (id: string) => void
  clearNotifications: () => void
  
  // Loading States
  loading: Record<string, boolean>
  setLoading: (key: string, isLoading: boolean) => void
  
  // Offline State
  isOnline: boolean
  setIsOnline: (isOnline: boolean) => void
  
  // Cache
  cache: Record<string, {
    data: any
    timestamp: number
    ttl: number
  }>
  setCache: (key: string, data: any, ttl?: number) => void
  getCache: <T>(key: string) => T | null
  clearCache: (key?: string) => void
}

export const useAppStore = create<AppState>()(
  persist(
    (set, get) => ({
      // Initial state
      theme: 'light',
      user: null,
      sidebarOpen: false,
      notifications: [],
      loading: {},
      isOnline: typeof navigator !== 'undefined' ? navigator.onLine : true,
      cache: {},
      
      // Actions
      setTheme: (theme) => set({ theme }),
      
      toggleTheme: () => 
        set((state) => ({ 
          theme: state.theme === 'light' ? 'dark' : 'light' 
        })),
      
      setUser: (user) => set({ user }),
      
      clearUser: () => set({ user: null }),
      
      setSidebarOpen: (sidebarOpen) => set({ sidebarOpen }),
      
      addNotification: (notification) =>
        set((state) => ({
          notifications: [
            {
              ...notification,
              id: Math.random().toString(36).substr(2, 9),
              timestamp: Date.now(),
            },
            ...state.notifications,
          ].slice(0, 10), // Keep only last 10 notifications
        })),
      
      removeNotification: (id) =>
        set((state) => ({
          notifications: state.notifications.filter((n) => n.id !== id),
        })),
      
      clearNotifications: () => set({ notifications: [] }),
      
      setLoading: (key, isLoading) =>
        set((state) => ({
          loading: {
            ...state.loading,
            [key]: isLoading,
          },
        })),
      
      setIsOnline: (isOnline) => set({ isOnline }),
      
      setCache: (key, data, ttl = 5 * 60 * 1000) => // Default 5 minutes
        set((state) => ({
          cache: {
            ...state.cache,
            [key]: {
              data,
              timestamp: Date.now(),
              ttl,
            },
          },
        })),
      
      getCache: (key) => {
        const cacheItem = get().cache[key]
        if (!cacheItem) return null
        
        const isExpired = Date.now() - cacheItem.timestamp > cacheItem.ttl
        if (isExpired) {
          get().clearCache(key)
          return null
        }
        
        return cacheItem.data
      },
      
      clearCache: (key) => {
        if (key) {
          const { [key]: _, ...rest } = get().cache
          set({ cache: rest })
        } else {
          set({ cache: {} })
        }
      },
    }),
    {
      name: 'diyapp-store',
      storage: createJSONStorage(() => localStorage),
      partialize: (state) => ({
        theme: state.theme,
        user: state.user,
        sidebarOpen: state.sidebarOpen,
      }),
    }
  )
)

// Selectors for optimized re-renders
export const useTheme = () => useAppStore((state) => state.theme)
export const useUser = () => useAppStore((state) => state.user)
export const useSidebarOpen = () => useAppStore((state) => state.sidebarOpen)
export const useIsLoading = (key: string) => 
  useAppStore((state) => state.loading[key] || false)
```

#!/usr/bin/env python3
"""
Setup script for BigQuery data warehouse for DIYAPP V12
"""

from google.cloud import bigquery
from google.oauth2 import service_account
import json
import os
from datetime import datetime

class BigQuerySetup:
    def __init__(self, project_id, dataset_id, credentials_path=None):
        """Initialize BigQuery client"""
        if credentials_path:
            credentials = service_account.Credentials.from_service_account_file(
                credentials_path,
                scopes=["https://www.googleapis.com/auth/cloud-platform"]
            )
            self.client = bigquery.Client(
                project=project_id,
                credentials=credentials
            )
        else:
            self.client = bigquery.Client(project=project_id)
        
        self.project_id = project_id
        self.dataset_id = dataset_id
        self.dataset_ref = f"{project_id}.{dataset_id}"
    
    def create_dataset(self, location="US"):
        """Create main dataset"""
        dataset = bigquery.Dataset(self.dataset_ref)
        dataset.location = location
        dataset.description = "DIYAPP V12 - Data Warehouse"
        dataset.labels = {
            "environment": "production",
            "version": "v12",
            "managed_by": "data_engineering"
        }
        
        # Dataset default settings
        dataset.default_table_expiration_ms = 90 * 24 * 60 * 60 * 1000  # 90 days
        
        dataset = self.client.create_dataset(dataset, exists_ok=True)
        print(f"Dataset {self.dataset_id} created/verified")
        return dataset
    
    def create_raw_tables(self):
        """Create raw layer tables (bronze)"""
        tables = [
            {
                "table_id": "raw_user_events",
                "schema": [
                    bigquery.SchemaField("event_id", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("event_type", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("event_data", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("user_id", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("session_id", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("timestamp", "TIMESTAMP", mode="REQUIRED"),
                    bigquery.SchemaField("app_version", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("environment", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("source", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("ingestion_timestamp", "TIMESTAMP", mode="REQUIRED"),
                    bigquery.SchemaField("raw_source", "STRING", mode="REQUIRED")
                ],
                "partitioning": bigquery.TimePartitioning(
                    type_=bigquery.TimePartitioningType.DAY,
                    field="timestamp"
                ),
                "clustering": ["event_type", "source"],
                "description": "Raw user events - Never delete"
            },
            {
                "table_id": "raw_llm_calls",
                "schema": [
                    bigquery.SchemaField("call_id", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("model_name", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("provider", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("input_tokens", "INTEGER", mode="REQUIRED"),
                    bigquery.SchemaField("output_tokens", "INTEGER", mode="REQUIRED"),
                    bigquery.SchemaField("total_tokens", "INTEGER", mode="REQUIRED"),
                    bigquery.SchemaField("cost_usd", "FLOAT", mode="REQUIRED"),
                    bigquery.SchemaField("latency_ms", "INTEGER", mode="REQUIRED"),
                    bigquery.SchemaField("user_id", "STRING", mode="NULLABLE"),
                    bigquery.SchemaField("feature_name", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("timestamp", "TIMESTAMP", mode="REQUIRED"),
                    bigquery.SchemaField("status", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("error_message", "STRING", mode="NULLABLE"),
                    bigquery.SchemaField("ingestion_timestamp", "TIMESTAMP", mode="REQUIRED")
                ],
                "partitioning": bigquery.TimePartitioning(
                    type_=bigquery.TimePartitioningType.DAY,
                    field="timestamp"
                ),
                "clustering": ["model_name", "provider", "feature_name"],
                "description": "Raw LLM API calls with cost tracking"
            },
            {
                "table_id": "raw_product_metrics",
                "schema": [
                    bigquery.SchemaField("metric_id", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("metric_name", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("metric_value", "FLOAT", mode="REQUIRED"),
                    bigquery.SchemaField("metric_dimensions", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("timestamp", "TIMESTAMP", mode="REQUIRED"),
                    bigquery.SchemaField("app_version", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("environment", "STRING", mode="REQUIRED"),
                    bigquery.SchemaField("ingestion_timestamp", "TIMESTAMP", mode="REQUIRED")
                ],
                "partitioning": bigquery.TimePartitioning(
                    type_=bigquery.TimePartitioningType.DAY,
                    field="timestamp"
                ),
                "clustering": ["metric_name", "environment"],
                "description": "Raw product metrics from application"
            }
        ]
        
        for table_config in tables:
            table_ref = f"{self.dataset_ref}.{table_config['table_id']}"
            table = bigquery.Table(table_ref, schema=table_config["schema"])
            
            if "partitioning" in table_config:
                table.time_partitioning = table_config["partitioning"]
            
            if "clustering" in table_config:
                table.clustering_fields = table_config["clustering"]
            
            table.description = table_config["description"]
            
            # Set retention policy for raw tables (never delete)
            table.expires = None
            
            table = self.client.create_table(table, exists_ok=True)
            print(f"Table {table_config['table_id']} created/verified")
    
    def create_staging_views(self):

const crypto = require('crypto');

const algorithm = 'aes-256-cbc';
const key = Buffer.from(process.env.ENCRYPTION_KEY, 'hex');
const iv = crypto.randomBytes(16);

exports.encrypt = (text) => {
  const cipher = crypto.createCipheriv(algorithm, key, iv);
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  return `${iv.toString('hex')}:${encrypted}`;
};

exports.decrypt = (encryptedText) => {
  const [ivText, encrypted] = encryptedText.split(':');
  const iv = Buffer.from(ivText, 'hex');
  const decipher = crypto.createDecipheriv(algorithm, key, iv);
  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  return decrypted;
};

# Guia de Autenticação - DIYAPP Evolution V12

## Como autenticar na API
A API DIYAPP usa **JWT (JSON Web Tokens)** para autenticação. Siga os passos abaixo:

### 1. Obter credenciais
- **Client ID** e **Client Secret** são fornecidos no painel de desenvolvedor: [https://diyapp.com/developers](https://diyapp.com/developers).
- Guarde essas credenciais em um local seguro (ex: variáveis de ambiente).

### 2. Gerar token de acesso
Faça uma requisição `POST` para `/auth/token` com suas credenciais:

**Request**:
```bash
curl -X POST https://api.diyapp.com/v12/auth/token \
  -H "Content-Type: application/json" \
  -d '{
    "client_id": "seu_client_id",
    "client_secret": "seu_client_secret"
  }'
```

**Response (200 OK)**:
```json
{
  "access_token": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...",
  "expires_in": 3600
}
```

### 3. Usar o token em requisições
Inclua o token no header `Authorization` de todas as requisições protegidas:

```bash
curl -X GET https://api.diyapp.com/v12/projects \
  -H "Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."
```

### Erros comuns
| Código | Erro               | Solução                                  |
|--------|--------------------|------------------------------------------|
| 401    | `invalid_credentials` | Verifique seu `client_id` e `client_secret`. |
| 401    | `token_expired`      | Gere um novo token via `/auth/token`.    |
| 403    | `insufficient_scope` | Solicite permissões adicionais no painel. |

## Contexto
Atualmente, workflows complexos no DIYAPP são executados sequencialmente, o que aumenta o tempo de processamento. A squad identificou a necessidade de um modo de execução paralela para:
- Reduzir o tempo de pipelines de CI/CD.
- Melhorar a escalabilidade de workflows com múltiplas etapas independentes.
- Isolar recursos para evitar contaminação entre etapas.

---

## Opções consideradas
| Opção               | Prós                                      | Contras                                      |
|---------------------|-------------------------------------------|----------------------------------------------|
| **Modo Hive**       | Isolamento de recursos, escalabilidade    | Complexidade de implementação                |
| **Kubernetes Jobs** | Nativo no K8s, fácil de implementar       | Falta de isolamento de recursos              |
| **AWS Step Functions** | Gerenciado, integração com AWS         | Vendor lock-in, custo                        |

---

## Decisão
Adotar o **Modo Hive** para refatoração paralela, com as seguintes características:
1. **Isolamento de recursos**: Cada etapa do workflow roda em um container separado.
2. **Comunicação via filas**: Etapas se comunicam via filas (ex: SQS ou RabbitMQ).
3. **Fallback para modo sequencial**: Se o Modo Hive falhar, o workflow volta para execução sequencial.

**Implementação**:
- Backend: Adicionar suporte a `parallel_steps` no modelo de workflows.
- Frontend: Novo editor visual para workflows paralelos.
- SRE: Configurar namespaces no Kubernetes para isolamento.

---

## Consequências
- **Positivas**:
  - Redução de 50-70% no tempo de execução de workflows complexos.
  - Melhor utilização de recursos do cluster.
- **Negativas**:
  - Aumento na complexidade do código (gerenciamento de filas, timeouts).
  - Necessidade de treinamento para usuários avançados.

#### **3. `public/voiceflow.js` (Lógica Principal)**
```javascript
// Configurações
const WHISPER_MODEL_URL = 'https://cdn.jsdelivr.net/npm/@xenova/transformers@2.6.0/dist/whisper-tiny.js';
const LLM_ENDPOINT = 'http://localhost:11434/api/generate'; // Ollama local
const TUTORIAL_STEPS = [
  "Primeiro, pegue a furadeira e faça um furo na madeira.",
  "Em seguida, insira o parafuso no furo.",
  "Por fim, aperte o parafuso com a chave de fenda."
];

// Elementos DOM
const startBtn = document.getElementById('startBtn');
const stopBtn = document.getElementById('stopBtn');
const transcriptText = document.getElementById('transcriptText');
const feedbackText = document.getElementById('feedbackText');
const tutorialVideo = document.getElementById('tutorialVideo');

// Variáveis de estado
let recognition;
let whisperWorker;
let currentStep = 0;

// Inicializa o Web Speech API
function initSpeechRecognition() {
  if (!('webkitSpeechRecognition' in window)) {
    alert("Seu navegador não suporta reconhecimento de voz. Tente o Chrome.");
    return;
  }

  recognition = new webkitSpeechRecognition();
  recognition.continuous = true;
  recognition.interimResults = true;
  recognition.lang = 'pt-BR';

  recognition.onresult = async (event) => {
    const transcript = event.results[event.results.length - 1][0].transcript;
    transcriptText.textContent = transcript;

    // Processa o comando com LLM
    const feedback = await processCommandWithLLM(transcript);
    feedbackText.textContent = feedback;

    // Avança ou retrocede o tutorial se o comando for válido
    if (feedback.includes("Avançar para")) {
      currentStep = Math.min(currentStep + 1, TUTORIAL_STEPS.length - 1);
      tutorialVideo.currentTime = currentStep * 10; // Simula avanço no vídeo
    } else if (feedback.includes("Voltar para")) {
      currentStep = Math.max(currentStep - 1, 0);
      tutorialVideo.currentTime = currentStep * 10;
    }
  };

  recognition.onerror = (event) => {
    console.error("Erro no reconhecimento de voz:", event.error);
    feedbackText.textContent = "Erro: Não consegui entender. Tente novamente.";
  };
}

// Processa o comando do usuário com LLM local (Ollama)
async function processCommandWithLLM(command) {
  try {
    const prompt = `
      Você é um assistente de tutoriais DIY. O usuário está na etapa "${TUTORIAL_STEPS[currentStep]}".
      O comando dele foi: "${command}".
      Responda de forma concisa (máx. 20 palavras) e, se aplicável, sugira avançar ou voltar a etapa.
      Exemplo 1: "Como faço o furo?" → "Use a furadeira no modo 2. Avançar para próxima etapa."
      Exemplo 2: "Errei o furo" → "Volte para a etapa 1 e tente novamente."
    `;

    const response = await fetch(LLM_ENDPOINT, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'llama3.1:8b',
        prompt: prompt,
        stream: false
      })
    });

    const data = await response.json();
    return data.response.trim();
  } catch (error) {
    console.error("Erro ao chamar LLM:", error);
    return "Desculpe, não consegui processar sua solicitação.";
  }
}

// Inicializa o Whisper.js para transcrição local (opcional)
async function initWhisper() {
  try {
    const { pipeline } = await import(WHISPER_MODEL_URL);
    whisperWorker = await pipeline('automatic-speech-recognition', 'Xenova/whisper-tiny');
    console.log("Whisper carregado com sucesso!");
  } catch (error) {
    console.error("Erro ao carregar Whisper:", error);
  }
}

// Event Listeners
startBtn.addEventListener('click', () => {
  initSpeechRecognition();
  recognition.start();
  startBtn.disabled = true;
  stopBtn.disabled = false;
  feedbackText.textContent = "Ouvindo... Diga algo como 'Como faço o furo?'";
});

stopBtn.addEventListener('click', () => {
  recognition.stop();
  startBtn.disabled = false;
  stopBtn.disabled = true;
});

// Inicializa o Whisper ao carregar a página
initWhisper();
```

#### **3. `src/controllers/voiceController.js`**
```javascript
import { transcribeAudio } from '../services/whisperService.js';
import { generateResponse } from '../services/llmController.js';
import { synthesizeSpeech } from '../services/ttsService.js';

let recognition;
let isListening = false;
let startTime;
let stepTimes = [];

// Inicializa Web Speech API
function initSpeechRecognition() {
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  if (!SpeechRecognition) {
    alert("Seu navegador não suporta reconhecimento de voz. Teste no Chrome/Edge.");
    return;
  }

  recognition = new SpeechRecognition();
  recognition.lang = 'pt-BR';
  recognition.interimResults = false;

  recognition.onresult = async (event) => {
    const transcript = event.results[0][0].transcript;
    document.getElementById('transcriptText').textContent = transcript;

    // Envia para o LLM e processa resposta
    const response = await generateResponse(transcript);
    document.getElementById('responseText').textContent = response;

    // Síntese de voz
    await synthesizeSpeech(response);

    // Calcula tempo da etapa
    const endTime = new Date();
    const timeTaken = (endTime - startTime) / 1000; // segundos
    stepTimes.push(timeTaken);
    updateMetrics();
  };

  recognition.onerror = (event) => {
    console.error("Erro no reconhecimento de voz:", event.error);
    stopListening();
  };
}

// Atualiza métricas no dashboard
function updateMetrics() {
  const avgTime = stepTimes.reduce((a, b) => a + b, 0) / stepTimes.length;
  document.getElementById('avgTime').textContent = `${avgTime.toFixed(1)}s`;

  // Simula NPS (em produção, coletar via formulário)
  const nps = Math.floor(Math.random() * 10) + 1;
  document.getElementById('npsScore').textContent = nps;
}

// Controles de voz
document.getElementById('startBtn').addEventListener('click', () => {
  if (!isListening) {
    initSpeechRecognition();
    recognition.start();
    isListening = true;
    startTime = new Date();
    document.getElementById('startBtn').disabled = true;
    document.getElementById('stopBtn').disabled = false;
  }
});

document.getElementById('stopBtn').addEventListener('click', () => {
  if (isListening) {
    recognition.stop();
    isListening = false;
    document.getElementById('startBtn').disabled = false;
    document.getElementById('stopBtn').disabled = true;
  }
});
```

// Integração com Ollama (Llama-3-8B local) ou API externa
export async function generateResponse(transcript) {
  try {
    // Exemplo: Prompt para tutoriais DIY
    const prompt = `
      Você é um assistente especializado em tutoriais DIY.
      O usuário disse: "${transcript}".
      Responda de forma clara e concisa, como um instrutor.
      Se a pergunta não for relacionada ao tutorial, responda: "Foco no projeto! Vamos continuar?".
    `;

    // Chamada para Ollama (local) ou API externa
    const response = await fetch('http://localhost:11434/api/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'llama3',
        prompt: prompt,
        stream: false
      })
    });

    const data = await response.json();
    return data.response || "Desculpe, não consegui gerar uma resposta.";
  } catch (error) {
    console.error("Erro no LLM:", error);
    return "Estou com dificuldades técnicas. Tente novamente.";
  }
}

// Usa ElevenLabs (API) ou Web Speech Synthesis (navegador)
export async function synthesizeSpeech(text) {
  // Opção 1: Web Speech Synthesis (gratuito, mas qualidade limitada)
  if ('speechSynthesis' in window) {
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'pt-BR';
    window.speechSynthesis.speak(utterance);
    return;
  }

  // Opção 2: ElevenLabs (qualidade premium, pago)
  try {
    const response = await fetch('https://api.elevenlabs.io/v1/text-to-speech/YOUR_VOICE_ID', {
      method: 'POST',
      headers: {
        'xi-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        text: text,
        model_id: 'eleven_multilingual_v2'
      })
    });

    const audioBlob = await response.blob();
    const audioUrl = URL.createObjectURL(audioBlob);
    const audio = new Audio(audioUrl);
    audio.play();
  } catch (error) {
    console.error("Erro na síntese de voz:", error);
    // Fallback para Web Speech Synthesis
    const utterance = new SpeechSynthesisUtterance(text);
    window.speechSynthesis.speak(utterance);
  }
}

  count = length(var.public_subnets)

  cidr_block              = var.public_subnets[count.index]
  availability_zone       = var.availability_zones[count.index]
  map_public_ip_on_launch = true

  tags = {
    Name = "${var.cluster_name}-public-${count.index}"
    "kubernetes.io/role/elb" = "1"
  }

resource "aws_subnet" "private" {
  count = length(var.private_subnets)

  vpc_id            = aws_vpc.main.id
  cidr_block        = var.private_subnets[count.index]
  availability_zone = var.availability_zones[count.index]

  tags = {
    Name = "${var.cluster_name}-private-${count.index}"
    "kubernetes.io/role/internal-elb" = "1"

  count = length(var.public_subnets)
  vpc   = true

  tags = {
    Name = "${var.cluster_name}-nat-${count.index}"

resource "aws_nat_gateway" "nat" {

### kubernetes/monitoring/prometheus/alerts.yaml

```yaml
groups:
- name: slo-alerts
  rules:
  - alert: HighErrorRate
    expr: rate(http_requests_total{status=~"5.."}[5m]) / rate(http_requests_total[5m]) > 0.001
    for: 5m
    labels:
      severity: critical
      slo: "error_rate"
    annotations:
      summary: "High error rate on {{ $labels.instance }}"
      description: "Error rate is {{ $value }} (threshold: 0.1%)"

  - alert: HighLatencyAPI
    expr: histogram_quantile(0.95, sum(rate(http_request_duration_seconds_bucket{service!~"llm-.*"}[5m])) by (le, service)) > 0.3
    for: 5m
    labels:
      severity: critical
      slo: "latency_api"
    annotations:
      summary: "High latency on API service {{ $labels.service }}"
      description: "P95 latency is {{ $value }}s (threshold: 300ms)"

  - alert: HighLatencyLLM
    expr: histogram_quantile(0.95, sum(rate(http_request_duration_seconds_bucket{service=~"llm-.*"}[5m])) by (le, service)) > 8
    for: 5m
    labels:
      severity: warning
      slo: "latency_llm"
    annotations:
      summary: "High latency on LLM service {{ $labels.service }}"
      description: "P95 latency is {{ $value }}s (threshold: 8s)"

  - alert: LowAvailability
    expr: (1 - (sum(increase(http_requests_total{status=~"5.."}[5m])) by (service) / sum(increase(http_requests_total[5m])) by (service))) < 0.999
    for: 5m
    labels:
      severity: critical
      slo: "availability"
    annotations:
      summary: "Low availability for service {{ $labels.service }}"
      description: "Availability is {{ $value }} (threshold: 99.9%)"

  - alert: ErrorBudgetBurn
    expr: (1 - (sum(increase(http_requests_total{status=~"5.."}[1h])) by (service) / sum(increase(http_requests_total[1h])) by (service))) < 0.999 and (1 - (sum(increase(http_requests_total{status=~"5.."}[5m])) by (service) / sum(increase(http_requests_total[5m])) by (service))) < 0.999
    for: 5m
    labels:
      severity: critical
      slo: "error_budget"
    annotations:
      summary: "Error budget burn for service {{ $labels.service }}"
      description: "Error budget is being consumed too quickly"

- name: kubernetes-alerts
  rules:
  - alert: KubernetesNodeNotReady
    expr: kube_node_status_condition{condition="Ready",status="true"} == 0
    for: 10m
    labels:
      severity: critical
    annotations:
      summary: "Node {{ $labels.node }} is not ready"
      description: "Node {{ $labels.node }} has been unready for more than 10 minutes"

  - alert: KubernetesPodCrashLooping
    expr: increase(kube_pod_container_status_restarts_total[10m]) > 0
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "Pod {{ $labels.pod }} is crash looping"
      description: "Pod {{ $labels.pod }} in namespace {{ $labels.namespace }} is crash looping"

  - alert: KubernetesPodNotReady
    expr: sum by (namespace, pod) (kube_pod_status_phase{phase=~"Pending|Unknown"}) > 0
    for: 15m
    labels:
      severity: warning
    annotations:
      summary: "Pod {{ $labels.pod }} is not ready"
      description: "Pod {{ $labels.pod }} in namespace {{ $labels.namespace }} has been in a non-ready state for more than 15 minutes"

  - alert: KubernetesDeploymentReplicasMismatch
    expr: kube_deployment_spec_replicas != kube_deployment_status_replicas_available
    for: 10m
    labels:
      severity: warning
    annotations:
      summary: "Deployment {{ $labels.deployment }} replica mismatch"

ERROR_BUDGET=$(kubectl get --raw /apis/metrics.k8s.io/v1beta1/nodes | jq -r '.items[0].usage.error_budget_consumption')

```javascript
// src/auth/auth.js
const passport = require('passport');
const { Issuer, Strategy } = require('openid-client');

async function setupOAuth2() {
  const issuer = await Issuer.discover('https://accounts.google.com'); // Example for Google
  const client = new issuer.Client({
    client_id: process.env.OAUTH_CLIENT_ID,
    client_secret: process.env.OAUTH_CLIENT_SECRET,
    redirect_uris: ['http://localhost:3000/callback'],
    response_types: ['code'],
  });

  passport.use('oidc', new Strategy({
    client,
    params: {
      redirect_uri: 'http://localhost:3000/callback',
      scope: 'openid profile email',
    },
  }, (tokenset, userinfo, done) => {
    return done(null, userinfo);
  }));

  passport.serializeUser((user, done) => done(null, user));
  passport.deserializeUser((user, done) => done(null, user));
}

module.exports = { setupOAuth2 };

const crypto = require('crypto');

function encrypt(text) {
  const cipher = crypto.createCipher('aes256', process.env.ENCRYPTION_KEY);
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  return encrypted;
}

function decrypt(encrypted) {
  const decipher = crypto.createDecipher('aes256', process.env.ENCRYPTION_KEY);

  return decrypted;
}

class Predictor {
    constructor() {

    }

    loadLogs() {
        // Load and parse logs
        const logPath = path.join(__dirname, 'logs', 'example.log');
        return fs.readFileSync(logPath, 'utf-8').split('\n').map(line => this.parseLog(line));
    }

    loadMetrics() {
        // Load and parse metrics
        const metricsPath = path.join(__dirname, 'metrics', 'exampleMetrics.json');
        return JSON.parse(fs.readFileSync(metricsPath, 'utf-8'));
    }

    parseLog(logLine) {
        // Simple log parsing logic
        const parts = logLine.split(' ');
        return { timestamp: parts[0], level: parts[1], message: parts.slice(2).join(' ') };
    }

    predictFailures() {
        // Basic predictive analysis logic (e.g., identifying patterns)
        return this.logs.filter(log => log.level === 'ERROR');
    }
}

## **10. Regras de Integração com o Modo Hive**
1. **Refatoração Paralela**:
   - Testes E2E e de contrato devem rodar em **ambos os branches** (legacy e novo) durante a transição.
   - Usar **feature flags** para alternar entre implementações sem quebrar testes.

2. **Autonomia**:
   - Todos os testes devem ser **determinísticos** (evitar dependências externas em testes de carga).
   - **Golden datasets** de LLMs devem ser versionados junto ao código.

3. **Feedback Rápido**:
   - Pipeline de CI deve falhar **imediatamente** se:
     - Testes E2E críticos falharem.
     - Drift de LLM ≥ 10%.
     - Varredura de segurança encontrar vulnerabilidades de severidade alta/crítica.

4. **Documentação**:
   - Cada teste deve ter um **comentário** explicando seu propósito e critérios de aceite.
   - Relatórios de segurança devem ser **anexados automaticamente** ao card de deploy.

Se houver migrações de DB incompatíveis:
1. Restaure o backup mais recente:
   ```bash
   kubectl exec -n diyapp-prod diyapp-db-0 -- pg_restore -U postgres -d diyapp_prod /backups/diyapp_prod_20240520.sql
   ```

## **4. Diagnóstico de Erros**
### **4.1. Erro: `CrashLoopBackOff`**
**Causa:** Falha na inicialização do container (ex: configuração inválida).
**Solução:**
1. Verifique os logs:
   ```bash
   kubectl logs -n diyapp-prod diyapp-core-7c8f5d6b8c-abc12 --previous
   ```
2. Valide as variáveis de ambiente:
   ```bash
   kubectl describe pod -n diyapp-prod diyapp-core-7c8f5d6b8c-abc12 | grep -A 10 "Environment"
   ```

### **4.2. Erro: `HIVE_MODE_TIMEOUT`**
**Causa:** Refatoração paralela excedeu o tempo limite (padrão: 30 minutos).
**Solução:**
1. Aumente o timeout no manifesto de deploy:
   ```yaml
   env:
     - name: HIVE_TIMEOUT_MINUTES
       value: "60"
   ```
2. Reinicie o pod:
   ```bash
   kubectl rollout restart deployment/diyapp-core -n diyapp-prod
   ```

### [Sessão Paralela: Inovação]
Vamos prototipar uma **feature flagship** para o DIYAPP Evolution V12: **"DIY Voice Assistant"** — um assistente por voz integrado que permite aos usuários controlar tutoriais, buscar materiais e receber orientações hands-free usando LLMs e reconhecimento de voz em tempo real.

Esta feature se enquadra no **Horizonte 3 (H3)** — reinventar o futuro com IA — e explora uma oportunidade de diferenciação competitiva: **interação por voz em ambientes de trabalho manuais**, onde o usuário não pode tocar na tela (ex: marcenaria, mecânica, eletrônica).

---

### 🔍 **Sinal de Mercado & Justificativa**
- **Competidores**: Nenhum produto DIY mainstream oferece assistente por voz integrado com tutoriais e busca contextual.
- **Usuários**: Tickets de suporte e entrevistas mostram que usuários frequentemente interrompem o trabalho para consultar o app, sujando telas ou perdendo foco.
- **Tecnologia**: Modelos de reconhecimento de voz (Whisper) e síntese (TTS) atingiram qualidade suficiente para uso em ambientes ruidosos. LLMs permitem respostas contextuais baseadas no tutorial atual.
- **Regulatório**: Nenhuma barreira significativa; uso de APIs de voz é amplamente adotado.

> **Hipótese (H3)**:
> *"Acreditamos que usuários de tutoriais DIY têm o problema de precisar interromper o trabalho para interagir com o app e que um assistente por voz integrado vai resultar em 40% de redução no tempo médio de interrupção por sessão, medido via logs de uso."*

import { startListening, stopListening } from './voice/recognizer.js';
import { speak } from './voice/synthesizer.js';
import { handleCommand } from './voice/assistant.js';
import { getCurrentTutorialContext } from './context/tutorialContext.js';

// DOM Elements
const startBtn = document.getElementById('start-btn');
const stopBtn = document.getElementById('stop-btn');
const messagesDiv = document.getElementById('messages');
const currentStepEl = document.getElementById('current-step');
const npsBtns = document.querySelectorAll('.nps-btn');
const npsResultEl = document.getElementById('nps-result');

// State
let isListening = false;
let npsScore = null;

// Initialize
function init() {
  updateTutorialContext();
  setupEventListeners();
}

function setupEventListeners() {
  startBtn.addEventListener('click', () => {
    if (!isListening) {
      startListening(onSpeechResult);
      isListening = true;
      startBtn.disabled = true;
      stopBtn.disabled = false;
      addMessage('Ouvindo...', 'user');
    }
  });

  stopBtn.addEventListener('click', () => {
    stopListening();
    isListening = false;
    startBtn.disabled = false;
    stopBtn.disabled = true;
  });

  npsBtns.forEach(btn => {
    btn.addEventListener('click', () => {
      npsScore = parseInt(btn.dataset.score);
      npsResultEl.textContent = `Obrigado! Sua avaliação: ${npsScore}`;
      npsBtns.forEach(b => b.disabled = true);
    });
  });
}

function onSpeechResult(text) {
  addMessage(text, 'user');
  const response = handleCommand(text, getCurrentTutorialContext());
  addMessage(response, 'assistant');
  speak(response);
}

function addMessage(text, sender) {
  const messageEl = document.createElement('div');
  messageEl.classList.add('message', sender);
  messageEl.textContent = text;
  messagesDiv.appendChild(messageEl);
  messagesDiv.scrollTop = messagesDiv.scrollHeight;
}

function updateTutorialContext() {
  const context = getCurrentTutorialContext();
  currentStepEl.textContent = `${context.title} - ${context.step}`;
}

init();

let recognition;

export function startListening(callback) {
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

  if (!SpeechRecognition) {
    alert('Reconhecimento de voz não suportado neste navegador. Tente Chrome ou Edge.');
    return;
  }

  recognition = new SpeechRecognition();
  recognition.lang = 'pt-BR';
  recognition.interimResults = false;
  recognition.maxAlternatives = 1;

  recognition.onresult = (event) => {
    const transcript = event.results[0][0].transcript;
    callback(transcript);
  };

  recognition.onerror = (event) => {
    console.error('Erro no reconhecimento:', event.error);
    callback(`Erro: ${event.error}`);
  };

  recognition.onend = () => {
    if (isListening) {
      recognition.start(); // Mantém ouvindo
    }
  };

  recognition.start();
}

export function stopListening() {
  if (recognition) {
    recognition.stop();
  }
}

### 6. `src/voice/assistant.js`

```javascript
import { queryLLM } from '../llm/llmClient.js';

export async function handleCommand(command, context) {
  const lowerCommand = command.toLowerCase();

  // Comandos simples (sem LLM)
  if (lowerCommand.includes('próximo passo') || lowerCommand.includes('avançar')) {
    return `O próximo passo é: ${context.nextStep || 'finalizar a montagem.'}`;
  }

  if (lowerCommand.includes('passo anterior') || lowerCommand.includes('voltar')) {
    return `O passo anterior foi: ${context.previousStep || 'preparação dos materiais.'}`;
  }

  if (lowerCommand.includes('quais ferramentas') || lowerCommand.includes('materiais')) {
    return `Você vai precisar de: ${context.materials.join(', ')}.`;
  }

  if (lowerCommand.includes('repetir') || lowerCommand.includes('dizer de novo')) {
    return `Claro! ${context.step}`;
  }

  if (lowerCommand.includes('ajuda') || lowerCommand.includes('comandos')) {
    return 'Você pode dizer: próximo passo, materiais necessários, repetir instrução, ou fazer uma pergunta.';
  }

  // Comandos complexos (com LLM)
  return await queryLLM(command, context);
}

### 7. `src/llm/llmClient.js`

```javascript
// Usa Ollama local (recomendado para privacidade e custo zero)
// ou OpenAI API como fallback

const USE_LOCAL_LLM = true;

export async function queryLLM(prompt, context) {
  const fullPrompt = `
    Você é um assistente especializado em tutoriais DIY (faça você mesmo).
    O usuário está no seguinte contexto:
    - Tutorial: ${context.title}
    - Passo atual: ${context.step}
    - Materiais: ${context.materials.join(', ')}

    Responda de forma clara, concisa e útil à seguinte pergunta: "${prompt}".
    Se a pergunta não fizer sentido no contexto, peça para reformular.
    Limite a resposta a 2 frases.
  `;

  if (USE_LOCAL_LLM) {
    try {
      const response = await fetch('http://localhost:11434/api/generate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: 'llama2',
          prompt: fullPrompt,
          stream: false
        })
      });
      const data = await response.json();
      return data.response.trim();
    } catch (e) {
      console.error('Erro ao chamar Ollama:', e);
      return 'Desculpe, não consegui processar sua pergunta. Tente novamente.';
    }
  } else {
    // Fallback para OpenAI (requer chave)
    try {
      const response = await fetch('https://api.openai.com/v1/chat/completions', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${import.meta.env.VITE_OPENAI_API_KEY}`
        },
        body: JSON.stringify({
          model: 'gpt-3.5-turbo',
          messages: [{ role: 'user', content: fullPrompt }],
          max_tokens: 60
        })
      });
      const data = await response.json();
      return data.choices[0].message.content.trim();
    } catch (e) {
      console.error('Erro ao chamar OpenAI:', e);
      return 'Desculpe, não consegui processar sua pergunta. Verifique sua conexão.';
    }
  }
}
