# DIYAPP Evolution - V12 Core - Arquitetura de Microsserviços

## 1. Documento de Arquitetura (ADR-001)

```
# ADR-001: Arquitetura de Microsserviços V12
Data: 2024-01-15
Status: Aceita
Autores: Tech Lead V12 Core

CONTEXTO:
A V11 do DIYAPP atingiu limites de escalabilidade devido à arquitetura monolítica. 
Necessitamos de:
1. Deploy independente por funcionalidade
2. Isolamento de falhas
3. Escala horizontal por componente crítico
4. Time-to-market reduzido para novas features
5. Suporte a múltiplas linguagens por domínio específico

DECISÃO:
Adotar arquitetura de microsserviços com os seguintes princípios:
- 8 serviços independentes com bounded contexts claros
- API Gateway para roteamento e agregação
- Message Broker para comunicação assíncrona
- Service Discovery dinâmico
- Circuit Breaker para resiliência
- Centralized Logging e Distributed Tracing

OPÇÕES CONSIDERADAS:
- Opção A: Monólito modularizado (Strangler Pattern) - Prós: Menor risco, migração gradual. Contras: Limites de escalabilidade permanecem.
- Opção B: Microsserviços completos - Prós: Máxima flexibilidade e escalabilidade. Contras: Complexidade operacional aumentada.
- Opção C: Miniserviços (3-4 serviços maiores) - Prós: Balance entre complexidade e benefícios. Contras: Acoplamento ainda presente.

Opção escolhida: B - Justificativa: O domínio DIYAPP tem bounded contexts bem definidos. A complexidade operacional será mitigada com automação CI/CD e observabilidade robusta.

CONSEQUÊNCIAS:
Positivas:
- Deploy independente por squad
- Escala por componente crítico (ex: processamento de IA)
- Stack tecnológica otimizada por domínio
- Isolamento de falhas

Negativas:
- Latência de rede entre serviços
- Gerenciamento distribuído de transações
- Monitoramento mais complexo
- Testes end-to-end mais desafiadores

Riscos:
1. Network latency impactando P95
   Mitigação: Implementar caching estratégico e connection pooling
2. Data consistency em operações distribuídas
   Mitigação: Adotar SAGA pattern com compensação
3. Debug em produção complexo
   Mitigação: Implementar Distributed Tracing com correlation IDs

REVISÃO: 2024-04-15 (após 3 meses em produção)
```

```yaml
# .github/workflows/ci.yml
name: CI Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  lint-and-test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [api-gateway, auth-service, content-service, ai-processor, notification-service, payment-service, analytics-service, file-service]
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Setup Node.js (for Node services)
      if: matrix.service == 'api-gateway' || matrix.service == 'notification-service'
      uses: actions/setup-node@v3
      with:
        node-version: '18'
        cache: 'npm'
        cache-dependency-path: services/${{ matrix.service }}/package-lock.json
    
    - name: Setup Python (for Python services)
      if: matrix.service == 'content-service' || matrix.service == 'ai-processor' || matrix.service == 'analytics-service'
      uses: actions/setup-python@v4
      with:
        python-version: '3.11'
        cache: 'pip'
    
    - name: Setup Go (for Go services)
      if: matrix.service == 'auth-service' || matrix.service == 'file-service'
      uses: actions/setup-go@v4
      with:
        go-version: '1.21'
    
    - name: Setup Java (for Java service)
      if: matrix.service == 'payment-service'
      uses: actions/setup-java@v3
      with:
        distribution: 'temurin'
        java-version: '17'
    
    - name: Run Lint
      run: |
        cd services/${{ matrix.service }}
        make lint || echo "Lint step optional for ${{ matrix.service }}"
    
    - name: Run Unit Tests
      run: |
        cd services/${{ matrix.service }}
        make test-unit
        # Generate coverage report
        bash <(curl -s https://codecov.io/bash) -t ${{ secrets.CODECOV_TOKEN }}
    
    - name: Build Docker Image
      run: |
        cd services/${{ matrix.service }}
        docker build -t diyapp/${{ matrix.service }}:${GITHUB_SHA::8} .
    
    - name: Scan for Vulnerabilities
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: 'diyapp/${{ matrix.service }}:${GITHUB_SHA::8}'
        format: 'sarif'
        output: 'trivy-results.sarif'
    
    - name: Upload Vulnerability Report
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: 'trivy-results.sarif'

  integration-tests:
    runs-on: ubuntu-latest
    needs: lint-and-test
    services:
      postgres:
        image: postgres:15
        env:
          POSTGRES_PASSWORD: postgres
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
      redis:
        image: redis:7-alpine
        options: >-
          --health-cmd "redis-cli ping"
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
      rabbitmq:
        image: rabbitmq:3-management
        options: >-
          --health-cmd "rabbitmq-diagnostics -q ping"
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Start Services
      run: docker-compose -f infrastructure/docker-compose.test.yml up -d
    
    - name: Run Integration Tests
      run: |
        make test-integration
        # Wait for services to be ready
        sleep 30
    
    - name: Run E2E Tests
      run: make test-e2e
    
    - name: Upload Test Results
      if: always()
      uses: actions/upload-artifact@v3
      with:
        name: test-results
        path: test-results/

```yaml
# .github/workflows/cd.yml
name: CD Pipeline

on:
  push:
    branches: [ main ]
  workflow_dispatch:
    inputs:
      environment:
        description: 'Deployment environment'
        required: true
        default: 'staging'
        type: choice
        options:
        - staging
        - production

jobs:
  deploy:
    runs-on: ubuntu-latest
    environment: ${{ github.event.inputs.environment || 'staging' }}
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Configure AWS Credentials
      if: github.event_name == 'workflow_dispatch' || github.ref == 'refs/heads/main'
      uses: aws-actions/configure-aws-credentials@v1
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: ${{ secrets.AWS_REGION }}
    
    - name: Login to Amazon ECR
      if: github.event_name == 'workflow_dispatch' || github.ref == 'refs/heads/main'
      id: login-ecr
      uses: aws-actions/amazon-ecr-login@v1
    
    - name: Build, tag, and push image to Amazon ECR
      if: github.event_name == 'workflow_dispatch' || github.ref == 'refs/heads/main'
      env:
        ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
        ECR_REPOSITORY: diyapp-v12
        IMAGE_TAG: ${{ github.sha }}
      run: |
        # Build and push each service
        for service in api-gateway auth-service content-service ai-processor notification-service payment-service analytics-service file-service; do
          docker build -t $ECR_REGISTRY/$ECR_REPOSITORY/$service:$IMAGE_TAG ./services/$service
          docker push $ECR_REGISTRY/$ECR_REPOSITORY/$service:$IMAGE_TAG
        done
    
    - name: Deploy to EKS
      if: github.event_name == 'workflow_dispatch' || github.ref == 'refs/heads/main'
      run: |
        aws eks update-kubeconfig --name diyapp-v12-cluster --region ${{ secrets.AWS_REGION }}
        kubectl apply -f infrastructure/k8s/namespace.yaml
        kubectl apply -f infrastructure/k8s/configs/
        kubectl apply -f infrastructure/k8s/services/
        kubectl rollout status deployment/api-gateway -n diyapp-v12 --timeout=300s
    
    - name: Run Smoke Tests
      if: github.event_name == 'workflow_dispatch' || github.ref == 'refs/heads/main'
      run: |
        ./scripts/smoke-tests.sh
    
    - name: Notify Deployment
      if: always()
      uses: 8398a7/action-slack@v3
      with:
        status: ${{ job.status }}
        channel: '#deployments'
        username: 'Deploy Bot'

## 2. Convenções de Código

### 2.1 Nomenclatura
- **Variáveis/Funções**: camelCase (JavaScript/TypeScript/Java), snake_case (Python), camelCase (Go)
- **Classes/Interfaces**: PascalCase
- **Constantes**: UPPER_SNAKE_CASE
- **Arquivos**: kebab-case
- **Branches**: feature/ISSUE-ID-short-description, fix/ISSUE-ID-description

### 2.2 Commits
Usar Conventional Commits:
```
feat: add user authentication endpoint
fix(api): resolve memory leak in request handler
chore: update dependencies
docs: update API documentation
test: add unit tests for payment service
refactor: simplify database connection logic
perf: optimize image processing algorithm
```

### 2.3 APIs REST
- Versionamento: `/api/v1/resource`
- Métodos HTTP semânticos
- Plural para collections: `/api/v1/users`
- Singular para resource específico: `/api/v1/users/{id}`
- Query params para filtros: `/api/v1/users?status=active&role=admin`
- Status codes apropriados
- Paginação: `limit`, `offset` ou `cursor`
- Rate limiting: `X-RateLimit-Limit`, `X-RateLimit-Remaining`

### 2.4 Documentação de APIs
- OpenAPI 3.0 para todos os endpoints
- Exemplos de request/response
- Erros documentados
- Autenticação requirements

## 3. Segurança

### 3.1 Autenticação/Autorização
- JWT com expiração curta (15min)
- Refresh tokens com expiração longa (7 dias)
- OAuth2 para integrações de terceiros
- RBAC (Role-Based Access Control)
- Validação de escopo por endpoint

### 3.2 Dados Sensíveis
- Nunca logar dados sensíveis
- Criptografia em repouso (AES-256)
- Criptografia em trânsito (TLS 1.3)
- Hash com salt para senhas (bcrypt/argon2)
- Secrets em vault (Hashicorp Vault/AWS Secrets Manager)

### 3.3 Input Validation
- Validação em todas as entradas
- Sanitização contra XSS
- Proteção contra SQL injection
- Limites de tamanho de payload
- Content-Type validation

## 4. Performance

### 4.1 Banco de Dados
- Índices para queries frequentes
- Evitar N+1 queries
- Connection pooling
- Read replicas para cargas pesadas de leitura
- Caching estratégico com Redis

### 4.2 API Design
- Paginação para listas grandes
- Campos parciais com GraphQL-like queries
- Compression (gzip/brotli)
- CDN para assets estáticos
- Cache headers apropriados

### 4.3 Monitoramento
- Métricas por endpoint (latência, erro rate)
- Alertas para P95 > 200ms
- Tracing distribuído com correlation IDs
- Logs estruturados (JSON)
- Health checks com readiness/liveness probes

## 5. Testes

### 5.1 Cobertura Mínima
- Unit tests: 80%+
- Integration tests: 70%+
- Critical paths: 100%

### 5.2 Tipos de Testes
- **Unit**: Testes isolados de funções/classes
- **Integration**: Testes com dependências (DB, cache, external APIs)
- **E2E**: Fluxos completos de usuário
- **Load**: Testes de performance sob carga
- **Security**: SAST/DAST scans

-- Função para criar partições automaticamente
CREATE OR REPLACE FUNCTION dbaas_monitoring.create_log_partition()
RETURNS TRIGGER AS $$
DECLARE
    partition_date DATE;
    partition_name TEXT;
    start_date DATE;
    end_date DATE;
BEGIN
    -- Cria partição para hoje + 7 dias no futuro
    FOR i IN 0..7 LOOP
        partition_date := CURRENT_DATE + i;
        partition_name := 'execution_logs_' || TO_CHAR(partition_date, 'YYYY_MM_DD');
        start_date := partition_date;
        end_date := partition_date + 1;
        
        -- Cria partição se não existir
        IF NOT EXISTS (
            SELECT 1 FROM pg_tables 
            WHERE schemaname = 'dbaas_monitoring' 
            AND tablename = partition_name
        ) THEN
            EXECUTE format('
                CREATE TABLE dbaas_monitoring.%I
                PARTITION OF dbaas_monitoring.execution_logs
                FOR VALUES FROM (%L) TO (%L)
                WITH (fillfactor = 95)
            ', partition_name, start_date, end_date);
            
            -- Cria índices específicos da partição
            EXECUTE format('
                CREATE INDEX ON dbaas_monitoring.%I (app_id, log_timestamp DESC)
                WHERE log_level IN (''ERROR'', ''FATAL'')
            ', partition_name);
            
            EXECUTE format('
                CREATE INDEX ON dbaas_monitoring.%I (log_timestamp DESC, component)
                INCLUDE (log_level, message)
            ', partition_name);
            
            EXECUTE format('
                CREATE INDEX ON dbaas_monitoring.%I USING gin (metadata)
                WHERE metadata IS NOT NULL
            ', partition_name);
            
            EXECUTE format('
                CREATE INDEX ON dbaas_monitoring.%I (execution_id, log_timestamp DESC)
            ', partition_name);
            
            RAISE NOTICE 'Partição criada: %', partition_name;
        END IF;
    END LOOP;
    
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- View para métricas agregadas por hora
CREATE OR REPLACE VIEW dbaas_monitoring.vw_hourly_metrics AS
SELECT 
    app_id,
    DATE_TRUNC('hour', metric_timestamp) as hour_bucket,
    metric_name,
    COUNT(*) as sample_count,
    AVG(metric_value) as avg_value,
    MIN(metric_value) as min_value,
    MAX(metric_value) as max_value,
    PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY metric_value) as p95_value,
    PERCENTILE_CONT(0.99) WITHIN GROUP (ORDER BY metric_value) as p99_value
FROM dbaas_monitoring.performance_metrics
WHERE metric_timestamp > NOW() - INTERVAL '24 hours'
GROUP BY app_id, DATE_TRUNC('hour', metric_timestamp), metric_name;

    MIN(CASE WHEN is_healthy THEN response_time_ms END) as min_healthy_response

-- Função para limpeza automática de dados antigos
CREATE OR REPLACE FUNCTION dbaas_monitoring.cleanup_old_data()
RETURNS INTEGER AS $$
DECLARE
    deleted_logs INTEGER;
    deleted_metrics INTEGER;
    app_record RECORD;
BEGIN
    deleted_logs := 0;
    deleted_metrics := 0;
    
    -- Para cada aplicação, remove dados mais antigos que retention_days
    FOR app_record IN SELECT app_id, retention_days FROM dbaas_monitoring.applications WHERE is_active = TRUE
    LOOP
        -- Remove logs antigos
        DELETE FROM dbaas_monitoring.execution_logs 
        WHERE app_id = app_record.app_id 
        AND log_timestamp < NOW() - (app_record.retention_days || ' days')::INTERVAL;
        
        GET DIAGNOSTICS deleted_logs = ROW_COUNT;
        
        -- Remove métricas antigas
        DELETE FROM dbaas_monitoring.performance_metrics 
        WHERE app_id = app_record.app_id 
        AND metric_timestamp < NOW() - (app_record.retention_days || ' days')::INTERVAL;
        
        GET DIAGNOSTICS deleted_metrics = ROW_COUNT;
        
        RAISE NOTICE 'App %: Removed % logs and % metrics older than % days', 
            app_record.app_id, deleted_logs, deleted_metrics, app_record.retention_days;
    END LOOP;
    
    RETURN deleted_logs + deleted_metrics;
END;
$$ LANGUAGE plpgsql;

class JSONFormatter(logging.Formatter):
    """Formatter para logs em JSON estruturado"""
    
    def format(self, record: logging.LogRecord) -> str:
        log_data: Dict[str, Any] = {
            "timestamp": self.formatTime(record),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
            "correlation_id": correlation_id_ctx.get(),
            "module": record.module,
            "function": record.funcName,
            "line": record.lineno,
        }
        
        # Adiciona campos extras
        if hasattr(record, 'extra_fields'):
            log_data.update(record.extra_fields)
        
        # Adiciona exception se houver
        if record.exc_info:
            log_data["exception"] = self.formatException(record.exc_info)
        
        return json.dumps(log_data)

def setup_logging(level: str = "INFO", format_type: str = "json"):
    """Configura sistema de logging"""
    
    logger = logging.getLogger()
    logger.setLevel(getattr(logging, level.upper()))
    
    # Remove handlers existentes
    logger.handlers.clear()
    
    # Console handler
    console_handler = logging.StreamHandler()
    
    if format_type == "json":
        console_handler.setFormatter(JSONFormatter())
    else:
        console_handler.setFormatter(
            logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
        )
    
    logger.addHandler(console_handler)
    
    return logger

# 3. CIRCUIT BREAKER - src/utils/circuit_breaker.py
import time
from typing import Callable, Any, Optional
from enum import Enum
from dataclasses import dataclass
import threading

class CircuitState(Enum):
    CLOSED = "CLOSED"
    OPEN = "OPEN"
    HALF_OPEN = "HALF_OPEN"

@dataclass
class CircuitBreakerStats:
    failures: int = 0
    successes: int = 0
    last_failure_time: Optional[float] = None
    state: CircuitState = CircuitState.CLOSED

class CircuitBreaker:
    """Implementação de Circuit Breaker com thread safety"""
    
    def __init__(
        self,
        failure_threshold: int = 5,
        recovery_timeout: int = 60,
        name: str = "default"
    ):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.name = name
        self._stats = CircuitBreakerStats()
        self._lock = threading.RLock()
    
    def execute(self, func: Callable, *args, **kwargs) -> Any:
        """Executa função com proteção de circuit breaker"""
        
        with self._lock:
            # Verifica se circuito está aberto
            if self._stats.state == CircuitState.OPEN:
                # Verifica se já pode tentar recovery
                if (self._stats.last_failure_time and 
                    time.time() - self._stats.last_failure_time > self.recovery_timeout):
                    self._stats.state = CircuitState.HALF_OPEN
                else:
                    raise CircuitBreakerOpenError(
                        f"Circuit breaker '{self.name}' is OPEN"
                    )
        
        try:
            # Executa função
            result = func(*args, **kwargs)
            
            with self._lock:
                # Sucesso - reseta contadores
                if self._stats.state == CircuitState.HALF_OPEN:
                    self._stats.state = CircuitState.CLOSED
                
                self._stats.successes += 1
                self._stats.failures = 0
            
            return result
            
        except Exception as e:
            with self._lock:
                # Falha - atualiza contadores
                self._stats.failures += 1
                self._stats.last_failure_time = time.time()
                
                # Verifica se deve abrir circuito
                if (self._stats.failures >= self.failure_threshold and 
                    self._stats.state != CircuitState.OPEN):
                    self._stats.state = CircuitState.OPEN
            
            raise
    
    def get_stats(self) -> Dict[str, Any]:
        """Retorna estatísticas do circuit breaker"""
        with self._lock:
            return {
                "name": self.name,
                "state": self._stats.state.value,
                "failures": self._stats.failures,
                "successes": self._stats.successes,
                "last_failure_time": self._stats.last_failure_time,
            }

class CircuitBreakerOpenError(Exception):
    """Exceção quando circuit breaker está aberto"""
    pass

from datetime import datetime, timedelta
from typing import Optional, Dict, Any
from jose import JWTError, jwt
from passlib.context import CryptContext
import secrets

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def verify_password(plain_password: str, hashed_password: str) -> bool:
    """Verifica senha"""
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    """Gera hash da senha"""
    return pwd_context.hash(password)

def create_access_token(
    data: Dict[str, Any],
    secret_key: str,
    algorithm: str,
    expires_delta: Optional[timedelta] = None
) -> str:
    """Cria token JWT"""
    to_encode = data.copy()
    
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=15)
    
    to_encode.update({"exp": expire, "iat": datetime.utcnow()})
    encoded_jwt = jwt.encode(to_encode, secret_key, algorithm=algorithm)
    return encoded_jwt

def decode_token(
    token: str,
    secret_key: str,
    algorithm: str
) -> Optional[Dict[str, Any]]:
    """Decodifica token JWT"""
    try:
        payload = jwt.decode(token, secret_key, algorithms=[algorithm])
        return payload
    except JWTError:
        return None

def generate_api_key() -> str:
    """Gera API key segura"""
    return secrets.token_urlsafe(32)

import redis
from typing import Optional, Any, Union
import json
import pickle
from functools import wraps
import time
from src.utils.circuit_breaker import CircuitBreaker
from src.utils.logger import get_logger

logger = get_logger(__name__)

class CacheService:
    """Serviço de cache com Redis e circuit breaker"""
    
    def __init__(self, redis_url: str, default_ttl: int = 300):
        self.redis_url = redis_url
        self.default_ttl = default_ttl
        self._client: Optional[redis.Redis] = None
        self.circuit_breaker = CircuitBreaker(
            failure_threshold=3,
            recovery_timeout=30,
            name="redis_cache"
        )
    
    @property
    def client(self) -> redis.Redis:
        """Lazy initialization do cliente Redis"""
        if self._client is None:
            try:
                self._client = redis.from_url(
                    self.redis_url,
                    socket_connect_timeout=5,
                    socket_timeout=5,
                    retry_on_timeout=True,
                    decode_responses=False
                )
                # Testa conexão
                self._client.ping()
                logger.info("Redis connection established")
            except Exception as e:
                logger.error(f"Failed to connect to Redis: {e}")
                raise
        
        return self._client
    
    def get(self, key: str) -> Optional[Any]:
        """Obtém valor do cache"""
        try:
            return self.circuit_breaker.execute(self._get, key)
        except Exception as e:
            logger.warning(f"Cache get failed for key {key}: {e}")
            return None
    
    def _get(self, key: str) -> Optional[Any]:
        """Implementação interna de get"""
        data = self.client.get(key)
        if data:
            try:
                return pickle.loads(data)
            except:
                # Fallback para JSON
                return json.loads(data.decode('utf-8'))
        return None
    
    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Define valor no cache"""
        try:
            return self.circuit_breaker.execute(self._set, key, value, ttl)
        except Exception as e:
            logger.warning(f"Cache set failed for key {key}: {e}")
            return False
    
    def _set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Implementação interna de set"""
        try:
            # Tenta serializar com pickle primeiro
            serialized = pickle.dumps(value)
        except:
            # Fallback para JSON
            serialized = json.dumps(value).encode('utf-8')
        
        expire = ttl or self.default_ttl
        return bool(self.client.setex(key, expire, serialized))
    
    def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        try:
            return self.circuit_breaker.execute(self._delete, key)

import React, { createContext, useContext, useEffect, useState } from 'react'

type Theme = 'light' | 'dark'

interface ThemeContextType {
  theme: Theme
  toggleTheme: () => void
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined)

export const useTheme = () => {
  const context = useContext(ThemeContext)
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider')
  }
  return context
}

interface ThemeProviderProps {
  children: React.ReactNode
}

export const ThemeProvider: React.FC<ThemeProviderProps> = ({ children }) => {
  const [theme, setTheme] = useState<Theme>(() => {
    const saved = localStorage.getItem('diyapp-theme')
    return (saved as Theme) || 'light'
  })

  useEffect(() => {
    const root = document.documentElement
    root.classList.remove('light', 'dark')
    root.classList.add(theme)
    localStorage.setItem('diyapp-theme', theme)
  }, [theme])

  const toggleTheme = () => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light')
  }

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  )
}

import React from 'react'

interface CardProps extends React.HTMLAttributes<HTMLDivElement> {
  variant?: 'default' | 'outline' | 'filled'
  padding?: 'none' | 'sm' | 'md' | 'lg'
}

  children,
  variant = 'default',
  padding = 'md',
  className,
  ...props
}) => {
  const variantClasses = {
    default: 'bg-white dark:bg-neutral-900 border border-neutral-200 dark:border-neutral-800',
    outline: 'border border-neutral-300 dark:border-neutral-700',
    filled: 'bg-neutral-50 dark:bg-neutral-800',
  }

  const paddingClasses = {
    none: '',
    sm: 'p-4',
    md: 'p-6',
    lg: 'p-8',
  }

  return (
    <div
      className={clsx(
        'rounded-xl shadow-sm',
        variantClasses[variant],
        paddingClasses[padding],
        className
      )}
      {...props}
    >
      {children}
    </div>
  )
}

import axios from 'axios'
import { toast } from 'react-hot-toast'

const API_BASE_URL = import.meta.env.VITE_API_URL || 'http://localhost:3001/api'

export const apiClient = axios.create({
  baseURL: API_BASE_URL,
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json',
  },
})

// Request interceptor para adicionar token de autenticação
apiClient.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('diyapp-token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// Response interceptor para tratamento global de erros
apiClient.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response) {
      switch (error.response.status) {
        case 401:
          localStorage.removeItem('diyapp-token')
          window.location.href = '/login'
          toast.error('Sessão expirada. Por favor, faça login novamente.')
          break
        case 403:
          toast.error('Você não tem permissão para realizar esta ação.')
          break
        case 429:
          toast.error('Muitas requisições. Por favor, aguarde um momento.')
          break
        case 500:
          toast.error('Erro interno do servidor. Tente novamente mais tarde.')
          break
        default:
          toast.error(error.response.data?.message || 'Erro na requisição')
      }
    } else if (error.request) {
      toast.error('Servidor não disponível. Verifique sua conexão.')
    } else {
      toast.error('Erro ao configurar requisição.')
    }
    return Promise.reject(error)
  }
)

// Funções de retry com backoff exponencial
export const retryWithBackoff = async <T>(
  fn: () => Promise<T>,
  maxRetries = 3,
  baseDelay = 1000
): Promise<T> => {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn()
    } catch (error) {
      if (i === maxRetries - 1) throw error
      const delay = baseDelay * Math.pow(2, i)
      await new Promise(resolve => setTimeout(resolve, delay))
    }
  }
  throw new Error('Max retries exceeded')
}

import { apiClient, retryWithBackoff } from './client'

export interface Execution {
  id: string
  agentId: string
  agentName: string
  status: 'running' | 'success' | 'failed' | 'pending'
  startedAt: string
  completedAt?: string
  duration?: number
  metrics?: {
    tokensUsed: number
    cost: number
    accuracy: number
  }
  logs: string[]
}

export interface ExecutionFilters {
  status?: Execution['status']
  agentId?: string
  dateRange?: {
    from: string
    to: string
  }
  search?: string
}

export const executionsApi = {
  // Buscar execuções com paginação e filtros
  getExecutions: async (page = 1, limit = 20, filters?: ExecutionFilters) => {
    return retryWithBackoff(() =>
      apiClient.get<{ data: Execution[]; total: number; page: number }>('/executions', {
        params: { page, limit, ...filters }
      })
    )
  },

  // Buscar execução específica
  getExecution: async (id: string) => {
    return retryWithBackoff(() =>
      apiClient.get<Execution>(`/executions/${id}`)
    )
  },

  // Stream de logs em tempo real (SSE)
  streamExecutionLogs: (executionId: string, onLog: (log: string) => void) => {
    const eventSource = new EventSource(
      `${import.meta.env.VITE_API_URL}/executions/${executionId}/logs/stream`
    )

    eventSource.onmessage = (event) => {
      onLog(event.data)
    }

    eventSource.onerror = () => {
      eventSource.close()
    }

    return () => eventSource.close()
  },

  // Cancelar execução
  cancelExecution: async (id: string) => {
    return retryWithBackoff(() =>
      apiClient.post(`/executions/${id}/cancel`)
    )
  },

  // Métricas agregadas
  getMetrics: async () => {
    return retryWithBackoff(() =>
      apiClient.get<{
        totalExecutions: number
        successRate: number
        avgDuration: number
        totalCost: number
        todayExecutions: number
      }>('/executions/metrics')
    )
  }
}
