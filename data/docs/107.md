# DIYAPP Evolution - V12 Core - Arquitetura de MicrosserviГ§os

## 1. VisГЈo Geral da Arquitetura V12

A V12 implementa uma arquitetura de microsserviГ§os orientada a agentes autГґnomos, onde cada agente Г© um serviГ§o independente com responsabilidades bem definidas. A comunicaГ§ГЈo ocorre via eventos e APIs REST/WebSocket.

### 1.1 PrincГ­pios Arquiteturais
- **Autonomia Total**: Cada agente pode operar independentemente
- **ResiliГЄncia**: Circuit breakers, retry policies e fallbacks em todos os serviГ§os
- **Observabilidade**: Logs estruturados, mГ©tricas e tracing distribuГ­do
- **Escalabilidade Horizontal**: Stateless por design, com session stickiness opcional

```javascript
// shared/libs/api-client.js
const axios = require('axios');
const CircuitBreaker = require('opossum');

class ResilientAPIClient {
  constructor(serviceName, baseURL) {
    this.serviceName = serviceName;
    this.client = axios.create({
      baseURL,
      timeout: 10000,
      headers: {
        'Content-Type': 'application/json',
        'X-Correlation-ID': this.generateCorrelationId()
      }
    });

    // Circuit breaker configuration
    this.breaker = new CircuitBreaker(
      async (method, url, data) => {
        const response = await this.client[method](url, data);
        return response.data;
      },
      {
        timeout: 5000,
        errorThresholdPercentage: 50,
        resetTimeout: 30000,
        rollingCountTimeout: 10000,
        rollingCountBuckets: 10
      }
    );

    // Fallback function
    this.breaker.fallback(() => ({
      status: 'fallback',
      message: `Service ${serviceName} unavailable`,
      timestamp: new Date().toISOString()
    }));
  }

  async request(method, url, data = null) {
    try {
      return await this.breaker.fire(method, url, data);
    } catch (error) {
      this.logError(error);
      throw error;
    }
  }

  generateCorrelationId() {
    return `corr_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }

  logError(error) {
    console.error({
      service: this.serviceName,
      error: error.message,
      timestamp: new Date().toISOString(),
      correlationId: error.config?.headers['X-Correlation-ID']
    });
  }
}

module.exports = ResilientAPIClient;
```

### 5.2 ComunicaГ§ГЈo AssГ­ncrona (Event Bus)
```javascript
// infrastructure/message-bus/event-bus.js
const amqp = require('amqplib');
const EventSchemas = require('../../shared/events/event-schemas');

class EventBus {
  constructor() {
    this.connection = null;
    this.channel = null;
    this.exchanges = {
      COMMANDS: 'agent.commands',
      EVENTS: 'agent.events',
      DEAD_LETTER: 'agent.dlx'
    };
  }

  async connect() {
    const rabbitmqUrl = process.env.RABBITMQ_URL || 'amqp://localhost';
    this.connection = await amqp.connect(rabbitmqUrl);
    this.channel = await this.connection.createChannel();
    
    // Declare exchanges
    await this.channel.assertExchange(this.exchanges.COMMANDS, 'direct', { durable: true });
    await this.channel.assertExchange(this.exchanges.EVENTS, 'topic', { durable: true });
    await this.channel.assertExchange(this.exchanges.DEAD_LETTER, 'fanout', { durable: true });
    
    console.log('EventBus connected and exchanges declared');
  }

  async publish(exchange, routingKey, event) {
    if (!this.channel) {
      throw new Error('EventBus not connected');
    }

    // Validate event schema
    const schema = EventSchemas[event.eventType];
    if (schema) {
      const { error } = schema.validate(event);
      if (error) {
        throw new Error(`Invalid event schema: ${error.message}`);
      }
    }

    const message = Buffer.from(JSON.stringify(event));
    const options = {
      persistent: true,
      timestamp: Date.now(),
      headers: {
        'x-event-type': event.eventType,
        'x-correlation-id': event.metadata?.correlationId || 'unknown'
      }
    };

    return this.channel.publish(exchange, routingKey, message, options);
  }

  async subscribe(queue, exchange, routingKey, handler) {
    if (!this.channel) {
      throw new Error('EventBus not connected');
    }

    // Declare queue with DLX
    const queueResult = await this.channel.assertQueue(queue, {
      durable: true,
      deadLetterExchange: this.exchanges.DEAD_LETTER,
      deadLetterRoutingKey: routingKey
    });

    // Bind queue to exchange
    await this.channel.bindQueue(queueResult.queue, exchange, routingKey);

    // Consume messages
    await this.channel.consume(queueResult.queue, async (msg) => {
      try {
        const event = JSON.parse(msg.content.toString());
        await handler(event);
        this.channel.ack(msg);
      } catch (error) {
        console.error('Error processing message:', error);
        
        // Reject with requeue based on retry count
        const retryCount = (msg.properties.headers['x-retry-count'] || 0) + 1;
        if (retryCount <= 3) {
          msg.properties.headers['x-retry-count'] = retryCount;
          this.channel.nack(msg, false, true); // Requeue
        } else {
          // Send to dead letter exchange
          this.channel.nack(msg, false, false);
        }
      }
    });

    console.log(`Subscribed to ${exchange} with routing key ${routingKey}`);
  }

  async disconnect() {
    if (this.channel) await this.channel.close();
    if (this.connection) await this.connection.close();
  }
}

module.exports = new EventBus();
```

@contextmanager
def get_db() -> Generator[Session, None, None]:
    """
    Context manager for database sessions with automatic cleanup.
    Ensures sessions are always closed, even on exceptions.
    """
    db = SessionLocal()
    try:
        yield db
        db.commit()
    except Exception as e:
        db.rollback()
        logger.error("database_session_rollback", error=str(e), error_type=type(e).__name__)
        raise
    finally:
        db.close()

def init_db() -> None:
    """
    Initialize database tables.
    Should be called on application startup.
    """
    Base.metadata.create_all(bind=engine)
    logger.info("database_tables_initialized")

from typing import TypeVar, Generic, Type, Optional, List, Dict, Any
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
import structlog

from src.database.session import Base as BaseModel

ModelType = TypeVar("ModelType", bound=BaseModel)
logger = structlog.get_logger(__name__)

class BaseRepository(Generic[ModelType]):
    """Base repository with CRUD operations."""
    
    def __init__(self, model: Type[ModelType], db: Session):
        self.model = model
        self.db = db
    
    def get(self, id: str) -> Optional[ModelType]:
        """Get by ID with correlation ID logging."""
        try:
            result = self.db.query(self.model).filter(self.model.id == id).first()
            return result
        except Exception as e:
            logger.error("repository_get_error", 
                        model=self.model.__name__, 
                        id=id, 
                        error=str(e))
            raise
    
    def get_multi(
        self, 
        skip: int = 0, 
        limit: int = 100,
        filters: Optional[Dict[str, Any]] = None
    ) -> List[ModelType]:
        """Get multiple records with pagination and filtering."""
        try:
            query = self.db.query(self.model)
            
            # Apply filters
            if filters:
                filter_conditions = []
                for key, value in filters.items():
                    if hasattr(self.model, key):
                        if isinstance(value, list):
                            filter_conditions.append(getattr(self.model, key).in_(value))
                        else:
                            filter_conditions.append(getattr(self.model, key) == value)
                
                if filter_conditions:
                    query = query.filter(and_(*filter_conditions))
            
            return query.offset(skip).limit(limit).all()
        except Exception as e:
            logger.error("repository_get_multi_error",
                        model=self.model.__name__,
                        skip=skip,
                        limit=limit,
                        error=str(e))
            raise
    
    def create(self, obj_in: Dict[str, Any]) -> ModelType:
        """Create new record with parameterized query."""
        try:
            db_obj = self.model(**obj_in)
            self.db.add(db_obj)
            self.db.flush()
            return db_obj
        except Exception as e:
            logger.error("repository_create_error",
                        model=self.model.__name__,
                        data=obj_in,
                        error=str(e))
            raise
    
    def update(self, db_obj: ModelType, obj_in: Dict[str, Any]) -> ModelType:
        """Update existing record."""
        try:
            for field in obj_in:
                if hasattr(db_obj, field):

                    self.db.add(db_obj)

                self.db.flush()
                return True

```javascript
const logger = require('./logger');

function observabilityMiddleware(req, res, next) {
  const start = Date.now();
  const loggerInstance = new logger(req.serviceName || 'unknown-service');

  // Log request
  loggerInstance.info('Incoming request', {
    method: req.method,
    url: req.url,
    ip: req.ip,
    userAgent: req.get('User-Agent'),
    correlationId: req.headers['x-correlation-id'] || generateCorrelationId()
  });

  // Capture response
  const originalSend = res.send;
  res.send = function(body) {
    const duration = (Date.now() - start) / 1000;
    
    // Record metrics
    loggerInstance.recordRequest(
      req.method,
      req.path,
      res.statusCode,
      duration
    );

    // Log response
    loggerInstance.info('Request completed', {
      method: req.method,
      url: req.url,
      statusCode: res.statusCode,
      duration: `${duration.toFixed(3)}s`,
      contentLength: res.get('Content-Length')
    });

    originalSend.call(this, body);
  };

  // Error handling
  res.on('error', (error) => {
    loggerInstance.error('Response error', error, {
      method: req.method,
      url: req.url,
      statusCode: res.statusCode
    });
  });

  next();
}

function generateCorrelationId() {
  return 'cid-' + Date.now() + '-' + Math.random().toString(36).substr(2, 9);
}

function agentPerformanceMiddleware(agentType) {
  return function(req, res, next) {
    const start = Date.now();
    
    res.on('finish', () => {

      // Log performance
      if (duration > 2) {
        loggerInstance.warn('Slow agent response', {
          agentType,
          duration,
          endpoint: req.path
        });
      }
    });
    
    next();
  };
}
```

```python
"""
Configuration module with security-focused settings
"""
from typing import Optional, List
from pydantic_settings import BaseSettings
from pydantic import AnyHttpUrl, validator


class Settings(BaseSettings):
    # Application
    app_name: str = "DIYAPP V12 Core"
    app_env: str = "development"
    app_secret_key: str
    app_debug: bool = False
    
    # CORS
    backend_cors_origins: List[AnyHttpUrl] = []
    
    @validator("backend_cors_origins", pre=True)
    def assemble_cors_origins(cls, v: str | List[str]) -> List[str] | str:
        if isinstance(v, str) and not v.startswith("["):
            return [i.strip() for i in v.split(",")]
        elif isinstance(v, (list, str)):
            return v
        raise ValueError(v)
    
    # Database
    database_url: str
    database_pool_size: int = 20
    database_max_overflow: int = 40
    
    # Redis
    redis_url: str = "redis://localhost:6379/0"
    redis_password: Optional[str] = None
    
    # JWT
    jwt_secret_key: str
    jwt_algorithm: str = "HS256"
    jwt_access_token_expire_minutes: int = 15  # STRIDE: Short-lived tokens
    jwt_refresh_token_expire_days: int = 7
    
    # Security
    bcrypt_rounds: int = 12
    rate_limit_per_minute: int = 60
    max_login_attempts: int = 5
    lockout_time_minutes: int = 15
    
    # Audit
    audit_log_enabled: bool = True

    # LLM Security (if applicable)
    llm_prompt_injection_detection: bool = True
    llm_output_sanitization: bool = True
    
    class Config:
        env_file = ".env"
        case_sensitive = True
```

```python
from jose import JWTError, jwt
from passlib.context import CryptContext
from cryptography.fernet import Fernet
import base64
import hashlib
import structlog

logger = structlog.get_logger(__name__)

# Password hashing context
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

# Encryption for sensitive data at rest
fernet_key = Fernet.generate_key()
cipher_suite = Fernet(fernet_key)


class SecurityManager:
    """STRIDE-based security manager"""
    
    @staticmethod
    def verify_password(plain_password: str, hashed_password: str) -> bool:
        """Verify a password against its hash"""
        return pwd_context.verify(plain_password, hashed_password)
    
    @staticmethod
    def get_password_hash(password: str) -> str:
        """Generate password hash"""
        return pwd_context.hash(password)
    
    @staticmethod
    def create_access_token(
        data: Dict[str, Any],
        secret_key: str,
        algorithm: str,
        expires_delta: Optional[timedelta] = None
    ) -> str:
        """Create JWT access token with short expiration"""
        to_encode = data.copy()
        if expires_delta:
            expire = datetime.utcnow() + expires_delta
        else:
            expire = datetime.utcnow() + timedelta(minutes=15)
        
        to_encode.update({
            "exp": expire,
            "iat": datetime.utcnow(),
            "type": "access",
            "jti": secrets.token_urlsafe(32)  # Unique token ID for repudiation tracking
        })
        
        encoded_jwt = jwt.encode(to_encode, secret_key, algorithm=algorithm)
        return encoded_jwt
    
    @staticmethod
    def create_refresh_token(
        data: Dict[str, Any],
        secret_key: str,
        algorithm: str,
        expires_delta: Optional[timedelta] = None
    ) -> str:
        """Create JWT refresh token"""
        to_encode = data.copy()
        if expires_delta:
            expire = datetime.utcnow() + expires_delta
        else:
            expire = datetime.utcnow() + timedelta(days=7)
        
        to_encode.update({
            "exp": expire,
            "iat": datetime.utcnow(),
            "type": "refresh",
            "jti": secrets.token_urlsafe(32)
        })
        
        encoded_jwt = jwt.encode(to_encode, secret_key, algorithm=algorithm)
        return encoded_jwt
    
    @staticmethod
    def verify_token(token: str, secret_key: str, algorithm: str) -> Dict[str, Any]:
        """Verify and decode JWT token"""
        try:
            payload = jwt.decode(token, secret_key, algorithms=[algorithm])
            return payload
        except JWTError as e:
            logger.error("token_verification_failed", error=str(e))
            raise
    
    @staticmethod
    def encrypt_sensitive_data(data: str) -> str:
        """Encrypt sensitive data at rest (AES-256 via Fernet)"""
        encrypted_data = cipher_suite.encrypt(data.encode())
        return base64.urlsafe_b64encode(encrypted_data).decode()
    
    @staticmethod
    def decrypt_sensitive_data(encrypted_data: str) -> str:
        """Decrypt sensitive data"""
        encrypted_bytes = base64.urlsafe_b64decode(encrypted_data)
        decrypted_data = cipher_suite.decrypt(encrypted_bytes)
        return decrypted_data.decode()
    
    @staticmethod
    def mask_pii(data: str, visible_chars: int = 4) -> str:
        """Mask PII data for logs and error messages"""
        if len(data) <= visible_chars * 2:
            return "*" * len(data)
        
        visible_start = data[:visible_chars]
        visible_end = data[-visible_chars:] if visible_chars > 0 else ""
        masked_middle = "*" * (len(data) - (visible_chars * 2))
        
        return f"{visible_start}{masked_middle}{visible_end}"
    
    @staticmethod
    def generate_csrf_token() -> str:
        """Generate CSRF token for form protection"""
        return secrets.token_urlsafe(32)
    
    @staticmethod
    def validate_csrf_token(token: str, expected_token: str) -> bool:
        """Validate CSRF token with timing-safe comparison"""
        return secrets.compare_digest(token, expected_token)
    
    @staticmethod
    def sanitize_sql_input(input_str: str) -> str:
        """Basic SQL injection prevention"""
        dangerous_patterns = [
            "'", "\"", ";", "--", "/*", "*/", "xp_", "sp_", 
            "union", "select", "insert", "update", "delete", "drop",
            "exec", "execute", "truncate"
        ]
        
        sanitized = input_str
        for pattern in dangerous_patterns:
            sanitized = sanitized.replace(pattern, "")
        
        return sanitized
    
    @staticmethod
    def sanitize_html_input(input_str: str) -> str:
        """Basic HTML/JS injection prevention"""
        import html
        return html.escape(input_str)
    
    @staticmethod
    def sanitize_llm_output(output: str) -> str:
        """Sanitize LLM output to prevent XSS and code injection"""
        sanitized = SecurityManager.sanitize_html_input(output)
        
        # Remove potentially dangerous patterns
        dangerous_patterns = [
            "<script>", "</script>", "javascript:", "onload=", "onerror=",
            "eval(", "Function(", "setTimeout(", "setInterval("
        ]
        
        for pattern in dangerous_patterns:
            sanitized = sanitized.replace(pattern, "")
        
        return sanitized
```

```javascript
class CommandParser {
  constructor() {
    this.availableCommands = {
      scale_service: ['service', 'instances', 'reason'],
      restart_service: ['service', 'reason', 'graceful'],
      reroute_traffic: ['from', 'to', 'percentage', 'reason'],
      adjust_resource: ['service', 'cpu', 'memory', 'reason'],
      run_maintenance: ['task', 'window', 'priority'],
      alert_team: ['level', 'message', 'channel'],
      backup_system: ['component', 'retention', 'reason']
    };
  }

  parseLLMResponse(response) {
    try {
      // Extract JSON from response (handling potential markdown or extra text)
      const jsonMatch = response.match(/\{[\s\S]*\}/);
      if (!jsonMatch) {
        throw new Error('No JSON found in response');
      }

      const parsed = JSON.parse(jsonMatch[0]);
      
      // Validate structure
      this.validateResponse(parsed);
      
      // Validate each command
      if (parsed.commands && Array.isArray(parsed.commands)) {
        parsed.commands.forEach(cmd => this.validateCommand(cmd));
      }

      return {
        success: true,
        data: parsed,
        raw: response
      };
    } catch (error) {
      return {
        success: false,
        error: `Failed to parse LLM response: ${error.message}`,
        raw: response
      };
    }
  }

  validateResponse(response) {
    const requiredFields = ['decision', 'commands', 'confidence', 'reasoning'];
    requiredFields.forEach(field => {
      if (!(field in response)) {
        throw new Error(`Missing required field: ${field}`);
      }
    });

    if (!Array.isArray(response.commands)) {
      throw new Error('Commands must be an array');
    }

    if (response.confidence < 0 || response.confidence > 1) {
      throw new Error('Confidence must be between 0 and 1');
    }
  }

  validateCommand(command) {
    if (!command.command) {
      throw new Error('Command missing "command" field');
    }

    if (!this.availableCommands[command.command]) {
      throw new Error(`Unknown command: ${command.command}`);
    }

    if (!command.parameters) {
      throw new Error(`Command ${command.command} missing parameters`);
    }

    // Check required parameters
    const requiredParams = this.availableCommands[command.command];
    requiredParams.forEach(param => {
      if (!(param in command.parameters)) {
        throw new Error(`Command ${command.command} missing required parameter: ${param}`);
      }
    });

    // Validate priority
    const validPriorities = ['low', 'medium', 'high', 'critical'];
    if (command.priority && !validPriorities.includes(command.priority)) {
      throw new Error(`Invalid priority: ${command.priority}`);
    }

    // Validate execution window
    const validWindows = ['immediate', 'next_window', 'scheduled'];
    if (command.execution_window && !validWindows.includes(command.execution_window)) {
      throw new Error(`Invalid execution window: ${command.execution_window}`);
    }
  }

  formatForExecution(parsedResponse) {
    return {
      timestamp: new Date().toISOString(),
      decisionId: `dec_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
      summary: parsedResponse.data.decision,
      confidence: parsedResponse.data.confidence,
      commands: parsedResponse.data.commands.map((cmd, index) => ({
        id: `cmd_${Date.now()}_${index}`,
        ...cmd,
        status: 'pending',
        createdAt: new Date().toISOString()
      })),
      metadata: {
        reasoning: parsedResponse.data.reasoning,
        parsedAt: new Date().toISOString()
      }
    };
  }
}

module.exports = CommandParser;
```
