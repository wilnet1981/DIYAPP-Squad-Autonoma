import pulumi
import pulumi_aws as aws
from typing import List, Dict
from .config import Config

class Networking:
    def __init__(self, config: Config):
        self.config = config
        self.vpc = None
        self.subnets = {}
        self.security_groups = {}
        self.create_vpc()
        self.create_subnets()
        self.create_security_groups()
    
    def create_vpc(self):
        """Cria VPC com subnets públicas e privadas"""
        self.vpc = aws.ec2.Vpc(
            f"{self.config.cluster_name}-vpc",
            cidr_block=self.config.vpc_cidr,
            enable_dns_hostnames=True,
            enable_dns_support=True,
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-vpc"}
        )
        
        # Internet Gateway
        self.igw = aws.ec2.InternetGateway(
            f"{self.config.cluster_name}-igw",
            vpc_id=self.vpc.id,
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-igw"}
        )
        
        # NAT Gateway (para subnets privadas)
        self.eip = aws.ec2.Eip(
            f"{self.config.cluster_name}-nat-eip",
            domain="vpc",
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-nat-eip"}
        )
        
        self.nat_gateway = aws.ec2.NatGateway(
            f"{self.config.cluster_name}-nat",
            allocation_id=self.eip.id,
            subnet_id=self.vpc.public_subnet_ids[0],  # Usa primeira subnet pública
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-nat"}
        )
        
        pulumi.export("vpc_id", self.vpc.id)
    
    def create_subnets(self):
        """Cria subnets públicas e privadas em múltiplas AZs"""
        azs = aws.get_availability_zones(state="available")
        
        # Subnets públicas
        self.public_subnets = []
        for i, az in enumerate(azs.names[:3]):  # Usa até 3 AZs
            subnet = aws.ec2.Subnet(
                f"{self.config.cluster_name}-public-{i+1}",
                vpc_id=self.vpc.id,
                cidr_block=f"10.0.{i}.0/24",
                availability_zone=az,
                map_public_ip_on_launch=True,
                tags={
                    **self.config.get_tags(),
                    "Name": f"{self.config.cluster_name}-public-{i+1}",
                    "kubernetes.io/role/elb": "1"
                }
            )
            self.public_subnets.append(subnet)
        
        # Subnets privadas
        self.private_subnets = []
        for i, az in enumerate(azs.names[:3]):
            subnet = aws.ec2.Subnet(
                f"{self.config.cluster_name}-private-{i+1}",
                vpc_id=self.vpc.id,
                cidr_block=f"10.0.{i+100}.0/24",
                availability_zone=az,
                tags={
                    **self.config.get_tags(),
                    "Name": f"{self.config.cluster_name}-private-{i+1}",
                    "kubernetes.io/role/internal-elb": "1"
                }
            )
            self.private_subnets.append(subnet)
        
        # Route tables
        public_rt = aws.ec2.RouteTable(
            f"{self.config.cluster_name}-public-rt",
            vpc_id=self.vpc.id,
            routes=[
                aws.ec2.RouteTableRouteArgs(
                    cidr_block="0.0.0.0/0",
                    gateway_id=self.igw.id
                )
            ],
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-public-rt"}
        )
        
        private_rt = aws.ec2.RouteTable(
            f"{self.config.cluster_name}-private-rt",
            vpc_id=self.vpc.id,
            routes=[
                aws.ec2.RouteTableRouteArgs(
                    cidr_block="0.0.0.0/0",
                    nat_gateway_id=self.nat_gateway.id
                )
            ],
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-private-rt"}
        )
        
        # Associa subnets às route tables
        for i, subnet in enumerate(self.public_subnets):
            aws.ec2.RouteTableAssociation(
                f"{self.config.cluster_name}-public-rta-{i+1}",
                subnet_id=subnet.id,
                route_table_id=public_rt.id
            )
        
        for i, subnet in enumerate(self.private_subnets):
            aws.ec2.RouteTableAssociation(
                f"{self.config.cluster_name}-private-rta-{i+1}",
                subnet_id=subnet.id,
                route_table_id=private_rt.id
            )
        
        pulumi.export("public_subnet_ids", [s.id for s in self.public_subnets])
        pulumi.export("private_subnet_ids", [s.id for s in self.private_subnets])
    
    def create_security_groups(self):
        """Cria security groups para EKS e RDS"""
        # Security Group para EKS
        self.eks_sg = aws.ec2.SecurityGroup(
            f"{self.config.cluster_name}-eks-sg",
            vpc_id=self.vpc.id,
            description="Security group for EKS cluster",
            ingress=[
                aws.ec2.SecurityGroupIngressArgs(
                    description="Node to node",
                    from_port=0,
                    to_port=65535,
                    protocol="tcp",
                    self=True
                ),
                aws.ec2.SecurityGroupIngressArgs(
                    description="Node to control plane",
                    from_port=443,
                    to_port=443,
                    protocol="tcp",
                    cidr_blocks=[self.config.vpc_cidr]
                )
            ],
            egress=[
                aws.ec2.SecurityGroupEgressArgs(
                    from_port=0,
                    to_port=0,
                    protocol="-1",
                    cidr_blocks=["0.0.0.0/0"]
                )
            ],
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-eks-sg"}
        )
        
        # Security Group para RDS
        self.rds_sg = aws.ec2.SecurityGroup(
            f"{self.config.cluster_name}-rds-sg",
            vpc_id=self.vpc.id,
            description="Security group for RDS",
            ingress=[
                aws.ec2.SecurityGroupIngressArgs(
                    description="PostgreSQL from EKS",
                    from_port=5432,
                    to_port=5432,
                    protocol="tcp",
                    security_groups=[self.eks_sg.id]
                )
            ],
            tags={**self.config.get_tags(), "Name": f"{self.config.cluster_name}-rds-sg"}
        )
        
        pulumi.export("eks_security_group_id", self.eks_sg.id)
        pulumi.export("rds_security_group_id", self.rds_sg.id)

    timestamp: datetime = None
    
    def __post_init__(self):
        if self.timestamp is None:
            self.timestamp = datetime.utcnow()
    
    def to_dict(self) -> Dict:
        return {
            **asdict(self),
            'timestamp': self.timestamp.isoformat()
        }

class CircuitBreaker:
    """Circuit breaker pattern para operações de cache"""
    
    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 30):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.failures = 0
        self.last_failure_time = None
        self.state = "CLOSED"  # CLOSED, OPEN, HALF_OPEN
        
    def record_success(self):
        """Registra sucesso e reseta o circuit breaker"""
        self.failures = 0
        self.state = "CLOSED"
        self.last_failure_time = None
        
    def record_failure(self):
        """Registra falha e atualiza estado do circuit breaker"""
        self.failures += 1
        self.last_failure_time = time.time()
        
        if self.failures >= self.failure_threshold:
            self.state = "OPEN"
            logger.warning(f"Circuit breaker OPENED after {self.failures} failures")
    
    def allow_request(self) -> bool:
        """Verifica se a requisição pode passar"""
        if self.state == "CLOSED":
            return True
            
        if self.state == "OPEN":
            if self.last_failure_time and \
               (time.time() - self.last_failure_time) > self.recovery_timeout:
                self.state = "HALF_OPEN"
                logger.info("Circuit breaker moved to HALF_OPEN state")
                return True
            return False
            
        return True  # HALF_OPEN permite tentativa

class RedisCacheService:
    """Serviço de cache Redis com resiliência embutida"""
    
    def __init__(
        self,
        host: str = "localhost",
        port: int = 6379,
        db: int = 0,
        password: Optional[str] = None,
        default_ttl: int = 3600,  # 1 hora padrão
        connection_timeout: int = 5,
        socket_timeout: int = 3,
        max_retries: int = 3,
        retry_delay: float = 0.1
    ):
        self.config = {
            'host': host,
            'port': port,
            'db': db,
            'password': password,
            'socket_timeout': socket_timeout,
            'socket_connect_timeout': connection_timeout,
            'retry_on_timeout': True,
            'max_connections': 10
        }
        self.default_ttl = default_ttl
        self.max_retries = max_retries
        self.retry_delay = retry_delay
        self.circuit_breaker = CircuitBreaker()
        self._client = None
        self._metrics_callback: Optional[Callable[[CacheMetrics], None]] = None
        
    @property
    def client(self) -> redis.Redis:
        """Lazy initialization do cliente Redis com reconnection"""
        if self._client is None:
            self._connect()
        return self._client
    
    def _connect(self):
        """Estabelece conexão com Redis com retry pattern"""
        for attempt in range(self.max_retries):
            try:
                self._client = redis.Redis(**self.config)
                # Testa a conexão
                self._client.ping()
                self.circuit_breaker.record_success()
                logger.info(f"Redis connected successfully to {self.config['host']}:{self.config['port']}")
                return
            except (ConnectionError, RedisTimeoutError) as e:
                wait_time = self.retry_delay * (2 ** attempt)  # Backoff exponencial
                logger.warning(f"Redis connection attempt {attempt + 1} failed: {e}. Retrying in {wait_time}s")
                time.sleep(wait_time)
                self.circuit_breaker.record_failure()
            except RedisError as e:
                logger.error(f"Redis connection error: {e}")
                self.circuit_breaker.record_failure()
                raise
        
        raise ConnectionError(f"Failed to connect to Redis after {self.max_retries} attempts")
    
    def _record_metrics(self, metrics: CacheMetrics):
        """Registra métricas da operação"""
        if self._metrics_callback:
            try:
                self._metrics_callback(metrics)
            except Exception as e:
                logger.error(f"Failed to record metrics: {e}")
        
        # Log estruturado
        log_data = {
            "level": "info" if metrics.success else "error",
            "service": "redis_cache",
            "operation": metrics.operation.value,
            "key": metrics.key,
            "duration_ms": round(metrics.duration_ms, 2),
            "success": metrics.success,
            "cache_hit": metrics.cache_hit,
            "fallback_used": metrics.fallback_used,
            "error_type": metrics.error_type,
            "timestamp": metrics.timestamp.isoformat()
        }
        
        if metrics.success:
            logger.info("Cache operation completed", extra=log_data)
        else:
            logger.error("Cache operation failed", extra=log_data)
    
    def set_metrics_callback(self, callback: Callable[[CacheMetrics], None]):
        """Define callback para coletar métricas"""
        self._metrics_callback = callback
    
    def _execute_with_resilience(self, operation: Callable, operation_type: CacheOperation, 
                                key: str, *args, **kwargs) -> Any:
        """Executa operação com circuit breaker, retry e fallback"""
        start_time = time.time()
        metrics = CacheMetrics(
            operation=operation_type,
            key=key,
            duration_ms=0,
            success=False
        )
        
        try:
            # Verifica circuit breaker
            if not self.circuit_breaker.allow_request():
                metrics.error_type = "circuit_breaker_open"
                metrics.fallback_used = True
                metrics.duration_ms = (time.time() - start_time) * 1000
                self._record_metrics(metrics)
                raise RedisError("Circuit breaker is OPEN")
            
            # Executa operação com retry
            for attempt in range(self.max_retries):
                try:
                    result = operation()
                    metrics.success = True
                    self.circuit_breaker.record_success()
                    
                    # Para operações GET, verifica se foi cache hit
                    if operation_type == CacheOperation.GET:
                        metrics.cache_hit = result is not None
                    
                    break
                except (ConnectionError, RedisTimeoutError, BusyLoadingError) as e:
                    if attempt == self.max_retries - 1:
                        metrics.error_type = type(e).__name__
                        self.circuit_breaker.record_failure()
                        raise
                    
                    wait_time = self.retry_delay * (2 ** attempt)
                    logger.warning(f"Redis {operation_type.value} retry {attempt + 1} for key {key}: {e}")
                    time.sleep(wait_time)
            
        except RedisError as e:
            metrics.error_type = type(e).__name__
            logger.error(f"Redis {operation_type.value} failed for key {key}: {e}")
            raise
        finally:
            metrics.duration_ms = (time.time() - start_time) * 1000
            self._record_metrics(metrics)
        
        return result
    
    def get(self, key: str, fallback_value: Any = None) -> Any:
        """Obtém valor do cache com fallback"""
        try:
            def operation():
                value = self.client.get(key)
                return json.loads(value) if value else None
            
            return self._execute_with_resilience(operation, CacheOperation.GET, key)
        except RedisError:
            if fallback_value is not None:
                logger.info(f"Using fallback value for key: {key}")
                return fallback_value
            raise
    
    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """Armazena valor no cache com TTL"""
        def operation():
            serialized = json.dumps(value, default=str)
            return self.client.setex(
                key, 
                ttl or self.default_ttl, 
                serialized
            )
        
        return self._execute_with_resilience(operation, CacheOperation.SET, key)
    
    def delete(self, key: str) -> bool:
        """Remove valor do cache"""
        def operation():
            return self.client.delete(key) > 0
        
        return self._execute_with_resilience(operation, CacheOperation.DELETE, key)
    
    def exists(self, key: str) -> bool:
        """Verifica se chave existe no cache"""
        def operation():
            return self.client.exists(key) > 0
        
        return self._execute_with_resilience(operation, CacheOperation.EXISTS, key)
    
    def invalidate_pattern(self, pattern: str) -> int:
        """Invalida todas as chaves que correspondem ao pattern"""
        start_time = time.time()
        metrics = CacheMetrics(
            operation=CacheOperation.INVALIDATE,
            key=pattern,
            duration_ms=0,
            success=False
        )
        
        try:
            keys = self.client.keys(pattern)
            if keys:
                deleted = self.client.delete(*keys)
                metrics.success = True
                logger.info(f"Invalidated {deleted} keys matching pattern: {pattern}")
                return deleted
            return 0
        except RedisError as e:
            metrics.error_type = type(e).__name__
            logger.error(f"Failed to invalidate pattern {pattern}: {e}")
            raise
        finally:
            metrics.duration_ms = (time.time() - start_time) * 1000
            self._record_metrics(metrics)
    
    def get_or_set(self, key: str, factory: Callable[[], Any], 
                  ttl: Optional[int] = None) -> Any:
        """Obtém do cache ou executa factory e armazena resultado"""
        cached = self.get(key)
        if cached is not None:
            return cached
        
        value = factory()
        self.set(key, value, ttl)
        return value
    
    def increment(self, key: str, amount: int = 1, ttl: Optional[int] = None) -> int:
        """Incrementa valor numérico (atomicamente)"""
        def operation():
            result = self.client.incrby(key, amount)
            if ttl and self.client.ttl(key) == -1:  # Se não tem TTL
                self.client.expire(key, ttl)
            return result
        
        return self._execute_with_resilience(operation, CacheOperation.SET, key)
    
    def get_many(self, keys: List[str]) -> Dict[str, Any]:
        """Obtém múltiplas chaves de uma vez"""
        results = {}
        for key in keys:
            try:
                value = self.get(key)
                if value is not None:
                    results[key] = value
            except RedisError:
                continue
        return results
    
    def set_many(self, items: Dict[str, Any], ttl: Optional[int] = None) -> bool:
        """Armazena múltiplos itens de uma vez"""
        pipeline = self.client.pipeline()
        for key, value in items.items():
            serialized = json.dumps(value, default=str)
            pipeline.setex(key, ttl or self.default_ttl, serialized)
        
        def operation():
            return pipeline.execute()
        
        return all(self._execute_with_resilience(operation, CacheOperation.SET, "batch_set"))
    
    def clear(self) -> bool:
        """Limpa todo o cache (use com cuidado!)"""
        logger.warning("Clearing entire Redis cache")
        return self.client.flushdb()
    
    def health_check(self) -> Dict[str, Any]:
        """Verifica saúde do serviço de cache"""
        start_time = time.time()
        health = {
            "status": "healthy",
            "service": "redis_cache",
            "timestamp": datetime.utcnow().isoformat(),
            "latency_ms": 0,
            "circuit_breaker_state": self.circuit_breaker.state,
            "connection_info": {
                "host": self.config['host'],
                "port": self.config['port'],
                "db": self.config['db']
            }
        }
        
        try:
            # Testa conexão e operação básica
            self.client.ping()
            test_key = "health_check"
            self.client.setex(test_key, 10, "test")
            self.client.get(test_key)
            self.client.delete(test_key)
            
            health["latency_ms"] = round((time.time() - start_time) * 1000, 2)
            
        except RedisError as e:
            health.update({
                "status": "unhealthy",
                "error": str(e),
                "error_type": type(e).__name__
            })
            logger.error(f"Redis health check failed: {e}")
        
        return health

def cached(ttl: int = 3600, key_prefix: str = "func"):
    """Decorator para cache automático de resultados de função"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Gera chave única baseada na função e argumentos
            key_parts = [key_prefix, func.__module__, func.__name__]
            
            # Adiciona hash dos argumentos
            if args or kwargs:
                arg_str = json.dumps({
                    'args': args,
                    'kwargs': kwargs
                }, sort_keys=True, default=str)
                arg_hash = hashlib.md5(arg_str.encode()).hexdigest()[:8]
                key_parts.append(arg_hash)
            
            cache_key = ":".join(key_parts)
            
            # Tenta obter do cache
            cache_service = RedisCacheService()  # Ou injetar via dependency
            cached_result = cache_service.get(cache_key)
            
            if cached_result is not None:
                logger.debug(f"Cache hit for {func.__name__}")
                return cached_result
            
            # Executa função e armazena no cache
            result = func(*args, **kwargs)
            cache_service.set(cache_key, result, ttl)
            
            return result
        
        return wrapper
    return decorator

def invalidates_cache(pattern: str):
    """Decorator para invalidar cache após execução de função"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            result = func(*args, **kwargs)
            
            try:
                cache_service = RedisCacheService()
                invalidated = cache_service.invalidate_pattern(pattern)

                logger.error(f"Failed to invalidate cache for pattern {pattern}: {e}")
                # Não propaga erro - cache invalidation não deve quebrar a operação principal
            
            return result
        return wrapper
    return decorator

import React, { HTMLAttributes, forwardRef } from 'react';
import './Card.css';

export interface CardProps extends HTMLAttributes<HTMLDivElement> {
  variant?: 'default' | 'outline' | 'filled';
  padding?: 'none' | 'sm' | 'md' | 'lg';
  hoverable?: boolean;
  clickable?: boolean;
}

const Card = forwardRef<HTMLDivElement, CardProps>(({
  children,
  variant = 'default',
  padding = 'md',
  hoverable = false,
  clickable = false,
  className = '',
  ...props
}, ref) => {
  const baseClasses = 'card';
  const variantClass = `card-${variant}`;
  const paddingClass = `card-padding-${padding}`;
  const hoverClass = hoverable ? 'card-hoverable' : '';
  const clickableClass = clickable ? 'card-clickable' : '';
  
  const combinedClasses = [
    baseClasses,
    variantClass,
    paddingClass,
    hoverClass,
    clickableClass,
    className
  ].filter(Boolean).join(' ');

  return (
    <div
      ref={ref}
      className={combinedClasses}
      role={clickable ? 'button' : undefined}
      tabIndex={clickable ? 0 : undefined}
      {...props}
    >
      {children}
    </div>
  );
});

Card.displayName = 'Card';

export default Card;

class AnomalyDetector {
    constructor(config = {}) {
        this.config = {
            windowSize: 24, // horas
            sensitivity: 2.5, // desvios padrão
            minDataPoints: 10,
            seasonality: {
                daily: true,
                weekly: true
            },
            ...config
        };

        this.models = new Map();
        this.metricsHistory = new Map();
        this.anomalies = [];
    }

    /**
     * Detecta anomalias em série temporal usando múltiplos métodos
     */
    async detectAnomalies(metricData, metricType) {
        try {
            const anomalies = [];
            const values = metricData.map(d => d.value);
            const timestamps = metricData.map(d => d.timestamp);

            if (values.length < this.config.minDataPoints) {
                logger.warn(`Dados insuficientes para ${metricType}: ${values.length} pontos`);
                return [];
            }

            // 1. Detecção por Z-Score
            const zScoreAnomalies = this.detectByZScore(values, timestamps, metricType);
            anomalies.push(...zScoreAnomalies);

            // 2. Detecção por IQR (Interquartile Range)
            const iqrAnomalies = this.detectByIQR(values, timestamps, metricType);
            anomalies.push(...iqrAnomalies);

            // 3. Detecção por Mudança de Tendência
            const trendAnomalies = this.detectTrendChange(values, timestamps, metricType);
            anomalies.push(...trendAnomalies);

            // 4. Detecção Sazonal
            if (this.config.seasonality.daily || this.config.seasonality.weekly) {
                const seasonalAnomalies = this.detectSeasonalAnomalies(values, timestamps, metricType);
                anomalies.push(...seasonalAnomalies);
            }

            // Consolidar anomalias (remover duplicatas)
            const uniqueAnomalies = this.consolidateAnomalies(anomalies);
            
            // Classificar severidade
            const classifiedAnomalies = uniqueAnomalies.map(anomaly => ({
                ...anomaly,
                severity: this.classifySeverity(anomaly),
                confidence: this.calculateConfidence(anomaly, values)
            }));

            // Armazenar no histórico
            this.storeAnomalies(classifiedAnomalies, metricType);

            return classifiedAnomalies;

        } catch (error) {
            logger.error(`Erro na detecção de anomalias para ${metricType}:`, error);
            return [];
        }
    }

    /**
     * Detecção por Z-Score
     */
    detectByZScore(values, timestamps, metricType) {
        const mean = ss.mean(values);
        const std = ss.standardDeviation(values);
        const anomalies = [];

        values.forEach((value, index) => {
            const zScore = Math.abs((value - mean) / std);
            
            if (zScore > this.config.sensitivity) {
                anomalies.push({
                    timestamp: timestamps[index],
                    value,
                    metricType,
                    method: 'z-score',
                    zScore,
                    threshold: this.config.sensitivity,
                    description: `Valor ${value.toFixed(2)} está ${zScore.toFixed(2)} desvios padrão da média`
                });
            }
        });

        return anomalies;
    }

    /**
     * Detecção por IQR
     */
    detectByIQR(values, timestamps, metricType) {
        const sorted = [...values].sort((a, b) => a - b);
        const q1 = ss.quantile(sorted, 0.25);
        const q3 = ss.quantile(sorted, 0.75);
        const iqr = q3 - q1;
        const lowerBound = q1 - (1.5 * iqr);
        const upperBound = q3 + (1.5 * iqr);

    rotateOldLogs() {
        const cutoffDate = new Date();
        cutoffDate.setDate(cutoffDate.getDate() - COMPLIANCE_CONFIG.AUDIT_LOG_RETENTION_DAYS);
        
        const logDir = 'logs/audit';
        fs.readdir(logDir, (err, files) => {
            if (err) return;
            
            files.forEach(file => {
                const filePath = path.join(logDir, file);
                const stats = fs.statSync(filePath);
                
                if (stats.mtime < cutoffDate && file !== 'audit.log' && file !== 'compliance.log') {
                    fs.unlinkSync(filePath);
                    this.logDeletion(filePath, 'log_rotation');
                }
            });
        });
    }

    class DataFlowMapper {
        constructor() {
            this.dataFlows = new Map();
            this.dataProcessors = new Map();
            this.loadDataFlowRegistry();
        }
        
        /**
         * Registra um novo fluxo de dados
         */
        registerDataFlow(flowId, description, dataCategories, purpose, legalBasis, processors = []) {
            const flow = {
                id: flowId,
                description,
                dataCategories,
                purpose,
                legalBasis,
                processors,
                createdAt: new Date().toISOString(),
                lastUpdated: new Date().toISOString(),
                dpiaRequired: this.requiresDPIA(dataCategories),
                retentionPeriod: this.calculateRetentionPeriod(dataCategories, purpose)
            };
            
            this.dataFlows.set(flowId, flow);
            this.saveDataFlowRegistry();
            
            auditLogger.logComplianceEvent('DATA_FLOW_REGISTERED', flow);
            
            return flow;
        }
        
        /**
         * Mapeia processadores de dados (LGPD Art. 5, VII)
         */
        registerDataProcessor(processorId, name, type, location, dpaSigned, dataCategories) {
            const processor = {
                id: processorId,
                name,
                type,
                location,
                dpaSigned,
                dpaSignedDate: dpaSigned ? new Date().toISOString() : null,
                dataCategories,
                securityMeasures: this.getRequiredSecurityMeasures(type, dataCategories),
                lastAudit: null
            };
            
            this.dataProcessors.set(processorId, processor);
            this.saveProcessorRegistry();
            
            return processor;
        }
        
        /**
         * Gera relatório de mapeamento para DPO
         */
        generateDataMappingReport() {
            const report = {
                generatedAt: new Date().toISOString(),
                totalDataFlows: this.dataFlows.size,
                totalProcessors: this.dataProcessors.size,
                dataFlows: Array.from(this.dataFlows.values()),
                processors: Array.from(this.dataProcessors.values()),
                complianceStatus: this.assessComplianceStatus()
            };
            
            // Salva relatório
            const reportPath = `reports/data-mapping-${new Date().toISOString().split('T')[0]}.json`;
            fs.writeFileSync(reportPath, JSON.stringify(report, null, 2));
            
            return report;
        }
        
        /**
         * Verifica se fluxo requer DPIA (LGPD Art. 38)
         */
        requiresDPIA(dataCategories) {
            const sensitiveCategories = COMPLIANCE_CONFIG.DATA_CATEGORIES.SENSITIVE;
            const hasSensitiveData = dataCategories.some(cat => 
                sensitiveCategories.includes(cat)
            );
            
            const largeScale = dataCategories.length > 3;
            const systematicMonitoring = dataCategories.includes('behavioral');
            
            return hasSensitiveData || largeScale || systematicMonitoring;
        }
        
        /**
         * Calcula período de retenção baseado na finalidade
         */
        calculateRetentionPeriod(dataCategories, purpose) {
            const retentionRules = {
                FINANCIAL: 365 * 5, // 5 anos para dados financeiros
                HEALTH: 365 * 20,   // 20 anos para dados de saúde
                CONTRACT: 365 * 10, // 10 anos após término do contrato
                CONSENT: 365 * 2,   // 2 anos após revogação do consentimento
                DEFAULT: 365 * 1    // 1 ano para outros casos
            };
            
            if (dataCategories.some(cat => cat.includes('financial'))) {
                return retentionRules.FINANCIAL;
            }
            
            if (dataCategories.some(cat => cat.includes('health'))) {
                return retentionRules.HEALTH;
            }
            
            return retentionRules.DEFAULT;
        }
        
        loadDataFlowRegistry() {
            try {
                if (fs.existsSync('data/compliance/data-flows.json')) {
                    const data = JSON.parse(fs.readFileSync('data/compliance/data-flows.json'));
                    this.dataFlows = new Map(data.flows);
                    this.dataProcessors = new Map(data.processors);
                }
            } catch (error) {
                console.warn('Could not load data flow registry, starting fresh');
            }
        }
        
        saveDataFlowRegistry() {
            const registryDir = 'data/compliance';
            if (!fs.existsSync(registryDir)) {
                fs.mkdirSync(registryDir, { recursive: true });
            }
            
            const data = {
                flows: Array.from(this.dataFlows.entries()),
                processors: Array.from(this.dataProcessors.entries()),
                lastUpdated: new Date().toISOString()
            };
            
            fs.writeFileSync(
                path.join(registryDir, 'data-flows.json'),
                JSON.stringify(data, null, 2)
            );
        }
        
        saveProcessorRegistry() {
            const registryDir = 'data/compliance';
            if (!fs.existsSync(registryDir)) {
                fs.mkdirSync(registryDir, { recursive: true });
            }
            
            fs.writeFileSync(
                path.join(registryDir, 'processors.json'),
                JSON.stringify(Array.from(this.dataProcessors.entries()), null, 2)
            );
        }
        
        assessComplianceStatus() {
            const issues = [];
            
            // Verifica DPAs assinados
            this.dataProcessors.forEach((processor, id) => {
                if (!processor.dpaSigned) {
                    issues.push(`Processor ${processor.name} missing DPA`);
                }
                
                if (processor.location === 'non-eu' && processor.dataCategories.includes('sensitive')) {
                    issues.push(`Sensitive data transfer to non-EU processor ${processor.name}`);
                }
            });
            
            // Verifica DPIAs necessárias
            this.dataFlows.forEach((flow, id) => {
                if (flow.dpiaRequired && !flow.dpiaCompleted) {
                    issues.push(`Data flow ${flow.id} requires DPIA`);
                }
                
                if (!flow.retentionPeriod) {
                    issues.push(`Data flow ${flow.id} missing retention period`);
                }
            });
            
            return {
                status: issues.length === 0 ? 'COMPLIANT' : 'NEEDS_ATTENTION',
                issues,
                lastAssessment: new Date().toISOString()
            };
        }
    }

    class DataRetentionManager {
        constructor() {
            this.retentionRules = new Map();
            this.loadRetentionRules();
        }
        
        /**
         * Aplica política de retenção automaticamente
         */
        async applyRetentionPolicy() {
            const now = new Date();
            const deletionQueue = [];
            
            // Itera sobre todos os fluxos de dados
            dataFlowMapper.dataFlows.forEach((flow, flowId) => {
                if (flow.retentionPeriod) {
                    const cutoffDate = new Date(now);
                    cutoffDate.setDate(cutoffDate.getDate() - flow.retentionPeriod);
                    
                    // Aqui integraria com o banco de dados real
                    // Por enquanto, simula a lógica
                    deletionQueue.push({
                        flowId,
                        cutoffDate: cutoffDate.toISOString(),
                        dataCategories: flow.dataCategories
                    });
                }
            });
            
            // Processa exclusões
            for (const item of deletionQueue) {
                await this.processDeletions(item);
            }
            
            return {
                processed: deletionQueue.length,
                timestamp: now.toISOString()
            };
        }
        
        async processDeletions(retentionRule) {
            // Implementação real integraria com o banco de dados
            // Esta é uma implementação de exemplo
            
            auditLogger.logDataDeletion(
                'SYSTEM',
                retentionRule.dataCategories.join(','),
                'BATCH_DELETION',
                'RETENTION_POLICY',
                retentionRule.retentionPeriod
            );
            
            // Simula exclusão
            await new Promise(resolve => setTimeout(resolve, 100));
            
            return {
                success: true,
                rule: retentionRule.flowId,
                deletedCount: 'simulated'
            };
        }
        
        /**
         * Processa solicitação de exclusão do titular (LGPD Art. 18, V)
         */
        async processUserDeletionRequest(userId, dataTypes, requestId) {
            const auditTrail = [];
            
            for (const dataType of dataTypes) {
                const startTime = Date.now();
                
                // Exclui dados do tipo especificado
                const result = await this.deleteUserData(userId, dataType);
                
                auditTrail.push({
                    dataType,
                    result,
                    duration: Date.now() - startTime
                });
                
                // Log da exclusão
                auditLogger.logDataDeletion(
                    userId,
                    dataType,
                    requestId,
                    'USER_REQUEST',
                    'N/A'
                );
            }
            
            // Notifica conclusão
            await this.notifyDeletionCompletion(userId, requestId, auditTrail);
            
            return {
                requestId,
                userId,
                auditTrail,
                completedAt: new Date().toISOString(),
                complianceVerified: true
            };
        }
        
        async deleteUserData(userId, dataType) {
            // Implementação real de exclusão
            // Esta é uma implementação de exemplo
            
            return {
                success: true,
                message: `Data type ${dataType} deleted for user ${userId}`,
                timestamp: new Date().toISOString()
            };
        }
        
        async notifyDeletionCompletion(userId, requestId, auditTrail) {

        }

            }

    hashDataId(dataId) {
        // Hash para anonimização em logs
        return crypto
            .createHash('sha256')
            .update(dataId + process.env.AUDIT_SALT || 'compliance-salt')
            .digest('hex')
            .substring(0, 32);
    }
    
    sanitizeChanges(changes) {
        // Remove dados sensíveis dos logs
        const sensitiveFields = ['password', 'token', 'secret', 'key', 'credit_card'];
        
        if (typeof changes === 'object') {
            const sanitized = { ...changes };
            sensitiveFields.forEach(field => {
                if (sanitized[field]) {
                    sanitized[field] = '[REDACTED]';
                }
            });
            return sanitized;
        }
        
        return changes;
    }
    
    assessOperationRisk(operation, target) {
        const highRiskOps = ['DELETE', 'GRANT_ADMIN', 'CHANGE_PERMISSIONS', 'EXPORT_DATA'];
        const sensitiveTargets = ['user_data', 'payment_info', 'audit_logs'];
        
        if (highRiskOps.includes(operation) || sensitiveTargets.includes(target)) {
            return 'HIGH';
        }
        
        return 'MEDIUM';
    }
