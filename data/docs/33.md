## 2. ADR 001: SERVICE BOUNDARIES & COMMUNICATION PATTERNS

```markdown
# ADR-001: Service Boundaries and Communication Patterns

Date: 2024-01-15
Status: Accepted
Authors: Tech Lead + Architecture Committee

## CONTEXT
DIYAPP V12 precisa escalar para 100+ usuários concorrentes com 99.9% uptime.
Sistema monolítico atual não permite deploy independente, escalabilidade granular
ou substituição de componentes sem downtime.

## DECISION
Adotar arquitetura de microsserviços com os seguintes princípios:
1. Cada serviço possui seu próprio banco de dados (Database per Service)
2. Comunicação síncrona via HTTP/REST para operações críticas
3. Comunicação assíncrona via message broker (RabbitMQ) para operações background
4. API Gateway como ponto único de entrada
5. Service Mesh (Linkerd) para comunicação service-to-service segura

## OPTIONS CONSIDERED
- Option A: Monolito modularizado - Prós: Simples, Contras: Escalabilidade limitada
- Option B: Microsserviços completos - Prós: Escalabilidade, Contras: Complexidade
- Option C: Arquitetura híbrida - Prós: Balanceado, Contras: Gerenciamento misto

## CONSEQUENCES
Positivas:
- Deploy independente por serviço
- Escalabilidade granular
- Stack tecnológica heterogênea
- Isolamento de falhas

Negativas:
- Complexidade operacional aumentada
- Latência em comunicações entre serviços
- Gerenciamento distribuído de transações

Risks:
- Distributed tracing necessário
- Service discovery obrigatório
- Monitoramento distribuído crítico
```

```typescript
export interface APIResponse<T> {
  success: boolean;
  data?: T;
  error?: {
    code: string;
    message: string;
    details?: Record<string, unknown>;
  };

    timestamp: string;
    requestId: string;
    version: string;
  };

export interface Task {
  id: string;
  title: string;
  description: string;
  status: 'pending' | 'in_progress' | 'completed' | 'failed';
  priority: 'low' | 'medium' | 'high' | 'critical';
  assignedTo?: string;
  createdAt: string;
  updatedAt: string;
  metadata?: Record<string, unknown>;
}

// services/task-service/src/controllers/task.controller.ts
import { Controller, Get, Post, Put, Delete, Param, Body, Query } from '@nestjs/common';
import { ApiTags, ApiOperation, ApiResponse } from '@nestjs/swagger';
import { TaskService } from '../services/task.service';
import { CreateTaskDto, UpdateTaskDto, TaskQueryDto } from '../dto/task.dto';
import { APIResponse, Task } from '../interfaces/api.contract';

@Controller('v1/tasks')
@ApiTags('tasks')
export class TaskController {
  constructor(private readonly taskService: TaskService) {}

  @Post()
  @ApiOperation({ summary: 'Create a new task' })
  @ApiResponse({ status: 201, description: 'Task created successfully' })
  @ApiResponse({ status: 400, description: 'Invalid input' })
  async createTask(@Body() createTaskDto: CreateTaskDto): Promise<APIResponse<Task>> {
    const task = await this.taskService.create(createTaskDto);
    return {
      success: true,
      data: task,
      metadata: {
        timestamp: new Date().toISOString(),
        requestId: this.getRequestId(),
        version: '1.0.0'
      }
    };
  }

  @Get(':id')
  @ApiOperation({ summary: 'Get task by ID' })
  @ApiResponse({ status: 200, description: 'Task found' })
  @ApiResponse({ status: 404, description: 'Task not found' })
  async getTask(@Param('id') id: string): Promise<APIResponse<Task>> {
    const task = await this.taskService.findById(id);
    return {
      success: true,
      data: task,
      metadata: {
        timestamp: new Date().toISOString(),
        requestId: this.getRequestId(),
        version: '1.0.0'
      }
    };
  }

  private getRequestId(): string {
    // Implementação de tracing distribuído
    return require('crypto').randomBytes(16).toString('hex');
  }
}
```

## 3. Configuração do Banco de Dados (src/config/database.js)

```javascript
const { Pool } = require('pg');
const config = require('./index');
const logger = require('../utils/logger');

class Database {
  constructor() {
    this.pool = null;
    this.isConnected = false;
    this.connectionAttempts = 0;
    this.maxConnectionAttempts = 5;
  }

  async connect() {
    try {
      this.pool = new Pool({
        host: config.database.host,
        port: config.database.port,
        database: config.database.name,
        user: config.database.user,
        password: config.database.password,
        max: config.database.maxConnections,
        idleTimeoutMillis: config.timeouts.database,
        connectionTimeoutMillis: config.database.connectionTimeout,
        
        // Configurações de resiliência
        maxUses: 7500, // Reconectar após 7500 queries
        allowExitOnIdle: false,
      });

      // Testar conexão
      const client = await this.pool.connect();
      await client.query('SELECT NOW()');
      client.release();
      
      this.isConnected = true;
      this.connectionAttempts = 0;
      
      logger.info('Database connected successfully');
      
      // Configurar listeners de erro
      this.pool.on('error', (err) => {
        logger.error('Unexpected database pool error', { error: err.message });
        this.isConnected = false;
      });
      
      return this.pool;
    } catch (error) {
      this.connectionAttempts++;
      logger.error('Database connection failed', {
        attempt: this.connectionAttempts,
        error: error.message,
      });
      
      if (this.connectionAttempts >= this.maxConnectionAttempts) {
        throw new Error(`Failed to connect to database after ${this.maxConnectionAttempts} attempts`);
      }
      
      // Retry com backoff exponencial
      const delay = Math.min(1000 * Math.pow(2, this.connectionAttempts), 30000);
      await new Promise(resolve => setTimeout(resolve, delay));
      return this.connect();
    }
  }

  async query(text, params) {
    if (!this.isConnected) {
      await this.connect();
    }

    const start = Date.now();
    try {
      const result = await this.pool.query(text, params);
      const duration = Date.now() - start;
      
      logger.debug('Database query executed', {
        query: text,
        duration,
        rowCount: result.rowCount,
      });
      
      return result;
    } catch (error) {
      const duration = Date.now() - start;
      logger.error('Database query failed', {
        query: text,
        params: params ? params.map(p => typeof p === 'string' ? p.substring(0, 100) : p) : [],
        duration,
        error: error.message,
      });
      
      // Reclassificar erros para tratamento apropriado
      if (error.code === '23505') { // Unique violation
        const customError = new Error('Duplicate entry');
        customError.code = 'DUPLICATE_ENTRY';
        customError.status = 409;
        throw customError;
      } else if (error.code === '23503') { // Foreign key violation
        const customError = new Error('Referenced entity not found');
        customError.code = 'FOREIGN_KEY_VIOLATION';
        customError.status = 404;
        throw customError;
      }
      
      throw error;
    }
  }

  async transaction(callback) {
    const client = await this.pool.connect();
    
    try {
      await client.query('BEGIN');
      const result = await callback(client);
      await client.query('COMMIT');
      return result;
    } catch (error) {
      await client.query('ROLLBACK');
      throw error;
    } finally {
      client.release();
    }
  }

  async disconnect() {
    if (this.pool) {
      await this.pool.end();
      this.isConnected = false;
      logger.info('Database disconnected');
    }
  }
}

module.exports = new Database();
```

## 5. Middleware de Autenticação (src/middleware/auth.js)

```javascript
const jwt = require('jsonwebtoken');
const config = require('../config');
const logger = require('../utils/logger');

class AuthMiddleware {
  constructor() {
    this.blacklist = new Set(); // Em produção, usar Redis
  }

  // Middleware de autenticação JWT
  authenticate(req, res, next) {
    const start = Date.now();
    const correlationId = req.headers['x-correlation-id'] || req.id || 'unknown';
    
    try {
      const authHeader = req.headers.authorization;
      if (!authHeader || !authHeader.startsWith('Bearer ')) {
        logger.warn('Authentication failed - no token', { correlationId });
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'Authentication token required',
          correlationId,
        });
      }

      const token = authHeader.substring(7);
      
      // Verificar se token está na blacklist
      if (this.blacklist.has(token)) {
        logger.warn('Authentication failed - token revoked', { correlationId });
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'Token has been revoked',
          correlationId,
        });
      }

      // Verificar token
      const decoded = jwt.verify(token, config.jwt.secret);
      
      // Adicionar usuário ao request
      req.user = {
        id: decoded.userId,
        email: decoded.email,
        role: decoded.role,
        permissions: decoded.permissions || [],
      };
      
      req.token = token;
      
      const duration = Date.now() - start;
      logger.debug('Authentication successful', {
        correlationId,
        userId: req.user.id,
        duration,
      });
      
      next();
    } catch (error) {
      const duration = Date.now() - start;
      
      if (error.name === 'TokenExpiredError') {
        logger.warn('Authentication failed - token expired', { correlationId, duration });
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'Token has expired',
          correlationId,
        });
      } else if (error.name === 'JsonWebTokenError') {
        logger.warn('Authentication failed - invalid token', { correlationId, duration, error: error.message });
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'Invalid token',
          correlationId,
        });
      }
      
      logger.error('Authentication error', { correlationId, duration, error: error.message });
      return res.status(500).json({
        error: 'Internal Server Error',
        message: 'Authentication failed',
        correlationId,
      });
    }
  }

  // Middleware de autorização baseada em roles
  authorize(requiredRoles = [], requiredPermissions = []) {
    return (req, res, next) => {
      const correlationId = req.headers['x-correlation-id'] || req.id || 'unknown';
      
      if (!req.user) {
        logger.warn('Authorization failed - no user', { correlationId });
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'Authentication required',
          correlationId,
        });
      }

      // Verificar roles
      if (requiredRoles.length > 0 && !requiredRoles.includes(req.user.role)) {
        logger.warn('Authorization failed - insufficient role', {
          correlationId,
          userId: req.user.id,
          userRole: req.user.role,
          requiredRoles,
        });
        return res.status(403).json({
          error: 'Forbidden',
          message: 'Insufficient role permissions',
          correlationId,
        });
      }

      // Verificar permissions específicas
      if (requiredPermissions.length > 0) {
        const hasAllPermissions = requiredPermissions.every(permission =>
          req.user.permissions.includes(permission)
        );
        
        if (!hasAllPermissions) {
          logger.warn('Authorization failed - insufficient permissions', {
            correlationId,
            userId: req.user.id,
            userPermissions: req.user.permissions,
            requiredPermissions,
          });
          return res.status(403).json({
            error: 'Forbidden',
            message: 'Insufficient permissions',
            correlationId,
          });
        }
      }

      next();
    };
  }

  // Revogar token (logout)
  async revokeToken(token) {
    try {
      const decoded = jwt.decode(token);
      if (decoded && decoded.exp) {
        const ttl = decoded.exp - Math.floor(Date.now() / 1000);
        if (ttl > 0) {
          this.blacklist.add(token);
          // Em produção: this.redis.setex(`blacklist:${token}`, ttl, '1')
        }
      }
      return true;
    } catch (error) {
      logger.error('Failed to revoke token', { error: error.message });
      return false;
    }
  }

  // Gerar tokens
  generateTokens(user) {
    const accessToken = jwt.sign(
      {
        userId: user.id,
        email: user.email,
        role: user.role,
        permissions: user.permissions || [],
      },
      config.jwt.secret,
      { expiresIn: config.jwt.expiresIn }
    );

    const refreshToken = jwt.sign(
      { userId: user.id },
      config.jwt.secret,
      { expiresIn: config.jwt.refreshExpiresIn }
    );

    return { accessToken, refreshToken };
  }
}

module.exports = new AuthMiddleware();
```

```typescript
import storage from 'redux-persist/lib/storage';
import authReducer from './slices/authSlice';

import uiReducer from './slices/uiSlice';

const persistConfig = {
  key: 'root',
  storage,
  whitelist: ['auth', 'user'],
  version: 1,
};

const rootReducer = combineReducers({
  auth: authReducer,
  user: userReducer,
  ui: uiReducer,
});

const persistedReducer = persistReducer(persistConfig, rootReducer);

export const store = configureStore({
  reducer: persistedReducer,
  middleware: (getDefaultMiddleware) =>
    getDefaultMiddleware({
      serializableCheck: {
        ignoredActions: ['persist/PERSIST', 'persist/REHYDRATE'],
      },
    }),
  devTools: process.env.NODE_ENV !== 'production',
});

export const persistor = persistStore(store);

export type RootState = ReturnType<typeof store.getState>;
```

```typescript
// src/store/slices/authSlice.ts
import { createSlice, createAsyncThunk, PayloadAction } from '@reduxjs/toolkit';
import { authApi } from '../../api/authApi';
import { AuthState, LoginCredentials, User } from '../../types/auth';

const initialState: AuthState = {
  user: null,
  token: null,
  isLoading: false,
  error: null,
  isAuthenticated: false,
};

export const login = createAsyncThunk(
  'auth/login',
  async (credentials: LoginCredentials, { rejectWithValue }) => {
    try {
      const response = await authApi.login(credentials);
      return response.data;
    } catch (error) {
      return rejectWithValue(error.response?.data?.message || 'Login failed');
    }
  }
);

export const logout = createAsyncThunk(
  'auth/logout',
  async (_, { rejectWithValue }) => {
    try {
      await authApi.logout();
      return null;
    } catch (error) {
      return rejectWithValue(error.response?.data?.message || 'Logout failed');
    }
  }
);

export const checkAuth = createAsyncThunk(
  'auth/checkAuth',
  async (_, { rejectWithValue }) => {
    try {
      const response = await authApi.checkAuth();
      return response.data;
    } catch (error) {
      return rejectWithValue(error.response?.data?.message || 'Auth check failed');
    }
  }
);

const authSlice = createSlice({
  name: 'auth',
  initialState,
  reducers: {
    clearError: (state) => {
      state.error = null;
    },
    setToken: (state, action: PayloadAction<string>) => {
      state.token = action.payload;
      state.isAuthenticated = true;
    },
  },
  extraReducers: (builder) => {
    builder
      // Login
      .addCase(login.pending, (state) => {
        state.isLoading = true;
        state.error = null;
      })
      .addCase(login.fulfilled, (state, action) => {
        state.isLoading = false;
        state.user = action.payload.user;
        state.token = action.payload.token;
        state.isAuthenticated = true;
        state.error = null;
      })
      .addCase(login.rejected, (state, action) => {
        state.isLoading = false;
        state.error = action.payload as string;
        state.isAuthenticated = false;
      })
      // Logout
      .addCase(logout.pending, (state) => {
        state.isLoading = true;
      })
      .addCase(logout.fulfilled, (state) => {
        state.isLoading = false;
        state.user = null;
        state.token = null;
        state.isAuthenticated = false;
        state.error = null;
      })
      .addCase(logout.rejected, (state, action) => {
        state.isLoading = false;
        state.error = action.payload as string;
      })
      // Check Auth
      .addCase(checkAuth.pending, (state) => {
        state.isLoading = true;
      })
      .addCase(checkAuth.fulfilled, (state, action) => {
        state.isLoading = false;
        state.user = action.payload.user;
        state.isAuthenticated = true;
        state.error = null;
      })
      .addCase(checkAuth.rejected, (state) => {
        state.isLoading = false;
        state.user = null;
        state.token = null;
        state.isAuthenticated = false;
      });
  },
});

export const { clearError, setToken } = authSlice.actions;
export default authSlice.reducer;
```

```typescript
import { createSlice, createAsyncThunk, PayloadAction } from '@reduxjs/toolkit';
import { userApi } from '../../api/userApi';
import { User, UserState } from '../../types/user';

const initialState: UserState = {
  profile: null,
  isLoading: false,
  error: null,
  preferences: {
    theme: 'light',
    notifications: true,
    language: 'en',
  },
};

export const fetchUserProfile = createAsyncThunk(
  'user/fetchProfile',
  async (userId: string, { rejectWithValue }) => {
    try {
      const response = await userApi.getProfile(userId);
      return response.data;
    } catch (error) {
      return rejectWithValue(error.response?.data?.message || 'Failed to fetch profile');
    }
  }
);

export const updateUserProfile = createAsyncThunk(
  'user/updateProfile',
  async (updates: Partial<User>, { rejectWithValue }) => {
    try {
      const response = await userApi.updateProfile(updates);
      return response.data;
    } catch (error) {
      return rejectWithValue(error.response?.data?.message || 'Failed to update profile');
    }
  }
);

export const updatePreferences = createAsyncThunk(
  'user/updatePreferences',
  async (preferences: Partial<UserState['preferences']>, { rejectWithValue }) => {
    try {
      const response = await userApi.updatePreferences(preferences);
      return response.data;
    } catch (error) {
      return rejectWithValue(error.response?.data?.message || 'Failed to update preferences');
    }
  }
);

const userSlice = createSlice({
  name: 'user',
  initialState,
  reducers: {
    setPreferences: (state, action: PayloadAction<Partial<UserState['preferences']>>) => {
      state.preferences = { ...state.preferences, ...action.payload };
    },
    clearUserError: (state) => {
      state.error = null;
    },
  },
  extraReducers: (builder) => {
    builder
      // Fetch Profile
      .addCase(fetchUserProfile.pending, (state) => {
        state.isLoading = true;
        state.error = null;
      })
      .addCase(fetchUserProfile.fulfilled, (state, action) => {
        state.isLoading = false;
        state.profile = action.payload;
        state.error = null;
      })
      .addCase(fetchUserProfile.rejected, (state, action) => {
        state.isLoading = false;
        state.error = action.payload as string;
      })
      // Update Profile
      .addCase(updateUserProfile.pending, (state) => {
        state.isLoading = true;
        state.error = null;
      })
      .addCase(updateUserProfile.fulfilled, (state, action) => {
        state.isLoading = false;
        state.profile = action.payload;
        state.error = null;
      })
      .addCase(updateUserProfile.rejected, (state, action) => {
        state.isLoading = false;
        state.error = action.payload as string;
      })
      // Update Preferences
      .addCase(updatePreferences.fulfilled, (state, action) => {
        state.preferences = action.payload;
      });
  },
});

export const { setPreferences, clearUserError } = userSlice.actions;
export default userSlice.reducer;
```

## 2. Sistema de Autenticação JWT/OAuth2

```javascript
// src/security/auth/jwt.js
const jwt = require('jsonwebtoken');
const crypto = require('crypto');
const SECURITY_CONFIG = require('../../config/security');

class JWTSecurity {
  constructor() {
    this.secret = SECURITY_CONFIG.keys.jwtSecret;
    this.algorithm = 'HS256';
  }

  /**
   * Generate JWT token with security best practices
   */
  generateToken(payload, options = {}) {
    const defaultPayload = {
      ...payload,
      iat: Math.floor(Date.now() / 1000),
      jti: crypto.randomBytes(16).toString('hex'), // Unique token ID
      iss: 'diyapp-v12', // Issuer
      aud: 'diyapp-users', // Audience
    };

    const tokenOptions = {
      algorithm: this.algorithm,
      expiresIn: options.expiresIn || SECURITY_CONFIG.owasp.accessControl.jwtExpiry,
    };

    return jwt.sign(defaultPayload, this.secret, tokenOptions);
  }

  /**
   * Verify JWT token with comprehensive validation
   */
  verifyToken(token) {
    try {
      const decoded = jwt.verify(token, this.secret, {
        algorithms: [this.algorithm],
        issuer: 'diyapp-v12',
        audience: 'diyapp-users',
        clockTolerance: 30, // 30 seconds tolerance
      });

      // Additional security checks
      if (!decoded.jti) {
        throw new Error('Missing token ID');
      }

      // Check token blacklist (for logout functionality)
      if (this.isTokenBlacklisted(decoded.jti)) {
        throw new Error('Token revoked');
      }

      return {
        valid: true,
        payload: decoded,
      };
    } catch (error) {
      return {
        valid: false,
        error: error.message,
      };
    }
  }

  /**
   * Generate refresh token with longer expiry
   */
  generateRefreshToken(userId) {
    const payload = {
      sub: userId,
      type: 'refresh',
    };

    return this.generateToken(payload, {
      expiresIn: SECURITY_CONFIG.owasp.accessControl.refreshTokenExpiry,
    });
  }

  /**
   * Token blacklist management (for logout)
   */
  blacklistToken(tokenId, expirySeconds = 86400) {
    // In production, use Redis or database
    const blacklist = this.getBlacklist();
    blacklist[tokenId] = Date.now() + (expirySeconds * 1000);
    this.saveBlacklist(blacklist);
  }

  isTokenBlacklisted(tokenId) {
    const blacklist = this.getBlacklist();
    const expiry = blacklist[tokenId];
    
    if (!expiry) return false;
    
    // Clean expired entries
    if (Date.now() > expiry) {
      delete blacklist[tokenId];
      this.saveBlacklist(blacklist);
      return false;
    }
    
    return true;
  }

  getBlacklist() {
    // In-memory storage for demo - use Redis in production
    return global.tokenBlacklist || {};
  }

  saveBlacklist(blacklist) {
    global.tokenBlacklist = blacklist;
  }

  /**
   * Password hashing with bcrypt
   */
  async hashPassword(password) {
    const bcrypt = require('bcrypt');
    const saltRounds = SECURITY_CONFIG.owasp.cryptography.saltRounds;
    return bcrypt.hash(password, saltRounds);
  }

  async verifyPassword(password, hash) {
    const bcrypt = require('bcrypt');
    return bcrypt.compare(password, hash);
  }

  /**
   * Validate password against OWASP recommendations
   */
  validatePassword(password) {
    const policy = SECURITY_CONFIG.owasp.authentication.passwordPolicy;
    const errors = [];

    if (password.length < policy.minLength) {
      errors.push(`Password must be at least ${policy.minLength} characters`);
    }

    if (policy.requireUppercase && !/[A-Z]/.test(password)) {
      errors.push('Password must contain at least one uppercase letter');
    }

    if (policy.requireLowercase && !/[a-z]/.test(password)) {
      errors.push('Password must contain at least one lowercase letter');
    }

    if (policy.requireNumbers && !/\d/.test(password)) {
      errors.push('Password must contain at least one number');
    }

    if (policy.requireSpecialChars && !/[!@#$%^&*(),.?":{}|<>]/.test(password)) {
      errors.push('Password must contain at least one special character');
    }

    if (policy.preventCommon) {
```

```javascript
class OAuth2Security {
  constructor() {
    this.clients = new Map(); // client_id -> client_secret, redirect_uris, etc.
    this.authorizationCodes = new Map(); // code -> { client_id, redirect_uri, user_id, expiry }
    this.accessTokens = new Map(); // token -> { client_id, user_id, scope, expiry }
  }

  /**
   * Register OAuth2 client
   */
  registerClient(clientInfo) {
    const clientId = crypto.randomBytes(16).toString('hex');
    const clientSecret = crypto.randomBytes(32).toString('hex');
    
    const client = {
      client_id: clientId,
      client_secret: clientSecret,
      redirect_uris: clientInfo.redirect_uris || [],
      scope: clientInfo.scope || 'read',
      grant_types: clientInfo.grant_types || ['authorization_code'],
      client_name: clientInfo.client_name,
      client_uri: clientInfo.client_uri,
      logo_uri: clientInfo.logo_uri,
      tos_uri: clientInfo.tos_uri,
      policy_uri: clientInfo.policy_uri,
      created_at: Date.now(),
    };

    this.clients.set(clientId, client);
    
    return {
      client_id: clientId,
      client_secret: clientSecret,
      ...client,
    };
  }

  /**
   * Validate OAuth2 client
   */
  validateClient(clientId, clientSecret, grantType) {
    const client = this.clients.get(clientId);
    
    if (!client) {
      return { valid: false, error: 'invalid_client' };
    }

    if (clientSecret && client.client_secret !== clientSecret) {
      return { valid: false, error: 'invalid_client' };
    }

    if (grantType && !client.grant_types.includes(grantType)) {
      return { valid: false, error: 'unauthorized_client' };
    }

    return { valid: true, client };
  }

  /**
   * Generate authorization code
   */
  generateAuthorizationCode(clientId, redirectUri, userId, scope) {
    const code = crypto.randomBytes(32).toString('hex');
    const expiry = Date.now() + (10 * 60 * 1000); // 10 minutes
    
    this.authorizationCodes.set(code, {
      client_id: clientId,
      redirect_uri: redirectUri,
      user_id: userId,
      scope: scope,
      expiry: expiry,
    });

    return code;
  }

  /**
   * Validate authorization code
   */
  validateAuthorizationCode(code, clientId, redirectUri) {
    const authCode = this.authorizationCodes.get(code);
    
    if (!authCode) {
      return { valid: false, error: 'invalid_grant' };
    }

    if (authCode.expiry < Date.now()) {
      this.authorizationCodes.delete(code);
      return { valid: false, error: 'invalid_grant' };
    }

    if (authCode.client_id !== clientId) {
      return { valid: false, error: 'invalid_grant' };
    }

    if (authCode.redirect_uri !== redirectUri) {
      return { valid: false, error: 'invalid_grant' };
    }

    // Clean up used code
    this.authorizationCodes.delete(code);

    return {
      valid: true,
      user_id: authCode.user_id,
      scope: authCode.scope,
    };
  }

  /**
   * Generate access token
   */
  generateAccessToken(clientId, userId, scope) {
    const token = crypto.randomBytes(48).toString('hex');
    const refreshToken = crypto.randomBytes(48).toString('hex');
    const expiry = Date.now() + (3600 * 1000); // 1 hour
    
    this.accessTokens.set(token, {
      client_id: clientId,
      user_id: userId,
      scope: scope,
      expiry: expiry,
      refresh_token: refreshToken,
    });

    return {
      access_token: token,
      token_type: 'Bearer',
      expires_in: 3600,
      refresh_token: refreshToken,
      scope: scope,
    };
  }

  /**
   * Validate access token
   */
  validateAccessToken(token) {
    const accessToken = this.accessTokens.get(token);
    
    if (!accessToken) {
      return { valid: false, error: 'invalid_token' };
    }

    if (accessToken.expiry < Date.now()) {
      this.accessTokens.delete(token);
      return { valid: false, error: 'invalid_token' };
    }

    return {
      valid: true,
      client_id: accessToken.client_id,
      user_id: accessToken.user_id,
      scope: accessToken.scope,
    };
  }

  /**
   * Refresh access token
   */
  refreshAccessToken(refreshToken) {
    // Find token by refresh token
    for (const [token, data] of this.accessTokens.entries()) {
      if (data.refresh_token === refreshToken) {
        // Remove old token
        this.accessTokens.delete(token);
        
        // Generate new token
        return this.generateAccessToken(
          data.client_id,
          data.user_id,
          data.scope
        );
      }
    }

    return { error: 'invalid_grant' };
  }

  /**
   * Revoke token
   */
  revokeToken(token, tokenTypeHint = 'access_token') {
    if (tokenTypeHint === 'access_token') {
      this.accessTokens.delete(token);
    } else if (tokenTypeHint === 'refresh_token') {
      // Find and remove by refresh token
      for (const [accessToken, data] of this.accessTokens.entries()) {
        if (data.refresh_token === token) {
          this.accessTokens.delete(accessToken);
          break;
        }
      }
    }
    
    return { revoked: true };
  }
}
```

```typescript
    template: string;

      `[data-template="${options.template}"]`

    await this.createButton.click();

  }

    return this.page.locator(`[data-project="${projectName}"]`);
  }

  async getProjectList(): Promise<string[]> {
    const projectElements = await this.page.locator('[data-project]').all();
    const projects: string[] = [];
    
    for (const element of projectElements) {
      const name = await element.getAttribute('data-project');
      if (name) projects.push(name);
    }
    
    return projects;
  }

  async getVisibleWidgets(): Promise<string[]> {
    const widgets = await this.page.locator('[data-widget]').all();
    const widgetIds: string[] = [];
    
    for (const widget of widgets) {
      const id = await widget.getAttribute('data-widget');
      if (id && await widget.isVisible()) {
        widgetIds.push(id);
      }
    }

    return boundingBox?.width || 0;
  }

    await this.page.goto('/settings');
  }

  }
```
