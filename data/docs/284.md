# DIYAPP Evolution - V12 Core - Arquitetura Técnica

## ADR-001: Arquitetura de Microsserviços Baseada em Eventos

**Data:** 2024-03-15
**Status:** Aceita
**Autores:** Tech Lead V12 Core

### CONTEXTO:
A V11 atual sofre com acoplamento forte entre módulos, deploy monolítico e dificuldade de escalar componentes individualmente. Necessitamos de uma arquitetura que permita:
1. Evolução independente dos serviços
2. Tolerância a falhas parcial (um serviço down não derruba todo o sistema)
3. Escalabilidade horizontal seletiva
4. Manutenção 100% autônoma por squads especializadas

### DECISÃO:
Adotar arquitetura de microsserviços com comunicação assíncrona via eventos usando Apache Kafka como message broker. Cada serviço será independente com seu próprio banco de dados e ciclo de vida.

### OPÇÕES CONSIDERADAS:
- **Opção A: Monólito modularizado** — Prós: Simplicidade inicial, deploy único. Contras: Acoplamento, escalabilidade limitada, deploy all-or-nothing.
- **Opção B: Microsserviços REST síncronos** — Prós: Separação clara, deploy independente. Contras: Acoplamento temporal, cascata de falhas, latência acumulada.
- **Opção C: Microsserviços baseados em eventos (escolhida)** — Prós: Desacoplamento completo, resiliência, escalabilidade individual. Contras: Complexidade operacional, eventual consistency.

### CONSEQUÊNCIAS:
**Positivas:**
- Deploy independente por serviço
- Escalabilidade granular
- Tolerância a falhas parcial
- Evolução tecnológica independente

**Negativas:**
- Maior complexidade operacional
- Necessidade de monitoramento distribuído
- Eventual consistency em alguns casos
- Overhead de infraestrutura

**Riscos:**
1. Latência inicial maior devido a comunicação assíncrona
2. Dificuldade de debug em fluxos distribuídos
3. Mitigação: Implementar tracing distribuído (OpenTelemetry) e dead letter queues

**REVISÃO:** 2024-06-15

```typescript
// services/shared/observability/metrics.ts
import prometheus from 'prom-client';
import { Histogram, Counter, Gauge, Registry } from 'prom-client';

export class MetricsCollector {
  private static instance: MetricsCollector;
  private registry: Registry;
  
  // HTTP Metrics
  public httpRequestDuration: Histogram<string>;
  public httpRequestTotal: Counter<string>;
  public httpRequestErrors: Counter<string>;
  
  // Business Metrics
  public orderProcessingDuration: Histogram<string>;
  public userRegistrationTotal: Counter<string>;
  public paymentSuccessRate: Gauge<string>;
  
  // System Metrics
  public memoryUsage: Gauge<string>;
  public cpuUsage: Gauge<string>;
  public eventProcessingLag: Gauge<string>;
  
  private constructor() {
    this.registry = new Registry();
    prometheus.collectDefaultMetrics({ register: this.registry });
    
    this.initializeMetrics();
  }
  
  public static getInstance(): MetricsCollector {
    if (!MetricsCollector.instance) {
      MetricsCollector.instance = new MetricsCollector();
    }
    return MetricsCollector.instance;
  }
  
  private initializeMetrics(): void {
    // HTTP Request Duration Histogram (P95 target: < 200ms)
    this.httpRequestDuration = new Histogram({
      name: 'http_request_duration_seconds',
      help: 'Duration of HTTP requests in seconds',
      labelNames: ['method', 'route', 'status_code'],
      buckets: [0.005, 0.01, 0.025, 0.05, 0.1, 0.2, 0.5, 1, 2, 5],
      registers: [this.registry]
    });
    
    // HTTP Request Counter
    this.httpRequestTotal = new Counter({
      name: 'http_requests_total',
      help: 'Total number of HTTP requests',
      labelNames: ['method', 'route', 'status_code'],
      registers: [this.registry]
    });
    
    // HTTP Error Counter
    this.httpRequestErrors = new Counter({
      name: 'http_request_errors_total',
      help: 'Total number of HTTP request errors',
      labelNames: ['method', 'route', 'error_type'],
      registers: [this.registry]
    });
    
    // Order Processing Duration
    this.orderProcessingDuration = new Histogram({
      name: 'order_processing_duration_seconds',
      help: 'Duration of order processing in seconds',
      labelNames: ['order_type', 'status'],
      buckets: [0.1, 0.25, 0.5, 1, 2, 5, 10],
      registers: [this.registry]
    });
    
    // Event Processing Lag
    this.eventProcessingLag = new Gauge({
      name: 'event_processing_lag_seconds',
      help: 'Lag in event processing in seconds',
      labelNames: ['topic', 'consumer_group'],
      registers: [this.registry]
    });
  }
  
  public recordHttpRequest(
    method: string,
    route: string,
    statusCode: number,
    duration: number
  ): void {
    this.httpRequestDuration
      .labels(method, route, statusCode.toString())
      .observe(duration);
    
    this.httpRequestTotal
      .labels(method, route, statusCode.toString())
      .inc();
    
    // Alert if P95 > 200ms
    if (duration > 0.2) {
      console.warn(`Slow request detected: ${method} ${route} took ${duration}s`);
    }
  }
  
  public recordError(method: string, route: string, errorType: string): void {
    this.httpRequestErrors
      .labels(method, route, errorType)
      .inc();
  }
  
  public async getMetrics(): Promise<string> {
    return this.registry.metrics();
  }
}

// Express middleware for automatic metrics collection
export const metricsMiddleware = (req: any, res: any, next: Function) => {
  const start = Date.now();
  const metrics = MetricsCollector.getInstance();
  
  // Capture response finish
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    metrics.recordHttpRequest(
      req.method,
      req.route?.path || req.path,
      res.statusCode,
      duration
    );
    
    if (res.statusCode >= 400) {
      metrics.recordError(
        req.method,
        req.route?.path || req.path,
        `http_${Math.floor(res.statusCode / 100)}xx`
      );
    }
  });
  
  next();
};
```

```typescript
// services/shared/observability/tracing.ts
import { NodeTracerProvider } from '@opentelemetry/sdk-trace-node';
import { SimpleSpanProcessor } from '@opentelemetry/sdk-trace-base';
import { JaegerExporter } from '@opentelemetry/exporter-jaeger';
import { Resource } from '@opentelemetry/resources';
import { SemanticResourceAttributes } from '@opentelemetry/semantic-conventions';
import { KafkaJsInstrumentation } from 'opentelemetry-instrumentation-kafkajs';
import { ExpressInstrumentation } from '@opentelemetry/instrumentation-express';
import { HttpInstrumentation } from '@opentelemetry/instrumentation-http';
import { registerInstrumentations } from '@opentelemetry/instrumentation';

export class TracingSetup {
  static initialize(serviceName: string): void {
    // Create tracer provider
    const provider = new NodeTracerProvider({
      resource: new Resource({
        [SemanticResourceAttributes.SERVICE_NAME]: serviceName,
        [SemanticResourceAttributes.DEPLOYMENT_ENVIRONMENT]: process.env.NODE_ENV || 'development',
      }),
    });

    // Configure Jaeger exporter
    const exporter = new JaegerExporter({
      endpoint: process.env.JAEGER_ENDPOINT || 'http://localhost:14268/api/traces',
    });

    // Add span processor
    provider.addSpanProcessor(new SimpleSpanProcessor(exporter));

    // Register the provider
    provider.register();

    // Register instrumentations
    registerInstrumentations({
      instrumentations: [
        new HttpInstrumentation(),
        new ExpressInstrumentation(),
        new KafkaJsInstrumentation(),
      ],
    });
  }
}

// Kafka consumer with tracing
export class TracedKafkaConsumer {
  constructor(private consumer: any, private tracer: any) {}
  
  async consume(topic: string, handler: Function): Promise<void> {
    await this.consumer.subscribe({ topic });
    
    await this.consumer.run({
      eachMessage: async ({ topic, partition, message }: any) => {
        const span = this.tracer.startSpan(`kafka.consume.${topic}`);
        
        try {
          // Extract trace context from message headers
          const traceContext = this.extractTraceContext(message.headers);
          
          // Process message
          await handler(message.value, message.headers);
          
          span.setStatus({ code: 1 }); // OK
        } catch (error) {
          span.setStatus({ code: 2, message: error.message }); // ERROR
          span.recordException(error);
          throw error;
        } finally {
          span.end();
        }
      },
    });
  }
  
  private extractTraceContext(headers: any): any {
    // Extract W3C trace context from Kafka headers
    return {
      traceparent: headers?.traceparent?.toString(),
      tracestate: headers?.tracestate?.toString(),
    };
  }
}
```

const jwt = require('jsonwebtoken');
const axios = require('axios');
const CircuitBreaker = require('circuit-breaker-js');
const logger = require('../../shared/utils/logger');

// Circuit breaker para chamadas ao serviço de auth
const authCircuitBreaker = new CircuitBreaker({
  timeoutDuration: 3000,
  volumeThreshold: 10,
  errorThreshold: 50,
  onCircuitOpen: () => logger.warn('Auth service circuit breaker opened'),
  onCircuitClose: () => logger.info('Auth service circuit breaker closed')
});

class AuthMiddleware {
  constructor() {
    this.jwtSecret = process.env.JWT_SECRET;
    this.authServiceUrl = process.env.AUTH_SERVICE_URL || 'http://localhost:3001';
  }

  async verifyToken(req, res, next) {
    const correlationId = req.headers['x-correlation-id'];
    const authHeader = req.headers.authorization;
    
    if (!authHeader) {
      return res.status(401).json({
        error: 'Unauthorized',
        message: 'No authorization token provided'
      });
    }

    const token = authHeader.replace('Bearer ', '');
    
    try {
      // Verificação local do JWT primeiro (mais rápido)
      const decoded = jwt.verify(token, this.jwtSecret);
      req.user = decoded;
      
      // Validação adicional no serviço de auth (com circuit breaker)
      const isValid = await authCircuitBreaker.run(async () => {
        const response = await axios.get(`${this.authServiceUrl}/api/validate-token`, {
          headers: {
            'Authorization': `Bearer ${token}`,
            'X-Correlation-ID': correlationId
          },
          timeout: 2000
        });
        return response.data.valid;
      });
      
      if (!isValid) {
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'Token is no longer valid'
        });
      }
      
      next();
    } catch (error) {
      logger.error({
        correlation_id: correlationId,
        error: error.message,
        action: 'token_verification'
      });
      
      if (error.name === 'TokenExpiredError') {
        return res.status(401).json({
          error: 'Token expired',
          message: 'Please refresh your token'
        });
      }
      
      if (error.name === 'JsonWebTokenError') {
        return res.status(401).json({
          error: 'Invalid token',
          message: 'Malformed or invalid token'
        });
      }
      
      // Se o serviço de auth estiver indisponível, ainda permitimos com JWT local
      // (fallback para resiliência)
      if (error.code === 'ECONNABORTED' || error.response?.status === 503) {
        logger.warn('Auth service unavailable, using local JWT validation only');
        try {
          const decoded = jwt.verify(token, this.jwtSecret);
          req.user = decoded;
          return next();
        } catch (jwtError) {
          return res.status(401).json({
            error: 'Unauthorized',
            message: 'Unable to validate token'
          });
        }
      }
      
      return res.status(401).json({
        error: 'Unauthorized',
        message: 'Invalid authentication token'
      });
    }
  }

  requireRole(role) {
    return (req, res, next) => {
      if (!req.user) {
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'User not authenticated'
        });
      }
      
      if (!req.user.roles || !req.user.roles.includes(role)) {
        return res.status(403).json({
          error: 'Forbidden',
          message: `Required role: ${role}`
        });
      }
      
      next();
    };
  }

  requirePermission(permission) {
    return (req, res, next) => {
      if (!req.user) {
        return res.status(401).json({
          error: 'Unauthorized',
          message: 'User not authenticated'
        });
      }
      
      if (!req.user.permissions || !req.user.permissions.includes(permission)) {
        return res.status(403).json({
          error: 'Forbidden',
          message: `Required permission: ${permission}`
        });
      }
      
      next();
    };
  }
}

```typescript
import { createSlice, PayloadAction } from '@reduxjs/toolkit'

interface UIState {
  theme: 'light' | 'dark'
  sidebarOpen: boolean
  notifications: Array<{
    id: string
    type: 'info' | 'success' | 'warning' | 'error'
    message: string
    duration?: number
  }>
  modals: Record<string, boolean>
}

const initialState: UIState = {
  theme: 'light',
  sidebarOpen: false,
  notifications: [],
  modals: {}
}

const uiSlice = createSlice({
  name: 'ui',
  initialState,
  reducers: {
    toggleTheme: (state) => {
      state.theme = state.theme === 'light' ? 'dark' : 'light'
    },
    setTheme: (state, action: PayloadAction<'light' | 'dark'>) => {
      state.theme = action.payload
    },
    toggleSidebar: (state) => {
      state.sidebarOpen = !state.sidebarOpen
    },
    setSidebarOpen: (state, action: PayloadAction<boolean>) => {
      state.sidebarOpen = action.payload
    },
    addNotification: (state, action: PayloadAction<Omit<UIState['notifications'][0], 'id'>>) => {
      const id = Date.now().toString()
      state.notifications.push({ id, ...action.payload })
    },
    removeNotification: (state, action: PayloadAction<string>) => {
      state.notifications = state.notifications.filter(n => n.id !== action.payload)
    },
    clearNotifications: (state) => {
      state.notifications = []
    },
    openModal: (state, action: PayloadAction<string>) => {
      state.modals[action.payload] = true
    },
    closeModal: (state, action: PayloadAction<string>) => {
      state.modals[action.payload] = false
    }
  }
})

export const {
  toggleTheme,
  setTheme,
  toggleSidebar,
  setSidebarOpen,
  addNotification,
  removeNotification,
  clearNotifications,
  openModal,
  closeModal
} = uiSlice.actions
```

```sql
WITH daily_users AS (
    SELECT
        DATE(event_timestamp) AS date,
        user_id,
        COUNT(DISTINCT session_id) AS session_count,
        AVG(duration_seconds) AS avg_session_duration
    FROM {{ ref('stg_events') }}
    WHERE event_type = 'session_start'
    GROUP BY 1, 2
),

daily_metrics AS (
    SELECT
        date,
        COUNT(DISTINCT user_id) AS daily_active_users,
        COUNT(DISTINCT CASE WHEN u.created_at::DATE = date THEN u.user_id END) AS new_users,
        COUNT(DISTINCT CASE WHEN u.created_at::DATE < date THEN u.user_id END) AS returning_users,
        SUM(session_count) AS total_sessions,
        AVG(avg_session_duration) AS avg_session_duration_seconds
    FROM daily_users du
    LEFT JOIN {{ ref('stg_users') }} u ON du.user_id = u.user_id
    GROUP BY 1
)

SELECT
    dm.date,
    dm.daily_active_users,
    dm.new_users,
    dm.returning_users,
    dm.total_sessions,
    dm.avg_session_duration_seconds,
    JSONB_BUILD_OBJECT(
        'feature_a', COUNT(DISTINCT CASE WHEN e.feature_name = 'feature_a' THEN e.user_id END),
        'feature_b', COUNT(DISTINCT CASE WHEN e.feature_name = 'feature_b' THEN e.user_id END),
        'feature_c', COUNT(DISTINCT CASE WHEN e.feature_name = 'feature_c' THEN e.user_id END)
    ) AS feature_adoption,
    CURRENT_TIMESTAMP AS created_at
FROM daily_metrics dm
LEFT JOIN {{ ref('stg_events') }} e ON dm.date = DATE(e.event_timestamp)
GROUP BY 1, 2, 3, 4, 5, 6
```
